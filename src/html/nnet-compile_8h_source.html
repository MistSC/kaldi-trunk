57  // Some generic information about each step of the computation... a step is an
58  // instance of a NetworkNode, but a NetworkNode may in general have multiple
59  // steps. A single step may turn into no commands (for input nodes), or
60  // multiple commands.
61  struct StepInfo {
62  int32 node_index; // network-node index
63  bool is_input; // true if step corresponds to an input to the computation.
64  int32 value; // sub-matrix index of value that this step outputs.
65  int32 deriv; // sub-matrix index of derivative at the output of this step; zero
66  // if not used (note: index zero is reserved for the empty
67  // matrix).
68 
69  // precomputed_indexes_index is the index into the
70  // component_precomputed_indexes array in the NnetComputation, or zero if
71  // none needed.
72  int32 precomputed_indexes_index;
73 
74  std::vector<Index> output_indexes; // Indexes that this step outputs.
75  std::vector<int32> output_cindex_ids; // cindex_ids corresponding to each
76  // of the output indexes.
77 
78  // If this component is of type kDescriptor (and note that the top-level
79  // Descriptor is a concatenation over >= 1 parts), then we set value_parts
80  // to a list of submatrix-indexes, each for the corresponding part of the
81  // value. If there is only one part, it will have one element which will be
82  // the same as "value".
83  std::vector<int32> value_parts;
84  // deriv_parts is as "value_parts", but for parts of the derivative (if
85  // we're doing backprop).
86  std::vector<int32> deriv_parts;
87 
88  // for nodes corresponding to descriptors, input_locations_list will contain
89  // information about the inputs to this descriptor, telling us for each row
90  // of the matrix what other matrix rows it is a summation over. this is a
91  // quantity indexed[part-index][row-index], then a list of pairs (step,
92  // row-index), that we store here to avoid computing it twice in forward and
93  // backprop.
94  std::vector<std::vector<std::vector<std::pair<int32,int32> > > > input_locations_list;
95 
96  StepInfo(): node_index(-1), is_input(false), value(0),
97  deriv(0), precomputed_indexes_index(0) { }
98  };
