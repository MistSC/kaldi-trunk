324 static int TinyCat(int argc, const char** argv) {
325  const char* name_in = argc > 0 && strcmp(argv[0], "-") ? argv[0] : NULL;
326  int fd_in = name_in ? _open(name_in, _O_RDONLY) : _fileno(stdin);
327  if (fd_in < 0)
328  return 1;
329 
330  int fd_out = _fileno(stdout);
331  _setmode(fd_in, _O_BINARY);
332  _setmode(fd_out, _O_BINARY);
333 
334  char buffer[100];
335  int last_read;
336  while ((last_read = _read(fd_in, buffer, sizeof(buffer))) > 0)
337  _write(fd_out, buffer, last_read);
338 
339  if (name_in) _close(fd_in);
340  return 0;
341 }
