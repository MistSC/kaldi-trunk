205  void Update(const CuMatrixBase<BaseFloat> &input, const CuMatrixBase<BaseFloat> &diff)
206  {
207  // initial gradient parameters
208  weight_grad_.Resize(wei_dim_3_, wei_dim_1_ * wei_dim_2_);
209  bias_grad_.Resize(wei_dim_3_);
210  // we use following hyperparameters from the option class
211  const BaseFloat lr = opts_.learn_rate * learn_rate_coef_;
212  const BaseFloat lr_bias = opts_.learn_rate * bias_learn_rate_coef_;
213  const BaseFloat mmt = opts_.momentum;
214  const BaseFloat l2 = opts_.l2_penalty;
215  const BaseFloat l1 = opts_.l1_penalty;
216 
217  // we will also need the number of frames in the mini-batch
218  const int32 batch_size = input.NumRows();
219 
220  // compute gradient
221  weight_grad_.AddMatMat(1.0, diff, kTrans, input, kNoTrans, mmt);
222  bias_grad_.AddRowSumMat(1.0, diff, mmt);
223  // l2 regularization
224  if (l2 != 0.0)
225  {
226  weight_.AddMat(-lr*l2*batch_size, weight_);
227  }
228  if (l1 != 0.0)
229  {
230  cu::RegularizeL1(&weight_, &weight_grad_, lr*l1*batch_size, lr);
231  }
232  //update
233  weight_.AddMat(-lr, weight_grad_);
234  bias_.AddVec(-lr_bias, bias_grad_);
235  }
