235  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out)
236  {
237  int32 batch_size = in.NumRows(); // get batch size
238  // give initial size
239  CuMatrix<BaseFloat> output(batch_size, wei_1_j1_ * wei_2_j2_);
240  CuMatrix<BaseFloat> temp(batch_size, wei_1_j1_ * wei_2_i2_);
241  // reshape to std::vector
242  std::vector< CuSubMatrix<BaseFloat>* > reshaped_input;
243  std::vector< CuSubMatrix<BaseFloat>* > reshaped_output;
244  std::vector< CuSubMatrix<BaseFloat>* > reshaped_temp;
245  // w1_vec and w2_vec used to parallel
246  std::vector< CuSubMatrix<BaseFloat>* > w1_vec;
247  std::vector< CuSubMatrix<BaseFloat>* > w2_vec;
248  CuSubMatrix<BaseFloat>* w1 = new CuSubMatrix<BaseFloat>(mode_1_wei_,0,wei_1_j1_,0,wei_1_i1_);
249  CuSubMatrix<BaseFloat>* w2 = new CuSubMatrix<BaseFloat>(mode_2_wei_,0,wei_2_j2_,0,wei_2_i2_);
250  // initial some vectors
251  for(int32 i = 0; i < batch_size; i++)
252  {
253  CuSubMatrix<BaseFloat> *temp_reshaped_input = new CuSubMatrix<BaseFloat>(in,i,wei_1_i1_,wei_2_i2_);
254  CuSubMatrix<BaseFloat> *temp_reshaped_temp = new CuSubMatrix<BaseFloat>(temp,i,wei_1_j1_,wei_2_i2_);
255  CuSubMatrix<BaseFloat> *temp_reshaped_output = new CuSubMatrix<BaseFloat>(output,i,wei_1_j1_,wei_2_j2_);
256  w1_vec.push_back(w1);
257  w2_vec.push_back(w2);
258  reshaped_input.push_back(temp_reshaped_input);
259  reshaped_temp.push_back(temp_reshaped_temp);
260  reshaped_output.push_back(temp_reshaped_output);
261  }
262 
263  AddMatMatBatched<BaseFloat>(1.0,reshaped_temp,w1_vec,kNoTrans,reshaped_input,kNoTrans,0.0);
264  AddMatMatBatched<BaseFloat>(1.0,reshaped_output,reshaped_temp,kNoTrans,w2_vec,kTrans,0.0);
265 
266  out->CopyFromMat(output);
267  CuVector<BaseFloat> bias_vec(wei_1_j1_ * wei_2_j2_);
268  bias_vec.CopyRowsFromMat(bias_);
269  out->AddVecToRows(1.0,bias_vec,1.0);
270  // delete pointers
271  for(std::vector< CuSubMatrix<BaseFloat>* >::iterator i=reshaped_input.begin(); i!=reshaped_input.end(); ++i)
272  delete *i;
273  for(std::vector< CuSubMatrix<BaseFloat>* >::iterator i=reshaped_output.begin(); i!=reshaped_output.end(); ++i)
274  delete *i;
275  for(std::vector< CuSubMatrix<BaseFloat>* >::iterator i=reshaped_temp.begin(); i!=reshaped_temp.end(); ++i)
276  delete *i;
277  delete w1;
278  delete w2;
279  }

328  void Update(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &diff)
329  {
330  // we use following hyperparameters from the option class
331  const BaseFloat lr = opts_.learn_rate * learn_rate_coef_;
332  const BaseFloat lr_bias = opts_.learn_rate * bias_learn_rate_coef_;
333  const BaseFloat mmt = opts_.momentum;
334  const BaseFloat l2 = opts_.l2_penalty;
335  const BaseFloat l1 = opts_.l1_penalty;
336 
337  // we will also need the number of frames in the mini-batch
338  const int32 batch_size = in.NumRows();
339  // give initial size
340  mode_1_wei_grad_.Resize(wei_1_j1_, wei_1_i1_);
341  mode_2_wei_grad_.Resize(wei_2_j2_, wei_2_i2_);
342  bias_grad_.Resize(wei_1_j1_, wei_2_j2_);
343  CuMatrix<BaseFloat> input(batch_size, wei_1_j1_ * wei_2_i2_);
344  CuMatrix<BaseFloat> dif(batch_size, wei_1_j1_ * wei_2_j2_);
345  CuMatrix<BaseFloat> temp_w1(batch_size, wei_1_j1_ * wei_2_i2_);
346  CuMatrix<BaseFloat> temp_w2(batch_size, wei_2_j2_ * wei_1_i1_);
347  CuMatrix<BaseFloat> w1_grad_batch(batch_size, wei_1_j1_ * wei_1_i1_);
348  CuMatrix<BaseFloat> w2_grad_batch(batch_size, wei_2_j2_ * wei_2_i2_);
349  // reshape to std::vector
350  std::vector< CuSubMatrix<BaseFloat>* > reshaped_input;
351  std::vector< CuSubMatrix<BaseFloat>* > reshaped_diff;
352  std::vector< CuSubMatrix<BaseFloat>* > reshaped_temp_w1;
353  std::vector< CuSubMatrix<BaseFloat>* > reshaped_temp_w2;
354  std::vector< CuSubMatrix<BaseFloat>* > reshaped_w1_grad_batch;
355  std::vector< CuSubMatrix<BaseFloat>* > reshaped_w2_grad_batch;
356  // w1_vec and w2_vec used to parallel
357  std::vector< CuSubMatrix<BaseFloat>* > w1_vec;
358  std::vector< CuSubMatrix<BaseFloat>* > w2_vec;
359  CuSubMatrix<BaseFloat>* w1 = new CuSubMatrix<BaseFloat>(mode_1_wei_,0,wei_1_j1_,0,wei_1_i1_);
360  CuSubMatrix<BaseFloat>* w2 = new CuSubMatrix<BaseFloat>(mode_2_wei_,0,wei_2_j2_,0,wei_2_i2_);
361 
362  // initial some vectors
363  for(int32 i = 0; i < in.NumRows(); i++)
364  {
365  CuSubMatrix<BaseFloat> *temp_reshaped_input = new CuSubMatrix<BaseFloat>(in,i,wei_1_i1_,wei_2_i2_);
366  CuSubMatrix<BaseFloat> *temp_reshaped_diff = new CuSubMatrix<BaseFloat>(diff,i,wei_1_j1_,wei_2_j2_);
367  CuSubMatrix<BaseFloat> *temp_reshaped_temp_w1 = new CuSubMatrix<BaseFloat>(temp_w1,i,wei_1_j1_,wei_2_i2_); //j1 * i2
368  CuSubMatrix<BaseFloat> *temp_reshaped_temp_w2 = new CuSubMatrix<BaseFloat>(temp_w2,i,wei_2_j2_,wei_1_i1_); //j2 * i1
369  CuSubMatrix<BaseFloat> *temp_w1_grad_batch = new CuSubMatrix<BaseFloat>(w1_grad_batch,i,wei_1_j1_,wei_1_i1_); //j1 * i1
370  CuSubMatrix<BaseFloat> *temp_w2_grad_batch = new CuSubMatrix<BaseFloat>(w2_grad_batch,i,wei_2_j2_,wei_2_i2_); //j2 * i2
371  w1_vec.push_back(w1);
372  w2_vec.push_back(w2);
373  reshaped_input.push_back(temp_reshaped_input);
374  reshaped_diff.push_back(temp_reshaped_diff);
375  reshaped_temp_w1.push_back(temp_reshaped_temp_w1);
376  reshaped_temp_w2.push_back(temp_reshaped_temp_w2);
377  reshaped_w1_grad_batch.push_back(temp_w1_grad_batch);
378  reshaped_w2_grad_batch.push_back(temp_w2_grad_batch);
379  }
380  // compute gradient
381  // we have w1 and w2 gradient
382  // compute w1 grad
383  AddMatMatBatched<BaseFloat>(1.0,reshaped_temp_w1,reshaped_diff,kNoTrans,w2_vec,kNoTrans,0.0);
384  AddMatMatBatched<BaseFloat>(1.0,reshaped_w1_grad_batch,reshaped_temp_w1,kNoTrans,reshaped_input,kTrans,0.0);
385  CuVector<BaseFloat> w1_grad_vec_temp(wei_1_j1_ * wei_1_i1_);
386  w1_grad_vec_temp.AddRowSumMat(1.0,w1_grad_batch);
387  mode_1_wei_grad_.CopyRowsFromVec(w1_grad_vec_temp);
388  mode_1_wei_grad_.AddMat(mmt,mode_1_wei_grad_);
389  // compute w2 grad
390  AddMatMatBatched<BaseFloat>(1.0,reshaped_temp_w2,reshaped_diff,kTrans,w1_vec,kNoTrans,0.0);
391  AddMatMatBatched<BaseFloat>(1.0,reshaped_w2_grad_batch,reshaped_temp_w2,kNoTrans,reshaped_input,kNoTrans,0.0);
392  CuVector<BaseFloat> w2_grad_vec_temp(wei_2_j2_ * wei_2_i2_);
393  w2_grad_vec_temp.AddRowSumMat(1.0,w2_grad_batch);
394  mode_2_wei_grad_.CopyRowsFromVec(w2_grad_vec_temp);
395  mode_2_wei_grad_.AddMat(mmt,mode_2_wei_grad_);
396  // bias
397  CuVector<BaseFloat> bias_grad_vec_temp(wei_1_j1_ * wei_2_j2_);
398  bias_grad_vec_temp.AddRowSumMat(1.0,diff);
399  bias_grad_.CopyRowsFromVec(bias_grad_vec_temp);
400  bias_grad_.AddMat(mmt,bias_grad_);
401  // l2 regularization
402  if (l2 != 0.0)
403  {
404  mode_1_wei_grad_.AddMat(-lr*l2*batch_size,mode_1_wei_grad_);
405  mode_2_wei_grad_.AddMat(-lr*l2*batch_size,mode_2_wei_grad_);
406  }
407  // l1 regularization
408  if (l1 != 0.0)
409  {
410  cu::RegularizeL1(&mode_1_wei_, &mode_1_wei_grad_, lr*l1*batch_size, lr);
411  cu::RegularizeL1(&mode_2_wei_, &mode_2_wei_grad_, lr*l1*batch_size, lr);
412  }
413  // update
414  mode_1_wei_.AddMat(-lr, mode_1_wei_grad_);
415  mode_2_wei_.AddMat(-lr,mode_2_wei_grad_);
416  bias_.AddMat(-lr_bias,bias_grad_);
417  // delete pointers
