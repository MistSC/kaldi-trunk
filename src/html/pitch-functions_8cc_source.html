78 BaseFloat NccfToPov(BaseFloat n) {
79  BaseFloat ndash = fabs(n);
80  if (ndash > 1.0) ndash = 1.0; // just in case it was slightly outside [-1, 1]
81 
82  BaseFloat r = -5.2 + 5.4 * Exp(7.5 * (ndash - 1.0)) + 4.8 * ndash -
83  2.0 * Exp(-10.0 * ndash) + 4.2 * Exp(20.0 * (ndash - 1.0));
84  // r is the approximate log-prob-ratio of voicing, log(p/(1-p)).
85  BaseFloat p = 1.0 / (1 + Exp(-1.0 * r));
86  KALDI_ASSERT(p - p == 0); // Check for NaN/inf
87  return p;
88 }

102 void ComputeCorrelation(const VectorBase<BaseFloat> &wave,
103  int32 first_lag, int32 last_lag,
104  int32 nccf_window_size,
105  VectorBase<BaseFloat> *inner_prod,
106  VectorBase<BaseFloat> *norm_prod) {
107  Vector<BaseFloat> zero_mean_wave(wave);
108  // TODO: possibly fix this, the mean normalization is done in a strange way.
109  SubVector<BaseFloat> wave_part(wave, 0, nccf_window_size);
110  // subtract mean-frame from wave
111  zero_mean_wave.Add(-wave_part.Sum() / nccf_window_size);
112  BaseFloat e1, e2, sum;
113  SubVector<BaseFloat> sub_vec1(zero_mean_wave, 0, nccf_window_size);
114  e1 = VecVec(sub_vec1, sub_vec1);
115  for (int32 lag = first_lag; lag <= last_lag; lag++) {
116  SubVector<BaseFloat> sub_vec2(zero_mean_wave, lag, nccf_window_size);
117  e2 = VecVec(sub_vec2, sub_vec2);
118  sum = VecVec(sub_vec1, sub_vec2);
119  (*inner_prod)(lag - first_lag) = sum;
120  (*norm_prod)(lag - first_lag) = e1 * e2;
121  }
122 }

131 void ComputeNccf(const VectorBase<BaseFloat> &inner_prod,
132  const VectorBase<BaseFloat> &norm_prod,
133  BaseFloat nccf_ballast,
134  VectorBase<BaseFloat> *nccf_vec) {
135  KALDI_ASSERT(inner_prod.Dim() == norm_prod.Dim() &&
136  inner_prod.Dim() == nccf_vec->Dim());
137  for (int32 lag = 0; lag < inner_prod.Dim(); lag++) {
138  BaseFloat numerator = inner_prod(lag),
139  denominator = pow(norm_prod(lag) + nccf_ballast, 0.5),
140  nccf;
141  if (denominator != 0.0) {
142  nccf = numerator / denominator;
143  } else {
144  KALDI_ASSERT(numerator == 0.0);
145  nccf = 0.0;
146  }
147  KALDI_ASSERT(nccf < 1.01 && nccf > -1.01);
148  (*nccf_vec)(lag) = nccf;
149  }

178 void ComputeLocalCost(const VectorBase<BaseFloat> &nccf_pitch,
179  const VectorBase<BaseFloat> &lags,
180  const PitchExtractionOptions &opts,
181  VectorBase<BaseFloat> *local_cost) {
182  // from the paper, eq. 5, local_cost = 1 - Phi(t,i)(1 - soft_min_f0 L_i)
183  // nccf is the nccf on this frame measured at the lags in "lags".
184  KALDI_ASSERT(nccf_pitch.Dim() == local_cost->Dim() &&
185  nccf_pitch.Dim() == lags.Dim());

187  // add the term -Phi(t,i):
188  local_cost->AddVec(-1.0, nccf_pitch);
189  // add the term soft_min_f0 Phi(t,i) L_i
190  local_cost->AddVecVec(opts.soft_min_f0, lags, nccf_pitch, 1.0);

306 void PitchFrameInfo::ComputeBacktraces(
307  const PitchExtractionOptions &opts,
308  const VectorBase<BaseFloat> &nccf_pitch,
309  const VectorBase<BaseFloat> &lags,
310  const VectorBase<BaseFloat> &prev_forward_cost_vec,
311  std::vector<std::pair<int32, int32> > *index_info,
312  VectorBase<BaseFloat> *this_forward_cost_vec) {
313  int32 num_states = nccf_pitch.Dim();
314 
315  Vector<BaseFloat> local_cost(num_states, kUndefined);
316  ComputeLocalCost(nccf_pitch, lags, opts, &local_cost);
317 
318  const BaseFloat delta_pitch_sq = pow(Log(1.0 + opts.delta_pitch), 2.0),
319  inter_frame_factor = delta_pitch_sq * opts.penalty_factor;
320 
321  // index local_cost, prev_forward_cost and this_forward_cost using raw pointer
322  // indexing not operator (), since this is the very inner loop and a lot of
323  // time is taken here.
324  const BaseFloat *prev_forward_cost = prev_forward_cost_vec.Data();
325  BaseFloat *this_forward_cost = this_forward_cost_vec->Data();
326 
327  if (index_info->empty())
328  index_info->resize(num_states);
329 
330  // make it a reference for more concise indexing.
331  std::vector<std::pair<int32, int32> > &bounds = *index_info;
332 
333  /* bounds[i].first will be a lower bound on the backpointer for state i,
334  bounds[i].second will be an upper bound on it. We progressively tighten
335  these bounds till we know the backpointers exactly.
336  */
337 
338  if (pitch_use_naive_search) {
339  // This branch is only taken in unit-testing code.
340  for (int32 i = 0; i < num_states; i++) {
341  BaseFloat best_cost = std::numeric_limits<BaseFloat>::infinity();
342  int32 best_j = -1;
343  for (int32 j = 0; j < num_states; j++) {
344  BaseFloat this_cost = (j - i) * (j - i) * inter_frame_factor
345  + prev_forward_cost[j];
346  if (this_cost < best_cost) {
347  best_cost = this_cost;
348  best_j = j;
349  }
350  }
351  this_forward_cost[i] = best_cost;
352  state_info_[i].backpointer = best_j;
353  }
354  } else {
355  int32 last_backpointer = 0;
356  for (int32 i = 0; i < num_states; i++) {
357  int32 start_j = last_backpointer;
358  BaseFloat best_cost = (start_j - i) * (start_j - i) * inter_frame_factor
359  + prev_forward_cost[start_j];
360  int32 best_j = start_j;
361 
362  for (int32 j = start_j + 1; j < num_states; j++) {
363  BaseFloat this_cost = (j - i) * (j - i) * inter_frame_factor
364  + prev_forward_cost[j];
365  if (this_cost < best_cost) {
366  best_cost = this_cost;
367  best_j = j;
368  } else { // as soon as the costs stop improving, we stop searching.
369  break; // this is a loose lower bound we're getting.
370  }
371  }
372  state_info_[i].backpointer = best_j;
373  this_forward_cost[i] = best_cost;
374  bounds[i].first = best_j; // this is now a lower bound on the
375  // backpointer.
376  bounds[i].second = num_states - 1; // we have no meaningful upper bound
377  // yet.
378  last_backpointer = best_j;
379  }
380 
381  // We iterate, progressively refining the upper and lower bounds until they
382  // meet and we know that the resulting backtraces are optimal. Each
383  // iteration takes time linear in num_states. We won't normally iterate as
384  // far as num_states; normally we only do two iterations; when printing out
385  // the number of iterations, it's rarely more than that (once I saw seven
386  // iterations). Anyway, this part of the computation does not dominate.
387  for (int32 iter = 0; iter < num_states; iter++) {
388  bool changed = false;
389  if (iter % 2 == 0) { // go backwards through the states
390  last_backpointer = num_states - 1;
391  for (int32 i = num_states - 1; i >= 0; i--) {
392  int32 lower_bound = bounds[i].first,
393  upper_bound = std::min(last_backpointer, bounds[i].second);
394  if (upper_bound == lower_bound) {
395  last_backpointer = lower_bound;
396  continue;
397  }
398  BaseFloat best_cost = this_forward_cost[i];
399  int32 best_j = state_info_[i].backpointer, initial_best_j = best_j;
400 
401  if (best_j == upper_bound) {
402  // if best_j already equals upper bound, don't bother tightening the
403  // upper bound, we'll tighten the lower bound when the time comes.
404  last_backpointer = best_j;
405  continue;
406  }
407  // Below, we have j > lower_bound + 1 because we know we've already
408  // evaluated lower_bound and lower_bound + 1 [via knowledge of
409  // this algorithm.]
410  for (int32 j = upper_bound; j > lower_bound + 1; j--) {
411  BaseFloat this_cost = (j - i) * (j - i) * inter_frame_factor
412  + prev_forward_cost[j];
413  if (this_cost < best_cost) {
414  best_cost = this_cost;
415  best_j = j;
416  } else { // as soon as the costs stop improving, we stop searching,
417  // unless the best j is still lower than j, in which case
418  // we obviously need to keep moving.
419  if (best_j > j)
420  break; // this is a loose lower bound we're getting.
421  }
422  }
423  // our "best_j" is now an upper bound on the backpointer.
424  bounds[i].second = best_j;
425  if (best_j != initial_best_j) {
426  this_forward_cost[i] = best_cost;
427  state_info_[i].backpointer = best_j;
428  changed = true;
429  }
430  last_backpointer = best_j;
431  }
432  } else { // go forwards through the states.
433  last_backpointer = 0;
434  for (int32 i = 0; i < num_states; i++) {
435  int32 lower_bound = std::max(last_backpointer, bounds[i].first),
436  upper_bound = bounds[i].second;
437  if (upper_bound == lower_bound) {
438  last_backpointer = lower_bound;
439  continue;
440  }
441  BaseFloat best_cost = this_forward_cost[i];
442  int32 best_j = state_info_[i].backpointer, initial_best_j = best_j;
443 
444  if (best_j == lower_bound) {
445  // if best_j already equals lower bound, we don't bother tightening
446  // the lower bound, we'll tighten the upper bound when the time
447  // comes.
448  last_backpointer = best_j;
449  continue;
450  }
451  // Below, we have j < upper_bound because we know we've already
452  // evaluated that point.
453  for (int32 j = lower_bound; j < upper_bound - 1; j++) {
454  BaseFloat this_cost = (j - i) * (j - i) * inter_frame_factor
455  + prev_forward_cost[j];
456  if (this_cost < best_cost) {
457  best_cost = this_cost;
458  best_j = j;
459  } else { // as soon as the costs stop improving, we stop searching,
460  // unless the best j is still higher than j, in which case
461  // we obviously need to keep moving.
462  if (best_j < j)
463  break; // this is a loose lower bound we're getting.
464  }
465  }
466  // our "best_j" is now a lower bound on the backpointer.
467  bounds[i].first = best_j;
468  if (best_j != initial_best_j) {
469  this_forward_cost[i] = best_cost;
470  state_info_[i].backpointer = best_j;
471  changed = true;
472  }
473  last_backpointer = best_j;
474  }
475  }
476  if (!changed)
477  break;
478  }
479  }
480  // The next statement is needed due to RecomputeBacktraces: we have to
481  // invalidate the previously computed best-state info.
482  cur_best_state_ = -1;
483  this_forward_cost_vec->AddVec(1.0, local_cost);
484 }

486 void PitchFrameInfo::SetBestState(
487  int32 best_state,
488  std::vector<std::pair<int32, BaseFloat> > &lag_nccf) {
489 
490  // This function would naturally be recursive, but we have coded this to avoid
491  // recursion, which would otherwise eat up the stack. Think of it as a static
492  // member function, except we do use "this" right at the beginning.
493 
494  std::vector<std::pair<int32, BaseFloat> >::reverse_iterator iter = lag_nccf.rbegin();
495 
496  PitchFrameInfo *this_info = this; // it will change in the loop.
497  while (this_info != NULL) {
498  PitchFrameInfo *prev_info = this_info->prev_info_;
499  if (best_state == this_info->cur_best_state_)
500  return; // no change
501  if (prev_info != NULL) // don't write anything for frame -1.
502  iter->first = best_state;
503  size_t state_info_index = best_state - this_info->state_offset_;
504  KALDI_ASSERT(state_info_index < this_info->state_info_.size());
505  this_info->cur_best_state_ = best_state;
506  best_state = this_info->state_info_[state_info_index].backpointer;
507  if (prev_info != NULL) // don't write anything for frame -1.
508  iter->second = this_info->state_info_[state_info_index].pov_nccf;
509  this_info = prev_info;
510  if (this_info != NULL) ++iter;
511  }

515  if (max_latency <= 0) return 0;
516 
517  int32 latency = 0;
518 
519  // This function would naturally be recursive, but we have coded this to avoid
520  // recursion, which would otherwise eat up the stack. Think of it as a static
521  // member function, except we do use "this" right at the beginning.
522  // This function is called only on the most recent PitchFrameInfo object.

525  PitchFrameInfo *this_info = this; // it will change in the loop.
526 
527 
528  for (; this_info != NULL && latency < max_latency;) {
529  int32 offset = this_info->state_offset_;

531  max_living_state - offset < this_info->state_info_.size());

534  max_living_state =
535  this_info->state_info_[max_living_state - offset].backpointer;
536  if (min_living_state == max_living_state) {
537  return latency;
538  }
539  this_info = this_info->prev_info_;
540  if (this_info != NULL) // avoid incrementing latency for frame -1,
541  latency++; // as it's not a real frame.
542  }

721  double outer_min_lag = 1.0 / opts.max_f0 -
722  (opts.upsample_filter_width/(2.0 * opts.resample_freq));
723  double outer_max_lag = 1.0 / opts.min_f0 +
724  (opts.upsample_filter_width/(2.0 * opts.resample_freq));
725  nccf_first_lag_ = ceil(opts.resample_freq * outer_min_lag);
726  nccf_last_lag_ = floor(opts.resample_freq * outer_max_lag);
727 
728  frames_latency_ = 0; // will be set in AcceptWaveform()
729 
730  // Choose the lags at which we resample the NCCF.
731  SelectLags(opts, &lags_);
732 
733  // upsample_cutoff is the filter cutoff for upsampling the NCCF, which is the
734  // Nyquist of the resampling frequency. The NCCF is (almost completely)
735  // bandlimited to around "lowpass_cutoff" (1000 by default), and when the
736  // spectrum of this bandlimited signal is convolved with the spectrum of an
737  // impulse train with frequency "resample_freq", which are separated by 4kHz,
738  // we get energy at -5000,-3000, -1000...1000, 3000..5000, etc. Filtering at
739  // half the Nyquist (2000 by default) is sufficient to get only the first
740  // repetition.
741  BaseFloat upsample_cutoff = opts.resample_freq * 0.5;
742 
743 
744  Vector<BaseFloat> lags_offset(lags_);
745  // lags_offset equals lags_ (which are the log-spaced lag values we want to
746  // measure the NCCF at) with nccf_first_lag_ / opts.resample_freq subtracted
747  // from each element, so we can treat the measured NCCF values as as starting
748  // from sample zero in a signal that starts at the point start /
749  // opts.resample_freq. This is necessary because the ArbitraryResample code
750  // assumes that the input signal starts from sample zero.
751  lags_offset.Add(-nccf_first_lag_ / opts.resample_freq);
752 
753  int32 num_measured_lags = nccf_last_lag_ + 1 - nccf_first_lag_;
754 
755  nccf_resampler_ = new ArbitraryResample(num_measured_lags, opts.resample_freq,
756  upsample_cutoff, lags_offset,
757  opts.upsample_filter_width);
758 
759  // add a PitchInfo object for frame -1 (not a real frame).
760  frame_info_.push_back(new PitchFrameInfo(lags_.Dim()));
761  // zeroes forward_cost_; this is what we want for the fake frame -1.
762  forward_cost_.Resize(lags_.Dim());
763 }

766 int32 OnlinePitchFeatureImpl::NumFramesAvailable(
767  int64 num_downsampled_samples, bool snip_edges) const {
768  int32 frame_shift = opts_.NccfWindowShift(),
769  frame_length = opts_.NccfWindowSize();

771  // of frames only if the input is not finished.

777  return (int32)(num_downsampled_samples * 1.0f / frame_shift + 0.5f);

783 void OnlinePitchFeatureImpl::UpdateRemainder(
784  const VectorBase<BaseFloat> &downsampled_wave_part) {
785  // frame_info_ has an extra element at frame-1, so subtract
786  // one from the length.
787  int64 num_frames = static_cast<int64>(frame_info_.size()) - 1,
788  next_frame = num_frames,
789  frame_shift = opts_.NccfWindowShift(),
790  next_frame_sample = frame_shift * next_frame;
791 
792  signal_sumsq_ += VecVec(downsampled_wave_part, downsampled_wave_part);
793  signal_sum_ += downsampled_wave_part.Sum();
794 
795  // next_frame_sample is the first sample index we'll need for the
796  // next frame.
797  int64 next_downsampled_samples_processed =
798  downsampled_samples_processed_ + downsampled_wave_part.Dim();
799 
800  if (next_frame_sample > next_downsampled_samples_processed) {
801  // this could only happen in the weird situation that the full frame length
802  // is less than the frame shift.
803  int32 full_frame_length = opts_.NccfWindowSize() + nccf_last_lag_;
804  KALDI_ASSERT(full_frame_length < frame_shift && "Code error");
805  downsampled_signal_remainder_.Resize(0);
806  } else {
807  Vector<BaseFloat> new_remainder(next_downsampled_samples_processed -
808  next_frame_sample);
809  // note: next_frame_sample is the index into the entire signal, of
810  // new_remainder(0).
811  // i is the absolute index of the signal.
812  for (int64 i = next_frame_sample;
813  i < next_downsampled_samples_processed; i++) {
814  if (i >= downsampled_samples_processed_) { // in current signal.
815  new_remainder(i - next_frame_sample) =
816  downsampled_wave_part(i - downsampled_samples_processed_);
817  } else { // in old remainder; only reach here if waveform supplied is
818  new_remainder(i - next_frame_sample) = // tiny.
819  downsampled_signal_remainder_(i - downsampled_samples_processed_ +
820  downsampled_signal_remainder_.Dim());
821  }
822  }
823  downsampled_signal_remainder_.Swap(&new_remainder);
824  }
825  downsampled_samples_processed_ = next_downsampled_samples_processed;
826 }

828 void OnlinePitchFeatureImpl::ExtractFrame(
829  const VectorBase<BaseFloat> &downsampled_wave_part,
830  int64 sample_index,
831  VectorBase<BaseFloat> *window) {
832  int32 full_frame_length = window->Dim();
833  int32 offset = static_cast<int32>(sample_index -
834  downsampled_samples_processed_);
835 
836  if (offset + full_frame_length > downsampled_wave_part.Dim()) {
837  // Requested frame is past end of the signal. This should only happen if
838  // input_finished_ == true, when we're flushing out the last couple of
839  // frames of signal. In this case we pad with zeros.
840  KALDI_ASSERT(input_finished_);
841  int32 new_full_frame_length = downsampled_wave_part.Dim() - offset;
842  KALDI_ASSERT(new_full_frame_length > 0);
843  window->SetZero();
844  SubVector<BaseFloat> sub_window(*window, 0, new_full_frame_length);
845  ExtractFrame(downsampled_wave_part, sample_index, &sub_window);
846  return;
847  }
848 
849  // "offset" is the offset of the start of the frame, into this
850  // signal.
851  if (offset >= 0) {
852  // frame is full inside the new part of the signal.
853  window->CopyFromVec(downsampled_wave_part.Range(offset, full_frame_length));
854  } else {
855  // frame is partly in the remainder and partly in the new part.
856  int32 remainder_offset = downsampled_signal_remainder_.Dim() + offset;
857  KALDI_ASSERT(remainder_offset >= 0); // or we didn't keep enough remainder.
858  KALDI_ASSERT(offset + full_frame_length > 0); // or we should have
859  // processed this frame last
860  // time.
861 
862  int32 old_length = -offset, new_length = offset + full_frame_length;
863  window->Range(0, old_length).CopyFromVec(
864  downsampled_signal_remainder_.Range(remainder_offset, old_length));
865  window->Range(old_length, new_length).CopyFromVec(
866  downsampled_wave_part.Range(0, new_length));
867  }
868  if (opts_.preemph_coeff != 0.0) {
869  BaseFloat preemph_coeff = opts_.preemph_coeff;
870  for (int32 i = window->Dim() - 1; i > 0; i--)
871  (*window)(i) -= preemph_coeff * (*window)(i-1);
872  (*window)(0) *= (1.0 - preemph_coeff);
873  }
874 }

912 // see comment with declaration. This is only relevant for online
913 // operation (it gets called for non-online mode, but is a no-op).
914 void OnlinePitchFeatureImpl::RecomputeBacktraces() {
915  KALDI_ASSERT(!opts_.nccf_ballast_online);
916  int32 num_frames = static_cast<int32>(frame_info_.size()) - 1;
917 
918  // The assertion reflects how we believe this function will be called.
919  KALDI_ASSERT(num_frames <= opts_.recompute_frame);
920  KALDI_ASSERT(nccf_info_.size() == static_cast<size_t>(num_frames));
921  if (num_frames == 0)
922  return;
923  double num_samp = downsampled_samples_processed_, sum = signal_sum_,
924  sumsq = signal_sumsq_, mean = sum / num_samp;
925  BaseFloat mean_square = sumsq / num_samp - mean * mean;
926 
927  bool must_recompute = false;
928  BaseFloat threshold = 0.01;
929  for (int32 frame = 0; frame < num_frames; frame++)
930  if (!ApproxEqual(nccf_info_[frame]->mean_square_energy,
931  mean_square, threshold))
932  must_recompute = true;
933 
934  if (!must_recompute) {
935  // Nothing to do. We'll reach here, for instance, if everything was in one
936  // chunk and opts_.nccf_ballast_online == false. This is the case for
937  // offline processing.
938  for (size_t i = 0; i < nccf_info_.size(); i++)
939  delete nccf_info_[i];
940  nccf_info_.clear();
941  return;
942  }
943 
944  int32 num_states = forward_cost_.Dim(),
945  basic_frame_length = opts_.NccfWindowSize();
946 
947  BaseFloat new_nccf_ballast = pow(mean_square * basic_frame_length, 2) *
948  opts_.nccf_ballast;
949 
950  double forward_cost_remainder = 0.0;
951  Vector<BaseFloat> forward_cost(num_states), // start off at zero.
952  next_forward_cost(forward_cost);
953  std::vector<std::pair<int32, int32 > > index_info;
954 
955  for (int32 frame = 0; frame < num_frames; frame++) {
956  NccfInfo &nccf_info = *nccf_info_[frame];
957  BaseFloat old_mean_square = nccf_info_[frame]->mean_square_energy,
958  avg_norm_prod = nccf_info_[frame]->avg_norm_prod,
959  old_nccf_ballast = pow(old_mean_square * basic_frame_length, 2) *
960  opts_.nccf_ballast,
961  nccf_scale = pow((old_nccf_ballast + avg_norm_prod) /
962  (new_nccf_ballast + avg_norm_prod),
963  static_cast<BaseFloat>(0.5));
964  // The "nccf_scale" is an estimate of the scaling factor by which the NCCF
965  // would change on this frame, on average, by changing the ballast term from
966  // "old_nccf_ballast" to "new_nccf_ballast". It's not exact because the
967  // "avg_norm_prod" is just an average of the product e1 * e2 of frame
968  // energies of the (frame, shifted-frame), but these won't change that much
969  // within a frame, and even if they do, the inaccuracy of the scaled NCCF
970  // will still be very small if the ballast term didn't change much, or if
971  // it's much larger or smaller than e1*e2. By doing it as a simple scaling,
972  // we save the overhead of the NCCF resampling, which is a considerable part
973  // of the whole computation.
974  nccf_info.nccf_pitch_resampled.Scale(nccf_scale);
975 
976  frame_info_[frame + 1]->ComputeBacktraces(
977  opts_, nccf_info.nccf_pitch_resampled, lags_,
978  forward_cost, &index_info, &next_forward_cost);
979 
980  forward_cost.Swap(&next_forward_cost);
981  BaseFloat remainder = forward_cost.Min();
982  forward_cost_remainder += remainder;
983  forward_cost.Add(-remainder);
984  }
985  KALDI_VLOG(3) << "Forward-cost per frame changed from "
986  << (forward_cost_remainder_ / num_frames) << " to "
987  << (forward_cost_remainder / num_frames);
988 
989  forward_cost_remainder_ = forward_cost_remainder;
990  forward_cost_.Swap(&forward_cost);
991 
992  int32 best_final_state;
993  forward_cost_.Min(&best_final_state);
994 
995  if (lag_nccf_.size() != static_cast<size_t>(num_frames))
996  lag_nccf_.resize(num_frames);
997 
998  frame_info_.back()->SetBestState(best_final_state, lag_nccf_);
999  frames_latency_ =
1000  frame_info_.back()->ComputeLatency(opts_.max_frames_latency);
1001  for (size_t i = 0; i < nccf_info_.size(); i++)
1002  delete nccf_info_[i];
1003  nccf_info_.clear();
1004 }

1015 void OnlinePitchFeatureImpl::AcceptWaveform(
1016  BaseFloat sampling_rate,
1017  const VectorBase<BaseFloat> &wave) {
1018  // flush out the last few samples of input waveform only if input_finished_ ==
1019  // true.
1020  const bool flush = input_finished_;
1021 
1022  Vector<BaseFloat> downsampled_wave;
1023  signal_resampler_->Resample(wave, flush, &downsampled_wave);
1024 
1025  // these variables will be used to compute the root-mean-square value of the
1026  // signal for the ballast term.
1027  double cur_sumsq = signal_sumsq_, cur_sum = signal_sum_;
1028  int64 cur_num_samp = downsampled_samples_processed_,
1029  prev_frame_end_sample = 0;
1030  if (!opts_.nccf_ballast_online) {
1031  cur_sumsq += VecVec(downsampled_wave, downsampled_wave);
1032  cur_sum += downsampled_wave.Sum();
1033  cur_num_samp += downsampled_wave.Dim();
1034  }
1035 
1036  // end_frame is the total number of frames we can now process, including
1037  // previously processed ones.
1038  int32 end_frame = NumFramesAvailable(
1039  downsampled_samples_processed_ + downsampled_wave.Dim(), opts_.snip_edges);
1040  // "start_frame" is the first frame-index we process
1041  int32 start_frame = frame_info_.size() - 1,
1042  num_new_frames = end_frame - start_frame;
1043 
1044  if (num_new_frames == 0) {
1045  UpdateRemainder(downsampled_wave);
1046  return;
1047  // continuing to the rest of the code would generate
1048  // an error when sizing matrices with zero rows, and
1049  // anyway is a waste of time.
1050  }
1051 
1052  int32 num_measured_lags = nccf_last_lag_ + 1 - nccf_first_lag_,
1053  num_resampled_lags = lags_.Dim(),
1054  frame_shift = opts_.NccfWindowShift(),
1055  basic_frame_length = opts_.NccfWindowSize(),
1056  full_frame_length = basic_frame_length + nccf_last_lag_;
1057 
1058  Vector<BaseFloat> window(full_frame_length),
1059  inner_prod(num_measured_lags),
1060  norm_prod(num_measured_lags);
1061  Matrix<BaseFloat> nccf_pitch(num_new_frames, num_measured_lags),
1062  nccf_pov(num_new_frames, num_measured_lags);
1063 
1064  Vector<BaseFloat> cur_forward_cost(num_resampled_lags);
1065 
1066 
1067  // Because the resampling of the NCCF is more efficient when grouped together,
1068  // we first compute the NCCF for all frames, then resample as a matrix, then
1069  // do the Viterbi [that happens inside the constructor of PitchFrameInfo].
1070 
1071  for (int32 frame = start_frame; frame < end_frame; frame++) {
1072  // start_sample is index into the whole wave, not just this part.
1073  int64 start_sample = static_cast<int64>(frame) * frame_shift;
1074  ExtractFrame(downsampled_wave, start_sample, &window);
1075  if (opts_.nccf_ballast_online) {
1076  // use only up to end of current frame to compute root-mean-square value.
1077  // end_sample will be the sample-index into "downsampled_wave", so
1078  // not really comparable to start_sample.
1079  int64 end_sample = start_sample + full_frame_length -
1080  downsampled_samples_processed_;
1081  KALDI_ASSERT(end_sample > 0); // or should have processed this frame last
1082  // time. Note: end_sample is one past last
1083  // sample.
1084  if (end_sample > downsampled_wave.Dim()) {
1085  KALDI_ASSERT(input_finished_);
1086  end_sample = downsampled_wave.Dim();
1087  }
1088  SubVector<BaseFloat> new_part(downsampled_wave, prev_frame_end_sample,
1089  end_sample - prev_frame_end_sample);
1090  cur_num_samp += new_part.Dim();
1091  cur_sumsq += VecVec(new_part, new_part);
1092  cur_sum += new_part.Sum();
1093  prev_frame_end_sample = end_sample;
1094  }
1095  double mean_square = cur_sumsq / cur_num_samp -
1096  pow(cur_sum / cur_num_samp, 2.0);
1097 
1098  ComputeCorrelation(window, nccf_first_lag_, nccf_last_lag_,
1099  basic_frame_length, &inner_prod, &norm_prod);
1100  double nccf_ballast_pov = 0.0,
1101  nccf_ballast_pitch = pow(mean_square * basic_frame_length, 2) *
1102  opts_.nccf_ballast,
1103  avg_norm_prod = norm_prod.Sum() / norm_prod.Dim();
1104  SubVector<BaseFloat> nccf_pitch_row(nccf_pitch, frame - start_frame);
1105  ComputeNccf(inner_prod, norm_prod, nccf_ballast_pitch,
1106  &nccf_pitch_row);
1107  SubVector<BaseFloat> nccf_pov_row(nccf_pov, frame - start_frame);
1108  ComputeNccf(inner_prod, norm_prod, nccf_ballast_pov,
1109  &nccf_pov_row);
1110  if (frame < opts_.recompute_frame)
1111  nccf_info_.push_back(new NccfInfo(avg_norm_prod, mean_square));
1112  }
1113 
1114  Matrix<BaseFloat> nccf_pitch_resampled(num_new_frames, num_resampled_lags);
1115  nccf_resampler_->Resample(nccf_pitch, &nccf_pitch_resampled);
1116  nccf_pitch.Resize(0, 0); // no longer needed.
1117  Matrix<BaseFloat> nccf_pov_resampled(num_new_frames, num_resampled_lags);
1118  nccf_resampler_->Resample(nccf_pov, &nccf_pov_resampled);
1119  nccf_pov.Resize(0, 0); // no longer needed.
1120 
1121  // We've finished dealing with the waveform so we can call UpdateRemainder
1122  // now; we need to call it before we possibly call RecomputeBacktraces()
1123  // below, which is why we don't do it at the very end.
1124  UpdateRemainder(downsampled_wave);
1125 
1126  std::vector<std::pair<int32, int32 > > index_info;
1127 
1128  for (int32 frame = start_frame; frame < end_frame; frame++) {
1129  int32 frame_idx = frame - start_frame;
1130  PitchFrameInfo *prev_info = frame_info_.back(),
1131  *cur_info = new PitchFrameInfo(prev_info);
1132  cur_info->SetNccfPov(nccf_pov_resampled.Row(frame_idx));
1133  cur_info->ComputeBacktraces(opts_, nccf_pitch_resampled.Row(frame_idx),
1134  lags_, forward_cost_, &index_info,
1135  &cur_forward_cost);
1136  forward_cost_.Swap(&cur_forward_cost);
1137  // Renormalize forward_cost so smallest element is zero.
1138  BaseFloat remainder = forward_cost_.Min();
1139  forward_cost_remainder_ += remainder;
1140  forward_cost_.Add(-remainder);
1141  frame_info_.push_back(cur_info);
1142  if (frame < opts_.recompute_frame)
1143  nccf_info_[frame]->nccf_pitch_resampled =
1144  nccf_pitch_resampled.Row(frame_idx);
1145  if (frame == opts_.recompute_frame - 1 && !opts_.nccf_ballast_online)
1146  RecomputeBacktraces();
1147  }
1148 
1149  // Trace back the best-path.
1150  int32 best_final_state;
1151  forward_cost_.Min(&best_final_state);
1152  lag_nccf_.resize(frame_info_.size() - 1); // will keep any existing data.
1153  frame_info_.back()->SetBestState(best_final_state, lag_nccf_);
1154  frames_latency_ =
1155  frame_info_.back()->ComputeLatency(opts_.max_frames_latency);
1156  KALDI_VLOG(4) << "Latency is " << frames_latency_;
1157 }

1203 void ComputeKaldiPitchFirstPass(
1204  const PitchExtractionOptions &opts,
1205  const VectorBase<BaseFloat> &wave,
1206  Matrix<BaseFloat> *output) {
1207 
1208  int32 cur_rows = 100;
1209  Matrix<BaseFloat> feats(cur_rows, 2);
1210 
1211  OnlinePitchFeature pitch_extractor(opts);
1212  KALDI_ASSERT(opts.frames_per_chunk > 0 &&
1213  "--simulate-first-pass-online option does not make sense "
1214  "unless you specify --frames-per-chunk");
1215 
1216  int32 cur_offset = 0, cur_frame = 0, samp_per_chunk =
1217  opts.frames_per_chunk * opts.samp_freq * 1.0e-03 * opts.frame_shift_ms;
1218 
1219  while (cur_offset < wave.Dim()) {
1220  int32 num_samp = std::min(samp_per_chunk, wave.Dim() - cur_offset);
1221  SubVector<BaseFloat> wave_chunk(wave, cur_offset, num_samp);
1222  pitch_extractor.AcceptWaveform(opts.samp_freq, wave_chunk);
1223  cur_offset += num_samp;
1224  if (cur_offset == wave.Dim())
1225  pitch_extractor.InputFinished();
