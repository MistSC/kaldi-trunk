35 void NnetOnlineComputer::Compute(const CuMatrixBase<BaseFloat> &input,
36  CuMatrix<BaseFloat> *output) {
37  KALDI_ASSERT(output != NULL);
38  KALDI_ASSERT(!finished_);
39  int32 dim = input.NumCols();
40 
41  // If input is empty, we also set output to zero size.
42  if (input.NumRows() == 0) {
43  output->Resize(0, 0);
44  return;
45  }
46 
47  // Checking if feature dimension matches that required by the neural network.
48  if (dim != nnet_.InputDim()) {
49  KALDI_ERR << "Feature dimension is " << dim << ", but network expects "
50  << nnet_.InputDim();
51  }
52  // num_effective_input_rows is the effective number of input rows we have, for
53  // purposes of computing how much output we will get. It is the number of
54  // actual input rows plus the amount of context stored at intermediate layers
55  // of the network (which if we have previously done the computation, will
56  // equal nnet_.LeftContext() + nnet_.RightContext()).
57  int32 num_effective_input_rows = 0;
58  // Initialize the first element of data_, with input
59  CuMatrix<BaseFloat> &input_data(data_[0]);
60  if (is_first_chunk_) {
61  is_first_chunk_ = false;
62  // assert that all the component-wise input buffers are empty
63  for (int32 i = 0; i < reusable_component_inputs_.size(); i++)
64  KALDI_ASSERT(reusable_component_inputs_[0].NumRows() == 0);
65  // Pad at the start of the file if necessary.
66  if ((pad_input_) && (nnet_.LeftContext() > 0)) {
67  input_data.Resize(nnet_.LeftContext() + input.NumRows(), dim);
68  input_data.Range(0, nnet_.LeftContext(), 0,
69  dim).CopyRowsFromVec(input.Row(0));
70  input_data.Range(nnet_.LeftContext(), input.NumRows(),
71  0, dim).CopyFromMat(input);
72  } else {
73  input_data.Resize(input.NumRows(), input.NumCols());
74  input_data.CopyFromMat(input);
75  }
76  num_effective_input_rows = input_data.NumRows();
77  } else {
78  int32 extra_input_rows = 0;
79  // checking if we did forward pass for any chunks before.
80  // if we did a forward pass, component input buffers would be non-empty
81  // these buffers store information equivalent to having an nnet_input
82  // buffer of (nnet_.LeftContext() + nnet_.RightContext())
83  for (int32 i = 0; i < reusable_component_inputs_.size(); i++) {
84  if (reusable_component_inputs_[i].NumRows() > 0) {
85  extra_input_rows = nnet_.LeftContext() + nnet_.RightContext();
86  break;
87  }
88  }
89  // add unprocessed input from the previous calls
90  input_data.Resize(input.NumRows() + unprocessed_buffer_.NumRows(), dim);
91  if (unprocessed_buffer_.NumRows() > 0)
92  input_data.Range(0, unprocessed_buffer_.NumRows(),
93  0, dim).CopyFromMat(unprocessed_buffer_);
94  input_data.Range(unprocessed_buffer_.NumRows(), input.NumRows(),
95  0, dim).CopyFromMat(input);
96  unprocessed_buffer_.Resize(0, 0); // clearing the unprocessed buffer
97  num_effective_input_rows = input_data.NumRows() + extra_input_rows;
98  }
99  if (num_effective_input_rows >=
100  nnet_.LeftContext() + nnet_.RightContext() + 1) {
101  // we have sufficient frames to compute at least one nnet output
102  nnet_.ComputeChunkInfo(num_effective_input_rows, 1, &chunk_info_);
103  // store the last frame as it might be needed for padding
104  last_seen_input_frame_ = input_data.Row(input_data.NumRows() - 1);
105  Propagate();
106  *output = data_.back();
107  } else {
108  // store the input in the unprocessed_buffer_
109  unprocessed_buffer_ = input_data;
110  // not enough input context so just return an empty array
111  output->Resize(0, 0);
112  }
113 }

115 void NnetOnlineComputer::Flush(CuMatrix<BaseFloat> *output) {
116  KALDI_ASSERT(!finished_ && !is_first_chunk_);
117  int32 num_frames_padding = (pad_input_ ? nnet_.RightContext() : 0);
118  int32 num_stored_frames = nnet_.LeftContext() + nnet_.RightContext();
119  int32 num_effective_input_rows = num_stored_frames + num_frames_padding;
120  // If the amount of output would be empty return at this point.
121  if (num_effective_input_rows < nnet_.LeftContext() + nnet_.RightContext() + 1) {
122  output->Resize(0, 0);
123  finished_ = true;
124  return;
125  }
126 
127  int32 dim = nnet_.InputDim();
128  CuMatrix<BaseFloat> &input_data(data_[0]);
129  KALDI_ASSERT(num_frames_padding > 0); // else we would have returned above.
130  input_data.Resize(num_frames_padding, dim);
131  input_data.CopyRowsFromVec(last_seen_input_frame_);
132 
133  // Note, we later modify this chunk-info, it isn't quite correct right now
134  // because we add extra data at intermediate layers, and the actual number of
135  // input rows doesn't equal num_effective_input_rows.
136  nnet_.ComputeChunkInfo(num_effective_input_rows, 1,
137  &chunk_info_);
138  Propagate();
139  *output = data_.back();
140  finished_ = true;
141 }

143 void NnetOnlineComputer::Propagate() {
144  // This method is like the normal nnet propagate, but we reuse the frames
145  // computed from the previous chunk, at each component.
146 
147  for (int32 c = 0; c < nnet_.NumComponents(); c++) {
148  // we assume that the chunks are always contiguous
149  chunk_info_[c].MakeOffsetsContiguous();
150  chunk_info_[c + 1].MakeOffsetsContiguous();
151 
152  const Component &component = nnet_.GetComponent(c);
153  CuMatrix<BaseFloat> &input_data = data_[c], &output_data = data_[c + 1];
154  CuMatrix<BaseFloat> input_data_temp;
155 
156  if (component.Context().size() > 1) {
157  int32 dim = component.InputDim();
158  if (reusable_component_inputs_[c].NumRows() > 0) {
159  // concatenate any frames computed by previous component
160  // in the last call, to the input of the current component
161  input_data_temp.Resize(reusable_component_inputs_[c].NumRows()
162  + input_data.NumRows(), dim);
163  input_data_temp.Range(0, reusable_component_inputs_[c].NumRows(),
164  0, dim).CopyFromMat(reusable_component_inputs_[c]);
165  input_data_temp.Range(reusable_component_inputs_[c].NumRows(),
166  input_data.NumRows(), 0, dim).CopyFromMat(
167  input_data);
168  input_data = input_data_temp;
169  }
170  // store any frames which can be reused in the next call
171  reusable_component_inputs_[c].Resize(component.Context().back() -
172  component.Context().front(), dim);
173  reusable_component_inputs_[c].CopyFromMat(
174  input_data.RowRange(input_data.NumRows() -
175  reusable_component_inputs_[c].NumRows(),
176  reusable_component_inputs_[c].NumRows()));
177  }
178 
179  // chunk_info objects provided assume that we added all the reusable
180  // context at the input of the nnet. However we are reusing hidden
181  // activations computed in the previous call.
182  // Hence we manipulate the chunk_info objects to reflect the state of the
183  // actual chunk, each component is computing, in the current Propagate.
184  // As before we always assume the chunks are contiguous.
185 
186  // modifying the input chunk_info
187  int32 chunk_size_assumed = chunk_info_[c].ChunkSize();
188  int32 last_offset = chunk_info_[c].GetOffset(chunk_size_assumed - 1);
189  int32 first_offset = last_offset - input_data.NumRows() + 1;
190  ChunkInfo input_chunk_info(chunk_info_[c].NumCols(),
191  chunk_info_[c].NumChunks(),
192  first_offset,
193  last_offset);
194  // modifying the output chunk_info
195  chunk_size_assumed = chunk_info_[c + 1].ChunkSize();
196  last_offset = chunk_info_[c + 1].GetOffset(chunk_size_assumed - 1);
197  first_offset = last_offset - (input_data.NumRows() -
198  (component.Context().back() -
199  component.Context().front())) + 1;
200  ChunkInfo output_chunk_info(chunk_info_[c + 1].NumCols(),
201  chunk_info_[c + 1].NumChunks(),
202  first_offset,
203  last_offset);
204  component.Propagate(input_chunk_info, output_chunk_info,
205  input_data, &output_data);
206  }
207 }
