140 template <typename OtherReal>
141 void CuSparseMatrix<Real>::CopyFromSmat(const SparseMatrix<OtherReal> &smat) {
142  num_rows_ = smat.NumRows();
143  num_cols_ = smat.NumCols();
144 #if HAVE_CUDA == 1
145  if (CuDevice::Instantiate().Enabled()) {
146  if (num_rows_ == 0 || num_cols_ == 0) {
147  elements_.Resize(0);
148  return;
149  }
150  // We first prepare <elements_> on CPU, we then move it to GPU by calling
151  // CopyFromVec. This piece of code should be changed if we change the data
152  // structure later.
153  std::vector<MatrixElement<Real> > cpu_elements;
154  for (int32 i = 0; i < smat.NumRows(); ++i) {
155  for (int32 j = 0; j < (smat.Data() + i)->NumElements(); ++j) {
156  MatrixElement<Real> cpu_element;
157  cpu_element.row = i;
158  cpu_element.column = ((smat.Data() + i)->Data() + j)->first;
159  cpu_element.weight = ((smat.Data() + i)->Data() + j)->second;
160  cpu_elements.push_back(cpu_element);
161  }
162  }
163  elements_.CopyFromVec(cpu_elements);
164  } else
165 #endif
166  {
167  this->Mat().CopyFromSmat(smat);
168  }
169 }

254  MatrixIndexT tmp_dim = num_rows_;
255  num_rows_ = smat->num_rows_;
256  smat->num_rows_ = tmp_dim;
257  tmp_dim = num_cols_;
258  num_cols_ = smat->num_cols_;
259  smat->num_cols_ = tmp_dim;
260  } else
261 #endif
262  {
263  Real dim = num_rows_;

265  smat->num_rows_ = dim;
266  dim = num_cols_;
267  num_cols_ = smat->num_cols_;
268  smat->num_cols_ = dim;
269  Mat().Swap(&(smat->Mat()));

299 template <typename Real>
300 Real TraceMatSmat(const CuMatrixBase<Real> &A,
301  const CuSparseMatrix<Real> &B,
302  MatrixTransposeType trans) {
303  if (A.NumCols() == 0) {
304  KALDI_ASSERT(B.NumCols() == 0);
305  return 0.0;
306  }
307  if (B.NumElements() == 0) {
308  return 0.0;
309  }
310  Real result = 0;
311 #if HAVE_CUDA == 1
312  if (CuDevice::Instantiate().Enabled()) {
313  if (trans == kTrans) {
314  KALDI_ASSERT(A.NumRows() == B.NumRows() && A.NumCols() == B.NumCols());
315  } else {
316  KALDI_ASSERT(A.NumCols() == B.NumRows() && A.NumRows() == B.NumCols());
317  }
318 
319  // The Sum() method in CuVector handles a bunch of logic, we use that to
320  // comptue the trace.
321  CuVector<Real> sum_vec(B.NumElements());
322  Timer tim;
323  dim3 dimBlock(CU1DBLOCK, 1);
324  dim3 dimGrid(n_blocks(B.NumElements(), CU1DBLOCK), 1);
325  if (trans == kNoTrans) {
326  cuda_trace_mat_smat(dimGrid, dimBlock, A.Data(), B.Data(),
327  A.Dim(), B.NumElements(), sum_vec.Data());
328  } else {
329  cuda_trace_mat_smat_trans(dimGrid, dimBlock, A.Data(), B.Data(),
330  A.Dim(), B.NumElements(), sum_vec.Data());
331  }
332  result = sum_vec.Sum();
333  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
334  } else
335 #endif
336  {
337  result = TraceMatSmat(A.Mat(), B.Mat(), trans);
338  }
339  return result;
340 }

351 void GeneralMatrix::CopyToMat(CuMatrixBase<BaseFloat> *cu_mat,
352  MatrixTransposeType trans) const {
353 #if HAVE_CUDA == 1
354  if (CuDevice::Instantiate().Enabled()) {
355  switch (Type()) {
356  case kFullMatrix: {
357  cu_mat->CopyFromMat(mat_);
358  break;
359  }
360  case kSparseMatrix: {
361  CuSparseMatrix<BaseFloat> smat(smat_);
362  smat.CopyToMat(cu_mat, trans);
363  break;
364  }
365  case kCompressedMatrix: {
366  Matrix<BaseFloat> mat(cmat_);
367  if (trans == kNoTrans) {
368  cu_mat->CopyFromMat(mat);
369  break;
370  } else {
371  CuMatrix<BaseFloat> temp_cu;
372  temp_cu.Swap(&mat);
373  cu_mat->CopyFromMat(temp_cu, kTrans);
374  break;
375  }
376  }
377  default:
378  KALDI_ERR << "Invalid GeneralMatrix type.";
379  }
380  return;
381  } else
382 #endif
383  {
384  CopyToMat(&(cu_mat->Mat()), trans);
385  }
386 }

389 template <typename Real>
390 template <typename OtherReal>
391 void CuSparseMatrix<Real>::CopyToMat(CuMatrixBase<OtherReal> *M,
392  MatrixTransposeType trans) const {
393  if (trans == kNoTrans) {
394  KALDI_ASSERT(M->NumRows() == NumRows() && M->NumCols() == NumCols());
395  } else {
396  KALDI_ASSERT(M->NumRows() == NumCols() && M->NumCols() == NumRows());
397  }
398  M->SetZero();
399  if (NumElements() == 0) {
400  return;
401  }
402 #if HAVE_CUDA == 1
403  if (CuDevice::Instantiate().Enabled()) {
404  Timer tim;
405  dim3 dimBlock(CU1DBLOCK, 1);
406  dim3 dimGrid(n_blocks(this->NumElements(), CU1DBLOCK), 1);
407  if (trans == kNoTrans) {
408  cuda_copy_from_smat(dimGrid, dimBlock, M->Data(),
409  this->Data(), M->Dim(), this->NumElements());
410  } else {
411  cuda_copy_from_smat_trans(dimGrid, dimBlock, M->Data(),
412  this->Data(), M->Dim(), this->NumElements());
413  }
414  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
415  } else
416 #endif
417  {
418  reinterpret_cast<const SparseMatrix<Real>*>(this)->CopyToMat(&(M->Mat()),
419  trans);
420  }
421 }

441 void GeneralMatrix::AddToMat(BaseFloat alpha,
442  CuMatrixBase<BaseFloat> *cu_mat,
443  MatrixTransposeType trans) const {
444  switch (Type()) {
445  case kFullMatrix: {
446 #if HAVE_CUDA == 1
447  if (CuDevice::Instantiate().Enabled()) {
448  CuMatrix<BaseFloat> cu_copy(mat_);
449  cu_mat->AddMat(alpha, cu_copy);
450  break;
451  }
452 #endif
453  cu_mat->Mat().AddMat(alpha, mat_);
454  break;
455  }
456  case kSparseMatrix: {
457 #if HAVE_CUDA == 1
458  if (CuDevice::Instantiate().Enabled()) {
459  // TODO: we could make this more efficient by
460  // implementing an AddSmat function in class CuMatrixBase.
461  CuSparseMatrix<BaseFloat> sparse_cu_mat(smat_);
462  CuMatrix<BaseFloat> cu_temp(
463  trans == kNoTrans ? sparse_cu_mat.NumRows() :
464  sparse_cu_mat.NumCols(),
465  trans == kNoTrans ? sparse_cu_mat.NumCols() :
466  sparse_cu_mat.NumRows(),
467  kUndefined);
468  sparse_cu_mat.CopyToMat(&cu_temp, trans);
469  cu_mat->AddMat(alpha, cu_temp, kNoTrans);
470  break;
471  }
472 #endif
473  smat_.AddToMat(alpha, &(cu_mat->Mat()), trans);
474  break;
475  }
476  case kCompressedMatrix: {
477  Matrix<BaseFloat> mat(cmat_);
478 #if HAVE_CUDA == 1
479  if (CuDevice::Instantiate().Enabled()) {
480  CuMatrix<BaseFloat> cu_mat_copy(mat);
481  cu_mat->AddMat(alpha, cu_mat_copy, trans);
482  break;
483  }
484 #endif
485  cu_mat->Mat().AddMat(alpha, mat, trans);
486  break;
487  }
488  default:
489  KALDI_ERR << "Invalid GeneralMatrix type.";
490  }
491 }
