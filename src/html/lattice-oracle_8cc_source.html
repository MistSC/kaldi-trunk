59 void MapWildCards(const LabelSet &wildcards, fst::StdVectorFst *ofst) {
60  // map all wildcards symbols to epsilons
61  for (fst::StateIterator<fst::StdVectorFst> siter(*ofst);
62  !siter.Done(); siter.Next()) {
63  fst::StdArc::StateId s = siter.Value();
64  for (fst::MutableArcIterator<fst::StdVectorFst> aiter(ofst, s);
65  !aiter.Done(); aiter.Next()) {
66  fst::StdArc arc(aiter.Value());
67  LabelSet::const_iterator it = wildcards.find(arc.ilabel);
68  if (it != wildcards.end()) {
69  KALDI_VLOG(4) << "MapWildCards: mapping symbol " << arc.ilabel
70  << " to epsilon" << endl;
71  arc.ilabel = 0;
72  }
73  it = wildcards.find(arc.olabel);
74  if (it != wildcards.end()) {
75  arc.olabel = 0;
76  }
77  aiter.SetValue(arc);
78  }
79  }
80 }

82 // convert from Lattice to standard FST
83 // also maps wildcard symbols to epsilons
84 // then removes epsilons
85 void ConvertLatticeToUnweightedAcceptor(const kaldi::Lattice &ilat,
86  const LabelSet &wildcards,
87  fst::StdVectorFst *ofst) {
88  // first convert from lattice to normal FST
89  fst::ConvertLattice(ilat, ofst);
90  // remove weights, project to output, sort according to input arg
91  fst::Map(ofst, fst::RmWeightMapper<fst::StdArc>());
92  fst::Project(ofst, fst::PROJECT_OUTPUT); // The words are on the output side
93  MapWildCards(wildcards, ofst);
94  fst::RmEpsilon(ofst); // Don't tolerate epsilons as they make it hard to
95  // tally errors
96  fst::ArcSort(ofst, fst::StdILabelCompare());
97 }

99 void CreateEditDistance(const fst::StdVectorFst &fst1,
100  const fst::StdVectorFst &fst2,
101  fst::StdVectorFst *pfst) {
102  typedef fst::StdArc StdArc;
103  typedef fst::StdArc::Weight Weight;
104  typedef fst::StdArc::Label Label;
105  Weight correct_cost(0.0);
106  Weight substitution_cost(1.0);
107  Weight insertion_cost(1.0);
108  Weight deletion_cost(1.0);
109 
110  // create set of output symbols in fst1
111  std::vector<Label> fst1syms, fst2syms;
112  GetOutputSymbols(fst1, false /*no epsilons*/, &fst1syms);
113  GetInputSymbols(fst2, false /*no epsilons*/, &fst2syms);
114 
115  pfst->AddState();
116  pfst->SetStart(0);
117  for (size_t i = 0; i < fst1syms.size(); i++)
118  pfst->AddArc(0, StdArc(fst1syms[i], 0, deletion_cost, 0)); // deletions
119 
120  for (size_t i = 0; i < fst2syms.size(); i++)
121  pfst->AddArc(0, StdArc(0, fst2syms[i], insertion_cost, 0)); // insertions
122 
123  // stupid implementation O(N^2)
124  for (size_t i = 0; i < fst1syms.size(); i++) {
125  Label label1 = fst1syms[i];
126  for (size_t j = 0; j < fst2syms.size(); j++) {
127  Label label2 = fst2syms[j];
128  Weight cost(label1 == label2 ? correct_cost : substitution_cost);
129  pfst->AddArc(0, StdArc(label1, label2, cost, 0)); // substitutions
130  }
131  }
132  pfst->SetFinal(0, Weight::One());
133  ArcSort(pfst, fst::StdOLabelCompare());
134 }

136 void CountErrors(const fst::StdVectorFst &fst,
137  int32 *correct,
138  int32 *substitutions,
139  int32 *insertions,
140  int32 *deletions,
141  int32 *num_words) {
142  typedef fst::StdArc::StateId StateId;
143  typedef fst::StdArc::Weight Weight;
144  *correct = *substitutions = *insertions = *deletions = *num_words = 0;
145 
146  // go through the first complete path in fst (there should be only one)
147  StateId src = fst.Start();
148  while (fst.Final(src)== Weight::Zero()) { // while not final
149  for (fst::ArcIterator<fst::StdVectorFst> aiter(fst, src);
150  !aiter.Done(); aiter.Next()) {
151  fst::StdArc arc = aiter.Value();
152  if (arc.ilabel == arc.olabel && arc.ilabel != 0) {
153  (*correct)++;
154  (*num_words)++;
155  } else if (arc.ilabel == 0 && arc.olabel != 0) {
156  (*deletions)++;
157  (*num_words)++;
158  } else if (arc.ilabel != 0 && arc.olabel == 0) {
159  (*insertions)++;
160  } else if (arc.ilabel != 0 && arc.olabel != 0) {
161  (*substitutions)++;
162  (*num_words)++;
163  } else {
164  KALDI_ASSERT(arc.ilabel == 0 && arc.olabel == 0);
165  }
166  src = arc.nextstate;
167  continue; // jump to next state
168  }
169  }
170 }
