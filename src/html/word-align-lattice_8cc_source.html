172  void ProcessFinal(Tuple tuple, StateId output_state) {
173  // ProcessFinal is only called if the input_state has
174  // final-prob of One(). [else it should be zero. This
175  // is because we called CreateSuperFinal().]
176 
177  if (tuple.comp_state.IsEmpty()) { // computation state doesn't have
178  // anything pending.
179  std::vector<int32> empty_vec;
180  CompactLatticeWeight cw(tuple.comp_state.FinalWeight(), empty_vec);
181  lat_out_->SetFinal(output_state, Plus(lat_out_->Final(output_state), cw));
182  } else {
183  // computation state has something pending, i.e. input or
184  // output symbols that need to be flushed out. Note: OutputArc() would
185  // have returned false or we wouldn't have been called, so we have to
186  // force it out.
187  CompactLatticeArc lat_arc;
188  tuple.comp_state.OutputArcForce(info_, tmodel_, &lat_arc, &error_);
189  lat_arc.nextstate = GetStateForTuple(tuple, true); // true == add to queue.
190  // The final-prob stuff will get called again from ProcessQueueElement().
191  // Note: because we did CreateSuperFinal(), this final-state on the input
192  // lattice will have no output arcs (and unit final-prob), so there will be
193  // no complications with processing the arcs from this state (there won't
194  // be any).
195  KALDI_ASSERT(output_state != lat_arc.nextstate);
196  lat_out_->AddArc(output_state, lat_arc);
197  }
198  }

201  void ProcessQueueElement() {
202  KALDI_ASSERT(!queue_.empty());
203  Tuple tuple = queue_.back().first;
204  StateId output_state = queue_.back().second;
205  queue_.pop_back();
206 
207  // First thing is-- we see whether the computation-state has something
208  // pending that it wants to output. In this case we don't do
209  // anything further. This is a chosen behavior similar to the
210  // epsilon-sequencing rules encoded by the filters in
211  // composition.
212  CompactLatticeArc lat_arc;
213  Tuple tuple2(tuple); // temp
214  if (tuple.comp_state.OutputArc(info_, tmodel_, &lat_arc, &error_)) {
215  // note: this function changes the tuple (when it returns true).
216  lat_arc.nextstate = GetStateForTuple(tuple, true); // true == add to queue,
217  // if not already present.
218  KALDI_ASSERT(output_state != lat_arc.nextstate);
219  lat_out_->AddArc(output_state, lat_arc);
220  } else {
221  // when there's nothing to output, we'll process arcs from the input-state.
222  // note: it would in a sense be valid to do both (i.e. process the stuff
223  // above, and also these), but this is a bit like the epsilon-sequencing
224  // stuff in composition: we avoid duplicate arcs by doing it this way.
225 
226  if (lat_.Final(tuple.input_state) != CompactLatticeWeight::Zero()) {
227  KALDI_ASSERT(lat_.Final(tuple.input_state) == CompactLatticeWeight::One());
228  // ... since we did CreateSuperFinal.
229  ProcessFinal(tuple, output_state);
230  }
231  // Now process the arcs. Note: final-state shouldn't have any arcs.
232  for(fst::ArcIterator<CompactLattice> aiter(lat_, tuple.input_state);
233  !aiter.Done(); aiter.Next()) {
234  const CompactLatticeArc &arc = aiter.Value();
235  Tuple next_tuple(tuple);
236  LatticeWeight weight;
237  next_tuple.comp_state.Advance(arc, &weight);
238  next_tuple.input_state = arc.nextstate;
239  StateId next_output_state = GetStateForTuple(next_tuple, true); // true == add to queue,
240  // if not already present.
241  // We add an epsilon arc here (as the input and output happens
242  // separately)... the epsilons will get removed later.
243  KALDI_ASSERT(next_output_state != output_state);
244  lat_out_->AddArc(output_state,
245  CompactLatticeArc(0, 0,
246  CompactLatticeWeight(weight, std::vector<int32>()),
247  next_output_state));
248  }
249  }

252  LatticeWordAligner(const CompactLattice &lat,
253  const TransitionModel &tmodel,
254  const WordBoundaryInfo &info,
255  int32 max_states,
256  CompactLattice *lat_out):
257  lat_(lat), tmodel_(tmodel), info_in_(info), info_(info),
258  max_states_(max_states), lat_out_(lat_out),
259  error_(false) {
260  bool test = true;
261  uint64 props = lat_.Properties(fst::kIDeterministic|fst::kIEpsilons, test);
262  if (props != fst::kIDeterministic) {
263  KALDI_WARN << "[Lattice has input epsilons and/or is not input-deterministic "
264  << "(in Mohri sense)]-- i.e. lattice is not deterministic. "
265  << "Word-alignment may be slow and-or blow up in memory.";
266  }
267  fst::CreateSuperFinal(&lat_); // Creates a super-final state, so the
268  // only final-probs are One().
269 
270  // Inside this class, we don't want to use zero for the silence
271  // or partial-word labels, as this will interfere with the RmEpsilon
272  // stage, where we don't want the arcs corresponding to silence or
273  // partial words to be removed-- only the arcs with nothing at all
274  // on them.
275  if (info_.partial_word_label == 0 || info_.silence_label == 0) {
276  int32 unused_label = 1 + HighestNumberedOutputSymbol(lat);
277  KALDI_ASSERT(unused_label > 0);
278  if (info_.partial_word_label == 0)
279  info_.partial_word_label = unused_label;
280  if (info_.silence_label == 0)
281  info_.silence_label = unused_label;
282  }
283  }
284 
285  // Removes epsilons; also removes unreachable states...
286  // not sure if these would exist if original was connected.
287  // This also replaces the temporary symbols for the silence
288  // and partial-words, with epsilons, if we wanted epsilons.
289  void RemoveEpsilonsFromLattice() {
290  // Remove epsilon arcs from output lattice.
291  RmEpsilon(lat_out_, true); // true = connect.
292  std::vector<int32> syms_to_remove;
293  if (info_in_.partial_word_label == 0)
294  syms_to_remove.push_back(info_.partial_word_label);
295  if (info_in_.silence_label == 0)
296  syms_to_remove.push_back(info_.silence_label);
297  if (!syms_to_remove.empty()) {
298  RemoveSomeInputSymbols(syms_to_remove, lat_out_);
299  Project(lat_out_, fst::PROJECT_INPUT);
300  }
301  }

303  bool AlignLattice() {
304  lat_out_->DeleteStates();
305  if (lat_.Start() == fst::kNoStateId) {
306  KALDI_WARN << "Trying to word-align empty lattice.";
307  return false;
308  }
309  ComputationState initial_comp_state;
310  Tuple initial_tuple(lat_.Start(), initial_comp_state);
311  StateId start_state = GetStateForTuple(initial_tuple, true); // True = add this to queue.
312  lat_out_->SetStart(start_state);
313 
314  while (!queue_.empty()) {
315  if (max_states_ > 0 && lat_out_->NumStates() > max_states_) {
316  KALDI_WARN << "Number of states in lattice exceeded max-states of "
317  << max_states_ << ", original lattice had "
318  << lat_.NumStates() << " states. Returning what we have.";
319  RemoveEpsilonsFromLattice();
320  return false;
321  }
322  ProcessQueueElement();
323  }
324 
325  RemoveEpsilonsFromLattice();
326 
327  return !error_;
328  }

346 bool LatticeWordAligner::ComputationState::OutputSilenceArc(
347  const WordBoundaryInfo &info, const TransitionModel &tmodel,
348  CompactLatticeArc *arc_out, bool *error) {
349  if (transition_ids_.empty()) return false;
350  int32 phone = tmodel.TransitionIdToPhone(transition_ids_[0]);
351  if (info.TypeOfPhone(phone) != WordBoundaryInfo::kNonWordPhone) return false;
352 
353  // we assume the start of transition_ids_ is the start of the phone [silence];
354  // this is a precondition.
355  size_t len = transition_ids_.size(), i;
356  // Keep going till we reach a "final" transition-id; note, if
357  // reorder==true, we have to go a bit further after this.
358  for (i = 0; i < len; i++) {
359  int32 tid = transition_ids_[i];
360  int32 this_phone = tmodel.TransitionIdToPhone(tid);
361  if (this_phone != phone && ! *error) { // error condition: should have reached final transition-id first.
362  *error = true;
363  KALDI_WARN << "Phone changed before final transition-id found "
364  "[broken lattice or mismatched model or wrong --reorder option?]";
365  }
366  if (tmodel.IsFinal(tid))
367  break;
368  }
369  if (i == len) return false; // fell off loop.
370  i++; // go past the one for which IsFinal returned true.
371  if (info.reorder) // we have to consume the following self-loop transition-ids.
372  while (i < len && tmodel.IsSelfLoop(transition_ids_[i])) i++;
373  if (i == len) return false; // we don't know if it ends here... so can't output arc.
374 
375  if (tmodel.TransitionIdToPhone(transition_ids_[i-1]) != phone
376  && ! *error) { // another check.
377  KALDI_WARN << "Phone changed unexpectedly in lattice "
378  "[broken lattice or mismatched model?]";
379  }
380  // interpret i as the number of transition-ids to consume.
381  std::vector<int32> tids_out(transition_ids_.begin(), transition_ids_.begin()+i);
382 
383  // consumed transition ids from our internal state.
384  *arc_out = CompactLatticeArc(info.silence_label, info.silence_label,
385  CompactLatticeWeight(weight_, tids_out), fst::kNoStateId);
386  transition_ids_.erase(transition_ids_.begin(), transition_ids_.begin()+i); // delete these
387  weight_ = LatticeWeight::One(); // we just output the weight.
388  return true;
389 }

443 bool LatticeWordAligner::ComputationState::OutputNormalWordArc(
444  const WordBoundaryInfo &info, const TransitionModel &tmodel,
445  CompactLatticeArc *arc_out, bool *error) {
446  if (transition_ids_.empty()) return false;
447  if (word_labels_.empty()) return false;
448  int32 begin_phone = tmodel.TransitionIdToPhone(transition_ids_[0]);
449  if (info.TypeOfPhone(begin_phone) != WordBoundaryInfo::kWordBeginPhone)
450  return false;
451  // we assume the start of transition_ids_ is the start of the phone.
452  // this is a precondition.
453  size_t len = transition_ids_.size(), i;
454 
455  // Eat up the transition-ids of this word-begin phone until we get to the
456  // "final" transition-id. [there may be self-loops following this though,
457  // if reorder==true]
458  for (i = 0; i < len && !tmodel.IsFinal(transition_ids_[i]); i++);
459  if (i == len) return false;
460  i++; // Skip over this final-transition.
461  if (info.reorder) // Skip over any reordered self-loops for this final-transition
462  for (; i < len && tmodel.IsSelfLoop(transition_ids_[i]); i++);
463  if (i == len) return false;
464  if (tmodel.TransitionIdToPhone(transition_ids_[i-1]) != begin_phone
465  && ! *error) { // another check.
466  KALDI_WARN << "Phone changed unexpectedly in lattice "
467  "[broken lattice or mismatched model?]";
468  *error = true;
469  }
470  // Now keep going till we hit a word-ending phone.
471  // Note: we don't expect anything except word-internal phones
472  // here, but we'll just print a warning if we get something
473  // else.
474  for (; i < len; i++) {
475  int32 this_phone = tmodel.TransitionIdToPhone(transition_ids_[i]);
476  if (info.TypeOfPhone(this_phone) == WordBoundaryInfo::kWordEndPhone)
477  break;
478  if (info.TypeOfPhone(this_phone) != WordBoundaryInfo::kWordInternalPhone
479  && !*error) {
480  KALDI_WARN << "Unexpected phone " << this_phone
481  << " found inside a word.";
482  *error = true;
483  }
484  }
485  if (i == len) return false;
486 
487  // OK, we hit a word-ending phone. Continue till we get to
488  // a "final-transition".
489 
490  // this variable just used for checks.
491  int32 final_phone = tmodel.TransitionIdToPhone(transition_ids_[i]);
492  for (; i < len; i++) {
493  int32 this_phone = tmodel.TransitionIdToPhone(transition_ids_[i]);
494  if (this_phone != final_phone && ! *error) {
495  *error = true;
496  KALDI_WARN << "Phone changed before final transition-id found "
497  "[broken lattice or mismatched model or wrong --reorder option?]";
498  }
499  if (tmodel.IsFinal(transition_ids_[i])) break;
500  }
501  if (i == len) return false;
502  i++;
503  // We got to the final-transition of the final phone;
504  // if reorder==true, continue eating up the self-loop.
505  if (info.reorder == true)
506  while (i < len && tmodel.IsSelfLoop(transition_ids_[i])) i++;
507  if (i == len) return false;
508  if (tmodel.TransitionIdToPhone(transition_ids_[i-1]) != final_phone
509  && ! *error) {
510  *error = true;
511  KALDI_WARN << "Phone changed while following final self-loop "
512  "[broken lattice or mismatched model or wrong --reorder option?]";
513  }
514 
515  // OK, we're ready to output the word.
516  // Interpret i as the number of transition-ids to consume.
517  std::vector<int32> tids_out(transition_ids_.begin(), transition_ids_.begin()+i);
518 
519  // consumed transition ids from our internal state.
520  int32 word = word_labels_[0];
521  *arc_out = CompactLatticeArc(word, word,
522  CompactLatticeWeight(weight_, tids_out),
523  fst::kNoStateId);
524  transition_ids_.erase(transition_ids_.begin(), transition_ids_.begin()+i); // delete these
525  word_labels_.erase(word_labels_.begin(), word_labels_.begin()+1); // remove the word we output.
526  weight_ = LatticeWeight::One(); // we just output the weight.
527  return true;
528 }

530 // Returns true if this vector of transition-ids could be a valid
531 // word. Note: the checks are not 100% exhaustive.
532 static bool IsPlausibleWord(const WordBoundaryInfo &info,
533  const TransitionModel &tmodel,
534  const std::vector<int32> &transition_ids) {
535  if (transition_ids.empty()) return false;
536  int32 first_phone = tmodel.TransitionIdToPhone(transition_ids.front()),
537  last_phone = tmodel.TransitionIdToPhone(transition_ids.back());
538  if ( (info.TypeOfPhone(first_phone) == WordBoundaryInfo::kWordBeginAndEndPhone
539  && first_phone == last_phone)
540  ||
541  (info.TypeOfPhone(first_phone) == WordBoundaryInfo::kWordBeginPhone &&
542  info.TypeOfPhone(last_phone) == WordBoundaryInfo::kWordEndPhone) ) {
543  if (! info.reorder) {
544  return (tmodel.IsFinal(transition_ids.back()));
545  } else {
546  int32 i = transition_ids.size() - 1;
547  while (i > 0 && tmodel.IsSelfLoop(transition_ids[i])) i--;
548  return tmodel.IsFinal(transition_ids[i]);
549  }
550  } else return false;
551 }

554 void LatticeWordAligner::ComputationState::OutputArcForce(
555  const WordBoundaryInfo &info, const TransitionModel &tmodel,
556  CompactLatticeArc *arc_out, bool *error) {
557 
558  KALDI_ASSERT(!IsEmpty());
559  if (!word_labels_.empty()
560  && !transition_ids_.empty()) { // We have at least one word to
561  // output, and some transition-ids. We assume that the normal OutputArc was called
562  // and failed, so this means we didn't see the end of that
563  // word.
564  int32 word = word_labels_[0];
565  if (! *error && !IsPlausibleWord(info, tmodel, transition_ids_)) {
566  *error = true;
567  KALDI_WARN << "Invalid word at end of lattice [partial lattice, forced out?]";
568  }
569  CompactLatticeWeight cw(weight_, transition_ids_);
570  *arc_out = CompactLatticeArc(word, word, cw, fst::kNoStateId);
571  weight_ = LatticeWeight::One();
572  transition_ids_.clear();
573  word_labels_.erase(word_labels_.begin(), word_labels_.begin()+1);
574  } else if (!word_labels_.empty() && transition_ids_.empty()) {
575  // We won't create arcs with these word labels on, as most likely
576  // this will cause errors down the road. This is an error
577  // condition anyway, in some sense.
578  if (! *error) {
579  *error = true;
580  KALDI_WARN << "Discarding word-ids at the end of a sentence, "
581  "that don't have alignments.";
582  }
583  CompactLatticeWeight cw(weight_, transition_ids_);
584  // This creates an epsilon arc with a weight on it, but
585  // no transition-ids since the vector is empty.
586  // The word labels are discarded.
587  *arc_out = CompactLatticeArc(0, 0, cw, fst::kNoStateId);
588  weight_ = LatticeWeight::One();
589  word_labels_.clear();
590  } else if (!transition_ids_.empty() && word_labels_.empty()) {
591  // Transition-ids but no word label-- either silence or partial word.
592  int32 first_phone = tmodel.TransitionIdToPhone(transition_ids_[0]);
593  if (info.TypeOfPhone(first_phone) == WordBoundaryInfo::kNonWordPhone) {
594  // first phone is silence...
595  if (first_phone != tmodel.TransitionIdToPhone(transition_ids_.back())
596  && ! *error) {
597  *error = true;
598  // Phone changed-- this is a code error, because the regular OutputArc
599  // should have output an arc (a silence arc) if that phone finished.
600  // So we make it fatal.
601  KALDI_ERR << "Broken silence arc at end of utterance (the phone "
602  "changed); code error";
603  }
604  if (!*error) { // Check that it ends at the end state of silence; error otherwise.
605  int32 i = transition_ids_.size() - 1;
606  if (info.reorder)
607  while(tmodel.IsSelfLoop(transition_ids_[i]) && i > 0) i--;
608  if (!tmodel.IsFinal(transition_ids_[i])) {
609  *error = true;
610  KALDI_WARN << "Broken silence arc at end of utterance (does not "
611  "reach end of silence)";
612  }
613  }
614  CompactLatticeWeight cw(weight_, transition_ids_);
615  *arc_out = CompactLatticeArc(info.silence_label, info.silence_label,
616  cw, fst::kNoStateId);
617  } else {
618  // Not silence phone -- treat as partial word (with no word label).
619  // This is in itself an error condition, i.e. the lattice was maybe
620  // forced out.
621  if (! *error) {
622  *error = true;
623  KALDI_WARN << "Partial word detected at end of utterance";
624  }
625  CompactLatticeWeight cw(weight_, transition_ids_);
626  *arc_out = CompactLatticeArc(info.partial_word_label, info.partial_word_label,
627  cw, fst::kNoStateId);
628  }
629  transition_ids_.clear();
630  weight_ = LatticeWeight::One();
631  } else {
632  KALDI_ERR << "Code error, word-aligning lattice"; // this shouldn't
633  // be able to happen; we don't call this function of they're both empty.
634  }
635 }

687  std::string line;

695  KALDI_ASSERT(p > 0);

700  else if (t == "begin") phone_to_type[p] = kWordBeginPhone;
701  else if (t == "singleton") phone_to_type[p] = kWordBeginAndEndPhone;
702  else if (t == "end") phone_to_type[p] = kWordEndPhone;
703  else if (t == "internal") phone_to_type[p] = kWordInternalPhone;
704  else
705  KALDI_ERR << "Invalid line in word-boundary file: " << line;
706  }
707  if (phone_to_type.empty())
708  KALDI_ERR << "Empty word-boundary file";

723  public:
724  WordAlignedLatticeTester(const CompactLattice &lat,

730  void Test() {
731  // First test that each aligned arc is valid.
732  typedef CompactLattice::StateId StateId ;
733  for (StateId s = 0; s < aligned_lat_.NumStates(); s++) {
734  for (fst::ArcIterator<CompactLattice> iter(aligned_lat_, s);
735  !iter.Done();
736  iter.Next()) {
737  TestArc(iter.Value());
738  }
739  if (aligned_lat_.Final(s) != CompactLatticeWeight::Zero()) {
740  TestFinal(aligned_lat_.Final(s));
741  }
742  }
743  TestEquivalent();
744  }
745  private:
746  void TestArc(const CompactLatticeArc &arc) {
747  if (! (TestArcSilence(arc) || TestArcNormalWord(arc) || TestArcOnePhoneWord(arc)
748  || TestArcEmpty(arc)))
749  KALDI_ERR << "Invalid arc in aligned CompactLattice: "
750  << arc.ilabel << " " << arc.olabel << " " << arc.nextstate
751  << " " << arc.weight;
752  }
753  bool TestArcEmpty(const CompactLatticeArc &arc) {
754  if (arc.ilabel != 0) return false; // Check there is no label. Note, ilabel==olabel.
755  const std::vector<int32> &tids = arc.weight.String();
756  return tids.empty();
757  }
758  bool TestArcSilence(const CompactLatticeArc &arc) {
759  // This only applies when silence doesn't have word labels.
760  if (arc.ilabel != info_.silence_label) return false; // Check the label is
761  // the silence label. Note, ilabel==olabel.
762  const std::vector<int32> &tids = arc.weight.String();
763  if (tids.empty()) return false;
764  int32 first_phone = tmodel_.TransitionIdToPhone(tids.front());
765  if (info_.TypeOfPhone(first_phone) != WordBoundaryInfo::kNonWordPhone)
766  return false;
767  for (size_t i = 0; i < tids.size(); i++)
768  if (tmodel_.TransitionIdToPhone(tids[i]) != first_phone) return false;
769 
770  if (!info_.reorder) return tmodel_.IsFinal(tids.back());
771  else {
772  for (size_t i = 0; i < tids.size(); i++) {
773  if(tmodel_.IsFinal(tids[i])) { // got the "final" transition, which is
774  // reordered to actually not be final. Make sure that all the
775  // rest of the transition ids are the self-loop of that same
776  // transition-state.
777  for(size_t j = i+1; j < tids.size(); j++) {
778  if (!(tmodel_.TransitionIdToTransitionState(tids[j])
779  == tmodel_.TransitionIdToTransitionState(tids[i]))) return false;
780  }
781  return true;
782  }
783  }
784  return false; // fell off loop. No final-state present.
785  }
786  }
787 
788  bool TestArcOnePhoneWord(const CompactLatticeArc &arc) {
789  if (arc.ilabel == 0) return false; // Check there's a label. Note, ilabel==olabel.
790  const std::vector<int32> &tids = arc.weight.String();
791  if (tids.empty()) return false;
792  int32 first_phone = tmodel_.TransitionIdToPhone(tids.front());
793  if (info_.TypeOfPhone(first_phone) !=
794  WordBoundaryInfo::kWordBeginAndEndPhone) return false;
795  for (size_t i = 0; i < tids.size(); i++)
796  if (tmodel_.TransitionIdToPhone(tids[i]) != first_phone) return false;
797 
798  if (!info_.reorder) return tmodel_.IsFinal(tids.back());
799  else {
800  for (size_t i = 0; i < tids.size(); i++) {
801  if(tmodel_.IsFinal(tids[i])) { // got the "final" transition, which is
802  // reordered to actually not be final. Make sure that all the
803  // rest of the transition ids are the self-loop of that same
804  // transition-state.
805  for(size_t j = i+1; j < tids.size(); j++) {
806  if (tmodel_.TransitionIdToTransitionState(tids[j])
807  != tmodel_.TransitionIdToTransitionState(tids[i])) return false;
808  }
809  return true;
810  }
811  }
812  return false; // fell off loop. No final-state present.
813  }
814  }
815 
816  bool TestArcNormalWord(const CompactLatticeArc &arc) {
817  if (arc.ilabel == 0) return false; // Check there's a label. Note, ilabel==olabel.
818  const std::vector<int32> &tids = arc.weight.String();
819  if (tids.empty()) return false;
820  int32 first_phone = tmodel_.TransitionIdToPhone(tids.front());
821  if (info_.TypeOfPhone(first_phone) != WordBoundaryInfo::kWordBeginPhone)
822  return false;
823  size_t i;
824  { // first phone.
825  int num_final = 0;
826  for (i = 0; i < tids.size(); i++) {
827  if (tmodel_.IsFinal(tids[i])) num_final++;
828  if (tmodel_.TransitionIdToPhone(tids[i]) != first_phone) break;
829  }
830  if (num_final != 1)
831  return false; // Something went wrong-- perhaps we
832  // got two beginning phones in a row.
833  }
834  { // middle phones. Skip over them.
835  while (i < tids.size() &&
836  info_.TypeOfPhone(tmodel_.TransitionIdToPhone(tids[i]))
837  == WordBoundaryInfo::kWordInternalPhone)
838  i++;
839  }
840  if (i == tids.size()) return false;
841  int32 final_phone = tmodel_.TransitionIdToPhone(tids[i]);
842  if (info_.TypeOfPhone(final_phone) != WordBoundaryInfo::kWordEndPhone)
843  return false; // not word-ending.
844  for (size_t j = i; j < tids.size(); j++) // make sure only this final phone till end.
845  if (tmodel_.TransitionIdToPhone(tids[j]) != final_phone)
846  return false; // Other phones after final phone.
847 
848  for (size_t j = i; j < tids.size(); j++) {
849  if (tmodel_.IsFinal(tids[j])) { // Found "final transition".. Note:
850  // may be "reordered" with its self loops.
851  if (!info_.reorder) return (j+1 == tids.size());
852  else {
853  // Make sure the only thing that follows this is self-loops
854  // of the final transition-state.
855  for (size_t k=j+1; k<tids.size(); k++)
856  if (tmodel_.TransitionIdToTransitionState(tids[k])
857  != tmodel_.TransitionIdToTransitionState(tids[j])
858  || !tmodel_.IsSelfLoop(tids[k]))
859  return false;
860  return true;
861  }
862  }
863  }
864  return false; // Found no final state.
865  }
866 
867  bool TestArcPartialWord(const CompactLatticeArc &arc) {
868  if (arc.ilabel != info_.partial_word_label) return false; // label should
869  // be the partial-word label.
870  const std::vector<int32> &tids = arc.weight.String();
871  if (tids.empty()) return false;
872  return true; // We're pretty liberal when it comes to partial words here.
873  }

877  KALDI_ERR << "Expect to have no strings on final-weights of lattices.";
878  }

880  CompactLattice aligned_lat(aligned_lat_);
881  if (info_.silence_label != 0) { // remove silence labels.

889  200/*path length (max?)*/))
890  KALDI_ERR << "Equivalence test failed (testing word-alignment of lattices.) "
891  << "Make sure your model and lattices match!";
892  }

897  const CompactLattice &aligned_lat_;
