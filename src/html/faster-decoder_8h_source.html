105  class Token {
106  public:
107  Arc arc_; // contains only the graph part of the cost;
108  // we can work out the acoustic part from difference between
109  // "cost_" and prev->cost_.
110  Token *prev_;
111  int32 ref_count_;
112  // if you are looking for weight_ here, it was removed and now we just have
113  // cost_, which corresponds to ConvertToCost(weight_).
114  double cost_;
115  inline Token(const Arc &arc, BaseFloat ac_cost, Token *prev):
116  arc_(arc), prev_(prev), ref_count_(1) {
117  if (prev) {
118  prev->ref_count_++;
119  cost_ = prev->cost_ + arc.weight.Value() + ac_cost;
120  } else {
121  cost_ = arc.weight.Value() + ac_cost;
122  }
123  }
124  inline Token(const Arc &arc, Token *prev):
125  arc_(arc), prev_(prev), ref_count_(1) {
126  if (prev) {
127  prev->ref_count_++;
128  cost_ = prev->cost_ + arc.weight.Value();
129  } else {
130  cost_ = arc.weight.Value();
131  }
132  }
133  inline bool operator < (const Token &other) {
134  return cost_ > other.cost_;
135  }

137  inline static void TokenDelete(Token *tok) {
138  while (--tok->ref_count_ == 0) {
139  Token *prev = tok->prev_;
140  delete tok;

142  else tok = prev;
143  }
