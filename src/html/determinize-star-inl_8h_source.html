192  void Determinize(bool *debug_ptr) {
193  assert(!determinized_);
194  // This determinizes the input fst but leaves it in the "special format"
195  // in "output_arcs_".
196  InputStateId start_id = ifst_->Start();
197  if (start_id == kNoStateId) { determinized_ = true; return; } // Nothing to do.
198  else { // Insert start state into hash and queue.
199  Element elem;
200  elem.state = start_id;
201  elem.weight = Weight::One();
202  elem.string = repository_.IdOfEmpty(); // Id of empty sequence.
203  vector<Element> vec;
204  vec.push_back(elem);
205  OutputStateId cur_id = SubsetToStateId(vec);
206  assert(cur_id == 0 && "Do not call Determinize twice.");
207  }
208  while (!Q_.empty()) {
209  pair<vector<Element>*, OutputStateId> cur_pair = Q_.front();
210  Q_.pop_front();
211  ProcessSubset(cur_pair);
212  if (debug_ptr && *debug_ptr) Debug(); // will exit.
213  if (max_states_ > 0 && output_arcs_.size() > max_states_) {
214  if (allow_partial_ == false) {
215  std::cerr << "Determinization aborted since passed " << max_states_
216  << " states.\n";
217  throw std::runtime_error("max-states reached in determinization");
218  } else {
219  KALDI_WARN << "Determinization terminated since passed " << max_states_
220  << " states, partial results will be generated.";
221  is_partial_ = true;
222  break;
223  }
224  }
225  }
226  determinized_ = true;
227  }

281  // Hashing function used in hash of subsets.
282  // A subset is a pointer to vector<Element>.
283  // The Elements are in sorted order on state id, and without repeated states.
284  // Because the order of Elements is fixed, we can use a hashing function that is
285  // order-dependent. However the weights are not included in the hashing function--
286  // we hash subsets that differ only in weight to the same key. This is not optimal
287  // in terms of the O(N) performance but typically if we have a lot of determinized
288  // states that differ only in weight then the input probably was pathological in some way,
289  // or even non-determinizable.
290  // We don't quantize the weights, in order to avoid inexactness in simple cases.
291  // Instead we apply the delta when comparing subsets for equality, and allow a small
292  // difference.
293 
294  class SubsetKey {
295  public:
296  size_t operator ()(const vector<Element> * subset) const { // hashes only the state and string.
297  size_t hash = 0, factor = 1;
298  for (typename vector<Element>::const_iterator iter = subset->begin();
299  iter != subset->end(); ++iter) {
300  hash *= factor;
301  hash += iter->state + 103333*iter->string;
302  factor *= 23531; // these numbers are primes.
303  }
304  return hash;
305  }
306  };

308  // This is the equality operator on subsets. It checks for exact match on state-id
309  // and string, and approximate match on weights.
310  class SubsetEqual {
311  public:
312  bool operator ()(const vector<Element> *s1,
313  const vector<Element> *s2) const {
314  size_t sz = s1->size();
315  assert(sz >= 0);
316  if (sz != s2->size()) return false;
317  typename vector<Element>::const_iterator iter1 = s1->begin(),
318  iter1_end = s1->end(), iter2 = s2->begin();
319  for (; iter1 < iter1_end; ++iter1, ++iter2) {
320  if (iter1->state != iter2->state ||
321  iter1->string != iter2->string ||
322  ! ApproxEqual(iter1->weight, iter2->weight, delta_))
323  return false;
324  }
325  return true;
326  }
327  float delta_;
328  SubsetEqual(float delta): delta_(delta) {}
329  SubsetEqual(): delta_(kDelta) {}
330  };

441  void ProcessFinal(const vector<Element> &closed_subset, OutputStateId state) {
442  // processes final-weights for this subset.
443  bool is_final = false;
444  StringId final_string = 0; // = 0 to keep compiler happy.
445  Weight final_weight = Weight::One(); // This value will never be accessed, and
446  // we just set it to avoid spurious compiler warnings. We avoid setting it
447  // to Zero() because floating-point infinities can sometimes generate
448  // interrupts and slow things down.
449  typename vector<Element>::const_iterator iter = closed_subset.begin(),
450  end = closed_subset.end();
451  for (; iter != end; ++iter) {
452  const Element &elem = *iter;
453  Weight this_final_weight = ifst_->Final(elem.state);
454  if (this_final_weight != Weight::Zero()) {
455  if (!is_final) { // first final-weight
456  final_string = elem.string;
457  final_weight = Times(elem.weight, this_final_weight);
458  is_final = true;
459  } else { // already have one.
460  if (final_string != elem.string) {
461  std::cerr << "DeterminizerStar: FST was not functional -> not determinizable\n";
462  throw std::runtime_error("Non-functional FST: cannot determinize.\n");
463  }
464  final_weight = Plus(final_weight, Times(elem.weight, this_final_weight));
465  }
466  }
467  }
468  if (is_final) {
469  // store final weights in TempArc structure, just like a transition.
470  TempArc temp_arc;
471  temp_arc.ilabel = 0;
472  temp_arc.nextstate = kNoStateId; // special marker meaning "final weight".
473  temp_arc.ostring = final_string;
474  temp_arc.weight = final_weight;
475  output_arcs_[state].push_back(temp_arc);
476  }
477  }

499  // ProcessTransitions handles transitions out of this subset of states.
500  // Ignores epsilon transitions (epsilon closure already handled that).
501  // Does not consider final states. Breaks the transitions up by ilabel,
502  // and creates a new transition in determinized FST, for each ilabel.
503  // Does this by creating a big vector of pairs <Label, Element> and then sorting them
504  // using a lexicographical ordering, and calling ProcessTransition for each range
505  // with the same ilabel.
506  // Side effects on repository, and (via ProcessTransition) on Q_, hash_,
507  // and output_arcs_.
508  void ProcessTransitions(const vector<Element> &closed_subset, OutputStateId state) {
509  vector<pair<Label, Element> > all_elems;
510  { // Push back into "all_elems", elements corresponding to all non-epsilon-input transitions
511  // out of all states in "closed_subset".
512  typename vector<Element>::const_iterator iter = closed_subset.begin(),
513  end = closed_subset.end();
514  for (; iter != end; ++iter) {
515  const Element &elem = *iter;
516  for (ArcIterator<Fst<Arc> > aiter(*ifst_, elem.state);
517  !aiter.Done(); aiter.Next()) {
518  const Arc &arc = aiter.Value();
519  if (arc.ilabel != 0) { // Non-epsilon transition -- ignore epsilons here.
520  pair<Label, Element> this_pr;
521  this_pr.first = arc.ilabel;
522  Element &next_elem(this_pr.second);
523  next_elem.state = arc.nextstate;
524  next_elem.weight = Times(elem.weight, arc.weight);
525  if (arc.olabel == 0) // output epsilon-- this is simple case so
526  // handle separately for efficiency
527  next_elem.string = elem.string;
528  else {
529  vector<Label> seq;
530  repository_.SeqOfId(elem.string, &seq);
531  seq.push_back(arc.olabel);
532  next_elem.string = repository_.IdOfSeq(seq);
533  }
534  all_elems.push_back(this_pr);
535  }
536  }
537  }
538  }
539  PairComparator pc;
540  std::sort(all_elems.begin(), all_elems.end(), pc);
541  // now sorted first on input label, then on state.
542  typedef typename vector<pair<Label, Element> >::const_iterator PairIter;
543  PairIter cur = all_elems.begin(), end = all_elems.end();
544  vector<Element> this_subset;
545  while (cur != end) {
546  // Process ranges that share the same input symbol.
547  Label ilabel = cur->first;
548  this_subset.clear();
549  while (cur != end && cur->first == ilabel) {
550  this_subset.push_back(cur->second);
551  cur++;
552  }
553  // We now have a subset for this ilabel.
554  ProcessTransition(state, ilabel, &this_subset);
555  }
556  }

558  // SubsetToStateId converts a subset (vector of Elements) to a StateId in the output
559  // fst. This is a hash lookup; if no such state exists, it adds a new state to the hash
560  // and adds a new pair to the queue.
561  // Side effects on hash_ and Q_, and on output_arcs_ [just affects the size].
562  OutputStateId SubsetToStateId(const vector<Element> &subset) { // may add the subset to the queue.
563  typedef typename SubsetHash::iterator IterType;
564  IterType iter = hash_.find(&subset);
565  if (iter == hash_.end()) { // was not there.
566  vector<Element> *new_subset = new vector<Element>(subset);
567  OutputStateId new_state_id = (OutputStateId) output_arcs_.size();
568  bool ans = hash_.insert(std::pair<const vector<Element>*,
569  OutputStateId>(new_subset,
570  new_state_id)).second;
571  assert(ans);
572  output_arcs_.push_back(vector<TempArc>());
573  if (allow_partial_ == false) {
574  // If --allow-partial is not requested, we do the old way.
575  Q_.push_front(pair<vector<Element>*, OutputStateId>(new_subset, new_state_id));
576  } else {
577  // If --allow-partial is requested, we do breadth first search. This
578  // ensures that when we return partial results, we return the states
579  // that are reachable by the fewest steps from the start state.
580  Q_.push_back(pair<vector<Element>*, OutputStateId>(new_subset, new_state_id));
581  }
582  return new_state_id;
583  } else {
584  return iter->second; // the OutputStateId.
585  }
586  }

590  // transitions out of it and adds new determinized states to the queue if necessary.
591  // The first stage is "EpsilonClosure" (follow epsilons to get a possibly larger set
592  // of (states, weights)). After that we ignore epsilons. We process the final-weight
593  // of the state, and then handle transitions out (this may add more determinized states
594  // to the queue).
595  void ProcessSubset(const pair<vector<Element>*, OutputStateId> & pair) {

602  // Now follow non-epsilon arcs [and also process final states]
603  ProcessFinal(closed_subset, state);
604 
605  // Now handle transitions out of these states.
606  ProcessTransitions(closed_subset, state);
607  }

657 template<class F>
658 void DeterminizerStar<F>::EpsilonClosure::
659  GetEpsilonClosure(const vector<Element> &input_subset,
660  vector<Element> *output_subset) {
661  ecinfo_.resize(0);
662  size_t size = input_subset.size();
663  // find whether input fst is known to be sorted in input label.
664  bool sorted =
665  ((ifst_->Properties(kILabelSorted, false) & kILabelSorted) != 0);
666 
667  // size is still the input_subset.size()
668  for (size_t i = 0; i < size; i++) {
669  ExpandOneElement(input_subset[i], sorted, input_subset[i].weight, true);
670  }
671 
672  size_t s = queue_2_.size();
673  if (s == 0) {
674  *output_subset = input_subset;
675  return;
676  } else {
677  // queue_2 not empty. Need to create the vector<info>
678  for (size_t i = 0; i < size; i++) {
679  // the weight has not been processed yet,
680  // so put all of them in the "weight_to_process"
681  ecinfo_.push_back(EpsilonClosureInfo(input_subset[i],
682  input_subset[i].weight,
683  false));
684  ecinfo_.back().element.weight = Weight::Zero(); // clear the weight
685 
686  if (id_to_index_.size() < input_subset[i].state + 1) {
687  id_to_index_.resize(2 * input_subset[i].state + 1, -1);
688  }
689  id_to_index_[input_subset[i].state] = ecinfo_.size() - 1;
690  }
691  }
692 
693  {
694  Element elem;
695  elem.weight = Weight::Zero();
696  for (size_t i = 0; i < s; i++) {
697  elem.state = queue_2_[i].state;
698  elem.string = queue_2_[i].string;
699  AddOneElement(elem, queue_2_[i].weight);
700  }
701  queue_2_.resize(0);
702  }
703 
704  int counter = 0; // relates to max-states option, used for test.
705  while (!queue_.empty()) {
706  InputStateId id = queue_.front();
707 
708  // no need to check validity of the index
709  // since anything in the queue we are sure they're in the "virtual set"
710  int index = id_to_index_[id];
711  EpsilonClosureInfo &info = ecinfo_[index];
712  Element &elem = info.element;
713  Weight unprocessed_weight = info.weight_to_process;
714 
715  elem.weight = Plus(elem.weight, unprocessed_weight);
716  info.weight_to_process = Weight::Zero();
717 
718  info.in_queue = false;
719  queue_.pop_front();
720 
721  if (max_states_ > 0 && counter++ > max_states_) {
722  std::cerr << "Determinization aborted since looped more than "
723  << max_states_ << " times during epsilon closure.\n";
724  throw std::runtime_error("looped more than max-states times"
725  " in determinization");
726  }
727 
728  // generally we need to be careful about iterator-invalidation problem
729  // here we pass a reference (elem), which could be an issue.
730  // In the beginning of ExpandOneElement, we make a copy of elem.string
731  // to avoid that issue
732  ExpandOneElement(elem, sorted, unprocessed_weight);
733  }
734 
735  {
736  // this sorting is based on StateId
737  sort(ecinfo_.begin(), ecinfo_.end());
738 
739  output_subset->clear();
740 
741  size = ecinfo_.size();
742  output_subset->reserve(size);
743  for (size_t i = 0; i < size; i++) {
744  EpsilonClosureInfo& info = ecinfo_[i];
745  if (info.weight_to_process != Weight::Zero()) {
746  info.element.weight = Plus(info.element.weight, info.weight_to_process);
747  }
748  output_subset->push_back(info.element);
749  }
750  }

753 template<class F>
754 void DeterminizerStar<F>::EpsilonClosure::
755  AddOneElement(const Element &elem, const Weight &unprocessed_weight) {
756  // first we try to find the element info in the ecinfo_ vector
757  int index = -1;
758  if (elem.state < id_to_index_.size()) {
759  index = id_to_index_[elem.state];
760  }
761  if (index != -1) {
762  if (index >= ecinfo_.size()) {
763  index = -1;
764  }
765  // since ecinfo_ might store outdated information, we need to check
766  else if (ecinfo_[index].element.state != elem.state) {
767  index = -1;
768  }
769  }
770 
771  if (index == -1) {
772  // was no such StateId: insert and add to queue.
773  ecinfo_.push_back(EpsilonClosureInfo(elem, unprocessed_weight, true));
774  size_t size = id_to_index_.size();
775  if (size < elem.state + 1) {
776  // double the size to reduce memory operations
777  id_to_index_.resize(2 * elem.state + 1, -1);
778  }
779  id_to_index_[elem.state] = ecinfo_.size() - 1;
780  queue_.push_back(elem.state);
781 
782  } else { // one is already there. Add weights.
783  EpsilonClosureInfo &info = ecinfo_[index];
784  if (info.element.string != elem.string) {
785  std::cerr << "DeterminizerStar: FST was not functional "
786  "-> not determinizable\n";
787  { // Print some debugging information. Can be helpful to debug
788  // the inputs when FSTs are mysteriously non-functional.
789  vector<Label> tmp_seq;
790  repository_->SeqOfId(info.element.string, &tmp_seq);
791  std::cerr << "First string: ";
792  for (size_t i = 0; i < tmp_seq.size(); i++)
793  std::cerr << tmp_seq[i] << " ";
794  std::cerr << "\nSecond string: ";
795  repository_->SeqOfId(elem.string, &tmp_seq);
796  for (size_t i = 0; i < tmp_seq.size(); i++)
797  std::cerr << tmp_seq[i] << " ";
798  std::cerr << "\n";
799  }
800  throw std::runtime_error("Non-functional FST: cannot determinize.\n");
801  }
802 
803  info.weight_to_process =
804  Plus(info.weight_to_process, unprocessed_weight);
805 
806  if (!info.in_queue) {
807  // this is because the code in "else" below: the
808  // iter->second.weight_to_process might not be Zero()
809  Weight weight = Plus(info.element.weight, info.weight_to_process);
810 
811  // What is done below is, we propagate the weight (by adding them
812  // to the queue only when the change is big enough;
813  // otherwise we just store the weight, until before returning
814  // we add the element.weight and weight_to_process together
815  if (! ApproxEqual(weight, info.element.weight, delta_)) {
816  // add extra part of weight to queue.
817  info.in_queue = true;
818  queue_.push_back(elem.state);
819  }
820  }
821  }

824 template<class F>
825 void DeterminizerStar<F>::EpsilonClosure::ExpandOneElement(
826  const Element &elem,
827  bool sorted,
828  const Weight &unprocessed_weight,
829  bool save_to_queue_2) {
830  StringId str = elem.string; // copy it here because there is an iterator-
831  // - invalidation problem (it really happens for some FSTs)
832 
833  // now we are going to propagate the "unprocessed_weight"
834  for (ArcIterator<Fst<Arc> > aiter(*ifst_, elem.state);
835  !aiter.Done(); aiter.Next()) {
836  const Arc &arc = aiter.Value();
837  if (sorted && arc.ilabel > 0) {
838  break;
839  // Break from the loop: due to sorting there will be no
840  // more transitions with epsilons as input labels.
841  }
842  if (arc.ilabel != 0) {
843  continue; // we only process epsilons here
844  }
845  Element next_elem;
846  next_elem.state = arc.nextstate;
847  next_elem.weight = Weight::Zero();
848  Weight next_unprocessed_weight
849  = Times(unprocessed_weight, arc.weight);
850 
851  // now must append strings
852  if (arc.olabel == 0) {
853  next_elem.string = str;
854  } else {
855  vector<Label> seq;
856  repository_->SeqOfId(str, &seq);
857  if (arc.olabel != 0)
858  seq.push_back(arc.olabel);
859  next_elem.string = repository_->IdOfSeq(seq);
860  }
861  if (save_to_queue_2) {
862  next_elem.weight = next_unprocessed_weight;
863  queue_2_.push_back(next_elem);
864  } else {
865  AddOneElement(next_elem, next_unprocessed_weight);
866  }
867  }

870 template<class F>
871 void DeterminizerStar<F>::Output(MutableFst<GallicArc<Arc> > *ofst,
872  bool destroy) {
873  assert(determinized_);
874  if (destroy) determinized_ = false;
875  typedef GallicWeight<Label, Weight> ThisGallicWeight;
876  typedef typename Arc::StateId StateId;
877  if (destroy)
878  FreeMostMemory();
879  StateId nStates = static_cast<StateId>(output_arcs_.size());
880  ofst->DeleteStates();
881  ofst->SetStart(kNoStateId);
882  if (nStates == 0) {
883  return;
884  }
885  for (StateId s = 0;s < nStates;s++) {
886  OutputStateId news = ofst->AddState();
887  assert(news == s);
888  }
889  ofst->SetStart(0);
890  // now process transitions.
891  for (StateId this_state = 0; this_state < nStates; this_state++) {
892  vector<TempArc> &this_vec(output_arcs_[this_state]);
893  typename vector<TempArc>::const_iterator iter = this_vec.begin(),
894  end = this_vec.end();
895  for (; iter != end; ++iter) {
896  const TempArc &temp_arc(*iter);
897  GallicArc<Arc> new_arc;
898  vector<Label> seq;
899  repository_.SeqOfId(temp_arc.ostring, &seq);
900  StringWeight<Label, STRING_LEFT> string_weight;
901  for (size_t i = 0;i < seq.size();i++) string_weight.PushBack(seq[i]);
902  ThisGallicWeight gallic_weight(string_weight, temp_arc.weight);
903 
904  if (temp_arc.nextstate == kNoStateId) { // is really final weight.
905  ofst->SetFinal(this_state, gallic_weight);
906  } else { // is really an arc.
907  new_arc.nextstate = temp_arc.nextstate;
908  new_arc.ilabel = temp_arc.ilabel;
909  new_arc.olabel = temp_arc.ilabel; // acceptor. input == output.
910  new_arc.weight = gallic_weight; // includes string and weight.
911  ofst->AddArc(this_state, new_arc);
912  }
913  }
914  // Free up memory. Do this inside the loop as ofst is also allocating memory
915  if (destroy) { vector<TempArc> temp; temp.swap(this_vec); }
916  }
917  if (destroy) { vector<vector<TempArc> > temp; temp.swap(output_arcs_); }
918 }

920 template<class F>
921 void DeterminizerStar<F>::Output(MutableFst<Arc> *ofst, bool destroy) {
922  assert(determinized_);
923  if (destroy) determinized_ = false;
924  // Outputs to standard fst.
925  OutputStateId num_states = static_cast<OutputStateId>(output_arcs_.size());
926  if (destroy)
927  FreeMostMemory();
928  ofst->DeleteStates();
929  if (num_states == 0) {
930  ofst->SetStart(kNoStateId);
931  return;
932  }
933  // Add basic states-- but will add extra ones to account for strings on output.
934  for (OutputStateId s = 0; s < num_states; s++) {
935  OutputStateId news = ofst->AddState();
936  assert(news == s);
937  }
938  ofst->SetStart(0);
939  for (OutputStateId this_state = 0; this_state < num_states; this_state++) {
940  vector<TempArc> &this_vec(output_arcs_[this_state]);
941 
942  typename vector<TempArc>::const_iterator iter = this_vec.begin(),
943  end = this_vec.end();
944  for (; iter != end; ++iter) {
945  const TempArc &temp_arc(*iter);
946  vector<Label> seq;
947  repository_.SeqOfId(temp_arc.ostring, &seq);
948  if (temp_arc.nextstate == kNoStateId) { // Really a final weight.
949  // Make a sequence of states going to a final state, with the strings as labels.
950  // Put the weight on the first arc.
951  OutputStateId cur_state = this_state;
952  for (size_t i = 0; i < seq.size();i++) {
953  OutputStateId next_state = ofst->AddState();
954  Arc arc;
955  arc.nextstate = next_state;
956  arc.weight = (i == 0 ? temp_arc.weight : Weight::One());
957  arc.ilabel = 0; // epsilon.
958  arc.olabel = seq[i];
959  ofst->AddArc(cur_state, arc);
960  cur_state = next_state;
961  }
962  ofst->SetFinal(cur_state, (seq.size() == 0 ? temp_arc.weight : Weight::One()));
963  } else { // Really an arc.
964  OutputStateId cur_state = this_state;
965  // Have to be careful with this integer comparison (i+1 < seq.size()) because unsigned.
966  // i < seq.size()-1 could fail for zero-length sequences.
967  for (size_t i = 0; i+1 < seq.size();i++) {
968  // for all but the last element of seq, create new state.
969  OutputStateId next_state = ofst->AddState();
970  Arc arc;
971  arc.nextstate = next_state;
972  arc.weight = (i == 0 ? temp_arc.weight : Weight::One());
973  arc.ilabel = (i == 0 ? temp_arc.ilabel : 0); // put ilabel on first element of seq.
974  arc.olabel = seq[i];
975  ofst->AddArc(cur_state, arc);
976  cur_state = next_state;
977  }
978  // Add the final arc in the sequence.
979  Arc arc;
980  arc.nextstate = temp_arc.nextstate;
981  arc.weight = (seq.size() <= 1 ? temp_arc.weight : Weight::One());
982  arc.ilabel = (seq.size() <= 1 ? temp_arc.ilabel : 0);
983  arc.olabel = (seq.size() > 0 ? seq.back() : 0);
984  ofst->AddArc(cur_state, arc);
985  }
986  }
987  // Free up memory. Do this inside the loop as ofst is also allocating memory
988  if (destroy) { vector<TempArc> temp; temp.swap(this_vec); }
989  }
990  if (destroy) {
991  vector<vector<TempArc> > temp;
992  temp.swap(output_arcs_);
993  repository_.Destroy();
994  }

997 template<class F> void DeterminizerStar<F>::
998 ProcessTransition(OutputStateId state, Label ilabel, vector<Element> *subset) {
999  // At input, "subset" may contain duplicates for a given dest state (but in sorted
1000  // order). This function removes duplicates from "subset", normalizes it, and adds
1001  // a transition to the dest. state (possibly affecting Q_ and hash_, if state did not
1002  // exist).
1003 
1004  typedef typename vector<Element>::iterator IterType;
1005  { // This block makes the subset have one unique Element per state, adding the weights.
1006  IterType cur_in = subset->begin(), cur_out = cur_in, end = subset->end();
1007  size_t num_out = 0;
1008  // Merge elements with same state-id
1009  while (cur_in != end) { // while we have more elements to process.
1010  // At this point, cur_out points to location of next place we want to put an element,
1011  // cur_in points to location of next element we want to process.
1012  if (cur_in != cur_out) *cur_out = *cur_in;
1013  cur_in++;
1014  while (cur_in != end && cur_in->state == cur_out->state) { // merge elements.
1015  if (cur_in->string != cur_out->string) {
1016  std::cerr << "DeterminizerStar: FST was not functional -> not determinizable\n";
1017  throw std::runtime_error("Non-functional FST: cannot determinize.\n");
1018  }
1019  cur_out->weight = Plus(cur_out->weight, cur_in->weight);
1020  cur_in++;
1021  }
1022  cur_out++;
1023  num_out++;
1024  }
1025  subset->resize(num_out);
1026  }
1027 
1028  StringId common_str;
1029  Weight tot_weight;
1030  { // This block computes common_str and tot_weight (essentially: the common divisor)
1031  // and removes them from the elements.
1032  vector<Label> seq;
1033 
1034  IterType begin = subset->begin(), iter, end = subset->end();
1035  { // This block computes "seq", which is the common prefix, and "common_str",
1036  // which is the StringId version of "seq".
1037  vector<Label> tmp_seq;
1038  for (iter = begin; iter != end; ++iter) {
1039  if (iter == begin) {
1040  repository_.SeqOfId(iter->string, &seq);
1041  } else {
1042  repository_.SeqOfId(iter->string, &tmp_seq);
1043  if (tmp_seq.size() < seq.size()) seq.resize(tmp_seq.size()); // size of shortest one.
1044  for (size_t i = 0; i < seq.size(); i++) // seq.size() is the shorter one at this point.
1045  if (tmp_seq[i] != seq[i]) seq.resize(i);
1046  }
1047  if (seq.size() == 0) break; // will not get any prefix.
1048  }
1049  common_str = repository_.IdOfSeq(seq);
1050  }
1051 
1052  { // This block computes "tot_weight".
1053  iter = begin;
1054  tot_weight = iter->weight;
1055  for (++iter; iter != end; ++iter)
1056  tot_weight = Plus(tot_weight, iter->weight);
1057  }
1058 
1059  // Now divide out common stuff from elements.
1060  size_t prefix_len = seq.size();
1061  for (iter = begin; iter != end; ++iter) {
1062  iter->weight = Divide(iter->weight, tot_weight);
1063  iter->string = repository_.RemovePrefix(iter->string, prefix_len);
1064  }
1065  }
1066 
1067  // Now add an arc to the state that the subset represents.
1068  // We may create a new state id for this (in SubsetToStateId).
1069  TempArc temp_arc;
1070  temp_arc.ilabel = ilabel;
1071  temp_arc.nextstate = SubsetToStateId(*subset); // may or may not really add the subset.
1072  temp_arc.ostring = common_str;
1073  temp_arc.weight = tot_weight;
1074  output_arcs_[state].push_back(temp_arc); // record the arc.
1075 }
