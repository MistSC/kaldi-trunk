45  void InitData(std::istream &is) {
46  // define options
47  std::vector<std::string> nested_nnet_proto;
48  std::vector<std::string> nested_nnet_filename;
49  // parse config
50  std::string token;
51  while (!is.eof()) {
52  ReadToken(is, false, &token);
53  if (token == "<NestedNnetFilename>") {
54  while(!is.eof()) {
55  std::string file_or_end;
56  ReadToken(is, false, &file_or_end);
57  if (file_or_end == "</NestedNnetFilename>") break;
58  nested_nnet_filename.push_back(file_or_end);
59  }
60  } else if (token == "<NestedNnetProto>") {
61  while(!is.eof()) {
62  std::string file_or_end;
63  ReadToken(is, false, &file_or_end);
64  if (file_or_end == "</NestedNnetProto>") break;
65  nested_nnet_proto.push_back(file_or_end);
66  }
67  } else KALDI_ERR << "Unknown token " << token << ", typo in config?"
68  << " (NestedNnetFilename|NestedNnetProto)";
69  is >> std::ws; // eat-up whitespace
70  }
71  // initialize
72  KALDI_ASSERT((nested_nnet_proto.size() > 0) ^ (nested_nnet_filename.size() > 0)); //xor
73  // read nnets from files
74  if (nested_nnet_filename.size() > 0) {
75  for (int32 i=0; i<nested_nnet_filename.size(); i++) {
76  Nnet nnet;
77  nnet.Read(nested_nnet_filename[i]);
78  nnet_.push_back(nnet);
79  KALDI_LOG << "Loaded nested <Nnet> from file : " << nested_nnet_filename[i];
80  }
81  }
82  // initialize nnets from prototypes
83  if (nested_nnet_proto.size() > 0) {
84  for (int32 i=0; i<nested_nnet_proto.size(); i++) {
85  Nnet nnet;
86  nnet.Init(nested_nnet_proto[i]);
87  nnet_.push_back(nnet);
88  KALDI_LOG << "Initialized nested <Nnet> from prototype : " << nested_nnet_proto[i];
89  }
90  }
91  // check dim-sum of nested nnets
92  int32 nnet_input_sum = 0, nnet_output_sum = 0;
93  for (int32 i=0; i<nnet_.size(); i++) {
94  nnet_input_sum += nnet_[i].InputDim();
95  nnet_output_sum += nnet_[i].OutputDim();
96  }
97  KALDI_ASSERT(InputDim() == nnet_input_sum);
98  KALDI_ASSERT(OutputDim() == nnet_output_sum);
99  }

101  void ReadData(std::istream &is, bool binary) {
102  // read
103  ExpectToken(is, binary, "<NestedNnetCount>");
104  int32 nnet_count;
105  ReadBasicType(is, binary, &nnet_count);
106  for (int32 i=0; i<nnet_count; i++) {
107  ExpectToken(is, binary, "<NestedNnet>");
108  int32 dummy;
109  ReadBasicType(is, binary, &dummy);
110  Nnet nnet;
111  nnet.Read(is, binary);
112  nnet_.push_back(nnet);
113  }
114  ExpectToken(is, binary, "</ParallelComponent>");
115 
116  // check dim-sum of nested nnets
117  int32 nnet_input_sum = 0, nnet_output_sum = 0;
118  for (int32 i=0; i<nnet_.size(); i++) {
119  nnet_input_sum += nnet_[i].InputDim();
120  nnet_output_sum += nnet_[i].OutputDim();
121  }
122  KALDI_ASSERT(InputDim() == nnet_input_sum);
123  KALDI_ASSERT(OutputDim() == nnet_output_sum);
124  }

147  void GetParams(Vector<BaseFloat>* wei_copy) const {
148  wei_copy->Resize(NumParams());
149  int32 offset = 0;
150  for (int32 i=0; i<nnet_.size(); i++) {
151  Vector<BaseFloat> wei_aux;
152  nnet_[i].GetParams(&wei_aux);
153  wei_copy->Range(offset, wei_aux.Dim()).CopyFromVec(wei_aux);
154  offset += wei_aux.Dim();
155  }
156  KALDI_ASSERT(offset == NumParams());
157  }

195  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
196  int32 input_offset = 0, output_offset = 0;
197  for (int32 i=0; i<nnet_.size(); i++) {
198  CuSubMatrix<BaseFloat> src(in.ColRange(input_offset, nnet_[i].InputDim()));
199  CuSubMatrix<BaseFloat> tgt(out->ColRange(output_offset, nnet_[i].OutputDim()));
200  //
201  CuMatrix<BaseFloat> tgt_aux;
202  nnet_[i].Propagate(src, &tgt_aux);
203  tgt.CopyFromMat(tgt_aux);
204  //
205  input_offset += nnet_[i].InputDim();
206  output_offset += nnet_[i].OutputDim();
207  }
208  }
