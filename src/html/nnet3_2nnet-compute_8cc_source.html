146 void NnetComputer::ExecuteCommand(int32 command) {
147  const NnetComputation::Command &c = computation_.commands[command];
148  try {
149  switch (c.command_type) {
150  case kAllocMatrixZeroed:
151  matrices_[c.arg1].Resize(computation_.matrices[c.arg1].num_rows,
152  computation_.matrices[c.arg1].num_cols,
153  kSetZero);
154  break;
155  case kAllocMatrixUndefined:
156  matrices_[c.arg1].Resize(computation_.matrices[c.arg1].num_rows,
157  computation_.matrices[c.arg1].num_cols,
158  kUndefined);
159  break;
160  case kDeallocMatrix:
161  matrices_[c.arg1].Resize(0, 0);
162  break;
163  case kAllocMatrixFromOther:
164  matrices_[c.arg1].Swap(&(matrices_[c.arg2]));
165  break;
166  case kAllocMatrixFromOtherZeroed:
167  matrices_[c.arg1].Swap(&(matrices_[c.arg2]));
168  matrices_[c.arg1].SetZero();
169  break;
170  case kPropagate: {
171  const Component *component = nnet_.GetComponent(c.arg1);
172  ComponentPrecomputedIndexes *indexes =
173  computation_.component_precomputed_indexes[c.arg2];
174  const CuSubMatrix<BaseFloat> input(GetSubMatrix(c.arg3));
175  CuSubMatrix<BaseFloat> output(GetSubMatrix(c.arg4));
176  component->Propagate(indexes, input, &output);
177  break;
178  }
179  case kStoreStats: {
180  KALDI_ASSERT(nnet_to_update_ != NULL);
181  Component *upd_component = nnet_to_update_->GetComponent(c.arg1);
182  CuSubMatrix<BaseFloat> output(GetSubMatrix(c.arg2));
183  upd_component->StoreStats(output);
184  break;
185  }
186  case kBackprop:
187  case kBackpropNoModelUpdate: {
188  std::ostringstream debug_str;
189  KALDI_ASSERT(nnet_to_update_ != NULL);
190  debug_str << nnet_.GetComponentName(c.arg1);
191  const Component *component = nnet_.GetComponent(c.arg1);
192  KALDI_ASSERT(!(computation_.need_model_derivative && !nnet_to_update_));
193  Component *upd_component = (nnet_to_update_ &&
194  c.command_type == kBackprop &&
195  computation_.need_model_derivative ?
196  nnet_to_update_->GetComponent(c.arg1) :
197  NULL);
198  ComponentPrecomputedIndexes *indexes =
199  computation_.component_precomputed_indexes[c.arg2];
200  const CuSubMatrix<BaseFloat> in_value(GetSubMatrix(c.arg3));
201  const CuSubMatrix<BaseFloat> out_value(GetSubMatrix(c.arg4));
202  const CuSubMatrix<BaseFloat> out_deriv(GetSubMatrix(c.arg5));
203  CuSubMatrix<BaseFloat> in_deriv(GetSubMatrix(c.arg6));
204  component->Backprop(debug_str.str(), indexes,
205  in_value, out_value, out_deriv, upd_component,
206  c.arg6 == 0 ? NULL : &in_deriv);
207  break;
208  }
209  case kMatrixCopy: {
210  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
211  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
212  dest.CopyFromMat(src);
213  break;
214  }
215  case kMatrixAdd: {
216  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
217  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
218  dest.AddMat(1.0, src);
219  break;
220  }
221  case kAddRows: {
222  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
223  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
224  const CuArray<int32> &indexes = computation_.indexes_cuda[c.arg3];
225  dest.AddRows(1.0, src, indexes);
226  break;
227  }
228  case kCopyRows: {
229  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
230  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
231  const CuArray<int32> &indexes = computation_.indexes_cuda[c.arg3];
232  dest.CopyRows(src, indexes);
233  break;
234  }
235  case kCopyRowsMulti: {
236  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
237  CuArray<const BaseFloat*> pointers;
238  GetPointers(c.arg2, dest.NumCols(), &pointers);
239  dest.CopyRows(pointers);
240  break;
241  }
242  case kCopyToRowsMulti: {
243  CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg1));
244  CuArray<BaseFloat*> pointers;
245  GetPointers(c.arg2, src.NumCols(), &pointers);
246  src.CopyToRows(pointers);
247  break;
248  }
249  case kAddRowsMulti: {
250  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
251  CuArray<const BaseFloat*> pointers;
252  GetPointers(c.arg2, dest.NumCols(), &pointers);
253  dest.AddRows(1.0, pointers);
254  break;
255  }
256  case kAddToRowsMulti: {
257  CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg1));
258  CuArray<BaseFloat*> pointers;
259  GetPointers(c.arg2, src.NumCols(), &pointers);
260  src.AddToRows(1.0, pointers);
261  break;
262  }
263  case kAddRowRanges: {
264  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
265  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
266  const CuArray<Int32Pair> &pairs = computation_.indexes_ranges_cuda[c.arg3];
267  dest.AddRowRanges(src, pairs);
268  break;
269  }
270  case kNoOperation: case kNoOperationMarker:
271  break;
272  default:
273  KALDI_ERR << "Invalid command in computation";
274  }
275  } catch (...) {
276  if (!debug_) {
277  std::string preamble;
278  computation_.GetCommandStrings(nnet_, &preamble, &command_strings_);
279  KALDI_WARN << "Printing some background info since error was detected";
280  KALDI_LOG << preamble;
281  for (int32 prev_c = 0; prev_c < command; prev_c++)
282  KALDI_LOG << command_strings_[prev_c];
283  }
284  // the following will re-throw the error, but now we've printed more info
285  // about what went wrong.
286  KALDI_ERR << "Error running command " << command_strings_[command];
287  }
288 }

300 void NnetComputer::GetPointers(int32 indexes_multi_index,
301  int32 num_cols,
302  CuArray<BaseFloat*> *pointers) {
303  KALDI_ASSERT(static_cast<size_t>(indexes_multi_index)
304  < computation_.indexes_multi.size());
305  const std::vector<std::pair<int32,int32> > &pairs =
306  computation_.indexes_multi[indexes_multi_index];
307  int32 size = pairs.size();
308  std::vector<BaseFloat*> vec(size);
309 
310  // the map "lookup" maps from submatrix index to the Data()
311  // pointer of that submatrix, and the corresponding Stride().
312  unordered_map<int32, std::pair<BaseFloat*, int32> > lookup;
313 
314  for (int32 i = 0; i < size; i++) {
315  int32 submatrix_index = pairs[i].first, row = pairs[i].second;
316  if (submatrix_index != -1) {
317  unordered_map<int32, std::pair<BaseFloat*, int32> >::iterator
318  iter = lookup.find(submatrix_index);
319  if (iter == lookup.end()) {
320  CuSubMatrix<BaseFloat> m = GetSubMatrix(submatrix_index);
321  lookup[submatrix_index] = std::pair<BaseFloat*, int32>(m.Data(),
322  m.Stride());
323  iter = lookup.find(submatrix_index);
324  }
325  BaseFloat *data = iter->second.first;
326  int32 stride = iter->second.second;
327  vec[i] = data + (row * stride);
328  } else {
329  // -1 is a marker that will be translated to NULL.
330  vec[i] = NULL;
331  }
332  }
333 #ifdef KALDI_PARANOID
334  for (int32 i = 0; i < size; i += 30 + RandInt(0, 9)) {
335  // Do a pseudo-random spot check that the row-indexes are not out of range.
336  int32 submatrix_index = pairs[i].first, row = pairs[i].second;
337  CuSubMatrix<BaseFloat> m = GetSubMatrix(submatrix_index);
338  KALDI_ASSERT(row >= 0 && row < m.NumRows() && num_cols == m.NumCols());
339  }
340 #endif
341  pointers->CopyFromVec(vec);
342 }

455 int32 NnetComputer::GetMatrixIndex(
456  const std::string &node_name, bool is_output, bool is_deriv) const {
457  int32 node_index = nnet_.GetNodeIndex(node_name);
458  if (node_index == -1)
459  KALDI_ERR << "No node named '" << node_name << "'in network.";
460  if (is_output) {
461  if (!nnet_.IsOutputNode(node_index))
462  KALDI_ERR << "Expecting output node; node named '"
463  << node_name << "' is not output node.";
464  } else {
465  if (nnet_.IsOutputNode(node_index))
466  KALDI_ERR << "Expecting input node or component node; node named '"
467  << node_name << "' is output node.";
468  }
469  unordered_map<int32, std::pair<int32, int32> >::const_iterator
470  iter = computation_.input_output_info.find(node_index),
471  end = computation_.input_output_info.end();
472  if (iter == end)
473  KALDI_ERR << "Not expecting input or output for node named '" << node_name
474  << "' (not in computation request)";
475  std::pair<int32,int32> locations = iter->second;
476  int32 location;
477  if (is_deriv) {
478  location = locations.second;
479  if (locations.second <= 0) // No deriv expected.
480  KALDI_ERR << "Not expecting derivative information for node named '"
481  << node_name << "' (not in computation request)";
482  } else {
483  location = locations.first;
484  }
485  KALDI_ASSERT(static_cast<size_t>(location) < matrices_.size());
486  return location;
487 }

489 void NnetComputer::CheckInputs(bool check_output_deriv) const {
490  unordered_map<int32, std::pair<int32, int32> >::const_iterator
491  iter = computation_.input_output_info.begin(),
492  end = computation_.input_output_info.end();
493  for (; iter != end; ++iter) {
494  int32 node_index = iter->first,
495  value_matrix_index = iter->second.first,
496  deriv_matrix_index = iter->second.second;
497  std::string name = nnet_.GetNodeName(node_index);
498  if (nnet_.IsOutputNode(node_index)) {
499  if (check_output_deriv && deriv_matrix_index > 0) {

502  KALDI_ERR << "Output-derivative required but not provided for node '"

505  } else {

507  if (matrices_[value_matrix_index].NumRows() == 0)
508  KALDI_ERR << "Input required but not provided for node '"
