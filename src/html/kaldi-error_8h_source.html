120 // Note on KALDI_ASSERT and KALDI_PARANOID_ASSERT
121 // The original (simple) version of the code was this
122 //
123 // #define KALDI_ASSERT(cond) if (!(cond))
124 // kaldi::KaldiAssertFailure_(__func__, __FILE__, __LINE__, #cond);
125 //
126 // That worked well, but we were concerned that it
127 // could potentially cause a performance issue due to failed branch
128 // prediction (best practice is to have the if branch be the commonly
129 // taken one).
130 // Therefore, we decided to move the call into the else{} branch.
131 // A single block {} around if /else does not work, because it causes
132 // syntax error (unmatched else block) in the following code:
133 //
134 // if (condition)
135 // KALDI_ASSERT(condition2);
136 // else
137 // SomethingElse();
138 //
139 // do {} while(0) -- note there is no semicolon at the end! --- works nicely
140 // and compilers will be able to optimize the loop away (as the condition
141 // is always false).
142 #ifndef NDEBUG
143 #define KALDI_ASSERT(cond) do { if ((cond)) ; else \
144  kaldi::KaldiAssertFailure_(__func__, __FILE__, __LINE__, #cond);}while(0)
145 #else
146 #define KALDI_ASSERT(cond)
147 #endif
148 // also see KALDI_COMPILE_TIME_ASSERT, defined in base/kaldi-utils.h,
149 // and KALDI_ASSERT_IS_INTEGER_TYPE and KALDI_ASSERT_IS_FLOATING_TYPE,
150 // also defined there.
151 // some more expensive asserts only checked if this defined
152 #ifdef KALDI_PARANOID
153 #define KALDI_PARANOID_ASSERT(cond) do { if ((cond)) ; else \
154  kaldi::KaldiAssertFailure_(__func__, __FILE__, __LINE__, #cond);}while(0)
155 #else
156 #define KALDI_PARANOID_ASSERT(cond)
157 #endif
