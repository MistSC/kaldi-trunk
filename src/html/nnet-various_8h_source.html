52  void InitData(std::istream &is) {
53  // define options
54  std::vector<int32> frame_offsets;
55  std::vector<std::vector<int32> > build_vector;
56  // parse config
57  std::string token;
58  while (!is.eof()) {
59  ReadToken(is, false, &token);
60  if (token == "<ReadVector>") {
61  ReadIntegerVector(is, false, &frame_offsets);
62  } else if (token == "<BuildVector>") {
63  // <BuildVector> 1:1:1000 1:1:1000 1 2 3 1:10 </BuildVector> [matlab indexing]
64  // read the colon-separated-lists:
65  while (!is.eof()) {
66  std::string colon_sep_list_or_end;
67  ReadToken(is, false, &colon_sep_list_or_end);
68  if (colon_sep_list_or_end == "</BuildVector>") break;
69  std::vector<int32> v;
70  SplitStringToIntegers(colon_sep_list_or_end, ":", false, &v);
71  build_vector.push_back(v);
72  }
73  } else {
74  KALDI_ERR << "Unknown token " << token << ", a typo in config?"
75  << " (ReadVector|BuildVector)";
76  }
77  is >> std::ws; // eat-up whitespace
78  }
79 
80  // build the vector, using <BuildVector> ... </BuildVector> inputs
81  if (build_vector.size() > 0) {
82  for (int32 i=0; i<build_vector.size(); i++) {
83  switch (build_vector[i].size()) {
84  case 1:
85  frame_offsets.push_back(build_vector[i][0]);
86  break;
87  case 2: { // assuming step 1
88  int32 min=build_vector[i][0], max=build_vector[i][1];
89  KALDI_ASSERT(min <= max);
90  for (int32 j=min; j<=max; j++) {
91  frame_offsets.push_back(j);
92  }}
93  break;
94  case 3: { // step can be negative -> flipped min/max
95  int32 min=build_vector[i][0], step=build_vector[i][1], max=build_vector[i][2];
96  KALDI_ASSERT((min <= max && step > 0) || (min >= max && step < 0));
97  for (int32 j=min; j<=max; j += step) {
98  frame_offsets.push_back(j);
99  }}
100  break;
101  case 0:
102  default:
103  KALDI_ERR << "Error parsing <BuildVector>";
104  }
105  }
106  }
107 
108  // copy to GPU
109  frame_offsets_ = frame_offsets;
110 
111  // check dim
112  KALDI_ASSERT(frame_offsets_.Dim()*InputDim() == OutputDim());
113  }

307  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
308  // resize vector when needed,
309  if (row_scales_.Dim() != in.NumRows()) {
310  row_scales_.Resize(in.NumRows());
311  }
312  // get the normalization scalars,
313  l2_aux_ = in;
314  l2_aux_.MulElements(l2_aux_); // x^2,
315  row_scales_.AddColSumMat(1.0,l2_aux_,0.0); // sum_of_cols(x^2),
316  row_scales_.ApplyPow(0.5); // L2norm = sqrt(sum_of_cols(x^2)),
317  row_scales_.InvertElements(); // 1/L2norm,
318  // compute the output,
319  out->CopyFromMat(in);
320  out->MulRowsVec(row_scales_); // re-normalize,
321  }
322 
323  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
324  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
325  in_diff->CopyFromMat(out_diff);
326  in_diff->MulRowsVec(row_scales_); // diff_by_x(s * x) = s,
327  }

331  CuVector<BaseFloat> row_scales_;
