165 template <typename Real>
166 void SparseVector<Real>::Read(std::istream &is, bool binary) {
167  if (binary) {
168  ExpectToken(is, binary, "SV");
169  ReadBasicType(is, binary, &dim_);
170  KALDI_ASSERT(dim_ >= 0);
171  int32 num_elems;
172  ReadBasicType(is, binary, &num_elems);
173  KALDI_ASSERT(num_elems >= 0 && num_elems <= dim_);
174  pairs_.resize(num_elems);
175  typename std::vector<std::pair<MatrixIndexT, Real> >::iterator
176  iter = pairs_.begin(), end = pairs_.end();
177  for (; iter != end; ++iter) {
178  ReadBasicType(is, binary, &(iter->first));
179  ReadBasicType(is, binary, &(iter->second));
180  }
181  } else {
182  // In text-mode, format is "dim=5 [ 0 0.2 3 0.9 ]
183  std::string str;
184  is >> str;
185  if (str.substr(0, 4) != "dim=")
186  KALDI_ERR << "Reading sparse vector, expected 'dim=xxx', got " << str;
187  std::string dim_str = str.substr(4, std::string::npos);
188  std::istringstream dim_istr(dim_str);
189  int32 dim = -1;
190  dim_istr >> dim;
191  if (dim < 0 || dim_istr.fail()) {
192  KALDI_ERR << "Reading sparse vector, expected 'dim=[int]', got " << str;
193  }
194  dim_ = dim;
195  is >> std::ws;
196  is >> str;
197  if (str != "[")
198  KALDI_ERR << "Reading sparse vector, expected '[', got " << str;
199  pairs_.clear();
200  while (1) {
201  is >> std::ws;
202  if (is.peek() == ']') {
203  is.get();
204  break;
205  }
206  MatrixIndexT i;
207  BaseFloat p;
208  is >> i >> p;
209  if (is.fail())
210  KALDI_ERR << "Error reading sparse vector, expecting numbers.";
211  KALDI_ASSERT(i >= 0 && i < dim
212  && (pairs_.empty() || i > pairs_.back().first));
213  pairs_.push_back(std::pair<MatrixIndexT, BaseFloat>(i, p));
214  }
215  }

229 template <typename Real>
230 SparseVector<Real>::SparseVector(
231  MatrixIndexT dim, const std::vector<std::pair<MatrixIndexT, Real> > &pairs):
232  dim_(dim),
233  pairs_(pairs) {
234  std::sort(pairs_.begin(), pairs_.end(),
235  sparse_vector_utils::CompareFirst<Real>());
236  typename std::vector<std::pair<MatrixIndexT, Real> >::iterator
237  out = pairs_.begin(), in = out, end = pairs_.end();
238  // special case: while there is nothing to be changed, skip over
239  // initial input (avoids unnecessary copying).
240  while (in + 1 < end && in[0].first != in[1].first && in[0].second != 0.0) {
241  in++;
242  out++;
243  }
244  while (in < end) {
245  // We reach this point only at the first element of
246  // each stretch of identical .first elements.
247  *out = *in;
248  ++in;
249  while (in < end && in->first == out->first) {
250  out->second += in->second; // this is the merge operation.
251  ++in;
252  }
253  if (out->second != Real(0.0)) // Don't keep zero elements.
254  out++;
255  }
256  pairs_.erase(out, end);
257  if (!pairs_.empty()) {
258  // range check.
259  KALDI_ASSERT(pairs_.front().first >= 0 && pairs_.back().first < dim_);
260  }
261 }

346  MatrixTransposeType trans) const {
347  if (trans == kNoTrans) {

359  other->SetZero();

361  const SparseVector<Real> &svec = rows_[row];
362  MatrixIndexT num_elems = svec.NumElements();
363  const std::pair<MatrixIndexT, Real> *sdata = svec.Data();
364  for (MatrixIndexT e = 0; e < num_elems; e++)
365  other_col_data[sdata[e].first * other_stride] = sdata[e].second;
366  }

618  sum += VecSvec(A.Row(r), B.Row(r));
619  } else {

621  MatrixIndexT Astride = A.Stride(), Acols = A.NumCols(), Arows = A.NumRows();
622  KALDI_ASSERT(Arows == B.NumCols() && Acols == B.NumRows());
623  sum = 0.0;
624  for (MatrixIndexT i = 0; i < Acols; i++, A_col_data++) {
625  Real col_sum = 0.0;
626  const SparseVector<Real> &svec = B.Row(i);
627  MatrixIndexT num_elems = svec.NumElements();
628  const std::pair<MatrixIndexT, Real> *sdata = svec.Data();
629  for (MatrixIndexT e = 0; e < num_elems; e++)
630  col_sum += A_col_data[Astride * sdata[e].first] * sdata[e].second;
631  sum += col_sum;
632  }
633  }
634  return sum;

818 void AppendGeneralMatrixRows(const std::vector<const GeneralMatrix *> &src,
819  GeneralMatrix *mat) {
820  mat->Clear();
821  int32 size = src.size();
822  if (size == 0)
823  return;
824  bool all_sparse = true;
825  for (int32 i = 0; i < size; i++) {
826  if (src[i]->Type() != kSparseMatrix && src[i]->NumRows() != 0) {
827  all_sparse = false;
828  break;
829  }
830  }
831  if (all_sparse) {
832  std::vector<SparseMatrix<BaseFloat> > sparse_mats(size);
833  for (int32 i = 0; i < size; i++)
834  sparse_mats[i] = src[i]->GetSparseMatrix();
835  SparseMatrix<BaseFloat> appended_mat;
836  appended_mat.AppendSparseMatrixRows(&sparse_mats);
837  mat->SwapSparseMatrix(&appended_mat);
838  } else {
839  int32 tot_rows = 0, num_cols = -1;
840  for (int32 i = 0; i < size; i++) {
841  const GeneralMatrix &src_mat = *(src[i]);
842  int32 src_rows = src_mat.NumRows(), src_cols = src_mat.NumCols();
843  if (src_rows != 0) {
844  tot_rows += src_rows;
845  if (num_cols == -1) num_cols = src_cols;
846  else if (num_cols != src_cols)
847  KALDI_ERR << "Appending rows of matrices with inconsistent num-cols: "
848  << num_cols << " vs. " << src_cols;
849  }
850  }
851  Matrix<BaseFloat> appended_mat(tot_rows, num_cols, kUndefined);
852  int32 row_offset = 0;
853  for (int32 i = 0; i < size; i++) {
854  const GeneralMatrix &src_mat = *(src[i]);
855  int32 src_rows = src_mat.NumRows();
856  if (src_rows != 0) {
857  SubMatrix<BaseFloat> dest_submat(appended_mat, row_offset, src_rows,
858  0, num_cols);
859  src_mat.CopyToMat(&dest_submat);
860  row_offset += src_rows;
861  }
862  }
863  KALDI_ASSERT(row_offset == tot_rows);
864  mat->SwapFullMatrix(&appended_mat);
865  }
866 }

868 void FilterCompressedMatrixRows(const CompressedMatrix &in,
869  const std::vector<bool> &keep_rows,
870  Matrix<BaseFloat> *out) {
871  KALDI_ASSERT(keep_rows.size() == static_cast<size_t>(in.NumRows()));
872  int32 num_kept_rows = 0;
873  std::vector<bool>::const_iterator iter = keep_rows.begin(),
874  end = keep_rows.end();
875  for (; iter != end; ++iter)
876  if (*iter)
877  num_kept_rows++;
878  if (num_kept_rows == 0)
879  KALDI_ERR << "No kept rows";
880  if (num_kept_rows == static_cast<int32>(keep_rows.size())) {
881  out->Resize(in.NumRows(), in.NumCols(), kUndefined);
882  in.CopyToMat(out);
883  return;
884  }
885  const BaseFloat heuristic = 0.33;
886  // should be > 0 and < 1.0. represents the performance hit we get from
887  // iterating row-wise versus column-wise in compressed-matrix uncompression.
888 
889  if (num_kept_rows > heuristic * in.NumRows()) {
890  // if quite a few of the the rows are kept, it may be more efficient
891  // to uncompress the entire compressed matrix, since per-column operation
892  // is more efficient.
893  Matrix<BaseFloat> full_mat(in);
894  FilterMatrixRows(full_mat, keep_rows, out);
895  } else {
896  out->Resize(num_kept_rows, in.NumCols(), kUndefined);
897 
898  iter = keep_rows.begin();
899  int32 out_row = 0;
900  for (int32 in_row = 0; iter != end; ++iter, ++in_row) {
901  if (*iter) {
902  SubVector<BaseFloat> dest(*out, out_row);
903  in.CopyRowToVec(in_row, &dest);
904  out_row++;
905  }
906  }
907  KALDI_ASSERT(out_row == num_kept_rows);
908  }
909 }

1055 template <class Real>
1056 Real SparseVector<Real>::Max(int32 *index_out) const {
1057  KALDI_ASSERT(dim_ > 0 && pairs_.size() <= static_cast<size_t>(dim_));
1058  Real ans = -std::numeric_limits<Real>::infinity();
1059  int32 index = 0;
1060  typename std::vector<std::pair<MatrixIndexT, Real> >::const_iterator
1061  iter = pairs_.begin(), end = pairs_.end();
1062  for (; iter != end; ++iter) {
1063  if (iter->second > ans) {
1064  ans = iter->second;
1065  index = iter->first;
1066  }
1067  }
1068  if (ans >= 0 || pairs_.size() == dim_) {
1069  // ans >= 0 will be the normal case.
1070  // if pairs_.size() == dim_ then we need to return
1071  // even a negative answer as there are no spaces (hence no unlisted zeros).
1072  *index_out = index;
1073  return ans;
1074  }
1075  // all the stored elements are < 0, but there are unlisted
1076  // elements -> pick the first unlisted element.
1077  // Note that this class requires that the indexes are sorted
1078  // and unique.
1079  index = 0; // "index" will always be the next index, that
1080  // we haven't seen listed yet.
1081  iter = pairs_.begin();
1082  for (; iter != end; ++iter) {
1083  if (iter->first > index) { // index "index" is not listed.
1084  *index_out = index;
1085  return 0.0;
1086  } else {
1087  // index is the next potential gap in the indexes.
1088  index = iter->first + 1;
1089  }
1090  }
1091  // we can reach here if either pairs_.empty(), or
1092  // pairs_ is nonempty but contains a sequence (0, 1, 2,...).
1093  if (!pairs_.empty())
1094  index = pairs_.back().first + 1;
1095  // else leave index at zero
1096  KALDI_ASSERT(index < dim_);
1097  *index_out = index;
1098  return 0.0;
1099 }
