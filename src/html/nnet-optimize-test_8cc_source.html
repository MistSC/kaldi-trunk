30 // Run the test wothout optimizations and with optimizations specified by the
31 // parameter. Only print warnings; we'll fail the whole test later.
32 static bool UnitTestNnetOptimizeWithOptions(NnetOptimizeOptions opt_config) {
33  //opt_config.convert_addition = false;
34  //opt_config.remove_assignments = false;
35  //opt_config.move_sizing_commands = false;
36  //opt_config.allocate_from_other = false;
37 
38  srand(0); // Every run must be deterministic.
39  for (int32 n = 0; n < 40; n++) {
40  struct NnetGenerationOptions gen_config;
41 
42  std::vector<std::string> configs;
43  GenerateConfigSequence(gen_config, &configs);
44  Nnet nnet;
45  for (size_t j = 0; j < configs.size(); j++) {
46  KALDI_LOG << "Input config[" << j << "] is: " << configs[j];
47  std::istringstream is(configs[j]);
48  nnet.ReadConfig(is);
49  }
50 
51  ComputationRequest request;
52  std::vector<Matrix<BaseFloat> > inputs;
53  ComputeExampleComputationRequestSimple(nnet, &request, &inputs);
54 
55  NnetComputation computation;
56  Compiler compiler(request, nnet);
57 
58  CompilerOptions opts;
59  compiler.CreateComputation(opts, &computation);
60  {
61  std::ostringstream os;
62  computation.Print(os, nnet);
63  KALDI_LOG << "Generated computation is: " << os.str();
64  }
65  CheckComputationOptions check_config;
66  // we can do the rewrite check since it's before optimization.
67  check_config.check_rewrite = true;
68  ComputationChecker checker(check_config, nnet, computation);
69  checker.Check();
70 
71  NnetComputation computation_opt(computation);
72 
73  {
74  Optimize(opt_config, nnet, request, &computation_opt);
75  std::ostringstream os;
76  computation_opt.Print(os, nnet);
77  KALDI_LOG << "Optimized computation is: " << os.str();
78  }
79 
80  NnetComputeOptions compute_opts;
81  if (RandInt(0, 1) == 0)
82  compute_opts.debug = true;
83 
84  computation.ComputeCudaIndexes();
85  computation_opt.ComputeCudaIndexes();
86  Nnet nnet_to_update(nnet); // copy of the nnet that we update... needed to
87  // test the consolidation of backprop commands,
88  // otherwise the optimized and non-optimized
89  // comptuations differ.
90  bool is_gradient = true; // with natural gradient, the consolidation would
91  // affect the final model params -> test just the
92  // gradient.
93  SetZero(is_gradient, &nnet_to_update);
94 
95  NnetComputer computer(compute_opts,
96  computation,
97  nnet,
98  &nnet_to_update);
99 
100  Nnet nnet_opt(nnet); // copy of the nnet for the optimized computation.
101  // necessary in case backprop changes parameters.
102  Nnet nnet_opt_to_update(nnet_opt);
103  SetZero(is_gradient, &nnet_opt_to_update);
104 
105  // NnetComputer for the optimized version of the computation.
106  NnetComputer computer_opt(compute_opts,
107  computation_opt,
108  nnet_opt,
109  &nnet_opt_to_update);
110 
111  // provide the input to the computations.
112  for (size_t i = 0; i < request.inputs.size(); i++) {
113  CuMatrix<BaseFloat> temp(inputs[i]);
114  KALDI_LOG << "Input sum is " << temp.Sum();
115  computer.AcceptInput(request.inputs[i].name, &temp);
116  CuMatrix<BaseFloat> temp2(inputs[i]);
117  computer_opt.AcceptInput(request.inputs[i].name, &temp2);
118  }
119  KALDI_LOG << "Running non-optimized forward computation";
120  computer.Forward();
121  KALDI_LOG << "Running optimized forward computation";
122  computer_opt.Forward();
123 
124  const CuMatrixBase<BaseFloat> &output(computer.GetOutput("output"));
125  KALDI_LOG << "Output sum (not optimized) is " << output.Sum();
126  const CuMatrixBase<BaseFloat> &output_opt(computer_opt.GetOutput("output"));
127  KALDI_LOG << "Output sum (optimized) is " << output_opt.Sum();
128  if (!ApproxEqual(output, output_opt)) {
129  KALDI_WARN << "Non-optimized and optimized versions of the computation give "
130  << "different outputs.";
131  return false;
132  }
133 
134  CuMatrix<BaseFloat> output_deriv(output.NumRows(), output.NumCols());
135  output_deriv.SetRandn();
136  CuMatrix<BaseFloat> output_deriv_opt(output_deriv);
137 
138  if (request.outputs[0].has_deriv) {
139  computer.AcceptOutputDeriv("output", &output_deriv);
140  computer_opt.AcceptOutputDeriv("output", &output_deriv_opt);
141  }
142 
143  KALDI_LOG << "Running non-optimized backward computation";
144  computer.Backward();
145  KALDI_LOG << "Running optimized backward computation";
146  computer_opt.Backward();
147  for (size_t i = 0; i < request.inputs.size(); i++) {
148  if (request.inputs[i].has_deriv) {
149  const CuMatrixBase<BaseFloat> &in_deriv =
150  computer.GetInputDeriv(request.inputs[i].name);
151  const CuMatrixBase<BaseFloat> &in_deriv_opt =
152  computer_opt.GetInputDeriv(request.inputs[i].name);
153  KALDI_LOG << "Input-deriv sum for input '" << request.inputs[i].name
154  << "' (non-optimized) is " << in_deriv.Sum();
155  KALDI_LOG << "Input-deriv sum for input '" << request.inputs[i].name
156  << "' (optimized) is " << in_deriv_opt.Sum();
157  if (!ApproxEqual(in_deriv, in_deriv_opt)) {
158  KALDI_WARN << "Non-optimized and optimized versions of the "
159  << "computation give different input-derivs.";
160  return false;
161  }
162  }
163  }
164 
165  if (!NnetParametersAreIdentical(nnet_to_update,
166  nnet_opt_to_update, 1.0e-05)) {
167  KALDI_WARN << "Neural networks differ after training, between "
168  << "optimized and non-optimized computation.";
169  return false;
170  }
171  }
172  return true;
173 }

176 // This test runs the computation with and without optimization, and checks that
177 // the outputs are the same.
178 static void UnitTestNnetOptimize() {
179  NnetOptimizeOptions optimize_all;
180  // randomly sometimes set min_deriv and max_deriv to small/large values,
181  // which will cause some of the LimitDerivativeTimes() code to be called
182  // (without really changing anything).
183  if (RandInt(0, 3) == 0) optimize_all.min_deriv_time = -200;
184  if (RandInt(0, 3) == 0) optimize_all.max_deriv_time = 1000;
185 
186  // this is useful for debugging as it removes nans:
187  // optimize_all.initialize_undefined = false;
188  bool success = UnitTestNnetOptimizeWithOptions(optimize_all);
189  if (success)
190  return;
191 
192  // Test failed with full optimization. Slowly retry with various
193  // optimizations switched off.
194  NnetOptimizeOptions optimize = optimize_all;
195  optimize.propagate_in_place = false;
196  bool succ_no_propagate_in_place = UnitTestNnetOptimizeWithOptions(optimize);
197 
198  optimize = optimize_all;
199  optimize.backprop_in_place = false;
200  bool succ_no_backprop_in_place = UnitTestNnetOptimizeWithOptions(optimize);
201 
202  optimize = optimize_all;
203  optimize.remove_assignments = false;
204  bool succ_no_remove_assignments = UnitTestNnetOptimizeWithOptions(optimize);
205 
206  optimize = optimize_all;
207  optimize.initialize_undefined = false;
208  bool succ_no_initialize_undefined = UnitTestNnetOptimizeWithOptions(optimize);
209 
210  optimize = optimize_all;
211  optimize.move_sizing_commands = false;
212  bool succ_no_move_sizing_commands = UnitTestNnetOptimizeWithOptions(optimize);
213 
214 #define KALDI_SUCCFAIL(b) ((b) ? "SUCCESS" : "FAILURE")
215  KALDI_ERR
216  << "Test failed with all optimizations enabled. Retried test with the "
217  << "following optimizations turned off:"
218  << "\n propagate_in_place ... " << KALDI_SUCCFAIL(succ_no_propagate_in_place)
219  << "\n backprop_in_place ... " << KALDI_SUCCFAIL(succ_no_backprop_in_place)
220  << "\n remove_assignments ... " << KALDI_SUCCFAIL(succ_no_remove_assignments)
221  << "\n initialize_undefined ... " << KALDI_SUCCFAIL(succ_no_initialize_undefined)
222  << "\n move_sizing_commands ... " << KALDI_SUCCFAIL(succ_no_move_sizing_commands);
223 #undef KALDI_SUCCFAIL
224 }
