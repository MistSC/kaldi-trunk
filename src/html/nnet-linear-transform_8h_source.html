44  void InitData(std::istream &is) {
45  // define options
46  float param_stddev = 0.1;
47  float learn_rate_coef = 1.0;
48  std::string read_matrix_file;
49  // parse config
50  std::string token;
51  while (!is.eof()) {
52  ReadToken(is, false, &token);
53  if (token == "<ParamStddev>") ReadBasicType(is, false, &param_stddev);
54  else if (token == "<LearnRateCoef>") ReadBasicType(is, false, &learn_rate_coef);
55  else if (token == "<ReadMatrix>") ReadToken(is, false, &read_matrix_file);
56  else KALDI_ERR << "Unknown token " << token << ", a typo in config?"
57  << " (ParamStddev|ReadMatrix|LearnRateCoef)";
58  is >> std::ws; // eat-up whitespace
59  }
60 
61  //
62  // initialize
63  //
64  if (read_matrix_file != "") { // load from file,
65  bool binary;
66  Input in(read_matrix_file, &binary);
67  linearity_.Read(in.Stream(), binary);
68  in.Close();
69  KALDI_LOG << "Loaded <LinearTransform> matrix from file : " << read_matrix_file;
70  } else { // random initialization,
71  linearity_.Resize(output_dim_, input_dim_);
72  for (int32 r=0; r<output_dim_; r++) {
73  for (int32 c=0; c<input_dim_; c++) {
74  linearity_(r,c) = param_stddev * RandGauss(); // 0-mean Gauss with given std_dev
75  }
76  }
77  }
78  //
79  learn_rate_coef_ = learn_rate_coef;
80  //
81 
82  // check dims,
83  KALDI_ASSERT(linearity_.NumRows() == output_dim_);
84  KALDI_ASSERT(linearity_.NumCols() == input_dim_);
85  }

132  void Update(const CuMatrixBase<BaseFloat> &input, const CuMatrixBase<BaseFloat> &diff) {
133  // we use following hyperparameters from the option class
134  const BaseFloat lr = opts_.learn_rate;
135  const BaseFloat mmt = opts_.momentum;
136  const BaseFloat l2 = opts_.l2_penalty;
137  const BaseFloat l1 = opts_.l1_penalty;
138  // we will also need the number of frames in the mini-batch
139  const int32 num_frames = input.NumRows();
140  // compute gradient (incl. momentum)
141  linearity_corr_.AddMatMat(1.0, diff, kTrans, input, kNoTrans, mmt);
142  // l2 regularization
143  if (l2 != 0.0) {
144  linearity_.AddMat(-lr*l2*num_frames, linearity_);
145  }
146  // l1 regularization
147  if (l1 != 0.0) {
148  cu::RegularizeL1(&linearity_, &linearity_corr_, lr*l1*num_frames, lr);
149  }
150  // update
151  linearity_.AddMat(-lr*learn_rate_coef_, linearity_corr_);
152  }
