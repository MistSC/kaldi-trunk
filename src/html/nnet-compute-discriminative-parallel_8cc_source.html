64 void DiscriminativeExamplesRepository::AcceptExample(
65  const DiscriminativeNnetExample &example) {
66  empty_semaphore_.Wait();
67  examples_mutex_.Lock();
68  examples_.push_back(new DiscriminativeNnetExample(example));
69  examples_mutex_.Unlock();

73 void DiscriminativeExamplesRepository::ExamplesDone() {
74  for (int32 i = 0; i < buffer_size_; i++)
75  empty_semaphore_.Wait();
76  examples_mutex_.Lock();
77  KALDI_ASSERT(examples_.empty());
78  examples_mutex_.Unlock();
79  done_ = true;
80  full_semaphore_.Signal();
81 }

83 DiscriminativeNnetExample*
84 DiscriminativeExamplesRepository::ProvideExample() {
85  full_semaphore_.Wait();
86  if (done_) {
87  KALDI_ASSERT(examples_.empty());
88  full_semaphore_.Signal(); // Increment the semaphore so
89  // the call by the next thread will not block.
90  return NULL; // no examples to return-- all finished.
91  } else {
92  examples_mutex_.Lock();
93  KALDI_ASSERT(!examples_.empty());
94  DiscriminativeNnetExample *ans = examples_.front();
95  examples_.pop_front();
96  examples_mutex_.Unlock();
97  empty_semaphore_.Signal();
98  return ans;
99  }
100 }

103 class DiscTrainParallelClass: public MultiThreadable {
104  public:
105  // This constructor is only called for a temporary object
106  // that we pass to the RunMultiThreaded function.
107  DiscTrainParallelClass(const AmNnet &am_nnet,
108  const TransitionModel &tmodel,
109  const NnetDiscriminativeUpdateOptions &opts,
110  bool store_separate_gradients,
111  DiscriminativeExamplesRepository *repository,
112  Nnet *nnet_to_update,
113  NnetDiscriminativeStats *stats):
114  am_nnet_(am_nnet), tmodel_(tmodel), opts_(opts),
115  store_separate_gradients_(store_separate_gradients),
116  repository_(repository),
117  nnet_to_update_(nnet_to_update),
118  nnet_to_update_orig_(nnet_to_update),
119  stats_ptr_(stats) { }
120 
121  // The following constructor is called multiple times within
122  // the RunMultiThreaded template function.
123  DiscTrainParallelClass(const DiscTrainParallelClass &other):
124  am_nnet_(other.am_nnet_), tmodel_(other.tmodel_), opts_(other.opts_),
125  store_separate_gradients_(other.store_separate_gradients_),
126  repository_(other.repository_), nnet_to_update_(other.nnet_to_update_),
127  nnet_to_update_orig_(other.nnet_to_update_orig_),
128  stats_ptr_(other.stats_ptr_) {
129  if (store_separate_gradients_) {
130  // To ensure correctness, we work on separate copies of the gradient
131  // object, which we'll sum at the end. This is used for exact gradient
132  // computation.
133  if (other.nnet_to_update_ != NULL) {
134  nnet_to_update_ = new Nnet(*(other.nnet_to_update_));
135  // our "nnet_to_update_" variable is a copy of the neural network
136  // we are to update (presumably a gradient). If we don't set these
137  // to zero we would end up adding multiple copies of the any initial
138  // gradient that "nnet_to_update_" contained when we initialize
139  // the first instance of the class.
140  nnet_to_update_->SetZero(true);
141  } else { // support case where we don't really need a gradient.
142  nnet_to_update_ = NULL;
143  }
144  }
145  }
146  // This does the main function of the class.
147  void operator () () {
148  DiscriminativeNnetExample *example;
149  while ((example = repository_->ProvideExample()) != NULL) {
150  // This is a function call to a function defined in
151  // nnet-compute-discriminative.h
152  NnetDiscriminativeUpdate(am_nnet_, tmodel_, opts_,
153  *example, nnet_to_update_, &stats_);
154  delete example;
155 
156  if (GetVerboseLevel() > 3) {
157  KALDI_VLOG(3) << "Printing local stats for thread " << thread_id_;
158  stats_.Print(opts_.criterion);
159  }
160  }
161  }
162 
163  ~DiscTrainParallelClass() {
164  if (nnet_to_update_orig_ != nnet_to_update_) {
165  // This branch is only taken if this instance of the class is
166  // one of the multiple instances allocated inside the RunMultiThreaded
167  // template function, *and* store_separate_gradients_ has been set to true.
168  // In the typical hogwild case, we don't do this.
169  nnet_to_update_orig_->AddNnet(1.0, *nnet_to_update_);
170  delete nnet_to_update_;
171  }
172  stats_ptr_->Add(stats_);
173  }
174  private:
175  const AmNnet &am_nnet_;
176  const TransitionModel &tmodel_;
177  const NnetDiscriminativeUpdateOptions &opts_;
178  bool store_separate_gradients_;
179  DiscriminativeExamplesRepository *repository_;
180  Nnet *nnet_to_update_;
181  Nnet *nnet_to_update_orig_;
182  NnetDiscriminativeStats *stats_ptr_;
183  NnetDiscriminativeStats stats_;
184 };

188 void NnetDiscriminativeUpdateParallel(
189  const AmNnet &am_nnet,
190  const TransitionModel &tmodel,
191  const NnetDiscriminativeUpdateOptions &opts,
192  int32 num_threads,
193  SequentialDiscriminativeNnetExampleReader *example_reader,
194  Nnet *nnet_to_update,
195  NnetDiscriminativeStats *stats) {
196 
197  DiscriminativeExamplesRepository repository;
198 
199  const bool store_separate_gradients = (nnet_to_update != &(am_nnet.GetNnet()));
200 
201  DiscTrainParallelClass c(am_nnet, tmodel, opts,
202  store_separate_gradients,
203  &repository, nnet_to_update, stats);
204 
205  {
206  // The initialization of the following class spawns the threads that
207  // process the examples. They get re-joined in its destructor.
208  MultiThreader<DiscTrainParallelClass> m(num_threads, c);
209 
210  for (; !example_reader->Done(); example_reader->Next()) {
211  repository.AcceptExample(example_reader->Value());
212  }
213  repository.ExamplesDone();
214  }
215  stats->Print(opts.criterion);
216 }
