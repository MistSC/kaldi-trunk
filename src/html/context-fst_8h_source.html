196 /*
197  Actual FST for ContextFst. Most of the work gets done in ContextFstImpl.
198 
199  A ContextFst is a transducer from symbols representing phones-in-context,
200  to phones. It is an on-demand FST. However, it does not create itself in the usual
201  way by expanding states by enumerating all their arcs. This is possible to enable
202  iterating over arcs, but it is not recommended. Instead, we define a special
203  Matcher class that knows how to request the specific arc corresponding to a particular
204  output label.
205 
206  This class requires a list of all the phones and disambiguation
207  symbols, plus the subsequential symbol. This is required to be able to
208  enumerate all output symbols (if we want to access it in an inefficient way), and
209  also to distinguish between phones and disambiguation symbols.
210 */
211

378 // ContextMatcher is a matcher type that is specialized to compose a ContextFst
379 // on the left, with an arbitrary FST on the right. It does so by, rather than
380 // using arc iterators (which would force a call to Expand in ContextFstImpl, which
381 // would expand all the states), uses the CreateArc function of ContextFst. This
382 // function is specific to the ContextFst type. ContextMatcher queries the
383 // type of the FST using FstType(), and verifies that the left hand FST is a context
384 // FST, and then uses a static cast to ContextFst. [We can't make it a template
385 // argument, as the template for ComposeFstOptions is only templated on a single FST
386 // class, and uses the same Matcher type for both the left and right].
387 
388 template <class Arc, class LabelT>
389 class ContextMatcher : public MatcherBase<Arc> { // CAREFUL: templated on arc, not on FST like normal Matcher.
390  public:
391  typedef Fst<Arc> FST; // basic FST type that we get passed
392  // because this is used in composition, typically one side will be a ContextFst,
393  // and one will be some other type which we just treat as Fst<Arc>
394  typedef ContextFst<Arc, LabelT> ContextF;
395  typedef typename Arc::StateId StateId;
396  typedef typename Arc::Label Label;
397  typedef typename Arc::Weight Weight;
398 
399  // If input is a context FST, and match_type == MATCH_OUTPUT,
400  // then we do matching by creating the arcs on-demand.
401  // Otherwise we do not match.
402 
403  ContextMatcher(const FST &fst, MatchType match_type)
404  : fst_(fst.Copy()),
405  match_label_(kNoLabel),
406  s_ (kNoStateId),
407  ready_(false) {
408  if (match_type == MATCH_OUTPUT && fst.Type() == (string)"context") {
409  match_type_ = MATCH_OUTPUT;
410  } else {
411  match_type_ = MATCH_NONE;

413  }

482 /* This is a specialization of Compose, where the left argument is of
483  type ContextFst.
484  For clarity we distinguish it with a different name.
485  It uses the special matcher which should be more efficient than
486  a normal matcher.
487  The fst ifst2 must have the subsequential loop (if not a left-context-only
488  system)
489 */
490 template<class Arc, class LabelT>
491 void ComposeContextFst(const ContextFst<Arc, LabelT> &ifst1, const Fst<Arc> &ifst2,
492  MutableFst<Arc> *ofst,
493  const ComposeOptions &opts = ComposeOptions()) {
494  ComposeFstOptions<Arc, ContextMatcher<Arc, LabelT> > nopts;
495  nopts.gc_limit = 0; // Cache only the most recent state for fastest copy.
496  *ofst = ComposeFst<Arc>(ifst1, ifst2, nopts);
497  if (opts.connect)
498  Connect(ofst);
499 }
