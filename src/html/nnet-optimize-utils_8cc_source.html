205 void ComputationRenumberer::CreateRenumbering(
206  int32 old_num_elements,
207  const std::vector<int32> &to_remove,
208  std::vector<int32> *renumbering) {
209  KALDI_ASSERT(IsSortedAndUniq(to_remove) && old_num_elements > 0);
210  renumbering->clear();
211  renumbering->resize(old_num_elements, 0);
212  int32 num_remove = to_remove.size();
213  for (int32 r = 0; r < num_remove; r++) {
214  int32 this_remove = to_remove[r];
215  // the "> 0" would be ">= 0" in a more generic context, but zero is
216  // not valid in this particular application.
217  KALDI_ASSERT(this_remove > 0 && this_remove < old_num_elements);
218  (*renumbering)[this_remove] = -1;
219  }
220  int32 cur_number = 0;
221  for (int32 i = 0; i < old_num_elements; i++) {
222  if ((*renumbering)[i] != -1)
223  (*renumbering)[i] = cur_number++;
224  }
225  KALDI_ASSERT(cur_number == old_num_elements -
226  static_cast<int32>(to_remove.size()));
227 }

304  num_matrices_new_ = CreateRenumbering(matrix_is_used_, &old_to_new_matrix_);
305 
306  unordered_map<NnetComputation::SubMatrixInfo, int32,

311  // -1's will appear wherever a particular submatrix was never used.

333  std::vector<int32*>::iterator iter = submatrix_args.begin(),

408 void ComputationRenumberer::RemoveUnusedIndexesMulti() {
409  int32 num_indexes_multi = computation_->indexes_multi.size();
410  if (num_indexes_multi == 0)
411  return; // Nothing to do. An optimization.
412  std::vector<bool> indexes_multi_used(num_indexes_multi, false);
413  std::vector<int32*> indexes_multi_args;
414  IdentifyIndexesMultiArgs(&(computation_->commands), &indexes_multi_args);
415  std::vector<int32*>::iterator iter = indexes_multi_args.begin(),
416  end = indexes_multi_args.end();
417  for (; iter != end; ++iter) {
418  int32 indexes_multi_index = **iter;
419  KALDI_ASSERT(indexes_multi_index >= 0 &&
420  indexes_multi_index < num_indexes_multi);
421  indexes_multi_used[indexes_multi_index] = 1;
422  }
423  // old->new mapping for the indexes_multi arrays. will remain -1 for
424  // ones that are unused.
425  std::vector<int32> old_to_new(num_indexes_multi, -1);
426  int32 new_num_indexes_multi = CreateRenumbering(indexes_multi_used,
427  &old_to_new);
428  if (new_num_indexes_multi == num_indexes_multi)
429  return; // Nothing to do. An optimization.
430  std::vector<std::vector<std::pair<int32, int32> > >
431  new_indexes_multi(new_num_indexes_multi);
432  for (int32 i = 0; i < num_indexes_multi; i++) {
433  if (old_to_new[i] != -1)
434  new_indexes_multi[old_to_new[i]].swap(computation_->indexes_multi[i]);

436  computation_->indexes_multi.swap(new_indexes_multi);
437  // renumber within the commands.
438  for (iter = indexes_multi_args.begin(); iter != end; ++iter)
439  **iter = old_to_new[**iter];
440 }

443 // removes duplicates within the indexes_multi_ array itself.
444 void ComputationRenumberer::RemoveIndexesMultiDuplicates() {
445  int32 cur_index = 0,
446  old_indexes_multi_size = computation_->indexes_multi.size();
447  if (old_indexes_multi_size == 0)
448  return;
449  // create index mapping from old to new. the use of map is generally not that
450  // efficient, but the idea here is that we can do most of the comparisons just
451  // based on the size of the vectors, and avoid even visiting most of their
452  // contents.
453  std::vector<int32> indexes_multi_old_to_new(old_indexes_multi_size);
454  typedef std::vector<std::pair<int32,int32> > PairVectorType;
455  typedef std::map<const PairVectorType*, int32,
456  PointerCompare<std::pair<int32,int32> > > MapType;
457  MapType indexes_multi_map;
458  for (int32 i = 0; i < computation_->indexes_multi.size(); i++) {
459  std::pair<MapType::iterator, bool> p =
460  indexes_multi_map.insert(std::pair<const PairVectorType*, int32>(
461  &(computation_->indexes_multi[i]), cur_index));
462  if (p.second) { // was inserted-- was not there already.
463  indexes_multi_old_to_new[i] = cur_index++;
464  } else {
465  int32 index_from_map = p.first->second;
466  indexes_multi_old_to_new[i] = index_from_map;
467  }
468  }
469  if (cur_index == old_indexes_multi_size)
470  return; // An optimization. No duplicates were found.
471  std::vector<PairVectorType> new_indexes_multi(cur_index);
472  for (int32 i = 0; i < old_indexes_multi_size; i++) {
473  int32 new_index = indexes_multi_old_to_new[i];
474  computation_->indexes_multi[i].swap(new_indexes_multi[new_index]);
475  }
476  computation_->indexes_multi.swap(new_indexes_multi);
477 
478  std::vector<int32*> indexes_multi_args;
479  IdentifyIndexesMultiArgs(&(computation_->commands), &indexes_multi_args);
480  std::vector<int32*>::const_iterator iter = indexes_multi_args.begin(),
481  end = indexes_multi_args.end();
482  for (; iter != end; ++iter)
483  **iter = indexes_multi_old_to_new[**iter];
484 }

487 void ComputationRenumberer::RenumberIndexes() {
488  int32 old_num_indexes = computation_->indexes.size();
489  if (old_num_indexes == 0)
490  return;
491  std::vector<int32*> indexes_args;
492  IdentifyIndexesArgs(&(computation_->commands), &indexes_args);
493 
494  std::vector<bool> indexes_seen(old_num_indexes, false);
495  std::vector<int32*>::const_iterator iter = indexes_args.begin(),
496  end = indexes_args.end();
497  for (; iter != end; ++iter)
498  indexes_seen[**iter] = true;
499 
500  std::vector<int32> old_to_new_index(old_num_indexes);
501  typedef std::map<const std::vector<int32>*, int32,
502  PointerCompare<int32> > MapType;
503  MapType indexes_map;
504 
505  int32 cur_index = 0;
506  for (int32 i = 0; i < old_num_indexes; i++) {
507  if (!indexes_seen[i]) {
508  old_to_new_index[i] = -1;
509  } else {
510  std::pair<MapType::iterator, bool> p =
511  indexes_map.insert(std::pair<const std::vector<int32>*, int32>(
512  &(computation_->indexes[i]), cur_index));
513  if (p.second) { // was inserted-- was not there already.
514  old_to_new_index[i] = cur_index++;
515  } else {
516  int32 index_from_map = p.first->second;
517  old_to_new_index[i] = index_from_map;
518  }
519  }
520  }
521  if (cur_index == old_num_indexes)
522  return; // An optimization. No changes to the numbering are made.
523  std::vector<std::vector<int32> > new_indexes(cur_index);
524  for (int32 i = 0; i < old_num_indexes; i++) {
525  int32 new_index = old_to_new_index[i];
526  if (new_index != -1)
527  computation_->indexes[i].swap(new_indexes[new_index]);
528  }
529  computation_->indexes.swap(new_indexes);
530 
531  // renumber the indexes inside the commmands.
532  for (iter = indexes_args.begin(); iter != end; ++iter) {
533  int32 old_index = **iter;
534  KALDI_ASSERT(old_index >= 0 && old_index < old_num_indexes);
535  int32 new_index = old_to_new_index[old_index];
536  KALDI_ASSERT(new_index >= 0);
537  **iter = new_index;
538  }
539 }

543  if (old_num_indexes_ranges == 0)

559  for (int32 i = 0; i < old_num_indexes_ranges; i++) {
560  if (!is_seen[i]) {
561  old_to_new_index[i] = -1;
562  } else {
563  std::pair<MapType::iterator, bool> p =
564  indexes_map.insert(
565  std::pair<const std::vector<std::pair<int32, int32> >*, int32>(
566  &(computation_->indexes_ranges[i]), cur_index));
567  if (p.second) { // was inserted-- was not there already.
568  old_to_new_index[i] = cur_index++;
569  } else {
570  int32 index_from_map = p.first->second;
571  old_to_new_index[i] = index_from_map;
572  }
573  }
574  }
575  if (cur_index == old_num_indexes_ranges)
576  return; // An optimization. No changes to the numbering are made.
577  std::vector<std::vector<std::pair<int32, int32> > > new_indexes_ranges(
578  cur_index);
579  for (int32 i = 0; i < old_num_indexes_ranges; i++) {
580  int32 new_index = old_to_new_index[i];
581  if (new_index != -1)
582  computation_->indexes_ranges[i].swap(new_indexes_ranges[new_index]);
583  }
584  computation_->indexes_ranges.swap(new_indexes_ranges);
585 
586  // renumber the indexes inside the commmands.
587  for (iter = indexes_ranges_args.begin(); iter != end; ++iter) {
588  int32 old_index = **iter;
589  KALDI_ASSERT(old_index >= 0 && old_index < old_num_indexes_ranges);
590  int32 new_index = old_to_new_index[old_index];
591  KALDI_ASSERT(new_index >= 0);
592  **iter = new_index;
593  }
594 }

706 bool VariableMergingOptimizer::MergeVariables() {
707  KALDI_ASSERT(!already_called_merge_variables_);
708  already_called_merge_variables_ = true;
709  if (!config_.optimize)
710  return false;
711  bool merged = false;
712  int32 num_commands = computation_->commands.size();
713  for (int32 command_index = 0; command_index < num_commands;
714  command_index++) {
715  // This loop looks for pairs of sub-matrix indexes s1,s2 that we could
716  // potentially merge into a single variable.
717  const NnetComputation::Command &c =
718  computation_->commands[command_index];
719  int32 s1 = -1, s2 = -1;
720  if (c.command_type == kMatrixCopy &&
721  config_.remove_assignments) {
722  s2 = c.arg1; // s2 is the written-to matrix.
723  s1 = c.arg2;
724  } else if (c.command_type == kPropagate &&
725  config_.propagate_in_place) {
726  const Component *component = nnet_.GetComponent(c.arg1);
727  if (component->Properties() & kPropagateInPlace) {
728  s1 = c.arg3;
729  s2 = c.arg4; // s2 is the written-to matrix.
730  }
731  } else if ((c.command_type == kBackprop ||
732  c.command_type == kBackpropNoModelUpdate) &&
733  config_.backprop_in_place) {
734  const Component *component = nnet_.GetComponent(c.arg1);
735  if (component->Properties() & kBackpropInPlace) {
736  s1 = c.arg5;
737  s2 = c.arg6; // s2 is the written-to matrix.
738  if (s1 == c.arg3 || s2 == c.arg3 || s1 == c.arg4 || s2 == c.arg4) {
739  // we don't think this should ever happen, but just out of an
740  // abundance of caution: if either of these submatrix indexes are the
741  // input-value or output-value args to Backprop, don't do the optimization.
742  s1 = -1;
743  s2 = -1;
744  }
745  }
746  }
747  if (s1 > 0 && s2 > 0) {
748  std::pair<bool,bool> p = MayBeMerged(command_index, s1, s2);
749  if (p.first) {
750  DoLeftMerge(command_index, s1, s2);
751  merged = true;
752  } else if (p.second) {
753  DoRightMerge(command_index, s1, s2);
754  merged = true;
755  }
756  }
757  }
758  if (merged) {
759  RenumberComputation(computation_);
760  RemoveNoOps(computation_);
761  }
762  return merged;
763 }

803 void VariableMergingOptimizer::DoRightMerge(int32 command_index,
804  int32 s1, int32 s2) {
805  // Prevent further optimizations touching s1 or s2 (we can
806  // try again in a later round of optimization, with a new
807  // instance of this class).
808  MarkAsDirty(s1);
809  MarkAsDirty(s2);
810 
811  int32 m1 = computation_->submatrices[s1].matrix_index,
812  m2 = computation_->submatrices[s2].matrix_index;
813  KALDI_ASSERT(m1 != m2 && m1 > 0 && m2 > 0);
814  { // modify submatrices for submatrices of m1 to effectively be sub-matrices of
815  // s2 instead (they will refer to m2 as the matrix_index).
816  std::vector<int32>::const_iterator iter = matrix_to_submatrix_[m1].begin(),
817  end = matrix_to_submatrix_[m1].end();
818  for (; iter != end; ++iter) {
819  int32 submatrix_index = *iter;
820  KALDI_ASSERT(computation_->submatrices[submatrix_index].matrix_index==m1);
821  computation_->submatrices[submatrix_index] =
822  GetSubMatrixOfSubMatrix(*computation_, submatrix_index, s2);
823  }
824  }
825  const std::vector<MatrixAccesses> &matrix_accesses = analyzer_.matrix_accesses;
826  // - If m1 was an input, replace it as an input with m2
827  bool replaced = ReplaceInInput(nnet_, m1, m2, computation_);
828  KALDI_ASSERT(replaced == matrix_accesses[m1].is_input);
829  if (replaced) { // Remove the command that allocates m2.
830  int32 alloc_command = matrix_accesses[m2].allocate_command;
831  KALDI_ASSERT(alloc_command != -1);
832  computation_->commands[alloc_command].command_type =
833  kNoOperation;
834  }
835  // we keep matrix m2 (so m2 is m_to_keep, m1 is m_to_discard).
836  DoMergeCommon(command_index, m2, m1);
837 }

839 void VariableMergingOptimizer::DoMergeCommon(int32 command_index,
840  int32 m_to_keep,
841  int32 m_to_discard) {
842  ComputationAnalysis analysis(*computation_, analyzer_);
843  NnetComputation::Command &c = computation_->commands[command_index];
844  const std::vector<MatrixAccesses> &matrix_accesses =
845  analyzer_.matrix_accesses;
846 
847  // - If it was case (a), replace the assignment command with a no-op.
848  if (c.command_type == kMatrixCopy) {
849  // remove the command.
850  c.command_type = kNoOperation;
851  c.arg1 = -1;
852  c.arg2 = -1;
853  }
854 
855  // - If both m_to_keep and m_to_discard have commands that deallocate them,
856  // keep only the allocation command for m_to_keep, and make sure it's after
857  // the last access of m_to_discard (otherwise delete any deallocation
858  // command).
859  int32 dealloc_keep = matrix_accesses[m_to_keep].deallocate_command,
860  dealloc_discard = matrix_accesses[m_to_discard].deallocate_command;
861  if (dealloc_keep != -1 && dealloc_discard != -1) {
862  KALDI_ASSERT(analysis.LastMatrixAccess(m_to_discard) < dealloc_keep);
863  computation_->commands[dealloc_discard].command_type = kNoOperation;
864  } else {
865  if (dealloc_keep != -1)
866  computation_->commands[dealloc_keep].command_type =
867  kNoOperation;
868  if (dealloc_discard != -1)
869  computation_->commands[dealloc_discard].command_type =
870  kNoOperation;
871  }
872 
873  // - If both m_to_keep and m_to_discard have commands that allocate them,
874  // keep only the allocation command for m_to_keep and make sure it's
875  // before the first access of m_to_discard.
876  // (otherwise delete any allocation command).
877  int32 alloc_keep = matrix_accesses[m_to_keep].allocate_command,
878  alloc_discard = matrix_accesses[m_to_discard].allocate_command;
879  if (alloc_keep != -1 && alloc_discard != -1) {
880  KALDI_ASSERT(analysis.FirstMatrixAccess(m_to_discard) > alloc_keep);
881  NnetComputation::Command
882  &keep_alloc_command = computation_->commands[alloc_keep],
883  &discard_alloc_command = computation_->commands[alloc_discard];
884  discard_alloc_command.command_type = kNoOperation;
885  if (keep_alloc_command.command_type == kAllocMatrixUndefined) {
886  keep_alloc_command.command_type = kAllocMatrixZeroed;
887  } else if (keep_alloc_command.command_type == kAllocMatrixFromOther) {
888  keep_alloc_command.command_type = kAllocMatrixFromOtherZeroed;
889  }
890  } else {
891  if (alloc_keep != -1)
892  computation_->commands[alloc_keep].command_type =
893  kNoOperation;
894  if (alloc_discard != -1)
895  computation_->commands[alloc_discard].command_type =
896  kNoOperation;
897  }
898 }

939 std::pair<bool,bool> VariableMergingOptimizer::MayBeMerged(
940  int32 command_index, int32 s1, int32 s2) const {
941  KALDI_ASSERT(s1 > 0 && s2 > 0 && static_cast<size_t>(command_index) <
942  computation_->commands.size());
943  if (!config_.allow_left_merge && !config_.allow_right_merge)
944  return std::pair<bool,bool>(false,false);
945  int32 m1 = computation_->submatrices[s1].matrix_index,
946  m2 = computation_->submatrices[s2].matrix_index;
947  // we can't merge two different submatrices of the same matrix.
948  if (m1 == m2) return std::pair<bool,bool>(false,false);
949  std::vector<int32> variable_indexes;
950  analyzer_.variables.AppendVariablesForSubmatrix(s1, &variable_indexes);
951  analyzer_.variables.AppendVariablesForSubmatrix(s2, &variable_indexes);
952  std::vector<int32>::iterator iter = variable_indexes.begin(),
953  end = variable_indexes.end();
954  // condition c5:
955  for (; iter != end; ++iter)
956  if (variable_dirty_[*iter])
957  return std::pair<bool,bool>(false,false);
958  const std::vector<MatrixAccesses> &matrix_accesses = analyzer_.matrix_accesses;
959  const MatrixAccesses &m1_access = matrix_accesses[m1],
960  &m2_access = matrix_accesses[m2];
961  // condition c1:
962  if ((m1_access.is_input && m2_access.is_input) ||
963  (m1_access.is_output && m2_access.is_output))
964  return std::pair<bool,bool>(false,false);
965  // condition c2:
966  if ((m1_access.is_input || m1_access.is_output ||
967  m2_access.is_input || m2_access.is_output) &&
968  (!computation_->IsWholeMatrix(s1) ||
969  !computation_->IsWholeMatrix(s2)))
970  return std::pair<bool,bool>(false,false);
971  bool left = config_.allow_left_merge,
972  right = config_.allow_right_merge;
973  // condition c3:
974  if (!computation_->IsWholeMatrix(s2)) left = false;
975  if (!computation_->IsWholeMatrix(s1)) right = false;
976  if (!left && !right) // save some time.
977  return std::pair<bool,bool>(false,false);
978  bool is_assignment = (computation_->commands[command_index].command_type ==
979  kMatrixCopy);
980  ComputationAnalysis analysis(*computation_, analyzer_);
981  if (is_assignment) {
982  if (analysis.FirstAccess(s2) == command_index &&
983  analysis.LastWriteAccess(s1) < command_index &&
984  analysis.LastAccess(s1) <
985  analysis.DataInvalidatedCommand(command_index, s2)) {
986  return std::pair<bool,bool>(left, right); // possible success.
987  }
988  } else {
989  if (analysis.FirstAccess(s2) == command_index &&
990  analysis.LastAccess(s1) == command_index) {
991  return std::pair<bool,bool>(left, right); // possible success.
992  }
993  }
994  // failure.
995  return std::pair<bool,bool>(false,false);
996 }

1023 int32 ModelUpdateConsolidator::ConsolidateSubmatrices(
1024  const std::vector<int32> &commands,
1025  const std::vector<int32> &submatrices) {
1026  int32 num_submatrices = submatrices.size();
1027  KALDI_ASSERT(num_submatrices > 1 && commands.size() == submatrices.size());
1028  int32 first_submatrix = submatrices[0];
1029  int32 num_cols = computation_->submatrices[first_submatrix].num_cols,
1030  num_rows = 0;
1031  NnetComputation::MatrixDebugInfo debug_info;
1032  for (int32 i = 0; i < num_submatrices; i++) {
1033  int32 submatrix = submatrices[i];
1034  num_rows += computation_->submatrices[submatrix].num_rows;
1035  KALDI_ASSERT(computation_->submatrices[submatrix].num_cols == num_cols);
1036  if (!computation_->matrix_debug_info.empty())
1037  AppendDebugInfoForSubmatrix(submatrix, &debug_info);
1038  }
1039  // new_whole_submatrix is a new submatrix index corresponding to the whole
1040  // of a new matrix that we are creating.
1041  int32 new_whole_submatrix = computation_->NewMatrix(num_rows, num_cols);
1042  // Add a command at the very start, to initialize this new matrix.
1043  int32 new_matrix_index =
1044  computation_->submatrices[new_whole_submatrix].matrix_index;
1045  // we can later on optimize this zeroed initialization to an undefined
1046  // initialization.
1047  extra_commands_[0].push_back(
1048  NnetComputation::Command(kAllocMatrixZeroed, new_matrix_index));
1049  final_deallocate_commands_.push_back(
1050  NnetComputation::Command(kDeallocMatrix, new_matrix_index));
1051  if (!computation_->matrix_debug_info.empty())
1052  computation_->matrix_debug_info[new_matrix_index].Swap(&debug_info);
1053 
1054  int32 row_offset = 0;
1055  for (int32 i = 0; i < num_submatrices; i++) {
1056  int32 submatrix_index = submatrices[i];
1057  int32 this_num_rows = computation_->submatrices[submatrix_index].num_rows;
1058  // submatrix corresponding to the part of the new matrix corresponding
1059  // to 'submatrices[i]'.
1060  int32 new_submatrix = computation_->NewSubMatrix(new_whole_submatrix,
1061  row_offset, this_num_rows,
1062  0, num_cols);
1063  // Just before command 'commands[i]', add a command that assigns to the
1064  // submatrix numbered 'new_submatrix' the contents of the submatrix numbered
1065  // 'submatrices[i]'. Note: we hope that a later pass of optimization
1066  // (VariableMergingOptimization) will remove this redundant copy by
1067  // having the operation that created it right directly to the location
1068  // we want it to be.
1069  NnetComputation::Command c(kMatrixCopy, new_submatrix, submatrices[i]);
1070  extra_commands_[commands[i]].push_back(c);
1071  row_offset += this_num_rows;
1072  }
1073  KALDI_ASSERT(row_offset == num_rows);
1074  return new_whole_submatrix;
1075 }

1294 void DerivativeTimeLimiter::MapSimpleMatrixCommand(NnetComputation::Command *c) {
1295  int32 submatrix1 = c->arg1,
1296  submatrix2 = c->arg2;
1297  int32 submatrix1_mapped = submatrix_map_if_deriv_[submatrix1],
1298  submatrix2_mapped = submatrix_map_if_deriv_[submatrix2];
1299  if (submatrix1_mapped == submatrix1 &&
1300  submatrix2_mapped == submatrix2) {
1301  // nothing to do.
1302  return;
1303  }
1304  if (submatrix1_mapped == 0 || submatrix2_mapped == 0) {
1305  // remove the operation-- it has nothing to do.
1306  c->command_type = kNoOperation;
1307  return;
1308  }
1309  // left_prune1 is the nmber of rows pruned away on the left for submatrix1.
1310  int32 orig_num_rows = computation_->submatrices[submatrix1].num_rows,
1311  left_prune1, left_prune2, right_prune1, right_prune2;
1312  GetPruneValues(submatrix1, submatrix1_mapped, &left_prune1, &right_prune1);
1313  GetPruneValues(submatrix2, submatrix2_mapped, &left_prune2, &right_prune2);
1314  if (left_prune1 == left_prune2 && right_prune1 == right_prune2) {
1315  // we took the same number of rows away from the left and right for
1316  // both arguments; the normal mapped values will work in this case
1317  c->arg1 = submatrix1_mapped;
1318  c->arg2 = submatrix2_mapped;
1319  } else {
1320  // there is some kind of mismatch- we'll prune back to what remains
1321  // after applying the maximum pruning on the left and right.
1322  int32 left_prune = std::max(left_prune1, left_prune2),
1323  right_prune = std::max(right_prune1, right_prune2);
1324  if (left_prune + right_prune >= orig_num_rows) {
1325  // everything was pruned away; remove the operation.
1326  c->command_type = kNoOperation;
1327  return;
1328  } else {
1329  int32 num_rows = orig_num_rows - left_prune - right_prune;
1330  // note: the call NewSubMatrix effectively gives us a sub-matrix of a
1331  // subm-matrix.
1332  c->arg1 = computation_->NewSubMatrix(submatrix1,
1333  left_prune, num_rows, 0, -1);
1334  c->arg2 = computation_->NewSubMatrix(submatrix2,
1335  left_prune, num_rows, 0, -1);
1336  }
1337  }
1338 }

1343 void DerivativeTimeLimiter::MapIndexesCommand(NnetComputation::Command *c) {
1344  int32 output_submatrix = c->arg1,
1345  input_submatrix = c->arg2;
1346  int32 input_submatrix_mapped = submatrix_map_if_deriv_[input_submatrix],
1347  output_submatrix_mapped = submatrix_map_if_deriv_[output_submatrix];
1348  // input_submatrix_mapped and output_submatrix_mapped map both submatrices to
1349  // just the portion that we are treating as nonzero.
1350 
1351  if (input_submatrix_mapped == input_submatrix &&
1352  output_submatrix_mapped == output_submatrix) {
1353  return; // nothing is changed.
1354  }
1355  if (input_submatrix_mapped == 0 ||
1356  output_submatrix_mapped == 0) {
1357  // Either input or output is all zeros; make the command a no-op.
1358  // It may not be obvious that in the case of kCopyRows it would
1359  // be valid to make this a no-op (because what if the existing
1360  // contents were nonzero?), but we insist that this optimization
1361  // come before optimizations, and we know that the originally
1362  // generated computation would not overwrite a nonzero value
1363  // (and there are no undefined values because we make sure to
1364  // initialize everything with zeros; ununitialized values are
1365  // allowed only at a later optimization stage.
1366  c->command_type = kNoOperation;
1367  return;

1369  const std::vector<int32> &old_indexes = computation_->indexes[c->arg3];
1370 
1371  int32 left_prune_input, left_prune_output;
1372  GetPruneValues(input_submatrix, input_submatrix_mapped,
1373  &left_prune_input, NULL);
1374  GetPruneValues(output_submatrix, output_submatrix_mapped,
1375  &left_prune_output, NULL);
1376  int32 new_num_input_rows =
1377  computation_->submatrices[input_submatrix_mapped].num_rows,
1378  new_num_output_rows =
1379  computation_->submatrices[output_submatrix_mapped].num_rows;
1380  std::vector<int32> new_indexes(new_num_output_rows);
1381  bool must_keep_command = false;
1382  for (int32 i = 0; i < new_num_output_rows; i++) {
1383  // the index into the 'new_indexes' vector is the row of the output
1384  // submatrix; the value is the row of the input submatrix.
1385  int32 orig_index = old_indexes[i + left_prune_output];
1386  if (orig_index == -1) {
1387  new_indexes[i] = -1;
1388  } else {
1389  int32 mapped_index = orig_index - left_prune_input;
1390  if (mapped_index >= 0 && mapped_index < new_num_input_rows) {
1391  new_indexes[i] = mapped_index;
1392  must_keep_command = true;
1393  } else {
1394  // input was out of range (i.e. it takes a value that we are asserting
1395  // is zero)-- use -1 as the index.
1396  new_indexes[i] = -1;
1397  }
1398  }
1399  }
1400  if (!must_keep_command) {
1401  c->command_type = kNoOperation;
1402  return;
1403  }
1404  int32 new_indexes_index = computation_->indexes.size();
1405  computation_->indexes.push_back(new_indexes);
1406  c->arg1 = output_submatrix_mapped;
1407  c->arg2 = input_submatrix_mapped;
1408  c->arg3 = new_indexes_index;
1409 }

1704 void DerivativeTimeLimiter::LimitMatrices(const std::vector<bool> &will_limit) {
1705  // first modify 'submatrices'.
1706  int32 num_submatrices = computation_->submatrices.size(),
1707  num_matrices = computation_->matrices.size();
1708  for (int32 s = 1; s < num_submatrices; s++) {
1709  NnetComputation::SubMatrixInfo &submat_info = computation_->submatrices[s];
1710  int32 m = submat_info.matrix_index;
1711  if (will_limit[m]) {
1712  // we need to do something...
1713  const MatrixPruneInfo &prune_info = matrix_prune_info_[m];
1714  int32 matrix_num_rows = prune_info.row_end - prune_info.row_begin;
1715  KALDI_ASSERT(matrix_num_rows > 0 &&
1716  matrix_num_rows < computation_->matrices[m].num_rows);
1717  KALDI_ASSERT(prune_info.partly_inside_range);
1718  int32 new_row_begin = submat_info.row_offset - prune_info.row_begin;
1719  if (new_row_begin >= 0 &&
1720  submat_info.num_rows + new_row_begin <= matrix_num_rows) {
1721  // If this submatrix is entirely inside the limited range of the matrix,
1722  // then we modify its row_offset to account for the truncation of
1723  // rows to the left.
1724  submat_info.row_offset = new_row_begin;
1725  } else {
1726  // This submatrix is not entirely the kept range of the matrix.
1727  // We assume that this submatrix is never accessed directly (as when
1728  // we modified the computation we ensured this). We
1729  // give it a valid but stupid size of num-rows=1, num-cols=1, so
1730  // that if it ever does get accessed it should produce an error.
1731  submat_info.row_offset = 0;
1732  submat_info.num_rows = 1;
1733  submat_info.col_offset = 0;
1734  submat_info.num_cols = 1;
1735  }
1736  }
1737  }
1738  // next modify 'matrices'
1739  for (int32 m = 1; m < num_matrices; m++) {
1740  if (will_limit[m]) {
1741  const MatrixPruneInfo &prune_info = matrix_prune_info_[m];
1742  NnetComputation::MatrixInfo &matrix_info = computation_->matrices[m];
1743  if (!computation_->matrix_debug_info.empty()) {
1744  NnetComputation::MatrixDebugInfo &debug_info =
1745  computation_->matrix_debug_info[m];
1746  std::vector<Cindex> &cindexes = debug_info.cindexes;
1747  KALDI_ASSERT(cindexes.size() == static_cast<size_t>(matrix_info.num_rows));
1748  cindexes.erase(cindexes.begin() + prune_info.row_end, cindexes.end());
1749  cindexes.erase(cindexes.begin(),
1750  cindexes.begin() + prune_info.row_begin);
1751  }
1752  matrix_info.num_rows = prune_info.row_end - prune_info.row_begin;
1753  // num_cols stays the same.
1754  }
1755  }

1758 void DerivativeTimeLimiter::PruneMatrices() {
1759  Analyzer analyzer;
1760  analyzer.Init(nnet_, *computation_);
1761  KALDI_ASSERT(computation_->matrices.size() == entire_submatrix_.size());
1762  int32 num_matrices = computation_->matrices.size();
1763  std::vector<bool> will_limit(num_matrices, false);
1764  bool will_limit_at_least_one = false;
1765  for (int32 m = 1; m < num_matrices; m++) {
1766  const MatrixAccesses &accesses = analyzer.matrix_accesses[m];
1767  const MatrixPruneInfo &matrix_prune_info = matrix_prune_info_[m];
1768  if (matrix_prune_info.fully_inside_range ||
1769  accesses.is_input || accesses.is_output ||
1770  !computation_->matrix_debug_info[m].is_deriv)
1771  continue; // nothing to do: it's inside the time-range or not a
1772  // derivative.
1773  // if we got here it's not completely inside the time range, not an input or
1774  // an output, and it's a derivative.
1775  if (!matrix_prune_info.partly_inside_range) {
1776  // completely outside time range. we can prune the matrix if it is not an
1777  // input or output, and is never accessed apart from allocation.
1778  if (accesses.accesses.empty() ||
1779  (accesses.accesses.size() == 1 &&
1780  accesses.accesses[0].command_index == accesses.allocate_command)) {
1781  // we prune the matrix away. the only thing we need to do here is
1782  // to remove the allocation and deallocation commands.
1783  // they should exist, because we just checked that it's not an input
1784  // or an output.
1785  KALDI_ASSERT(accesses.allocate_command >= 0 &&
1786  accesses.deallocate_command >= 0);
1787  computation_->commands[accesses.allocate_command].command_type =
1788  kNoOperation;
1789  computation_->commands[accesses.deallocate_command].command_type =
1790  kNoOperation;
1791  }
1792  } else {
1793  // the matrix is partly inside the time range, it's a derivative, and not
1794  // an input or an output.
1795  if (CanLimitMatrix(analyzer, m)) {
1796  will_limit[m] = true;
1797  will_limit_at_least_one = true;
1798  }
1799  }
1800  }
1801  if (will_limit_at_least_one)
1802  LimitMatrices(will_limit);
1803 }
