42 bool IsSimpleNnet(const Nnet &nnet) {
43  // check that we have just one output node and it is
44  // called "output".
45  if (NumOutputNodes(nnet) != 1 ||
46  nnet.GetNodeIndex("output") == -1 ||
47  !nnet.IsOutputNode(nnet.GetNodeIndex("output")))
48  return false;
49  // check that there is an input node named "input".
50  if (nnet.GetNodeIndex("input") == -1 ||
51  !nnet.IsInputNode(nnet.GetNodeIndex("input")))
52  return false;
53  // if there was just one input, then it was named
54  // "input" and everything checks out.
55  if (NumInputNodes(nnet) == 1)
56  return true;
57  // Otherwise, there should be 2 inputs and one
58  // should be called "ivector".
59  return NumInputNodes(nnet) == 2 &&
60  nnet.GetNodeIndex("ivector") != -1 &&
61  nnet.IsInputNode(nnet.GetNodeIndex("ivector"));
62 }

79 // this non-exported function is used in ComputeSimpleNnetContext
80 // to compute the left and right context of the nnet for a particular
81 // window size and shift-length.
82 static void ComputeSimpleNnetContextForShift(
83  const Nnet &nnet,
84  int32 input_start,
85  int32 window_size,
86  int32 *left_context,
87  int32 *right_context) {
88 
89  int32 input_end = input_start + window_size;
90  IoSpecification input;
91  input.name = "input";
92  IoSpecification output;
93  output.name = "output";
94  IoSpecification ivector; // we might or might not use this.
95  ivector.name = "ivector";
96 
97  int32 n = rand() % 10;
98  // in the IoSpecification for now we we will request all the same indexes at
99  // output that we requested at input.
100  for (int32 t = input_start; t < input_end; t++) {
101  input.indexes.push_back(Index(n, t));
102  output.indexes.push_back(Index(n, t));
103  }
104  // the assumption here is that the network just requires the ivector at time
105  // t=0.
106  ivector.indexes.push_back(Index(n, 0));
107 
108  ComputationRequest request;
109  request.inputs.push_back(input);
110  request.outputs.push_back(output);
111  if (nnet.GetNodeIndex("ivector") != -1)
112  request.inputs.push_back(ivector);
113  std::vector<std::vector<bool> > computable;
114  EvaluateComputationRequest(nnet, request, &computable);
115 
116  KALDI_ASSERT(computable.size() == 1);
117  std::vector<bool> &output_ok = computable[0];
118  std::vector<bool>::iterator iter =
119  std::find(output_ok.begin(), output_ok.end(), true);
120  int32 first_ok = iter - output_ok.begin();
121  int32 first_not_ok = std::find(iter, output_ok.end(), false) -
122  output_ok.begin();
123  if (first_ok == window_size || first_not_ok <= first_ok)
124  KALDI_ERR << "No outputs were computable (perhaps not a simple nnet?)";
125  *left_context = first_ok;
126  *right_context = window_size - first_not_ok;
127 }

129 void ComputeSimpleNnetContext(const Nnet &nnet,
130  int32 *left_context,
131  int32 *right_context) {
132  KALDI_ASSERT(IsSimpleNnet(nnet));
133  int32 modulus = nnet.Modulus();
134  // modulus >= 1 is a number such that the network ought to be
135  // invariant to time shifts (of both the input and output) that
136  // are a multiple of this number. We need to test all shifts modulo
137  // this number in case the left and right context vary at all within
138  // this range.
139 
140  std::vector<int32> left_contexts(modulus + 1);
141  std::vector<int32> right_contexts(modulus + 1);
142 
143  // This will crash if the total context (left + right) is greater
144  // than window_size.
145  int32 window_size = 100;
146  // by going "<= modulus" instead of "< modulus" we do one more computation
147  // than we really need; it becomes a sanity check.
148  for (int32 input_start = 0; input_start <= modulus; input_start++)
149  ComputeSimpleNnetContextForShift(nnet, input_start, window_size,
150  &(left_contexts[input_start]),
151  &(right_contexts[input_start]));
152  KALDI_ASSERT(left_contexts[0] == left_contexts[modulus] &&
153  "nnet does not have the properties we expect.");
154  KALDI_ASSERT(right_contexts[0] == right_contexts[modulus] &&
155  "nnet does not have the properties we expect.");
156  *left_context =
157  *std::max_element(left_contexts.begin(), left_contexts.end());
158  *right_context =
159  *std::max_element(right_contexts.begin(), right_contexts.end());
160 }

186 void ComponentDotProducts(const Nnet &nnet1,
187  const Nnet &nnet2,
188  VectorBase<BaseFloat> *dot_prod) {
189  KALDI_ASSERT(nnet1.NumComponents() == nnet2.NumComponents());
190  int32 updatable_c = 0;
191  for (int32 c = 0; c < nnet1.NumComponents(); c++) {
192  const Component *comp1 = nnet1.GetComponent(c),
193  *comp2 = nnet2.GetComponent(c);
194  if (comp1->Properties() & kUpdatableComponent) {
195  const UpdatableComponent
196  *u_comp1 = dynamic_cast<const UpdatableComponent*>(comp1),
197  *u_comp2 = dynamic_cast<const UpdatableComponent*>(comp2);
198  KALDI_ASSERT(u_comp1 != NULL && u_comp2 != NULL);
199  dot_prod->Data()[updatable_c] = u_comp1->DotProduct(*u_comp2);
200  updatable_c++;
201  }
202  }
203  KALDI_ASSERT(updatable_c == dot_prod->Dim());
204 }
