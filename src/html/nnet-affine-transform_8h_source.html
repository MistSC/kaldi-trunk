46  void InitData(std::istream &is) {
47  // define options
48  float bias_mean = -2.0, bias_range = 2.0, param_stddev = 0.1;
49  float learn_rate_coef = 1.0, bias_learn_rate_coef = 1.0;
50  float max_norm = 0.0;
51  // parse config
52  std::string token;
53  while (!is.eof()) {
54  ReadToken(is, false, &token);
55  if (token == "<ParamStddev>") ReadBasicType(is, false, &param_stddev);
56  else if (token == "<BiasMean>") ReadBasicType(is, false, &bias_mean);
57  else if (token == "<BiasRange>") ReadBasicType(is, false, &bias_range);
58  else if (token == "<LearnRateCoef>") ReadBasicType(is, false, &learn_rate_coef);
59  else if (token == "<BiasLearnRateCoef>") ReadBasicType(is, false, &bias_learn_rate_coef);
60  else if (token == "<MaxNorm>") ReadBasicType(is, false, &max_norm);
61  else KALDI_ERR << "Unknown token " << token << ", a typo in config?"
62  << " (ParamStddev|BiasMean|BiasRange|LearnRateCoef|BiasLearnRateCoef)";
63  is >> std::ws; // eat-up whitespace
64  }
65 
66  //
67  // initialize
68  //
69  Matrix<BaseFloat> mat(output_dim_, input_dim_);
70  for (int32 r=0; r<output_dim_; r++) {
71  for (int32 c=0; c<input_dim_; c++) {
72  mat(r,c) = param_stddev * RandGauss(); // 0-mean Gauss with given std_dev
73  }
74  }
75  linearity_ = mat;
76  //
77  Vector<BaseFloat> vec(output_dim_);
78  for (int32 i=0; i<output_dim_; i++) {
79  // +/- 1/2*bias_range from bias_mean:
80  vec(i) = bias_mean + (RandUniform() - 0.5) * bias_range;
81  }
82  bias_ = vec;
83  //
84  learn_rate_coef_ = learn_rate_coef;
85  bias_learn_rate_coef_ = bias_learn_rate_coef;
86  max_norm_ = max_norm;
87  //
88  }

107  KALDI_ASSERT(linearity_.NumCols() == input_dim_);
108  KALDI_ASSERT(bias_.Dim() == output_dim_);
109  }

146  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
147  // precopy bias
148  out->AddVecToRows(1.0, bias_, 0.0);
149  // multiply by weights^t
150  out->AddMatMat(1.0, in, kNoTrans, linearity_, kTrans, 1.0);
151  }

160  void Update(const CuMatrixBase<BaseFloat> &input, const CuMatrixBase<BaseFloat> &diff) {
161  // we use following hyperparameters from the option class
162  const BaseFloat lr = opts_.learn_rate * learn_rate_coef_;
163  const BaseFloat lr_bias = opts_.learn_rate * bias_learn_rate_coef_;
164  const BaseFloat mmt = opts_.momentum;
165  const BaseFloat l2 = opts_.l2_penalty;
166  const BaseFloat l1 = opts_.l1_penalty;
167  // we will also need the number of frames in the mini-batch
168  const int32 num_frames = input.NumRows();
169  // compute gradient (incl. momentum)
170  linearity_corr_.AddMatMat(1.0, diff, kTrans, input, kNoTrans, mmt);
171  bias_corr_.AddRowSumMat(1.0, diff, mmt);
172  // l2 regularization
173  if (l2 != 0.0) {
174  linearity_.AddMat(-lr*l2*num_frames, linearity_);
175  }
176  // l1 regularization
177  if (l1 != 0.0) {
178  cu::RegularizeL1(&linearity_, &linearity_corr_, lr*l1*num_frames, lr);
179  }
180  // update
181  linearity_.AddMat(-lr, linearity_corr_);
182  bias_.AddVec(-lr_bias, bias_corr_);
183  // max-norm
184  if (max_norm_ > 0.0) {
185  CuMatrix<BaseFloat> lin_sqr(linearity_);
186  lin_sqr.MulElements(linearity_);
187  CuVector<BaseFloat> l2(OutputDim());
188  l2.AddColSumMat(1.0, lin_sqr, 0.0);
189  l2.ApplyPow(0.5); // we have per-neuron L2 norms
190  CuVector<BaseFloat> scl(l2);
191  scl.Scale(1.0/max_norm_);
192  scl.ApplyFloor(1.0);
193  scl.InvertElements();
194  linearity_.MulRowsVec(scl); // shink to sphere!
195  }
196  }
