80 OutputType ClassifyWxfilename(const std::string &filename) {
81  const char *c = filename.c_str();
82  if (*c == '\0' || (*c == '-' && c[1] == '\0')) return kStandardOutput; // ""
83  // or "-".
84  else if (*c == '|') return kPipeOutput; // An output pipe like "|blah".
85  else if (isspace(*c) || isspace(c[filename.length()-1])) return kNoOutput; //
86  // Leading or trailing space: can't interpret this.
87  else if ((*c == 't'||*c == 'b') && c[1] == ',') {
88  // We have detected that the user has supplied a wspecifier
89  // or rspecifier (as in kaldi-table.h) where a wxfilename was
90  // needed. Since this is almost certain not to be a real filename
91  // (and would cause confusion if it were a real filename), we
92  // refuse to deal with it.
93  return kNoOutput;
94  } else {
95  const char *d = c;
96  while (d[1] != '\0') d++; // go to last char.
97  if (*d == '|' || isspace(*d)) return kNoOutput; // An input pipe (not
98  // allowed in this context) or trailing space which is just wrong.
99  else if (isdigit(*d)) {
100  // OK, it could be a file, but we have to see if it's an offset into a
101  // file, which is not allowed.
102  while (isdigit(*d) && d > c) d--;
103  if (*d == ':') return kNoOutput; // Filename is like some_file:12345;
104  // not allowed,
105  else
106  return kFileOutput;
107  // as we cannot write to an offset into a file (and if we interpreted it
108  // as an actual filename, the reading code would misinterpret it as an
109  // offset.
110  } else {
111  // at this point it matched no other pattern so we assume a filename, but
112  // we check for '|' as it's a common source of errors to have pipe
113  // commands without the pipe in the right place. Say that it can't be
114  // classified.
115  if (strchr(c, '|') != NULL) {
116  KALDI_WARN << "Trying to classify wxfilename with pipe symbol in the"
117  " wrong place (pipe without | at the beginning?): " <<
118  filename;
119  return kNoOutput;
120  }
121  return kFileOutput; // matched no other pattern: assume it's an actual
122  // filename.
123  }
124  }
125 }

128 InputType ClassifyRxfilename(const std::string &filename) {
129  const char *c = filename.c_str();
130  if (*c == '\0' || (*c == '-' && c[1] == '\0')) return kStandardInput; // ""
131  // or "-".
132  else if (*c == '|') return kNoInput; // An output pipe like "|blah": not
133  // valid for input.
134  else if (isspace(*c) || isspace(c[filename.length()-1])) return kNoInput; //
135  // Leading or trailing space.
136  else if ((*c == 't'||*c == 'b') && c[1] == ',') {
137  // We have detected that the user has supplied a wspecifier
138  // or rspecifier (as in kaldi-table.h) where a wxfilename was
139  // needed. Since this is almost certain not to be a real filename
140  // (and would cause a lot of confusion if it were a real filename), we
141  // refuse to deal with it upfront.
142  return kNoInput;
143  } else {
144  const char *d = c;
145  while (d[1] != '\0') d++; // go to last char.
146  if (*d == '|') return kPipeInput; // an input pipe.
147  if (isspace(*d)) return kNoInput; // trailing space which is never valid.
148  else if (isdigit(*d)) {

157  // at this point it matched no other pattern so we assume a filename, but
158  // we check for '|' as it's a common source of errors to have pipe
159  // commands without the pipe in the right place. Say that it can't be
160  // classified in this case.

539  public:
540  // splits a filename like /my/file:123 into /my/file and the
541  // number 123. Crashes if not this format.
542  static void SplitFilename(const std::string &rxfilename,
543  std::string *filename,
544  size_t *offset) {
545  size_t pos = rxfilename.find_last_of(':');
546  KALDI_ASSERT(pos != std::string::npos); // would indicate error in calling
547  // code, as the filename is supposed to be of the correct form at this
548  // point.
549  *filename = std::string(rxfilename, 0, pos);
550  std::string number(rxfilename, pos+1);
551  bool ans = ConvertStringToInteger(number, offset);
552  if (!ans)
553  KALDI_ERR << "Cannot get offset from filename " << rxfilename
554  << " (possibly you compiled in 32-bit and have a >32-bit"
555  << " byte offset into a file; you'll have to compile 64-bit.";
556  }

558  bool Seek(size_t offset) {
559  size_t cur_pos = is_.tellg();
560  if (cur_pos == offset) return true;
561  else if (cur_pos<offset && cur_pos+100 > offset) {
562  // We're close enough that it may be faster to just
563  // read that data, rather than seek.
564  for (size_t i = cur_pos; i < offset; i++)
565  is_.get();
566  return (is_.tellg() == std::streampos(offset));
567  }
568  // Try to actually seek.
569  is_.seekg(offset, std::ios_base::beg);
570  if (is_.fail()) { // failbit or badbit is set [error happened]
571  is_.close();
572  return false; // failure.
573  } else {
574  is_.clear(); // Clear any failure bits (e.g. eof).
575  return true; // success.
576  }
577  }

579  // This Open routine is unusual in that it is designed to work even
580  // if it was already open. This for efficiency when seeking multiple
581  // times.
582  virtual bool Open(const std::string &rxfilename, bool binary) {
583  if (is_.is_open()) {
584  // We are opening when we have an already-open file.
585  // We may have to seek within this file, or else close it and
586  // open a different one.
587  std::string tmp_filename;
588  size_t offset;
589  SplitFilename(rxfilename, &tmp_filename, &offset);
590  if (tmp_filename == filename_ && binary == binary_) { // Just seek
591  is_.clear(); // clear fail bit, etc.
592  return Seek(offset);
593  } else {
594  is_.close(); // don't bother checking error status of is_.
595  filename_ = tmp_filename;
596  is_.open(MapOsPath(filename_).c_str(),
597  binary ? std::ios_base::in | std::ios_base::binary
598  : std::ios_base::in);
599  if (!is_.is_open()) return false;
600  else
601  return Seek(offset);
602  }
603  } else {
604  size_t offset;
605  SplitFilename(rxfilename, &filename_, &offset);
606  binary_ = binary;
607  is_.open(MapOsPath(filename_).c_str(),
608  binary ? std::ios_base::in | std::ios_base::binary
609  : std::ios_base::in);
610  if (!is_.is_open()) return false;
611  else
612  return Seek(offset);
613  }
614  }
