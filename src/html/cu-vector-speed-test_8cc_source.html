41 template<typename Real> void TestCuVectorSoftmax(int32 dim) {
42  BaseFloat time_in_secs = 0.02;
43  CuVector<Real> M(dim);
44  M.SetRandn();
45 
46  Timer tim;

51 
52  BaseFloat fdim = dim;

191 template<typename Real> void CudaVectorSpeedTest() {
192  std::vector<int32> sizes;
193  sizes.push_back(16);
194  sizes.push_back(32);
195  sizes.push_back(64);
196  sizes.push_back(128);
197  sizes.push_back(256);
198  sizes.push_back(1024);
199  int32 ns = sizes.size();
200  for (int32 s = 0; s < ns; s++)
201  TestCuVectorSoftmax<Real>(sizes[s]);
202  for (int32 s = 0; s < ns; s++)
203  TestCuVectorSum<Real>(sizes[s]);
204  for (int32 s = 0; s < ns; s++)
205  TestCuVectorVecVecOne<Real>(sizes[s]);
206  for (int32 s = 0; s < ns; s++) {
207  TestCuVectorAddDiagMatMat<Real>(sizes[s], kNoTrans, kNoTrans);
208  TestCuVectorAddDiagMatMat<Real>(sizes[s], kNoTrans, kTrans);
209  TestCuVectorAddDiagMatMat<Real>(sizes[s], kTrans, kNoTrans);
210  TestCuVectorAddDiagMatMat<Real>(sizes[s], kTrans, kTrans);
211  }
212  for (int32 s = 0; s < ns; s++) {
213  TestCuVectorAddDiagMat2<Real>(sizes[s], kNoTrans);
214  TestCuVectorAddDiagMat2<Real>(sizes[s], kTrans);
215  }
216  for (int32 s = 0; s < ns; s++) {
217  TestCuVectorAddRowSumMat<Real>(sizes[s], kNoTrans);
218  TestCuVectorAddRowSumMat<Real>(sizes[s], kTrans);
219  }
220  for (int32 s = 0; s < ns; s++) {
221  TestCuVectorAddColSumMat<Real>(sizes[s], kNoTrans);
222  TestCuVectorAddColSumMat<Real>(sizes[s], kTrans);
223  }
224 
225 }
