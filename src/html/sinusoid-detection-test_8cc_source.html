30 void AddSinusoidSimple(BaseFloat samp_freq,
31  const Sinusoid &sinusoid,
32  VectorBase<BaseFloat> *signal) {
33  for (int32 i = 0; i < signal->Dim(); i++)
34  (*signal)(i) += sinusoid.amplitude *
35  cos(M_2PI * sinusoid.freq / samp_freq * i + sinusoid.phase);
36 }

111 void UnitTestSinusoidDetector() {
112  BaseFloat samp_freq = 4000 + (rand() % 2000);
113  int32 num_samp = 128 + rand() % 400;
114  SinusoidDetector detector(samp_freq, num_samp);
115 
116  for (int32 i = 0; i < 40; i++) {
117 
118  Vector<BaseFloat> signal(num_samp);
119 
120  // Sinusoid ref_sinusoid(1.3, 312.5, M_PI * 0.0);
121  // Sinusoid ref_sinusoid(1.3, 324.125, M_PI * 0.5);
122 
123  BaseFloat nyquist = samp_freq * 0.5;
124  BaseFloat freq = nyquist * RandUniform();
125  BaseFloat amplitude = RandUniform();
126  BaseFloat phase = M_2PI * RandUniform();
127 
128  Sinusoid ref_sinusoid(amplitude, freq, phase);
129 
130  AddSinusoid(samp_freq, ref_sinusoid, &signal);
131 
132 
133  BaseFloat orig_energy = VecVec(signal, signal);
134  KALDI_LOG << "Real frequency is " << freq << ", amplitude "
135  << amplitude << ", phase " << phase << ", samp-freq "
136  << samp_freq;
137  KALDI_LOG << "Total energy of signal (with sinusoid) is " << orig_energy;
138 
139  Sinusoid sinusoid;
140  BaseFloat min_energy = 0.0;
141  BaseFloat energy = detector.DetectSinusoid(min_energy,
142  signal, &sinusoid);
143 
144  Vector<BaseFloat> new_signal(signal);
145  sinusoid.phase += M_PI; // Reverse the phase.
146  AddSinusoid(samp_freq, sinusoid, &new_signal);
147  BaseFloat delta_energy = VecVec(signal, signal) -
148  VecVec(new_signal, new_signal);
149  KALDI_LOG << "Projected delta energy = " << energy
150  << " and observed was " << delta_energy;
151 
152  BaseFloat remaining_energy = VecVec(new_signal, new_signal);
153  if (remaining_energy > 0.01 * orig_energy) {
154  KALDI_WARN << "Energy remaining is " << remaining_energy
155  << " vs. original " << orig_energy;
156  BaseFloat relative_freq = freq / nyquist;
157  BaseFloat inv_num_samp = 1.0 / num_samp;
158  // We only tolerate this kind of error for very ridiculous frequency,
159  // close to zero or the Nyquist.
160  KALDI_ASSERT(relative_freq < inv_num_samp ||
161  relative_freq > 1.0 - inv_num_samp);
162  }
163  }

228 void AddFreqToSignal(BaseFloat base_freq,
229  BaseFloat samp_freq,
230  BaseFloat tolerance,
231  BaseFloat gain,
232  VectorBase<BaseFloat> *signal) {
233  BaseFloat error_scale = (2 * RandUniform() - 1) * tolerance;
234  BaseFloat freq = base_freq * (1.0 + error_scale);
235  KALDI_VLOG(3) << "base-freq = " << base_freq << ", factor = " << error_scale;
236  for (int32 i = 0; i < signal->Dim(); i++)
237  (*signal)(i) += gain * sin(i * 2.0 * 3.14159 * freq / samp_freq);
238 }

241 void GenerateDtmfTestCase(
242  BaseFloat sampling_rate,
243  Vector<BaseFloat> *signal,
244  std::vector<MultiSinusoidDetectorOutput> *ref_output) {
245  // the "ref_output" should correlate with the first of each run of frames with the same label.
246 
247  BaseFloat min_duration_secs = 0.04; // min duration of dtmf or non-tone segments.
248  BaseFloat min_dialtone_duration_secs = 0.1;
249  BaseFloat frequency_tolerance = 0.035;
250  BaseFloat dialtone_frequency_tolerance = 0.4 * (440.0 - 425.0) / 440.0;
251 
252  int32 num_events = 2 * (5 + rand() % 5) + 1; // odd number.
253  int32 tot_signal_dim = 0;
254 
255  ref_output->resize(num_events);
256  std::vector<Vector<BaseFloat> > all_signals(num_events);
257  for (int32 i = 0; i < num_events; i++) {
258  MultiSinusoidDetectorOutput &this_output = (*ref_output)[i];
259  Vector<BaseFloat> &this_signal = all_signals[i];
260  BaseFloat duration_secs = min_duration_secs * (1 + rand() % 3);
261  int32 num_samp = sampling_rate * duration_secs;
262  tot_signal_dim += num_samp;
263 
264  this_signal.Resize(num_samp);
265  this_signal.SetRandn();
266 
267  if (i % 2 == 0); // do nothing;
268  else if (rand() % 2 == 0 && duration_secs >= min_dialtone_duration_secs) {
269  // dialtone.
270  BaseFloat freq;
271  if (rand() % 3 == 0) { freq = 350; }
272  else if (rand() % 2 == 0) { freq = 440; }
273  else { freq = 425; }
274  BaseFloat gain = 10.0 * (1.0 + rand() % 2);
275  AddFreqToSignal(freq, sampling_rate, dialtone_frequency_tolerance,
276  gain, &(this_signal));
277  this_output.freq1 = freq;
278  } else {
279  // dtmf. use a subset of tones as examples.
280  BaseFloat freq1, freq2;
281  char c;
282  if (rand() % 4 == 0) {
283  c = '8'; freq1 = 852; freq2 = 1336;
284  } else if (rand() % 3 == 0) {
285  c = '0'; freq1 = 941; freq2 = 1336;
286  } else if (rand() % 2 == 0) {
287  c = '#'; freq1 = 941; freq2 = 1477;
288  } else {
289  c = '1'; freq1 = 697; freq2 = 1209;
290  }
291  BaseFloat base_gain = 10.0 * (1.0 + (rand() % 3)),
292  gain_factor = 1.0 + 0.1 * (-2 + rand() % 5),
293  gain1 = base_gain, gain2 = gain_factor * base_gain;
294  AddFreqToSignal(freq1, sampling_rate, frequency_tolerance, gain1,
295  &(this_signal));
296  AddFreqToSignal(freq2, sampling_rate, frequency_tolerance, gain2,
297  &(this_signal));
298  this_output.freq1 = freq1;
299  this_output.freq2 = freq2;
300  }
301  }
302  signal->Resize(tot_signal_dim);
303  int32 signal_offset = 0;
304  for (int32 i = 0; i < num_events; i++) {
305  int32 this_dim = all_signals[i].Dim();
306  signal->Range(signal_offset, this_dim).CopyFromVec(all_signals[i]);
307  signal_offset += this_dim;
308  }
