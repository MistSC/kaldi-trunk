36 bool DecodeUtterance(LatticeBiglmFasterDecoder &decoder, // not const but is really an input.
37  DecodableInterface &decodable, // not const but is really an input.
38  const TransitionModel &trans_model,
39  const fst::SymbolTable *word_syms,
40  std::string utt,
41  double acoustic_scale,
42  bool determinize,
43  bool allow_partial,
44  Int32VectorWriter *alignment_writer,
45  Int32VectorWriter *words_writer,
46  CompactLatticeWriter *compact_lattice_writer,
47  LatticeWriter *lattice_writer,
48  double *like_ptr) { // puts utterance's like in like_ptr on success.
49  using fst::VectorFst;
50 
51  if (!decoder.Decode(&decodable)) {
52  KALDI_WARN << "Failed to decode file " << utt;
53  return false;
54  }
55  if (!decoder.ReachedFinal()) {
56  if (allow_partial) {
57  KALDI_WARN << "Outputting partial output for utterance " << utt
58  << " since no final-state reached\n";
59  } else {
60  KALDI_WARN << "Not producing output for utterance " << utt
61  << " since no final-state reached and "
62  << "--allow-partial=false.\n";
63  return false;
64  }
65  }
66 
67  double likelihood;
68  LatticeWeight weight;
69  int32 num_frames;
70  { // First do some stuff with word-level traceback...
71  VectorFst<LatticeArc> decoded;
72  decoder.GetBestPath(&decoded);
73  if (decoded.NumStates() == 0)
74  // Shouldn't really reach this point as already checked success.
75  KALDI_ERR << "Failed to get traceback for utterance " << utt;
76 
77  std::vector<int32> alignment;
78  std::vector<int32> words;
79  GetLinearSymbolSequence(decoded, &alignment, &words, &weight);
80  num_frames = alignment.size();
81  if (words_writer->IsOpen())
82  words_writer->Write(utt, words);
83  if (alignment_writer->IsOpen())
84  alignment_writer->Write(utt, alignment);
85  if (word_syms != NULL) {
86  std::cerr << utt << ' ';
87  for (size_t i = 0; i < words.size(); i++) {
88  std::string s = word_syms->Find(words[i]);
89  if (s == "")
90  KALDI_ERR << "Word-id " << words[i] <<" not in symbol table.";
91  std::cerr << s << ' ';
92  }
93  std::cerr << '\n';
94  }
95  likelihood = -(weight.Value1() + weight.Value2());
96  }
97 
98  // Get lattice, and do determinization if requested.
99  Lattice lat;
100  decoder.GetRawLattice(&lat);
101  if (lat.NumStates() == 0)
102  KALDI_ERR << "Unexpected problem getting lattice for utterance " << utt;
103  fst::Connect(&lat);
104  if (determinize) {
105  CompactLattice clat;
106  if (!DeterminizeLatticePhonePrunedWrapper(
107  trans_model,
108  &lat,
109  decoder.GetOptions().lattice_beam,
110  &clat,
111  decoder.GetOptions().det_opts))
112  KALDI_WARN << "Determinization finished earlier than the beam for "
113  << "utterance " << utt;
114  // We'll write the lattice without acoustic scaling.
115  if (acoustic_scale != 0.0)
116  fst::ScaleLattice(fst::AcousticLatticeScale(1.0 / acoustic_scale), &clat);
117  compact_lattice_writer->Write(utt, clat);
118  } else {
119  Lattice fst;
120  decoder.GetRawLattice(&fst);
121  if (fst.NumStates() == 0)
122  KALDI_ERR << "Unexpected problem getting lattice for utterance "
123  << utt;
124  fst::Connect(&fst); // Will get rid of this later... shouldn't have any
125  // disconnected states there, but we seem to.
126  if (acoustic_scale != 0.0) // We'll write the lattice without acoustic scaling
127  fst::ScaleLattice(fst::AcousticLatticeScale(1.0 / acoustic_scale), &fst);
128  lattice_writer->Write(utt, fst);
129  }
130  KALDI_LOG << "Log-like per frame for utterance " << utt << " is "
131  << (likelihood / num_frames) << " over "
132  << num_frames << " frames.";
133  KALDI_VLOG(2) << "Cost for utterance " << utt << " is "
134  << weight.Value1() << " + " << weight.Value2();
135  *like_ptr = likelihood;
136  return true;
137 }

155  "this decoder applies the difference during decoding\n"

157  "oldlm-fst-in newlm-fst-in features-rspecifier"
158  " lattice-wspecifier [ words-wspecifier [alignments-wspecifier] ]\n";

160  Timer timer;

162  BaseFloat acoustic_scale = 0.1;
163  LatticeBiglmFasterDecoderConfig config;
164 
165  std::string word_syms_filename;
166  config.Register(&po);
167  po.Register("acoustic-scale", &acoustic_scale, "Scaling factor for acoustic likelihoods");
168 
169  po.Register("word-symbol-table", &word_syms_filename, "Symbol table for words [for debug output]");
170  po.Register("allow-partial", &allow_partial, "If true, produce output even if end state was not reached.");
171 
172  po.Read(argc, argv);
173 
174  if (po.NumArgs() < 6 || po.NumArgs() > 8) {
175  po.PrintUsage();
176  exit(1);
177  }
178 
179  std::string model_in_filename = po.GetArg(1),
180  fst_in_str = po.GetArg(2),
181  old_lm_fst_rxfilename = po.GetArg(3),
182  new_lm_fst_rxfilename = po.GetArg(4),
183  feature_rspecifier = po.GetArg(5),
184  lattice_wspecifier = po.GetArg(6),
185  words_wspecifier = po.GetOptArg(7),
186  alignment_wspecifier = po.GetOptArg(8);
187

189  AmDiagGmm am_gmm;
190  {
191  bool binary;
192  Input ki(model_in_filename, &binary);
193  trans_model.Read(ki.Stream(), binary);
194  am_gmm.Read(ki.Stream(), binary);
195  }
196 
197  VectorFst<StdArc> *old_lm_fst = ReadFstKaldi(old_lm_fst_rxfilename);
198  ApplyProbabilityScale(-1.0, old_lm_fst); // Negate old LM probs...
199 
200  VectorFst<StdArc> *new_lm_fst = ReadFstKaldi(new_lm_fst_rxfilename);
201 
202  fst::BackoffDeterministicOnDemandFst<StdArc> old_lm_dfst(*old_lm_fst);
203  fst::BackoffDeterministicOnDemandFst<StdArc> new_lm_dfst(*new_lm_fst);
204  fst::ComposeDeterministicOnDemandFst<StdArc> compose_dfst(&old_lm_dfst,
205  &new_lm_dfst);
206  fst::CacheDeterministicOnDemandFst<StdArc> cache_dfst(&compose_dfst);
207 
208  bool determinize = config.determinize_lattice;
209  CompactLatticeWriter compact_lattice_writer;
210  LatticeWriter lattice_writer;
211  if (! (determinize ? compact_lattice_writer.Open(lattice_wspecifier)
212  : lattice_writer.Open(lattice_wspecifier)))
213  KALDI_ERR << "Could not open table for writing lattices: "
214  << lattice_wspecifier;
215 
216  Int32VectorWriter words_writer(words_wspecifier);
217 
218  Int32VectorWriter alignment_writer(alignment_wspecifier);
219 
220  fst::SymbolTable *word_syms = NULL;
221  if (word_syms_filename != "")
222  if (!(word_syms = fst::SymbolTable::ReadText(word_syms_filename)))
223  KALDI_ERR << "Could not read symbol table from file "
224  << word_syms_filename;
225 
226  double tot_like = 0.0;
227  kaldi::int64 frame_count = 0;
228  int num_success = 0, num_fail = 0;
229 
230 
231  if (ClassifyRspecifier(fst_in_str, NULL, NULL) == kNoRspecifier) {
232  SequentialBaseFloatMatrixReader feature_reader(feature_rspecifier);
233  // Input FST is just one FST, not a table of FSTs.
234  VectorFst<StdArc> *decode_fst = fst::ReadFstKaldi(fst_in_str);
235 
236  {
237  LatticeBiglmFasterDecoder decoder(*decode_fst, config, &cache_dfst);
238 
239  for (; !feature_reader.Done(); feature_reader.Next()) {
240  std::string utt = feature_reader.Key();
241  Matrix<BaseFloat> features (feature_reader.Value());
242  feature_reader.FreeCurrent();
243  if (features.NumRows() == 0) {
244  KALDI_WARN << "Zero-length utterance: " << utt;
245  num_fail++;
246  continue;
247  }
248 
249  DecodableAmDiagGmmScaled gmm_decodable(am_gmm, trans_model, features,
250  acoustic_scale);
251 
252 
253  double like;
254  if (DecodeUtterance(decoder, gmm_decodable, trans_model, word_syms,
255  utt, acoustic_scale, determinize, allow_partial,
256  &alignment_writer, &words_writer,
257  &compact_lattice_writer, &lattice_writer,
258  &like)) {
259  tot_like += like;
260  frame_count += features.NumRows();
261  num_success++;
262  } else num_fail++;
263  }
264  }
265  delete decode_fst; // delete this only after decoder goes out of scope.
266  } else { // We have different FSTs for different utterances.
267  SequentialTableReader<fst::VectorFstHolder> fst_reader(fst_in_str);
268  RandomAccessBaseFloatMatrixReader feature_reader(feature_rspecifier);
269  for (; !fst_reader.Done(); fst_reader.Next()) {
270  std::string utt = fst_reader.Key();
271  if (!feature_reader.HasKey(utt)) {
272  KALDI_WARN << "Not decoding utterance " << utt
273  << " because no features available.";
274  num_fail++;
275  continue;
276  }
277  const Matrix<BaseFloat> &features = feature_reader.Value(utt);
278  if (features.NumRows() == 0) {
279  KALDI_WARN << "Zero-length utterance: " << utt;
280  num_fail++;
281  continue;
282  }
283  LatticeBiglmFasterDecoder decoder(fst_reader.Value(), config,
284  &cache_dfst);
285  DecodableAmDiagGmmScaled gmm_decodable(am_gmm, trans_model, features,
286  acoustic_scale);
287  double like;
288  if (DecodeUtterance(decoder, gmm_decodable, trans_model, word_syms, utt,
289  acoustic_scale, determinize, allow_partial,
290  &alignment_writer, &words_writer,
291  &compact_lattice_writer, &lattice_writer,
292  &like)) {
293  tot_like += like;
294  frame_count += features.NumRows();
295  num_success++;
296  } else num_fail++;
297  }
298  }
299 
300  double elapsed = timer.Elapsed();
301  KALDI_LOG << "Time taken "<< elapsed
302  << "s: real-time factor assuming 100 frames/sec is "
303  << (elapsed*100.0/frame_count);
304  KALDI_LOG << "Done " << num_success << " utterances, failed for "
305  << num_fail;
306  KALDI_LOG << "Overall log-likelihood per frame is " << (tot_like/frame_count) << " over "
307  << frame_count<<" frames.";
308 
309  delete word_syms;
310  if (num_success != 0) return 0;
311  else return 1;
312  } catch(const std::exception &e) {
313  std::cerr << e.what();
314  return -1;
315  }
