37 Component* Component::ReadNew(std::istream &is, bool binary) {
38  std::string token;
39  ReadToken(is, binary, &token); // e.g. "<SigmoidComponent>".
40  token.erase(0, 1); // erase "<".
41  token.erase(token.length()-1); // erase ">".
42  Component *ans = NewComponentOfType(token);
43  if (!ans)
44  KALDI_ERR << "Unknown component type " << token;
45  ans->Read(is, binary);
46  return ans;
47 }

156 void UpdatableComponent::ReadUpdatableCommon(std::istream &is, bool binary) {
157  std::ostringstream opening_tag;
158  opening_tag << '<' << this->Type() << '>';
159  std::string token;
160  ReadToken(is, binary, &token);
161  if (token == opening_tag.str()) {
162  // if the first token is the opening tag, then
163  // ignore it and get the next tag.
164  ReadToken(is, binary, &token);
165  }
166  if (token == "<LearningRateFactor>") {
167  ReadBasicType(is, binary, &learning_rate_factor_);
168  ReadToken(is, binary, &token);
169  } else {
170  learning_rate_factor_ = 1.0;
171  }
172  if (token == "<IsGradient>") {
173  ReadBasicType(is, binary, &is_gradient_);
174  ReadToken(is, binary, &token);
175  } else {
176  is_gradient_ = false;
177  }
178  if (token == "<LearningRate>") {
179  ReadBasicType(is, binary, &learning_rate_);
180  } else {
181  KALDI_ERR << "Expected token <LearningRate>, got "
182  << token;
183  }
184 }

217 void NonlinearComponent::StoreStatsInternal(
218  const CuMatrixBase<BaseFloat> &out_value,
219  const CuMatrixBase<BaseFloat> *deriv) {
220  KALDI_ASSERT(out_value.NumCols() == InputDim());
221  // Check we have the correct dimensions.
222  if (value_sum_.Dim() != InputDim() ||
223  (deriv != NULL && deriv_sum_.Dim() != InputDim())) {
224  mutex_.Lock();
225  if (value_sum_.Dim() != InputDim()) {
226  value_sum_.Resize(InputDim());
227  count_ = 0.0;
228  }
229  if (deriv != NULL && deriv_sum_.Dim() != InputDim()) {
230  deriv_sum_.Resize(InputDim());
231  count_ = 0.0;
232  value_sum_.SetZero();
233  }
234  mutex_.Unlock();
235  }
236  count_ += out_value.NumRows();
237  CuVector<BaseFloat> temp(InputDim());
238  temp.AddRowSumMat(1.0, out_value, 0.0);
239  value_sum_.AddVec(1.0, temp);
240  if (deriv != NULL) {
241  temp.AddRowSumMat(1.0, *deriv, 0.0);
242  deriv_sum_.AddVec(1.0, temp);
243  }
244 }

278 void NonlinearComponent::Add(BaseFloat alpha, const Component &other_in) {
279  const NonlinearComponent *other =
280  dynamic_cast<const NonlinearComponent*>(&other_in);
281  KALDI_ASSERT(other != NULL);
282  if (value_sum_.Dim() == 0 && other->value_sum_.Dim() != 0)
283  value_sum_.Resize(other->value_sum_.Dim());
284  if (deriv_sum_.Dim() == 0 && other->deriv_sum_.Dim() != 0)
285  deriv_sum_.Resize(other->deriv_sum_.Dim());
286  if (other->value_sum_.Dim() != 0)
287  value_sum_.AddVec(alpha, other->value_sum_);
288  if (other->deriv_sum_.Dim() != 0)
289  deriv_sum_.AddVec(alpha, other->deriv_sum_);
290  count_ += alpha * other->count_;
291 }

293 void NonlinearComponent::Read(std::istream &is, bool binary) {
294  std::ostringstream ostr_beg, ostr_end;
295  ostr_beg << "<" << Type() << ">"; // e.g. "<SigmoidComponent>"
296  ostr_end << "</" << Type() << ">"; // e.g. "</SigmoidComponent>"
297  ExpectOneOrTwoTokens(is, binary, ostr_beg.str(), "<Dim>");
298  ReadBasicType(is, binary, &dim_); // Read dimension.
299  std::string tok; // TODO: remove back-compatibility code.
300  ReadToken(is, binary, &tok);
301  if (tok == "<ValueSum>") {
302  // this branch is for back compatibility. TODO: delete it
303  // after Dec 2015.
304  value_sum_.Read(is, binary);
305  ExpectToken(is, binary, "<DerivSum>");
306  deriv_sum_.Read(is, binary);
307  ExpectToken(is, binary, "<Count>");
308  ReadBasicType(is, binary, &count_);
309  ExpectToken(is, binary, ostr_end.str());
310  } else {
311  // The new format is more readable as we write values that are normalized by
312  // the count.
313  KALDI_ASSERT(tok == "<ValueAvg>");
314  value_sum_.Read(is, binary);
315  ExpectToken(is, binary, "<DerivAvg>");
316  deriv_sum_.Read(is, binary);
317  ExpectToken(is, binary, "<Count>");
318  ReadBasicType(is, binary, &count_);
319  value_sum_.Scale(count_);
320  deriv_sum_.Scale(count_);
321  ExpectToken(is, binary, ostr_end.str());
322  }
323 }

325 void NonlinearComponent::Write(std::ostream &os, bool binary) const {
326  std::ostringstream ostr_beg, ostr_end;
327  ostr_beg << "<" << Type() << ">"; // e.g. "<SigmoidComponent>"
328  ostr_end << "</" << Type() << ">"; // e.g. "</SigmoidComponent>"
329  WriteToken(os, binary, ostr_beg.str());
330  WriteToken(os, binary, "<Dim>");
331  WriteBasicType(os, binary, dim_);
332  // Write the values and derivatives in a count-normalized way, for
333  // greater readability in text form.
334  WriteToken(os, binary, "<ValueAvg>");
335  Vector<BaseFloat> temp(value_sum_);
336  if (count_ != 0.0) temp.Scale(1.0 / count_);
337  temp.Write(os, binary);
338  WriteToken(os, binary, "<DerivAvg>");
339 
340  temp.Resize(deriv_sum_.Dim(), kUndefined);
341  temp.CopyFromVec(deriv_sum_);
342  if (count_ != 0.0) temp.Scale(1.0 / count_);
343  temp.Write(os, binary);
344  WriteToken(os, binary, "<Count>");
345  WriteBasicType(os, binary, count_);
346  WriteToken(os, binary, ostr_end.str());
347 }
