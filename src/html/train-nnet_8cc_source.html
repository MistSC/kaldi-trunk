27 class NnetExampleBackgroundReader {
28  public:
29  NnetExampleBackgroundReader(int32 minibatch_size,
30  Nnet *nnet,
31  SequentialNnetExampleReader *reader):
32  minibatch_size_(minibatch_size), nnet_(nnet), reader_(reader),
33  finished_(false) {
34  // When this class is created, it spawns a thread which calls ReadExamples()
35  // in the background.
36  pthread_attr_t pthread_attr;
37  pthread_attr_init(&pthread_attr);
38  int32 ret;
39  // below, Run is the static class-member function.
40  if ((ret=pthread_create(&thread_, &pthread_attr,
41  Run, static_cast<void*>(this)))) {
42  const char *c = strerror(ret);
43  if (c == NULL) { c = "[NULL]"; }
44  KALDI_ERR << "Error creating thread, errno was: " << c;
45  }
46  // the following call is a signal that no-one is currently using the examples_ and
47  // formatted_examples_ class members.
48  consumer_semaphore_.Signal();
49  }
50  ~NnetExampleBackgroundReader() {
51  if (KALDI_PTHREAD_PTR(thread_) == 0)
52  KALDI_ERR << "No thread to join.";
53  if (pthread_join(thread_, NULL))
54  KALDI_ERR << "Error rejoining thread.";
55  }
56 
57  // This will be called in a background thread. It's responsible for
58  // reading and formatting the examples.
59  void ReadExamples() {
60  KALDI_ASSERT(minibatch_size_ > 0);
61  int32 minibatch_size = minibatch_size_;
62 
63 
64  // Loop over minibatches...
65  while (true) {
66  // When the following call succeeds we interpret it as a signal that
67  // we are free to write to the class-member variables examples_ and formatted_examples_.
68  consumer_semaphore_.Wait();
69 
70  examples_.clear();
71  examples_.reserve(minibatch_size);
72  // Read the examples.
73  for (; examples_.size() < minibatch_size && !reader_->Done(); reader_->Next())
74  examples_.push_back(reader_->Value());
75 
76  // Format the examples as a single matrix. The reason we do this here is
77  // that it's a somewhat CPU-intensive operation (involves decompressing
78  // the matrix), so we do it in a separate thread from the one that's
79  // controlling the GPU (assuming we're using a GPU), so we can get better
80  // GPU utilization. If we have no GPU this doesn't hurt us.
81  if (examples_.empty()) {
82  formatted_examples_.Resize(0, 0);
83  total_weight_ = 0.0;
84  } else {
85  FormatNnetInput(*nnet_, examples_, &formatted_examples_);
86  total_weight_ = TotalNnetTrainingWeight(examples_);
87  }
88 
89  bool finished = examples_.empty();
90 
91  // The following call alerts the main program thread (that calls
92  // GetNextMinibatch() that it can how use the contents of
93  // examples_ and formatted_examples_.
94  producer_semaphore_.Signal();
95 
96  // If we just read an empty minibatch (because no more examples),
97  // then return.
98  if (finished)
99  return;
100  }
101  }
102 
103  // this wrapper can be passed to pthread_create.
104  static void* Run(void *ptr_in) {
105  NnetExampleBackgroundReader *ptr =
106  reinterpret_cast<NnetExampleBackgroundReader*>(ptr_in);
107  ptr->ReadExamples();
108  return NULL;
109  }
110 
111  // This call makes available the next minibatch of input. It returns
112  // true if it got some, and false if there was no more available.
113  // It is an error if you call this function after it has returned false.
114  bool GetNextMinibatch(std::vector<NnetExample> *examples,
115  Matrix<BaseFloat> *formatted_examples,
116  double *total_weight) {
117  KALDI_ASSERT(!finished_);
118  // wait until examples_ and formatted_examples_ have been created by
119  // the background thread.
120  producer_semaphore_.Wait();
121  // the calls to swap and Swap are lightweight.
122  examples_.swap(*examples);
123  formatted_examples_.Swap(formatted_examples);
124  *total_weight = total_weight_;
125 
126  // signal the background thread that it is now free to write
127  // again to examples_ and formatted_examples_.
128  consumer_semaphore_.Signal();
129 
130  if (examples->empty()) {
131  finished_ = true;
132  return false;
133  } else {
134  return true;
135  }
136  }
137 
138  private:
139  int32 minibatch_size_;
140  Nnet *nnet_;
141  SequentialNnetExampleReader *reader_;
142  pthread_t thread_;
143 
144  std::vector<NnetExample> examples_;
145  Matrix<BaseFloat> formatted_examples_;
146  double total_weight_; // total weight, from TotalNnetTrainingWeight(examples_).
147  // better to compute this in the background thread.
148 
149  Semaphore producer_semaphore_;
150  Semaphore consumer_semaphore_;
151 
152  bool finished_;
153 };

157 int64 TrainNnetSimple(const NnetSimpleTrainerConfig &config,
158  Nnet *nnet,
159  SequentialNnetExampleReader *reader,
160  double *tot_weight_ptr,
161  double *tot_logprob_ptr) {
162  int64 num_egs_processed = 0;
163  double tot_weight = 0.0, tot_logprob = 0.0;
164  NnetExampleBackgroundReader background_reader(config.minibatch_size,
165  nnet, reader);
166  KALDI_ASSERT(config.minibatches_per_phase > 0);
167  while (true) {
168  // Iterate over phases. A phase of training is just a certain number of
169  // minibatches, and its only significance is that it's the periodicity with
170  // which we print diagnostics.
171  double tot_weight_this_phase = 0.0, tot_logprob_this_phase = 0.0;
172 
173  int32 i;
174  for (i = 0; i < config.minibatches_per_phase; i++) {
175  std::vector<NnetExample> examples;
176  Matrix<BaseFloat> examples_formatted;
177  double minibatch_total_weight; // this will normally equal minibatch size.
178  if (!background_reader.GetNextMinibatch(&examples, &examples_formatted,
179  &minibatch_total_weight))
180  break;
181  tot_logprob_this_phase += DoBackprop(*nnet, examples, &examples_formatted,
182  nnet, NULL);
183  tot_weight_this_phase += minibatch_total_weight;
184  num_egs_processed += examples.size();
185  }
186  if (i != 0) {
187  KALDI_LOG << "Training objective function (this phase) is "
188  << (tot_logprob_this_phase / tot_weight_this_phase) << " over "
189  << tot_weight_this_phase << " frames.";
190  }
191  tot_weight += tot_weight_this_phase;
192  tot_logprob += tot_logprob_this_phase;
193  if (i != config.minibatches_per_phase) {
194  // did not get all the minibatches we wanted because no more input.
195  // this is true if and only if we did "break" in the loop over i above.
196  break;
197  }
198  }
199  if (tot_weight == 0.0) {
200  KALDI_WARN << "No data seen.";
201  } else {
202  KALDI_LOG << "Did backprop on " << tot_weight
203  << " examples, average log-prob per frame is "
204  << (tot_logprob / tot_weight);
205  KALDI_LOG << "[this line is to be parsed by a script:] log-prob-per-frame="
206  << (tot_logprob / tot_weight);
207  }
208  if (tot_weight_ptr) *tot_weight_ptr = tot_weight;
209  if (tot_logprob_ptr) *tot_logprob_ptr = tot_logprob;
210  return num_egs_processed;
211 }
