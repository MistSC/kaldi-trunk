49  void InitData(std::istream &is) {
50  // parse config
51  std::string token;
52  while (!is.eof()) {
53  ReadToken(is, false, &token);
54  if (token == "<PoolSize>") ReadBasicType(is, false, &pool_size_);
55  else if (token == "<PoolStep>") ReadBasicType(is, false, &pool_step_);
56  else if (token == "<PoolStride>") ReadBasicType(is, false, &pool_stride_);
57  else KALDI_ERR << "Unknown token " << token << ", a typo in config?"
58  << " (PoolSize|PoolStep|PoolStride)";
59  is >> std::ws; // eat-up whitespace
60  }
61  // check
62  KALDI_ASSERT(pool_size_ != 0 && pool_step_ != 0 && pool_stride_ != 0);
63  }

98  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
99  // useful dims
100  int32 num_patches = input_dim_ / pool_stride_;
101  int32 num_pools = 1 + (num_patches - pool_size_) / pool_step_;
102 
103  // do the max-pooling (pools indexed by q)
104  for (int32 q = 0; q < num_pools; q++) {
105  // get output buffer of the pool
106  CuSubMatrix<BaseFloat> pool(out->ColRange(q*pool_stride_, pool_stride_));
107  pool.Set(-1e20); // reset (large negative value)
108  for (int32 r = 0; r < pool_size_; r++) { // max
109  int32 p = r + q * pool_step_; // p = input patch
110  pool.Max(in.ColRange(p*pool_stride_, pool_stride_));
111  }
112  }
113  }

115  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
116  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
117  // useful dims
118  int32 num_patches = input_dim_ / pool_stride_;
119  int32 num_pools = 1 + (num_patches - pool_size_) / pool_step_;
120 
121  //
122  // here we note how many diff matrices are summed for each input patch,
123  std::vector<int32> patch_summands(num_patches, 0);
124  // this metainfo will be used to divide diff of patches
125  // used in more than one pool.
126  //
127 
128  in_diff->SetZero(); // reset
129 
130  for(int32 q=0; q<num_pools; q++) { // sum
131  for(int32 r=0; r<pool_size_; r++) {
132  int32 p = r + q * pool_step_; // patch number
133  //
134  CuSubMatrix<BaseFloat> in_p(in.ColRange(p*pool_stride_, pool_stride_));
135  CuSubMatrix<BaseFloat> out_q(out.ColRange(q*pool_stride_, pool_stride_));
136  //
137  CuSubMatrix<BaseFloat> tgt(in_diff->ColRange(p*pool_stride_, pool_stride_));
138  CuMatrix<BaseFloat> src(out_diff.ColRange(q*pool_stride_, pool_stride_));
139 
140  // Only the pool-inputs with 'max-values' are used to back-propagate into,
141  // the rest of derivatives is zeroed-out by a mask.
142  CuMatrix<BaseFloat> mask;
143  in_p.EqualElementMask(out_q, &mask);
144  src.MulElements(mask);
145  tgt.AddMat(1.0, src);
146 
147  patch_summands[p] += 1;
148  }
149  }
150 
151  // divide diff by #summands (compensate for patches used in more pools)
152  for(int32 p=0; p<num_patches; p++) {
153  CuSubMatrix<BaseFloat> tgt(in_diff->ColRange(p*pool_stride_, pool_stride_));
154  KALDI_ASSERT(patch_summands[p] > 0); // patch at least in one pool
155  tgt.Scale(1.0/patch_summands[p]);
156  }
157  }
158 
159  private:
160  int32 pool_size_, // input patches used for pooling
161  pool_step_, // shift used for pooling (allow overlapping pools)
162  pool_stride_; // stride used to cut input matrix to a vector of matrices
163 };
