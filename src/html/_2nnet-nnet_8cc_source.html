29 // returns dimension that this node outputs.
30 int32 NetworkNode::Dim(const Nnet &nnet) const {
31  int32 ans;
32  switch (node_type) {
33  case kInput: case kDimRange:
34  ans = dim;
35  break;
36  case kDescriptor:
37  ans = descriptor.Dim(nnet);
38  break;
39  case kComponent:
40  ans = nnet.GetComponent(u.component_index)->OutputDim();
41  break;
42  default:
43  ans = 0; // suppress compiler warning
44  KALDI_ERR << "Invalid node type.";
45  }
46  KALDI_ASSERT(ans > 0);
47  return ans;
48 }

58 std::string Nnet::GetAsConfigLine(int32 node_index, bool include_dim) const {
59  std::ostringstream ans;
60  KALDI_ASSERT(node_index < nodes_.size() &&
61  nodes_.size() == node_names_.size());
62  const NetworkNode &node = nodes_[node_index];
63  const std::string &name = node_names_[node_index];
64  switch (node.node_type) {
65  case kInput:
66  ans << "input-node name=" << name << " dim=" << node.dim;
67  break;
68  case kDescriptor:
69  // assert that it's an output-descriptor, not one describing the input to
70  // a component-node.
71  KALDI_ASSERT(IsOutputNode(node_index));
72  ans << "output-node name=" << name << " input=";
73  node.descriptor.WriteConfig(ans, node_names_);
74  if (include_dim)
75  ans << " dim=" << node.Dim(*this);
76  ans << " objective=" << (node.u.objective_type == kLinear ? "linear" :
77  "quadratic");
78  break;
79  case kComponent:
80  ans << "component-node name=" << name << " component="
81  << component_names_[node.u.component_index] << " input=";
82  KALDI_ASSERT(nodes_[node_index-1].node_type == kDescriptor);
83  nodes_[node_index-1].descriptor.WriteConfig(ans, node_names_);
84  if (include_dim)
85  ans << " input-dim=" << nodes_[node_index-1].Dim(*this)
86  << " output-dim=" << node.Dim(*this);
87  break;
88  case kDimRange:
89  ans << "dim-range-node name=" << name << " input-node="
90  << node_names_[node.u.node_index] << " dim-offset="
91  << node.dim_offset << " dim=" << node.dim;
92  break;
93  default:
94  KALDI_ERR << "Unknown node type.";
95  }
96  return ans.str();
97 }

161 void Nnet::ReadConfig(std::istream &config_is) {
162 
163  std::vector<std::string> lines;
164  // Write into "lines" a config file corresponding to whatever
165  // nodes we currently have. Because the numbering of nodes may
166  // change, it's most convenient to convert to the text representation
167  // and combine the existing and new config lines in that representation.
168  const bool include_dim = false;
169  GetConfigLines(include_dim, &lines);
170 
171  // we'll later regenerate what we need from nodes_ and node_name_ from the
172  // string representation.
173  nodes_.clear();
174  node_names_.clear();
175 
176  int32 num_lines_initial = lines.size();
177 
178  ReadConfigFile(config_is, &lines);
179  // now "lines" will have comments removed and empty lines stripped out
180 
181  std::vector<std::string> first_tokens(lines.size());
182  std::vector<ConfigLine> config_lines(lines.size());
183  for (size_t i = 0; i < lines.size(); i++) {
184  std::istringstream is(lines[i]);
185  std::string first_token;
186  is >> first_token;
187  first_tokens[i] = first_token;
188  std::string rest_of_line;
189  getline(is, rest_of_line);
190  if (!config_lines[i].ParseLine(rest_of_line))
191  KALDI_ERR << "Could not parse config-file line '" << lines[i] << "'";
192 
193  }
194 
195  // the next line will possibly remove some elements from "first_tokens" and
196  // "config_lines" so nothing is doubly defined.
197  RemoveRedundantConfigLines(num_lines_initial, &first_tokens, &config_lines);
198 
199  int32 initial_num_components = components_.size();
200  for (int32 pass = 0; pass <= 1; pass++) {
201  for (size_t i = 0; i < config_lines.size(); i++) {
202  if (first_tokens[i] == "component") {
203  if (pass == 0)
204  ProcessComponentConfigLine(initial_num_components,
205  &(config_lines[i]));
206  } else if (first_tokens[i] == "component-node") {
207  ProcessComponentNodeConfigLine(pass, &(config_lines[i]));
208  } else if (first_tokens[i] == "input-node") {
209  if (pass == 0)
210  ProcessInputNodeConfigLine(&(config_lines[i]));
211  } else if (first_tokens[i] == "output-node") {
212  ProcessOutputNodeConfigLine(pass, &(config_lines[i]));
213  } else if (first_tokens[i] == "dim-range-node") {
214  ProcessDimRangeNodeConfigLine(pass, &(config_lines[i]));
215  } else {
216  KALDI_ERR << "Invalid config-file line ('" << first_tokens[i]
217  << "' not expected): " << config_lines[i].WholeLine();
218  }
219  }
220  }
221  Check();
222 }

224 // called only on pass 0 of ReadConfig.
225 void Nnet::ProcessComponentConfigLine(
226  int32 initial_num_components,
227  ConfigLine *config) {
228  std::string name, type;
229  if (!config->GetValue("name", &name))
230  KALDI_ERR << "Expected field name=<component-name> in config line: "
231  << config->WholeLine();
232  if (!IsToken(name)) // e.g. contains a space.
233  KALDI_ERR << "Component name '" << name << "' is not allowed, in line: "
234  << config->WholeLine();
235  if (!config->GetValue("type", &type))
236  KALDI_ERR << "Expected field type=<component-type> in config line: "
237  << config->WholeLine();
238  Component *new_component = Component::NewComponentOfType(type);
239  if (new_component == NULL)
240  KALDI_ERR << "Unknown component-type '" << type
241  << "' in config file. Check your code version and config.";
242  // the next call will call KALDI_ERR or KALDI_ASSERT and die if something
243  // went wrong.
244  new_component->InitFromConfig(config);
245  int32 index = GetComponentIndex(name);
246  if (index != -1) { // Replacing existing component.
247  if (index >= initial_num_components) {
248  // that index was something we added from this config.
249  KALDI_ERR << "You are adding two components with the same name: '"
250  << name << "'";
251  }
252  delete components_[index];
253  components_[index] = new_component;
254  } else {
255  components_.push_back(new_component);
256  component_names_.push_back(name);
257  }
258  if (config->HasUnusedValues())
259  KALDI_ERR << "Unused values '" << config->UnusedValues()
260  << " in config line: " << config->WholeLine();
261 }

264 void Nnet::ProcessComponentNodeConfigLine(
265  int32 pass,
266  ConfigLine *config) {
267 
268  std::string name;
269  if (!config->GetValue("name", &name))
270  KALDI_ERR << "Expected field name=<component-name> in config line: "
271  << config->WholeLine();
272 
273  std::string input_name = name + std::string("_input");
274  int32 input_node_index = GetNodeIndex(input_name),
275  node_index = GetNodeIndex(name);
276 
277  if (pass == 0) {
278  KALDI_ASSERT(input_node_index == -1 && node_index == -1);
279  // just set up the node types and names for now, we'll properly set them up
280  // on pass 1.
281  nodes_.push_back(NetworkNode(kDescriptor));
282  nodes_.push_back(NetworkNode(kComponent));
283  node_names_.push_back(input_name);
284  node_names_.push_back(name);
285  return;
286  } else {
287  KALDI_ASSERT(input_node_index != -1 && node_index == input_node_index + 1);
288  std::string component_name, input_descriptor;
289  if (!config->GetValue("component", &component_name))
290  KALDI_ERR << "Expected component=<component-name>, in config line: "
291  << config->WholeLine();
292  int32 component_index = GetComponentIndex(component_name);
293  if (component_index == -1)
294  KALDI_ERR << "No component named '" << component_name
295  << "', in config line: " << config->WholeLine();
296  nodes_[node_index].u.component_index = component_index;
297 
298  if (!config->GetValue("input", &input_descriptor))
299  KALDI_ERR << "Expected input=<input-descriptor>, in config line: "
300  << config->WholeLine();
301  std::vector<std::string> tokens;
302  if (!DescriptorTokenize(input_descriptor, &tokens))
303  KALDI_ERR << "Error tokenizing descriptor in config line "
304  << config->WholeLine();
305  std::vector<std::string> node_names_temp;
306  GetSomeNodeNames(&node_names_temp);
307  tokens.push_back("end of input");
308  const std::string *next_token = &(tokens[0]);
309  if (!nodes_[input_node_index].descriptor.Parse(node_names_temp,
310  &next_token))
311  KALDI_ERR << "Error parsing Descriptor in config line: "
312  << config->WholeLine();
313  if (config->HasUnusedValues())
314  KALDI_ERR << "Unused values '" << config->UnusedValues()
315  << " in config line: " << config->WholeLine();
316  }

346 void Nnet::ProcessOutputNodeConfigLine(
347  int32 pass,
348  ConfigLine *config) {
349  std::string name;
350  if (!config->GetValue("name", &name))
351  KALDI_ERR << "Expected field name=<input-name> in config line: "
352  << config->WholeLine();
353  int32 node_index = GetNodeIndex(name);
354  if (pass == 0) {
355  KALDI_ASSERT(node_index == -1);
356  nodes_.push_back(NetworkNode(kDescriptor));
357  node_names_.push_back(name);
358  } else {
359  KALDI_ASSERT(node_index != -1);
360  std::string input_descriptor;
361  if (!config->GetValue("input", &input_descriptor))
362  KALDI_ERR << "Expected input=<input-descriptor>, in config line: "
363  << config->WholeLine();
364  std::vector<std::string> tokens;
365  if (!DescriptorTokenize(input_descriptor, &tokens))
366  KALDI_ERR << "Error tokenizing descriptor in config line "
367  << config->WholeLine();
368  tokens.push_back("end of input");
369  // if the following fails it will die.
370  std::vector<std::string> node_names_temp;
371  GetSomeNodeNames(&node_names_temp);
372  const std::string *next_token = &(tokens[0]);
373  if (!nodes_[node_index].descriptor.Parse(node_names_temp, &next_token))
374  KALDI_ERR << "Error parsing descriptor (input=...) in config line "
375  << config->WholeLine();
376  std::string objective_type;
377  if (config->GetValue("objective", &objective_type)) {
378  if (objective_type == "linear") {
379  nodes_[node_index].u.objective_type = kLinear;
380  } else if (objective_type == "quadratic") {
381  nodes_[node_index].u.objective_type = kQuadratic;
382  } else {
383  KALDI_ERR << "Invalid objective type: " << objective_type;
384  }
385  } else {
386  // the default objective type is linear. This is what we use
387  // for softmax objectives; the LogSoftmaxLayer is included as the
388  // last layer, in this case.
389  nodes_[node_index].u.objective_type = kLinear;
390  }
391  if (config->HasUnusedValues())
392  KALDI_ERR << "Unused values '" << config->UnusedValues()
393  << " in config line: " << config->WholeLine();
394  }
395 }

461 // note: the input to this function is a config generated from the nnet,
462 // containing the node info, concatenated with a config provided by the user.
463 //static
464 void Nnet::RemoveRedundantConfigLines(int32 num_lines_initial,
465  std::vector<std::string> *first_tokens,
466  std::vector<ConfigLine> *configs) {
467  int32 num_lines = first_tokens->size();
468  KALDI_ASSERT(configs->size() == num_lines &&
469  num_lines_initial <= num_lines);
470  // node names and component names live in different namespaces.
471  unordered_map<std::string, int32, StringHasher> node_name_to_most_recent_line;
472  unordered_set<std::string, StringHasher> component_names;
473  typedef unordered_map<std::string, int32, StringHasher>::iterator IterType;
474 
475  std::vector<bool> to_remove(num_lines, false);
476  for (int32 line = 0; line < num_lines; line++) {
477  std::string first_token = (*first_tokens)[line];
478  ConfigLine &config_line = (*configs)[line];
479  std::string name;
480  if (!config_line.GetValue("name", &name))
481  KALDI_ERR << "Config line has no field 'name=xxx': "
482  << config_line.WholeLine();
483  if (!IsValidName(name))
484  KALDI_ERR << "Name '" << name << "' is not allowable, in line: "
485  << config_line.WholeLine();
486  if ((*first_tokens)[line] == "component") {
487  // a line starting with "component"... components live in their own
488  // namespace. No repeats are allowed because we never wrote them
489  // to the config generated from the nnet.
490  if (!component_names.insert(name).second) {
491  // we could not insert it because it was already there.
492  KALDI_ERR << "Component name " << name
493  << " appears twice in the same config file.";
494  }
495  } else {
496  // the line defines some sort of network node, e.g. component-node.
497  IterType iter = node_name_to_most_recent_line.find(name);
498  if (iter != node_name_to_most_recent_line.end()) {
499  // name is repeated.
500  int32 prev_line = iter->second;
501  if (prev_line >= num_lines_initial) {
502  // user-provided config contained repeat of node with this name.
503  KALDI_ERR << "Node name " << name
504  << " appears twice in the same config file.";
505  }
506  // following assert checks that the config-file generated
507  // from an actual nnet does not contain repeats.. that
508  // would be a bug so check it with assert.
509  KALDI_ASSERT(line >= num_lines_initial);
510  to_remove[prev_line] = true;
511  }
512  node_name_to_most_recent_line[name] = line;
513  }
514  }
515  // Now remove any lines with to_remove[i] = true.
516  std::vector<std::string> first_tokens_out;
517  std::vector<ConfigLine> configs_out;
518  first_tokens_out.reserve(num_lines);
519  configs_out.reserve(num_lines);
520  for (int32 i = 0; i < num_lines; i++) {
521  if (!to_remove[i]) {
522  first_tokens_out.push_back((*first_tokens)[i]);
523  configs_out.push_back((*configs)[i]);
524  }
525  }
526  first_tokens->swap(first_tokens_out);
527  configs->swap(configs_out);
528 }

565 void Nnet::Read(std::istream &is, bool binary) {
566  Destroy();
567  ExpectToken(is, binary, "<Nnet3>");
568  std::ostringstream config_file_out;
569  std::string cur_line;
570  getline(is, cur_line); // Eat up a single newline.
571  if (!(cur_line == "" || cur_line == "\r"))
572  KALDI_ERR << "Expected newline in config file, got " << cur_line;
573  while (getline(is, cur_line)) {
574  // config-file part of file is terminated by an empty line.
575  if (cur_line == "" || cur_line == "\r")
576  break;
577  config_file_out << cur_line << std::endl;
578  }
579  // Now we read the Components; later we try to parse the config_lines.
580  ExpectToken(is, binary, "<NumComponents>");
581  int32 num_components;
582  ReadBasicType(is, binary, &num_components);
583  KALDI_ASSERT(num_components >= 0 && num_components < 100000);
584  components_.resize(num_components, NULL);
585  component_names_.resize(num_components);
586  for (int32 c = 0; c < num_components; c++) {
587  ExpectToken(is, binary, "<ComponentName>");
588  ReadToken(is, binary, &(component_names_[c]));
589  components_[c] = Component::ReadNew(is, binary);
590  }
591  ExpectToken(is, binary, "</Nnet3>");
592  std::istringstream config_file_in(config_file_out.str());
593  this->ReadConfig(config_file_in);

596 void Nnet::Write(std::ostream &os, bool binary) const {
597  WriteToken(os, binary, "<Nnet3>");
598  os << std::endl;
599  std::vector<std::string> config_lines;
600  const bool include_dim = false;
601  GetConfigLines(include_dim, &config_lines);

610  WriteToken(os, binary, "<NumComponents>");
611  WriteBasicType(os, binary, num_components);

616  WriteToken(os, binary, component_names_[c]);
617  components_[c]->Write(os, binary);
618  if (!binary)
619  os << std::endl;
620  }
621  WriteToken(os, binary, "</Nnet3>");
622 }

660 void Nnet::Check() const {
661  int32 num_nodes = nodes_.size(),
662  num_input_nodes = 0,
663  num_output_nodes = 0;
664  KALDI_ASSERT(num_nodes != 0);
665  std::vector<bool> component_used(components_.size());
666  for (int32 n = 0; n < num_nodes; n++) {
667  const NetworkNode &node = nodes_[n];
668  std::string node_name = node_names_[n];
669  KALDI_ASSERT(GetNodeIndex(node_name) == n);
670  switch (node.node_type) {
671  case kInput:
672  KALDI_ASSERT(node.dim > 0);
673  num_input_nodes++;
674  break;
675  case kDescriptor: {
676  if (IsOutputNode(n))
677  num_output_nodes++;
678  std::vector<int32> node_deps;
679  node.descriptor.GetNodeDependencies(&node_deps);
680  SortAndUniq(&node_deps);
681  for (size_t i = 0; i < node_deps.size(); i++) {
682  int32 src_node = node_deps[i];
683  KALDI_ASSERT(src_node >= 0 && src_node < num_nodes);
684  NodeType src_type = nodes_[src_node].node_type;
685  if (src_type != kInput && src_type != kDimRange &&
686  src_type != kComponent)
687  KALDI_ERR << "Invalid source node type in Descriptor: source node "
688  << node_names_[src_node];
689  }
690  break;
691  }
692  case kComponent: {
693  KALDI_ASSERT(n > 0 && nodes_[n-1].node_type == kDescriptor);
694  const NetworkNode &src_node = nodes_[n-1];
695  const Component *c = GetComponent(node.u.component_index);
696  component_used[node.u.component_index] = true;
697  int32 src_dim = src_node.Dim(*this), input_dim = c->InputDim();
698  if (src_dim != input_dim) {
699  KALDI_ERR << "Dimension mismatch for network-node "
700  << node_name << ": input-dim "
701  << src_dim << " versus component-input-dim "
702  << input_dim;
703  }
704  break;
705  }
706  case kDimRange: {
707  int32 input_node = node.u.node_index;
708  KALDI_ASSERT(input_node >= 0 && input_node < num_nodes);
709  NodeType input_type = nodes_[input_node].node_type;
710  if (input_type != kInput && input_type != kComponent)
711  KALDI_ERR << "Invalid source node type in DimRange node: source node "
712  << node_names_[input_node];
713  int32 input_dim = nodes_[input_node].Dim(*this);
714  if (!(node.dim > 0 && node.dim_offset >= 0 &&
715  node.dim + node.dim_offset <= input_dim)) {
716  KALDI_ERR << "Invalid node dimensions for DimRange node: " << node_name
717  << ": input-dim=" << input_dim << ", dim=" << node.dim
718  << ", dim-offset=" << node.dim_offset;
719  }
720  break;
721  }
722  default:
723  KALDI_ERR << "Invalid node type for node " << node_name;
724  }
725  }
726 
727  int32 num_components = components_.size();
728  for (int32 c = 0; c < num_components; c++) {
729  const std::string &component_name = component_names_[c];
730  if (!component_used[c]) {
731  KALDI_WARN << "Orphan component " << component_name;
732  }
733  KALDI_ASSERT(GetComponentIndex(component_name) == c &&
734  "Duplicate component names?");
735  }
736  KALDI_ASSERT(num_input_nodes > 0);
737  KALDI_ASSERT(num_output_nodes > 0);
738 }
