123  bool GetNBestLattice(fst::MutableFst<CompactLatticeArc> *fst_out,
124  bool *was_final, int32 *nbest, BaseFloat *nbest_beam,
125  bool use_final_probs = true) {
126  // GetNBestLattice gets the n-best decoding output. If "use_final_probs"
127  // is true AND we reached a final state, it limits itself to final states;
128  // otherwise it does not take into account final-probs.
129  // fst_out will be empty (Start() == kNoStateId) if nothing was available.
130  // It returns number of paths if it got output (thus, fst_out will be
131  // nonempty), otherwise zero.
132  int n_paths = 0;
133  BaseFloat worst_final = 0.0,
134  best_final = std::numeric_limits<BaseFloat>::infinity();
135  *was_final = ReachedFinal();
136  Elem *last_toks = toks_.Clear(); // P <- C , C = {}
137  Token *tok = token_store_.CreateTok(0, NULL);
138  StateId end_state = 1E9; // some imaginary super end state
139  tok->c = Weight::Zero();
140  tok->I = NULL;
141  toks_.Insert(end_state, tok);
142  Elem *best_e = toks_.Find(end_state);
143 
144  if (!(*was_final)) { // only look for best tokens in this frame
145  for (Elem *e = last_toks, *e_tail; e != NULL; e = e_tail) {
146  token_store_.CombineN(best_e, e->val);
147  e_tail = e->tail;
148  toks_.Delete(e);
149  }
150  } else { // find best tokens in final states
151  for (Elem *e = last_toks, *e_tail; e != NULL; e = e_tail) {
152  Token *source = e->val;
153  Weight fw = use_final_probs ? fst_.Final(e->key) : Weight::One();
154  if (fw != Weight::Zero()) {
155  source->c = Times(source->c, fw);
156  token_store_.CombineN(best_e, source);
157  } else {
158  token_store_.DeleteTok(source);
159  }
160  e_tail = e->tail;
161  toks_.Delete(e);
162  }
163  }
164  // start building output FST
165  fst_out->DeleteStates();
166  StateId start_state = fst_out->AddState();
167  fst_out->SetStart(start_state);
168  last_toks = toks_.Clear(); // just tokens in super end state this time
169  if (last_toks->val == NULL) return false; // No output.
170  // go through tokens in imaginary super end state
171  for (Elem *e = last_toks, *e_tail = NULL; e != NULL; e = e_tail) {
172  Token *best_tok = e->val;
173  BaseFloat amscore = best_tok->ca.Value(),
174  lmscore = best_tok->c.Value() - amscore;
175  if (KALDI_ISINF(amscore) || KALDI_ISINF(lmscore)) {
176  KALDI_WARN << "infinity token! probably too narrow beam to retrieve n best";
177  e_tail = e->tail;
178  toks_.Delete(e);
179  continue; // skip that token
180  }
181  LatticeWeight path_w(lmscore, amscore);
182  CompactLatticeWeight path_weight(path_w, vector<int32>());
183 
184  std::vector<CompactLatticeArc*> arcs_reverse; // reverse order output arcs
185  // outer loop for word tokens
186  for (Token *tok = best_tok; tok != NULL; tok = tok->previous) {
187  // inner loop for input label tokens
188  std::vector<int32> str_rev, str;
189  for (SeqToken *stok = tok->I; stok != NULL; stok = stok->previous) {
190  str_rev.push_back(stok->i);
191  }
192  // reverse vector
193  std::vector<int32>::reverse_iterator rit;
194  for (rit = str_rev.rbegin(); rit < str_rev.rend(); ++rit)
195  str.push_back(*rit);
196  arcs_reverse.push_back(new CompactLatticeArc(
197  tok->o, tok->o, CompactLatticeWeight(LatticeWeight::One(), str), 0));
198  // no weight info (tok->c), no state info
199  }
200  token_store_.DeleteTok(best_tok);
201 
202  StateId cur_state = start_state;
203  for (ssize_t i = static_cast<ssize_t>(arcs_reverse.size())-1; i >= 0; i--) {
204  CompactLatticeArc *arc = arcs_reverse[i];
205  arc->nextstate = fst_out->AddState();
206  fst_out->AddArc(cur_state, *arc);
207  cur_state = arc->nextstate;
208  delete arc;
209  }
210  fst_out->SetFinal(cur_state, path_weight);
211  e_tail = e->tail;
212  toks_.Delete(e);
213  n_paths++;
214  BaseFloat p_score = amscore + lmscore;
215  if (p_score > worst_final) worst_final = p_score;
216  if (p_score < best_final) best_final = p_score;
217  }
218  if (nbest_beam) *nbest_beam = worst_final - best_final;
219  if (nbest) *nbest = n_paths;
220 
221  RemoveEpsLocal(fst_out);
222  return true;
223  }

247  class Token {
248  public:
249  // here will be the c and I of 'full tokens'
250  Weight c; // c (total weight)
251  Weight ca; // acoustic part of c
252  SeqToken *I; // sequence I
253  Label o; // o
254  Token *previous; // t'
255  int32 refs; // reference counter (for memory management)
256  unsigned hash; // hashing the output symbol sequence
257  inline bool operator < (const Token &other) {
258  return c.Value() > other.c.Value();
259  // This makes sense for log + tropical semiring.
260  }
261  inline bool Equal(Token *other) { // compares output sequences of Tokens
262  if (hash != other->hash) return false;
263  Token *t1 = this, *t2 = other;
264  while (t1 != NULL && t2 != NULL) {
265  if (t1->o != t2->o) return false;
266  t1 = t1->previous; t2 = t2->previous;
267  if (t1 == t2) { return true; }
268  if ((!t1) || (!t2)) { return false; }
269  }
270  KALDI_ASSERT(false); // should never reach this point
271  return true;
272  }

283  inline void DeleteSeq(SeqToken *seq) {
284  // delete seq token: either decrease reference count or put to linked list
285  seq->refs--;
286  if (seq->refs > 0) return;
287  // really kill sequence token
288  // clean up unused sequence tokens recursively backwards
289  if (seq->previous != NULL) DeleteSeq(seq->previous);
290  // save the unused sequence token in linked list (abusing *previous)
291  seq->previous = free_st_head_;
292  free_st_head_ = seq;
293  }
294  inline SeqToken *NewSeq() {
295  // new seq token: either take from linked list or extend list
296  SeqToken *tmp;
297  if (free_st_head_) { // take from free list
298  tmp = free_st_head_;
299  free_st_head_ = free_st_head_->previous;
300  } else { // make new entries in free list
301  tmp = new SeqToken[allocate_block_size_];
302  for (size_t i = 0; i + 1 < allocate_block_size_; i++) {
303  tmp[i].previous = tmp + i + 1;
304  }
305  tmp[allocate_block_size_ - 1].previous = NULL;
306  allocated_s_.push_back(tmp);
307  free_st_head_ = tmp->previous;
308  }
309  // initialize new sequence token (append or start new sequence)
310  tmp->refs = 1;
311  return tmp;
312  }
313  inline SeqToken *CreateSeq(Label input, SeqToken *prev) {

333  inline void DeleteTok(Token *tok) {
334  // delete token: either decrease reference count or put to linked list
335  tok->refs--;
336  if (tok->refs > 0) return;
337  // really kill token
338  // clean up unused tokens recursively backwards
339  if (tok->previous != NULL) {
340  DeleteTok(tok->previous);
341  }
342  if (tok->I != NULL) { // delete sequence I
343  DeleteSeq(tok->I);
344  }
345  // save the unused token in linked list (abusing *previous)
346  tok->previous = free_t_head_;
347  free_t_head_ = tok;
348  }
349  inline Token *CreateTok(Label output, Token *prev) {
350  // new token: either take from linked list or extend list
351  Token *tmp;
352  if (free_t_head_) { // take from free list
353  tmp = free_t_head_;
354  free_t_head_ = free_t_head_->previous;
355  } else { // make new entries in free list
356  tmp = new Token[allocate_block_size_];
357  for (size_t i = 0; i + 1 < allocate_block_size_; i++) {
358  tmp[i].previous = tmp + i + 1;
359  }
360  tmp[allocate_block_size_ - 1].previous = NULL;
361  allocated_t_.push_back(tmp);
362  free_t_head_ = tmp->previous;
363  }
364  // initialize data
365  tmp->refs = 1;
366  tmp->c = Weight::Zero();
367  tmp->I = NULL;
368  tmp->o = output;
369  tmp->previous = prev;

371  prev->refs++;
372  tmp->hash = prev->hash * 97 + static_cast<unsigned>(output);
373  } else {

379  inline Token* Combine(Token *tok1, Token *tok2) { // Viterbi version
380  KALDI_ASSERT(tok1);

382  if (tok1 == tok2) return tok1;

392  inline bool CombineN(Elem *head, Token *new_tok) { // n-best version
393  if (!new_tok) return false;
394  Elem *e = head;
395  StateId state = e->key;
396  BaseFloat new_weight = static_cast<BaseFloat>(new_tok->c.Value());
397  size_t count = 0;
398  BaseFloat worst_weight = 0.0; // small == low cost
399  Elem *worst_elem = NULL;
400  do {
401  count++;
402  Token *tok = e->val;
403  if (tok == new_tok) { return false; }
404  BaseFloat w = static_cast<BaseFloat>(tok->c.Value());
405  if (w > worst_weight) {
406  worst_weight = w;
407  worst_elem = e;
408  }
409  if (tok->Equal(new_tok)) { // if they have the same output sequence
410  if (w < new_weight) {
411  DeleteTok(new_tok);
412  return false;
413  } else {
414  DeleteTok(tok);
415  e->val = new_tok;
416  return true;
417  }
418  }
419  e = e->tail;
420  } while ( (e != NULL) && (e->key == state) );
421  // if we are here, no Token with the same output sequence was found
422  if (count < n_best_) {
423  toks_->InsertMore(state, new_tok);
424  return true;
425  } else {
426  if (worst_weight < new_weight) {
427  DeleteTok(new_tok);
428  return false;
429  } else {
430  DeleteTok(worst_elem->val);
431  worst_elem->val = new_tok;
432  return true;
433  }
434  }

436  inline Token* Advance(Token *source, Arc &arc, int32 frame,
437  BaseFloat cutoff) {
438  // compute new weight
439  Weight w = Times(source->c, arc.weight);
440  Weight amscore = Weight::One();
441  if (arc.ilabel > 0) { // emitting arc
442  amscore = Weight(- decodable_->LogLikelihood(frame, arc.ilabel));
443  w = Times(w, amscore);
444  }
445  Weight wa = Times(source->ca, amscore);
446  if (w.Value() > cutoff) { // prune
447  return NULL;
448  }
449  // create new token
450  Token *tok;
451  if (arc.olabel > 0) { // create new token
452  // find or create corresponding Token
453  tok = CreateTok(arc.olabel, source); // "new" Token
454  tok->I = CreateSeq(arc.ilabel, NULL); // new sequence I starts
455  } else { // append sequence I
456  tok = CreateTok(source->o, source->previous); // copy previous Token
457  tok->I = CreateSeq(arc.ilabel, source->I);
458  }
459  tok->c = w;
460  tok->ca = wa;
461  return tok;
462  }

501  BaseFloat GetCutoff(Elem *list_head, size_t *tok_count,
502  BaseFloat *adaptive_beam, Elem **best_elem) {
503  BaseFloat best_weight = 1.0e+10; // positive == high cost == bad.
504  size_t count = 0;
505  // find best token
506  tmp_array_.clear();
507  for (Elem *e = list_head; e != NULL; e = e->tail, count++) {
508  BaseFloat w = static_cast<BaseFloat>(e->val->c.Value());
509  tmp_array_.push_back(w); // ??check this - not always necessary
510  if (w < best_weight) {
511  best_weight = w;
512  if (best_elem) *best_elem = e;
513  }
514  }
515  if (tok_count != NULL) *tok_count = count;
516  // compute adaptive beam
517  if ((opts_.max_active == std::numeric_limits<int32>::max()) ||
518  (tmp_array_.size() <= static_cast<size_t>(opts_.max_active))) {
519  if (adaptive_beam != NULL) *adaptive_beam = opts_.beam;
520  return best_weight + opts_.beam;
521  } else {
522  // the lowest elements (lowest costs, highest likes)
523  // will be put in the left part of tmp_array.
524  std::nth_element(tmp_array_.begin(),
525  tmp_array_.begin() + opts_.max_active,
526  tmp_array_.end());
527  // return the tighter of the two beams.
528  BaseFloat ans = std::min(best_weight + opts_.beam,
529  *(tmp_array_.begin() + opts_.max_active));
530  if (adaptive_beam)
531  *adaptive_beam = std::min(opts_.beam,
532  ans - best_weight + opts_.beam_delta);
533  return ans;
534  }
535  }

545  // PropagateEmitting returns the likelihood cutoff used.
546  BaseFloat PropagateEmitting(int32 frame) {
547  Elem *last_toks = toks_.Clear(); // P <- C , C = {}
548  size_t tok_cnt;
549  BaseFloat adaptive_beam;
550  Elem *best_elem = NULL;
551  BaseFloat weight_cutoff = GetCutoff(last_toks, &tok_cnt,
552  &adaptive_beam, &best_elem);
553  PossiblyResizeHash(tok_cnt); // This makes sure the hash is always big enough.
554 
555  // This is the cutoff we use after adding in the log-likes (i.e.
556  // for the next frame). This is a bound on the cutoff we will use
557  // on the next frame.
558  BaseFloat next_weight_cutoff = 1.0e+10;
559 
560  // First process the best token to get a hopefully
561  // reasonably tight bound on the next cutoff.
562  if (best_elem) {
563  StateId state = best_elem->key;
564  Token *tok = best_elem->val;
565  for (fst::ArcIterator<fst::Fst<Arc> > aiter(fst_, state);
566  !aiter.Done();
567  aiter.Next()) {
568  Arc arc = aiter.Value();
569  if (arc.ilabel != 0) { // propagate..
570  arc.weight = Times(arc.weight,
571  Weight(- decodable_->LogLikelihood(frame, arc.ilabel)));
572  BaseFloat new_weight = arc.weight.Value() + tok->c.Value();
573  if (new_weight + adaptive_beam < next_weight_cutoff)
574  next_weight_cutoff = new_weight + adaptive_beam;
575  }
576  }
577  }
578 
579  // int32 n = 0, np = 0;
580 
581  // the tokens are now owned here, in last_toks, and the hash is empty.
582  // 'owned' is a complex thing here; the point is we need to call DeleteElem
583  // on each elem 'e' to let toks_ know we're done with them.
584 
585  for (Elem *e = last_toks, *e_tail; e != NULL; e = e_tail) {
586  // for all (s,t) in P
587  // n++;
588  // because we delete "e" as we go.
589  StateId state = e->key;
590  Token *tok = e->val;
591  if (tok->c.Value() < weight_cutoff) { // not pruned.
592  // np++;
593  // KALDI_ASSERT(state == tok->arc_.nextstate);
594  for (fst::ArcIterator<fst::Fst<Arc> > aiter(fst_, state);
595  !aiter.Done(); aiter.Next()) {
596  // for all a in A(state)
597  Arc arc = aiter.Value();
598  if (arc.ilabel != 0) { // propagate only emitting
599  Token *new_tok =
600  token_store_.Advance(tok, arc, frame, next_weight_cutoff);
601  if (new_tok) {
602  Elem *e_found = toks_.Find(arc.nextstate);
603  if (e_found == NULL) {
604  toks_.Insert(arc.nextstate, new_tok);
605  } else {
606  token_store_.CombineN(e_found, new_tok);
607  }
608  }
609  }
610  }
611  }
612  e_tail = e->tail; // several tokens with the same key can follow
613  token_store_.DeleteTok(e->val);
614  toks_.Delete(e);
615  }
616  // std::cerr << n << ', ' << np << ', ' <<adaptive_beam<<' ';
617  return adaptive_beam;

620  void PropagateEpsilon(BaseFloat adaptive_beam) {
621  // Processes nonemitting arcs for one frame. Propagates within
622  // cur_toks_.
623  KALDI_ASSERT(queue_.empty());
624  queue_.max_load_factor(1.0);
625  float best_weight = 1.0e+10;
626  for (const Elem *e = toks_.GetList(); e != NULL; e = e->tail) {
627  // queue_.push_back(e->key);
628  queue_.insert(e->key);
629  best_weight = std::min(best_weight, e->val->c.Value());
630  }
631  BaseFloat cutoff = best_weight + adaptive_beam;
632 
633  while (!queue_.empty()) {
634  // StateId state = queue_.back();
635  StateId state = *(queue_.begin());
636  // queue_.pop_back();
637  queue_.erase(queue_.begin());
638  Elem *elem = toks_.Find(state); // would segfault if state not
639  // in toks_ but this can't happen.
640 
641  // we have to pop all tokens with the same state
642  // this may create some unneccessary repetitions, since only the new token
643  // needs to be forwarded, but I don't know yet how to solve this
644  while (elem && elem->key == state) {
645  Token *tok = elem->val;
646  elem = elem->tail;
647 
648  if (tok->c.Value() > cutoff) { // Don't bother processing successors.
649  continue;
650  }
651  // KALDI_ASSERT(tok != NULL && state == tok->arc_.nextstate);
652  KALDI_ASSERT(tok != NULL);
653  for (fst::ArcIterator<fst::Fst<Arc> > aiter(fst_, state);
654  !aiter.Done(); aiter.Next()) {
655  // for all a in A(state)
656  Arc arc = aiter.Value();
657  if (arc.ilabel == 0) { // propagate nonemitting only...
658  Token *new_tok = token_store_.Advance(tok, arc, -1, cutoff); // -1:eps
659  if (new_tok) {
660  Elem *e_found = toks_.Find(arc.nextstate);
661  if (e_found == NULL) {
662  toks_.Insert(arc.nextstate, new_tok);
663  // queue_.push_back(arc.nextstate);
664  queue_.insert(arc.nextstate); // might be pushed several times
665  } else {
666  if (token_store_.CombineN(e_found, new_tok)) { // C was updated
667  // queue_.push_back(arc.nextstate);
668  queue_.insert(arc.nextstate);
669  }
670  }
671  }
672  } // if nonemitting
673  } // for Arciterator
674  } // while
675  }
676  }

679  // more than one list (e.g. for current and previous frames), but only one of
680  // them at a time can be indexed by StateId.
681  TokenHash toks_;
682  const fst::Fst<fst::StdArc> &fst_;
683  NBestDecoderOptions opts_;
684  typedef unordered_set<StateId> StateQueue;
685  StateQueue queue_; // used in PropagateEpsilon,
686  // std::vector<StateId> queue_; // temp variable used in PropagateEpsilon,
687  std::vector<BaseFloat> tmp_array_; // used in GetCutoff.
688  // make it class member to avoid internal new/delete.
689  TokenStore token_store_;
690  DecodableInterface *decodable_;
691 
692  // It might seem unclear why we call ClearToks(toks_.Clear()).
693  // There are two separate cleanup tasks we need to do at when we start a new file.
694  // one is to delete the Token objects in the list; the other is to delete
695  // the Elem objects. toks_.Clear() just clears them from the hash and gives ownership
696  // to the caller, who then has to call toks_.Delete(e) for each one. It was designed
697  // this way for convenience in propagating tokens from one frame to the next.
698  void ClearToks(Elem *list) {
699  for (Elem *e = list, *e_tail; e != NULL; e = e_tail) {
700  token_store_.DeleteTok(e->val);
701  e_tail = e->tail;
702  toks_.Delete(e);
703  }
704  }
