71 static void UnitTestPieces() {
72  KALDI_LOG << "=== UnitTestPieces() ===\n";
73  for (int32 n = 0; n < 10; n++) {
74  // the parametrization object
75  PitchExtractionOptions op1;
76  ProcessPitchOptions op2;
77  op2.delta_pitch_noise_stddev = 0.0; // to avoid mismatch of delta_log_pitch
78  // brought by rand noise.
79  op1.nccf_ballast_online = true; // this is necessary for the computation
80  // to be identical regardless how many pieces we break the signal into.
81 
82  int32 size = 10000 + rand() % 50000;
83 
84  Vector<BaseFloat> v(size);
85  // init with noise plus a sine-wave whose frequency is changing randomly.
86 
87  double cur_freq = 200.0, normalized_time = 0.0;
88 
89  for (int32 i = 0; i < size; i++) {
90  v(i) = RandGauss() + cos(normalized_time * M_2PI);
91  cur_freq += RandGauss(); // let the frequency wander a little.
92  if (cur_freq < 100.0) cur_freq = 100.0;
93  if (cur_freq > 300.0) cur_freq = 300.0;
94  normalized_time += cur_freq / op1.samp_freq;
95  }
96 
97  Matrix<BaseFloat> m1, m1p;
98 
99  // trying to have same opts as baseline.
100  ComputeKaldiPitch(op1, v, &m1);
101  ProcessPitch(op2, m1, &m1p);
102 
103  Matrix<BaseFloat> m2, m2p;
104 
105  { // compute it online with multiple pieces.
106  OnlinePitchFeature pitch_extractor(op1);
107  OnlineProcessPitch process_pitch(op2, &pitch_extractor);
108  int32 start_samp = 0;
109  while (start_samp < v.Dim()) {
110  int32 num_samp = rand() % (v.Dim() + 1 - start_samp);
111  SubVector<BaseFloat> v_part(v, start_samp, num_samp);
112  pitch_extractor.AcceptWaveform(op1.samp_freq, v_part);
113  start_samp += num_samp;
114  }
115  pitch_extractor.InputFinished();
116  int32 num_frames = pitch_extractor.NumFramesReady();
117  m2.Resize(num_frames, 2);
118  m2p.Resize(num_frames, process_pitch.Dim());
119  for (int32 frame = 0; frame < num_frames; frame++) {
120  SubVector<BaseFloat> row(m2, frame);
121  pitch_extractor.GetFrame(frame, &row);
122  SubVector<BaseFloat> rowp(m2p, frame);
123  process_pitch.GetFrame(frame, &rowp);
124  }
125  }
126  AssertEqual(m1, m2);
127  if (!ApproxEqual(m1p, m2p)) {
128  KALDI_ERR << "Post-processed pitch differs: " << m1p << " vs. " << m2p;
129  }
130  KALDI_LOG << "Test passed :)\n";
131  }

244 static void UnitTestComputeGPE() {
245  KALDI_LOG << "=== UnitTestComputeGPE ===\n";
246  int32 wrong_pitch = 0, tot_voiced = 0, tot_unvoiced = 0, num_frames = 0;
247  BaseFloat tol = 0.1, avg_d_kpitch = 0, real_pitch = 0;
248  for (int32 i = 1; i < 11; i++) {
249  std::string wavefile;
250  std::string num;
251  if (i < 6) {
252  num = "f" + ConvertIntToString(i) + "nw0000";
253  } else {
254  num = "m" + ConvertIntToString(i-5) + "nw0000";
255  }
256  Matrix<BaseFloat> gross_pitch;
257  std::string pitchfile = "keele/keele-true-lags/"+num+".txt";
258  std::ifstream pitch(pitchfile.c_str());
259  gross_pitch.Read(pitch, false);
260  Matrix<BaseFloat> kaldi_pitch;
261  std::string kfile = "keele/tmp/+"+num+"-kaldi.txt";
262  std::ifstream kpitch(kfile.c_str());
263  kaldi_pitch.Read(kpitch, false);
264  num_frames = std::min(kaldi_pitch.NumRows(),gross_pitch.NumRows());
265  for (int32 j = 1; j < num_frames; j++) {
266  if (gross_pitch(j,0) > 0.0) {
267  tot_voiced++;
268  real_pitch = 20000.0/gross_pitch(j,0);
269  if (fabs((real_pitch - kaldi_pitch(j,1))/real_pitch) > tol)
270  wrong_pitch++;
271  } else if (gross_pitch(j,0) == 0.0 && gross_pitch(j-1,0) == 0.0) {
272  tot_unvoiced++;
273  avg_d_kpitch += fabs(kaldi_pitch(j,1) - kaldi_pitch(j-1,1));
274  }
275  }
276  }
277  BaseFloat GPE = 1.0 * wrong_pitch / tot_voiced;
278  KALDI_LOG << " Gross Pitch Error with Rel.Error " << tol << " is " << GPE;
279  KALDI_LOG << "Average Kaldi delta_pitch for unvoiced regions " << avg_d_kpitch/tot_unvoiced;
280 }
