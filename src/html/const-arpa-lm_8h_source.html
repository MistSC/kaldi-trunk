232  num_words_(num_words), overflow_buffer_size_(overflow_buffer_size),
233  lm_states_size_(lm_states_size), unigram_states_(unigram_states),
234  overflow_buffer_(overflow_buffer), lm_states_(lm_states) {
235  KALDI_ASSERT(unigram_states_ != NULL);
236  KALDI_ASSERT(overflow_buffer_ != NULL);
237  KALDI_ASSERT(lm_states_ != NULL);
238  KALDI_ASSERT(ngram_order_ > 0);
239  KALDI_ASSERT(bos_symbol_ < num_words_ && bos_symbol_ > 0);
240  KALDI_ASSERT(eos_symbol_ < num_words_ && eos_symbol_ > 0);
241  KALDI_ASSERT(unk_symbol_ < num_words_ &&
242  (unk_symbol_ > 0 || unk_symbol_ == -1));
243  lm_states_end_ = lm_states_ + lm_states_size_ - 1;
244  memory_assigned_ = false;
245  initialized_ = true;
246  }

361  int32** unigram_states_;
362 
363  // Technically a 32-bit number cannot represent a possibly 64-bit pointer. We
364  // therefore use "relative" address instead of "absolute" address, which will
365  // be a small number most of the time. This buffer is for the case where the
366  // relative address has more than 30-bits.
367  int32** overflow_buffer_;
368 
369  // Memory chunk that contains the actual LmStates. One LmState has the
370  // following structure:
371  //
372  // struct LmState {
373  // float logprob;
374  // float backoff_logprob;
375  // int32 num_children;
376  // std::pair<int32, int32> [] children;
377  // }
378  //
379  // Note that the floating point representation has 4 bytes, int32 also has 4
380  // bytes, therefore one LmState will occupy the following number of bytes:
381  //
382  // x = 1 + 1 + 1 + 2 * children.size() = 3 + 2 * children.size()
383  int32* lm_states_;
