27 void MinimumBayesRisk::MbrDecode() {
28 
29  for (size_t counter = 0; ; counter++) {
30  NormalizeEps(&R_);
31  AccStats(); // writes to gamma_
32  double delta_Q = 0.0; // change in objective function.
33 
34  one_best_times_.clear();
35  one_best_confidences_.clear();
36 
37  // Caution: q in the line below is (q-1) in the algorithm
38  // in the paper; both R_ and gamma_ are indexed by q-1.
39  for (size_t q = 0; q < R_.size(); q++) {
40  if (do_mbr_) { // This loop updates R_ [indexed same as gamma_].
41  // gamma_[i] is sorted in reverse order so most likely one is first.
42  const vector<pair<int32, BaseFloat> > &this_gamma = gamma_[q];
43  double old_gamma = 0, new_gamma = this_gamma[0].second;
44  int32 rq = R_[q], rhat = this_gamma[0].first; // rq: old word, rhat: new.
45  for (size_t j = 0; j < this_gamma.size(); j++)
46  if (this_gamma[j].first == rq) old_gamma = this_gamma[j].second;
47  delta_Q += (old_gamma - new_gamma); // will be 0 or negative; a bound on
48  // change in error.
49  if (rq != rhat)
50  KALDI_VLOG(2) << "Changing word " << rq << " to " << rhat;
51  R_[q] = rhat;
52  }
53  if (R_[q] != 0) {
54  one_best_times_.push_back(times_[q]);
55  BaseFloat confidence = 0.0;
56  for (int32 j = 0; j < gamma_[q].size(); j++)
57  if (gamma_[q][j].first == R_[q]) confidence = gamma_[q][j].second;
58  one_best_confidences_.push_back(confidence);
59  }
60  }
61  KALDI_VLOG(2) << "Iter = " << counter << ", delta-Q = " << delta_Q;
62  if (delta_Q == 0) break;
63  if (counter > 100) {
64  KALDI_WARN << "Iterating too many times in MbrDecode; stopping.";
65  break;
66  }
67  }
68  RemoveEps(&R_);
69 }

99  for (int32 q = 1; q <= Q; q++)
100  alpha_dash(1, q) = alpha_dash(1, q-1) + l(0, r(q)); // Line 7.
101  for (int32 n = 2; n <= N; n++) {
102  double alpha_n = kLogZeroDouble;
103  for (size_t i = 0; i < pre_[n].size(); i++) {
104  const Arc &arc = arcs_[pre_[n][i]];
105  alpha_n = LogAdd(alpha_n, alpha(arc.start_node) + arc.loglike);
106  }
107  alpha(n) = alpha_n; // Line 10.
108  // Line 11 omitted: matrix was initialized to zero.
109  for (size_t i = 0; i < pre_[n].size(); i++) {
110  const Arc &arc = arcs_[pre_[n][i]];
111  int32 s_a = arc.start_node, w_a = arc.word;
112  BaseFloat p_a = arc.loglike;
113  for (int32 q = 0; q <= Q; q++) {
114  if (q == 0) {
115  alpha_dash_arc(q) = // line 15.
116  alpha_dash(s_a, q) + l(w_a, 0) + delta();
117  } else { // a1,a2,a3 are the 3 parts of min expression of line 17.
118  int32 r_q = r(q);
119  double a1 = alpha_dash(s_a, q-1) + l(w_a, r_q),
120  a2 = alpha_dash(s_a, q) + l(w_a, 0) + delta(),
121  a3 = alpha_dash_arc(q-1) + l(0, r_q);
122  alpha_dash_arc(q) = std::min(a1, std::min(a2, a3));
123  }
124  // line 19:
125  alpha_dash(n, q) += Exp(alpha(s_a) + p_a - alpha(n)) * alpha_dash_arc(q);
126  }
127  }
128  }

133 void MinimumBayesRisk::AccStats() {
134  using std::map;
135 
136  int32 N = static_cast<int32>(pre_.size()) - 1,
137  Q = static_cast<int32>(R_.size());
138 
139  Vector<double> alpha(N+1); // index (1...N)
140  Matrix<double> alpha_dash(N+1, Q+1); // index (1...N, 0...Q)
141  Vector<double> alpha_dash_arc(Q+1); // index 0...Q
142  Matrix<double> beta_dash(N+1, Q+1); // index (1...N, 0...Q)
143  Vector<double> beta_dash_arc(Q+1); // index 0...Q
144  vector<char> b_arc(Q+1); // integer in {1,2,3}; index 1...Q
145  vector<map<int32, double> > gamma(Q+1); // temp. form of gamma.
146  // index 1...Q [word] -> occ.
147 
148  // The tau arrays below are the sums over words of the tau_b
149  // and tau_e timing quantities mentioned in Appendix C of
150  // the paper... we are using these to get averaged times for
151  // the sausage bins, not specifically for the 1-best output.
152  Vector<double> tau_b(Q+1), tau_e(Q+1);
153 
154  double Ltmp = EditDistance(N, Q, alpha, alpha_dash, alpha_dash_arc);
155  if (L_ != 0 && Ltmp > L_) { // L_ != 0 is to rule out 1st iter.
156  KALDI_WARN << "Edit distance increased: " << Ltmp << " > "
157  << L_;
158  }
159  L_ = Ltmp;
160  KALDI_VLOG(2) << "L = " << L_;
161  // omit line 10: zero when initialized.
162  beta_dash(N, Q) = 1.0; // Line 11.
163  for (int32 n = N; n >= 2; n--) {
164  for (size_t i = 0; i < pre_[n].size(); i++) {
165  const Arc &arc = arcs_[pre_[n][i]];
166  int32 s_a = arc.start_node, w_a = arc.word;
167  BaseFloat p_a = arc.loglike;
168  alpha_dash_arc(0) = alpha_dash(s_a, 0) + l(w_a, 0) + delta(); // line 14.
169  for (int32 q = 1; q <= Q; q++) { // this loop == lines 15-18.
170  int32 r_q = r(q);
171  double a1 = alpha_dash(s_a, q-1) + l(w_a, r_q),
172  a2 = alpha_dash(s_a, q) + l(w_a, 0) + delta(),
173  a3 = alpha_dash_arc(q-1) + l(0, r_q);
174  if (a1 <= a2) {
175  if (a1 <= a3) { b_arc[q] = 1; alpha_dash_arc(q) = a1; }
176  else { b_arc[q] = 3; alpha_dash_arc(q) = a3; }
177  } else {
178  if (a2 <= a3) { b_arc[q] = 2; alpha_dash_arc(q) = a2; }
179  else { b_arc[q] = 3; alpha_dash_arc(q) = a3; }
180  }
181  }
182  beta_dash_arc.SetZero(); // line 19.
183  for (int32 q = Q; q >= 1; q--) {
184  // line 21:
185  beta_dash_arc(q) += Exp(alpha(s_a) + p_a - alpha(n)) * beta_dash(n, q);
186  switch (static_cast<int>(b_arc[q])) { // lines 22 and 23:
187  case 1:
188  beta_dash(s_a, q-1) += beta_dash_arc(q);
189  // next: gamma(q, w(a)) += beta_dash_arc(q)
190  AddToMap(w_a, beta_dash_arc(q), &(gamma[q]));
191  // next: accumulating times, see decl for tau_b,tau_e
192  tau_b(q) += state_times_[s_a] * beta_dash_arc(q);
193  tau_e(q) += state_times_[n] * beta_dash_arc(q);
194  break;
195  case 2:
196  beta_dash(s_a, q) += beta_dash_arc(q);
197  break;
198  case 3:
199  beta_dash_arc(q-1) += beta_dash_arc(q);
200  // next: gamma(q, epsilon) += beta_dash_arc(q)
201  AddToMap(0, beta_dash_arc(q), &(gamma[q]));
202  // next: accumulating times, see decl for tau_b,tau_e
203  // WARNING: there was an error in Appendix C. If we followed
204  // the instructions there the next line would say state_times_[sa], but
205  // it would be wrong. I will try to publish an erratum.
206  tau_b(q) += state_times_[n] * beta_dash_arc(q);
207  tau_e(q) += state_times_[n] * beta_dash_arc(q);
208  break;
209  default:
210  KALDI_ERR << "Invalid b_arc value"; // error in code.
211  }
212  }
213  beta_dash_arc(0) += Exp(alpha(s_a) + p_a - alpha(n)) * beta_dash(n, 0);
214  beta_dash(s_a, 0) += beta_dash_arc(0); // line 26.
215  }
216  }
217  beta_dash_arc.SetZero(); // line 29.
218  for (int32 q = Q; q >= 1; q--) {
219  beta_dash_arc(q) += beta_dash(1, q);
220  beta_dash_arc(q-1) += beta_dash_arc(q);
221  AddToMap(0, beta_dash_arc(q), &(gamma[q]));
222  // the statements below are actually redundant because
223  // state_times_[1] is zero.
224  tau_b(q) += state_times_[1] * beta_dash_arc(q);
225  tau_e(q) += state_times_[1] * beta_dash_arc(q);
226  }
227  for (int32 q = 1; q <= Q; q++) { // a check (line 35)
228  double sum = 0.0;
229  for (map<int32, double>::iterator iter = gamma[q].begin();
230  iter != gamma[q].end(); ++iter) sum += iter->second;
231  if (fabs(sum - 1.0) > 0.1)
232  KALDI_WARN << "sum of gamma[" << q << ",s] is " << sum;
233  }
234  // The next part is where we take gamma, and convert
235  // to the class member gamma_, which is using a different
236  // data structure and indexed from zero, not one.
237  gamma_.clear();
238  gamma_.resize(Q);
239  for (int32 q = 1; q <= Q; q++) {
240  for (map<int32, double>::iterator iter = gamma[q].begin();
241  iter != gamma[q].end(); ++iter)
242  gamma_[q-1].push_back(std::make_pair(iter->first, static_cast<BaseFloat>(iter->second)));
243  // sort gamma_[q-1] from largest to smallest posterior.
244  GammaCompare comp;
245  std::sort(gamma_[q-1].begin(), gamma_[q-1].end(), comp);
246  }
247  // We do the same conversion for the state times tau_b and tau_e:
248  // they get turned into the times_ data member, which has zero-based
249  // indexing.
250  times_.clear();
251  times_.resize(Q);
252  for (int32 q = 1; q <= Q; q++) {
253  times_[q-1].first = tau_b(q);
254  times_[q-1].second = tau_e(q);
255  if (times_[q-1].first > times_[q-1].second) // this is quite bad.
256  KALDI_WARN << "Times out of order";
257  if (q > 1 && times_[q-2].second > times_[q-1].first) {
258  // We previously had a warning here, but now we'll just set both
259  // those values to their average. It's quite possible for this
260  // condition to happen, but it seems like it would have a bad effect
261  // on the downstream processing, so we fix it.
262  double avg = 0.5 * (times_[q-2].second + times_[q-1].first);
263  times_[q-2].second = times_[q-1].first = avg;
264  }
265  }

268 void MinimumBayesRisk::PrepareLatticeAndInitStats(CompactLattice *clat) {
269  KALDI_ASSERT(clat != NULL);
270 
271  CreateSuperFinal(clat); // Add super-final state to clat... this is
272  // one of the requirements of the MBR algorithm, as mentioned in the
273  // paper (i.e. just one final state).
274 
275  // Topologically sort the lattice, if not already sorted.
276  kaldi::uint64 props = clat->Properties(fst::kFstProperties, false);
277  if (!(props & fst::kTopSorted)) {
278  if (fst::TopSort(clat) == false)
279  KALDI_ERR << "Cycles detected in lattice.";
280  }
281  CompactLatticeStateTimes(*clat, &state_times_); // work out times of
282  // the states in clat
283  state_times_.push_back(0); // we'll convert to 1-based numbering.
284  for (size_t i = state_times_.size()-1; i > 0; i--)
285  state_times_[i] = state_times_[i-1];
286 
287  // Now we convert the information in "clat" into a special internal
288  // format (pre_, post_ and arcs_) which allows us to access the
289  // arcs preceding any given state.
290  // Note: in our internal format the states will be numbered from 1,
291  // which involves adding 1 to the OpenFst states.
292  int32 N = clat->NumStates();
293  pre_.resize(N+1);
294 
295  // Careful: "Arc" is a class-member struct, not an OpenFst type of arc as one
296  // would normally assume.
297  for (int32 n = 1; n <= N; n++) {
298  for (fst::ArcIterator<CompactLattice> aiter(*clat, n-1);
299  !aiter.Done();
300  aiter.Next()) {
301  const CompactLatticeArc &carc = aiter.Value();
302  Arc arc; // in our local format.
303  arc.word = carc.ilabel; // == carc.olabel
304  arc.start_node = n;
305  arc.end_node = carc.nextstate + 1; // convert to 1-based.
306  arc.loglike = - (carc.weight.Weight().Value1() +
307  carc.weight.Weight().Value2());
308  // loglike: sum graph/LM and acoustic cost, and negate to
309  // convert to loglikes. We assume acoustic scaling is already done.
310 
311  pre_[arc.end_node].push_back(arcs_.size()); // record index of this arc.
312  arcs_.push_back(arc);
313  }
314  }
315 }

317 MinimumBayesRisk::MinimumBayesRisk(const CompactLattice &clat_in, bool do_mbr):
318  do_mbr_(do_mbr) {
319  CompactLattice clat(clat_in); // copy.
320 
321  PrepareLatticeAndInitStats(&clat);
322 
323  // We don't need to look at clat.Start() or clat.Final(state):
324  // we know clat.Start() == 0 since it's topologically sorted,
325  // and clat.Final(state) is Zero() except for One() at the last-
326  // numbered state, thanks to CreateSuperFinal and the topological
327  // sorting.
328 
329  { // Now set R_ to one best in the FST.
330  RemoveAlignmentsFromCompactLattice(&clat); // will be more efficient
331  // in best-path if we do this.
332  Lattice lat;
333  ConvertLattice(clat, &lat); // convert from CompactLattice to Lattice.
334  fst::VectorFst<fst::StdArc> fst;
335  ConvertLattice(lat, &fst); // convert from lattice to normal FST.
336  fst::VectorFst<fst::StdArc> fst_shortest_path;
337  fst::ShortestPath(fst, &fst_shortest_path); // take shortest path of FST.
338  std::vector<int32> alignment, words;
339  fst::TropicalWeight weight;
340  GetLinearSymbolSequence(fst_shortest_path, &alignment, &words, &weight);
341  KALDI_ASSERT(alignment.empty()); // we removed the alignment.
342  R_ = words;
343  L_ = 0.0; // Set current edit-distance to 0 [just so we know
344  // when we're on the 1st iter.]
345  }
346 
347  MbrDecode();
348 
349 }
