31 // test that determinization proceeds correctly on general
32 // FSTs (not guaranteed determinzable, but we use the
33 // max-states option to stop it getting out of control).
34 template<class Arc> void TestDeterminizeGeneral() {
35  int max_states = 100; // don't allow more det-states than this.
36  for(int i = 0; i < 100; i++) {
37  VectorFst<Arc> *fst = RandFst<Arc>();
38  std::cout << "FST before determinizing is:\n";
39  {
40 #ifdef HAVE_OPENFST_GE_10400
41  FstPrinter<Arc> fstprinter(*fst, NULL, NULL, NULL, false, true, "\t");
42 #else
43  FstPrinter<Arc> fstprinter(*fst, NULL, NULL, NULL, false, true);
44 #endif
45  fstprinter.Print(&std::cout, "standard output");
46  }
47  VectorFst<Arc> ofst;
48  try {
49  DeterminizeStar<Fst<Arc> >(*fst, &ofst, kDelta, NULL, max_states);
50  std::cout << "FST after determinizing is:\n";
51  {
52 #ifdef HAVE_OPENFST_GE_10400
53  FstPrinter<Arc> fstprinter(ofst, NULL, NULL, NULL, false, true, "\t");
54 #else
55  FstPrinter<Arc> fstprinter(ofst, NULL, NULL, NULL, false, true);
56 #endif
57  fstprinter.Print(&std::cout, "standard output");
58  }
59  assert(RandEquivalent(*fst, ofst, 5/*paths*/, 0.01/*delta*/, kaldi::Rand()/*seed*/, 100/*path length, max*/));
60  } catch (...) {
61  std::cout << "Failed to determinize *this FST (probably not determinizable)\n";
62  }
63  delete fst;
64  }
65 }
