39  // phone sequence. Make it "forwards" so the ctx_dep object can interpret it
40  // right. will also have to reverse the FST we produce.

44  << phone_window.size() << ", context-dependency object "

48  int32 phone = phone_window[P];

50  if (config.reverse)

52  "FST with a non-central \"central position\" P (i.e. asymmetric "
53  "context), but forgot to initialize the ContextFst object with P "
54  "as N-1-P (or it could be a simpler problem)";
55  else
56  KALDI_ERR << "phone == 0. Some mismatch happened, or there is "
57  "a code error.";
58  }

60  const HmmTopology &topo = trans_model.GetTopo();
61  const HmmTopology::TopologyEntry &entry = topo.TopologyForPhone(phone);
62 
63  // vector of the pdfs, indexed by pdf-class (pdf-classes must start from zero
64  // and be contiguous).

66  for (int32 pdf_class = 0;
67  pdf_class < static_cast<int32>(pdfs.size());
68  pdf_class++) {

72  ctx_ss << phone_window[i] << ' ';
73  KALDI_ERR << "GetHmmAsFst: context-dependency object could not produce "
74  << "an answer: pdf-class = " << pdf_class << " ctx-window = "
75  << ctx_ss.str() << ". This probably points "
76  "to either a coding error in some graph-building process, "
77  "a mismatch of topology with context-dependency object, the "
78  "wrong FST being passed on a command-line, or something of "
79  " that general nature.";
80  }
81  }
82  std::pair<int32, std::vector<int32> > cache_index(phone, pdfs);
83  if (cache != NULL) {
84  HmmCacheType::iterator iter = cache->find(cache_index);
85  if (iter != cache->end())
86  return iter->second;
87  }

89  VectorFst<StdArc> *ans = new VectorFst<StdArc>;

91  typedef StdArc Arc;
92  typedef Arc::Weight Weight;
93  typedef Arc::StateId StateId;
94  typedef Arc::Label Label;
95 
96  std::vector<StateId> state_ids;
97  for (size_t i = 0; i < entry.size(); i++)
98  state_ids.push_back(ans->AddState());
99  KALDI_ASSERT(state_ids.size() != 0); // Or empty topology entry.
100  ans->SetStart(state_ids[0]);
101  StateId final = state_ids.back();
102  ans->SetFinal(final, Weight::One());
103 
104  for (int32 hmm_state = 0;
105  hmm_state < static_cast<int32>(entry.size());
106  hmm_state++) {
107  int32 pdf_class = entry[hmm_state].pdf_class, pdf;
108  if (pdf_class == kNoPdf) pdf = kNoPdf; // nonemitting state.
109  else {
110  KALDI_ASSERT(pdf_class < static_cast<int32>(pdfs.size()));
111  pdf = pdfs[pdf_class];
112  }
113  int32 trans_idx;
114  for (trans_idx = 0;
115  trans_idx < static_cast<int32>(entry[hmm_state].transitions.size());
116  trans_idx++) {
117  BaseFloat log_prob;
118  Label label;
119  int32 dest_state = entry[hmm_state].transitions[trans_idx].first;
120  bool is_self_loop = (dest_state == hmm_state);
121  if (is_self_loop)
122  continue; // We will add self-loops in at a later stage of processing,
123  // not in this function.
124  if (pdf_class == kNoPdf) {
125  // no pdf, hence non-estimated probability.
126  // [would not happen with normal topology] . There is no transition-state
127  // involved in this case.
128  log_prob = Log(entry[hmm_state].transitions[trans_idx].second);
129  label = 0;
130  } else { // normal probability.
131  int32 trans_state =
132  trans_model.TripleToTransitionState(phone, hmm_state, pdf);
133  int32 trans_id =
134  trans_model.PairToTransitionId(trans_state, trans_idx);
135  log_prob = trans_model.GetTransitionLogProbIgnoringSelfLoops(trans_id);
136  // log_prob is a negative number (or zero)...
137  label = trans_id;
138  }
139  // Will add probability-scale later (we may want to push first).
140  ans->AddArc(state_ids[hmm_state],
141  Arc(label, label, Weight(-log_prob), state_ids[dest_state]));
142  }
143  }
144 
145  if (config.reverse) {
146  VectorFst<StdArc> *tmp = new VectorFst<StdArc>;
147  fst::Reverse(*ans, tmp);
148  fst::RemoveEpsLocal(tmp); // this is safe and will not blow up.
149  if (config.push_weights) // Push to make it stochastic again.
150  PushInLog<REWEIGHT_TO_INITIAL>(tmp, kPushWeights, config.push_delta);
151  delete ans;
152  ans = tmp;
153  } else {
154  fst::RemoveEpsLocal(ans); // this is safe and will not blow up.
155  }
156 
157  // Now apply probability scale.
158  // We waited till after the possible weight-pushing steps,
159  // because weight-pushing needs "real" weights in order to work.
160  ApplyProbabilityScale(config.transition_scale, ans);
161  if (cache != NULL)
162  (*cache)[cache_index] = ans;
163  return ans;
164 }

190  // final-states, which we usually will.]

198  state_ids.push_back(ans->AddState());

200  ans->SetStart(state_ids[0]);

202  ans->SetFinal(final, Weight::One());

206  hmm_state++) {
207  int32 pdf_class = entry[hmm_state].pdf_class, pdf;
208  if (pdf_class == kNoPdf) pdf = kNoPdf; // nonemitting state; not generally used.
209  else {
210  bool ans = ctx_dep.Compute(phone_window, pdf_class, &pdf);
211  KALDI_ASSERT(ans && "Context-dependency computation failed.");
212  }
213  int32 trans_idx;
214  for (trans_idx = 0;
215  trans_idx < static_cast<int32>(entry[hmm_state].transitions.size());
216  trans_idx++) {

218  Label label;
219  int32 dest_state = entry[hmm_state].transitions[trans_idx].first;
220  if (pdf_class == kNoPdf) {
221  // no pdf, hence non-estimated probability. very unusual case. [would
222  // not happen with normal topology] . There is no transition-state
223  // involved in this case.
224  KALDI_ASSERT(hmm_state != dest_state);
225  log_prob = Log(entry[hmm_state].transitions[trans_idx].second);
226  label = 0;
227  } else { // normal probability.
228  int32 trans_state =
229  trans_model.TripleToTransitionState(phone, hmm_state, pdf);
230  int32 trans_id =
231  trans_model.PairToTransitionId(trans_state, trans_idx);
232  log_prob = prob_scale * trans_model.GetTransitionLogProb(trans_id);
233  // log_prob is a negative number (or zero)...
234  label = trans_id;
235  }
236  ans->AddArc(state_ids[hmm_state],
237  Arc(label, label, Weight(-log_prob), state_ids[dest_state]));
238  }
239  }
240  return ans;
241 }

312 void GetIlabelMapping (const std::vector<std::vector<int32> > &ilabel_info_old,
313  const ContextDependencyInterface &ctx_dep,
314  const TransitionModel &trans_model,
315  std::vector<int32> *old2new_map) {
316  KALDI_ASSERT(old2new_map != NULL);
317 
325  std::map<std::pair<int32, std::vector<int32> >, int32 >
326  pair_to_physical;
327 
328  int32 N = ctx_dep.ContextWidth(),
329  P = ctx_dep.CentralPosition();
330  int32 num_syms_old = ilabel_info_old.size();
331 
334  std::vector<int32> old2old_map(num_syms_old);
335  old2old_map[0] = 0;
336  for (int32 i = 1; i < num_syms_old; i++) {
337  const std::vector<int32> &vec = ilabel_info_old[i];
338  if (vec.size() == 1 && vec[0] <= 0) { // disambig.
339  old2old_map[i] = i;
340  } else {
341  KALDI_ASSERT(vec.size() == static_cast<size_t>(N));
342  // work out the vector of context-dependent phone
343  int32 central_phone = vec[P];
344  int32 num_pdf_classes = trans_model.GetTopo().NumPdfClasses(central_phone);
345  std::vector<int32> state_seq(num_pdf_classes); // Indexed by pdf-class
346  for (int32 pdf_class = 0; pdf_class < num_pdf_classes; pdf_class++) {
347  if (!ctx_dep.Compute(vec, pdf_class, &(state_seq[pdf_class]))) {
348  std::ostringstream ss;
349  WriteIntegerVector(ss, false, vec);
350  KALDI_ERR << "tree did not succeed in converting phone window "<<ss.str();
351  }
352  }
353  std::pair<int32, std::vector<int32> > pr(central_phone, state_seq);
354  std::map<std::pair<int32, std::vector<int32> >, int32 >::iterator iter
355  = pair_to_physical.find(pr);
356  if (iter == pair_to_physical.end()) { // first time we saw something like this.
357  pair_to_physical[pr] = i;
358  old2old_map[i] = i;
359  } else { // seen it before. look up in the map, the index we point to.
360  old2old_map[i] = iter->second;
361  }
362  }
363  }
364 
365  std::vector<bool> seen(num_syms_old, false);
366  for (int32 i = 0; i < num_syms_old; i++)
367  seen[old2old_map[i]] = true;
368 
369  // Now work out the elements of old2new_map corresponding to
370  // things that are first in their equivalence class. We're just
371  // compacting the labels to those for which seen[i] == true.
372  int32 cur_id = 0;
373  old2new_map->resize(num_syms_old);
374  for (int32 i = 0; i < num_syms_old; i++)
375  if (seen[i])
376  (*old2new_map)[i] = cur_id++;
377  // fill in the other elements of old2new_map.
378  for (int32 i = 0; i < num_syms_old; i++)
379  (*old2new_map)[i] = (*old2new_map)[old2old_map[i]];
380 }

409 class TidToTstateMapper {
410 public:
411  // Function object used in MakePrecedingInputSymbolsSameClass and
412  // MakeFollowingInputSymbolsSameClass (as called by AddSelfLoopsBefore
413  // and AddSelfLoopsAfter). It maps transition-ids to transition-states
414  // (and -1 to -1, 0 to 0 and disambiguation symbols to 0). It also
415  // checks that there are no self-loops in the graph (i.e. in the labels
416  // it is called with). This is just a convenient place to put this check.
417 
418  // This maps valid transition-ids to transition states, maps kNoLabel to -1, and
419  // maps all other symbols (i.e. epsilon symbols and disambig symbols) to zero.
420  // Its point is to provide an equivalence class on labels that's relevant to what
421  // the self-loop will be on the following (or preceding) state.
422  TidToTstateMapper(const TransitionModel &trans_model,
423  const std::vector<int32> &disambig_syms):
424  trans_model_(trans_model),
425  disambig_syms_(disambig_syms) { }
426  typedef int32 Result;
427  int32 operator() (int32 label) const {
428  if (label == static_cast<int32>(fst::kNoLabel)) return -1; // -1 -> -1
429  else if (label >= 1 && label <= trans_model_.NumTransitionIds()) {
430  if (trans_model_.IsSelfLoop(label))
431  KALDI_ERR << "AddSelfLoops: graph already has self-loops.";
432  return trans_model_.TransitionIdToTransitionState(label);
433  } else { // 0 or (presumably) disambiguation symbol. Map to zero
434  if (label != 0)
435  KALDI_ASSERT(std::binary_search(disambig_syms_.begin(), disambig_syms_.end(), label)); // or invalid label
436  return 0;
437  }
438  }
439 
440 private:
441  const TransitionModel &trans_model_;
442  const std::vector<int32> &disambig_syms_; // sorted.
443 };

445 static void AddSelfLoopsBefore(const TransitionModel &trans_model,
446  const std::vector<int32> &disambig_syms,
447  BaseFloat self_loop_scale,
448  fst::VectorFst<fst::StdArc> *fst) {
449  using namespace fst;
450  typedef StdArc Arc;
451  typedef Arc::Label Label;
452  typedef Arc::StateId StateId;
453  typedef Arc::Weight Weight;
454 
455  TidToTstateMapper f(trans_model, disambig_syms);
456  // Duplicate states as necessary so that each state has at most one self-loop
457  // on it.
458  MakePrecedingInputSymbolsSameClass(true, fst, f);
459 
460  int32 kNoTransState = f(kNoLabel);
461  KALDI_ASSERT(kNoTransState == -1);
462 
463  // use the following to keep track of the transition-state for each state.
464  std::vector<int32> state_in(fst->NumStates(), kNoTransState);
465 
466  // This first loop just works out the label into each state,
467  // and converts the transitions in the graph from transition-states
468  // to transition-ids.
469 
470  for (StateIterator<VectorFst<Arc> > siter(*fst);
471  !siter.Done();
472  siter.Next()) {
473  StateId s = siter.Value();
474  for (MutableArcIterator<VectorFst<Arc> > aiter(fst, s);
475  !aiter.Done();
476  aiter.Next()) {
477  Arc arc = aiter.Value();
478  int32 trans_state = f(arc.ilabel);
479  if (state_in[arc.nextstate] == kNoTransState)
480  state_in[arc.nextstate] = trans_state;
481  else {
482  KALDI_ASSERT(state_in[arc.nextstate] == trans_state);
483  // or probably an error in MakePrecedingInputSymbolsSame.
484  }
485  }
486  }
487 
488  KALDI_ASSERT(state_in[fst->Start()] == kNoStateId || state_in[fst->Start()] == 0);
489  // or MakePrecedingInputSymbolsSame failed.
490 
491  // The next loop looks at each graph state, adds the self-loop [if needed] and
492  // multiples all the out-transitions' probs (and final-prob) by the
493  // forward-prob for that state (which is one minus self-loop-prob). We do it
494  // like this to maintain stochasticity (i.e. rather than multiplying the arcs
495  // with the corresponding labels on them by this probability).
496 
497  for (StateId s = 0; s < static_cast<StateId>(state_in.size()); s++) {
498  if (state_in[s] > 0) { // defined, and not eps or a disambiguation symbol...
499  int32 trans_state = static_cast<int32>(state_in[s]);
500  // First multiply all probabilities by "forward" probability.
501  BaseFloat log_prob = trans_model.GetNonSelfLoopLogProb(trans_state);
502  fst->SetFinal(s, Times(fst->Final(s), Weight(-log_prob*self_loop_scale)));
503  for (MutableArcIterator<MutableFst<Arc> > aiter(fst, s);
504  !aiter.Done();
505  aiter.Next()) {
506  Arc arc = aiter.Value();
507  arc.weight = Times(arc.weight, Weight(-log_prob*self_loop_scale));
508  aiter.SetValue(arc);
509  }
510  // Now add self-loop, if needed.
511  int32 trans_id = trans_model.SelfLoopOf(trans_state);
512  if (trans_id != 0) { // has self-loop.
513  BaseFloat log_prob = trans_model.GetTransitionLogProb(trans_id);
514  fst->AddArc(s, Arc(trans_id, 0, Weight(-log_prob*self_loop_scale), s));
515  }
516  }
517  }
518 }

525 static void AddSelfLoopsAfter(const TransitionModel &trans_model,
526  const std::vector<int32> &disambig_syms,
527  BaseFloat self_loop_scale,
528  fst::VectorFst<fst::StdArc> *fst) {
529  using namespace fst;
530  typedef StdArc Arc;
531  typedef Arc::Label Label;
532  typedef Arc::StateId StateId;
533  typedef Arc::Weight Weight;
534 
535  // Duplicate states as necessary so that each state has at most one self-loop
536  // on it.
537  TidToTstateMapper f(trans_model, disambig_syms);
538  MakeFollowingInputSymbolsSameClass(true, fst, f);
539 
540  StateId num_states = fst->NumStates();
541  for (StateId s = 0; s < num_states; s++) {
542  int32 my_trans_state = f(kNoLabel);
543  KALDI_ASSERT(my_trans_state == -1);
544  for (MutableArcIterator<VectorFst<Arc> > aiter(fst, s);
545  !aiter.Done();
546  aiter.Next()) {
547  Arc arc = aiter.Value();
548  if (my_trans_state == -1) my_trans_state = f(arc.ilabel);
549  else KALDI_ASSERT(my_trans_state == f(arc.ilabel)); // or MakeFollowingInputSymbolsSameClass failed.
550  if (my_trans_state > 0) { // transition-id; multiply weight...
551  BaseFloat log_prob = trans_model.GetNonSelfLoopLogProb(my_trans_state);
552  arc.weight = Times(arc.weight, Weight(-log_prob*self_loop_scale));
553  aiter.SetValue(arc);
554  }
555  }
556  if (fst->Final(s) != Weight::Zero()) {
557  KALDI_ASSERT(my_trans_state == kNoLabel || my_trans_state == 0); // or MakeFollowingInputSymbolsSameClass failed.
558  }

582 // IsReordered returns true if the transitions were possibly reordered. This reordering
583 // can happen in AddSelfLoops, if the "reorder" option was true.
584 // This makes the out-transition occur before the self-loop transition.
585 // The function returns false (no reordering) if there is not enough information in
586 // the alignment to tell (i.e. no self-loop were taken), and in this case the calling
587 // code doesn't care what the answer is.
588 // The "alignment" vector contains a sequence of TransitionIds.
589 
590 static bool IsReordered(const TransitionModel &trans_model,
591  const std::vector<int32> &alignment) {
592  for (size_t i = 0; i + 1 < alignment.size(); i++) {
593  int32 tstate1 = trans_model.TransitionIdToTransitionState(alignment[i]),
594  tstate2 = trans_model.TransitionIdToTransitionState(alignment[i+1]);
595  if (tstate1 != tstate2) {
596  bool is_loop_1 = trans_model.IsSelfLoop(alignment[i]),
597  is_loop_2 = trans_model.IsSelfLoop(alignment[i+1]);
598  KALDI_ASSERT(!(is_loop_1 && is_loop_2)); // Invalid.
599  if (is_loop_1) return true; // Reordered. self-loop is last.
600  if (is_loop_2) return false; // Not reordered. self-loop is first.
601  }
602  }
603 
604  // Just one trans-state in whole sequence.
605  if (alignment.empty()) return false;
606  else {
607  bool is_loop_front = trans_model.IsSelfLoop(alignment.front()),
608  is_loop_back = trans_model.IsSelfLoop(alignment.back());
609  if (is_loop_front) return false; // Not reordered. Self-loop is first.
610  if (is_loop_back) return true; // Reordered. Self-loop is last.
611  return false; // We really don't know in this case but calling code should
612  // not care.
613  }
614 }

616 // SplitToPhonesInternal takes as input the "alignment" vector containing
617 // a sequence of transition-ids, and appends a single vector to
618 // "split_output" for each instance of a phone that occurs in the
619 // output.
620 // Returns true if the alignment passes some non-exhaustive consistency
621 // checks (if the input does not start at the start of a phone or does not
622 // end at the end of a phone, we should expect that false will be returned).
623 
624 static bool SplitToPhonesInternal(const TransitionModel &trans_model,
625  const std::vector<int32> &alignment,
626  bool reordered,
627  std::vector<std::vector<int32> > *split_output) {
628  if (alignment.empty()) return true; // nothing to split.
629  std::vector<size_t> end_points; // points at which phones end [in an
630  // stl iterator sense, i.e. actually one past the last transition-id within
631  // each phone]..
632 
633  bool was_ok = true;
634  for (size_t i = 0; i < alignment.size(); i++) {
635  int32 trans_id = alignment[i];
636  if (trans_model.IsFinal(trans_id)) { // is final-prob
637  if (!reordered) end_points.push_back(i+1);
638  else { // reordered.
639  while (i+1 < alignment.size() &&
640  trans_model.IsSelfLoop(alignment[i+1])) {
641  KALDI_ASSERT(trans_model.TransitionIdToTransitionState(alignment[i]) ==
642  trans_model.TransitionIdToTransitionState(alignment[i+1]));
643  i++;
644  }
645  end_points.push_back(i+1);
646  }
647  } else if (i+1 == alignment.size()) {
648  // need to have an end-point at the actual end.
649  // but this is an error- should have been detected already.
650  was_ok = false;
651  end_points.push_back(i+1);
652  } else {
653  int32 this_state = trans_model.TransitionIdToTransitionState(alignment[i]),
654  next_state = trans_model.TransitionIdToTransitionState(alignment[i+1]);
655  if (this_state == next_state) continue; // optimization.
656  int32 this_phone = trans_model.TransitionStateToPhone(this_state),
657  next_phone = trans_model.TransitionStateToPhone(next_state);
658  if (this_phone != next_phone) {
659  // The phone changed, but this is an error-- we should have detected this via the
660  // IsFinal check.
661  was_ok = false;
662  end_points.push_back(i+1);
663  }
664  }
665  }
666 
667  size_t cur_point = 0;
668  for (size_t i = 0; i < end_points.size(); i++) {
669  split_output->push_back(std::vector<int32>());
670  // The next if-statement checks if the initial trans-id at the current end
671  // point is the initial-state of the current phone if that initial-state
672  // is emitting (a cursory check that the alignment is plausible).
673  int32 trans_state =
674  trans_model.TransitionIdToTransitionState(alignment[cur_point]);
675  int32 phone = trans_model.TransitionStateToPhone(trans_state);
676  int32 pdf_class = trans_model.GetTopo().TopologyForPhone(phone)[0].pdf_class;
677  if (pdf_class != kNoPdf) // initial-state of the current phone is emitting
678  if (trans_model.TransitionStateToHmmState(trans_state) != 0)
679  was_ok = false;
680  for (size_t j = cur_point; j < end_points[i]; j++)
681  split_output->back().push_back(alignment[j]);
682  cur_point = end_points[i];
683  }
684  return was_ok;
685 }

790 static bool ComputeNewPhoneLengths(const HmmTopology &topology,
791  const std::vector<int32> &mapped_phones,
792  const std::vector<int32> &old_lengths,
793  int32 subsample_factor,
794  std::vector<int32> *new_lengths) {
795  int32 phone_sequence_length = old_lengths.size();
796  std::vector<int32> min_lengths(phone_sequence_length);
797  new_lengths->resize(phone_sequence_length);
798  for (int32 i = 0; i < phone_sequence_length; i++)
799  min_lengths[i] = topology.MinLength(mapped_phones[i]);
800  int32 cur_time_elapsed = 0;
801  for (int32 i = 0; i < phone_sequence_length; i++) {
802  // Note: the '+ subsample_factor - 1' here is needed so that
803  // the subsampled alignments have the same length as features
804  // subsampled with 'subsample-feats'.
805  int32 subsampled_time =
806  (cur_time_elapsed + subsample_factor - 1) / subsample_factor;
807  cur_time_elapsed += old_lengths[i];
808  int32 next_subsampled_time =
809  (cur_time_elapsed + subsample_factor - 1) / subsample_factor;
810  (*new_lengths)[i] = next_subsampled_time - subsampled_time;
811  }
812  bool changed = true;
813  while (changed) {
814  changed = false;
815  for (int32 i = 0; i < phone_sequence_length; i++) {
816  if ((*new_lengths)[i] < min_lengths[i]) {
817  changed = true;
818  // we need at least one extra frame.. just try to get one frame for now.
819  // Get it from the left or the right, depending which one has the closest
820  // availability of a 'spare' frame.
821  int32 min_distance = std::numeric_limits<int32>::max(),
822  best_other_phone_index = -1,
823  cur_distance = 0;
824  // first try to the left.
825  for (int32 j = i - 1; j >= 0; j--) {
826  if ((*new_lengths)[j] > min_lengths[j]) {
827  min_distance = cur_distance;
828  best_other_phone_index = j;
829  break;
830  } else {
831  cur_distance += (*new_lengths)[j];
832  }
833  }
834  // .. now to the right.
835  cur_distance = 0;
836  for (int32 j = i + 1; j < phone_sequence_length; j++) {
837  if ((*new_lengths)[j] > min_lengths[j]) {
838  if (cur_distance < min_distance) {
839  min_distance = cur_distance;
840  best_other_phone_index = j;
841  }
842  break;
843  } else {
844  cur_distance += (*new_lengths)[j];
845  }
846  }
847  if (best_other_phone_index == -1)
848  return false;
849  // assign an extra frame to this phone...
850  (*new_lengths)[i]++;
851  // and borrow it from the place that we found.
852  (*new_lengths)[best_other_phone_index]--;
853  }
854  }
855  }
856  return true;
857 }

859 bool ConvertAlignment(const TransitionModel &old_trans_model,
860  const TransitionModel &new_trans_model,
861  const ContextDependencyInterface &new_ctx_dep,
862  const std::vector<int32> &old_alignment,
863  int32 subsample_factor,
864  bool new_is_reordered,
865  const std::vector<int32> *phone_map,
866  std::vector<int32> *new_alignment) {
867  bool old_is_reordered = IsReordered(old_trans_model, old_alignment);
868  KALDI_ASSERT(new_alignment != NULL);
869  new_alignment->clear();
870  new_alignment->reserve(old_alignment.size());
871  std::vector<std::vector<int32> > old_split; // split into phones.
872  if (!SplitToPhones(old_trans_model, old_alignment, &old_split))
873  return false;
874  int32 phone_sequence_length = old_split.size();
875  std::vector<int32> mapped_phones(phone_sequence_length);
876  for (size_t i = 0; i < phone_sequence_length; i++) {
877  KALDI_ASSERT(!old_split[i].empty());
878  mapped_phones[i] = old_trans_model.TransitionIdToPhone(old_split[i][0]);
879  if (phone_map != NULL) { // Map the phone sequence.
880  int32 sz = phone_map->size();
881  if (mapped_phones[i] < 0 || mapped_phones[i] >= sz ||
882  (*phone_map)[mapped_phones[i]] == -1)
883  KALDI_ERR << "ConvertAlignment: could not map phone " << mapped_phones[i];
884  mapped_phones[i] = (*phone_map)[mapped_phones[i]];
885  }
886  }
887 
888  // the sizes of each element of 'new_split' indicate the length of alignment
889  // that we want for each phone in the new sequence.
890  std::vector<std::vector<int32> > new_split(phone_sequence_length);
891  if (subsample_factor == 1 &&
892  old_trans_model.GetTopo() == new_trans_model.GetTopo()) {
893  // we know the old phone lengths will be fine.
894  for (size_t i = 0; i < phone_sequence_length; i++)
895  new_split[i].resize(old_split[i].size());
896  } else {
897  // .. they may not be fine.
898  std::vector<int32> old_lengths(phone_sequence_length), new_lengths;
899  for (int32 i = 0; i < phone_sequence_length; i++)
900  old_lengths[i] = old_split[i].size();
901  if (!ComputeNewPhoneLengths(new_trans_model.GetTopo(),
902  mapped_phones, old_lengths,
903  subsample_factor, &new_lengths)) {
904  KALDI_WARN << "Failed to produce suitable phone lengths";
905  return false;
906  }
907  for (int32 i = 0; i < phone_sequence_length; i++)
908  new_split[i].resize(new_lengths[i]);
909  }
910 
911  int32 N = new_ctx_dep.ContextWidth(),
912  P = new_ctx_dep.CentralPosition();
913 
914  // by starting at -N and going to phone_sequence_length + N, we're
915  // being generous and not bothering to work out the exact
916  // array bounds.
917  for (int32 win_start = -N;
918  win_start < static_cast<int32>(phone_sequence_length + N);
919  win_start++) { // start of a context window.
920  int32 central_pos = win_start + P;
921  if (static_cast<size_t>(central_pos) < phone_sequence_length) {
922  // i.e. if (central_pos >= 0 && central_pos < phone_sequence_length)
923  std::vector<int32> new_phone_window(N, 0);
924  for (int32 offset = 0; offset < N; offset++)
925  if (static_cast<size_t>(win_start+offset) < phone_sequence_length)
926  new_phone_window[offset] = mapped_phones[win_start+offset];
927  const std::vector<int32> &old_alignment_for_phone = old_split[central_pos];
928  std::vector<int32> &new_alignment_for_phone = new_split[central_pos];
929 
930  ConvertAlignmentForPhone(old_trans_model, new_trans_model, new_ctx_dep,
931  old_alignment_for_phone, new_phone_window,
932  old_is_reordered, new_is_reordered,
933  &new_alignment_for_phone);
934  new_alignment->insert(new_alignment->end(),
935  new_alignment_for_phone.begin(),
936  new_alignment_for_phone.end());
937  }
938  }
939  KALDI_ASSERT(new_alignment->size() ==
940  (old_alignment.size() + subsample_factor - 1)/subsample_factor);
941  return true;
942 }

944 // Returns the scaled, but not negated, log-prob, with the given scaling factors.
945 static BaseFloat GetScaledTransitionLogProb(const TransitionModel &trans_model,

947  BaseFloat transition_scale,
948  BaseFloat self_loop_scale) {

950  return trans_model.GetTransitionLogProb(trans_id) * transition_scale;
951  } else {
952  if (trans_model.IsSelfLoop(trans_id)) {
953  return self_loop_scale * trans_model.GetTransitionLogProb(trans_id);

957  + transition_scale * trans_model.GetTransitionLogProbIgnoringSelfLoops(trans_id);

959  // (self_loop_scale - transition_scale) * trans_model.GetNonSelfLoopLogProb(trans_state)

961  // this simplifies if self_loop_scale == 0.0
962  }

1032 // This function takes a phone-sequence with word-start and word-end
1033 // tokens in it, and a word-sequence, and outputs the pronunciations
1034 // "prons"... the format of "prons" is, each element is a vector,
1035 // where the first element is the word (or zero meaning no word, e.g.
1036 // for optional silence introduced by the lexicon), and the remaining
1037 // elements are the phones in the word's pronunciation.
1038 // It returns false if it encounters a problem of some kind, e.g.
1039 // if the phone-sequence doesn't seem to have the right number of
1040 // words in it.
1041 bool ConvertPhnxToProns(const std::vector<int32> &phnx,
1042  const std::vector<int32> &words,
1043  int32 word_start_sym,
1044  int32 word_end_sym,
1045  std::vector<std::vector<int32> > *prons) {
1046  size_t i = 0, j = 0;
1047 
1048  while (i < phnx.size()) {
1049  if (phnx[i] == 0) return false; // zeros not valid here.
1050  if (phnx[i] == word_start_sym) { // start a word...
1051  std::vector<int32> pron;
1052  if (j >= words.size()) return false; // no word left..
1053  if (words[j] == 0) return false; // zero word disallowed.
1054  pron.push_back(words[j++]);
1055  i++;
1056  while (i < phnx.size()) {
1057  if (phnx[i] == 0) return false;
1058  if (phnx[i] == word_start_sym) return false; // error.
1059  if (phnx[i] == word_end_sym) { i++; break; }
1060  pron.push_back(phnx[i]);
1061  i++;
1062  }
1063  // check we did see the word-end symbol.
1064  if (!(i > 0 && phnx[i-1] == word_end_sym))
1065  return false;
1066  prons->push_back(pron);
1067  } else if (phnx[i] == word_end_sym) {
1068  return false; // error.
1069  } else {
1070  // start a non-word sequence of phones (e.g. opt-sil).
1071  std::vector<int32> pron;
1072  pron.push_back(0); // 0 serves as the word-id.
1073  while (i < phnx.size()) {
1074  if (phnx[i] == 0) return false;
1075  if (phnx[i] == word_start_sym) break;
1076  if (phnx[i] == word_end_sym) return false; // error.
1077  pron.push_back(phnx[i]);
1078  i++;
1079  }
1080  prons->push_back(pron);
1081  }
1082  }
1083  return (j == words.size());
1084 }
