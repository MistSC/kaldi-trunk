29 void GenRandStats(int32 dim, int32 num_stats, int32 N, int32 P,
30  const std::vector<int32> &phone_ids,
31  const std::vector<int32> &phone2hmm_length,
32  const std::vector<bool> &is_ctx_dep,
33  bool ensure_all_phones_covered,
34  BuildTreeStatsType *stats_out) {
35 
36  KALDI_ASSERT(dim > 0);
37  KALDI_ASSERT(num_stats > 0);
38  KALDI_ASSERT(N > 0);
39  KALDI_ASSERT(P < N);
40  KALDI_ASSERT(phone_ids.size() != 0);
41  KALDI_ASSERT(stats_out != NULL && stats_out->empty());
42  int32 max_phone = *std::max_element(phone_ids.begin(), phone_ids.end());
43  KALDI_ASSERT(phone2hmm_length.size() >= static_cast<size_t>(1 + max_phone));
44  KALDI_ASSERT(is_ctx_dep.size() >= static_cast<size_t>(1 + max_phone));
45 
46  // Make sure phone id's distinct.
47  {
48  std::vector<int32> tmp(phone_ids);
49  SortAndUniq(&tmp);
50  KALDI_ASSERT(tmp.size() == phone_ids.size());
51  }
52  size_t num_phones = phone_ids.size();
53 
54  // Decide on an underlying "mean" for phones...
55  Matrix<BaseFloat> phone_vecs(max_phone+1, dim);
56  for (int32 i = 0;i < max_phone+1;i++)
57  for (int32 j = 0;j < dim;j++) phone_vecs(i, j) = RandGauss() * (2.0 / (j+1));
58 
59 
60  std::map<EventType, Clusterable*> stats_tmp;
61 
62  std::vector<bool> covered(1 + max_phone, false);
63 
64  bool all_covered = false;
65  for (int32 i = 0;i < num_stats || (ensure_all_phones_covered && !all_covered);i++) {
66  // decide randomly on a phone-in-context.
67  std::vector<int32> phone_vec(N);
68  for (size_t i = 0;i < (size_t)N;i++) phone_vec[i] = phone_ids[(Rand() % num_phones)];
69 
70  int32 hmm_length = phone2hmm_length[phone_vec[P]];
71  KALDI_ASSERT(hmm_length > 0);
72  covered[phone_vec[P]] = true;
73 
74  // For each position [in the central phone]...
75  for (int32 j = 0; j < hmm_length; j++) {
76  // create event vector.
77  EventType event_vec;
78  event_vec.push_back(std::make_pair(kPdfClass, (EventValueType)j)); // record the position.
79  for (size_t pos = 0; pos < (size_t)N; pos++) {
80  if (pos == (size_t)(P) || is_ctx_dep[phone_vec[P]])
81  event_vec.push_back(std::make_pair((EventKeyType)pos, (EventValueType)phone_vec[pos]));
82  // The if-statement above ensures we do not record the context of "context-free"
83  // phone (e.g., silence).
84  }
85 
86  Vector<BaseFloat> mean(dim); // mean of Gaussian.
87  GaussClusterable *this_stats = new GaussClusterable(dim, 0.1); // 0.1 is var floor.
88  { // compute stats; this block attempts to simulate the process of "real" data
89  // collection and does not correspond to any code you would write in a real
90  // scenario.
91  Vector<BaseFloat> weights(N); // weight of each component.
92  for (int32 k = 0; k < N; k++) {
93  BaseFloat k_pos = (N - 0.5 - k) / N; // between 0 and 1, less for lower k...
94  BaseFloat j_pos = (hmm_length - 0.5 - j) / hmm_length;
95  // j_pos is between 0 and 1, less for lower j.
96 
97  BaseFloat weight = j_pos*k_pos + (1.0-j_pos)*(1.0-k_pos);
98  // if j_pos close to zero, gives larger weight to k_pos close
99  // to zero.
100  if (k == P) weight += 1.0;
101  weights(k) = weight;
102  }
103  KALDI_ASSERT(weights.Sum() != 0);
104  weights.Scale(1.0 / weights.Sum());
105  for (int32 k = 0; k < N; k++)
106  mean.AddVec(weights(k), phone_vecs.Row(phone_vec[k]));
107  BaseFloat count;
108  if (Rand() % 2 == 0) count = 1000.0 * RandUniform();
109  else count = 100.0 * RandUniform();
110 
111  int32 num_samples = 10;
112  for (size_t p = 0;p < (size_t)num_samples; p++) {
113  Vector<BaseFloat> sample(mean); // copy mean.
114  for (size_t d = 0; d < (size_t)dim; d++) sample(d) += RandGauss(); // unit var.
115  this_stats->AddStats(sample, count / num_samples);
116  }
117  }
118 
119  if (stats_tmp.count(event_vec) != 0) {
120  stats_tmp[event_vec]->Add(*this_stats);
121  delete this_stats;
122  } else {
123  stats_tmp[event_vec] = this_stats;
124  }
125  }
126  all_covered = true;
127  for (size_t i = 0; i< num_phones; i++) if (!covered[phone_ids[i]]) all_covered = false;
128  }
129  CopyMapToVector(stats_tmp, stats_out);
130  KALDI_ASSERT(stats_out->size() > 0);
131 }

239 // This function, called from BuildTreeTwoLevel, computes the mapping from the
240 // leaves of the big tree to the leaves of the small tree. It does this by
241 // working out which stats correspond to which leaf of the big tree, then
242 // mapping those stats to the leaves of the small tree and seeing where they go.
243 // It only works of all leaves of the big tree have stats-- but they should have
244 // stats, or there would have been an error in tree building.
245 
246 static void ComputeTreeMapping(const EventMap &small_tree,
247  const EventMap &big_tree,
248  const BuildTreeStatsType &stats,
249  std::vector<int32> *leaf_map) {
250  std::vector<BuildTreeStatsType> split_stats_small; // stats split by small tree
251  int32 num_leaves_big = big_tree.MaxResult() + 1,
252  num_leaves_small = small_tree.MaxResult() + 1;
253  SplitStatsByMap(stats, small_tree, &split_stats_small);
254  KALDI_ASSERT(static_cast<int32>(split_stats_small.size()) <=
255  num_leaves_small);
256  leaf_map->clear();
257  leaf_map->resize(num_leaves_big, -1); // fill with -1.
258 
259  std::vector<int32> small_leaves_unseen; // a list of small leaves that had no stats..
260  // this is used as a workaround for when there are no stats at leaves...
261  // it's really an error condition and it will cause errors later (e.g. when
262  // you initialize your model), but at this point we will try to handle it
263  // gracefully.
264 
265  for (int32 i = 0; i < num_leaves_small; i++) {
266  if (static_cast<size_t>(i) >= split_stats_small.size() ||
267  split_stats_small[i].empty()) {
268  KALDI_WARN << "No stats mapping to " << i << " in small tree. "
269  << "Continuing but this is a serious error.";
270  small_leaves_unseen.push_back(i);
271  } else {
272  for (size_t j = 0; j < split_stats_small[i].size(); j++) {
273  int32 leaf = 0; // = 0 to keep compiler happy. Leaf in big tree.
274  bool ok = big_tree.Map(split_stats_small[i][j].first, &leaf);
275  if (!ok)
276  KALDI_ERR << "Could not map stats with big tree: probable code error.";
277  if (leaf < 0 || leaf >= num_leaves_big)
278  KALDI_ERR << "Leaf out of range: " << leaf << " vs. " << num_leaves_big;
279  if ((*leaf_map)[leaf] != -1 && (*leaf_map)[leaf] != i)
280  KALDI_ERR << "Inconsistent mapping for big tree: "
281  << i << " vs. " << (*leaf_map)[leaf];
282  (*leaf_map)[leaf] = i;
283  }
284  }
285  }
286  // Now make sure that all leaves in the big tree have a leaf in the small tree
287  // assigned to them. If not we try to clean up... this should never normally
288  // happen and if it does it's due to trying to assign tree roots to unseen phones,
289  // which will anyway cause an error in a later stage of system building.
290  for (int32 leaf = 0; leaf < num_leaves_big; leaf++) {
291  int32 small_leaf = (*leaf_map)[leaf];
292  if (small_leaf == -1) {
293  KALDI_WARN << "In ComputeTreeMapping, could not get mapping from leaf "
294  << leaf;
295  if (!small_leaves_unseen.empty()) {
296  small_leaf = small_leaves_unseen.back();
297  KALDI_WARN << "Assigning it to unseen small-tree leaf " << small_leaf;
298  small_leaves_unseen.pop_back();
299  (*leaf_map)[leaf] = small_leaf;
300  } else {
301  KALDI_WARN << "Could not find any unseen small-tree leaf to assign "
302  << "it to. Making it zero, but this is bad. ";
303  (*leaf_map)[leaf] = 0;
304  }
305  } else if (small_leaf < 0 || small_leaf >= num_leaves_small)
306  KALDI_ERR << "Leaf in leaf mapping out of range: for big-map leaf "
307  << leaf << ", mapped to " << small_leaf << ", vs. "
308  << num_leaves_small;
309  }
310 }

314  const std::vector<std::vector<int32> > &phone_sets,

326  EventMap *first_level_tree = BuildTree(qopts, phone_sets,

329  max_leaves_first, 0.0, P);

332

335  for (size_t i = 0; i < phone_sets.size(); i++)

337  nonsplit_phones.insert(nonsplit_phones.end(), phone_sets[i].begin(), phone_sets[i].end());
338  std::sort(nonsplit_phones.begin(), nonsplit_phones.end());

342  FilterStatsByKey(stats, P, nonsplit_phones, false, // retain only those not
343  // in "nonsplit_phones"
344  &filtered_stats);
345

348 
349  BaseFloat smallest_split = 0.0;
350 
351  BaseFloat impr;
352  EventMap *tree = SplitDecisionTree(*first_level_tree,
353  filtered_stats,
354  qopts, 0.0, max_leaves_second,
355  &num_leaves, &impr, &smallest_split);
356 
357  KALDI_LOG << "Building second-level tree: increased #leaves from "
358  << old_num_leaves << " to " << num_leaves << ", smallest split was "
359  << smallest_split;
360 
361  BaseFloat normalizer = SumNormalizer(stats),
362  impr_normalized = impr / normalizer;
363 
364  KALDI_LOG << "After second decision tree split, num-leaves = "
365  << num_leaves << ", like-impr = " << impr_normalized
366  << " per frame over " << normalizer << " frames.";
367 
368  if (cluster_leaves) { // Cluster the leaves of the tree.
369  KALDI_LOG << "Clustering leaves of larger tree.";
370  BaseFloat objf_before_cluster = ObjfGivenMap(stats, *tree);
371 
372  // Now do the clustering.
373  int32 num_removed = 0;
374  EventMap *tree_clustered = ClusterEventMapRestrictedByMap(*tree,
375  stats,
376  smallest_split,
377  *first_level_tree,
378  &num_removed);
379  KALDI_LOG << "BuildTreeTwoLevel: removed " << num_removed << " leaves.";
380 
381  int32 num_leaves = 0;
382  EventMap *tree_renumbered = RenumberEventMap(*tree_clustered, &num_leaves);
383 
384  BaseFloat objf_after_cluster = ObjfGivenMap(stats, *tree_renumbered);
385 
386  KALDI_LOG << "Objf change due to clustering "
387  << ((objf_after_cluster-objf_before_cluster) / SumNormalizer(stats))
388  << " per frame.";
389  KALDI_LOG << "Num-leaves now "<< num_leaves;
390  delete tree;
391  delete tree_clustered;
392  tree = tree_renumbered;
393  }
394 
395  ComputeTreeMapping(*first_level_tree,
396  *tree,
397  stats,
398  leaf_map);
399 
400  { // Next do another renumbering of "tree" so that leaves with the
401  // same value in "first_level_tree" are contiguous.
402  std::vector<std::pair<int32, int32> > leaf_pairs;
403  for (size_t i = 0; i < leaf_map->size(); i++)
404  leaf_pairs.push_back(std::make_pair((*leaf_map)[i], static_cast<int32>(i)));
405  // pair of (small-tree-number, big-tree-number).
406  std::sort(leaf_pairs.begin(), leaf_pairs.end());
407  std::vector<int32> old2new_map(leaf_map->size()),
408  new_leaf_map(leaf_map->size());
409  // Note: old2new_map maps from old indices to new indices, in the
410  // renumbering; new_leaf_map maps from 2nd-level tree indices to
411  // 1st-level tree indices.
412  for (size_t i = 0; i < leaf_pairs.size(); i++) {
413  int32 old_number = leaf_pairs[i].second, new_number = i;
414  old2new_map[old_number] = new_number;
415  new_leaf_map[new_number] = (*leaf_map)[old_number];
416  }
417  *leaf_map = new_leaf_map;
418  EventMap *renumbered_tree = MapEventMapLeaves(*tree, old2new_map);
419  delete tree;
420  tree = renumbered_tree;
421  }
422 
423  delete first_level_tree;
424  return tree;
425 }

483 static void ObtainSetsOfPhones(const std::vector<std::vector<int32> > &phone_sets, // the original phone sets, may
484  // just be individual phones.
485  const std::vector<int32> &assignments, // phone-sets->clusters
486  const std::vector<int32> &clust_assignments, // clust->parent
487  int32 num_leaves, // number of clusters present..
488  std::vector<std::vector<int32> > *sets_out) {
489  KALDI_ASSERT(sets_out != NULL);
490  sets_out->clear();
491  std::vector<std::vector<int32> > raw_sets(clust_assignments.size());
492 
493  KALDI_ASSERT(num_leaves < static_cast<int32>(clust_assignments.size()));
494  KALDI_ASSERT(assignments.size() == phone_sets.size());
495  for (size_t i = 0; i < assignments.size(); i++) {
496  int32 clust = assignments[i]; // this is an index into phone_sets.
497  KALDI_ASSERT(clust>=0 && clust < num_leaves);
498  for (size_t j = 0; j < phone_sets[i].size(); j++) {
499  // and not just a hole.
500  raw_sets[clust].push_back(phone_sets[i][j]);
501  }
502  }
503  // for all clusters including the top-level cluster:
504  // [note that the top-level cluster contains all phones, but it may actually
505  // be useful because sometimes we cluster just the non-silence phones, so
506  // the list of all phones is a way of asking about silence in such a way
507  // that epsilon (end-or-begin-of-utterance) gets lumped with silence.
508  for (int32 j = 0; j < static_cast<int32>(clust_assignments.size()); j++) {
509  int32 parent = clust_assignments[j];
510  std::sort(raw_sets[j].begin(), raw_sets[j].end());
511  KALDI_ASSERT(IsSortedAndUniq(raw_sets[j])); // should be no dups.
512  if (parent < static_cast<int32>(clust_assignments.size())-1) { // parent is not out of range [i.e. not the top one]...
513  // add all j's phones to its parent.
514  raw_sets[parent].insert(raw_sets[parent].end(),
515  raw_sets[j].begin(),
516  raw_sets[j].end());
517  }
518  }
519  // Reverse the 'raw_sets' so the most important things (top-level questions)
520  // appear at the front... this will end up mattering because of the
521  // --truncate-leftmost-questions option to compile-questions.
522  std::reverse(raw_sets.begin(), raw_sets.end());
523 
524  // Now add the original sets-of-phones to the raw sets, to make sure all of
525  // these are present. (The main reason they might be absent is if the stats
526  // are empty, but we want to ensure they are all there regardless). note these
527  // will be actual singleton sets if the sets-of-phones each contain just one
528  // phone, which in some sense is the normal situation.
529  for (size_t i = 0; i < phone_sets.size(); i++) {
530  raw_sets.push_back(phone_sets[i]);
531  }
532  // Remove duplicate sets from "raw_sets".
533  RemoveDuplicates(&raw_sets);
534  sets_out->reserve(raw_sets.size());
