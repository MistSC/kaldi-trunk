36 int32 LatticeStateTimes(const Lattice &lat, vector<int32> *times) {
37  if (!lat.Properties(fst::kTopSorted, true))
38  KALDI_ERR << "Input lattice must be topologically sorted.";
39  KALDI_ASSERT(lat.Start() == 0);
40  int32 num_states = lat.NumStates();
41  times->clear();
42  times->resize(num_states, -1);
43  (*times)[0] = 0;
44  for (int32 state = 0; state < num_states; state++) {
45  int32 cur_time = (*times)[state];
46  for (fst::ArcIterator<Lattice> aiter(lat, state); !aiter.Done();
47  aiter.Next()) {
48  const LatticeArc &arc = aiter.Value();
49 
50  if (arc.ilabel != 0) { // Non-epsilon input label on arc
51  // next time instance
52  if ((*times)[arc.nextstate] == -1) {
53  (*times)[arc.nextstate] = cur_time + 1;
54  } else {
55  KALDI_ASSERT((*times)[arc.nextstate] == cur_time + 1);
56  }
57  } else { // epsilon input label on arc
58  // Same time instance
59  if ((*times)[arc.nextstate] == -1)
60  (*times)[arc.nextstate] = cur_time;
61  else
62  KALDI_ASSERT((*times)[arc.nextstate] == cur_time);
63  }
64  }
65  }
66  return (*std::max_element(times->begin(), times->end()));
67 }

179  }

181  }

186 template<class LatType> // could be Lattice or CompactLattice
187 bool PruneLattice(BaseFloat beam, LatType *lat) {
188  typedef typename LatType::Arc Arc;
189  typedef typename Arc::Weight Weight;
190  typedef typename Arc::StateId StateId;
191 
192  KALDI_ASSERT(beam > 0.0);
193  if (!lat->Properties(fst::kTopSorted, true)) {
194  if (fst::TopSort(lat) == false) {
195  KALDI_WARN << "Cycles detected in lattice";
196  return false;
197  }
198  }
199  // We assume states before "start" are not reachable, since
200  // the lattice is topologically sorted.
201  int32 start = lat->Start();
202  int32 num_states = lat->NumStates();
203  if (num_states == 0) return false;
204  std::vector<double> forward_cost(num_states,
205  std::numeric_limits<double>::infinity()); // viterbi forward.
206  forward_cost[start] = 0.0; // lattice can't have cycles so couldn't be
207  // less than this.
208  double best_final_cost = std::numeric_limits<double>::infinity();
209  // Update the forward probs.
210  // Thanks to Jing Zheng for finding a bug here.
211  for (int32 state = 0; state < num_states; state++) {
212  double this_forward_cost = forward_cost[state];
213  for (fst::ArcIterator<LatType> aiter(*lat, state);
214  !aiter.Done();
215  aiter.Next()) {
216  const Arc &arc(aiter.Value());
217  StateId nextstate = arc.nextstate;
218  KALDI_ASSERT(nextstate > state && nextstate < num_states);
219  double next_forward_cost = this_forward_cost +
220  ConvertToCost(arc.weight);
221  if (forward_cost[nextstate] > next_forward_cost)
222  forward_cost[nextstate] = next_forward_cost;
223  }
224  Weight final_weight = lat->Final(state);
225  double this_final_cost = this_forward_cost +
226  ConvertToCost(final_weight);
227  if (this_final_cost < best_final_cost)
228  best_final_cost = this_final_cost;
229  }
230  int32 bad_state = lat->AddState(); // this state is not final.
231  double cutoff = best_final_cost + beam;
232 
233  // Go backwards updating the backward probs (which share memory with the
234  // forward probs), and pruning arcs and deleting final-probs. We prune arcs
235  // by making them point to the non-final state "bad_state". We'll then use
236  // Trim() to remove unnecessary arcs and states. [this is just easier than
237  // doing it ourselves.]
238  std::vector<double> &backward_cost(forward_cost);
239  for (int32 state = num_states - 1; state >= 0; state--) {
240  double this_forward_cost = forward_cost[state];
241  double this_backward_cost = ConvertToCost(lat->Final(state));
242  if (this_backward_cost + this_forward_cost > cutoff
243  && this_backward_cost != std::numeric_limits<double>::infinity())
244  lat->SetFinal(state, Weight::Zero());
245  for (fst::MutableArcIterator<LatType> aiter(lat, state);
246  !aiter.Done();
247  aiter.Next()) {
248  Arc arc(aiter.Value());
249  StateId nextstate = arc.nextstate;
250  KALDI_ASSERT(nextstate > state && nextstate < num_states);
251  double arc_cost = ConvertToCost(arc.weight),
252  arc_backward_cost = arc_cost + backward_cost[nextstate],
253  this_fb_cost = this_forward_cost + arc_backward_cost;
254  if (arc_backward_cost < this_backward_cost)
255  this_backward_cost = arc_backward_cost;
256  if (this_fb_cost > cutoff) { // Prune the arc.
257  arc.nextstate = bad_state;
258  aiter.SetValue(arc);
259  }
260  }
261  backward_cost[state] = this_backward_cost;
262  }
263  fst::Connect(lat);
264  return (lat->NumStates() > 0);
265 }

272 BaseFloat LatticeForwardBackward(const Lattice &lat, Posterior *post,
273  double *acoustic_like_sum) {
274  // Note, Posterior is defined as follows: Indexed [frame], then a list
275  // of (transition-id, posterior-probability) pairs.
276  // typedef std::vector<std::vector<std::pair<int32, BaseFloat> > > Posterior;
277  using namespace fst;
278  typedef Lattice::Arc Arc;
279  typedef Arc::Weight Weight;
280  typedef Arc::StateId StateId;
281 
282  if (acoustic_like_sum) *acoustic_like_sum = 0.0;
283 
284  // Make sure the lattice is topologically sorted.
285  if (lat.Properties(fst::kTopSorted, true) == 0)
286  KALDI_ERR << "Input lattice must be topologically sorted.";
287  KALDI_ASSERT(lat.Start() == 0);
288 
289  int32 num_states = lat.NumStates();
290  vector<int32> state_times;
291  int32 max_time = LatticeStateTimes(lat, &state_times);
292  std::vector<double> alpha(num_states, kLogZeroDouble);
293  std::vector<double> &beta(alpha); // we re-use the same memory for
294  // this, but it's semantically distinct so we name it differently.
295  double tot_forward_prob = kLogZeroDouble;
296 
297  post->clear();
298  post->resize(max_time);
299 
300  alpha[0] = 0.0;
301  // Propagate alphas forward.
302  for (StateId s = 0; s < num_states; s++) {
303  double this_alpha = alpha[s];
304  for (ArcIterator<Lattice> aiter(lat, s); !aiter.Done(); aiter.Next()) {
305  const Arc &arc = aiter.Value();
306  double arc_like = -ConvertToCost(arc.weight);
307  alpha[arc.nextstate] = LogAdd(alpha[arc.nextstate], this_alpha + arc_like);
308  }
309  Weight f = lat.Final(s);
310  if (f != Weight::Zero()) {
311  double final_like = this_alpha - (f.Value1() + f.Value2());
312  tot_forward_prob = LogAdd(tot_forward_prob, final_like);
313  KALDI_ASSERT(state_times[s] == max_time &&
314  "Lattice is inconsistent (final-prob not at max_time)");
315  }
316  }
317  for (StateId s = num_states-1; s >= 0; s--) {
318  Weight f = lat.Final(s);
319  double this_beta = -(f.Value1() + f.Value2());
320  for (ArcIterator<Lattice> aiter(lat, s); !aiter.Done(); aiter.Next()) {
321  const Arc &arc = aiter.Value();
322  double arc_like = -ConvertToCost(arc.weight),
323  arc_beta = beta[arc.nextstate] + arc_like;
324  this_beta = LogAdd(this_beta, arc_beta);
325  int32 transition_id = arc.ilabel;
326 
327  // The following "if" is an optimization to avoid un-needed exp().
328  if (transition_id != 0 || acoustic_like_sum != NULL) {
329  double posterior = Exp(alpha[s] + arc_beta - tot_forward_prob);
330 
331  if (transition_id != 0) // Arc has a transition-id on it [not epsilon]
332  (*post)[state_times[s]].push_back(std::make_pair(transition_id,
333  static_cast<kaldi::BaseFloat>(posterior)));
334  if (acoustic_like_sum != NULL)
335  *acoustic_like_sum -= posterior * arc.weight.Value2();
336  }
337  }
338  if (acoustic_like_sum != NULL && f != Weight::Zero()) {
339  double final_logprob = - ConvertToCost(f),
340  posterior = Exp(alpha[s] + final_logprob - tot_forward_prob);
341  *acoustic_like_sum -= posterior * f.Value2();
342  }
343  beta[s] = this_beta;
344  }
345  double tot_backward_prob = beta[0];
346  if (!ApproxEqual(tot_forward_prob, tot_backward_prob, 1e-8)) {
347  KALDI_WARN << "Total forward probability over lattice = " << tot_forward_prob
348  << ", while total backward probability = " << tot_backward_prob;
349  }
350  // Now combine any posteriors with the same transition-id.
351  for (int32 t = 0; t < max_time; t++)
352  MergePairVectorSumming(&((*post)[t]));
353  return tot_backward_prob;
354 }

408 // Computes (normal or Viterbi) alphas and betas; returns (total-prob, or
409 // best-path negated cost) Note: in either case, the alphas and betas are
410 // negated costs. Requires that lat be topologically sorted. This code
411 // will work for either CompactLattice or Latice.
412 template<typename LatticeType>
413 static double ComputeLatticeAlphasAndBetas(const LatticeType &lat,
414  bool viterbi,
415  vector<double> *alpha,
416  vector<double> *beta) {
417  typedef typename LatticeType::Arc Arc;
418  typedef typename Arc::Weight Weight;
419  typedef typename Arc::StateId StateId;
420 
421  StateId num_states = lat.NumStates();
422  KALDI_ASSERT(lat.Properties(fst::kTopSorted, true) == fst::kTopSorted);
423  KALDI_ASSERT(lat.Start() == 0);
424  alpha->resize(num_states, kLogZeroDouble);
425  beta->resize(num_states, kLogZeroDouble);
426 
427  double tot_forward_prob = kLogZeroDouble;
428  (*alpha)[0] = 0.0;
429  // Propagate alphas forward.
430  for (StateId s = 0; s < num_states; s++) {
431  double this_alpha = (*alpha)[s];
432  for (fst::ArcIterator<LatticeType> aiter(lat, s); !aiter.Done();
433  aiter.Next()) {
434  const Arc &arc = aiter.Value();
435  double arc_like = -ConvertToCost(arc.weight);
436  (*alpha)[arc.nextstate] = LogAddOrMax(viterbi, (*alpha)[arc.nextstate],
437  this_alpha + arc_like);
438  }
439  Weight f = lat.Final(s);
440  if (f != Weight::Zero()) {
441  double final_like = this_alpha - ConvertToCost(f);
442  tot_forward_prob = LogAddOrMax(viterbi, tot_forward_prob, final_like);
443  }
444  }
445  for (StateId s = num_states-1; s >= 0; s--) { // it's guaranteed signed.
446  double this_beta = -ConvertToCost(lat.Final(s));
447  for (fst::ArcIterator<LatticeType> aiter(lat, s); !aiter.Done();
448  aiter.Next()) {
449  const Arc &arc = aiter.Value();
450  double arc_like = -ConvertToCost(arc.weight),
451  arc_beta = (*beta)[arc.nextstate] + arc_like;
452  this_beta = LogAddOrMax(viterbi, this_beta, arc_beta);
453  }
454  (*beta)[s] = this_beta;
455  }
456  double tot_backward_prob = (*beta)[lat.Start()];
457  if (!ApproxEqual(tot_forward_prob, tot_backward_prob, 1e-8)) {
458  KALDI_WARN << "Total forward probability over lattice = " << tot_forward_prob
459  << ", while total backward probability = " << tot_backward_prob;
460  }
461  // Split the difference when returning... they should be the same.
462  return 0.5 * (tot_backward_prob + tot_forward_prob);
463 }

478 void CompactLatticeLimitDepth(int32 max_depth_per_frame,
479  CompactLattice *clat) {
480  typedef CompactLatticeArc Arc;
481  typedef Arc::Weight Weight;
482  typedef Arc::StateId StateId;
483 
484  if (clat->Start() == fst::kNoStateId) {
485  KALDI_WARN << "Limiting depth of empty lattice.";
486  return;
487  }
488  if (clat->Properties(fst::kTopSorted, true) == 0) {
489  if (!TopSort(clat))
490  KALDI_ERR << "Topological sorting of lattice failed.";
491  }
492 
493  vector<int32> state_times;
494  int32 T = CompactLatticeStateTimes(*clat, &state_times);
495 
496  // The alpha and beta quantities here are "viterbi" alphas and beta.
497  std::vector<double> alpha;
498  std::vector<double> beta;
499  bool viterbi = true;
500  double best_prob = ComputeLatticeAlphasAndBetas(*clat, viterbi,
501  &alpha, &beta);
502 
503  std::vector<std::vector<LatticeArcRecord> > arc_records(T);
504 
505  StateId num_states = clat->NumStates();
506  for (StateId s = 0; s < num_states; s++) {
507  for (fst::ArcIterator<CompactLattice> aiter(*clat, s); !aiter.Done();
508  aiter.Next()) {
509  const Arc &arc = aiter.Value();
510  LatticeArcRecord arc_record;
511  arc_record.state = s;
512  arc_record.arc = aiter.Position();
513  arc_record.logprob =
514  (alpha[s] + beta[arc.nextstate] - ConvertToCost(arc.weight))
515  - best_prob;
516  KALDI_ASSERT(arc_record.logprob < 0.1); // Should be zero or negative.
517  int32 num_frames = arc.weight.String().size(), start_t = state_times[s];
518  for (int32 t = start_t; t < start_t + num_frames; t++) {
519  KALDI_ASSERT(t < T);
520  arc_records[t].push_back(arc_record);
521  }
522  }
523  }
524  StateId dead_state = clat->AddState(); // A non-coaccesible state which we use
525  // to remove arcs (make them end
526  // there).
527  size_t max_depth = max_depth_per_frame;
528  for (int32 t = 0; t < T; t++) {
529  size_t size = arc_records[t].size();
530  if (size > max_depth) {
531  // we sort from worst to best, so we keep the later-numbered ones,
532  // and delete the lower-numbered ones.
533  size_t cutoff = size - max_depth;
534  std::nth_element(arc_records[t].begin(),
535  arc_records[t].begin() + cutoff,
536  arc_records[t].end());
537  for (size_t index = 0; index < cutoff; index++) {
538  LatticeArcRecord record(arc_records[t][index]);
539  fst::MutableArcIterator<CompactLattice> aiter(clat, record.state);
540  aiter.Seek(record.arc);
541  Arc arc = aiter.Value();
542  if (arc.nextstate != dead_state) { // not already killed.
543  arc.nextstate = dead_state;
544  aiter.SetValue(arc);
545  }
546  }
547  }
548  }
549  Connect(clat);
550  TopSortCompactLatticeIfNeeded(clat);
551 }

684  BaseFloat b,
685  BaseFloat max_silence_error,
686  Lattice *lat) {

730  // known before, are still known, except for whether or not the
731  // lattice was weighted.
732  lat->SetProperties(props,
733  ~(fst::kWeighted|fst::kUnweighted));

740 BaseFloat LatticeForwardBackwardMpeVariants(
741  const TransitionModel &trans,
742  const std::vector<int32> &silence_phones,
743  const Lattice &lat,
744  const std::vector<int32> &num_ali,
745  std::string criterion,
746  bool one_silence_class,
747  Posterior *post) {
748  using namespace fst;
749  typedef Lattice::Arc Arc;
750  typedef Arc::Weight Weight;
751  typedef Arc::StateId StateId;
752 
753  KALDI_ASSERT(criterion == "mpfe" || criterion == "smbr");
754  bool is_mpfe = (criterion == "mpfe");
755 
756  if (lat.Properties(fst::kTopSorted, true) == 0)
757  KALDI_ERR << "Input lattice must be topologically sorted.";
758  KALDI_ASSERT(lat.Start() == 0);
759 
760  int32 num_states = lat.NumStates();
761  vector<int32> state_times;
762  int32 max_time = LatticeStateTimes(lat, &state_times);
763  KALDI_ASSERT(max_time == static_cast<int32>(num_ali.size()));
764  std::vector<double> alpha(num_states, kLogZeroDouble),
765  alpha_smbr(num_states, 0), //forward variable for sMBR
766  beta(num_states, kLogZeroDouble),
767  beta_smbr(num_states, 0); //backward variable for sMBR
768 
769  double tot_forward_prob = kLogZeroDouble;
770  double tot_forward_score = 0;
771 
772  post->clear();
773  post->resize(max_time);
774 
775  alpha[0] = 0.0;
776  // First Pass Forward,
777  for (StateId s = 0; s < num_states; s++) {
778  double this_alpha = alpha[s];
779  for (ArcIterator<Lattice> aiter(lat, s); !aiter.Done(); aiter.Next()) {
780  const Arc &arc = aiter.Value();
781  double arc_like = -ConvertToCost(arc.weight);
782  alpha[arc.nextstate] = LogAdd(alpha[arc.nextstate], this_alpha + arc_like);
783  }
784  Weight f = lat.Final(s);
785  if (f != Weight::Zero()) {
786  double final_like = this_alpha - (f.Value1() + f.Value2());
787  tot_forward_prob = LogAdd(tot_forward_prob, final_like);
788  KALDI_ASSERT(state_times[s] == max_time &&
789  "Lattice is inconsistent (final-prob not at max_time)");
790  }
791  }
792  // First Pass Backward,
793  for (StateId s = num_states-1; s >= 0; s--) {
794  Weight f = lat.Final(s);
795  double this_beta = -(f.Value1() + f.Value2());
796  for (ArcIterator<Lattice> aiter(lat, s); !aiter.Done(); aiter.Next()) {
797  const Arc &arc = aiter.Value();
798  double arc_like = -ConvertToCost(arc.weight),
799  arc_beta = beta[arc.nextstate] + arc_like;
800  this_beta = LogAdd(this_beta, arc_beta);
801  }
802  beta[s] = this_beta;
803  }
804  // First Pass Forward-Backward Check
805  double tot_backward_prob = beta[0];
806  // may loose the condition somehow here 1e-6 (was 1e-8)
807  if (!ApproxEqual(tot_forward_prob, tot_backward_prob, 1e-6)) {
808  KALDI_ERR << "Total forward probability over lattice = " << tot_forward_prob
809  << ", while total backward probability = " << tot_backward_prob;
810  }
811 
812  alpha_smbr[0] = 0.0;
813  // Second Pass Forward, calculate forward for MPFE/SMBR
814  for (StateId s = 0; s < num_states; s++) {
815  double this_alpha = alpha[s];
816  for (ArcIterator<Lattice> aiter(lat, s); !aiter.Done(); aiter.Next()) {
817  const Arc &arc = aiter.Value();
818  double arc_like = -ConvertToCost(arc.weight);
819  double frame_acc = 0.0;
820  if (arc.ilabel != 0) {
821  int32 cur_time = state_times[s];
822  int32 phone = trans.TransitionIdToPhone(arc.ilabel),
823  ref_phone = trans.TransitionIdToPhone(num_ali[cur_time]);
824  bool phone_is_sil = std::binary_search(silence_phones.begin(),
825  silence_phones.end(),
826  phone),
827  ref_phone_is_sil = std::binary_search(silence_phones.begin(),
828  silence_phones.end(),
829  ref_phone),
830  both_sil = phone_is_sil && ref_phone_is_sil;
831  if (!is_mpfe) { // smbr.
832  int32 pdf = trans.TransitionIdToPdf(arc.ilabel),
833  ref_pdf = trans.TransitionIdToPdf(num_ali[cur_time]);
834  if (!one_silence_class) // old behavior
835  frame_acc = (pdf == ref_pdf && !phone_is_sil) ? 1.0 : 0.0;
836  else
837  frame_acc = (pdf == ref_pdf || both_sil) ? 1.0 : 0.0;
838  } else {
839  if (!one_silence_class) // old behavior
840  frame_acc = (phone == ref_phone && !phone_is_sil) ? 1.0 : 0.0;
841  else
842  frame_acc = (phone == ref_phone || both_sil) ? 1.0 : 0.0;
843  }
844  }
845  double arc_scale = Exp(alpha[s] + arc_like - alpha[arc.nextstate]);
846  alpha_smbr[arc.nextstate] += arc_scale * (alpha_smbr[s] + frame_acc);
847  }
848  Weight f = lat.Final(s);
849  if (f != Weight::Zero()) {
850  double final_like = this_alpha - (f.Value1() + f.Value2());
851  double arc_scale = Exp(final_like - tot_forward_prob);
852  tot_forward_score += arc_scale * alpha_smbr[s];
853  KALDI_ASSERT(state_times[s] == max_time &&
854  "Lattice is inconsistent (final-prob not at max_time)");
855  }
856  }
857  // Second Pass Backward, collect Mpe style posteriors
858  for (StateId s = num_states-1; s >= 0; s--) {
859  for (ArcIterator<Lattice> aiter(lat, s); !aiter.Done(); aiter.Next()) {
860  const Arc &arc = aiter.Value();
861  double arc_like = -ConvertToCost(arc.weight),
862  arc_beta = beta[arc.nextstate] + arc_like;
863  double frame_acc = 0.0;
864  int32 transition_id = arc.ilabel;
865  if (arc.ilabel != 0) {
866  int32 cur_time = state_times[s];
867  int32 phone = trans.TransitionIdToPhone(arc.ilabel),
868  ref_phone = trans.TransitionIdToPhone(num_ali[cur_time]);
869  bool phone_is_sil = std::binary_search(silence_phones.begin(),
870  silence_phones.end(), phone),
871  ref_phone_is_sil = std::binary_search(silence_phones.begin(),
872  silence_phones.end(),
873  ref_phone),
874  both_sil = phone_is_sil && ref_phone_is_sil;
875  if (!is_mpfe) { // smbr.
876  int32 pdf = trans.TransitionIdToPdf(arc.ilabel),
877  ref_pdf = trans.TransitionIdToPdf(num_ali[cur_time]);
878  if (!one_silence_class) // old behavior
879  frame_acc = (pdf == ref_pdf && !phone_is_sil) ? 1.0 : 0.0;
880  else
881  frame_acc = (pdf == ref_pdf || both_sil) ? 1.0 : 0.0;
882  } else {
883  if (!one_silence_class) // old behavior
884  frame_acc = (phone == ref_phone && !phone_is_sil) ? 1.0 : 0.0;
885  else
886  frame_acc = (phone == ref_phone || both_sil) ? 1.0 : 0.0;
887  }
888  }
889  double arc_scale = Exp(beta[arc.nextstate] + arc_like - beta[s]);
890  // check arc_scale NAN,
891  // this is to prevent partial paths in Lattices
892  // i.e., paths don't survive to the final state
893  if (KALDI_ISNAN(arc_scale)) arc_scale = 0;
894  beta_smbr[s] += arc_scale * (beta_smbr[arc.nextstate] + frame_acc);
895 
896  if (transition_id != 0) { // Arc has a transition-id on it [not epsilon]
897  double posterior = Exp(alpha[s] + arc_beta - tot_forward_prob);
898  double acc_diff = alpha_smbr[s] + frame_acc + beta_smbr[arc.nextstate]
899  - tot_forward_score;
900  double posterior_smbr = posterior * acc_diff;
901  (*post)[state_times[s]].push_back(std::make_pair(transition_id,
902  static_cast<BaseFloat>(posterior_smbr)));
903  }
904  }
905  }
906 
907  //Second Pass Forward Backward check
908  double tot_backward_score = beta_smbr[0]; // Initial state id == 0
909  // may loose the condition somehow here 1e-5/1e-4
910  if (!ApproxEqual(tot_forward_score, tot_backward_score, 1e-4)) {
911  KALDI_ERR << "Total forward score over lattice = " << tot_forward_score
912  << ", while total backward score = " << tot_backward_score;
913  }
914 
915  // Output the computed posteriors
916  for (int32 t = 0; t < max_time; t++)
917  MergePairVectorSumming(&((*post)[t]));
918  return tot_forward_score;
919 }

1044  CompactLattice *shortest_path) {

1047  CompactLattice clat_copy(clat);
1048  if (!TopSort(&clat_copy))
1049  KALDI_ERR << "Was not able to topologically sort lattice (cycles found?)";
1050  CompactLatticeShortestPath(clat_copy, shortest_path);
1051  return;
1052  }
1053  // Now we can assume it's topologically sorted.
1054  shortest_path->DeleteStates();
1055  if (clat.Start() == kNoStateId) return;
1056  KALDI_ASSERT(clat.Start() == 0); // since top-sorted.
1057  typedef CompactLatticeArc Arc;
1058  typedef Arc::StateId StateId;
1059  typedef CompactLatticeWeight Weight;
1060  vector<std::pair<double, StateId> > best_cost_and_pred(clat.NumStates() + 1);
1061  StateId superfinal = clat.NumStates();
1062  for (StateId s = 0; s <= clat.NumStates(); s++) {
1063  best_cost_and_pred[s].first = numeric_limits<double>::infinity();
1064  best_cost_and_pred[s].second = fst::kNoStateId;
1065  }
1066  best_cost_and_pred[0].first = 0;
1067  for (StateId s = 0; s < clat.NumStates(); s++) {
1068  double my_cost = best_cost_and_pred[s].first;
1069  for (ArcIterator<CompactLattice> aiter(clat, s);
1070  !aiter.Done();
1071  aiter.Next()) {
1072  const Arc &arc = aiter.Value();
1073  double arc_cost = ConvertToCost(arc.weight),
1074  next_cost = my_cost + arc_cost;
1075  if (next_cost < best_cost_and_pred[arc.nextstate].first) {
1076  best_cost_and_pred[arc.nextstate].first = next_cost;
1077  best_cost_and_pred[arc.nextstate].second = s;
1078  }
1079  }
1080  double final_cost = ConvertToCost(clat.Final(s)),
1081  tot_final = my_cost + final_cost;
1082  if (tot_final < best_cost_and_pred[superfinal].first) {
1083  best_cost_and_pred[superfinal].first = tot_final;
1084  best_cost_and_pred[superfinal].second = s;
1085  }
1086  }
1087  std::vector<StateId> states; // states on best path.
1088  StateId cur_state = superfinal;
1089  while (cur_state != 0) {
1090  StateId prev_state = best_cost_and_pred[cur_state].second;
1091  if (prev_state == kNoStateId) {
1092  KALDI_WARN << "Failure in best-path algorithm for lattice (infinite costs?)";
1093  return; // return empty best-path.
1094  }
1095  states.push_back(prev_state);
1096  KALDI_ASSERT(cur_state != prev_state && "Lattice with cycles");
1097  cur_state = prev_state;
1098  }
1099  std::reverse(states.begin(), states.end());
1100  for (size_t i = 0; i < states.size(); i++)
1101  shortest_path->AddState();
1102  for (StateId s = 0; static_cast<size_t>(s) < states.size(); s++) {
1103  if (s == 0) shortest_path->SetStart(s);
1104  if (static_cast<size_t>(s + 1) < states.size()) { // transition to next state.
1105  bool have_arc = false;
1106  Arc cur_arc;
