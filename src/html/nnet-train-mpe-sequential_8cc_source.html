44 void LatticeAcousticRescore(const Matrix<BaseFloat> &log_like,
45  const TransitionModel &trans_model,
46  const std::vector<int32> &state_times,
47  Lattice *lat) {
48  kaldi::uint64 props = lat->Properties(fst::kFstProperties, false);
49  if (!(props & fst::kTopSorted))
50  KALDI_ERR << "Input lattice must be topologically sorted.";
51 
52  KALDI_ASSERT(!state_times.empty());
53  std::vector<std::vector<int32> > time_to_state(log_like.NumRows());
54  for (size_t i = 0; i < state_times.size(); i++) {
55  KALDI_ASSERT(state_times[i] >= 0);
56  if (state_times[i] < log_like.NumRows()) // end state may be past this..
57  time_to_state[state_times[i]].push_back(i);
58  else
59  KALDI_ASSERT(state_times[i] == log_like.NumRows()
60  && "There appears to be lattice/feature mismatch.");
61  }
62 
63  for (int32 t = 0; t < log_like.NumRows(); t++) {
64  for (size_t i = 0; i < time_to_state[t].size(); i++) {
65  int32 state = time_to_state[t][i];
66  for (fst::MutableArcIterator<Lattice> aiter(lat, state); !aiter.Done();
67  aiter.Next()) {
68  LatticeArc arc = aiter.Value();
69  int32 trans_id = arc.ilabel;
70  if (trans_id != 0) { // Non-epsilon input label on arc
71  int32 pdf_id = trans_model.TransitionIdToPdf(trans_id);
72  arc.weight.SetValue2(-log_like(t, pdf_id) + arc.weight.Value2());
73  aiter.SetValue(arc);
74  }
75  }
76  }
77  }
78 }

84 int main(int argc, char *argv[]) {
85  using namespace kaldi;
86  using namespace kaldi::nnet1;
87  typedef kaldi::int32 int32;
88  try {
89  const char *usage =
90  "Perform iteration of Neural Network MPE/sMBR training by stochastic "
91  "gradient descent.\n"
92  "The network weights are updated on each utterance.\n"
93  "Usage: nnet-train-mpe-sequential [options] <model-in> <transition-model-in> "
94  "<feature-rspecifier> <den-lat-rspecifier> <ali-rspecifier> [<model-out>]\n"
95  "e.g.: \n"
96  " nnet-train-mpe-sequential nnet.init trans.mdl scp:train.scp scp:denlats.scp ark:train.ali "
97  "nnet.iter1\n";
98 
99  ParseOptions po(usage);
100 
101  NnetTrainOptions trn_opts; trn_opts.learn_rate=0.00001;
102  trn_opts.Register(&po);
103 
104  bool binary = true;
105  po.Register("binary", &binary, "Write output in binary mode");
106 
107  std::string feature_transform;
108  po.Register("feature-transform", &feature_transform,
109  "Feature transform in Nnet format");
110  std::string silence_phones_str;
111  po.Register("silence-phones", &silence_phones_str, "Colon-separated list "
112  "of integer id's of silence phones, e.g. 46:47");
113 
114  PdfPriorOptions prior_opts;
115  prior_opts.Register(&po);
116 
117  bool one_silence_class = false;
118  BaseFloat acoustic_scale = 1.0,
119  lm_scale = 1.0,
120  old_acoustic_scale = 0.0;
121  po.Register("acoustic-scale", &acoustic_scale,
122  "Scaling factor for acoustic likelihoods");
123  po.Register("lm-scale", &lm_scale,
124  "Scaling factor for \"graph costs\" (including LM costs)");
125  po.Register("old-acoustic-scale", &old_acoustic_scale,
126  "Add in the scores in the input lattices with this scale, rather "
127  "than discarding them.");
128  po.Register("one-silence-class", &one_silence_class, "If true, newer "
129  "behavior which will tend to reduce insertions.");
130  kaldi::int32 max_frames = 6000; // Allow segments maximum of one minute by default
131  po.Register("max-frames",&max_frames, "Maximum number of frames a segment can have to be processed");
132  bool do_smbr = false;
133  po.Register("do-smbr", &do_smbr, "Use state-level accuracies instead of "
134  "phone accuracies.");
135 
136  std::string use_gpu="yes";
137  po.Register("use-gpu", &use_gpu, "yes|no|optional, only has effect if compiled with CUDA");
138 
139  po.Read(argc, argv);
140 
141  if (po.NumArgs() != 6) {
142  po.PrintUsage();
143  exit(1);
144  }
145 
146  std::string model_filename = po.GetArg(1),
147  transition_model_filename = po.GetArg(2),
148  feature_rspecifier = po.GetArg(3),
149  den_lat_rspecifier = po.GetArg(4),
150  ref_ali_rspecifier = po.GetArg(5);
151 
152  std::string target_model_filename;
153  target_model_filename = po.GetArg(6);
154 
155  std::vector<int32> silence_phones;
156  if (!kaldi::SplitStringToIntegers(silence_phones_str, ":", false,
157  &silence_phones))
158  KALDI_ERR << "Invalid silence-phones string " << silence_phones_str;
159  kaldi::SortAndUniq(&silence_phones);
160  if (silence_phones.empty())
161  KALDI_LOG << "No silence phones specified.";
162 
163  // Select the GPU
164 #if HAVE_CUDA == 1
165  CuDevice::Instantiate().SelectGpuId(use_gpu);
166 #endif
167 
168  Nnet nnet_transf;
169  if (feature_transform != "") {
170  nnet_transf.Read(feature_transform);
171  }
172 
173  Nnet nnet;
174  nnet.Read(model_filename);
175  // using activations directly: remove softmax, if present
176  if (nnet.GetComponent(nnet.NumComponents()-1).GetType() ==
177  kaldi::nnet1::Component::kSoftmax) {
178  KALDI_LOG << "Removing softmax from the nnet " << model_filename;
179  nnet.RemoveComponent(nnet.NumComponents()-1);
180  } else {
181  KALDI_LOG << "The nnet was without softmax " << model_filename;
182  }
183  nnet.SetTrainOptions(trn_opts);
184 
185  // Read the class-frame-counts, compute priors
186  PdfPrior log_prior(prior_opts);
187 
188  // Read transition model
189  TransitionModel trans_model;
190  ReadKaldiObject(transition_model_filename, &trans_model);
191 
192  SequentialBaseFloatMatrixReader feature_reader(feature_rspecifier);
193  RandomAccessLatticeReader den_lat_reader(den_lat_rspecifier);
194  RandomAccessInt32VectorReader ref_ali_reader(ref_ali_rspecifier);
195 
196  CuMatrix<BaseFloat> feats, feats_transf, nnet_out, nnet_diff;
197  Matrix<BaseFloat> nnet_out_h;
198 
199  Timer time;
200  double time_now = 0;
201  KALDI_LOG << "TRAINING STARTED";
202 
203  int32 num_done = 0, num_no_ref_ali = 0, num_no_den_lat = 0,
204  num_other_error = 0;
205 
206  kaldi::int64 total_frames = 0;
207  double total_frame_acc = 0.0, utt_frame_acc;
208 
209  // do per-utterance processing
210  for (; !feature_reader.Done(); feature_reader.Next()) {
211  std::string utt = feature_reader.Key();
212  if (!den_lat_reader.HasKey(utt)) {
213  KALDI_WARN << "Utterance " << utt << ": found no lattice.";
214  num_no_den_lat++;
215  continue;
216  }
217  if (!ref_ali_reader.HasKey(utt)) {
218  KALDI_WARN << "Utterance " << utt << ": found no reference alignment.";
219  num_no_ref_ali++;
220  continue;
221  }
222 
223  // 1) get the features, numerator alignment
224  const Matrix<BaseFloat> &mat = feature_reader.Value();
225  const std::vector<int32> &ref_ali = ref_ali_reader.Value(utt);
226  // check for temporal length of numerator alignments
227  if (static_cast<MatrixIndexT>(ref_ali.size()) != mat.NumRows()) {
228  KALDI_WARN << "Numerator alignment has wrong length "
229  << ref_ali.size() << " vs. "<< mat.NumRows();
230  num_other_error++;
231  continue;
232  }
233  if (mat.NumRows() > max_frames) {
234  KALDI_WARN << "Utterance " << utt << ": Skipped because it has "
235  << mat.NumRows() << " frames, which is more than " << max_frames << ".";
236  num_other_error++;
237  continue;
238  }
239  // 2) get the denominator lattice, preprocess
240  Lattice den_lat = den_lat_reader.Value(utt);
241  if (den_lat.Start() == -1) {
242  KALDI_WARN << "Empty lattice for utt " << utt;
243  num_other_error++;
244  continue;
245  }
246  if (old_acoustic_scale != 1.0) {
247  fst::ScaleLattice(fst::AcousticLatticeScale(old_acoustic_scale),
248  &den_lat);
249  }
250  // optional sort it topologically
251  kaldi::uint64 props = den_lat.Properties(fst::kFstProperties, false);
252  if (!(props & fst::kTopSorted)) {
253  if (fst::TopSort(&den_lat) == false)
254  KALDI_ERR << "Cycles detected in lattice.";
255  }
256  // get the lattice length and times of states
257  vector<int32> state_times;
258  int32 max_time = kaldi::LatticeStateTimes(den_lat, &state_times);
259  // check for temporal length of denominator lattices
260  if (max_time != mat.NumRows()) {
261  KALDI_WARN << "Denominator lattice has wrong length "
262  << max_time << " vs. " << mat.NumRows();
263  num_other_error++;
264  continue;
265  }
266 
267  // get actual dims for this utt and nnet
268  int32 num_frames = mat.NumRows(),
269  num_fea = mat.NumCols(),
270  num_pdfs = nnet.OutputDim();
271 
272  // 3) propagate the feature to get the log-posteriors (nnet w/o sofrmax)
273  // push features to GPU
274  feats.Resize(num_frames, num_fea, kUndefined);
275  feats.CopyFromMat(mat);
276  // possibly apply transform
277  nnet_transf.Feedforward(feats, &feats_transf);
278  // propagate through the nnet (assuming w/o softmax)
279  nnet.Propagate(feats_transf, &nnet_out);
280  // subtract the log_prior
281  if (prior_opts.class_frame_counts != "") {
282  log_prior.SubtractOnLogpost(&nnet_out);
283  }
284  // transfer it back to the host
285  nnet_out_h.Resize(num_frames, num_pdfs, kUndefined);
286  nnet_out.CopyToMat(&nnet_out_h);
287  // release the buffers we don't need anymore
288  feats.Resize(0,0);
289  feats_transf.Resize(0,0);
290  nnet_out.Resize(0,0);
291 
292  // 4) rescore the latice
293  LatticeAcousticRescore(nnet_out_h, trans_model, state_times, &den_lat);
294  if (acoustic_scale != 1.0 || lm_scale != 1.0)
295  fst::ScaleLattice(fst::LatticeScale(lm_scale, acoustic_scale), &den_lat);
296 
297  kaldi::Posterior post;
298 
299  if (do_smbr) { // use state-level accuracies, i.e. sMBR estimation
300  utt_frame_acc = LatticeForwardBackwardMpeVariants(
301  trans_model, silence_phones, den_lat, ref_ali, "smbr",
302  one_silence_class, &post);
303  } else { // use phone-level accuracies, i.e. MPFE (minimum phone frame error)
304  utt_frame_acc = LatticeForwardBackwardMpeVariants(
305  trans_model, silence_phones, den_lat, ref_ali, "mpfe",
306  one_silence_class, &post);
307  }
308 
309  // 6) convert the Posterior to a matrix,
310  PosteriorToMatrixMapped(post, trans_model, &nnet_diff);
311  nnet_diff.Scale(-1.0); // need to flip the sign of derivative,
312 
313  KALDI_VLOG(1) << "Lattice #" << num_done + 1 << " processed"
314  << " (" << utt << "): found " << den_lat.NumStates()
315  << " states and " << fst::NumArcs(den_lat) << " arcs.";
316 
317  KALDI_VLOG(1) << "Utterance " << utt << ": Average frame accuracy = "
318  << (utt_frame_acc/num_frames) << " over " << num_frames
319  << " frames,"
320  << " diff-range(" << nnet_diff.Min() << "," << nnet_diff.Max() << ")";
321 
322  // 7) backpropagate through the nnet,
323  nnet.Backpropagate(nnet_diff, NULL);
324  nnet_diff.Resize(0,0); // release GPU memory,
325 
326  // increase time counter
327  total_frame_acc += utt_frame_acc;
328  total_frames += num_frames;
329  num_done++;
330 
331  if (num_done % 100 == 0) {
332  time_now = time.Elapsed();
333  KALDI_VLOG(1) << "After " << num_done << " utterances: time elapsed = "
334  << time_now/60 << " min; processed " << total_frames/time_now
335  << " frames per second.";
336 #if HAVE_CUDA==1
337  // check the GPU is not overheated
338  CuDevice::Instantiate().CheckGpuHealth();
339 #endif
340  }
341 
342  // Gradient logging,
343  if (kaldi::g_kaldi_verbose_level >= 1 &&
344  (num_done == 1 || num_done % 1000 == 0)) {
345  // first utterance, or every 1000th utterance,
346  KALDI_VLOG(1) << nnet.InfoPropagate();
347  KALDI_VLOG(1) << nnet.InfoBackPropagate();
348  KALDI_VLOG(1) << nnet.InfoGradient();
349  } else if (kaldi::g_kaldi_verbose_level >= 2) {
350  // vlog 2 => every utterance,
351  KALDI_VLOG(2) << nnet.InfoPropagate();
352  KALDI_VLOG(2) << nnet.InfoBackPropagate();
353  KALDI_VLOG(2) << nnet.InfoGradient();
354  }
355  }
356 
357  // after last utterance,
358  KALDI_VLOG(1) << nnet.InfoPropagate();
359  KALDI_VLOG(1) << nnet.InfoBackPropagate();
360  KALDI_VLOG(1) << nnet.InfoGradient();
361 
362  // add the softmax layer back before writing,
363  KALDI_LOG << "Appending the softmax " << target_model_filename;
364  nnet.AppendComponent(new Softmax(nnet.OutputDim(),nnet.OutputDim()));
365  // store the nnet,
366  nnet.Write(target_model_filename, binary);
367 
368  time_now = time.Elapsed();
369  KALDI_LOG << "TRAINING FINISHED; "
370  << "Time taken = " << time_now/60 << " min; processed "
371  << (total_frames/time_now) << " frames per second.";
372 
373  KALDI_LOG << "Done " << num_done << " files, "
374  << num_no_ref_ali << " with no reference alignments, "
375  << num_no_den_lat << " with no lattices, "
376  << num_other_error << " with other errors.";
377 
378  KALDI_LOG << "Overall average frame-accuracy is "
379  << (total_frame_acc/total_frames) << " over " << total_frames
380  << " frames.";
381 
382 #if HAVE_CUDA == 1
383  CuDevice::Instantiate().PrintProfile();
384 #endif
385 
386  return 0;
387  } catch(const std::exception &e) {
388  std::cerr << e.what();
389  return -1;
390  }
