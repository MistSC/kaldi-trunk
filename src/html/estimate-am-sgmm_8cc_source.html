102 void MleAmSgmmAccs::Read(std::istream &in_stream, bool binary,
103  bool add) {
104  uint32 tmp_uint32;
105  string token;
106 
107  ExpectToken(in_stream, binary, "<SGMMACCS>");
108 
109  ExpectToken(in_stream, binary, "<NUMSTATES>");
110  ReadBasicType(in_stream, binary, &tmp_uint32);
111  num_states_ = static_cast<int32>(tmp_uint32);
112  ExpectToken(in_stream, binary, "<NUMGaussians>");
113  ReadBasicType(in_stream, binary, &tmp_uint32);
114  num_gaussians_ = static_cast<int32>(tmp_uint32);
115  ExpectToken(in_stream, binary, "<FEATUREDIM>");
116  ReadBasicType(in_stream, binary, &tmp_uint32);
117  feature_dim_ = static_cast<int32>(tmp_uint32);
118  ExpectToken(in_stream, binary, "<PHONESPACEDIM>");
119  ReadBasicType(in_stream, binary, &tmp_uint32);
120  phn_space_dim_ = static_cast<int32>(tmp_uint32);
121  ExpectToken(in_stream, binary, "<SPKSPACEDIM>");
122  ReadBasicType(in_stream, binary, &tmp_uint32);
123  spk_space_dim_ = static_cast<int32>(tmp_uint32);
124 
125  ReadToken(in_stream, binary, &token);
126 
127  while (token != "</SGMMACCS>") {
128  if (token == "<Y>") {
129  Y_.resize(num_gaussians_);
130  for (size_t i = 0; i < Y_.size(); i++) {
131  Y_[i].Read(in_stream, binary, add);
132  }
133  } else if (token == "<Z>") {
134  Z_.resize(num_gaussians_);
135  for (size_t i = 0; i < Z_.size(); i++) {
136  Z_[i].Read(in_stream, binary, add);
137  }
138  } else if (token == "<R>") {
139  R_.resize(num_gaussians_);
140  if (gamma_s_.Dim() == 0) gamma_s_.Resize(num_gaussians_);
141  for (size_t i = 0; i < R_.size(); i++) {
142  R_[i].Read(in_stream, binary, add);
143  }
144  } else if (token == "<S>") {
145  S_.resize(num_gaussians_);
146  for (size_t i = 0; i < S_.size(); i++) {
147  S_[i].Read(in_stream, binary, add);
148  }
149  } else if (token == "<y>") {
150  y_.resize(num_states_);
151  for (int32 j = 0; j < num_states_; j++) {
152  y_[j].Read(in_stream, binary, add);
153  }
154  } else if (token == "<gamma>") {
155  gamma_.resize(num_states_);
156  for (int32 j = 0; j < num_states_; j++) {
157  gamma_[j].Read(in_stream, binary, add);
158  }
159  // Don't read gamma_s, it's just a temporary variable and
160  // not part of the permanent (non-speaker-specific) accs.
161  } else if (token == "<total_like>") {
162  double total_like;
163  ReadBasicType(in_stream, binary, &total_like);
164  if (add)
165  total_like_ += total_like;
166  else
167  total_like_ = total_like;
168  } else if (token == "<total_frames>") {
169  double total_frames;
170  ReadBasicType(in_stream, binary, &total_frames);
171  if (add)
172  total_frames_ += total_frames;
173  else
174  total_frames_ = total_frames;
175  } else {
176  KALDI_ERR << "Unexpected token '" << token << "' in model file ";
177  }
178  ReadToken(in_stream, binary, &token);
179  }

248  KALDI_ASSERT(gamma_.size() != 0);

251  for (int32 j = 0; j < num_states_; j++) {
252  KALDI_ASSERT(y_[j].NumRows() == model.NumSubstates(j));
253  KALDI_ASSERT(y_[j].NumCols() == phn_space_dim_);
254  if (!nz && y_[j](0, 0) != 0) { nz = true; }

278 void MleAmSgmmAccs::ResizeAccumulators(const AmSgmm &model,
279  SgmmUpdateFlagsType flags) {
280  num_states_ = model.NumPdfs();
281  num_gaussians_ = model.NumGauss();
282  feature_dim_ = model.FeatureDim();
283  phn_space_dim_ = model.PhoneSpaceDim();
284  spk_space_dim_ = model.SpkSpaceDim();
285 
286  if (flags & (kSgmmPhoneProjections | kSgmmCovarianceMatrix)) {
287  Y_.resize(num_gaussians_);
288  for (int32 i = 0; i < num_gaussians_; i++) {
289  Y_[i].Resize(feature_dim_, phn_space_dim_);
290  }
291  } else {
292  Y_.clear();
293  }
294 
295  if (flags & kSgmmSpeakerProjections) {
296  if (spk_space_dim_ == 0) {
297  KALDI_ERR << "Cannot set up accumulators for speaker projections "
298  << "because speaker subspace has not been set up";
299  }
300  gamma_s_.Resize(num_gaussians_);
301  Z_.resize(num_gaussians_);
302  R_.resize(num_gaussians_);
303  for (int32 i = 0; i < num_gaussians_; i++) {
304  Z_[i].Resize(feature_dim_, spk_space_dim_);
305  R_[i].Resize(spk_space_dim_);
306  }
307  } else {
308  gamma_s_.Resize(0);
309  Z_.clear();
310  R_.clear();
311  }
312 
313  if (flags & kSgmmCovarianceMatrix) {
314  S_.resize(num_gaussians_);
315  for (int32 i = 0; i < num_gaussians_; i++) {
316  S_[i].Resize(feature_dim_);
317  }
318  } else {
319  S_.clear();
320  }
321 
322  if (flags & (kSgmmPhoneVectors | kSgmmPhoneWeightProjections |
323  kSgmmCovarianceMatrix | kSgmmSubstateWeights |
324  kSgmmPhoneProjections)) {
325  gamma_.resize(num_states_);
326  total_frames_ = total_like_ = 0;
327  for (int32 j = 0; j < num_states_; j++) {
328  gamma_[j].Resize(model.NumSubstates(j), num_gaussians_);
329  }
330  } else {
331  gamma_.clear();
332  total_frames_ = total_like_ = 0;
333  }
334 
335  if (flags & kSgmmPhoneVectors) {
336  y_.resize(num_states_);
337  for (int32 j = 0; j < num_states_; j++) {
338  y_[j].Resize(model.NumSubstates(j), phn_space_dim_);
339  }
340  } else {
341  y_.clear();
342  }
343 }

345 BaseFloat MleAmSgmmAccs::Accumulate(const AmSgmm &model,
346  const SgmmPerFrameDerivedVars &frame_vars,
347  const VectorBase<BaseFloat> &v_s, // may be empty
348  int32 j, BaseFloat weight,
349  SgmmUpdateFlagsType flags) {
350  // Calculate Gaussian posteriors and collect statistics
351  Matrix<BaseFloat> posteriors;
352  BaseFloat log_like = model.ComponentPosteriors(frame_vars, j, &posteriors);
353  posteriors.Scale(weight);
354  BaseFloat count = AccumulateFromPosteriors(model, frame_vars, posteriors,
355  v_s, j, flags);
356  // Note: total_frames_ is incremented in AccumulateFromPosteriors().
357  total_like_ += count * log_like;
358  return log_like;
359 }

362 BaseFloat MleAmSgmmAccs::AccumulateFromPosteriors(
363  const AmSgmm &model,
364  const SgmmPerFrameDerivedVars &frame_vars,
365  const Matrix<BaseFloat> &posteriors,
366  const VectorBase<BaseFloat> &v_s, // may be empty
367  int32 j,
368  SgmmUpdateFlagsType flags) {
369  double tot_count = 0.0;
370  const vector<int32> &gselect = frame_vars.gselect;
371  // Intermediate variables
372  Vector<BaseFloat> gammat(gselect.size());
373  Vector<BaseFloat> xt_jmi(feature_dim_), mu_jmi(feature_dim_),
374  zt_jmi(spk_space_dim_);
375 
376  int32 num_substates = model.NumSubstates(j);
377  for (int32 ki = 0; ki < static_cast<int32>(gselect.size()); ki++) {
378  int32 i = gselect[ki];
379 
380  for (int32 m = 0; m < num_substates; m++) {
381  // Eq. (39): gamma_{jmi}(t) = p (j, m, i|t)
382  BaseFloat gammat_jmi = RandPrune(posteriors(ki, m), rand_prune_);
383 
384  // Accumulate statistics for non-zero gaussian posterior
385  if (gammat_jmi != 0.0) {
386  tot_count += gammat_jmi;
387  if (flags & (kSgmmPhoneVectors | kSgmmPhoneWeightProjections |
388  kSgmmCovarianceMatrix | kSgmmSubstateWeights |
389  kSgmmPhoneProjections)) {
390  // Eq. (40): gamma_{jmi} = \sum_t gamma_{jmi}(t)
391  gamma_[j](m, i) += gammat_jmi;
392  }
393 
394  if (flags & kSgmmPhoneVectors) {
395  // Eq. (41): y_{jm} = \sum_{t, i} \gamma_{jmi}(t) z_{i}(t)
396  // Suggestion: move this out of the loop over m
397  y_[j].Row(m).AddVec(gammat_jmi, frame_vars.zti.Row(ki));
398  }
399 
400  if (flags & (kSgmmPhoneProjections | kSgmmCovarianceMatrix)) {
401  // Eq. (42): Y_{i} = \sum_{t, j, m} \gamma_{jmi}(t) x_{i}(t) v_{jm}^T
402  Y_[i].AddVecVec(gammat_jmi, frame_vars.xti.Row(ki),
403  model.StateVectors(j).Row(m));
404  }
405 
406  if (flags & kSgmmCovarianceMatrix)
407  gammat(ki) += gammat_jmi;
408 
409  // Accumulate for speaker projections
410  if (flags & kSgmmSpeakerProjections) {
411  KALDI_ASSERT(spk_space_dim_ > 0);
412  // Eq. (43): x_{jmi}(t) = x_k(t) - M{i} v_{jm}
413  model.GetSubstateMean(j, m, i, &mu_jmi);
414  xt_jmi.CopyFromVec(frame_vars.xt);
415  xt_jmi.AddVec(-1.0, mu_jmi);
416  // Eq. (44): Z_{i} = \sum_{t, j, m} \gamma_{jmi}(t) x_{jmi}(t) v^{s}'
417  if (v_s.Dim() != 0) // interpret empty v_s as zero.
418  Z_[i].AddVecVec(gammat_jmi, xt_jmi, v_s);
419  // Eq. (49): \gamma_{i}^{(s)} = \sum_{t\in\Tau(s), j, m} gamma_{jmi}
420  // Will be used when you call CommitStatsForSpk(), to update R_.
421  gamma_s_(i) += gammat_jmi;
422  }
423  } // non-zero posteriors
424  } // loop over substates
425  } // loop over selected Gaussians
426 
427  if (flags & kSgmmCovarianceMatrix) {
428  for (int32 ki = 0; ki < static_cast<int32>(gselect.size()); ki++) {
429  int32 i = gselect[ki];
430  // Eq. (47): S_{i} = \sum_{t, j, m} \gamma_{jmi}(t) x_{i}(t) x_{i}(t)^T
431  if (gammat(ki) != 0.0)
432  S_[i].AddVec2(gammat(ki), frame_vars.xti.Row(ki));
433  }
434  }
435  total_frames_ += tot_count;
436  return tot_count;
437 }

458 BaseFloat MleAmSgmmUpdater::Update(const MleAmSgmmAccs &accs,
459  AmSgmm *model,
460  SgmmUpdateFlagsType flags) {
461  KALDI_ASSERT((flags & (kSgmmPhoneVectors | kSgmmPhoneProjections |
462  kSgmmPhoneWeightProjections | kSgmmCovarianceMatrix |
463  kSgmmSubstateWeights | kSgmmSpeakerProjections)) != 0);
464 
465  if (flags & kSgmmPhoneProjections)
466  ComputeQ(accs, *model, &Q_);
467  if (flags & kSgmmCovarianceMatrix)
468  ComputeSMeans(accs, *model, &S_means_);
469 
470  // quantities used in both vector and weights updates...
471  vector< SpMatrix<double> > H;
472  // "smoothing" matrices, weighted sums of above.
473  SpMatrix<double> H_sm;
474  Vector<double> y_sm; // "smoothing" vectors
475  if ((flags & (kSgmmPhoneVectors | kSgmmPhoneWeightProjections))
476  || update_options_.renormalize_V) {
477  model->ComputeH(&H);
478  ComputeSmoothingTerms(accs, *model, H, &H_sm,
479  (flags & kSgmmPhoneVectors) ? &y_sm : NULL);
480  }
481 
482  BaseFloat tot_impr = 0.0;
483 
484  if (flags & kSgmmPhoneVectors) {
485  if (update_options_.check_v) {
486  KALDI_ASSERT(update_options_.tau_vec == 0 &&
487  "You cannot combine the check-v and tau-vec options.");
488  tot_impr += UpdatePhoneVectorsChecked(accs, model, H);
489  } else {
490  tot_impr += UpdatePhoneVectors(accs, model, H, H_sm, y_sm);
491  }
492  }
493  if (flags & kSgmmPhoneProjections) {
494  if (update_options_.tau_map_M > 0.0)
495  tot_impr += MapUpdateM(accs, model); // MAP adaptation of M
496  else
497  tot_impr += UpdateM(accs, model);
498  }
499 
500  if (flags & kSgmmPhoneWeightProjections) {
501  if (update_options_.use_sequential_weight_update) {
502  tot_impr += UpdateWSequential(accs, model);
503  } else {
504  tot_impr += UpdateWParallel(accs, model);
505  }
506  }
507  if (flags & kSgmmCovarianceMatrix)
508  tot_impr += UpdateVars(accs, model);
509  if (flags & kSgmmSubstateWeights)
510  tot_impr += UpdateSubstateWeights(accs, model);
511  if (flags & kSgmmSpeakerProjections) {
512  tot_impr += UpdateN(accs, model);
513  if (update_options_.renormalize_N)
514  RenormalizeN(accs, model); // if you renormalize N you have to
515  // alter any speaker vectors you're keeping around, as well.
516  }
517 
518  if (update_options_.renormalize_V)
519  RenormalizeV(accs, model, H_sm);
520 
521  KALDI_LOG << "*Overall auxf improvement, combining all parameters, is "
522  << tot_impr;
523 
524  KALDI_LOG << "***Overall data likelihood is "
525  << (accs.total_like_/accs.total_frames_)

551 // Note: we seem to have also included in this variable
552 // the term - (Y_i M_I^T + M_i Y_i^T).
553 void MleAmSgmmUpdater::ComputeSMeans(const MleAmSgmmAccs &accs,
554  const AmSgmm &model,
555  std::vector< SpMatrix<double> > *S_means) {
556  S_means->resize(accs.num_gaussians_);
557  Matrix<double> YM_MY(accs.feature_dim_, accs.feature_dim_);
558  Vector<BaseFloat> mu_jmi(accs.feature_dim_);
559  for (int32 i = 0; i < accs.num_gaussians_; i++) {
560  // YM_MY = - (Y_{i} M_{i}^T)
561  YM_MY.AddMatMat(-1.0, accs.Y_[i], kNoTrans,
562  Matrix<double>(model.M_[i]), kTrans, 0.0);
563  // Add its own transpose: YM_MY = - (Y_{i} M_{i}^T + M_{i} Y_{i}^T)
564  {
565  Matrix<double> M(YM_MY, kTrans);
566  YM_MY.AddMat(1.0, M);
567  }
568  (*S_means)[i].Resize(accs.feature_dim_, kUndefined);
569  (*S_means)[i].CopyFromMat(YM_MY); // Sigma_{i} = -(YM' + MY')
570 
571  for (int32 j = 0; j < accs.num_states_; j++) {
572  for (int32 m = 0; m < model.NumSubstates(j); m++) {
573  if (accs.gamma_[j](m, i) != 0.0) {
574  // Sigma_{i} += gamma_{jmi} * mu_{jmi}*mu_{jmi}^T
575  mu_jmi.AddMatVec(1.0, model.M_[i], kNoTrans, model.v_[j].Row(m), 0.0);
576  (*S_means)[i].AddVec2(static_cast<BaseFloat>(accs.gamma_[j](m, i)), mu_jmi);
577  }
578  }
579  }
580  KALDI_ASSERT(1.0 / (*S_means)[i](0, 0) != 0.0);

585 void MleAmSgmmUpdater::ComputeSmoothingTerms(const MleAmSgmmAccs &accs,

587  const vector<SpMatrix<double> > &H,
588  SpMatrix<double> *H_sm,
589  Vector<double> *y_sm) const {

616  (*y_sm).Scale(1.0 / sum);

625  Vector<double> avgVec(accs.phn_space_dim_);

627  HInv.Invert();
628  avgVec.AddSpVec(1.0, HInv, (*y_sm), 0.0);
629  (*y_sm).AddSpVec(1.0, (*H_sm), avgVec, 0.0);

677 // Runs the phone vectors update for a subset of states (called
678 // multi-threaded).
679 void MleAmSgmmUpdater::UpdatePhoneVectorsInternal(
680  const MleAmSgmmAccs &accs,
681  AmSgmm *model,
682  const std::vector<SpMatrix<double> > &H,
683  const SpMatrix<double> &H_sm,
684  const Vector<double> &y_sm,
685  double *auxf_impr,
686  double *like_impr,
687  int32 num_threads,
688  int32 thread_id) const {
689 
690  int32 block_size = (accs.num_states_ + (num_threads-1)) / num_threads,
691  j_start = block_size * thread_id,
692  j_end = std::min(accs.num_states_, j_start + block_size);
693 
694  for (int32 j = j_start; j < j_end; j++) {
695  double state_count = 0.0, state_auxf_impr = 0.0, state_like_impr = 0.0;
696  Vector<double> w_jm(accs.num_gaussians_);
697  for (int32 m = 0; m < model->NumSubstates(j); m++) {
698  double gamma_jm = accs.gamma_[j].Row(m).Sum();
699  state_count += gamma_jm;
700  Vector<double> g_jm(accs.phn_space_dim_); // computed using eq. 58
701  SpMatrix<double> H_jm(accs.phn_space_dim_); // computed using eq. 59
702  // First compute normal H_jm.
703 
704  // need weights for this ...
705  // w_jm = softmax([w_{k1}^T ... w_{kD}^T] * v_{jkm}) eq.(7)
706  w_jm.AddMatVec(1.0, Matrix<double>(model->w_), kNoTrans,
707  Vector<double>(model->v_[j].Row(m)), 0.0);
708  w_jm.ApplySoftMax();
709  g_jm.CopyFromVec(accs.y_[j].Row(m));
710 
711  for (int32 i = 0; i < accs.num_gaussians_; i++) {
712  double gamma_jmi = accs.gamma_[j](m, i);
713  double quadratic_term = std::max(gamma_jmi, gamma_jm * w_jm(i));
714  double scalar = gamma_jmi - gamma_jm * w_jm(i) + quadratic_term
715  * VecVec(model->w_.Row(i), model->v_[j].Row(m));
716  g_jm.AddVec(scalar, model->w_.Row(i));
717  if (gamma_jmi != 0.0) {
718  H_jm.AddSp(gamma_jmi, H[i]); // The most important term..
719  }
720  if (quadratic_term > 1.0e-10) {
721  H_jm.AddVec2(static_cast<BaseFloat>(quadratic_term), model->w_.Row(i));
722  }
723  }
724  SpMatrix<double> H_jm_dash(H_jm); // with ad-hoc smoothing term.
725  Vector<double> g_jm_dash(g_jm); // with ad-hoc smoothing term.
726 
727  // H_jm_dash = H_jm + (smoothing term)
728  H_jm_dash.AddSp(update_options_.tau_vec, H_sm);
729  // g_jm_dash.BlasGemv(update_options_.mTauVec, H_sm, kNoTrans, e_1, 1.0);
730  // g_jm_dash = g_jm + (smoothing term)
731  g_jm_dash.AddVec(update_options_.tau_vec, y_sm);
732 
733  // if (gamma_jm == 0) continue;
734  // no, we still want to update even with zero count.
735 #ifdef KALDI_PARANOID
736  if (update_options_.tau_vec > 0)
737  KALDI_ASSERT(H_jm_dash.IsPosDef());
738 #endif
739  Vector<double> vhat_jm(model->v_[j].Row(m));
740  SolverOptions opts;
741  opts.name = "v";
742  opts.K = update_options_.max_cond;
743  opts.eps = update_options_.epsilon;
744  double objf_impr_with_prior =
745  SolveQuadraticProblem(H_jm_dash,
746  g_jm_dash,
747  opts,
748  &vhat_jm);
749 
750  SpMatrix<BaseFloat> H_jm_flt(H_jm);
751 
752  double objf_impr_noprior =
753  (VecVec(vhat_jm, g_jm)
754  - 0.5 * VecSpVec(vhat_jm, H_jm, vhat_jm))
755  - (VecVec(model->v_[j].Row(m), g_jm)
756  - 0.5 * VecSpVec(model->v_[j].Row(m), H_jm_flt, model->v_[j].Row(m)));
757  model->v_[j].Row(m).CopyFromVec(vhat_jm);
758  if (j < 3 && m < 2 && thread_id == 0) {
759  KALDI_LOG << "Objf impr for j = " << (j) << " m = " << (m) << " is "
760  << (objf_impr_with_prior / (gamma_jm + 1.0e-20))
761  << " (with ad-hoc prior) "
762  << (objf_impr_noprior / (gamma_jm + 1.0e-20))
763  << " (no prior) over " << (gamma_jm) << " frames";
764  }
765  state_auxf_impr += objf_impr_with_prior;
766  state_like_impr += objf_impr_noprior;
767  }
768 
769  *auxf_impr += state_auxf_impr;
770  *like_impr += state_like_impr;
771  if (j < 10 && thread_id == 0) {
772  KALDI_LOG << "Objf impr for state j = " << (j) << " is "
773  << (state_auxf_impr / (state_count + 1.0e-20))
774  << " (with ad-hoc prior) "
775  << (state_like_impr / (state_count + 1.0e-20))
776  << " (no prior) over " << (state_count) << " frames";
777  }
778  }

1027  v_jm.Scale(0.5);
1028  }
1029  }

1048  weight_2nd_deriv.AddVec2(static_cast<BaseFloat>(scalar), wi);
1049  }

1051  total_linear_term.AddVec(1.0, num_deriv);
1052  total_linear_term.AddVec(-1.0, den_deriv);
1053  if (gamma_jm > 0.0)
1054  weight_2nd_deriv.AddVec2(-1.0/gamma_jm, den_deriv);

1057  // we want the derivatives around zero, not around the current point.
1058  // Correction for this.
1059

1068  total_linear_term, opts, &v_jm);
1069  }

1072  tot_count += gamma_jm;
1073  tot_objf_impr += objf_impr;
1074  tot_auxf_impr += auxf_impr;
1075  if (backtrack_iter == max_backtrack) {
1076  KALDI_WARN << "Backtracked " << max_backtrack << " times [not updating]";
1077  } else {
1078  model->v_[j].Row(m).CopyFromVec(v_jm);
1079  }

1082  << objf_impr << " vs. quadratic auxf impr (before backtrack) "
1083  << auxf_impr;
1084  }

1106  for (int32 m = 0; m < model->NumSubstates(j); m++) {

1113  if (fixed_eigs != 0) {

1115  << fixed_eigs << " eigenvalues.";
1116  }
1117  KALDI_LOG << "Eigenvalues of scatter of vectors v is : ";
1118  Sigma.PrintEigs("Sigma");
1119  if (!Sigma.IsPosDef()) {
1120  KALDI_LOG << "Not renormalizing v because scatter is not positive definite"
1121  << " -- maybe first iter?";
1122  return;
1123  }
1124 
1125  // Want to make variance of v unit and H_sm (like precision matrix) diagonal.
1126  TpMatrix<double> L(accs.phn_space_dim_);
1127  L.Cholesky(Sigma);
1128  TpMatrix<double> LInv(L);
1129  LInv.Invert();
1130

1135  H_sm_proj.AddMat2Sp(1.0, tmpL, kTrans, H_sm, 0.0);

1137  // This is right because we would transform the vectors themselves
1138  // by L^{-1}, and H_sm is like the inverse of the vectors,
1139  // so it's {L^{-1}}^{-T} = L^T.

1149  // Why? Because transform on H_sm is T =U^T L^T
1150  // and we want T^{-T} by normal rules of vector/covector and we
1151  // have (U^T L^T)^{-T} = (L U)^{-1} = U^T L^{-1}.
1152  Matrix<double> Trans(accs.phn_space_dim_, accs.phn_space_dim_); // T^{-T}
1153  Matrix<double> tmpLInv(accs.phn_space_dim_, accs.phn_space_dim_);

1155  Trans.AddMatMat(1.0, U, kTrans, tmpLInv, kNoTrans, 0.0);

1157  TransInv.Invert(); // T in above...
1158 
1159 #ifdef KALDI_PARANOID
1160  {
1161  SpMatrix<double> H_sm_tmp(accs.phn_space_dim_);

1163  KALDI_ASSERT(H_sm_tmp.IsDiagonal(0.1));
1164  }
1165  {
1166  SpMatrix<double> Sigma_tmp(accs.phn_space_dim_);
1167  Sigma_tmp.AddMat2Sp(1.0, Trans, kNoTrans, Sigma, 0.0);
1168  KALDI_ASSERT(Sigma_tmp.IsUnit(0.1));
1169  }
1170 #endif
1171 
1172  for (int32 j = 0; j < accs.num_states_; j++) {
1173  for (int32 m = 0; m < model->NumSubstates(j); m++) {
1174  Vector<double> tmp(accs.phn_space_dim_);
1175  tmp.AddMatVec(1.0, Trans, kNoTrans, Vector<double>(model->v_[j].Row(m)), 0.0);
1176  model->v_[j].Row(m).CopyFromVec(tmp);
1177  }
1178  }
1179  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1180  Vector<double> tmp(accs.phn_space_dim_);
1181  tmp.AddMatVec(1.0, TransInv, kTrans, Vector<double>(model->w_.Row(i)), 0.0);
1182  model->w_.Row(i).CopyFromVec(tmp);

1185  // Multiplying on right not left so must not transpose TransInv.
1186  tmpM.AddMatMat(1.0, Matrix<double>(model->M_[i]), kNoTrans,
1187  TransInv, kNoTrans, 0.0);
1188  model->M_[i].CopyFromMat(tmpM);

1233 // Estimate the parameters of a Gaussian prior over the M matrices. There are
1234 // as many mean matrices as UBM size and two covariance matrices for the rows
1235 // of M and columns of M. The prior means M_i are fixed to the unadapted values.
1236 // This is what was done in Lu, et al. "Maximum a posteriori adaptation of
1237 // subspace Gaussian mixture models for cross-lingual speech recognition",
1238 // ICASSP 2012.
1239 void MleAmSgmmUpdater::ComputeMPrior(AmSgmm *model) {
1240  KALDI_ASSERT(update_options_.map_M_prior_iters > 0);
1241  int32 Ddim = model->FeatureDim();
1242  int32 Sdim = model->PhoneSpaceDim();
1243  int32 nGaussians = model->NumGauss();
1244 
1245  // inverse variance of the columns of M: dim is # of rows
1246  model->col_cov_inv_.Resize(Ddim);
1247  // inverse covariance of the rows of M: dim is # of columns
1248  model->row_cov_inv_.Resize(Sdim);
1249 
1250  model->col_cov_inv_.SetUnit();
1251  model->row_cov_inv_.SetUnit();
1252 
1253  if (model->M_prior_.size() == 0) {
1254  model->M_prior_.resize(nGaussians);
1255  for (int32 i = 0; i < nGaussians; i++) {
1256  model->M_prior_[i].Resize(Ddim, Sdim);
1257  model->M_prior_[i].CopyFromMat(model->M_[i]); // We initialize Mpri as this
1258  }
1259  }
1260 
1261  if (update_options_.full_col_cov || update_options_.full_row_cov) {
1262  Matrix<double> avg_M(Ddim, Sdim); // average of the Gaussian prior means
1263  for (int32 i = 0; i < nGaussians; i++)
1264  avg_M.AddMat(1.0, Matrix<double>(model->M_prior_[i]));
1265  avg_M.Scale(1.0 / nGaussians);
1266 
1267  Matrix<double> MDiff(Ddim, Sdim);
1268  for (int32 iter = 0; iter < update_options_.map_M_prior_iters; iter++) {
1269  { // diagnostic block.
1270  double prior_like = -0.5 * nGaussians * (Ddim * Sdim * Log(2 * M_PI)
1271  + Sdim * (-model->row_cov_inv_.LogPosDefDet())
1272  + Ddim * (-model->col_cov_inv_.LogPosDefDet()));
1273  for (int32 i = 0; i < nGaussians; i++) {
1274  MDiff.CopyFromMat(Matrix<double>(model->M_prior_[i]));
1275  MDiff.AddMat(-1.0, avg_M); // MDiff = M_{i} - avg(M)
1276  SpMatrix<double> tmp(Ddim);
1277  // tmp = MDiff.Omega_r^{-1}*MDiff^T.
1278  tmp.AddMat2Sp(1.0, MDiff, kNoTrans,
1279  SpMatrix<double>(model->row_cov_inv_), 0.0);
1280  prior_like -= 0.5 * TraceSpSp(tmp, SpMatrix<double>(model->col_cov_inv_));
1281  }
1282  KALDI_LOG << "Before iteration " << iter
1283  << " of updating prior over M, log like per dimension modeled is "
1284  << prior_like / (nGaussians * Ddim * Sdim);
1285  }
1286 
1287  // First estimate the column covariances (\Omega_r in paper)
1288  if (update_options_.full_col_cov) {
1289  size_t limited;
1290  model->col_cov_inv_.SetZero();
1291  for (int32 i = 0; i < nGaussians; i++) {
1292  MDiff.CopyFromMat(Matrix<double>(model->M_prior_[i]));
1293  MDiff.AddMat(-1.0, avg_M); // MDiff = M_{i} - avg(M)
1294  // Omega_r += 1/(D*I) * Mdiff * Omega_c^{-1} * Mdiff^T
1295  model->col_cov_inv_.AddMat2Sp(1.0 / (Ddim * nGaussians),
1296  Matrix<BaseFloat>(MDiff), kNoTrans,
1297  model->row_cov_inv_, 1.0);
1298  }
1299  model->col_cov_inv_.PrintEigs("col_cov");
1300  limited = model->col_cov_inv_.LimitCond(update_options_.max_cond,
1301  true /*invert the matrix*/);
1302  if (limited != 0) {
1303  KALDI_LOG << "Computing column covariances for M: limited " << limited
1304  << " singular values, max condition is "
1305  << update_options_.max_cond;
1306  }
1307  }
1308 
1309  // Now estimate the row covariances (\Omega_c in paper)
1310  if (update_options_.full_row_cov) {
1311  size_t limited;
1312  model->row_cov_inv_.SetZero();
1313  for (int32 i = 0; i < nGaussians; i++) {
1314  MDiff.CopyFromMat(Matrix<double>(model->M_prior_[i]));
1315  MDiff.AddMat(-1.0, avg_M); // MDiff = M_{i} - avg(M)
1316  // Omega_c += 1/(S*I) * Mdiff^T * Omega_r^{-1} * Mdiff.
1317  model->row_cov_inv_.AddMat2Sp(1.0 / (Sdim * nGaussians),
1318  Matrix<BaseFloat>(MDiff), kTrans,
1319  model->col_cov_inv_, 1.0);
1320  }
1321  model->row_cov_inv_.PrintEigs("row_cov");
1322  limited = model->row_cov_inv_.LimitCond(update_options_.max_cond,
1323  true /*invert the matrix*/);
1324  if (limited != 0) {
1325  KALDI_LOG << "Computing row covariances for M: limited " << limited
1326  << " singular values, max condition is "
1327  << update_options_.max_cond;
1328  }
1329  }
1330  } // end iterations
1331  }

1488  for (int32 j = 0; j < accs.num_states_; j++) tot_count += accs.gamma_[j].Sum();
1489 
1490  for (int iter = 0; iter < update_options_.weight_projections_iters; iter++) {
1491  F_i.SetZero();
1492  g_i.SetZero();
1493  double k_like_before = 0.0;
1494 
1495  UpdateWParallelClass c(accs, *model, w, &F_i, &g_i, &k_like_before);
1496  RunMultiThreaded(c);
1497 
1498  Matrix<double> w_orig(w);
1499  double k_predicted_like_impr = 0.0, k_like_after = 0.0;
1500  double min_step = 0.001, step_size;
1501  for (step_size = 1.0; step_size >= min_step; step_size /= 2) {
1502  k_predicted_like_impr = 0.0;

1504

1506  opts.name = "w";
1507  opts.K = update_options_.max_cond;
1508  opts.eps = update_options_.epsilon;

1513  // returns objf impr with step_size = 1,
1514  // but it may not be 1 so we recalculate it.

1521  0.5 * VecSpVec(delta_w, this_F_i, delta_w);

1524  // we checked inside SolveQuadraticProblem.

1526

1528  KALDI_LOG << "Predicted objf impr for w (not per frame), iter = " <<
1529  (iter) << ", i = " << (i) << " is " << (predicted_impr);
1530  }
1531  k_predicted_like_impr += predicted_impr;
1532  w.Row(i).AddVec(1.0, delta_w);
1533  }

1536  for (int32 m = 0; m < model->NumSubstates(j); m++) {
1537  w_jm_vec.AddMatVec(1.0, w, kNoTrans, Vector<double>(model->v_[j].Row(m)), 0.0);
1538  w_jm_vec.Add((-1.0) * w_jm_vec.LogSumExp());
1539  k_like_after += VecVec(w_jm_vec, accs.gamma_[j].Row(m));
1540  }
1541  }

1543  << "predicted per-frame like impr "

1546  << (tot_count) << " frames";

1558  } else {
1559  break;
1560  }
1561  }
1562  if (step_size < min_step) {
1563  // Undo any step as we have no confidence that this is right.
1564  w.CopyFromMat(w_orig);
1565  } else {
1566  tot_predicted_like_impr += k_predicted_like_impr;
1567  tot_like_after += k_like_after;
1568  tot_like_before += k_like_before;
1569  }
1570  }
1571 
1572  model->w_.CopyFromMat(w);
1573 
1574  tot_predicted_like_impr /= tot_count;
1575  tot_like_after = (tot_like_after - tot_like_before) / tot_count;
1576  KALDI_LOG << "**Overall objf impr for w is " << tot_predicted_like_impr
1577  << ", actual " << tot_like_after << ", over "
1578  << tot_count << " frames";

1582 double MleAmSgmmUpdater::UpdateWSequential(
1583  const MleAmSgmmAccs &accs, AmSgmm *model) {
1584  // Sequential version, in paper.
1585  /* This is the approach for the weight projections that
1586  * I originally implemented, in which we test the auxiliary function
1587  improvement for each i that we update. This requires some
1588  careful bookkeeping. It means that we need to store the
1589  total of the un-normalized weights for each j, m. */
1590 
1591  KALDI_LOG << "Updating weight projections [original approach, checking each"
1592  << "Gaussian component].";
1593 
1594  SpMatrix<double> v_vT(accs.phn_space_dim_);
1595  // tot_like_{after, before} are totals over multiple iterations,
1596  // not valid likelihoods...
1597  // but difference is valid (when divided by tot_count).
1598  double tot_delta_predicted = 0.0, tot_delta_observed = 0.0,
1599  tot_count = 0.0;
1600 
1601  Vector<double> w_jm(accs.num_gaussians_);
1602  Vector<double> g_i(accs.phn_space_dim_);
1603  SpMatrix<double> F_i(accs.phn_space_dim_);
1604 
1605  double k_count = 0.0;
1606  // Total count in each substate.
1607  std::vector< Vector<double> > gamma_jm(accs.num_states_);
1608  for (int32 j = 0; j < accs.num_states_; j++) { // Initialize gamma_jm
1609  gamma_jm[j].Resize(model->NumSubstates(j));
1610  for (int32 m = 0; m < model->NumSubstates(j); m++) {
1611  k_count += (gamma_jm[j](m) = accs.gamma_[j].Row(m).Sum());
1612  }
1613  }
1614 
1615  Matrix<double> w(model->w_);
1616 
1617  for (int iter = 0; iter < update_options_.weight_projections_iters; iter++) {
1618  double k_delta_predicted = 0.0, k_delta_observed = 0.0;
1619 
1620  // log total of un-normalized weights for each j, m
1621  std::vector< Vector<double> > weight_tots(accs.num_states_);
1622 
1623  // Initialize weight_tots
1624  for (int32 j = 0; j < accs.num_states_; j++) {
1625  weight_tots[j].Resize(model->NumSubstates(j));
1626  for (int32 m = 0; m < model->NumSubstates(j); m++) {
1627  w_jm.AddMatVec(1.0, w, kNoTrans, Vector<double>(model->v_[j].Row(m)), 0.0);
1628  weight_tots[j](m) = w_jm.LogSumExp();
1629  }
1630  }
1631 
1632  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1633  F_i.SetZero();
1634  g_i.SetZero();
1635  SubVector<double> w_i = w.Row(i);
1636 
1637  for (int32 j = 0; j < accs.num_states_; j++) {
1638  for (int32 m = 0; m < model->NumSubstates(j); m++) {
1639  double this_unnormalized_weight = VecVec(w_i, model->v_[j].Row(m));
1640  double normalizer = weight_tots[j](m);
1641  double this_log_w = this_unnormalized_weight - normalizer,
1642  this_w = Exp(this_log_w),
1643  substate_count = gamma_jm[j](m),
1644  this_count = accs.gamma_[j](m, i);
1645 
1646  double linear_term = this_count - substate_count * this_w;
1647  double quadratic_term = std::max(this_count, substate_count * this_w);
1648 
1649  g_i.AddVec(linear_term, model->v_[j].Row(m));
1650  // should not ever be zero, but check anyway.
1651  if (quadratic_term != 0.0)
1652  F_i.AddVec2(static_cast<BaseFloat>(quadratic_term), model->v_[j].Row(m));
1653  }
1654  }
1655 
1656  SolverOptions opts;
1657  opts.name = "w";
1658  opts.K = update_options_.max_cond;
1659  opts.eps = update_options_.epsilon;
1660 
1661  // auxf is formulated in terms of change in w.
1662  Vector<double> delta_w(accs.phn_space_dim_);
1663  // returns objf impr with step_size = 1,
1664  // but it may not be 1 so we recalculate it.
1665  SolveQuadraticProblem(F_i,
1666  g_i,
1667  opts,
1668  &delta_w);
1669 
1670  try { // In case we have a problem in LogSub.
1671  double step_size, min_step = 0.0001;
1672  for (step_size = 1.0; step_size >= min_step; step_size /= 2) {
1673  Vector<double> new_w_i(w_i);
1674  // copy it in case we do not commit this change.
1675  std::vector<Vector<double> > new_weight_tots(weight_tots);
1676  new_w_i.AddVec(step_size, delta_w);
1677  double predicted_impr = step_size * VecVec(delta_w, g_i) -
1678  0.5 * step_size * step_size * VecSpVec(delta_w, F_i, delta_w);
1679  if (predicted_impr < -0.1) {
1680  KALDI_WARN << "Negative predicted auxf improvement " <<
1681  (predicted_impr) << ", not updating this gaussian " <<
1682  "(either numerical problems or a code mistake.";
1683  break;
1684  }
1685  // Now compute observed objf change.
1686  double observed_impr = 0.0, this_tot_count = 0.0;
1687 
1688  for (int32 j = 0; j < accs.num_states_; j++) {
1689  for (int32 m = 0; m < model->NumSubstates(j); m++) {
1690  double old_unnorm_weight = VecVec(w_i, model->v_[j].Row(m)),
1691  new_unnorm_weight = VecVec(new_w_i, model->v_[j].Row(m)),
1692  substate_count = gamma_jm[j](m),
1693  this_count = accs.gamma_[j](m, i);
1694  this_tot_count += this_count;
1695  observed_impr += this_count * // from numerator.
1696  (new_unnorm_weight - old_unnorm_weight);
1697  double old_normalizer = new_weight_tots[j](m), delta;
1698  if (new_unnorm_weight > old_unnorm_weight) {
1699  delta = LogAdd(0, LogSub(new_unnorm_weight - old_normalizer,
1700  old_unnorm_weight - old_normalizer));
1701  } else {
1702  delta = LogSub(0, LogSub(old_unnorm_weight - old_normalizer,
1703  new_unnorm_weight - old_normalizer));
1704  // The if-statement above is equivalent to:
1705  // delta = LogAdd(LogSub(0,
1706  // old_unnorm_weight-old_normalizer),
1707  // new_unnorm_weight-old_normalizer)
1708  // but has better behaviour numerically.
1709  }
1710  observed_impr -= substate_count * delta;
1711  new_weight_tots[j](m) += delta;
1712  }
1713  }
1714  if (observed_impr < 0.0) { // failed, so we reduce step size.
1715  KALDI_LOG << "Updating weights, for i = " << (i) << ", predicted "
1716  "auxf: " << (predicted_impr/(this_tot_count + 1.0e-20))
1717  << ", observed " << observed_impr/(this_tot_count + 1.0e-20)
1718  << " over " << this_tot_count << " frames. Reducing step size "
1719  << "to " << (step_size/2);
1720  if (predicted_impr / (this_tot_count + 1.0e-20) < 1.0e-07) {
1721  KALDI_WARN << "Not updating this weight vector as auxf decreased"
1722  << " probably due to numerical issues (since small change).";
1723  break;
1724  }
1725  } else {
1726  if (i < 10)
1727  KALDI_LOG << "Updating weights, for i = " << (i)
1728  << ", auxf change per frame is" << ": predicted " <<
1729  (predicted_impr /(this_tot_count + 1.0e-20)) << ", observed "
1730  << (observed_impr / (this_tot_count + 1.0e-20))
1731  << " over " << (this_tot_count) << " frames.";
1732 
1733  k_delta_predicted += predicted_impr;
1734  k_delta_observed += observed_impr;
1735  w.Row(i).CopyFromVec(new_w_i);
1736  weight_tots = new_weight_tots; // Copy over normalizers.
1737  break;
1738  }
1739  }
1740  } catch(...) {
1741  KALDI_LOG << "Warning: weight update for i = " << i
1742  << " failed, possible numerical problem.";
1743  }
1744  }
1745  KALDI_LOG << "For iteration " << iter << ", updating w gives predicted "
1746  << "per-frame like impr " << (k_delta_predicted / k_count) <<
1747  ", observed " << (k_delta_observed / k_count) << ", over " << (k_count)
1748  << " frames";
1749  if (iter == 0) tot_count += k_count;
1750  tot_delta_predicted += k_delta_predicted;
1751  tot_delta_observed += k_delta_observed;
1752  }
1753 
1754  model->w_.CopyFromMat(w);
1755 
1756  tot_delta_observed /= tot_count;
1757  tot_delta_predicted /= tot_count;
1758  KALDI_LOG << "**Overall objf impr for w is " << tot_delta_predicted
1759  << ", observed " << tot_delta_observed << ", over "
1760  << tot_count << " frames";
1761  return tot_delta_observed;
1762 }

1811  const MleAmSgmmAccs &accs, AmSgmm *model) {

1813  Vector<double> gamma_i(accs.num_gaussians_);
1814  for (int32 j = 0; j < accs.num_states_; j++) {
1815  for (int32 m = 0; m < model->NumSubstates(j); m++) {
1816  gamma_i.AddVec(1.0, accs.gamma_[j].Row(m));
1817  }
1818  }

1820  if (tot_count == 0) {
1821  KALDI_WARN << "Not renormalizing N, since there are no counts.";
1822  return;
1823  }
1824 
1825  SpMatrix<double> RTot(accs.spk_space_dim_);
1826  // for (int32 i = 0; i < accs.num_gaussians_; i++) {

1828  // }

1830  RTot.AddSp(gamma_i(i), accs.R_[i]);
1831  }
1832  RTot.Scale(1.0 / tot_count);
1833  Matrix<double> U(accs.spk_space_dim_, accs.spk_space_dim_);
1834  Vector<double> eigs(accs.spk_space_dim_);

1840  }

1848  Matrix<double> Ntmp(accs.feature_dim_, accs.spk_space_dim_);
1849  for (int32 i = 0; i < accs.num_gaussians_; i++) {

1856 double MleAmSgmmUpdater::UpdateVars(const MleAmSgmmAccs &accs,
1857  AmSgmm *model) {
1858  KALDI_ASSERT(S_means_.size() == static_cast<size_t>(accs.num_gaussians_) &&
1859  "Must call PreComputeStats before updating the covariances.");
1860  SpMatrix<double> Sigma_i(accs.feature_dim_), Sigma_i_ml(accs.feature_dim_);
1861  double tot_objf_impr = 0.0, tot_t = 0.0;
1862  SpMatrix<double> covfloor(accs.feature_dim_);
1863  Vector<double> gamma_vec(accs.num_gaussians_);
1864  Vector<double> objf_improv(accs.num_gaussians_);
1865 
1866  // First pass over all (shared) Gaussian components to calculate the
1867  // ML estimate of the covariances, and the total covariance for flooring.
1868  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1869  double gamma_i = 0;
1870  for (int32 j = 0; j < accs.num_states_; j++)
1871  for (int32 m = 0, end = model->NumSubstates(j); m < end; m++)
1872  gamma_i += accs.gamma_[j](m, i);
1873 
1874  // Eq. (75): Sigma_{i}^{ml} = 1/gamma_{i} [S_{i} + S_{i}^{(means)} - ...
1875  // Y_{i} M_{i}^T - M_{i} Y_{i}^T]
1876  // Note the S_means_ already contains the Y_{i} M_{i}^T terms.
1877  Sigma_i_ml.CopyFromSp(S_means_[i]);
1878  Sigma_i_ml.AddSp(1.0, accs.S_[i]);
1879 
1880  gamma_vec(i) = gamma_i;
1881  covfloor.AddSp(1.0, Sigma_i_ml);
1882  // inverting small values e.g. 4.41745328e-40 seems to generate inf,
1883  // although would be fixed up later.
1884  if (gamma_i > 1.0e-20) {
1885  Sigma_i_ml.Scale(1 / (gamma_i + 1.0e-20));
1886  } else {
1887  Sigma_i_ml.SetUnit();
1888  }
1889  KALDI_ASSERT(1.0 / Sigma_i_ml(0, 0) != 0.0);
1890  // Eq. (76): Compute the objective function with the old parameter values
1891  objf_improv(i) = model->SigmaInv_[i].LogPosDefDet() -
1892  TraceSpSp(SpMatrix<double>(model->SigmaInv_[i]), Sigma_i_ml);
1893 
1894  model->SigmaInv_[i].CopyFromSp(Sigma_i_ml); // inverted in the next loop.
1895  }
1896 
1897  // Compute the covariance floor.
1898  if (gamma_vec.Sum() == 0) { // If no count, use identity.
1899  KALDI_WARN << "Updating variances: zero counts. Setting floor to unit.";
1900  covfloor.SetUnit();
1901  } else { // else, use the global average covariance.
1902  covfloor.Scale(update_options_.cov_floor / gamma_vec.Sum());
1903  int32 tmp;
1904  if ((tmp = covfloor.LimitCondDouble(update_options_.max_cond)) != 0) {
1905  KALDI_WARN << "Covariance flooring matrix is poorly conditioned. Fixed "
1906  << "up " << (tmp) << " eigenvalues.";
1907  }
1908  }
1909 
1910  if (update_options_.cov_diag_ratio > 1000) {
1911  KALDI_LOG << "Assuming you want to build a diagonal system since "
1912  << "cov_diag_ratio is large: making diagonal covFloor.";
1913  for (int32 i = 0; i < covfloor.NumRows(); i++)
1914  for (int32 j = 0; j < i; j++)
1915  covfloor(i, j) = 0.0;
1916  }
1917 
1918  // Second pass over all (shared) Gaussian components to calculate the
1919  // floored estimate of the covariances, and update the model.
1920  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1921  Sigma_i.CopyFromSp(model->SigmaInv_[i]);
1922  Sigma_i_ml.CopyFromSp(Sigma_i);
1923  // In case of insufficient counts, make the covariance matrix diagonal.
1924  // cov_diag_ratio is 2 by default, set to very large to always get diag-cov
1925  if (gamma_vec(i) < update_options_.cov_diag_ratio * accs.feature_dim_) {
1926  KALDI_WARN << "For Gaussian component " << i << ": Too low count "
1927  << gamma_vec(i) << " for covariance matrix estimation. Setting to "
1928  << "diagonal";
1929  for (int32 d = 0; d < accs.feature_dim_; d++)
1930  for (int32 e = 0; e < d; e++)
1931  Sigma_i(d, e) = 0.0; // SpMatrix, can only set lower traingular part
1932 
1933  int floored = Sigma_i.ApplyFloor(covfloor);
1934  if (floored > 0) {
1935  KALDI_WARN << "For Gaussian component " << i << ": Floored " << floored
1936  << " covariance eigenvalues.";
1937  }
1938  model->SigmaInv_[i].CopyFromSp(Sigma_i);
1939  model->SigmaInv_[i].InvertDouble();
1940  } else { // Updating the full covariance matrix.
1941  try {
1942  int floored = Sigma_i.ApplyFloor(covfloor);
1943  if (floored > 0) {
1944  KALDI_WARN << "For Gaussian component " << i << ": Floored "
1945  << floored << " covariance eigenvalues.";
1946  }
1947  model->SigmaInv_[i].CopyFromSp(Sigma_i);
1948  model->SigmaInv_[i].InvertDouble();
1949 
1950  objf_improv(i) += Sigma_i.LogPosDefDet() +
1951  TraceSpSp(SpMatrix<double>(model->SigmaInv_[i]), Sigma_i_ml);
1952  objf_improv(i) *= (-0.5 * gamma_vec(i)); // Eq. (76)
1953

2006  }

2052  const SgmmPerFrameDerivedVars &frame_vars,
2053  const Matrix<BaseFloat> &posteriors,
2054  int32 j) {

2056  int32 feature_dim = model.FeatureDim(),
2057  spk_space_dim = model.SpkSpaceDim();
2058  KALDI_ASSERT(spk_space_dim != 0);
2059  const vector<int32> &gselect = frame_vars.gselect;
2060

2062  Vector<double> xt_jmi(feature_dim), mu_jmi(feature_dim),
2063  zt_jmi(spk_space_dim);
2064  int32 num_substates = model.NumSubstates(j);
2065  for (int32 ki = 0; ki < static_cast<int32>(gselect.size()); ki++) {
2066  int32 i = gselect[ki];

2068  // Eq. (39): gamma_{jmi}(t) = p (j, m, i|t)
2069  BaseFloat gammat_jmi = RandPrune(posteriors(ki, m), rand_prune_);
2070  if (gammat_jmi != 0.0) {
2071  tot_count += gammat_jmi;

2073  xt_jmi.CopyFromVec(frame_vars.xt);
2074  xt_jmi.AddVec(-1.0, mu_jmi);
2075  // Eq. (48): z{jmi}(t) = N_{i}^{T} \Sigma_{i}^{-1} x_{jmi}(t)
2076  zt_jmi.AddMatVec(1.0, NtransSigmaInv_[i], kNoTrans, xt_jmi, 0.0);
2077  // Eq. (49): \gamma_{i}^{(s)} = \sum_{t\in\Tau(s), j, m} gamma_{jmi}
2078  gamma_s_(i) += gammat_jmi;

2080  y_s_.AddVec(gammat_jmi, zt_jmi);
2081  }
2082  }
2083  }
2084  return tot_count;
