33 // enum used to store various binary component properties.
34 // We give it a name ComponentProperties, but don't use this
35 // type for the bitmasks: instead use int32 for this type, e.g.
36 // int32 properties = kSimpleComponent|kBackpropNeedsOutput.
37 enum ComponentProperties {
38  kSimpleComponent = 0x001, // true if number of rows of input equals number of rows
39  // of output and this component doesn't care about the indexes
40  // (i.e. it maps each row of input to each row of output without
41  // regard to the index values). Will normally be true.
42  kUpdatableComponent = 0x002, // true if the component has parameters that can
43  // be updated. Components that return this flag
44  // must be dynamic_castable to type
45  // UpdatableComponent (but components of type
46  // UpdatableComponent do not have to return this
47  // flag, e.g. if this instance is not really
48  // updatable).
49  kLinearInInput = 0x004, // true if the component's output is always a
50  // linear function of its input, i.e. alpha times
51  // input gives you alpha times output.
52  kLinearInParameters = 0x008, // true if an updatable component's output is always a
53  // linear function of its parameters, i.e. alpha times
54  // parameters gives you alpha times output. This is true
55  // for all updatable components we envisage.
56  kPropagateInPlace = 0x010, // true if we can do the propagate operation in-place
57  // (input and output matrices are the same).
58  // Note: if doing backprop, you'd also need to check
59  // that the kBackpropNeedsInput property is not true.
60  kPropagateAdds = 0x020, // true if the Propagate function adds to, rather
61  // than setting, its output. The Component chooses
62  // whether to add or set, and the calling code has to
63  // accommodate it.
64  kReordersIndexes = 0x040, // true if the ReorderIndexes function might reorder
65  // the indexes (otherwise we can skip calling it).
66  // Must not be set for simple components.
67  kBackpropAdds = 0x080, // true if the Backprop function adds to, rather than
68  // setting, the "in_deriv" output. The Component chooses
69  // whether to add or set, and the calling code has to
70  // accommodate it.
71  kBackpropNeedsInput = 0x100, // true if backprop operation needs access to
72  // forward-pass input.
73  kBackpropNeedsOutput = 0x200, // true if backprop operation needs access to
74  // forward-pass output (e.g. true for Sigmoid).
75  kBackpropInPlace = 0x400, // true if we can do the backprop operation in-place
76  // (input and output matrices may be the same).
77  kStoresStats = 0x800 // true if the StoreStats operation stores
78  // statistics e.g. on average node activations and
79  // derivatives of the nonlinearity, (as it does for
80  // Tanh, Sigmoid, ReLU and Softmax).
81 };

84 // This is a base class for a helper-class of class Component, which is used to
85 // store any pre-computed indexes it needs for its forward and backward
86 // computations. For components which are not "Simple" components (i.e. the
87 // kSimpleComponent property is false), and which may therefore "care" about
