33 void Compiler::CreateComputation(const CompilerOptions &opts,
34  NnetComputation *computation) {
35  computation->Clear();
36  ComputationGraphBuilder builder(nnet_, request_, &graph_);
37  builder.Compute();
38  if (!builder.AllOutputsAreComputable()) {
39  builder.ExplainWhyAllOutputsNotComputable(); // prints logging info
40  KALDI_ERR << "Not all outputs were computable, cannot create computation.";
41  }
42  builder.Prune();
43  // see function declaration's comment for meaning of "phases".
44  std::vector<std::vector<int32> > phases;
45  ComputeComputationPhases(nnet_, graph_, &phases);
46  std::vector<std::vector<int32> > steps;
47  ComputeComputationSteps(nnet_, request_, phases, &graph_, &steps);
48  phases.clear();
49  CreateLocationInfo(steps);
50  std::vector<bool> deriv_needed;
51  ComputeDerivNeeded(steps, &deriv_needed);
52  CreateStepInfo(deriv_needed, &steps, computation);
53  AddCommands(deriv_needed, computation);
54  if (opts.output_debug_info)
55  OutputDebugInfo(computation);
56 }

58 void Compiler::AddCommands(const std::vector<bool> &deriv_needed,
59  NnetComputation *computation) {
60  SetInputOutputInfo(computation);
61  computation->need_model_derivative = request_.need_model_derivative;
62  int32 arbitrary_factor = 8;
63  computation->commands.reserve(computation->matrices.size()
64  * arbitrary_factor);
65  AllocateMatrices(computation);
66  SetUpPrecomputedIndexes(computation);
67  int32 num_steps = steps_.size();
68  for (int32 step = 0; step < num_steps; step++)
69  DoForwardComputation(step, computation);
70  // mark the end of the forward phase.
71  computation->commands.push_back(
72  NnetComputation::Command(kNoOperationMarker));
73  for (int32 step = num_steps - 1; step >= 0; step--)
74  if (deriv_needed[step])
75  DoBackwardComputation(step, computation);
76  DeallocateMatrices(computation);
77 }

82  unordered_set<int32> *dep_steps) {
83  dep_steps->clear();
84  std::vector<int32>::const_iterator step_iter = this_step.begin(),
85  step_end = this_step.end();

102 void Compiler::ComputeDerivNeeded(
103  const std::vector<std::vector<int32> > &steps,
104  std::vector<bool> *deriv_needed) {
105  deriv_needed->clear();
106  int32 num_steps = steps.size();
107  deriv_needed->resize(num_steps, false);
108 
109  for (int32 step = 0; step < num_steps; step++) {
110  const std::vector<int32> &this_step = steps[step];
111  if (this_step.empty()) // empty steps are theoretically possible, e.g.
112  continue; // if a non-simple Component requires no input.
113  int32 cindex_id = this_step[0];
114  int32 node_index = graph_.cindexes[cindex_id].first;
115  bool is_input = graph_.is_input[cindex_id];
116 
117  std::string node_name = nnet_.GetNodeNames()[node_index];
118  unordered_set<int32> input_steps;
119  ComputeStepDependencies(this_step, &input_steps);
120 
121  unordered_set<int32>::iterator iter = input_steps.begin(),
122  end = input_steps.end();
123  // if some step that we depends on needs a derivative, we need the derivative.
124  for (; iter != end; ++iter) {
125  int32 dep_step = *iter;
126  KALDI_ASSERT(dep_step < step);
127  if ((*deriv_needed)[dep_step])
128  (*deriv_needed)[step] = true;
129  }
130  // if this step is an input and the user requested the derivative w.r.t. that
131  // input, we need the derivative.
132  if (is_input) {
133  int32 input_index = request_.IndexForInput(node_name);
134  KALDI_ASSERT(input_index != -1);
135  if (request_.inputs[input_index].has_deriv)
136  (*deriv_needed)[step] = true;
137  }
138  // if this step is an output and the user is providing the derivative w.r.t. that
139  // output, we need a place to store the derivative, so we set (*deriv_needed) to
140  // true.
141  if (nnet_.IsOutputNode(node_index)) {
142  int32 output_index = request_.IndexForOutput(node_name);
143  KALDI_ASSERT(output_index != -1);
144  if (request_.outputs[output_index].has_deriv)
145  (*deriv_needed)[step] = true;
146  }
147  // If this is an updatable Component node and the user requested model
148  // derivatives (e.g. during training), we need this step's derivative.
149  if (nnet_.IsComponentNode(node_index) && request_.need_model_derivative) {
150  const NetworkNode &node = nnet_.GetNode(node_index);
151  const Component *c = nnet_.GetComponent(node.u.component_index);
152  if (c->Properties() & kUpdatableComponent)
153  (*deriv_needed)[step] = true;
154  }
155  }
156  if (GetVerboseLevel() >= 5) {
157  std::ostringstream os;
158  os << "deriv_needed = ";
159  for (int32 i = 0; i < deriv_needed->size(); i++)
160  os << ((*deriv_needed)[i] ? "t" : "f");
161  os << "\n";
162  KALDI_VLOG(5) << os.str();
163  }
164 }

169 void Compiler::CreateStepInfo(
170  const std::vector<bool> &deriv_needed,
171  std::vector<std::vector<int32> > *by_step,
172  NnetComputation *computation) {
173  KALDI_ASSERT(!by_step->empty());
174  int32 num_steps = by_step->size();
175  steps_.resize(num_steps);
176  for (int32 step = 0; step < num_steps; step++) {
177  StepInfo &this_info = steps_[step];
178  this_info.output_cindex_ids.swap((*by_step)[step]);
179  int32 num_ids = this_info.output_cindex_ids.size();
180  this_info.output_indexes.resize(num_ids);
181  for (int32 row_index = 0; row_index < num_ids; row_index++)
182  this_info.output_indexes[row_index] =
183  graph_.cindexes[this_info.output_cindex_ids[row_index]].second;
184  KALDI_ASSERT(num_ids > 0);
185  // node id's of all Cindexes are the same, so just use first one.
186  this_info.node_index =
187  graph_.cindexes[this_info.output_cindex_ids.front()].first;
188  const NetworkNode &node = nnet_.GetNode(this_info.node_index);
189  int32 num_rows = num_ids, num_cols = node.Dim(nnet_);
190 
191  if (node.node_type != kDimRange) {
192  this_info.value = computation->NewMatrix(num_rows, num_cols);
193  if (deriv_needed[step])
194  this_info.deriv = computation->NewMatrix(num_rows, num_cols);
195  if (node.node_type == kComponent)
196  KALDI_PARANOID_ASSERT(step > 0 && steps_[step-1].output_indexes ==
197  this_info.output_indexes);
198  } else {
199  // kDimRange. Will just be a sub-matrix of a Component or Input node.
200  int32 cindex_id = this_info.output_cindex_ids.front(),
201  input_cindex_id = graph_.dependencies[cindex_id][0],
202  input_step = cindex_id_to_location_[input_cindex_id].first;
203  KALDI_ASSERT(input_step != -1 && input_step < step);
204  KALDI_PARANOID_ASSERT(this_info.output_indexes ==
205  steps_[input_step].output_indexes);
206  this_info.value = computation->NewSubMatrix(steps_[input_step].value,
207  0, -1,
208  node.dim_offset, node.dim);
209  if (deriv_needed[step])
210  this_info.deriv = computation->NewSubMatrix(steps_[input_step].deriv,
211  0, -1,
212  node.dim_offset, node.dim);
213  }
214  if (node.node_type == kDescriptor) {
215  // we have a couple of things to do: set up input_locations_list which
216  // says where we copy the data from, and also set up value_parts and
217  // possibly deriv_parts.
218  const Descriptor &desc = node.descriptor;
219  int32 num_parts = desc.NumParts();
220  KALDI_ASSERT(num_parts > 0);
221  // set up input_locations_list.
222  this_info.input_locations_list.resize(num_parts);
223  for (int32 part = 0; part < num_parts; part++)
224  ComputeInputLocationsList(step, part,
225  &(this_info.input_locations_list[part]));
226  // set up value_parts and deriv_parts.
227  if (num_parts == 1) {
228  this_info.value_parts.push_back(this_info.value);
229  if (deriv_needed[step])
230  this_info.deriv_parts.push_back(this_info.deriv);
231  } else { // num_parts > 1.
232  int32 cur_dim_offset = 0;
233  // Have multiple parts, so need to set up sub-matrices.
234  this_info.value_parts.resize(num_parts);
235  if (deriv_needed[step])
236  this_info.deriv_parts.resize(num_parts);
237  for (int32 p = 0; p < num_parts; p++) {
238  const SumDescriptor &this_part = desc.Part(p);
239  int32 this_dim = this_part.Dim(nnet_);
240  this_info.value_parts[p] =
241  computation->NewSubMatrix(this_info.value,
242  0, -1,
243  cur_dim_offset, this_dim);
244  if (deriv_needed[step])
245  this_info.deriv_parts[p] =
246  computation->NewSubMatrix(this_info.deriv,
247  0, -1,
248  cur_dim_offset, this_dim);
249  cur_dim_offset += this_dim;
250  }
251  KALDI_ASSERT(cur_dim_offset == desc.Dim(nnet_));
252  }
253  }
254  }

302 void Compiler::DoForwardComputation(int32 step,
303  NnetComputation *computation) const {
304  KALDI_ASSERT(step < static_cast<int32>(steps_.size()));
305  const StepInfo &step_info = steps_[step];
306  const NetworkNode &node = nnet_.GetNode(step_info.node_index);
307  switch (node.node_type) {
308  case kInput: case kDimRange: break; // Nothing to do.
309  case kComponent:
310  AddPropagateStep(step, computation);
311  break;
312  case kDescriptor:
313  DoForwardComputationDescriptor(step, computation);
314  break;
315  default:
316  KALDI_ERR << "Invalid node type";
317  }

329 // The output vector "locations" is indexed first by output row-index i
330 // (i.e. the index of output_indexes or output_cindex_ids), and then is a list
331 // of input locations for that row-index, sorted in the natural order of
332 // Cindexes (but not necessarily unique). The semantics is that the i'th row of
333 // the output becomes a sum over the rows in the i'th list (or zero if that list
334 // is empty). These locations will be pairs [step-index, row-index].
335 void Compiler::ComputeInputLocationsList(
336  int32 step, int32 part_index,
337  std::vector<std::vector<std::pair<int32, int32> > > *submat_locations_list)
338  const {
339 
340  KALDI_ASSERT(static_cast<size_t>(step) < steps_.size());
341  const StepInfo &step_info = steps_[step];
342  const std::vector<Index> &output_indexes = step_info.output_indexes;
343  const NetworkNode &node = nnet_.GetNode(step_info.node_index);
344  const SumDescriptor &descriptor = node.descriptor.Part(part_index);
345 
346  int32 num_indexes = output_indexes.size();
347  submat_locations_list->clear();
348  submat_locations_list->resize(num_indexes);
349 
350  for (int32 i = 0; i < num_indexes; i++) {
351  std::vector<int32> input_cindex_ids;
352  const Index &index = output_indexes[i];
353  std::vector<Cindex> input_cindexes;
354  CindexSet cindex_set(graph_);
355  bool ans = descriptor.IsComputable(index, cindex_set, &input_cindexes);
356  // earlier compilation stages should have checked that it is computable,
357  // and the graph should still contain required inputs.
358  KALDI_ASSERT(ans);
359  std::sort(input_cindexes.begin(), input_cindexes.end());
360  int32 size = input_cindexes.size();
361  input_cindex_ids.resize(size);
362  for (int32 j = 0; j < size; j++) {
363  int32 c = graph_.GetCindexId(input_cindexes[j]);
364  KALDI_ASSERT(c != -1);
365  input_cindex_ids[j] = c;
366  }
367  std::vector<std::pair<int32, int32> > &this_locations_list =
368  (*submat_locations_list)[i];
369  this_locations_list.resize(size);
370  for (int32 j = 0; j < size; j++)
371  this_locations_list[j] = cindex_id_to_location_[input_cindex_ids[j]];
372  }
373 }

441 void Compiler::DoForwardComputationFromIndexes(
442  int32 value_submatrix_index,
443  int32 input_submatrix_index,
444  bool is_first_term_in_sum,
445  const std::vector<int32> &indexes,
446  NnetComputation *computation) const {
447 
448  int32 input_num_rows =
449  computation->submatrices[input_submatrix_index].num_rows,
450  num_rows = indexes.size();
451  if (input_num_rows == num_rows) {
452  int32 i;
453  for (i = 0; i < num_rows; i++)
454  if (indexes[i] != i)
455  break;
456  if (i == num_rows) { // Simplest case: just matrix addition.
457  CommandType ctype =
458  (is_first_term_in_sum ?
459  kMatrixCopy : kMatrixAdd);
460  computation->commands.push_back(
461  NnetComputation::Command(ctype, value_submatrix_index,
462  input_submatrix_index));
463 
464  return;
465  }
466  }
467  // if we got to here, it's not just a case of matrix-copy or matrix-add,
468  // but it's still from a single source matrix.
469  // TODO: detect the case where the indexes are contiguous, but possibly
470  // with -1's at the beginning or end (e.g. [ -1 2 3 4 5 6 7 8 ]) and make
471  // it a standard matrix-copy command with new sub-matrices added as needed,
472  // possibly with a subset of the rows in the original sub-matrices.
473  int32 indexes_index = computation->indexes.size();
474  computation->indexes.push_back(indexes);
475  CommandType ctype =
476  (is_first_term_in_sum ?
477  kCopyRows : kAddRows);
478  computation->commands.push_back(
479  NnetComputation::Command(ctype, value_submatrix_index,
480  input_submatrix_index, indexes_index));
481  return;
482 }

563 void Compiler::DoBackwardComputationFromSubmatLocations(
564  int32 deriv_submatrix_index,
565  const std::vector<std::pair<int32, int32> > &submat_locations,
566  NnetComputation *computation) const {
567  // This function creates a command to handle an individual piece of the
568  // Descriptor, for backprop. Note: because the backprop case is a little
569  // trickier to implement efficiently on the GPU, there may be cases
570  // which we will refuse to implement backprop for if we get here.
571 
572  int32 num_rows = submat_locations.size();
573  std::vector<std::pair<int32, int32> >::const_iterator
574  iter = submat_locations.begin(), end = submat_locations.end();
575  int32 first_submat = iter->first;
576  for (++iter; iter != end; ++iter)
577  if (iter->first != first_submat)
578  break;
579  bool all_same_submatrix = (iter == end);
580  if (all_same_submatrix) {
581  int32 input_deriv_submatrix_index = first_submat;
582  std::vector<int32> indexes(num_rows);
583  for (int32 i = 0; i < num_rows; i++)
584  indexes[i] = submat_locations[i].second;
585  DoBackwardComputationFromIndexes(deriv_submatrix_index,
586  input_deriv_submatrix_index,
587  indexes,
588  computation);
589  return;
590  } else {
591  // There are multiple source matrices.
592  std::vector<std::pair<int32, int32> > submat_locations_sorted;
593  std::sort(submat_locations_sorted.begin(), submat_locations_sorted.end());
594  if (IsSortedAndUniq(submat_locations_sorted)) {
595  // There are no repeats in any of the submat locations. This means that
596  // we can just use kAddToRowsMulti (i.e. AddToRows with pointer
597  // destination). If there were repeats, the CUDA kernel would require
598  // special synchronization so we don't allow it.
599  int32 indexes_multi_index = computation->indexes_multi.size();
600  computation->indexes_multi.push_back(submat_locations);
601  computation->commands.push_back(
602  NnetComputation::Command(kAddToRowsMulti,
603  deriv_submatrix_index,
604  indexes_multi_index));
605  return;
606  }
607  // If you reach this point, there is a case that wasn't handled. Our
608  // intended strategy to handle it, if it's ever needed, is to create a
609  // temporary matrix consisting of all the unique submat_locations in the
610  // input. We would first recurse to DoBackwardComputationFromIndexes, and
611  // let it write to this temporary matrix; and then do the kAddToRowsMulti
612  // command as above to go from the temporary matrix to the multiple
613  // matrices.
614  KALDI_ERR << "This case not handled.";
615  }
616 }

618 void Compiler::DoBackwardComputationFromIndexes(
619  int32 deriv_submatrix_index,
620  int32 input_deriv_submatrix_index,
621  const std::vector<int32> &indexes,
622  NnetComputation *computation) const {
623 
624  int32 num_rows = computation->submatrices[deriv_submatrix_index].num_rows,
625  input_num_rows =
626  computation->submatrices[input_deriv_submatrix_index].num_rows;
627  KALDI_ASSERT(indexes.size() == num_rows);
628  if (input_num_rows == num_rows) {
629  int32 i;
630  for (i = 0; i < num_rows; i++)
631  if (indexes[i] != i)
632  break;
633  if (i == num_rows) { // Simplest case: just matrix addition.
634  computation->commands.push_back(
635  NnetComputation::Command(kMatrixAdd,
636  input_deriv_submatrix_index,
637  deriv_submatrix_index));
638 
639  return;
640  }
641  }
642  if (input_num_rows >= num_rows) {
643  // If there are no repeated elements in the "indexes" array, we can reverse
644  // the mapping and make it an operation of type kAddRows. TODO: change this
645  // to use kAddToRows, kCopyToRows, when implemented (will be more
646  // efficient).
647  std::vector<int32> reverse_indexes(input_num_rows, -1);
648  int32 i;
649  for (i = 0; i < num_rows; i++) {
650  int32 index_i = indexes[i];
651  KALDI_ASSERT(index_i >= 0 && index_i < input_num_rows);
652  if (reverse_indexes[index_i] == -1)
653  reverse_indexes[index_i] = i;
654  else
655  break;
656  }
657  if (i == num_rows) {
658  // There were no repeated elements, and this strategy will work.
659  int32 indexes_index = computation->indexes.size();
660  computation->indexes.push_back(reverse_indexes);
661  computation->commands.push_back(
662  NnetComputation::Command(kAddRows,
663  input_deriv_submatrix_index,
664  deriv_submatrix_index,
665  indexes_index));
666  return;
667  }
668  }
669  std::vector<std::pair<int32, int32> > ranges;
670  if (HasContiguousProperty(indexes, &ranges)) {
671  // the operation can be set up as AddRowRanges.
672  if (static_cast<int32>(ranges.size()) != input_num_rows) {
673  KALDI_ASSERT(static_cast<int32>(ranges.size()) < input_num_rows);
674  // extend with (-1, -1) pairs.
675  ranges.resize(input_num_rows, std::pair<int32,int32>(-1, -1));
676  }
677  int32 indexes_ranges_index = computation->indexes_ranges.size();
678  computation->indexes_ranges.push_back(ranges);
679  computation->commands.push_back(
680  NnetComputation::Command(kAddRowRanges,
681  input_deriv_submatrix_index,
682  deriv_submatrix_index,
683  indexes_ranges_index));
684  // TODO: if any of these ranges are quite long (summing over many rows), the
685  // resulting code could be inefficient because the AddRowRanges kernels
686  // takes time linear in the length of the range. Using a temporary matrix
687  // with an intermediate size would make this more efficient in that case, so
688  // the one command would be two commands (plus commands to set up and
689  // destroy the temporary matrix).
690  return;
691  }
692 
693  // If you ever reach here, it means someone has used a type of network that we
694  // don't yet support in the backprop. Basically this case can be handled by
695  // creating a temporary matrix to reorder the matrix at deriv_submatrix_index,
696  // (using CopyRows), and doing AddRowRanges from that.
697  // It wouldn't be too much work.
698  KALDI_ERR << "This case not implemented yet.";
699 }

809 void Compiler::AllocateMatrices(NnetComputation *computation) const {
810  KALDI_ASSERT(computation->commands.empty());
811  // Work out which matrices are inputs to the computation (or output-derivs,
812  // which are also supplied as inputs to the computation); we won't be setting
813  // these up.
814  unordered_set<int32> input_and_oderiv_matrices;
815  int32 num_steps = steps_.size();
816  for (int32 step = 0; step < num_steps; step++) {
817  const StepInfo &this_info = steps_[step];
818  if (this_info.output_cindex_ids.empty())
819  continue;
820  int32 first_cindex_id = this_info.output_cindex_ids.front(),
821  node_index = this_info.node_index;
822  bool is_input = graph_.is_input[first_cindex_id],
823  is_output = nnet_.IsOutputNode(node_index);
824  if (is_input) {
825  int32 value_submatrix_index = this_info.value,
826  value_matrix_index =
827  computation->submatrices[value_submatrix_index].matrix_index;
828  input_and_oderiv_matrices.insert(value_matrix_index);
829  }
830  if (is_output && this_info.deriv != 0) {
831  int32 deriv_submatrix_index = this_info.deriv,
832  deriv_matrix_index =
833  computation->submatrices[deriv_submatrix_index].matrix_index;
834  input_and_oderiv_matrices.insert(deriv_matrix_index);
835  }
836  }
837 
838  for (int32 m = 1; m < computation->matrices.size(); m++) {
839  // Later in the optimization phase, it turns out that zeroing is not
840  // necessary for some matrices, we'll turn these commands into
841  // kAllocMatrixUndefined.
842  // We don't set up the matrices that are inputs to the computation;
843  // this happens when the user provides the input.
844  if (input_and_oderiv_matrices.count(m) == 0) {
845  NnetComputation::Command c(kAllocMatrixZeroed, m);
846  computation->commands.push_back(c);
847  }
848  }

891 void Compiler::DeallocateMatrices(NnetComputation *computation) {
892  // This adds the commands to destroy all the matrices- but not the
893  // ones that might be needed as outputs of the computation. The ones that
894  // are spared from destruction are those corresponding to outputs of the
895  // computation, and those corresponding to input derivatives that were
896  // requested by the user.
897  int32 num_matrices = computation->matrices.size();
898  std::vector<bool> will_destroy(num_matrices, true);
899 
900  int32 num_steps = steps_.size();
901  for (int32 step = 0; step < num_steps; step++) {
902  const StepInfo &step_info = steps_[step];
903  if (nnet_.IsOutputNode(step_info.node_index)) {
904  // steps corresponding to output nodes need to have their "value" kept.
905  int32 value_matrix_index =
906  computation->submatrices[step_info.value].matrix_index;
907  will_destroy[value_matrix_index] = false;
908  } else if (nnet_.IsInputNode(step_info.node_index)) {
909  // steps corresponding to input nodes need to have their "deriv" kept, but
910  // only if the corresponding input derivative was requested. (we don't
911  // need to worry about whether outputs were requested, because if they
912  // were not requested we would not be computing them in the first place).
913  std::string input_name = nnet_.GetNodeNames()[step_info.node_index];
914  int32 i = 0, num_inputs = request_.inputs.size();
915  bool has_deriv = false;
916  for (; i < num_inputs; i++) {
917  if (input_name == request_.inputs[i].name) {
918  has_deriv = request_.inputs[i].has_deriv;
919  break;
920  }
921  }
922  KALDI_ASSERT(i != num_inputs); // assert we found an input-request with
923  // this name
924  if (has_deriv) {
925  int32 deriv_matrix_index =
926  computation->submatrices[step_info.deriv].matrix_index;
927  will_destroy[deriv_matrix_index] = false;
928  }
929  }
930  }
931  // note: matrix-index 0 is the empty matrix.
932  for (int32 m = 1; m < num_matrices; m++)
933  if (will_destroy[m])
934  computation->commands.push_back(
935  NnetComputation::Command(kDeallocMatrix, m));
936 }
