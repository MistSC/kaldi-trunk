25 ForwardingDescriptor *GenRandForwardingDescriptor(int32 num_nodes) {
26  if (Rand() % 2 != 0) {
27  return new SimpleForwardingDescriptor(Rand() % num_nodes);
28  } else {
29  int32 r = Rand() % 4;
30  if (r == 0) {
31  Index offset;
32  offset.t = Rand() % 5;
33  offset.x = Rand() % 2;
34  return
35  new OffsetForwardingDescriptor(GenRandForwardingDescriptor(num_nodes),
36  offset);
37  } else if (r == 1) {
38  std::vector<ForwardingDescriptor*> vec;
39  int32 n = 1 + Rand() % 3;
40  for (int32 i = 0; i < n; i++)
41  vec.push_back(GenRandForwardingDescriptor(num_nodes));
42  return new SwitchingForwardingDescriptor(vec);
43  } else if (r == 2) {
44  return new RoundingForwardingDescriptor(
45  GenRandForwardingDescriptor(num_nodes), 1 + Rand() % 4);
46  } else {
47  return new ReplaceIndexForwardingDescriptor(
48  GenRandForwardingDescriptor(num_nodes),
49  (Rand() % 2 == 0 ? ReplaceIndexForwardingDescriptor::kT :
50  ReplaceIndexForwardingDescriptor::kX),
51  -2 + Rand() % 4);
52  }
53  }
54 }

56 // generates a random descriptor.
57 SumDescriptor *GenRandSumDescriptor(
58  int32 num_nodes) {
59  if (Rand() % 3 != 0) {
60  bool not_required = (Rand() % 5 == 0);
61  if (not_required)
62  return new OptionalSumDescriptor(GenRandSumDescriptor(num_nodes));
63  else
64  return new SimpleSumDescriptor(GenRandForwardingDescriptor(num_nodes));
65  } else {
66  return new BinarySumDescriptor((Rand() % 2 == 0 ? BinarySumDescriptor::kSum:
67  BinarySumDescriptor::kFailover),
68  GenRandSumDescriptor(num_nodes),
69  GenRandSumDescriptor(num_nodes));
70  }

178 // normalizes the text form of a descriptor.
179 std::string NormalizeTextDescriptor(const std::vector<std::string> &node_names,
180  const std::string &desc_str) {
181  std::vector<std::string> tokens;
182  DescriptorTokenize(desc_str, &tokens);
183  tokens.push_back("end of input");
184  const std::string *next_token = &(tokens[0]);
185  GeneralDescriptor *gen_desc = GeneralDescriptor::Parse(node_names,
186  &next_token);
187  if (*next_token != "end of input")
188  KALDI_ERR << "Parsing Descriptor, expected end of input but got "
189  << "'" << *next_token << "'";
190  Descriptor *desc = gen_desc->ConvertToDescriptor();
191  std::ostringstream ostr;
192  desc->WriteConfig(ostr, node_names);
193  delete gen_desc;
194  delete desc;
195  KALDI_LOG << "Result of normalizing " << desc_str << " is: " << ostr.str();
196  return ostr.str();
197 }
