427 // This class, used in multi-core implementation of the updates of the "w_i"
428 // quantities, was previously in estimate-am-sgmm.cc, but is being moved to the
429 // header so it can be used in estimate-am-sgmm-ebw.cc. It is responsible for
430 // computing, in parallel, the F_i and g_i quantities used in the updates of
431 // w_i.
432 class UpdateWClass: public MultiThreadable {
433  public:
434  UpdateWClass(const MleAmSgmm2Accs &accs,
435  const AmSgmm2 &model,
436  const Matrix<double> &w,
437  const std::vector<Matrix<double> > &log_a,
438  Matrix<double> *F_i,
439  Matrix<double> *g_i,
440  double *tot_like):
441  accs_(accs), model_(model), w_(w), log_a_(log_a),
442  F_i_ptr_(F_i), g_i_ptr_(g_i), tot_like_ptr_(tot_like) {
443  tot_like_ = 0.0;
444  F_i_.Resize(F_i->NumRows(), F_i->NumCols());
445  g_i_.Resize(g_i->NumRows(), g_i->NumCols());
446  }
447 
448  ~UpdateWClass() {
449  F_i_ptr_->AddMat(1.0, F_i_, kNoTrans);
450  g_i_ptr_->AddMat(1.0, g_i_, kNoTrans);
451  *tot_like_ptr_ += tot_like_;
452  }
453 
454  inline void operator() () {
455  // Note: give them local copy of the sums we're computing,
456  // which will be propagated to the total sums in the destructor.
457  MleAmSgmm2Updater::UpdateWGetStats(accs_, model_, w_, log_a_,
458  &F_i_, &g_i_, &tot_like_,
459  num_threads_, thread_id_);
460  }
461  private:
462  const MleAmSgmm2Accs &accs_;
463  const AmSgmm2 &model_;
464  const Matrix<double> &w_;
465  const std::vector<Matrix<double> > &log_a_;
466  Matrix<double> *F_i_ptr_;
467  Matrix<double> *g_i_ptr_;
468  Matrix<double> F_i_;
469  Matrix<double> g_i_;
470  double *tot_like_ptr_;
471  double tot_like_;
472 };
