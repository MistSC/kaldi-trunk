47 template<typename Real>
48 void CuMatrix<Real>::Resize(MatrixIndexT rows, MatrixIndexT cols,
49  MatrixResizeType resize_type) {
50  // This code does not currently support the other resize_type options.
51  KALDI_ASSERT(resize_type == kSetZero || resize_type == kUndefined);
52  if (rows * cols == 0) KALDI_ASSERT(rows == 0 && cols == 0);
53  if (this->num_rows_ == rows && this->num_cols_ == cols) {
54  if (resize_type == kSetZero) this->SetZero();
55  return;
56  }
57 
58  if (this->num_rows_ != 0)
59  this->Destroy();
60  if (rows == 0) return;
61 #if HAVE_CUDA == 1
62  if (CuDevice::Instantiate().Enabled()) {
63  Timer tim;
64  MatrixIndexT row_bytes = cols * sizeof(Real);
65  size_t pitch;
66  this->data_ = static_cast<Real*>(CuDevice::Instantiate().MallocPitch(
67  row_bytes, rows, &pitch));
68  this->num_rows_ = rows;
69  this->num_cols_ = cols;
70  this->stride_ = pitch / sizeof(Real);
71  if (resize_type == kSetZero) this->SetZero();
72  CuDevice::Instantiate().AccuProfile("CuMatrix::Resize", tim.Elapsed());
73  } else
74 #endif
75  { // Let the initializer of Matrix<Real> handle the allocation,
76  // and then just do Swap which will switch the pointers.
77  // This wastes a few instructions but is simple to code.
78  Matrix<Real> mat(rows, cols, resize_type);
79  this->Swap(&mat);
80  }
81 }

122 template<typename Real>
123 void CuMatrix<Real>::Swap(Matrix<Real> *mat) {
124 #if HAVE_CUDA == 1
125  if (CuDevice::Instantiate().Enabled()) {
126  if (this->num_rows_ == 0) {
127  if (mat->num_rows_ != 0) {
128  // *this is empty, but mat is nonempty.
129  this->Resize(mat->num_rows_, mat->num_cols_, kUndefined);
130  this->CopyFromMat(*mat);
131  mat->Resize(0, 0);
132  }
133  // else both are empty.
134  } else { // *this is nonempty.
135  if (mat->num_rows_ != 0) {
136  // Both *this and *mat are nonempty. Recurse to simpler cases.
137  // this could be done more efficiently in the case where
138  // the size does not change.
139  Matrix<Real> temp;
140  this->Swap(&temp); // now temp is full, *this is empty.
141  mat->Swap(&temp); // now mat has data from *this, temp has
142  // data from mat.
143  this->Swap(&temp); // copy data in mat to *this, which is now empty.
144  } else { // *this is full but *mat is empty.
145  mat->Resize(this->num_rows_, this->num_cols_, kUndefined);
146  this->CopyToMat(mat);
147  this->Destroy();
148  }
149  }
150  } else
151 #endif
152  {
153  std::swap(mat->data_, this->data_);
154  std::swap(mat->num_cols_, this->num_cols_);
155  std::swap(mat->num_rows_, this->num_rows_);
156  std::swap(mat->stride_, this->stride_);
157  }

160 template <typename Real>
161 void CuMatrixBase<Real>::CopyFromBlock(const CuBlockMatrix<Real> &B,
162  MatrixTransposeType trans) {
163  this->SetZero();
164  if (trans == kNoTrans) {
165  KALDI_ASSERT(NumRows() == B.NumRows() && NumCols() == B.NumCols());
166  int32 row_offset = 0, col_offset = 0;
167  for (int32 b = 0; b < B.NumBlocks(); b++) {
168  const CuMatrixBase<Real> &block = B.Block(b);
169  int32 num_rows = block.NumRows(), num_cols = block.NumCols();
170  CuSubMatrix<Real> this_block(*this, row_offset, num_rows,
171  col_offset, num_cols);
172  this_block.CopyFromMat(block);
173  row_offset += num_rows;
174  col_offset += num_cols;
175  }
176  KALDI_ASSERT(row_offset == NumRows() && col_offset == NumCols());
177  } else {
178  KALDI_ASSERT(NumRows() == B.NumCols() && NumCols() == B.NumRows());
179  int32 row_offset = 0, col_offset = 0;
180  for (int32 b = 0; b < B.NumBlocks(); b++) {
181  const CuMatrixBase<Real> &block = B.Block(b);
182  int32 num_rows = block.NumCols(), num_cols = block.NumRows();
183  CuSubMatrix<Real> this_block(*this, row_offset, num_rows,
184  col_offset, num_cols);
185  this_block.CopyFromMat(block, kTrans);
186  row_offset += num_rows;
187  col_offset += num_cols;
188  }
189  KALDI_ASSERT(row_offset == NumRows() && col_offset == NumCols());
190  }

207 template<class OtherReal>
208 void CuMatrixBase<Real>::CopyFromMat(const CuMatrixBase<OtherReal> &M,
209  MatrixTransposeType trans) {
210  if (sizeof(Real) == sizeof(OtherReal) &&
211  static_cast<const void*>(M.Data()) ==
212  static_cast<const void*>(this->Data())) {
213  if (M.Data() == NULL)
214  return;
215  // CopyFromMat called on same data. Nothing to do (except sanity checks)
216  KALDI_ASSERT(trans == kNoTrans && M.NumRows() == NumRows() &&
217  M.NumCols() == NumCols() && M.Stride() == Stride());
218  return;
219  }
220 #if HAVE_CUDA == 1
221  if (CuDevice::Instantiate().Enabled()) {
222  if (trans == kNoTrans) {
223  KALDI_ASSERT(M.NumRows() == num_rows_ && M.NumCols() == num_cols_);
224  } else {
225  KALDI_ASSERT(M.NumCols() == num_rows_ && M.NumRows() == num_cols_);
226  }
227  if (M.num_rows_ == 0) return; // Nothing to do.
228  Timer tim;
229  if (sizeof(Real) == sizeof(OtherReal) && trans == kNoTrans ) {
230  MatrixIndexT dst_pitch = stride_ * sizeof(Real);
231  MatrixIndexT src_pitch = M.Stride() * sizeof(Real);
232  MatrixIndexT width = M.NumCols() * sizeof(Real);
233  CU_SAFE_CALL(cudaMemcpy2D(data_, dst_pitch, M.data_, src_pitch,
234  width, M.num_rows_, cudaMemcpyDeviceToDevice));
235  } else {
236  dim3 dimGrid, dimBlock;
237  GetBlockSizesForSimpleMatrixOperation(NumRows(), NumCols(),
238  &dimGrid, &dimBlock);
239  if (trans == kNoTrans) {
240  cuda_copy_from_mat(dimGrid, dimBlock, data_, M.data_, Dim(), M.Dim());
241  } else {
242  cuda_copy_from_mat_trans(dimGrid, dimBlock, data_, M.data_, Dim(), M.Dim());
243  }
244  }
245  CuDevice::Instantiate().AccuProfile("CuMatrixBase::CopyFromMat(from other CuMatrixBase)", tim.Elapsed());
246  } else
247 #endif
248  {
249  Mat().CopyFromMat(M.Mat(), trans);
250  }

587 template<typename Real>
588 bool CuMatrixBase<Real>::IsUnit(Real tol) const {
589  // want to return:
590  //FrobeniusNorm(*this - I) <= tol * NumRows(), i.e.:
591  //sqrt (trace((*this - I)(*this-I)) <= tol * NumRows()
592  // trace((*this - I)(*this - I)) <= tol * NumRows()
593  // trace(*this * *this) + trace(I) - 2 * trace(*this) <= tol * NumRows()
594  // trace(*this * *this) + dim - 2*this.Trace() <= tol * NumRows()
595  KALDI_ASSERT(this->NumRows() == this->NumCols());
596  return (TraceMatMat(*this, *this, kTrans) + this->NumRows() - 2.0 * this->Trace() <=
597  tol * this->NumRows());
598 }

1607 template<typename Real>
1608 void CuMatrixBase<Real>::Cholesky(CuMatrixBase<Real> *inv_cholesky) {
1609  KALDI_ASSERT(this->NumRows() == this->NumCols());
1610  const int32 block_size = 64; // We can tune this.
1611 #if HAVE_CUDA == 1
1612  bool have_gpu = CuDevice::Instantiate().Enabled();
1613 #else
1614  bool have_gpu = false;
1615 #endif
1616  if (this->NumRows() == 0) {
1617  return;
1618  }
1619  if (inv_cholesky == NULL && this->NumRows() >= block_size * 2 && have_gpu) {
1620  // Even if the user did not request the inverse Cholesky, for large enough
1621  // matrices (on GPUs) it's going to be more efficient to compute it anyway
1622  // as the recursion depends on it.
1623  CuMatrix<Real> inv(this->NumRows(), this->NumCols());
1624  Cholesky(&inv);
1625  return;
1626  }
1627  if (this->NumRows() <= block_size || inv_cholesky == NULL || !have_gpu) {
1628  // Don't recurse: compute the Cholesky (and inverse Cholesky, if requested)
1629  // directly, on the CPu.
1630  int32 dim = this->NumRows();
1631  CuSpMatrix<Real> this_sp(dim, kUndefined);
1632  this_sp.CopyFromMat(*this, kTakeLower);
1633  SpMatrix<Real> this_sp_cpu(this_sp);
1634  TpMatrix<Real> C_cpu(dim);
1635  C_cpu.Cholesky(this_sp_cpu);
1636  CuTpMatrix<Real> C(C_cpu);
1637  this->CopyFromTp(C);
1638  if (inv_cholesky != NULL) {
1639  C_cpu.Invert(); // Get inverse Cholesky on CPU.
1640  C.CopyFromTp(C_cpu);
1641  inv_cholesky->CopyFromTp(C); // Copy inverse Cholesky from CPU.
1642  }
1643  return;
1644  }
1645  // At this point, if none of the other cases apply, we recurse.
1646 
1647  // The selection of dim1 is a heuristic. We could also just take half.
1648  int32 tot_dim = this->NumRows();
1649  int32 dim1;
1650  // Break it up into a whole number of blocks, for better memory alignment.
1651  // The line below, setting dim1 can be decided on a heuristic basis: from
1652  // the point of view of correctness, it can really be any value
1653  // 0 < dim1 < tot_dim.
1654  dim1 = block_size * std::max<int32>(1, tot_dim / (2 * block_size));
1655 
1656  int32 dim2 = tot_dim - dim1;
1657  CuSubMatrix<Real> this_11(*this, 0, dim1, 0, dim1),
1658  this_12(*this, 0, dim1, dim1, dim2),
1659  this_21(*this, dim1, dim2, 0, dim1),
1660  this_22(*this, dim1, dim2, dim1, dim2);
1661  CuSubMatrix<Real> inv_11(*inv_cholesky, 0, dim1, 0, dim1),
1662  inv_12(*inv_cholesky, 0, dim1, dim1, dim2),
1663  inv_21(*inv_cholesky, dim1, dim2, 0, dim1),
1664  inv_22(*inv_cholesky, dim1, dim2, dim1, dim2);
1665  /*
1666  Here is the math on block-wise Cholesky. We'll use a Matlab-like notation for blocks of a matrix,
1667  e.g. [ A B; C D ], and also for transposes, e.g. A' is the transpose of A.
1668  Let A be the input matrix; we want to compute both its Cholesky L and its inverse Cholesky, which
1669  we'll call M.
1670  OK. let L = [ L11 0; L21 L22 ] be the Cholesky factor of A.
1671  We have A = L L' = [ L11 0; L21 L22 ] * [ L11' L21'; 0 L22' ]. Multiplying it out,
1672  if A = [ A11 A12; A21 A22 ]; then
1673  A11 = L11 L11', A21 = L21 L11', A22 = L21 L21' + L22 L22', and A12 = A21'.
1674 
1675  We also want an expression for the inverse of L (we call this M).
1676  If M = [ M11 0; M21 M22 ], then it's not hard to see that
1677  M11 = inv(L11), M22 = inv(L22).
1678  We can work out M21 as follows. We know that [ L11 0; L21 L22 ] [ M11 0; M21 M22 ] = [ I 0; 0 I ].
1679  Considering the zero on the bottom of the rhs, we have: L21 M11 + L22 M21 = 0, which gives us:
1680  M21 = - L22^{-1} L21 M11 = - M22 L21 M11.
1681 
1682  Next, we want expressions for L21 and L22. From the equation A21 = L21 L11', we have:
1683  L21 = A21 inv(L11') = A21 M11'
1684  We can compute L22 and M22 recursively by doing Cholesky (and computing the inverse Cholesky)
1685  on the quantity T = (A22 - L21 L21'). [we give it the name T just for easy reference.]
1686 
1687  Computationally, we do this as follows:
1688  (1) Recurse to get L11 and M11.
1689  (2) Compute L21 = A21 M11'
1690  (3) Compute T = A22 - L21 L21'
1691  (4) Recurse on T to get L22 and M22.
1692  (5) Compute M21 = -M22 L21 M11.
1693  Next, we have to consider the in-place nature of the computation, since L overwrites A
1694  [M has its own storage, in "inv_cholesky"].
1695  We address this here:
1696  (1) is in-place [L11 replaces A11, M11 has its own storage].
1697  (2) L21 gets written where M21 belongs.
1698  (3) T replaces A22.
1699  (4) is in-place [L22 replaces T where A22 was, M22 has its own storage]
1700  (5):(a) we first compute the transpose of (L21 M11) is done in the upper part of A/L,
1701  where A12 or L12 would be. Define a temporary expression
1702  U = (L21 M11)' = M11' L21'; this goes where A12 or L12 would be.
1703  (b) copy L21 to where it should be, in *this.
1704  (c) Compute M21 = -M22 U', in the correct place for M21.
1705  (d) zero L12 and M12. */
1706 
1707  // (1) compute L11 and M11.
1708  this_11.Cholesky(&inv_11);
1709  // (2) compute L21 = A21 M11'. For now it's in the "wrong place", where M21 should be.
1710  inv_21.AddMatMat(1.0, this_21, kNoTrans, inv_11, kTrans, 0.0);
1711  // (3) compute T = A22 - L21 L21'. Note: only the lower triangle of T will be valid, but
1712  // that's OK because Cholesky will ignore the upper part.
1713  this_22.SymAddMat2(-1.0, inv_21, kNoTrans, 1.0);
1714  // (4) Recurse to compute L22 and M22.
1715  this_22.Cholesky(&inv_22);
1716  // (5)(a) compute U = M11' L21'. We use the storage of this_12 for this. Note that L21 is
1717  // currently where M21 should be.
1718  this_12.AddMatMat(1.0, inv_11, kTrans, inv_21, kTrans, 0.0);
1719  // (5)(b) copy L21 to where it should be.
1720  this_21.CopyFromMat(inv_21);
1721  // (5)(c) compute M21 = -M22 U'.
1722  inv_21.AddMatMat(-1.0, inv_22, kNoTrans, this_12, kTrans, 0.0);
1723  // (5)(d) zero L12 and M12.
1724  this_12.SetZero();
1725  inv_12.SetZero();
1726 }

1765 Real TraceMatMat(const CuMatrixBase<Real> &A,
1766  const CuMatrixBase<Real> &B,
1767  MatrixTransposeType trans) {
1768  if (A.num_rows_ == 0) {
1769  KALDI_ASSERT(B.num_rows_ == 0);
1770  return 0.0;
1771  }
1772  Real result = 0;
1773 #if HAVE_CUDA == 1
1774  if (CuDevice::Instantiate().Enabled()) {
1775  if (trans == kNoTrans) {
1776  KALDI_ASSERT(A.NumRows() == B.NumCols() && A.NumCols() == B.NumRows());
1777  } else {
1778  KALDI_ASSERT(A.NumRows() == B.NumRows() && A.NumCols() == B.NumCols());
1779  }
1780  if (A.NumRows() * A.NumCols() > 16384) {
1781  // This version in which we don't use a special-purpose kernel, but
1782  // do AddDiagMat on a temporary vector and returns its sum, seems to be
1783  // faster for larger matrices. The cutoff is approximate and
1784  // we only looked at the time on square matrices, which
1785  // is what we test in cu-matrix-speed-test.cc.
1786  CuVector<Real> sum_vec(A.NumRows());
1787  sum_vec.AddDiagMatMat(1.0, A, kNoTrans,
1788  B, trans, 0.0);
1789  return sum_vec.Sum();
1790  } else {
1791  Timer tim;
1792  // the sizes of result_vec must match what we
1793  // call the kernels with, in cu-kernels.cu
1794  CuVector<Real> result_vec(trans == kTrans ? 4 : 2, kUndefined);
1795  if (trans == kNoTrans) {
1796  cuda_trace_mat_mat(A.Data(), B.Data(), A.Dim(), B.Stride(), result_vec.Data());
1797  } else {
1798  cuda_trace_mat_mat_trans(A.Data(), B.Data(), A.Dim(), B.Stride(), result_vec.Data());
1799  }
1800  CU_SAFE_CALL(cudaGetLastError());
1801  Vector<Real> result_cpu(result_vec); // copying from CUDA faster than summing in CUDA.
1802  result = result_cpu.Sum();
1803  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
1804  }
1805  } else
1806 #endif
1807  {
1808  result = TraceMatMat(A.Mat(), B.Mat(), trans);
1809  }
1810  return result;
1811 }

1822 template<typename Real>
1823 void AddMatMatBatched(const Real alpha, std::vector<CuSubMatrix<Real>* > &C,
1824  const std::vector<CuSubMatrix<Real>* > &A, MatrixTransposeType transA,
1825  const std::vector<CuSubMatrix<Real>* > &B, MatrixTransposeType transB,
1826  const Real beta) {
1827  // the batch counts should be the same
1828  int32 batchCountA = static_cast<int32>(A.size()), batchCountB = static_cast<int32>(B.size()), batchCountC = static_cast<int32>(C.size());
1829  KALDI_ASSERT((batchCountA == batchCountB) && (batchCountB == batchCountC));
1830  int32 batchCount = batchCountA;
1831 
1832  if (batchCount == 0) return;
1833 
1834  // all elements must have the same num-rows, num-cols and stride
1835  for (int32 i = 0; i < batchCount-1; i++) {
1836  KALDI_ASSERT(A[i]->NumRows() == A[i+1]->NumRows());
1837  KALDI_ASSERT(A[i]->NumCols() == A[i+1]->NumCols());
1838  KALDI_ASSERT(A[i]->Stride() == A[i+1]->Stride());
1839  KALDI_ASSERT(B[i]->NumRows() == B[i+1]->NumRows());
1840  KALDI_ASSERT(B[i]->NumCols() == B[i+1]->NumCols());
1841  KALDI_ASSERT(B[i]->Stride() == B[i+1]->Stride());
1842  KALDI_ASSERT(C[i]->NumRows() == C[i+1]->NumRows());
1843  KALDI_ASSERT(C[i]->NumCols() == C[i+1]->NumCols());
1844  KALDI_ASSERT(C[i]->Stride() == C[i+1]->Stride());
1845  }
1846  // CUBLAS is col-major, cudamatrix is row-major, how to do the mapping?
1847  // keep trans..., just swap A&B matrices: A->B B->A
1848  MatrixIndexT m = ((transB==kTrans)? B[0]->NumRows() : B[0]->NumCols());
1849  MatrixIndexT n = ((transA==kTrans)? A[0]->NumCols() : A[0]->NumRows());
1850  MatrixIndexT k = ((transB==kTrans)? B[0]->NumCols() : B[0]->NumRows());
1851  MatrixIndexT k1 = ((transA==kTrans)? A[0]->NumRows() : A[0]->NumCols());
1852 
1853  KALDI_ASSERT(m == C[0]->NumCols());
1854  KALDI_ASSERT(n == C[0]->NumRows());
1855  KALDI_ASSERT(k == k1);
1856 
1857  if (m == 0) return;
1858 
1859 #if HAVE_CUDA == 1
1860  if (CuDevice::Instantiate().Enabled()) {
1861  Timer tim;
1862 
1863  // place all data (A[batchCount], B[batchCount], C[batchCount]) in memory together
1864  // in order to make only one call of cudaMemcpy
1865  Real **device_abc_array = (Real**)CuDevice::Instantiate().Malloc(3*batchCount*sizeof(Real*));
1866  const Real **device_a_array = const_cast<const Real**>(device_abc_array);
1867  const Real **device_b_array = const_cast<const Real**>(device_abc_array) + batchCount;
1868  Real **device_c_array = device_abc_array + 2 * batchCount;
1869  const Real **host_abc_array = new const Real*[3*batchCount];
1870  const Real **host_a_array = host_abc_array;
1871  const Real **host_b_array = host_abc_array + batchCount;
1872  const Real **host_c_array = host_abc_array + 2 * batchCount;
1873 
1874  for (int32 i = 0; i < batchCount; i++) {
1875  host_a_array[i] = A[i]->data_;
1876  host_b_array[i] = B[i]->data_;
1877  host_c_array[i] = C[i]->data_;
1878  }
1879 
1880  CU_SAFE_CALL(cudaMemcpy(device_abc_array, host_abc_array, 3*batchCount*sizeof(Real*), cudaMemcpyHostToDevice));
1881 
1882  CU_SAFE_CALL(cublas_gemmBatched(GetCublasHandle(),
1883  (transB==kTrans? CUBLAS_OP_T:CUBLAS_OP_N),
1884  (transA==kTrans? CUBLAS_OP_T:CUBLAS_OP_N),
1885  m, n, k, alpha, device_b_array, B[0]->Stride(),
1886  device_a_array, A[0]->Stride(), beta,
1887  device_c_array, C[0]->Stride(), batchCount));
1888 
1889  CuDevice::Instantiate().Free(device_abc_array);
1890  delete[] host_abc_array;
1891 
1892  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
1893  } else
1894 #endif
1895  {
1896  for (int32 i = 0; i< batchCount; i++) {
1897  C[i]->Mat().AddMatMat(alpha, A[i]->Mat(), transA, B[i]->Mat(), transB, beta);
1898  }
1899  }
1900 }
