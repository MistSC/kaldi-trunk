29 int32 NumFrames(int32 nsamp,
30  const FrameExtractionOptions &opts) {
31  int32 frame_shift = opts.WindowShift();
32  int32 frame_length = opts.WindowSize();
33  KALDI_ASSERT(frame_shift != 0 && frame_length != 0);
34  if (opts.snip_edges) {
35  if (static_cast<int32>(nsamp) < frame_length)
36  return 0;
37  else
38  return (1 + ((nsamp - frame_length) / frame_shift));
39  // view the expression above as: nsamp-frame_length is how much room we
40  // have to shift the frame within the waveform; frame_shift is how much
41  // we shift it each time and the ratio is how many times we can shift
42  // it (integer arithmetic rounds down).
43  } else {
44  return (int32)(nsamp * 1.0f / frame_shift + 0.5f);
45  // if --snip-edges=false, the number of frames would be determined by
46  // rounding the (file-length / frame-shift) to the nearest integer
47  }
48 }

58  if (preemph_coeff == 0.0) return;

60  for (int32 i = waveform->Dim()-1; i > 0; i--)

67 FeatureWindowFunction::FeatureWindowFunction(const FrameExtractionOptions &opts) {
68  int32 frame_length = opts.WindowSize();
69  KALDI_ASSERT(frame_length > 0);
70  window.Resize(frame_length);
71  for (int32 i = 0; i < frame_length; i++) {
72  BaseFloat i_fl = static_cast<BaseFloat>(i);
73  if (opts.window_type == "hanning") {
74  window(i) = 0.5 - 0.5*cos(M_2PI * i_fl / (frame_length-1));
75  } else if (opts.window_type == "hamming") {
76  window(i) = 0.54 - 0.46*cos(M_2PI * i_fl / (frame_length-1));
77  } else if (opts.window_type == "povey") { // like hamming but goes to zero at edges.
78  window(i) = pow(0.5 - 0.5*cos(M_2PI * i_fl / (frame_length-1)), 0.85);
79  } else if (opts.window_type == "rectangular") {
80  window(i) = 1.0;
81  } else {
82  KALDI_ERR << "Invalid window type " << opts.window_type;
83  }
84  }
85 }

91 void ExtractWindow(const VectorBase<BaseFloat> &wave,
92  int32 f, // with 0 <= f < NumFrames(feats, opts)
93  const FrameExtractionOptions &opts,
94  const FeatureWindowFunction &window_function,
95  Vector<BaseFloat> *window,
96  BaseFloat *log_energy_pre_window) {
97  int32 frame_shift = opts.WindowShift();
98  int32 frame_length = opts.WindowSize();
99  KALDI_ASSERT(window_function.window.Dim() == frame_length);
100  KALDI_ASSERT(frame_shift != 0 && frame_length != 0);
101 
102  Vector<BaseFloat> wave_part(frame_length);
103  if (opts.snip_edges) {
104  int32 start = frame_shift*f, end = start + frame_length;
105  KALDI_ASSERT(start >= 0 && end <= wave.Dim());
106  wave_part.CopyFromVec(wave.Range(start, frame_length));
107  } else {
108  // If opts.snip_edges = false, we allow the frames to go slightly over the
109  // edges of the file; we'll extend the data by reflection.
110  int32 mid = frame_shift * (f + 0.5),
111  begin = mid - frame_length / 2,
112  end = begin + frame_length,
113  begin_limited = std::max<int32>(0, begin),
114  end_limited = std::min(end, wave.Dim()),
115  length_limited = end_limited - begin_limited;
116 
117  // Copy the main part. Usually this will be the entire window.
118  wave_part.Range(begin_limited - begin, length_limited).
119  CopyFromVec(wave.Range(begin_limited, length_limited));
120 
121  // Deal with any end effects by reflection, if needed. This code will
122  // rarely be reached, so we don't concern ourselves with efficiency.
123  for (int32 f = begin; f < 0; f++) {
124  int32 reflected_f = -f;
125  // The next statement will only have an effect in the case of files
126  // shorter than a single frame, it's to avoid a crash in those cases.
127  reflected_f = reflected_f % wave.Dim();
128  wave_part(f - begin) = wave(reflected_f);
129  }
130  for (int32 f = wave.Dim(); f < end; f++) {
131  int32 distance_to_end = f - wave.Dim();
132  // The next statement will only have an effect in the case of files
133  // shorter than a single frame, it's to avoid a crash in those cases.
134  distance_to_end = distance_to_end % wave.Dim();
135  int32 reflected_f = wave.Dim() - 1 - distance_to_end;
136  wave_part(f - begin) = wave(reflected_f);
137  }
138  }
139  KALDI_ASSERT(window != NULL);
140  int32 frame_length_padded = opts.PaddedWindowSize();
141 
142  if (window->Dim() != frame_length_padded)
143  window->Resize(frame_length_padded);
144 
145  SubVector<BaseFloat> window_part(*window, 0, frame_length);
146  window_part.CopyFromVec(wave_part);
147 
148  if (opts.dither != 0.0) Dither(&window_part, opts.dither);
149 
150  if (opts.remove_dc_offset)
151  window_part.Add(-window_part.Sum() / frame_length);
152 
153  if (log_energy_pre_window != NULL) {
154  BaseFloat energy = std::max(VecVec(window_part, window_part),
155  std::numeric_limits<BaseFloat>::min());
156  *log_energy_pre_window = Log(energy);
157  }
158 
159  if (opts.preemph_coeff != 0.0)
160  Preemphasize(&window_part, opts.preemph_coeff);
161 
162  window_part.MulElements(window_function.window);
163 
164  if (frame_length != frame_length_padded)
165  SubVector<BaseFloat>(*window, frame_length,
166  frame_length_padded-frame_length).SetZero();
167 }

210 DeltaFeatures::DeltaFeatures(const DeltaFeaturesOptions &opts): opts_(opts) {
211  KALDI_ASSERT(opts.order >= 0 && opts.order < 1000); // just make sure we don't get binary junk.
212  // opts will normally be 2 or 3.
213  KALDI_ASSERT(opts.window > 0 && opts.window < 1000); // again, basic sanity check.
214  // normally the window size will be two.
215 
216  scales_.resize(opts.order+1);
217  scales_[0].Resize(1);
218  scales_[0](0) = 1.0; // trivial window for 0th order delta [i.e. baseline feats]
219 
220  for (int32 i = 1; i <= opts.order; i++) {
221  Vector<BaseFloat> &prev_scales = scales_[i-1],
222  &cur_scales = scales_[i];
223  int32 window = opts.window; // this code is designed to still
224  // work if instead we later make it an array and do opts.window[i-1],
225  // or something like that. "window" is a parameter specifying delta-window
226  // width which is actually 2*window + 1.
227  KALDI_ASSERT(window != 0);
228  int32 prev_offset = (static_cast<int32>(prev_scales.Dim()-1))/2,
229  cur_offset = prev_offset + window;
230  cur_scales.Resize(prev_scales.Dim() + 2*window); // also zeros it.
231 
232  BaseFloat normalizer = 0.0;
233  for (int32 j = -window; j <= window; j++) {
234  normalizer += j*j;
235  for (int32 k = -prev_offset; k <= prev_offset; k++) {
236  cur_scales(j+k+cur_offset) +=
237  static_cast<BaseFloat>(j) * prev_scales(k+prev_offset);
238  }
239  }
240  cur_scales.Scale(1.0 / normalizer);
241  }
242 }

244 void DeltaFeatures::Process(const MatrixBase<BaseFloat> &input_feats,
245  int32 frame,
246  VectorBase<BaseFloat> *output_frame) const {
247  KALDI_ASSERT(frame < input_feats.NumRows());
248  int32 num_frames = input_feats.NumRows(),
249  feat_dim = input_feats.NumCols();
250  KALDI_ASSERT(static_cast<int32>(output_frame->Dim()) == feat_dim * (opts_.order+1));
251  output_frame->SetZero();
252  for (int32 i = 0; i <= opts_.order; i++) {
253  const Vector<BaseFloat> &scales = scales_[i];
254  int32 max_offset = (scales.Dim() - 1) / 2;
255  SubVector<BaseFloat> output(*output_frame, i*feat_dim, feat_dim);
256  for (int32 j = -max_offset; j <= max_offset; j++) {
257  // if asked to read
258  int32 offset_frame = frame + j;
259  if (offset_frame < 0) offset_frame = 0;
260  else if (offset_frame >= num_frames)
261  offset_frame = num_frames - 1;
262  BaseFloat scale = scales(j + max_offset);
263  if (scale != 0.0)
264  output.AddVec(scale, input_feats.Row(offset_frame));
265  }
266  }

285 void ShiftedDeltaFeatures::Process(const MatrixBase<BaseFloat> &input_feats,
286  int32 frame,
287  SubVector<BaseFloat> *output_frame) const {
288  KALDI_ASSERT(frame < input_feats.NumRows());
289  int32 num_frames = input_feats.NumRows(),
290  feat_dim = input_feats.NumCols();
291  KALDI_ASSERT(static_cast<int32>(output_frame->Dim())
292  == feat_dim * (opts_.num_blocks + 1));
293  output_frame->SetZero();
294 
295  // The original features
296  SubVector<BaseFloat> output(*output_frame, 0, feat_dim);
297  output.AddVec(1.0, input_feats.Row(frame));
298 
299  // Concatenate the delta-blocks. Each block is block_shift
300  // (usually 3) frames apart.
301  for (int32 i = 0; i < opts_.num_blocks; i++) {
302  int32 max_offset = (scales_.Dim() - 1) / 2;
303  SubVector<BaseFloat> output(*output_frame, (i + 1) * feat_dim, feat_dim);
304  for (int32 j = -max_offset; j <= max_offset; j++) {
305  int32 offset_frame = frame + j + i * opts_.block_shift;
306  if (offset_frame < 0) offset_frame = 0;
307  else if (offset_frame >= num_frames)
308  offset_frame = num_frames - 1;
309  BaseFloat scale = scales_(j + max_offset);
310  if (scale != 0.0)
311  output.AddVec(scale, input_feats.Row(offset_frame));
312  }
313  }

360 void InitIdftBases(int32 n_bases, int32 dimension, Matrix<BaseFloat> *mat_out) {
361  BaseFloat angle = M_PI / static_cast<BaseFloat>(dimension - 1);
362  BaseFloat scale = 1.0f / (2.0 * static_cast<BaseFloat>(dimension - 1));
363  mat_out->Resize(n_bases, dimension);
364  for (int32 i = 0; i < n_bases; i++) {
365  (*mat_out)(i, 0) = 1.0 * scale;
366  BaseFloat i_fl = static_cast<BaseFloat>(i);
367  for (int32 j = 1; j < dimension - 1; j++) {
368  BaseFloat j_fl = static_cast<BaseFloat>(j);
369  (*mat_out)(i, j) = 2.0 * scale * cos(angle * i_fl * j_fl);
370  }
371 
372  (*mat_out)(i, dimension -1)
373  = scale * cos(angle * i_fl * static_cast<BaseFloat>(dimension-1));
374  }

435 // Internal version of SlidingWindowCmn with double-precision arguments.
436 void SlidingWindowCmnInternal(const SlidingWindowCmnOptions &opts,
437  const MatrixBase<double> &input,
438  MatrixBase<double> *output) {
439  opts.Check();
440  int32 num_frames = input.NumRows(), dim = input.NumCols();
441 
442  int32 last_window_start = -1, last_window_end = -1;
443  Vector<double> cur_sum(dim), cur_sumsq(dim);
444 
445  for (int32 t = 0; t < num_frames; t++) {
446  int32 window_start, window_end; // note: window_end will be one
447  // past the end of the window we use for normalization.
448  if (opts.center) {
449  window_start = t - (opts.cmn_window / 2);
450  window_end = window_start + opts.cmn_window;
451  } else {
452  window_start = t - opts.cmn_window;
453  window_end = t + 1;
454  }
455  if (window_start < 0) { // shift window right if starts <0.
456  window_end -= window_start;
457  window_start = 0; // or: window_start -= window_start
458  }
459  if (!opts.center) {
460  if (window_end > t)
461  window_end = std::max(t + 1, opts.min_window);
462  }
463  if (window_end > num_frames) {
464  window_start -= (window_end - num_frames);
465  window_end = num_frames;
466  if (window_start < 0) window_start = 0;
467  }
468  if (last_window_start == -1) {
469  SubMatrix<double> input_part(input,
470  window_start, window_end - window_start,
471  0, dim);
472  cur_sum.AddRowSumMat(1.0, input_part , 0.0);
473  if (opts.normalize_variance)
474  cur_sumsq.AddDiagMat2(1.0, input_part, kTrans, 0.0);
475  } else {
476  if (window_start > last_window_start) {
477  KALDI_ASSERT(window_start == last_window_start + 1);
478  SubVector<double> frame_to_remove(input, last_window_start);
479  cur_sum.AddVec(-1.0, frame_to_remove);
480  if (opts.normalize_variance)
481  cur_sumsq.AddVec2(-1.0, frame_to_remove);
482  }
483  if (window_end > last_window_end) {
484  KALDI_ASSERT(window_end == last_window_end + 1);
485  SubVector<double> frame_to_add(input, last_window_end);
486  cur_sum.AddVec(1.0, frame_to_add);
487  if (opts.normalize_variance)
488  cur_sumsq.AddVec2(1.0, frame_to_add);
489  }
490  }
491  int32 window_frames = window_end - window_start;
492  last_window_start = window_start;
493  last_window_end = window_end;
494 
495  KALDI_ASSERT(window_frames > 0);
496  SubVector<double> input_frame(input, t),
497  output_frame(*output, t);
498  output_frame.CopyFromVec(input_frame);
499  output_frame.AddVec(-1.0 / window_frames, cur_sum);
500 
501  if (opts.normalize_variance) {
502  if (window_frames == 1) {
503  output_frame.Set(0.0);
504  } else {
505  Vector<double> variance(cur_sumsq);
506  variance.Scale(1.0 / window_frames);
507  variance.AddVec2(-1.0 / (window_frames * window_frames), cur_sum);
508  // now "variance" is the variance of the features in the window,
509  // around their own mean.
510  int32 num_floored = variance.ApplyFloor(1.0e-10);
511  if (num_floored > 0 && num_frames > 1) {
512  KALDI_WARN << "Flooring variance When normalizing variance, floored " << num_floored
513  << " elements; num-frames was " << window_frames;
514  }
515  variance.ApplyPow(-0.5); // get inverse standard deviation.
516  output_frame.MulElements(variance);
517  }
518  }
519  }
520 }
