40 void UnitTestNnetCompute() {
41  for (int32 n = 0; n < 20; n++) {
42  struct NnetGenerationOptions gen_config;
43 
44 
45  std::vector<std::string> configs;
46  GenerateConfigSequence(gen_config, &configs);
47  Nnet nnet;
48  for (size_t j = 0; j < configs.size(); j++) {
49  KALDI_LOG << "Input config[" << j << "] is: " << configs[j];
50  std::istringstream is(configs[j]);
51  nnet.ReadConfig(is);
52  }
53 
54  ComputationRequest request;
55  std::vector<Matrix<BaseFloat> > inputs;
56  ComputeExampleComputationRequestSimple(nnet, &request, &inputs);
57 
58  NnetComputation computation;
59  Compiler compiler(request, nnet);
60 
61  CompilerOptions opts;
62  compiler.CreateComputation(opts, &computation);
63  {
64  std::ostringstream os;
65  computation.Print(os, nnet);
66  KALDI_LOG << "Generated computation is: " << os.str();
67  }
68  CheckComputationOptions check_config;
69  // we can do the rewrite check since it's before optimization.
70  check_config.check_rewrite = true;
71  ComputationChecker checker(check_config, nnet, computation);
72  checker.Check();
73 
74  if (RandInt(0, 1) == 0) {
75  NnetOptimizeOptions opt_config;
76 
77  Optimize(opt_config, nnet, request, &computation);
78  {
79  std::ostringstream os;
80  computation.Print(os, nnet);
81  KALDI_LOG << "Optimized computation is: " << os.str();
82  }
83  }
84 
85  NnetComputeOptions compute_opts;
86  if (RandInt(0, 1) == 0)
87  compute_opts.debug = true;
88 
89  computation.ComputeCudaIndexes();
90  NnetComputer computer(compute_opts,
91  computation,
92  nnet,
93  &nnet);
94  // provide the input to the computation.
95  for (size_t i = 0; i < request.inputs.size(); i++) {
96  CuMatrix<BaseFloat> temp(inputs[i]);
97  KALDI_LOG << "Input sum is " << temp.Sum();
98  computer.AcceptInput(request.inputs[i].name, &temp);
99  }
100  computer.Forward();
101  const CuMatrixBase<BaseFloat> &output(computer.GetOutput("output"));
102 
103  TestNnetDecodable(request, inputs, nnet, output);
104 
105  KALDI_LOG << "Output sum is " << output.Sum();
106  CuMatrix<BaseFloat> output_deriv(output.NumRows(), output.NumCols());
107  output_deriv.SetRandn();
108  // output_deriv sum won't be informative so don't print it.
109  if (request.outputs[0].has_deriv)
110  computer.AcceptOutputDeriv("output", &output_deriv);
111  computer.Backward();
112  for (size_t i = 0; i < request.inputs.size(); i++) {
113  if (request.inputs[i].has_deriv) {
114  const CuMatrixBase<BaseFloat> &in_deriv =
115  computer.GetInputDeriv(request.inputs[i].name);
116  KALDI_LOG << "Input-deriv sum for input '"
117  << request.inputs[i].name << "' is " << in_deriv.Sum();
118  }
119  }
120  }
121 }
