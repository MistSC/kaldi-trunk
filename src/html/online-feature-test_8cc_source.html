40 void GetOutput(OnlineFeatureInterface *a,
41  Matrix<BaseFloat> *output) {
42  int32 dim = a->Dim();
43  int32 frame_num = 0;
44  OnlineCacheFeature cache(a);
45 
46  std::vector<Vector<BaseFloat>* > cached_frames;
47  while (true) {
48  Vector<BaseFloat> garbage(dim);
49  cache.GetFrame(frame_num, &garbage);
50  cached_frames.push_back(new Vector<BaseFloat>(garbage));
51  if (cache.IsLastFrame(frame_num))
52  break;
53  frame_num++;
54  }
55 
56  KALDI_ASSERT(cached_frames.size() == a->NumFramesReady());
57 
58  output->Resize(cached_frames.size(), dim);
59  for (int32 i = 0; i < cached_frames.size(); i++) {
60  output->CopyRowFromVec(*(cached_frames[i]), i);
61  delete cached_frames[i];
62  }
63  cached_frames.clear();
64  cache.ClearCache();
65 }

67 // Only generate random length for each piece
68 bool RandomSplit(int32 wav_dim,
69  std::vector<int32> *piece_dim,
70  int32 num_pieces,
71  int32 trials = 5) {
72  piece_dim->clear();
73  piece_dim->resize(num_pieces, 0);
74 
75  int32 dim_mean = wav_dim / (num_pieces * 2);
76  int32 cnt = 0;
77  while (true) {
78  int32 dim_total = 0;
79  for (int32 i = 0; i < num_pieces - 1; i++) {
80  (*piece_dim)[i] = dim_mean + rand() % dim_mean;
81  dim_total += (*piece_dim)[i];
82  }
83  (*piece_dim)[num_pieces - 1] = wav_dim - dim_total;
84 
85  if (dim_total > 0 && dim_total < wav_dim)
86  break;
87  if (++cnt > trials)
88  return false;
89  }
90  return true;
91 }

161  KALDI_ASSERT(wave.Data().NumRows() == 1);

164  // the parametrization object
165  MfccOptions op;
166  op.frame_opts.dither = 0.0;
167  op.frame_opts.preemph_coeff = 0.0;
168  op.frame_opts.window_type = "hamming";
169  op.frame_opts.remove_dc_offset = false;
170  op.frame_opts.round_to_power_of_two = true;
171  op.frame_opts.samp_freq = wave.SampFreq();
172  op.mel_opts.low_freq = 0.0;
173  op.htk_compat = false;
174  op.use_energy = false; // C0 not energy.
175  Mfcc mfcc(op);
176 
177  // compute mfcc offline
178  Matrix<BaseFloat> mfcc_feats;
179  mfcc.Compute(waveform, 1.0, &mfcc_feats, NULL); // vtln not supported
180 
181  // compare
182  // The test waveform is about 1.44s long, so
183  // we try to break it into from 5 pieces to 9(not essential to do so)
184  for (int32 num_piece = 5; num_piece < 10; num_piece++) {
185  OnlineMfcc online_mfcc(op);
186  std::vector<int32> piece_length(num_piece, 0);
187 
188  bool ret = RandomSplit(waveform.Dim(), &piece_length, num_piece);
189  KALDI_ASSERT(ret);
190 
191  int32 offset_start = 0;
192  for (int32 i = 0; i < num_piece; i++) {
193  Vector<BaseFloat> wave_piece(
194  waveform.Range(offset_start, piece_length[i]));
195  online_mfcc.AcceptWaveform(wave.SampFreq(), wave_piece);
196  offset_start += piece_length[i];
197  }
198  online_mfcc.InputFinished();
199 
200  Matrix<BaseFloat> online_mfcc_feats;
201  GetOutput(&online_mfcc, &online_mfcc_feats);
202 
203  AssertEqual(mfcc_feats, online_mfcc_feats);
204  }
