204 template<class Arc, class inttype> void TestStringRepository() {
205  typedef typename Arc::Label Label;
206 
207  StringRepository<Label, inttype> sr;
208 
209  int N = 1000;
210  if (sizeof(inttype) == 1) N = 64;
211  vector<vector<Label> > strings(N);
212  vector<inttype> ids(N);
213 
214  for (size_t i = 0;i < N;i++) {
215  size_t len = kaldi::Rand() % 4;
216  vector<Label> vec;
217  for (size_t j = 0;j < len;j++) vec.push_back( (kaldi::Rand()%10) + 150*(kaldi::Rand()%2)); // make it have reasonable range.
218  if (i < 500 && vec.size() == 0) ids[i] = sr.IdOfEmpty();
219  else if (i < 500 && vec.size() == 1) ids[i] = sr.IdOfLabel(vec[0]);
220  else ids[i] = sr.IdOfSeq(vec);
221 
222  strings[i] = vec;
223  }
224 
225  for (size_t i = 0;i < N;i++) {
226  vector<Label> tmpv;
227  tmpv.push_back(10); // just put in garbage.
228  sr.SeqOfId(ids[i], &tmpv);
229  assert(tmpv == strings[i]);
230  assert(sr.IdOfSeq(strings[i]) == ids[i]);
231  if (strings[i].size() == 0) assert(ids[i] == sr.IdOfEmpty());
232  if (strings[i].size() == 1) assert(ids[i] == sr.IdOfLabel(strings[i][0]));
233 
234  if (sizeof(inttype) != 1) {
235  size_t prefix_len = kaldi::Rand() % (strings[i].size() + 1);
236  inttype s2 = sr.RemovePrefix(ids[i], prefix_len);
237  vector<Label> vec2;
238  sr.SeqOfId(s2, &vec2);
239  for (size_t j = 0;j < strings[i].size()-prefix_len;j++) {
240  assert(vec2[j] == strings[i][j+prefix_len]);
241  }
242  }
243 
244  }
245 }
