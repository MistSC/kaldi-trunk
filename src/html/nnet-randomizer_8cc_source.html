45 void MatrixRandomizer::AddData(const CuMatrixBase<BaseFloat>& m) {
46  // pre-allocate before 1st use
47  if(data_.NumCols() == 0) {
48  data_.Resize(conf_.randomizer_size,m.NumCols());
49  }
50  // optionally put previous left-over to front
51  if (data_begin_ > 0) {
52  KALDI_ASSERT(data_begin_ <= data_end_); // sanity check
53  int32 leftover = data_end_ - data_begin_;
54  KALDI_ASSERT(leftover < data_begin_); // no overlap
55  if(leftover > 0) {
56  data_.RowRange(0,leftover).CopyFromMat(data_.RowRange(data_begin_,leftover));
57  }
58  data_begin_ = 0; data_end_ = leftover;
59  data_.RowRange(leftover,data_.NumRows()-leftover).SetZero(); // zeroing the rest
60  }
61  // extend the buffer if necessary
62  if(data_.NumRows() < data_end_ + m.NumRows()) {
63  CuMatrix<BaseFloat> data_aux(data_);
64  data_.Resize(data_end_ + m.NumRows() + 1000, data_.NumCols()); // +1000 row extra
65  data_.RowRange(0,data_aux.NumRows()).CopyFromMat(data_aux);
66  }
67  // copy the data
68  data_.RowRange(data_end_,m.NumRows()).CopyFromMat(m);
69  data_end_ += m.NumRows();
70 }

72 void MatrixRandomizer::Randomize(const std::vector<int32>& mask) {
73  KALDI_ASSERT(data_begin_ == 0);
74  KALDI_ASSERT(data_end_ > 0);
75  KALDI_ASSERT(data_end_ == mask.size());
76  // Copy to auxiliary buffer for unshuffled data
77  data_aux_ = data_;
78  // Put the mask to GPU
79  CuArray<int32> mask_in_gpu(mask.size());
80  mask_in_gpu.CopyFromVec(mask);
81  // Randomize the data, mask is used to index rows in source matrix:
82  // (Here the vector 'mask_in_gpu' is typically shorter than number of rows in 'data_aux_',
83  // because the the buffer 'data_aux_' is larger than capacity 'randomizer_size'.
84  // The extra rows in 'data_aux_' do not contain speech frames and are not copied
85  // from 'data_aux_', the extra rows in 'data_' are unchanged by cu::Randomize.)
86  cu::Randomize(data_aux_, mask_in_gpu, &data_);
87 }
