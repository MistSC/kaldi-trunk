33 template<class T> void WriteBasicType(std::ostream &os,
34  bool binary, T t) {
35  // Compile time assertion that this is not called with a wrong type.
36  KALDI_ASSERT_IS_INTEGER_TYPE(T);
37  if (binary) {
38  char len_c = (std::numeric_limits<T>::is_signed ? 1 : -1)
39  * static_cast<char>(sizeof(t));
40  os.put(len_c);
41  os.write(reinterpret_cast<const char *>(&t), sizeof(t));
42  } else {
43  if (sizeof(t) == 1)
44  os << static_cast<int16>(t) << " ";
45  else
46  os << t << " ";
47  }
48  if (os.fail()) {
49  throw std::runtime_error("Write failure in WriteBasicType.");
50  }

91 template<class T> inline void WriteIntegerVector(std::ostream &os, bool binary,
92  const std::vector<T> &v) {
93  // Compile time assertion that this is not called with a wrong type.
94  KALDI_ASSERT_IS_INTEGER_TYPE(T);
95  if (binary) {
96  char sz = sizeof(T); // this is currently just a check.
97  os.write(&sz, 1);
98  int32 vecsz = static_cast<int32>(v.size());
99  KALDI_ASSERT((size_t)vecsz == v.size());
100  os.write(reinterpret_cast<const char *>(&vecsz), sizeof(vecsz));
101  if (vecsz != 0) {
102  os.write(reinterpret_cast<const char *>(&(v[0])), sizeof(T)*vecsz);
103  }
104  } else {
105  // focus here is on prettiness of text form rather than
106  // efficiency of reading-in.
107  // reading-in is dominated by low-level operations anyway:
108  // for efficiency use binary.
109  os << "[ ";
110  typename std::vector<T>::const_iterator iter = v.begin(), end = v.end();
111  for (; iter != end; ++iter) {
112  if (sizeof(T) == 1)
113  os << static_cast<int16>(*iter) << " ";
114  else
115  os << *iter << " ";
116  }
117  os << "]\n";
118  }
119  if (os.fail()) {
120  throw std::runtime_error("Write failure in WriteIntegerVector.");
121  }
122 }

125 template<class T> inline void ReadIntegerVector(std::istream &is,
126  bool binary,
127  std::vector<T> *v) {
128  KALDI_ASSERT_IS_INTEGER_TYPE(T);
129  KALDI_ASSERT(v != NULL);
130  if (binary) {
131  int sz = is.peek();
132  if (sz == sizeof(T)) {
133  is.get();
134  } else { // this is currently just a check.
135  KALDI_ERR << "ReadIntegerVector: expected to see type of size "
136  << sizeof(T) << ", saw instead " << sz << ", at file position "
137  << is.tellg();
138  }
139  int32 vecsz;
140  is.read(reinterpret_cast<char *>(&vecsz), sizeof(vecsz));
141  if (is.fail() || vecsz < 0) goto bad;
142  v->resize(vecsz);
143  if (vecsz > 0) {
144  is.read(reinterpret_cast<char *>(&((*v)[0])), sizeof(T)*vecsz);
145  }
146  } else {
147  std::vector<T> tmp_v; // use temporary so v doesn't use extra memory
148  // due to resizing.
149  is >> std::ws;
150  if (is.peek() != static_cast<int>('[')) {
151  KALDI_ERR << "ReadIntegerVector: expected to see [, saw "
152  << is.peek() << ", at file position " << is.tellg();
153  }
154  is.get(); // consume the '['.
155  is >> std::ws; // consume whitespace.
156  while (is.peek() != static_cast<int>(']')) {
157  if (sizeof(T) == 1) { // read/write chars as numbers.
158  int16 next_t;
159  is >> next_t >> std::ws;
160  if (is.fail()) goto bad;
161  else
162  tmp_v.push_back((T)next_t);
163  } else {
164  T next_t;
165  is >> next_t >> std::ws;
166  if (is.fail()) goto bad;
167  else
168  tmp_v.push_back(next_t);
169  }
170  }
171  is.get(); // get the final ']'.
172  *v = tmp_v; // could use std::swap to use less temporary memory, but this
173  // uses less permanent memory.
174  }
175  if (!is.fail()) return;
176  bad:
177  KALDI_ERR << "ReadIntegerVector: read failure at file position "
178  << is.tellg();
179 }

182 // Initialize an opened stream for writing by writing an optional binary
183 // header and modifying the floating-point precision.
184 inline void InitKaldiOutputStream(std::ostream &os, bool binary) {
185  // This does not throw exceptions (does not check for errors).
186  if (binary) {
187  os.put('\0');
188  os.put('B');
189  }
190  // Note, in non-binary mode we may at some point want to mess with
191  // the precision a bit.
192  // 7 is a bit more than the precision of float..
193  if (os.precision() < 7)
194  os.precision(7);
195 }
