67  virtual bool Open(const std::string &rspecifier) {
68  if (state_ != kUninitialized)
69  if (!Close()) // call Close() yourself to suppress this exception.
70  KALDI_ERR << "TableReader::Open, error closing previous input: "
71  << "rspecifier was " << rspecifier_;
72  bool binary;
73  rspecifier_ = rspecifier;
74  RspecifierType rs = ClassifyRspecifier(rspecifier, &script_rxfilename_,
75  &opts_);
76  KALDI_ASSERT(rs == kScriptRspecifier);
77  if (!script_input_.Open(script_rxfilename_, &binary)) { // Failure on Open
78  KALDI_WARN << "Failed to open script file "
79  << PrintableRxfilename(script_rxfilename_);
80  state_ = kUninitialized;
81  return false;
82  } else { // Open succeeded.
83  if (binary) { // script file should not be binary file..
84  state_ = kError; // bad script file.
85  script_input_.Close();
86  return false;
87  } else {
88  state_ = kFileStart;
89  Next();
90  if (state_ == kError) {
91  script_input_.Close();
92  return false;
93  }
94  if (opts_.permissive) { // Next() will have preloaded.
95  KALDI_ASSERT(state_ == kLoadSucceeded || state_ == kEof);
96  } else {
97  KALDI_ASSERT(state_ == kHaveScpLine || state_ == kEof);
98  }
99  return true; // Success.
100  }
101  }
102  }

104  virtual bool IsOpen() const {
105  switch (state_) {
106  case kEof: case kError: case kHaveScpLine: case kLoadSucceeded:
107  case kLoadFailed: return true;
108  case kUninitialized: return false;

110  // is not valid

116  virtual bool Done() const {
117  switch (state_) {
118  case kHaveScpLine: return false;
119  case kLoadSucceeded: case kLoadFailed: return false;
120  // These cases are because we want LoadCurrent()
121  // to be callable after Next() and to not change the Done() status
122  // [only Next() should change the Done() status].
123  case kEof: case kError: return true; // Error condition, like Eof, counts
124  // as Done(); the destructor/Close() will inform the user of the error.
125  default: KALDI_ERR << "Done() called on TableReader object at the wrong"
126  " time.";
127  return false;
128  }

131  virtual std::string Key() {
132  // Valid to call this whenever Done() returns false.
133  switch (state_) {
134  case kHaveScpLine: case kLoadSucceeded: case kLoadFailed: break;
135  default:
136  // coding error.
137  KALDI_ERR << "Key() called on TableReader object at the wrong time.";
138  }
139  return key_;
140  }
141  const T &Value() {
142  StateType orig_state = state_;
143  if (state_ == kHaveScpLine) LoadCurrent(); // Takes
144  // state_ to kLoadSucceeded or kLoadFailed.
145  if (state_ == kLoadFailed) { // this can happen due to
146  // a file listed in an scp file not existing, or
147  // read failure, failure of a command, etc.
148  if (orig_state == kHaveScpLine)
149  KALDI_ERR << "TableReader: failed to load object from "
150  << PrintableRxfilename(data_rxfilename_)
151  << " (to suppress this error, add the permissive "
152  << "(p, ) option to the rspecifier.";
153 
154  else // orig_state_ was kLoadFailed, which only could have happened
155  // if the user called FreeCurrent().
156  KALDI_ERR << "TableReader: you called Value() after FreeCurrent().";
157  } else if (state_ != kLoadSucceeded) {
158  // This would be a coding error.
159  KALDI_ERR << "TableReader: Value() called at the wrong time.";
160  }
161  return holder_.Value();
162  }
163  void FreeCurrent() {
164  if (state_ == kLoadSucceeded) {
165  holder_.Clear();
166  state_ = kLoadFailed;
167  } else {
168  KALDI_WARN << "TableReader: FreeCurrent called at the wrong time.";
169  }
170  }
171  void Next() {
172  while (1) {
173  NextScpLine();
174  if (Done()) return;
175  if (opts_.permissive) {
176  // Permissive mode means, when reading scp files, we treat keys whose
177  // scp entry cannot be read as nonexistent. This means trying to read.
178  if (LoadCurrent()) return; // Success.
179  // else try the next scp line.
180  } else {
181  return; // We go the next key; Value() will crash if we can't
182  // read the scp line.
183  }
184  }

187  virtual bool Close() {
188  // Close() will succeed if the stream was not in an error
189  // state. To clean up, it also closes the Input objects if
190  // they're open.
191  if (script_input_.IsOpen())
192  script_input_.Close();
193  if (data_input_.IsOpen())
194  data_input_.Close();
195  if (state_ == kLoadSucceeded)
196  holder_.Clear();
197  if (!this->IsOpen())
198  KALDI_ERR << "Close() called on input that was not open.";
199  StateType old_state = state_;
200  state_ = kUninitialized;
201  if (old_state == kError) {
202  if (opts_.permissive) {
203  KALDI_WARN << "Close() called on scp file with read error, ignoring the"
204  " error because permissive mode specified.";
205  return true;
206  } else {
207  return false; // User will do something with the error status.
208  }
209  } else {
210  return true;
211  }

215  if (state_ == kError)
216  KALDI_ERR << "TableReader: reading script file failed: from scp "
217  << PrintableRxfilename(script_rxfilename_);
218  // If you don't want this exception to be thrown you can
219  // call Close() and check the status.

224  bool LoadCurrent() {
225  // Attempts to load object whose rxfilename is on the current scp line.
226  if (state_ != kHaveScpLine)
227  KALDI_ERR << "TableReader: LoadCurrent() called at the wrong time.";
228  bool ans;
229  // note, NULL means it doesn't read the binary-mode header
230  if (Holder::IsReadInBinary()) {
231  ans = data_input_.Open(data_rxfilename_, NULL);
232  } else {

234  }

237  KALDI_WARN << "TableReader: failed to open file "
238  << PrintableRxfilename(data_rxfilename_);
239  state_ = kLoadFailed;
240  return false;
241  } else {
242  if (holder_.Read(data_input_.Stream())) {
243  state_ = kLoadSucceeded;
244  return true;
245  } else { // holder_ will not contain data.
246  KALDI_WARN << "TableReader: failed to load object from "
247  << PrintableRxfilename(data_rxfilename_);
248  state_ = kLoadFailed;
249  return false;
250  }
251  }

254  // Reads the next line in the script file.
255  void NextScpLine() {
256  switch (state_) {
257  case kLoadSucceeded: holder_.Clear(); break;
258  case kHaveScpLine: case kLoadFailed: case kFileStart: break;
259  default:
260  // No other states are valid to call Next() from.
261  KALDI_ERR << "Reading script file: Next called wrongly.";
262  }
263  std::string line;
264  if (getline(script_input_.Stream(), line)) {
265  SplitStringOnFirstSpace(line, &key_, &data_rxfilename_);
266  if (!key_.empty() && !data_rxfilename_.empty()) {
267  // Got a valid line.
268  state_ = kHaveScpLine;
269  } else {
270  // Got an invalid line.
271  state_ = kError; // we can't make sense of this
272  // scp file and will now die.
273  }
274  } else {
275  state_ = kEof; // nothing more in the scp file.
276  // Might as well close the input streams as don't need them.
277  script_input_.Close();
278  if (data_input_.IsOpen())
279  data_input_.Close();
280  }
281  }

313 // This is the implementation for SequentialTableReader
314 // when it's an archive. Note that the archive format is:
315 // key1 [space] object1 key2 [space]
316 // object2 ... eof.
317 // "object1" is the output of the Holder::Write function and will
318 // typically contain a binary header (in binary mode) and then
319 // the output of object.Write(os, binary).
320 // The archive itself does not care whether it is in binary
321 // or text mode, for reading purposes.
322 
323  template<class Holder> class SequentialTableReaderArchiveImpl:
324  public SequentialTableReaderImplBase<Holder> {
325  public:
326  typedef typename Holder::T T;
327 
328  SequentialTableReaderArchiveImpl(): state_(kUninitialized) { }

330  virtual bool Open(const std::string &rspecifier) {
331  if (state_ != kUninitialized) {
332  if (!Close()) { // call Close() yourself to suppress this exception.
333  if (opts_.permissive)
334  KALDI_WARN << "TableReader::Open, error closing previous input "
335  "(only warning, since permissive mode).";
336  else
337  KALDI_ERR << "TableReader::Open, error closing previous input.";
338  }
339  }
340  rspecifier_ = rspecifier;
341  RspecifierType rs = ClassifyRspecifier(rspecifier,
342  &archive_rxfilename_,
343  &opts_);
344  KALDI_ASSERT(rs == kArchiveRspecifier);
345 
346  bool ans;
347  // NULL means don't expect binary-mode header
348  if (Holder::IsReadInBinary())
349  ans = input_.Open(archive_rxfilename_, NULL);
350  else
351  ans = input_.OpenTextMode(archive_rxfilename_);
352  if (!ans) { // header.
353  KALDI_WARN << "TableReader: failed to open stream "
354  << PrintableRxfilename(archive_rxfilename_);
355  state_ = kUninitialized; // Failure on Open
356  return false; // User should print the error message.
357  }
358  state_ = kFileStart;
359  Next();
360  if (state_ == kError) {
361  KALDI_WARN << "Error beginning to read archive file (wrong filename?): "
362  << PrintableRxfilename(archive_rxfilename_);
363  input_.Close();
364  state_ = kUninitialized;
365  return false;
366  }
367  KALDI_ASSERT(state_ == kHaveObject || state_ == kEof);
368  return true;
369  }
370 
371  virtual void Next() {
372  switch (state_) {
373  case kHaveObject:
374  holder_.Clear();
375  break;
376  case kFileStart: case kFreedObject:
377  break;
378  default:
379  KALDI_ERR << "TableReader: Next() called wrongly.";
380  }
381  std::istream &is = input_.Stream();
382  is.clear(); // Clear any fail bits that may have been set... just in case
383  // this happened in the Read function.
384  is >> key_; // This eats up any leading whitespace and gets the string.
385  if (is.eof()) {
386  state_ = kEof;
387  return;
388  }
389  if (is.fail()) { // This shouldn't really happen, barring file-system
390  // errors.
391  KALDI_WARN << "Error reading archive "
392  << PrintableRxfilename(archive_rxfilename_);
393  state_ = kError;
394  return;
395  }
396  int c;
397  if ((c = is.peek()) != ' ' && c != '\t' && c != '\n') { // We expect a
398  // space ' ' after the key.
399  // We also allow tab [which is consumed] and newline [which is not], just
400  // so we can read archives generated by scripts that may not be fully
401  // aware of how this format works.
402  KALDI_WARN << "Invalid archive file format: expected space after key "
403  << key_ << ", got character "
404  << CharToString(static_cast<char>(is.peek())) << ", reading "
405  << PrintableRxfilename(archive_rxfilename_);
406  state_ = kError;
407  return;
408  }
409  if (c != '\n') is.get(); // Consume the space or tab.
410  if (holder_.Read(is)) {
411  state_ = kHaveObject;
412  return;
413  } else {
414  KALDI_WARN << "Object read failed, reading archive "
415  << PrintableRxfilename(archive_rxfilename_);
416  state_ = kError;
417  return;
418  }
419  }

476  if (input_.IsOpen())
477  input_.Close();
478  if (state_ == kHaveObject)

1214 // Types of RandomAccessTableReader:
1215 // In principle, we would like to have four types of RandomAccessTableReader:
1216 // the 4 combinations [scp, archive], [seekable, not-seekable],
1217 // where if something is seekable we only store a file offset. However,
1218 // it seems sufficient for now to only implement two of these, in both
1219 // cases assuming it's not seekable so we never store file offsets and always
1220 // store either the scp line or the data in the archive. The reasons are:
1221 // (1)
1222 // For scp files, storing the actual entry is not that much more expensive
1223 // than storing the file offsets (since the entries are just filenames), and
1224 // avoids a lot of fseek operations that might be expensive.
1225 // (2)
1226 // For archive files, there is no real reason, if you have the archive file
1227 // on disk somewhere, why you wouldn't access it via its associated scp.
1228 // [i.e. write it as ark, scp]. The main reason to read archives directly
1229 // is if they are part of a pipe, and in this case it's not seekable, so
1230 // we implement only this case.
1231 //
1232 // Note that we will rarely in practice have to keep in memory everything in
1233 // the archive, as long as things are only read once from the archive (the
1234 // "o, " or "once" option) and as long as we keep our keys in sorted order;
1235 // to take advantage of this we need the "s, " (sorted) option, so we would
1236 // read archives as e.g. "s, o, ark:-" (this is the rspecifier we would use if
1237 // it was the standard input and these conditions held).
