43 static LangModelFst::LmWeight PrintCompletePath(fst::SymbolTable *pst,
44  fst::StdVectorFst *pfst,
45  fst::StdArc::StateId s,
46  LangModelFst::LmWeight score) {
47  fst::ArcIterator<fst::StdVectorFst> ai(*pfst, s);
48  for (ai.Reset(); !ai.Done(); ai.Next()) {
49  std::cout << pst->Find(ai.Value().ilabel) << " ";
50  fst::StdArc::Weight w = score; // initialize with current score
51  // reset weight to 0 if we are going through the initial state again
52  if (s == pfst->Start()) {
53  w = fst::StdArc::Weight::One();
54  }
55  std::cout << " \tcurrent score " << w;
56  w = fst::Times(w, ai.Value().weight); // add in value from current arc
57  std::cout << " added arc " << ai.Value().weight;
58  fst::StdArc::Weight fw = pfst->Final(ai.Value().nextstate);
59  if (fw != fst::StdArc::Weight::Zero()) {
60  w = fst::Times(w, fw); // add in destination state weight if final
61  std::cout << " added state weight " << w << '\n';
62  }
63  std::cout << '\n';
64  score = PrintCompletePath(pst, pfst, ai.Value().nextstate, w);
65  }
66  // test this after recursive call in case there are arcs out of a final state
67  if (pfst->Final(s) == fst::StdArc::Weight::One()) {
68  // we hit final state, stop there
69  // std::cout << " total score: " << score << '\n';
70  }
71  return score;
72 }

87 static fst::StdVectorFst* CreateGenFst(fst::SymbolTable *pst) {
88  fst::StdArc::StateId initId, midId, finalId;
89  fst::StdVectorFst *genFst = new fst::StdVectorFst;
90  pst->AddSymbol(epsilon); // added if not there
91  int64 boslab = pst->AddSymbol(startOfSentence); // added if not there
92  int64 eoslab = pst->AddSymbol(endOfSentence); // added if not there
93  genFst->SetInputSymbols(pst);
94  genFst->SetOutputSymbols(pst);
95 
96  initId = genFst->AddState();
97  midId = genFst->AddState();
98  finalId = genFst->AddState();
99  genFst->SetStart(initId); // initial state
100  genFst->SetFinal(finalId, fst::StdArc::Weight::One()); // final state
101  genFst->AddArc(initId, fst::StdArc(boslab, boslab, 0, midId));
102  genFst->AddArc(midId, fst::StdArc(eoslab, eoslab, 0, finalId));
103  // add a loop for each symbol except epsilon, begin and end of sentence
104  fst::SymbolTableIterator si(*pst);
105  for (si.Reset(); !si.Done(); si.Next()) {
106  if (si.Value() == boslab ||
107  si.Value() == eoslab ||
108  si.Value() == 0) continue;
109  genFst->AddArc(midId, fst::StdArc(si.Value(), si.Value(), 0, midId));
110  }
111  return genFst;
112 }

115 static fst::StdVectorFst* CreateRandPathFst(int n, fst::StdVectorFst *genFst) {
116  typedef fst::UniformArcSelector<fst::StdArc> UniformSelector;
117 
118  int nTrials = 50;
119  UniformSelector uniform_sel;
120  fst::RandGenOptions<UniformSelector > opts(uniform_sel,
121  MAX_SENTENCE_LENGTH, n);
122 
123  for (int i = 0; i < nTrials; i++) {
124  fst::StdVectorFst *tmpFst = new fst::StdVectorFst;
125  RandGen(*genFst, tmpFst, opts);
126  if (tmpFst->Properties(fst::kCoAccessible, true)) {
127  // std::cout << "Got valid random path after " << i << " tries" << '\n';
128  return tmpFst;
129  }
130  // not good, try another
131  delete tmpFst;
132  }
133  // couldn't generate it within allowed trials
134  std::cerr << " Warning: couldn't generate complete paths within " << nTrials;
135  std::cerr << " trials and " << MAX_SENTENCE_LENGTH << " max length" << '\n';
136  return NULL;
137 }

321 
322  std::cout << "Testing normal small arpa file" << '\n';
323  infile = "input.arpa";
324  success &= kaldi::TestLmTableReadWrite(ntests, infile, outfile);
325 
326  ntests = 2;
327  // note that we use latest value of 'infile' as the tested language model
328  for (int i = 1; i <= ntests; i++) {
329  std::ostringstream intext("");
330  std::ostringstream refscore("");
331  // these inputN.txt sentences have been scored

335  success &= kaldi::TestLmTableEvalScore(infile,

339
