27 void TestLatticeStringRepository() {
28  typedef int32 IntType;
29 
30  LatticeStringRepository<IntType> sr;
31  typedef LatticeStringRepository<IntType>::Entry Entry;
32 
33  for(int i = 0; i < 100; i++) {
34  int len = kaldi::Rand() % 5;
35  vector<IntType> str(len), str2(kaldi::Rand() % 4);
36  const Entry *e = NULL;
37  for(int i = 0; i < len; i++) {
38  str[i] = kaldi::Rand() % 5;
39  e = sr.Successor(e, str[i]);
40  }
41  sr.ConvertToVector(e, &str2);
42  assert(str == str2);
43 
44  int len2 = kaldi::Rand() % 5;
45  str2.resize(len2);
46  const Entry *f = sr.EmptyString(); // NULL
47  for(int i = 0; i < len2; i++) {
48  str2[i] = kaldi::Rand() % 5;
49  f = sr.Successor(f, str2[i]);
50  }
51  vector<IntType> prefix, prefix2(kaldi::Rand() % 10),
52  prefix3;
53  for(int i = 0; i < len && i < len2; i++) {
54  if (str[i] == str2[i]) prefix.push_back(str[i]);
55  else break;
56  }
57  const Entry *g = sr.CommonPrefix(e, f);
58  sr.ConvertToVector(g, &prefix2);
59  sr.ConvertToVector(e, &prefix3);
60  sr.ReduceToCommonPrefix(f, &prefix3);
61  assert(prefix == prefix2);
62  assert(prefix == prefix3);
63  assert(sr.IsPrefixOf(g, e));
64  assert(sr.IsPrefixOf(g, f));
65  if (str.size() > prefix.size())
66  assert(!sr.IsPrefixOf(e, g));
67  }
68 }

71 // test that determinization proceeds correctly on general
72 // FSTs (not guaranteed determinzable, but we use the
73 // max-states option to stop it getting out of control).
74 template<class Arc> void TestDeterminizeLattice() {
75  typedef typename Arc::Weight Weight;
76  typedef int32 Int;
77  typedef ArcTpl<CompactLatticeWeightTpl<Weight, Int> > CompactArc;
78 
79  for(int i = 0; i < 100; i++) {
80  RandFstOptions opts;
81  opts.n_states = 4;
82  opts.n_arcs = 10;
83  opts.n_final = 2;
84  opts.allow_empty = false;
85  opts.weight_multiplier = 0.5; // impt for the randomly generated weights
86  // to be exactly representable in float,
87  // or this test fails because numerical differences can cause symmetry in
88  // weights to be broken, which causes the wrong path to be chosen as far
89  // as the string part is concerned.
90 
91  VectorFst<Arc> *fst = RandFst<Arc>();
92  std::cout << "FST before lattice-determinizing is:\n";
93  {
94 #ifdef HAVE_OPENFST_GE_10400
95  FstPrinter<Arc> fstprinter(*fst, NULL, NULL, NULL, false, true, "\t");
96 #else
97  FstPrinter<Arc> fstprinter(*fst, NULL, NULL, NULL, false, true);
98 #endif
99  fstprinter.Print(&std::cout, "standard output");
100  }
101  VectorFst<Arc> det_fst;
102  try {
103  DeterminizeLatticeOptions lat_opts;
104  lat_opts.max_mem = 100;
105 
106  if (!DeterminizeLattice<TropicalWeight, int32>(*fst, &det_fst, lat_opts, NULL))
107  throw std::runtime_error("could not determinize");
108  std::cout << "FST after lattice-determinizing is:\n";
109  {
110 #ifdef HAVE_OPENFST_GE_10400
111  FstPrinter<Arc> fstprinter(det_fst, NULL, NULL, NULL, false, true, "\t");
112 #else
113  FstPrinter<Arc> fstprinter(det_fst, NULL, NULL, NULL, false, true);
114 #endif
115  fstprinter.Print(&std::cout, "standard output");
116  }
117  assert(det_fst.Properties(kIDeterministic, true) & kIDeterministic);
118  // OK, now determinize it a different way and check equivalence.
119  // [note: it's not normal determinization, it's taking the best path
120  // for any input-symbol sequence....
121  VectorFst<CompactArc> compact_fst, compact_det_fst;
122  ConvertLattice<Weight, Int>(*fst, &compact_fst, false);
123  std::cout << "Compact FST is:\n";
124  {
125 #ifdef HAVE_OPENFST_GE_10400
126  FstPrinter<CompactArc> fstprinter(compact_fst, NULL, NULL, NULL, false, true, "\t");
127 #else
128  FstPrinter<CompactArc> fstprinter(compact_fst, NULL, NULL, NULL, false, true);
129 #endif
130  fstprinter.Print(&std::cout, "standard output");
131  }
132  if (kaldi::Rand() % 2 == 1)
133  ConvertLattice<Weight, Int>(det_fst, &compact_det_fst, false);
134  else
135  if (!DeterminizeLattice<TropicalWeight, int32>(*fst, &compact_det_fst, lat_opts, NULL))
136  throw std::runtime_error("could not determinize");
137 
138  std::cout << "Compact version of determinized FST is:\n";
139  {
140 #ifdef HAVE_OPENFST_GE_10400
141  FstPrinter<CompactArc> fstprinter(compact_det_fst, NULL, NULL, NULL, false, true, "\t");
142 #else
143  FstPrinter<CompactArc> fstprinter(compact_det_fst, NULL, NULL, NULL, false, true);
144 #endif
145  fstprinter.Print(&std::cout, "standard output");
146  }
147 
148  assert(RandEquivalent(compact_det_fst, compact_fst, 5/*paths*/, 0.01/*delta*/, kaldi::Rand()/*seed*/, 100/*path length, max*/));
149  } catch (...) {
150  std::cout << "Failed to lattice-determinize this FST (probably not determinizable)\n";
151  }
152  delete fst;
153  }
154 }
