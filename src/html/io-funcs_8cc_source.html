170 int PeekToken(std::istream &is, bool binary) {
171  if (!binary) is >> std::ws; // consume whitespace.
172  bool read_bracket;
173  if (static_cast<char>(is.peek()) == '<') {
174  read_bracket = true;
175  is.get();

177  read_bracket = false;

183  }

188 void ExpectToken(std::istream &is, bool binary, const char *token) {
189  int pos_at_start = is.tellg();
190  KALDI_ASSERT(token != NULL);
191  CheckToken(token); // make sure it's valid (can be read back)
192  if (!binary) is >> std::ws; // consume whitespace.
193  std::string str;
194  is >> str;
195  is.get(); // consume the space.
196  if (is.fail()) {
197  KALDI_ERR << "Failed to read token [started at file position "
198  << pos_at_start << "], expected " << token;
199  }
200  if (strcmp(str.c_str(), token) != 0) {
201  KALDI_ERR << "Expected token \"" << token << "\", got instead \""
202  << str <<"\".";
203  }
204 }
