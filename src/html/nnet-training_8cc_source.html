26 NnetTrainer::NnetTrainer(const NnetTrainerOptions &config,
27  Nnet *nnet):
28  config_(config),
29  nnet_(nnet),
30  compiler_(*nnet, config_.optimize_config),
31  num_minibatches_processed_(0) {
32  if (config.zero_component_stats)
33  ZeroComponentStats(nnet);
34  if (config.momentum == 0.0 && config.max_param_change == 0.0) {
35  delta_nnet_= NULL;
36  } else {
37  KALDI_ASSERT(config.momentum >= 0.0 &&
38  config.max_param_change >= 0.0);
39  delta_nnet_ = nnet_->Copy();
40  bool is_gradient = false; // setting this to true would disable the
41  // natural-gradient updates.
42  SetZero(is_gradient, delta_nnet_);
43  }
44 }

47 void NnetTrainer::Train(const NnetExample &eg) {
48  bool need_model_derivative = true;
49  ComputationRequest request;
50  GetComputationRequest(*nnet_, eg, need_model_derivative,
51  config_.store_component_stats,
52  &request);
53  const NnetComputation *computation = compiler_.Compile(request);
54 
55  NnetComputer computer(config_.compute_config, *computation,
56  *nnet_,
57  (delta_nnet_ == NULL ? nnet_ : delta_nnet_));
58  // give the inputs to the computer object.
59  computer.AcceptInputs(*nnet_, eg.io);
60  computer.Forward();
61 
62  this->ProcessOutputs(eg, &computer);
63  computer.Backward();
64 
65  if (delta_nnet_ != NULL) {
66  BaseFloat scale = (1.0 - config_.momentum);
67  if (config_.max_param_change != 0.0) {
68  BaseFloat param_delta =
69  std::sqrt(DotProduct(*delta_nnet_, *delta_nnet_)) * scale;
70  if (param_delta > config_.max_param_change) {
71  if (param_delta - param_delta != 0.0) {
72  KALDI_WARN << "Infinite parameter change, will not apply.";
73  SetZero(false, delta_nnet_);
74  } else {
75  scale *= config_.max_param_change / param_delta;
76  KALDI_LOG << "Parameter change too big: " << param_delta << " > "
77  << "--max-param-change=" << config_.max_param_change
78  << ", scaling by " << config_.max_param_change / param_delta;
79  }
80  }
81  }
82  AddNnet(*delta_nnet_, scale, nnet_);
83  ScaleNnet(config_.momentum, delta_nnet_);
84  }
85 }

87 void NnetTrainer::ProcessOutputs(const NnetExample &eg,
88  NnetComputer *computer) {
89  std::vector<NnetIo>::const_iterator iter = eg.io.begin(),
90  end = eg.io.end();
91  for (; iter != end; ++iter) {
92  const NnetIo &io = *iter;
93  int32 node_index = nnet_->GetNodeIndex(io.name);
94  KALDI_ASSERT(node_index >= 0);
95  if (nnet_->IsOutputNode(node_index)) {
96  ObjectiveType obj_type = nnet_->GetNode(node_index).u.objective_type;
97  BaseFloat tot_weight, tot_objf;
98  bool supply_deriv = true;
99  ComputeObjectiveFunction(io.features, obj_type, io.name,
100  supply_deriv, computer,
101  &tot_weight, &tot_objf);
102  objf_info_[io.name].UpdateStats(io.name, config_.print_interval,
103  num_minibatches_processed_++,
104  tot_weight, tot_objf);
105  }
106  }

122 void ObjectiveFunctionInfo::UpdateStats(
123  const std::string &output_name,
124  int32 minibatches_per_phase,
125  int32 minibatch_counter,
126  BaseFloat this_minibatch_weight,
127  BaseFloat this_minibatch_tot_objf) {
128  int32 phase = minibatch_counter / minibatches_per_phase;
129  if (phase != current_phase) {
130  KALDI_ASSERT(phase == current_phase + 1); // or doesn't really make sense.
131  PrintStatsForThisPhase(output_name, minibatches_per_phase);
132  current_phase = phase;
133  tot_weight_this_phase = 0.0;
134  tot_objf_this_phase = 0.0;
135  }
136  tot_weight_this_phase += this_minibatch_weight;
137  tot_objf_this_phase += this_minibatch_tot_objf;
138  tot_weight += this_minibatch_weight;
139  tot_objf += this_minibatch_tot_objf;
140 }

142 void ObjectiveFunctionInfo::PrintStatsForThisPhase(
143  const std::string &output_name,
144  int32 minibatches_per_phase) const {
145  int32 start_minibatch = current_phase * minibatches_per_phase,
146  end_minibatch = start_minibatch + minibatches_per_phase - 1;
147  KALDI_LOG << "Average objective function for '" << output_name
148  << "' for minibatches " << start_minibatch
149  << '-' << end_minibatch << " is "
150  << (tot_objf_this_phase / tot_weight_this_phase) << " over "
151  << tot_weight_this_phase << " frames.";
152 }

154 bool ObjectiveFunctionInfo::PrintTotalStats(const std::string &name) const {
155  KALDI_LOG << "Overall average objective function for '" << name << "' is "

167 void ComputeObjectiveFunction(const GeneralMatrix &supervision,
168  ObjectiveType objective_type,
169  const std::string &output_name,
170  bool supply_deriv,
171  NnetComputer *computer,
172  BaseFloat *tot_weight,
173  BaseFloat *tot_objf) {
174  const CuMatrixBase<BaseFloat> &output = computer->GetOutput(output_name);
175 
176  if (output.NumCols() != supervision.NumCols())
177  KALDI_ERR << "Nnet versus example output dimension (num-classes) "
178  << "mismatch for '" << output_name << "': " << output.NumCols()
179  << " (nnet) vs. " << supervision.NumCols() << " (egs)\n";
180 
181  switch (objective_type) {
182  case kLinear: {
183  // objective is x * y.
184  switch (supervision.Type()) {
185  case kSparseMatrix: {
186  const SparseMatrix<BaseFloat> &post = supervision.GetSparseMatrix();
187  CuSparseMatrix<BaseFloat> cu_post(post);
188  // The cross-entropy objective is computed by a simple dot product,
189  // because after the LogSoftmaxLayer, the output is already in the form
190  // of log-likelihoods that are normalized to sum to one.
191  *tot_weight = cu_post.Sum();
192  *tot_objf = TraceMatSmat(output, cu_post, kTrans);
193  if (supply_deriv) {
194  CuMatrix<BaseFloat> output_deriv(output.NumRows(), output.NumCols(),
195  kUndefined);
196  cu_post.CopyToMat(&output_deriv);
197  computer->AcceptOutputDeriv(output_name, &output_deriv);
198  }
199  break;
200  }
201  case kFullMatrix: {
202  // there is a redundant matrix copy in here if we're not using a GPU
203  // but we don't anticipate this code branch being used in many cases.
204  CuMatrix<BaseFloat> cu_post(supervision.GetFullMatrix());
205  *tot_weight = cu_post.Sum();
206  *tot_objf = TraceMatMat(output, cu_post, kTrans);
207  if (supply_deriv)
208  computer->AcceptOutputDeriv(output_name, &cu_post);
209  break;
210  }
211  case kCompressedMatrix: {
212  Matrix<BaseFloat> post;
213  supervision.GetMatrix(&post);
214  CuMatrix<BaseFloat> cu_post;
215  cu_post.Swap(&post);
216  *tot_weight = cu_post.Sum();
217  *tot_objf = TraceMatMat(output, cu_post, kTrans);
218  if (supply_deriv)
219  computer->AcceptOutputDeriv(output_name, &cu_post);
220  break;
221  }
222  }
223  break;
224  }
225  case kQuadratic: {
226  // objective is -0.5 (x - y)^2
227  CuMatrix<BaseFloat> diff(supervision.NumRows(),
228  supervision.NumCols(),
229  kUndefined);
230  diff.CopyFromGeneralMat(supervision);
231  diff.AddMat(-1.0, output);
232  *tot_weight = diff.NumRows();
233  *tot_objf = -0.5 * TraceMatMat(diff, diff, kTrans);
234  if (supply_deriv)
235  computer->AcceptOutputDeriv(output_name, &diff);
236  break;
237  }
238  default:
239  KALDI_ERR << "Objective function type " << objective_type
240  << " not handled.";
241  }
242 }
