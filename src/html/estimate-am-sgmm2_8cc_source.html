122 void MleAmSgmm2Accs::Read(std::istream &in_stream, bool binary,
123  bool add) {
124  ExpectToken(in_stream, binary, "<SGMMACCS>");
125  ExpectToken(in_stream, binary, "<NUMPDFS>");
126  ReadBasicType(in_stream, binary, &num_pdfs_);
127  ExpectToken(in_stream, binary, "<NUMGROUPS>");
128  ReadBasicType(in_stream, binary, &num_groups_);
129  ExpectToken(in_stream, binary, "<NUMGaussians>");
130  ReadBasicType(in_stream, binary, &num_gaussians_);
131  ExpectToken(in_stream, binary, "<FEATUREDIM>");
132  ReadBasicType(in_stream, binary, &feature_dim_);
133  ExpectToken(in_stream, binary, "<PHONESPACEDIM>");
134  ReadBasicType(in_stream, binary, &phn_space_dim_);
135  ExpectToken(in_stream, binary, "<SPKSPACEDIM>");
136  ReadBasicType(in_stream, binary, &spk_space_dim_);
137 
138  string token;
139  ReadToken(in_stream, binary, &token);
140 
141  while (token != "</SGMMACCS>") {
142  if (token == "<Y>") {
143  Y_.resize(num_gaussians_);
144  for (size_t i = 0; i < Y_.size(); i++) {
145  Y_[i].Read(in_stream, binary, add);
146  }
147  } else if (token == "<Z>") {
148  Z_.resize(num_gaussians_);
149  for (size_t i = 0; i < Z_.size(); i++) {
150  Z_[i].Read(in_stream, binary, add);
151  }
152  } else if (token == "<R>") {
153  R_.resize(num_gaussians_);
154  if (gamma_s_.Dim() == 0) gamma_s_.Resize(num_gaussians_);
155  for (size_t i = 0; i < R_.size(); i++) {
156  R_[i].Read(in_stream, binary, add);
157  }
158  } else if (token == "<S>") {
159  S_.resize(num_gaussians_);
160  for (size_t i = 0; i < S_.size(); i++) {
161  S_[i].Read(in_stream, binary, add);
162  }
163  } else if (token == "<y>") {
164  y_.resize(num_groups_);
165  for (int32 j1 = 0; j1 < num_groups_; j1++) {
166  y_[j1].Read(in_stream, binary, add);
167  }
168  } else if (token == "<gamma>") {
169  gamma_.resize(num_groups_);
170  for (int32 j1 = 0; j1 < num_groups_; j1++) {
171  gamma_[j1].Read(in_stream, binary, add);
172  }
173  // Don't read gamma_s, it's just a temporary variable and
174  // not part of the permanent (non-speaker-specific) accs.
175  } else if (token == "<a>") {
176  a_.resize(num_groups_);
177  for (int32 j1 = 0; j1 < num_groups_; j1++) {
178  a_[j1].Read(in_stream, binary, add);
179  }
180  } else if (token == "<gamma_c>") {
181  gamma_c_.resize(num_pdfs_);
182  for (int32 j2 = 0; j2 < num_pdfs_; j2++) {
183  gamma_c_[j2].Read(in_stream, binary, add);
184  }
185  } else if (token == "<t>") {
186  t_.Read(in_stream, binary, add);
187  } else if (token == "<U>") {
188  U_.resize(num_gaussians_);
189  for (int32 i = 0; i < num_gaussians_; i++) {
190  U_[i].Read(in_stream, binary, add);
191  }
192  } else if (token == "<total_like>") {
193  double total_like;
194  ReadBasicType(in_stream, binary, &total_like);
195  if (add)
196  total_like_ += total_like;
197  else
198  total_like_ = total_like;
199  } else if (token == "<total_frames>") {
200  double total_frames;
201  ReadBasicType(in_stream, binary, &total_frames);
202  if (add)
203  total_frames_ += total_frames;
204  else
205  total_frames_ = total_frames;
206  } else {
207  KALDI_ERR << "Unexpected token '" << token << "' in model file ";
208  }
209  ReadToken(in_stream, binary, &token);
210  }

365 void MleAmSgmm2Accs::ResizeAccumulators(const AmSgmm2 &model,
366  SgmmUpdateFlagsType flags,
367  bool have_spk_vecs) {
368  num_pdfs_ = model.NumPdfs();
369  num_groups_ = model.NumGroups();
370  num_gaussians_ = model.NumGauss();
371  feature_dim_ = model.FeatureDim();
372  phn_space_dim_ = model.PhoneSpaceDim();
373  spk_space_dim_ = model.SpkSpaceDim();
374  total_frames_ = total_like_ = 0;
375 
376  if (flags & (kSgmmPhoneProjections | kSgmmCovarianceMatrix)) {
377  Y_.resize(num_gaussians_);
378  for (int32 i = 0; i < num_gaussians_; i++) {
379  Y_[i].Resize(feature_dim_, phn_space_dim_);
380  }
381  } else {
382  Y_.clear();
383  }
384 
385  if (flags & (kSgmmSpeakerProjections | kSgmmSpeakerWeightProjections)) {
386  gamma_s_.Resize(num_gaussians_);
387  } else {
388  gamma_s_.Resize(0);
389  }
390 
391  if (flags & kSgmmSpeakerProjections) {
392  if (spk_space_dim_ == 0) {
393  KALDI_ERR << "Cannot set up accumulators for speaker projections "
394  << "because speaker subspace has not been set up";
395  }
396  Z_.resize(num_gaussians_);
397  R_.resize(num_gaussians_);
398  for (int32 i = 0; i < num_gaussians_; i++) {
399  Z_[i].Resize(feature_dim_, spk_space_dim_);
400  R_[i].Resize(spk_space_dim_);
401  }
402  } else {
403  Z_.clear();
404  R_.clear();
405  }
406 
407  if (flags & kSgmmCovarianceMatrix) {
408  S_.resize(num_gaussians_);
409  for (int32 i = 0; i < num_gaussians_; i++) {
410  S_[i].Resize(feature_dim_);
411  }
412  } else {
413  S_.clear();
414  }
415 
416  if (flags & (kSgmmPhoneVectors | kSgmmPhoneWeightProjections |
417  kSgmmCovarianceMatrix | kSgmmPhoneProjections)) {
418  gamma_.resize(num_groups_);
419  for (int32 j1 = 0; j1 < num_groups_; j1++) {
420  gamma_[j1].Resize(model.NumSubstatesForGroup(j1), num_gaussians_);
421  }
422  } else {
423  gamma_.clear();
424  }
425 
426  if (flags & (kSgmmPhoneVectors | kSgmmPhoneWeightProjections)
427  && model.HasSpeakerDependentWeights() && have_spk_vecs) { // SSGMM code.
428  a_.resize(num_groups_);
429  for (int32 j1 = 0; j1 < num_groups_; j1++) {
430  a_[j1].Resize(model.NumSubstatesForGroup(j1),
431  num_gaussians_);
432  }
433  } else {
434  a_.clear();
435  }
436 
437  if (flags & kSgmmSpeakerWeightProjections) {
438  KALDI_ASSERT(model.HasSpeakerDependentWeights() &&
439  "remove the flag \"u\" if you don't have u set up.");
440  a_s_.Resize(num_gaussians_);
441  t_.Resize(num_gaussians_, spk_space_dim_);
442  U_.resize(num_gaussians_);
443  for (int32 i = 0; i < num_gaussians_; i++)
444  U_[i].Resize(spk_space_dim_);
445  } else {
446  a_s_.Resize(0);
447  t_.Resize(0, 0);
448  U_.resize(0);
449  }
450 
451  if (true) { // always set up gamma_c_; it's nominally for
452  // estimation of substate weights, but it's also required when
453  // GetStateOccupancies() is called.
454  gamma_c_.resize(num_pdfs_);
455  for (int32 j2 = 0; j2 < num_pdfs_; j2++) {
456  gamma_c_[j2].Resize(model.NumSubstatesForPdf(j2));
457  }
458  }
459 
460 
461  if (flags & kSgmmPhoneVectors) {
462  y_.resize(num_groups_);
463  for (int32 j1 = 0; j1 < num_groups_; j1++) {
464  y_[j1].Resize(model.NumSubstatesForGroup(j1), phn_space_dim_);
465  }
466  } else {
467  y_.clear();
468  }
469 }

471 BaseFloat MleAmSgmm2Accs::Accumulate(const AmSgmm2 &model,
472  const Sgmm2PerFrameDerivedVars &frame_vars,
473  int32 j2,
474  BaseFloat weight,
475  Sgmm2PerSpkDerivedVars *spk_vars) {
476  // Calculate Gaussian posteriors and collect statistics
477  Matrix<BaseFloat> posteriors;
478  BaseFloat log_like = model.ComponentPosteriors(frame_vars, j2, spk_vars, &posteriors);
479  posteriors.Scale(weight);

481  j2, spk_vars);
482  // Note: total_frames_ is incremented in AccumulateFromPosteriors().
483  total_like_ += count * log_like;
484  return log_like;

487 BaseFloat MleAmSgmm2Accs::AccumulateFromPosteriors(
488  const AmSgmm2 &model,
489  const Sgmm2PerFrameDerivedVars &frame_vars,
490  const Matrix<BaseFloat> &posteriors,
491  int32 j2,
492  Sgmm2PerSpkDerivedVars *spk_vars) {
493  double tot_count = 0.0;
494  const vector<int32> &gselect = frame_vars.gselect;
495  // Intermediate variables
496  Vector<BaseFloat> gammat(gselect.size()), // sum of gammas over mix-weight.
497  a_is_part(gselect.size()); //
498  Vector<BaseFloat> xt_jmi(feature_dim_), mu_jmi(feature_dim_),
499  zt_jmi(spk_space_dim_);
500 
501  int32 j1 = model.Pdf2Group(j2);
502  int32 num_substates = model.NumSubstatesForGroup(j1);
503 
504  for (int32 m = 0; m < num_substates; m++) {
505  BaseFloat d_jms = model.GetDjms(j1, m, spk_vars);
506  BaseFloat gammat_jm = 0.0;
507  for (int32 ki = 0; ki < static_cast<int32>(gselect.size()); ki++) {
508  int32 i = gselect[ki];
509 
510  // Eq. (39): gamma_{jmi}(t) = p (j, m, i|t)
511  BaseFloat gammat_jmi = RandPrune(posteriors(ki, m), rand_prune_);
512  if (gammat_jmi == 0.0) continue;
513  gammat(ki) += gammat_jmi;
514  if (gamma_s_.Dim() != 0)
515  gamma_s_(i) += gammat_jmi;
516  gammat_jm += gammat_jmi;
517 
518  // Accumulate statistics for non-zero gaussian posteriors
519  tot_count += gammat_jmi;
520  if (!gamma_.empty()) {
521  // Eq. (40): gamma_{jmi} = \sum_t gamma_{jmi}(t)
522  gamma_[j1](m, i) += gammat_jmi;
523  }
524  if (!y_.empty()) {
525  // Eq. (41): y_{jm} = \sum_{t, i} \gamma_{jmi}(t) z_{i}(t)
526  // Suggestion: move this out of the loop over m
527  y_[j1].Row(m).AddVec(gammat_jmi, frame_vars.zti.Row(ki));
528  }
529  if (!Y_.empty()) {
530  // Eq. (42): Y_{i} = \sum_{t, j, m} \gamma_{jmi}(t) x_{i}(t) v_{jm}^T
531  Y_[i].AddVecVec(gammat_jmi, frame_vars.xti.Row(ki),
532  model.v_[j1].Row(m));
533  }
534  // Accumulate for speaker projections
535  if (!Z_.empty()) {
536  KALDI_ASSERT(spk_space_dim_ > 0);
537  // Eq. (43): x_{jmi}(t) = x_k(t) - M{i} v_{jm}
538  model.GetSubstateMean(j1, m, i, &mu_jmi);
539  xt_jmi.CopyFromVec(frame_vars.xt);
540  xt_jmi.AddVec(-1.0, mu_jmi);
541  // Eq. (44): Z_{i} = \sum_{t, j, m} \gamma_{jmi}(t) x_{jmi}(t) v^{s}'
542  if (spk_vars->v_s.Dim() != 0) // interpret empty v_s as zero.
543  Z_[i].AddVecVec(gammat_jmi, xt_jmi, spk_vars->v_s);
544  // Eq. (49): \gamma_{i}^{(s)} = \sum_{t\in\Tau(s), j, m} gamma_{jmi}
545  // Will be used when you call CommitStatsForSpk(), to update R_.
546  }
547  } // loop over selected Gaussians
548  if (gammat_jm != 0.0) {
549  if (!a_.empty()) { // SSGMM code.
550  KALDI_ASSERT(d_jms > 0);
551  // below is eq. 40 in the MSR techreport. Caution: there
552  // was an error in the original techreport. The index i
553  // in the summation and the quantity \gamma_{jmi}^{(t)}
554  // should be differently named, e.g. i'.
555  a_[j1].Row(m).AddVec(gammat_jm / d_jms, spk_vars->b_is);
556  }
557  if (a_s_.Dim() != 0) { // [SSGMM]
558  KALDI_ASSERT(d_jms > 0);
559  KALDI_ASSERT(!model.w_jmi_.empty());
560  a_s_.AddVec(gammat_jm / d_jms, model.w_jmi_[j1].Row(m));
561  }
562  if (!gamma_c_.empty())
563  gamma_c_[j2](m) += gammat_jm;
564  }
565  } // loop over substates
566 
567  if (!S_.empty()) {
568  for (int32 ki = 0; ki < static_cast<int32>(gselect.size()); ki++) {
569  // Eq. (47): S_{i} = \sum_{t, j, m} \gamma_{jmi}(t) x_{i}(t) x_{i}(t)^T
570  if (gammat(ki) != 0.0) {
571  int32 i = gselect[ki];
572  S_[i].AddVec2(gammat(ki), frame_vars.xti.Row(ki));
573  }
574  }
575  }
576  total_frames_ += tot_count;
577  return tot_count;
578 }

612 void MleAmSgmm2Updater::Update(const MleAmSgmm2Accs &accs,
613  AmSgmm2 *model,
614  SgmmUpdateFlagsType flags) {
615  // Q_{i}, quadratic term for phonetic subspace estimation. Dim is [I][S][S]
616  std::vector< SpMatrix<double> > Q;
617 
618  // Eq (74): S_{i}^{(means)}, scatter of substate mean vectors for estimating
619  // the shared covariance matrices. [Actually this variable contains also the
620  // term -(Y_i M_i^T + M_i Y_I^T).] Dimension is [I][D][D].
621  std::vector< SpMatrix<double> > S_means;
622  std::vector<Matrix<double> > log_a;
623 
624  Vector<double> gamma_i(accs.num_gaussians_);
625  for (int32 j1 = 0; j1 < accs.num_groups_; j1++)
626  gamma_i.AddRowSumMat(1.0, accs.gamma_[j1]); // add sum of rows of
627  // accs.gamma_[j1], to gamma_i.
628 
629  if (flags & kSgmmPhoneProjections)
630  ComputeQ(accs, *model, &Q);
631  if (flags & kSgmmCovarianceMatrix)
632  ComputeSMeans(accs, *model, &S_means);
633  if (!accs.a_.empty())
634  ComputeLogA(accs, &log_a);
635 
636  // quantities used in both vector and weights updates...
637  vector< SpMatrix<double> > H;
638  // "smoothing" matrices, weighted sums of above.
639  SpMatrix<double> H_sm; // weighted sum of H. Used e.g. in renormalizing phonetic space.
640  if ((flags & (kSgmmPhoneVectors | kSgmmPhoneWeightProjections))
641  || options_.renormalize_V)
642  model->ComputeH(&H);
643 
644  BaseFloat tot_impr = 0.0;
645 
646  if (flags & kSgmmPhoneVectors)
647  tot_impr += UpdatePhoneVectors(accs, H, log_a, model);
648  if (flags & kSgmmPhoneProjections) {
649  if (options_.tau_map_M > 0.0)
650  tot_impr += MapUpdateM(accs, Q, gamma_i, model); // MAP adaptation of M
651  else
652  tot_impr += UpdateM(accs, Q, gamma_i, model);
653  }
654  if (flags & kSgmmPhoneWeightProjections)
655  tot_impr += UpdateW(accs, log_a, gamma_i, model);
656  if (flags & kSgmmCovarianceMatrix)
657  tot_impr += UpdateVars(accs, S_means, gamma_i, model);
658  if (flags & kSgmmSubstateWeights)
659  tot_impr += UpdateSubstateWeights(accs, model);
660  if (flags & kSgmmSpeakerProjections)
661  tot_impr += UpdateN(accs, gamma_i, model);
662  if (flags & kSgmmSpeakerWeightProjections)
663  tot_impr += UpdateU(accs, gamma_i, model);
664 
665  if ((flags & kSgmmSpeakerProjections) && (options_.renormalize_N))
666  RenormalizeN(accs, gamma_i, model); // if you renormalize N you have to
667  // alter any speaker vectors you're keeping around, as well.
668  // So be careful with this option.
669 
670  if (options_.renormalize_V)
671  RenormalizeV(accs, model, gamma_i, H);
672 
673  KALDI_LOG << "*Overall auxf improvement, combining all parameters, is "
674  << tot_impr;
675 
676  KALDI_LOG << "***Overall data likelihood is "
677  << (accs.total_like_/accs.total_frames_)
678  << " over " << accs.total_frames_ << " frames.";
679 
680  model->n_.clear(); // has become invalid.
681  model->w_jmi_.clear(); // has become invalid.
682  // we updated the v or w quantities.
683 }

686 void MleAmSgmm2Updater::ComputeQ(const MleAmSgmm2Accs &accs,
687  const AmSgmm2 &model,
688  std::vector< SpMatrix<double> > *Q) {
689  Q->resize(accs.num_gaussians_);
690  for (int32 i = 0; i < accs.num_gaussians_; i++) {
691  (*Q)[i].Resize(accs.phn_space_dim_);
692  for (int32 j1 = 0; j1 < accs.num_groups_; j1++) {
693  for (int32 m = 0; m < model.NumSubstatesForGroup(j1); m++) {
694  if (accs.gamma_[j1](m, i) > 0.0) {
695  (*Q)[i].AddVec2(static_cast<BaseFloat>(accs.gamma_[j1](m, i)),
696  model.v_[j1].Row(m));
697  }
698  }
699  }
700  }
701 }
702 
703 // Compute the S_i^{(means)} quantities (Eq. 74).
704 // Note: we seem to have also included in this variable
705 // the term - (Y_i M_I^T + M_i Y_i^T).
706 void MleAmSgmm2Updater::ComputeSMeans(const MleAmSgmm2Accs &accs,
707  const AmSgmm2 &model,
708  std::vector< SpMatrix<double> > *S_means) {
709  S_means->resize(accs.num_gaussians_);
710  Matrix<double> YM_MY(accs.feature_dim_, accs.feature_dim_);
711  Vector<BaseFloat> mu_jmi(accs.feature_dim_);
712  for (int32 i = 0; i < accs.num_gaussians_; i++) {
713  // YM_MY = - (Y_{i} M_{i}^T)
714  YM_MY.AddMatMat(-1.0, accs.Y_[i], kNoTrans,
715  Matrix<double>(model.M_[i]), kTrans, 0.0);
716  // Add its own transpose: YM_MY = - (Y_{i} M_{i}^T + M_{i} Y_{i}^T)
717  {
718  Matrix<double> M(YM_MY, kTrans);
719  YM_MY.AddMat(1.0, M);
720  }
721  (*S_means)[i].Resize(accs.feature_dim_, kUndefined);
722  (*S_means)[i].CopyFromMat(YM_MY); // Sigma_{i} = -(YM' + MY')
723 
724  for (int32 j1 = 0; j1 < accs.num_groups_; j1++) {
725  for (int32 m = 0; m < model.NumSubstatesForGroup(j1); m++) {
726  if (accs.gamma_[j1](m, i) != 0.0) {
727  // Sigma_{i} += gamma_{jmi} * mu_{jmi}*mu_{jmi}^T
728  mu_jmi.AddMatVec(1.0, model.M_[i], kNoTrans, model.v_[j1].Row(m), 0.0);
729  (*S_means)[i].AddVec2(static_cast<BaseFloat>(accs.gamma_[j1](m, i)), mu_jmi);
730  }
731  }
732  }
733  KALDI_ASSERT(1.0 / (*S_means)[i](0, 0) != 0.0);
734  }

800 void MleAmSgmm2Updater::ComputeLogA(const MleAmSgmm2Accs &accs,
801  std::vector<Matrix<double> > *log_a) {
802  // This computes the logarithm of the statistics a_{jmi} defined
803  // in Eq. 40 of the SSGMM techreport. Although the log of a_{jmi} never
804  // explicitly appears in the techreport, it happens to be more convenient
805  // in the code to use the log of it.
806  // Note: because of the way a is computed, for each (j,m) the
807  // entries over i should always be all zero or all nonzero.
808  int32 num_zeros = 0;
809  KALDI_ASSERT(accs.a_.size() == accs.num_groups_);
810  log_a->resize(accs.num_groups_);
811  for (int32 j1 = 0; j1 < accs.num_groups_; j1++) {
812  int32 num_substates = accs.a_[j1].NumRows();
813  KALDI_ASSERT(num_substates > 0);
814  (*log_a)[j1].Resize(num_substates, accs.num_gaussians_);
815  for (int32 m = 0; m < num_substates; m++) {
816  if (accs.a_[j1](m, 0) == 0.0) { // Zero accs.
817  num_zeros++;
818  if (accs.gamma_[j1].Row(m).Sum() != 0.0)
819  KALDI_WARN << "Inconsistency between a and gamma stats. [BAD!]";
820  // leave the row zero. This means the sub-state saw no stats.
821  } else {
822  (*log_a)[j1].Row(m).CopyFromVec(accs.a_[j1].Row(m));
823  (*log_a)[j1].Row(m).ApplyLog();
824  }
825  }
826  }
827  if (num_zeros != 0)
828  KALDI_WARN << num_zeros
829  << " sub-states with zero \"a\" (and presumably gamma) stats.";
830 }

832 void MleAmSgmm2Updater::UpdatePhoneVectorsInternal(
833  const MleAmSgmm2Accs &accs,
834  const vector< SpMatrix<double> > &H,
835  const vector< Matrix<double> > &log_a,
836  AmSgmm2 *model,
837  double *auxf_impr_ptr,
838  int32 num_threads,
839  int32 thread_id) const {
840 
841  int32 J1 = accs.num_groups_, block_size = (J1 + (num_threads-1)) / num_threads,
842  j1_start = block_size * thread_id,
843  j1_end = std::min(accs.num_groups_, j1_start + block_size);
844 
845  double tot_auxf_impr = 0.0;
846 
847  for (int32 j1 = j1_start; j1 < j1_end; j1++) {
848  for (int32 m = 0; m < model->NumSubstatesForGroup(j1); m++) {
849  double gamma_jm = accs.gamma_[j1].Row(m).Sum();
850  SpMatrix<double> X_jm(accs.phn_space_dim_); // = \sum_i \gamma_{jmi} H_i
851 
852  for (int32 i = 0; i < accs.num_gaussians_; i++) {
853  double gamma_jmi = accs.gamma_[j1](m, i);
854  if (gamma_jmi != 0.0)
855  X_jm.AddSp(gamma_jmi, H[i]);
856  }
857 
858  Vector<double> v_jm_orig(model->v_[j1].Row(m)),
859  v_jm(v_jm_orig);
860 
861  double exact_auxf_start = 0.0, exact_auxf = 0.0, approx_auxf_impr = 0.0;
862  int32 backtrack_iter, max_backtrack = 10;
863  for (backtrack_iter = 0; backtrack_iter < max_backtrack; backtrack_iter++) {
864  // Note: the 1st time we go through this loop we have not yet updated
865  // v_jm and it has the old value; the 2nd time, it has the updated value
866  // and we will typically break at this point, after verifying that
867  // the auxf has improved.
868 
869  // w_jm = softmax([w_{k1}^T ... w_{kD}^T] * v_{jkm}) eq.(7)
870  Vector<double> w_jm(accs.num_gaussians_);
871  w_jm.AddMatVec(1.0, Matrix<double>(model->w_), kNoTrans,
872  v_jm, 0.0);
873  if (!log_a.empty()) w_jm.AddVec(1.0, log_a[j1].Row(m)); // SSGMM techreport eq. 42
874  w_jm.Add(-w_jm.LogSumExp()); // it is now log w_jm
875 
876 
877  exact_auxf = VecVec(w_jm, accs.gamma_[j1].Row(m))
878  + VecVec(v_jm, accs.y_[j1].Row(m))
879  -0.5 * VecSpVec(v_jm, X_jm, v_jm);
880 
881  if (backtrack_iter == 0) {
882  exact_auxf_start = exact_auxf;
883  } else {
884  if (exact_auxf >= exact_auxf_start) {
885  break; // terminate backtracking.
886  } else {
887  KALDI_LOG << "Backtracking computation of v_jm for j = " << j1
888  << " and m = " << m << " because auxf changed by "
889  << (exact_auxf-exact_auxf_start) << " [vs. predicted:] "
890  << approx_auxf_impr;
891  v_jm.AddVec(1.0, v_jm_orig);
892  v_jm.Scale(0.5);
893  }
894  }
895 
896  if (backtrack_iter == 0) { // computing updated value.
897  w_jm.ApplyExp(); // it is now w_jm
898  SpMatrix<double> H_jm(X_jm);
899  Vector<double> g_jm(accs.y_[j1].Row(m));
900  for (int32 i = 0; i < accs.num_gaussians_; i++) {
901  double gamma_jmi = accs.gamma_[j1](m, i);
902  double quadratic_term = std::max(gamma_jmi, gamma_jm * w_jm(i));
903  double scalar = gamma_jmi - gamma_jm * w_jm(i) + quadratic_term
904  * VecVec(model->w_.Row(i), model->v_[j1].Row(m));
905  g_jm.AddVec(scalar, model->w_.Row(i));
906  if (quadratic_term > 1.0e-10) {
907  H_jm.AddVec2(static_cast<BaseFloat>(quadratic_term), model->w_.Row(i));
908  }
909  }
910 
911  SolverOptions opts;
912  opts.name = "v";
913  opts.K = options_.max_cond;
914  opts.eps = options_.epsilon;
915 
916  approx_auxf_impr = SolveQuadraticProblem(H_jm, g_jm, opts, &v_jm);
917  }
918  }
919  double exact_auxf_impr = exact_auxf - exact_auxf_start;
920  tot_auxf_impr += exact_auxf_impr;
921  if (backtrack_iter == max_backtrack) {
922  KALDI_WARN << "Backtracked " << max_backtrack << " times [not updating]";
923  } else {
924  model->v_[j1].Row(m).CopyFromVec(v_jm);
925  }
926 
927  if (j1 < 3 && m < 3) {
928  KALDI_LOG << "Auxf impr for j = " << j1 << " m = " << m << " is "
929  << (exact_auxf_impr/gamma_jm+1.0e-20) << " per frame over "
930  << gamma_jm << " frames.";
931  }
932  }
933  }
934  *auxf_impr_ptr = tot_auxf_impr;
935 }

938 void MleAmSgmm2Updater::RenormalizeV(const MleAmSgmm2Accs &accs,
939  AmSgmm2 *model,
940  const Vector<double> &gamma_i,
941  const vector<SpMatrix<double> > &H) {
942  // Compute H^{(sm)}, the "smoothing" matrix-- average of H's.
943  SpMatrix<double> H_sm(accs.phn_space_dim_);
944  for (int32 i = 0; i < accs.num_gaussians_; i++)
945  H_sm.AddSp(gamma_i(i), H[i]);
946  KALDI_ASSERT(gamma_i.Sum() > 0.0);
947  H_sm.Scale(1.0 / gamma_i.Sum());
948 
949  SpMatrix<double> Sigma(accs.phn_space_dim_);
950  int32 count = 0;
951  for (int32 j1 = 0; j1 < accs.num_groups_; j1++) {
952  for (int32 m = 0; m < model->NumSubstatesForGroup(j1); m++) {
953  count++;
954  Sigma.AddVec2(static_cast<BaseFloat>(1.0), model->v_[j1].Row(m));
955  }
956  }
957  if (!Sigma.IsPosDef()) {
958  KALDI_LOG << "Not renormalizing v because scatter is not positive definite"
959  << " -- maybe first iter?";
960  return;
961  }
962  Sigma.Scale(1.0 / count);
963  KALDI_LOG << "Scatter of vectors v is : ";
964  Sigma.PrintEigs("Sigma");
965 
966  // Want to make variance of v unit and H_sm (like precision matrix) diagonal.
967  TpMatrix<double> L(accs.phn_space_dim_);
968  L.Cholesky(Sigma);
969  TpMatrix<double> LInv(L);
970  LInv.Invert();
971 
972  Matrix<double> tmpL(accs.phn_space_dim_, accs.phn_space_dim_);
973  tmpL.CopyFromTp(L);
974 
975  SpMatrix<double> H_sm_proj(accs.phn_space_dim_);
976  H_sm_proj.AddMat2Sp(1.0, tmpL, kTrans, H_sm, 0.0);
977  // H_sm_proj := L^{T} * H_sm * L.
978  // This is right because we would transform the vectors themselves
979  // by L^{-1}, and H_sm is like the inverse of the vectors,
980  // so it's {L^{-1}}^{-T} = L^T.
981 
982  Matrix<double> U(accs.phn_space_dim_, accs.phn_space_dim_);
983  Vector<double> eigs(accs.phn_space_dim_);
984  H_sm_proj.SymPosSemiDefEig(&eigs, &U, 1.0); // 1.0 means no checking +ve def -> faster
985  KALDI_LOG << "Note on the next diagnostic: the first number is generally not "
986  << "that meaningful as it relates to the static offset";
987  H_sm_proj.PrintEigs("H_sm_proj (Significance of dims in vector space.. note)");
988 
989  // Transform on vectors is U^T L^{-1}.
990  // Why? Because transform on H_sm is T =U^T L^T
991  // and we want T^{-T} by normal rules of vector/covector and we
992  // have (U^T L^T)^{-T} = (L U)^{-1} = U^T L^{-1}.
993  Matrix<double> Trans(accs.phn_space_dim_, accs.phn_space_dim_); // T^{-T}
994  Matrix<double> tmpLInv(accs.phn_space_dim_, accs.phn_space_dim_);
995  tmpLInv.CopyFromTp(LInv);
996  Trans.AddMatMat(1.0, U, kTrans, tmpLInv, kNoTrans, 0.0);
997  Matrix<double> TransInv(Trans);
998  TransInv.Invert(); // T in above...
999 
1000 #ifdef KALDI_PARANOID
1001  {

1074 // Estimate the parameters of a Gaussian prior over the M matrices. There are
1075 // as many mean matrices as UBM size and two covariance matrices for the rows
1076 // of M and columns of M. The prior means M_i are fixed to the unadapted values.
1077 // This is what was done in Lu, et al. "Maximum a posteriori adaptation of
1078 // subspace Gaussian mixture models for cross-lingual speech recognition",
1079 // ICASSP 2012.
1080 void MleAmSgmm2Updater::ComputeMPrior(AmSgmm2 *model) {
1081  KALDI_ASSERT(options_.map_M_prior_iters > 0);
1082  int32 Ddim = model->FeatureDim();
1083  int32 Sdim = model->PhoneSpaceDim();
1084  int32 nGaussians = model->NumGauss();
1085 
1086  // inverse variance of the columns of M: dim is # of rows
1087  model->col_cov_inv_.Resize(Ddim);
1088  // inverse covariance of the rows of M: dim is # of columns
1089  model->row_cov_inv_.Resize(Sdim);
1090 
1091  model->col_cov_inv_.SetUnit();
1092  model->row_cov_inv_.SetUnit();
1093 
1094  if (model->M_prior_.size() == 0) {
1095  model->M_prior_.resize(nGaussians);
1096  for (int32 i = 0; i < nGaussians; i++) {
1097  model->M_prior_[i].Resize(Ddim, Sdim);
1098  model->M_prior_[i].CopyFromMat(model->M_[i]); // We initialize Mpri as this
1099  }
1100  }
1101 
1102  if (options_.full_col_cov || options_.full_row_cov) {
1103  Matrix<double> avg_M(Ddim, Sdim); // average of the Gaussian prior means
1104  for (int32 i = 0; i < nGaussians; i++)
1105  avg_M.AddMat(1.0, Matrix<double>(model->M_prior_[i]));
1106  avg_M.Scale(1.0 / nGaussians);
1107 
1108  Matrix<double> MDiff(Ddim, Sdim);
1109  for (int32 iter = 0; iter < options_.map_M_prior_iters; iter++) {
1110  { // diagnostic block.
1111  double prior_like = -0.5 * nGaussians * (Ddim * Sdim * Log(2 * M_PI)
1112  + Sdim * (-model->row_cov_inv_.LogPosDefDet())
1113  + Ddim * (-model->col_cov_inv_.LogPosDefDet()));
1114  for (int32 i = 0; i < nGaussians; i++) {
1115  MDiff.CopyFromMat(Matrix<double>(model->M_prior_[i]));
1116  MDiff.AddMat(-1.0, avg_M); // MDiff = M_{i} - avg(M)
1117  SpMatrix<double> tmp(Ddim);
1118  // tmp = MDiff.Omega_r^{-1}*MDiff^T.
1119  tmp.AddMat2Sp(1.0, MDiff, kNoTrans,
1120  SpMatrix<double>(model->row_cov_inv_), 0.0);
1121  prior_like -= 0.5 * TraceSpSp(tmp, SpMatrix<double>(model->col_cov_inv_));
1122  }
1123  KALDI_LOG << "Before iteration " << iter
1124  << " of updating prior over M, log like per dimension modeled is "
1125  << prior_like / (nGaussians * Ddim * Sdim);
1126  }
1127 
1128  // First estimate the column covariances (\Omega_r in paper)
1129  if (options_.full_col_cov) {
1130  size_t limited;
1131  model->col_cov_inv_.SetZero();
1132  for (int32 i = 0; i < nGaussians; i++) {
1133  MDiff.CopyFromMat(Matrix<double>(model->M_prior_[i]));
1134  MDiff.AddMat(-1.0, avg_M); // MDiff = M_{i} - avg(M)
1135  // Omega_r += 1/(D*I) * Mdiff * Omega_c^{-1} * Mdiff^T
1136  model->col_cov_inv_.AddMat2Sp(1.0 / (Ddim * nGaussians),
1137  Matrix<BaseFloat>(MDiff), kNoTrans,
1138  model->row_cov_inv_, 1.0);
1139  }
1140  model->col_cov_inv_.PrintEigs("col_cov");
1141  limited = model->col_cov_inv_.LimitCond(options_.max_cond,
1142  true /*invert the matrix*/);
1143  if (limited != 0) {
1144  KALDI_LOG << "Computing column covariances for M: limited " << limited
1145  << " singular values, max condition is "
1146  << options_.max_cond;
1147  }
1148  }
1149 
1150  // Now estimate the row covariances (\Omega_c in paper)
1151  if (options_.full_row_cov) {
1152  size_t limited;
1153  model->row_cov_inv_.SetZero();
1154  for (int32 i = 0; i < nGaussians; i++) {
1155  MDiff.CopyFromMat(Matrix<double>(model->M_prior_[i]));
1156  MDiff.AddMat(-1.0, avg_M); // MDiff = M_{i} - avg(M)
1157  // Omega_c += 1/(S*I) * Mdiff^T * Omega_r^{-1} * Mdiff.
1158  model->row_cov_inv_.AddMat2Sp(1.0 / (Sdim * nGaussians),
1159  Matrix<BaseFloat>(MDiff), kTrans,
1160  model->col_cov_inv_, 1.0);
1161  }
1162  model->row_cov_inv_.PrintEigs("row_cov");
1163  limited = model->row_cov_inv_.LimitCond(options_.max_cond,
1164  true /*invert the matrix*/);
1165  if (limited != 0) {
1166  KALDI_LOG << "Computing row covariances for M: limited " << limited
1167  << " singular values, max condition is "
1168  << options_.max_cond;
1169  }
1170  }
1171  } // end iterations
1172  }

1252 void MleAmSgmm2Updater::UpdateWGetStats(const MleAmSgmm2Accs &accs,
1253  const AmSgmm2 &model,
1254  const Matrix<double> &w,
1255  const std::vector<Matrix<double> > &log_a,

1258  double *tot_like,
1259  int32 num_threads,
1260  int32 thread_id) {
1261 
1262  // Accumulate stats from a block of states (this gets called in parallel).
1263  int32 block_size = (accs.num_groups_ + (num_threads-1)) / num_threads,
1264  j1_start = block_size * thread_id,
1265  j1_end = std::min(accs.num_groups_, j1_start + block_size);
1266 
1267  // Unlike in the report the inner most loop is over Gaussians, where
1268  // per-gaussian statistics are accumulated. This is more memory demanding
1269  // but more computationally efficient, as outer product v_{jvm} v_{jvm}^T
1270  // is computed only once for all gaussians.
1271 
1272  SpMatrix<double> v_vT(accs.phn_space_dim_);
1273 
1274  for (int32 j1 = j1_start; j1 < j1_end; j1++) {
1275  int32 num_substates = model.NumSubstatesForGroup(j1);
1276  Matrix<double> w_j(num_substates, accs.num_gaussians_);
1277  // The linear term and quadratic term for each Gaussian-- two scalars
1278  // for each Gaussian, they appear in the accumulation formulas.
1279  Matrix<double> linear_term(num_substates, accs.num_gaussians_);
1280  Matrix<double> quadratic_term(num_substates, accs.num_gaussians_);
1281  Matrix<double> v_vT_m(num_substates,
1282  (accs.phn_space_dim_*(accs.phn_space_dim_+1))/2);
1283 
1284  // w_jm = softmax([w_{k1}^T ... w_{kD}^T] * v_{jkm}) eq.(7)
1285  Matrix<double> v_j_double(model.v_[j1]);
1286  w_j.AddMatMat(1.0, v_j_double, kNoTrans, w, kTrans, 0.0);
1287  if (!log_a.empty()) w_j.AddMat(1.0, log_a[j1]); // SSGMM techreport eq. 42
1288 
1289  for (int32 m = 0; m < model.NumSubstatesForGroup(j1); m++) {
1290  SubVector<double> w_jm(w_j, m);
1291  double gamma_jm = accs.gamma_[j1].Row(m).Sum();
1292  w_jm.Add(-1.0 * w_jm.LogSumExp());
1293  *tot_like += VecVec(w_jm, accs.gamma_[j1].Row(m));
1294  w_jm.ApplyExp();
1295  v_vT.SetZero();
1296  // v_vT := v_{jkm} v_{jkm}^T
1297  v_vT.AddVec2(static_cast<BaseFloat>(1.0), v_j_double.Row(m));
1298  v_vT_m.Row(m).CopyFromPacked(v_vT); // a bit wasteful, but does not dominate.
1299 
1300  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1301  // Suggestion: g_jkm can be computed more efficiently
1302  // using the Vector/Matrix routines for all i at once
1303  // linear term around cur value.
1304  linear_term(m, i) = accs.gamma_[j1](m, i) - gamma_jm * w_jm(i);
1305  quadratic_term(m, i) = std::max(accs.gamma_[j1](m, i),
1306  gamma_jm * w_jm(i));
1307  }
1308  } // loop over substates
1309  g_i->AddMatMat(1.0, linear_term, kTrans, v_j_double, kNoTrans, 1.0);
1310  F_i->AddMatMat(1.0, quadratic_term, kTrans, v_vT_m, kNoTrans, 1.0);
1311  } // loop over states
1312 }

1315 double MleAmSgmm2Updater::UpdateW(const MleAmSgmm2Accs &accs,
1316  const std::vector<Matrix<double> > &log_a,
1317  const Vector<double> &gamma_i,
1318  AmSgmm2 *model) {
1319  KALDI_LOG << "Updating weight projections";
1320 
1321  // tot_like_{after, before} are totals over multiple iterations,
1322  // not valid likelihoods. but difference is valid (when divided by tot_count).
1323  double tot_predicted_like_impr = 0.0, tot_like_before = 0.0,
1324  tot_like_after = 0.0;
1325 
1326  Matrix<double> g_i(accs.num_gaussians_, accs.phn_space_dim_);
1327  // View F_i as a vector of SpMatrix.
1328  Matrix<double> F_i(accs.num_gaussians_,
1329  (accs.phn_space_dim_*(accs.phn_space_dim_+1))/2);
1330 
1331  Matrix<double> w(model->w_);
1332  double tot_count = gamma_i.Sum();
1333 
1334  for (int iter = 0; iter < options_.weight_projections_iters; iter++) {
1335  F_i.SetZero();
1336  g_i.SetZero();
1337  double k_like_before = 0.0;
1338 
1339  UpdateWClass c(accs, *model, w, log_a, &F_i, &g_i, &k_like_before);
1340  RunMultiThreaded(c);
1341 
1342  Matrix<double> w_orig(w);
1343  double k_predicted_like_impr = 0.0, k_like_after = 0.0;
1344  double min_step = 0.001, step_size;
1345 
1346  SolverOptions opts;
1347  opts.name = "w";
1348  opts.K = options_.max_cond;
1349  opts.eps = options_.epsilon;
1350 
1351  for (step_size = 1.0; step_size >= min_step; step_size /= 2) {
1352  k_predicted_like_impr = 0.0;
1353  k_like_after = 0.0;
1354 
1355  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1356  // auxf is formulated in terms of change in w.
1357  Vector<double> delta_w(accs.phn_space_dim_);
1358  // returns objf impr with step_size = 1,
1359  // but it may not be 1 so we recalculate it.
1360  SpMatrix<double> this_F_i(accs.phn_space_dim_);
1361  this_F_i.CopyFromVec(F_i.Row(i));
1362  SolveQuadraticProblem(this_F_i, g_i.Row(i), opts, &delta_w);
1363 
1364  delta_w.Scale(step_size);
1365  double predicted_impr = VecVec(delta_w, g_i.Row(i)) -
1366  0.5 * VecSpVec(delta_w, this_F_i, delta_w);
1367 
1368  // should never be negative because
1369  // we checked inside SolveQuadraticProblem.
1370  KALDI_ASSERT(predicted_impr >= -1.0e-05);
1371 
1372  if (i < 10)
1373  KALDI_LOG << "Predicted objf impr for w, iter = " << iter
1374  << ", i = " << i << " is "
1375  << (predicted_impr/gamma_i(i)+1.0e-20)
1376  << " per frame over " << gamma_i(i) << " frames.";
1377  k_predicted_like_impr += predicted_impr;
1378  w.Row(i).AddVec(1.0, delta_w);
1379  }
1380  for (int32 j1 = 0; j1 < accs.num_groups_; j1++) {
1381  int32 M = model->NumSubstatesForGroup(j1);
1382  Matrix<double> w_j(M, accs.num_gaussians_);
1383  w_j.AddMatMat(1.0, Matrix<double>(model->v_[j1]), kNoTrans,
1384  w, kTrans, 0.0);
1385  if (!log_a.empty()) w_j.AddMat(1.0, log_a[j1]); // SSGMM techreport eq. 42
1386  for (int32 m = 0; m < M; m++) {
1387  SubVector<double> w_jm(w_j, m);
1388  w_jm.Add(-1.0 * w_jm.LogSumExp());
1389  }
1390  k_like_after += TraceMatMat(w_j, accs.gamma_[j1], kTrans);
1391  }
1392  KALDI_VLOG(2) << "For iteration " << iter << ", updating w gives "
1393  << "predicted per-frame like impr "
1394  << (k_predicted_like_impr / tot_count) << ", actual "
1395  << ((k_like_after - k_like_before) / tot_count) << ", over "
1396  << tot_count << " frames";
1397  if (k_like_after < k_like_before) {
1398  w.CopyFromMat(w_orig); // Undo what we computed.
1399  if (fabs(k_like_after - k_like_before) / tot_count < 1.0e-05) {
1400  k_like_after = k_like_before;
1401  KALDI_WARN << "Not updating weights as not increasing auxf and "
1402  << "probably due to numerical issues (since small change).";
1403  break;
1404  } else {
1405  KALDI_WARN << "Halving step size for weights as likelihood did "
1406  << "not increase";
1407  }
1408  } else {
1409  break;
1410  }
1411  }
1412  if (step_size < min_step) {
1413  // Undo any step as we have no confidence that this is right.
1414  w.CopyFromMat(w_orig);
1415  } else {
1416  tot_predicted_like_impr += k_predicted_like_impr;
1417  tot_like_after += k_like_after;
1418  tot_like_before += k_like_before;
1419  }
1420  }
1421 
1422  model->w_.CopyFromMat(w);
1423  model->w_jmi_.clear(); // invalidated.
1424 
1425  tot_predicted_like_impr /= tot_count;
1426  tot_like_after = (tot_like_after - tot_like_before) / tot_count;
1427  KALDI_LOG << "**Overall objf impr for w is " << tot_predicted_like_impr
1428  << ", actual " << tot_like_after << ", over "
1429  << tot_count << " frames";
1430  return tot_like_after;

1520  const Vector<double> &gamma_i,
1521  AmSgmm2 *model) {

1523  double tot_count = gamma_i.Sum();
1524  if (tot_count == 0) {
1525  KALDI_WARN << "Not renormalizing N, since there are no counts.";
1526  return;
1527  }
1528 
1529  SpMatrix<double> RTot(accs.spk_space_dim_);
1530  // for (int32 i = 0; i < accs.num_gaussians_; i++) {
1531  // RTot.AddSp(1.0, accs.R_[i]);
1532  // }
1533  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1534  RTot.AddSp(gamma_i(i), accs.R_[i]);
1535  }
1536  RTot.Scale(1.0 / tot_count);
1537  Matrix<double> U(accs.spk_space_dim_, accs.spk_space_dim_);
1538  Vector<double> eigs(accs.spk_space_dim_);

1541  Vector<double> sqrteigs(accs.spk_space_dim_);
1542  for (int32 t = 0; t < accs.spk_space_dim_; t++) {
1543  sqrteigs(t) = sqrt(eigs(t));
1544  }
1545  // e.g. diag(eigs)^{-0.5} * U' * RTot * U * diag(eigs)^{-0.5} = 1

1547  // i.e. not (on left: diag(eigs)^{-0.5} * U')
1548  // but: (inverse it: U . diag(eigs)^{0.5},
1549  // transpose it: diag(eigs)^{0.5} U^T. Need to do this on the right to N
1550  // (because N has the spk vecs on the right), so N := N U diag(eigs)^{0.5}
1551  U.MulColsVec(sqrteigs);
1552  Matrix<double> Ntmp(accs.feature_dim_, accs.spk_space_dim_);
1553  for (int32 i = 0; i < accs.num_gaussians_; i++) {

1560 double MleAmSgmm2Updater::UpdateVars(const MleAmSgmm2Accs &accs,
1561  const std::vector< SpMatrix<double> > &S_means,
1562  const Vector<double> &gamma_i,
1563  AmSgmm2 *model) {
1564  KALDI_ASSERT(S_means.size() == static_cast<size_t>(accs.num_gaussians_));
1565 
1566  SpMatrix<double> Sigma_i(accs.feature_dim_), Sigma_i_ml(accs.feature_dim_);
1567  double tot_objf_impr = 0.0, tot_t = 0.0;
1568  SpMatrix<double> covfloor(accs.feature_dim_);
1569  Vector<double> objf_improv(accs.num_gaussians_);
1570 
1571  // First pass over all (shared) Gaussian components to calculate the
1572  // ML estimate of the covariances, and the total covariance for flooring.
1573  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1574  // Eq. (75): Sigma_{i}^{ml} = 1/gamma_{i} [S_{i} + S_{i}^{(means)} - ...
1575  // Y_{i} M_{i}^T - M_{i} Y_{i}^T]
1576  // Note the S_means already contains the Y_{i} M_{i}^T terms.
1577  Sigma_i_ml.CopyFromSp(S_means[i]);
1578  Sigma_i_ml.AddSp(1.0, accs.S_[i]);
1579 
1580  covfloor.AddSp(1.0, Sigma_i_ml);
1581  // inverting small values e.g. 4.41745328e-40 seems to generate inf,
1582  // although would be fixed up later.
1583  if (gamma_i(i) > 1.0e-20) {
1584  Sigma_i_ml.Scale(1 / (gamma_i(i) + 1.0e-20));
1585  } else {
1586  Sigma_i_ml.SetUnit();
1587  }
1588  KALDI_ASSERT(1.0 / Sigma_i_ml(0, 0) != 0.0);
1589  // Eq. (76): Compute the objective function with the old parameter values
1590  objf_improv(i) = model->SigmaInv_[i].LogPosDefDet() -
1591  TraceSpSp(SpMatrix<double>(model->SigmaInv_[i]), Sigma_i_ml);
1592 
1593  model->SigmaInv_[i].CopyFromSp(Sigma_i_ml); // inverted in the next loop.
1594  }
1595 
1596  // Compute the covariance floor.
1597  if (gamma_i.Sum() == 0) { // If no count, use identity.
1598  KALDI_WARN << "Updating variances: zero counts. Setting floor to unit.";
1599  covfloor.SetUnit();
1600  } else { // else, use the global average covariance.
1601  covfloor.Scale(options_.cov_floor / gamma_i.Sum());
1602  int32 tmp;
1603  if ((tmp = covfloor.LimitCondDouble(options_.max_cond)) != 0) {
1604  KALDI_WARN << "Covariance flooring matrix is poorly conditioned. Fixed "
1605  << "up " << tmp << " eigenvalues.";
1606  }
1607  }
1608 
1609  if (options_.cov_diag_ratio > 1000) {
1610  KALDI_LOG << "Assuming you want to build a diagonal system since "
1611  << "cov_diag_ratio is large: making diagonal covFloor.";
1612  for (int32 i = 0; i < covfloor.NumRows(); i++)
1613  for (int32 j = 0; j < i; j++)
1614  covfloor(i, j) = 0.0;
1615  }
1616 
1617  // Second pass over all (shared) Gaussian components to calculate the
1618  // floored estimate of the covariances, and update the model.
1619  for (int32 i = 0; i < accs.num_gaussians_; i++) {
1620  Sigma_i.CopyFromSp(model->SigmaInv_[i]);
1621  Sigma_i_ml.CopyFromSp(Sigma_i);
1622  // In case of insufficient counts, make the covariance matrix diagonal.
1623  // cov_diag_ratio is 2 by default, set to very large to always get diag-cov
1624  if (gamma_i(i) < options_.cov_diag_ratio * accs.feature_dim_) {
1625  KALDI_WARN << "For Gaussian component " << i << ": Too low count "
1626  << gamma_i(i) << " for covariance matrix estimation. Setting to "
1627  << "diagonal";
1628  for (int32 d = 0; d < accs.feature_dim_; d++)
1629  for (int32 e = 0; e < d; e++)
1630  Sigma_i(d, e) = 0.0; // SpMatrix, can only set lower triangular part
1631 
1632  int floored = Sigma_i.ApplyFloor(covfloor);
1633  if (floored > 0) {
1634  KALDI_WARN << "For Gaussian component " << i << ": Floored " << floored
1635  << " covariance eigenvalues.";
1636  }
1637  model->SigmaInv_[i].CopyFromSp(Sigma_i);
1638  model->SigmaInv_[i].InvertDouble();
1639  } else { // Updating the full covariance matrix.
1640  try {
1641  int floored = Sigma_i.ApplyFloor(covfloor);
1642  if (floored > 0) {
1643  KALDI_WARN << "For Gaussian component " << i << ": Floored "
1644  << floored << " covariance eigenvalues.";
1645  }
1646  model->SigmaInv_[i].CopyFromSp(Sigma_i);
1647  model->SigmaInv_[i].InvertDouble();
1648 
1649  objf_improv(i) += Sigma_i.LogPosDefDet() +
1650  TraceSpSp(SpMatrix<double>(model->SigmaInv_[i]), Sigma_i_ml);
1651  objf_improv(i) *= (-0.5 * gamma_i(i)); // Eq. (76)
1652 
1653  tot_objf_impr += objf_improv(i);
1654  tot_t += gamma_i(i);
1655  if (i < 5) {
1656  KALDI_VLOG(2) << "objf impr from variance update =" << objf_improv(i)
1657  / (gamma_i(i) + 1.0e-20) << " over " << (gamma_i(i))
1658  << " frames for i = " << (i);
1659  }
1660  } catch(...) {
1661  KALDI_WARN << "Updating within-class covariance matrix i = " << (i)
1662  << ", numerical problem";
1663  // This is a catch-all thing in case of unanticipated errors, but
1664  // flooring should prevent this occurring for the most part.
1665  model->SigmaInv_[i].SetUnit(); // Set to unit.
1666  }
1667  }
1668  }
1669  KALDI_LOG << "**Overall objf impr for variance update = "
1670  << (tot_objf_impr / (tot_t+ 1.0e-20))
1671  << " over " << tot_t << " frames";
1672  return tot_objf_impr / (tot_t + 1.0e-20);
1673 }

1851 // Basic update, no SSGMM.
1852 void MleSgmm2SpeakerAccs::UpdateWithU(const AmSgmm2 &model,
1853  Vector<BaseFloat> *v_s_ptr,
1854  BaseFloat *objf_impr_out,
1855  BaseFloat *count_out) {
1856  double tot_gamma = gamma_s_.Sum();
1857  KALDI_ASSERT(y_s_.Dim() != 0);
1858  int32 T = y_s_.Dim(); // speaker-subspace dim.
1859  int32 num_gauss = gamma_s_.Dim();
1860  if (v_s_ptr->Dim() != T) v_s_ptr->Resize(T); // will set it to zero.
1861 
1862  // Eq. (84): H^{(s)} = \sum_{i} \gamma_{i}(s) H_{i}^{spk}
1863  SpMatrix<double> H_s(T);
1864 
1865  for (int32 i = 0; i < num_gauss; i++)
1866  H_s.AddSp(gamma_s_(i), H_spk_[i]);
1867 
1868  Vector<double> v_s(*v_s_ptr);
1869  int32 num_iters = 5, // don't set this to 1, as we discard last iter.
1870  num_backtracks = 0,
1871  max_backtracks = 10;
1872  Vector<double> auxf(num_iters);
1873  Matrix<double> v_s_per_iter(num_iters, T);
1874  // The update for v^{(s)} is the one described in the technical report
1875  // section 5.1 (eq. 33 and below).
1876 
1877  for (int32 iter = 0; iter < num_iters; iter++) { // converges very fast,
1878  // and each iteration is fast, so don't need to make this configurable.
1879  v_s_per_iter.Row(iter).CopyFromVec(v_s);
1880 
1881  SpMatrix<double> F(H_s); // the 2nd-order quadratic term on this iteration...
1882  // F^{(p)} in the techerport.
1883  Vector<double> g(y_s_); // g^{(p)} in the techreport.
1884  g.AddSpVec(-1.0, H_s, v_s, 1.0);
1885  Vector<double> log_b_is(num_gauss); // b_i^{(s)}, indexed by i.
1886  log_b_is.AddMatVec(1.0, Matrix<double>(model.u_), kNoTrans, v_s, 0.0);
1887  Vector<double> tilde_w_is(log_b_is);
1888  Vector<double> log_a_s_(a_s_);
1889  log_a_s_.ApplyLog();
1890  tilde_w_is.AddVec(1.0, log_a_s_);
1891  tilde_w_is.Add(-1.0 * tilde_w_is.LogSumExp()); // normalize.
1892  // currently tilde_w_is is in log form.
1893  auxf(iter) = VecVec(v_s, y_s_) - 0.5 * VecSpVec(v_s, H_s, v_s)
1894  + VecVec(gamma_s_, tilde_w_is); // "new" term (weights)
1895 
1896  if (iter > 0 && auxf(iter) < auxf(iter-1) &&
1897  !ApproxEqual(auxf(iter), auxf(iter-1))) { // auxf did not improve.
1898  // backtrack halfway, and do this iteration again.
1899  KALDI_WARN << "Backtracking in speaker vector update, on iter "
1900  << iter << ", auxfs are " << auxf(iter-1) << " -> "
1901  << auxf(iter);
1902  v_s.Scale(0.5);
1903  v_s.AddVec(0.5, v_s_per_iter.Row(iter-1));
1904  if (++num_backtracks >= max_backtracks) {
1905  KALDI_WARN << "Backtracked " << max_backtracks
1906  << " times in speaker-vector update.";
1907  // backtrack all the way, and terminate:
1908  v_s_per_iter.Row(num_iters-1).CopyFromVec(v_s_per_iter.Row(iter-1));
1909  // the following statement ensures we will get
1910  // the appropriate auxiliary-function.
1911  auxf(num_iters-1) = auxf(iter-1);
1912  break;
1913  }
1914  iter--;
1915  }
1916  tilde_w_is.ApplyExp();
1917  for (int32 i = 0; i < num_gauss; i++) {
1918  g.AddVec(gamma_s_(i) - tot_gamma * tilde_w_is(i), model.u_.Row(i));
1919  F.AddVec2(tot_gamma * tilde_w_is(i), model.u_.Row(i));
