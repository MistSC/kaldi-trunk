77  void InitData(std::istream &is) {
78  // define options
79  BaseFloat bias_mean = -2.0, bias_range = 2.0, param_stddev = 0.1;
80  // parse config
81  std::string token;
82  while (!is.eof()) {
83  ReadToken(is, false, &token);
84  if (token == "<ParamStddev>") ReadBasicType(is, false, &param_stddev);
85  else if (token == "<BiasMean>") ReadBasicType(is, false, &bias_mean);
86  else if (token == "<BiasRange>") ReadBasicType(is, false, &bias_range);
87  else if (token == "<PatchDim>") ReadBasicType(is, false, &patch_dim_);
88  else if (token == "<PatchStep>") ReadBasicType(is, false, &patch_step_);
89  else if (token == "<PatchStride>") ReadBasicType(is, false, &patch_stride_);
90  else if (token == "<LearnRateCoef>") ReadBasicType(is, false, &learn_rate_coef_);
91  else if (token == "<BiasLearnRateCoef>") ReadBasicType(is, false, &bias_learn_rate_coef_);
92  else if (token == "<MaxNorm>") ReadBasicType(is, false, &max_norm_);
93  else KALDI_ERR << "Unknown token " << token << ", a typo in config?"
94  << " (ParamStddev|BiasMean|BiasRange|PatchDim|PatchStep|PatchStride)";
95  is >> std::ws; // eat-up whitespace
96  }
97 
98  //
99  // Sanity checks:
100  //
101  // splice (input are spliced frames):
102  KALDI_ASSERT(input_dim_ % patch_stride_ == 0);
103  int32 num_splice = input_dim_ / patch_stride_;
104  KALDI_LOG << "num_splice " << num_splice;
105  // number of patches:
106  KALDI_ASSERT((patch_stride_ - patch_dim_) % patch_step_ == 0);
107  int32 num_patches = 1 + (patch_stride_ - patch_dim_) / patch_step_;
108  KALDI_LOG << "num_patches " << num_patches;
109  // filter dim:
110  int32 filter_dim = num_splice * patch_dim_;
111  KALDI_LOG << "filter_dim " << filter_dim;
112  // num filters:
113  KALDI_ASSERT(output_dim_ % num_patches == 0);
114  int32 num_filters = output_dim_ / num_patches;
115  KALDI_LOG << "num_filters " << num_filters;
116  //
117 
118  //
119  // Initialize parameters
120  //
121  Matrix<BaseFloat> mat(num_filters, filter_dim);
122  for(int32 r=0; r<num_filters; r++) {
123  for(int32 c=0; c<filter_dim; c++) {
124  mat(r,c) = param_stddev * RandGauss(); // 0-mean Gauss with given std_dev
125  }
126  }
127  filters_ = mat;
128  //
129  Vector<BaseFloat> vec(num_filters);
130  for(int32 i=0; i<num_filters; i++) {
131  // +/- 1/2*bias_range from bias_mean:
132  vec(i) = bias_mean + (RandUniform() - 0.5) * bias_range;
133  }
134  bias_ = vec;
135  //
136  }

233  void PropagateFnc(const CuMatrixBase<BaseFloat> &in,
234  CuMatrixBase<BaseFloat> *out) {
235  // useful dims
236  int32 num_splice = input_dim_ / patch_stride_;
237  int32 num_patches = 1 + (patch_stride_ - patch_dim_) / patch_step_;
238  int32 num_filters = filters_.NumRows();
239  int32 num_frames = in.NumRows();
240  int32 filter_dim = filters_.NumCols();
241 
242  // we will need the buffers
243  if (vectorized_feature_patches_.NumRows() != num_frames) {
244  vectorized_feature_patches_.Resize(num_frames,
245  filter_dim * num_patches, kUndefined);
246  feature_patch_diffs_.Resize(num_frames, filter_dim * num_patches, kSetZero);
247  }
248 
249  /* Prepare feature patches, the layout is:
250  * |----------|----------|----------|---------| (in = spliced frames)
251  * xxx xxx xxx xxx (x = selected elements)
252  *
253  * xxx : patch dim
254  * xxx
255  * ^---: patch step
256  * |----------| : patch stride
257  *
258  * xxx-xxx-xxx-xxx : filter dim
259  *
260  */
261  // build-up a column selection map:
262  column_map_.resize(filter_dim * num_patches);
263  for (int32 p=0, index=0; p<num_patches; p++) {
264  for (int32 s=0; s<num_splice; s++) {
265  for (int32 d=0; d<patch_dim_; d++, index++) {
266  column_map_[index] = p * patch_step_ + s * patch_stride_ + d;
267  }
268  }
269  }
270  // select the columns
271  CuArray<int32> cu_column_map(column_map_);
272  vectorized_feature_patches_.CopyCols(in, cu_column_map);
273 
274  // compute filter activations
275  for (int32 p=0; p<num_patches; p++) {
276  CuSubMatrix<BaseFloat> tgt(out->ColRange(p * num_filters, num_filters));
277  CuSubMatrix<BaseFloat> patch(vectorized_feature_patches_.ColRange(
278  p * filter_dim, filter_dim));
279  tgt.AddVecToRows(1.0, bias_, 0.0); // add bias
280  // apply all filters
281  tgt.AddMatMat(1.0, patch, kNoTrans, filters_, kTrans, 1.0);
282  }
283  }

285  /*
286  This function does an operation similar to reversing a map,
287  except it handles maps that are not one-to-one by outputting
288  the reversed map as a vector of lists.
289  @param[in] forward_indexes is a vector of int32, each of whose
290  elements is between 0 and input_dim - 1.
291  @param[in] input_dim. See definitions of forward_indexes and
292  backward_indexes.
293  @param[out] backward_indexes is a vector of dimension input_dim
294  of lists, The list at (backward_indexes[i]) is a list
295  of all indexes j such that forward_indexes[j] = i.
296  */
297  void ReverseIndexes(const std::vector<int32> &forward_indexes,
298  std::vector<std::vector<int32> > *backward_indexes) {
299  int32 i;
300  int32 size = forward_indexes.size();
301  backward_indexes->resize(input_dim_);
302  int32 reserve_size = 2+ forward_indexes.size() / input_dim_;
303  std::vector<std::vector<int32> >::iterator iter = backward_indexes->begin(),
304  end = backward_indexes->end();
305  for (; iter != end; ++iter)
306  iter->reserve(reserve_size);
307  for (int32 j = 0; j < size; j++) {
308  i = forward_indexes[j];
309  KALDI_ASSERT(i < input_dim_);
310  (*backward_indexes)[i].push_back(j);
311  }
312  }

314  /*
315  This function transforms a vector of lists into a list of vectors,
316  padded with -1.
317  @param[in] The input vector of lists. Let in.size() be D, and let
318  the longest list length (i.e. the max of in[i].size()) be L.
319  @param[out] The output list of vectors. The length of the list will
320  be L, each vector-dimension will be D (i.e. out[i].size() == D),
321  and if in[i] == j, then for some k we will have that
322  out[k][j] = i. The output vectors are padded with -1
323  where necessary if not all the input lists have the same side.
324  */
325  void RearrangeIndexes(const std::vector<std::vector<int32> > &in,
326  std::vector<std::vector<int32> > *out) {
327  int32 D = in.size();
328  int32 L = 0;
329  for (int32 i = 0; i < D; i++)
330  if (in[i].size() > L)
331  L = in[i].size();
332  out->resize(L);
333  for (int32 i = 0; i < L; i++)
334  (*out)[i].resize(D, -1);
335  for (int32 i = 0; i < D; i++) {
336  for (int32 j = 0; j < in[i].size(); j++) {
337  (*out)[j][i] = in[i][j];
338  }
339  }
340  }

342  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in,
343  const CuMatrixBase<BaseFloat> &out,
344  const CuMatrixBase<BaseFloat> &out_diff,
345  CuMatrixBase<BaseFloat> *in_diff) {
346  // useful dims
347  int32 num_patches = 1 + (patch_stride_ - patch_dim_) / patch_step_;
348  int32 num_filters = filters_.NumRows();
349  int32 filter_dim = filters_.NumCols();
350 
351  // backpropagate to vector of matrices
352  // (corresponding to position of a filter)
353  for (int32 p=0; p<num_patches; p++) {
354  CuSubMatrix<BaseFloat> patch_diff(feature_patch_diffs_.ColRange(
355  p * filter_dim, filter_dim));
356  CuSubMatrix<BaseFloat> out_diff_patch(out_diff.ColRange(
357  p * num_filters, num_filters));
358  patch_diff.AddMatMat(1.0, out_diff_patch, kNoTrans,
359  filters_, kNoTrans, 0.0);
360  }
361 
362  // sum the derivatives into in_diff, we will compensate #summands
363  std::vector<std::vector<int32> > reversed_column_map;
364  ReverseIndexes(column_map_, &reversed_column_map);
365  std::vector<std::vector<int32> > rearranged_column_map;
366  RearrangeIndexes(reversed_column_map, &rearranged_column_map);
367  for (int32 p = 0; p < rearranged_column_map.size(); p++) {
368  CuArray<int32> cu_cols(rearranged_column_map[p]);
369  in_diff->AddCols(feature_patch_diffs_, cu_cols);
370  }
371  }

374  void Update(const CuMatrixBase<BaseFloat> &input,
375  const CuMatrixBase<BaseFloat> &diff) {
376  // useful dims
377  int32 num_patches = 1 + (patch_stride_ - patch_dim_) / patch_step_;
378  int32 num_filters = filters_.NumRows();
379  int32 filter_dim = filters_.NumCols();
380 
381  // we use following hyperparameters from the option class
382  const BaseFloat lr = opts_.learn_rate;
383 
384  //
385  // calculate the gradient
386  //
387  filters_grad_.Resize(num_filters, filter_dim, kSetZero); // reset
388  bias_grad_.Resize(num_filters, kSetZero); // reset
389  // use all the patches
390  for (int32 p=0; p<num_patches; p++) { // sum
391  CuSubMatrix<BaseFloat> diff_patch(diff.ColRange(p * num_filters,
392  num_filters));
393  CuSubMatrix<BaseFloat> patch(vectorized_feature_patches_.ColRange(
394  p * filter_dim, filter_dim));
395  filters_grad_.AddMatMat(1.0, diff_patch, kTrans, patch, kNoTrans, 1.0);
396  bias_grad_.AddRowSumMat(1.0, diff_patch, 1.0);
397  }
398 
399  //
400  // update
401  //
402  filters_.AddMat(-lr*learn_rate_coef_, filters_grad_);
403  bias_.AddVec(-lr*bias_learn_rate_coef_, bias_grad_);
404  //
405 
406  // max-norm
407  if (max_norm_ > 0.0) {
408  CuMatrix<BaseFloat> lin_sqr(filters_);
409  lin_sqr.MulElements(filters_);
410  CuVector<BaseFloat> l2(filters_.NumRows());
411  l2.AddColSumMat(1.0, lin_sqr, 0.0);
412  l2.ApplyPow(0.5); // we have per-neuron L2 norms
413  CuVector<BaseFloat> scl(l2);
414  scl.Scale(1.0/max_norm_);
415  scl.ApplyFloor(1.0);
416  scl.InvertElements();
417  filters_.MulRowsVec(scl); // shink to sphere!
418  }
419 
420  }
