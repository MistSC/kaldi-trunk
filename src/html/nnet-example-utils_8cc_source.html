29 // get a sorted list of all NnetIo names in all examples in the list (will
30 // normally be just the strings "input" and "output", but maybe also "ivector").
31 static void GetIoNames(const std::vector<NnetExample> &src,
32  std::vector<std::string> *names_vec) {
33  std::set<std::string> names;
34  std::vector<NnetExample>::const_iterator iter = src.begin(), end = src.end();
35  for (; iter != end; ++iter) {
36  std::vector<NnetIo>::const_iterator iter2 = iter->io.begin(),
37  end2 = iter->io.end();
38  for (; iter2 != end2; ++iter2)
39  names.insert(iter2->name);
40  }
41  CopySetToVector(names, names_vec);
42 }

44 // Get feature "sizes" for each NnetIo name, which are the total number of
45 // Indexes for that NnetIo (needed to correctly size the output matrix). Also
46 // make sure the dimensions are consistent for each name.
47 static void GetIoSizes(const std::vector<NnetExample> &src,
48  const std::vector<std::string> &names,
49  std::vector<int32> *sizes) {
50  std::vector<int32> dims(names.size(), -1); // just for consistency checking.
51  sizes->clear();
52  sizes->resize(names.size(), 0);
53  std::vector<std::string>::const_iterator names_begin = names.begin(),
54  names_end = names.end();
55  std::vector<NnetExample>::const_iterator iter = src.begin(), end = src.end();
56  for (; iter != end; ++iter) {
57  std::vector<NnetIo>::const_iterator iter2 = iter->io.begin(),
58  end2 = iter->io.end();
59  for (; iter2 != end2; ++iter2) {
60  const NnetIo &io = *iter2;
61  std::vector<std::string>::const_iterator names_iter =
62  std::lower_bound(names_begin, names_end, io.name);
63  KALDI_ASSERT(*names_iter == io.name);
64  int32 i = names_iter - names_begin;
65  int32 this_dim = io.features.NumCols();
66  if (dims[i] == -1)
67  dims[i] = this_dim;
68  else if(dims[i] != this_dim) {
69  KALDI_ERR << "Merging examples with inconsistent feature dims: "
70  << dims[i] << " vs. " << this_dim << " for '"
71  << io.name << "'.";
72  }
73  KALDI_ASSERT(io.features.NumRows() == io.indexes.size());
74  int32 this_size = io.indexes.size();
75  (*sizes)[i] += this_size;
76  }
77  }
78 }

83 // Do the final merging of NnetIo, once we have obtained the names, dims and
84 // sizes for each feature/supervision type.
85 static void MergeIo(const std::vector<NnetExample> &src,
86  const std::vector<std::string> &names,
87  const std::vector<int32> &sizes,
88  bool compress,
89  NnetExample *merged_eg) {
90  int32 num_feats = names.size();
91  std::vector<int32> cur_size(num_feats, 0);
92  std::vector<std::vector<GeneralMatrix const*> > output_lists(num_feats);
93  merged_eg->io.clear();
94  merged_eg->io.resize(num_feats);
95  for (int32 f = 0; f < num_feats; f++) {
96  NnetIo &io = merged_eg->io[f];
97  int32 size = sizes[f];
98  KALDI_ASSERT(size > 0);
99  io.name = names[f];
100  io.indexes.resize(size);
101  }
102 
103  std::vector<std::string>::const_iterator names_begin = names.begin(),
104  names_end = names.end();
105  std::vector<NnetExample>::const_iterator iter = src.begin(), end = src.end();
106  for (int32 n = 0; iter != end; ++iter,++n) {
107  std::vector<NnetIo>::const_iterator iter2 = iter->io.begin(),
108  end2 = iter->io.end();
109  for (; iter2 != end2; ++iter2) {
110  const NnetIo &io = *iter2;
111  std::vector<std::string>::const_iterator names_iter =
112  std::lower_bound(names_begin, names_end, io.name);
113  KALDI_ASSERT(*names_iter == io.name);
114  int32 f = names_iter - names_begin;
115  int32 this_size = io.indexes.size(),
116  &this_offset = cur_size[f];
117  KALDI_ASSERT(this_size + this_offset <= sizes[f]);
118  output_lists[f].push_back(&(io.features));
119  NnetIo &output_io = merged_eg->io[f];
120  std::copy(io.indexes.begin(), io.indexes.end(),
121  output_io.indexes.begin() + this_offset);
122  std::vector<Index>::iterator output_iter = output_io.indexes.begin();
123  // Set the n index to be different for each of the original examples.
124  for (int32 i = this_offset; i < this_offset + this_size; i++) {
125  // we could easily support merging already-merged egs, but I don't see a
126  // need for it right now.
127  KALDI_ASSERT(output_iter[i].n == 0 &&
128  "Merging already-merged egs? Not currentlysupported.");
129  output_iter[i].n = n;
130  }
131  this_offset += this_size; // note: this_offset is a reference.
132  }
133  }
134  KALDI_ASSERT(cur_size == sizes);
135  for (int32 f = 0; f < num_feats; f++) {
136  AppendGeneralMatrixRows(output_lists[f],
137  &(merged_eg->io[f].features));
138  if (compress) {
139  // the following won't do anything if the features were sparse.
140  merged_eg->io[f].features.Compress();
141  }
142  }
143 }

147 void MergeExamples(const std::vector<NnetExample> &src,
148  bool compress,
149  NnetExample *merged_eg) {
150  KALDI_ASSERT(!src.empty());
151  std::vector<std::string> io_names;
152  GetIoNames(src, &io_names);
153  // the sizes are the total number of Indexes we have across all examples.
154  std::vector<int32> io_sizes;
155  GetIoSizes(src, io_names, &io_sizes);
156  MergeIo(src, io_names, io_sizes, compress, merged_eg);
157 }

159 void ShiftExampleTimes(int32 t_offset,
160  const std::vector<std::string> &exclude_names,
161  NnetExample *eg) {
162  if (t_offset == 0)
163  return;
164  std::vector<NnetIo>::iterator iter = eg->io.begin(),
165  end = eg->io.end();
166  for (; iter != end; iter++) {
167  bool name_is_excluded = false;
168  std::vector<std::string>::const_iterator
169  exclude_iter = exclude_names.begin(),
170  exclude_end = exclude_names.end();
171  for (; exclude_iter != exclude_end; ++exclude_iter) {
172  if (iter->name == *exclude_iter) {
173  name_is_excluded = true;
174  break;
175  }
176  }
177  if (!name_is_excluded) {
178  // name is not something like "ivector" that we exclude from shifting.
179  std::vector<Index>::iterator index_iter = iter->indexes.begin(),
180  index_end = iter->indexes.end();
181  for (; index_iter != index_end; ++index_iter)
182  index_iter->t += t_offset;
183  }
184  }
185 }

187 void GetComputationRequest(const Nnet &nnet,
188  const NnetExample &eg,
189  bool need_model_derivative,
190  bool store_component_stats,
191  ComputationRequest *request) {
192  request->inputs.clear();
193  request->inputs.reserve(eg.io.size());
194  request->outputs.clear();
195  request->outputs.reserve(eg.io.size());
196  request->need_model_derivative = need_model_derivative;
197  request->store_component_stats = store_component_stats;
198  for (size_t i = 0; i < eg.io.size(); i++) {
199  const NnetIo &io = eg.io[i];
200  const std::string &name = io.name;
201  int32 node_index = nnet.GetNodeIndex(name);
202  if (node_index == -1 &&
203  !nnet.IsInputNode(node_index) && !nnet.IsOutputNode(node_index))
204  KALDI_ERR << "Nnet example has input or output named '" << name
205  << "', but no such input or output node is in the network.";
206 
207  std::vector<IoSpecification> &dest =
208  nnet.IsInputNode(node_index) ? request->inputs : request->outputs;
209  dest.resize(dest.size() + 1);
210  IoSpecification &io_spec = dest.back();
211  io_spec.name = name;
212  io_spec.indexes = io.indexes;
213  io_spec.has_deriv = nnet.IsOutputNode(node_index) && need_model_derivative;
214  }
215  // check to see if something went wrong.
216  if (request->inputs.empty())
217  KALDI_ERR << "No inputs in computation request.";
218  if (request->outputs.empty())
219  KALDI_ERR << "No outputs in computation request.";
220 }
