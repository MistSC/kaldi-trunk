52  void InitData(std::istream &is) {
53  // parse config
54  std::string token;
55  while (!is.eof()) {
56  ReadToken(is, false, &token);
57  if (token == "<FmapXLen>") ReadBasicType(is, false, &fmap_x_len_);
58  else if (token == "<FmapYLen>") ReadBasicType(is, false, &fmap_y_len_);
59  else if (token == "<PoolXLen>") ReadBasicType(is, false, &pool_x_len_);
60  else if (token == "<PoolYLen>") ReadBasicType(is, false, &pool_y_len_);
61  else if (token == "<PoolXStep>") ReadBasicType(is, false, &pool_x_step_);
62  else if (token == "<PoolYStep>") ReadBasicType(is, false, &pool_y_step_);
63  else KALDI_ERR << "Unknown token " << token << ", a typo in config?"
64  << " (FmapXLen|FmapYLen|PoolXLen|PoolYLen|PoolXStep|PoolYStep)";
65  is >> std::ws; // eat-up whitespace
66  }
67  // check
68  KALDI_ASSERT(fmap_x_len_ * fmap_y_len_ * pool_x_len_ * pool_y_len_ * pool_x_step_ * pool_y_step_ != 0);
69  }

122  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
123  // useful dims
124  int32 num_input_fmaps = input_dim_ / (fmap_x_len_ * fmap_y_len_);
125  int out_fmap_cnt = 0;
126  for (int32 m = 0; m < fmap_x_len_-pool_x_len_+1; m = m+pool_x_step_) {
127  for (int32 n = 0; n < fmap_y_len_-pool_y_len_+1; n = n+pool_y_step_) {
128  int32 st = 0;
129  st = (m * fmap_y_len_ + n) * num_input_fmaps;
130  CuSubMatrix<BaseFloat> pool(out->ColRange(out_fmap_cnt * num_input_fmaps, num_input_fmaps));
131  pool.Set(-1e20); // reset (large neg value)
132  for (int32 i = 0; i < pool_x_len_; i++) {
133  for (int32 j = 0; j < pool_y_len_; j++) {
134  int32 c = 0;
135  c = st + i * (num_input_fmaps * fmap_y_len_)
136  + j * num_input_fmaps;
137  pool.Max(in.ColRange(c, num_input_fmaps));
138  }
139  }
140  out_fmap_cnt++;
141  }
142  }
143  }

145  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
146  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
147  // useful dims
148  int32 num_input_fmaps = input_dim_ / (fmap_x_len_ * fmap_y_len_);
149  int32 inp_fmap_size = fmap_x_len_ * fmap_y_len_;
150 
151  //
152  // here we note how many diff matrices are summed for each input patch,
153  std::vector<int32> patch_summands(inp_fmap_size, 0);
154  // this metainfo will be used to divide diff of patches
155  // used in more than one pool.
156  //
157 
158  in_diff->SetZero(); // reset
159 
160  int out_fmap_cnt = 0;
161  for (int32 m = 0; m < fmap_x_len_-pool_x_len_+1; m = m+pool_x_step_) {
162  for (int32 n = 0; n < fmap_y_len_-pool_y_len_+1; n = n+pool_y_step_) {
163  int32 st = 0;
164  st = (m*fmap_y_len_+n)*num_input_fmaps;
165 
166  for (int32 i = 0; i < pool_x_len_; i++) {
167  for (int32 j = 0; j < pool_y_len_; j++) {
168  int32 c = 0;
169  c = st + i * (num_input_fmaps * fmap_y_len_)
170  + j * num_input_fmaps;
171  //
172  CuSubMatrix<BaseFloat> in_p(in.ColRange(c, num_input_fmaps));
173  CuSubMatrix<BaseFloat> out_p(out.ColRange(out_fmap_cnt*num_input_fmaps, num_input_fmaps));
174  //
175 
176  CuSubMatrix<BaseFloat> tgt(in_diff->ColRange(c, num_input_fmaps));
177  CuMatrix<BaseFloat> src(out_diff.ColRange(out_fmap_cnt*num_input_fmaps, num_input_fmaps));
178 
179  CuMatrix<BaseFloat> mask;
180  in_p.EqualElementMask(out_p, &mask);
181  src.MulElements(mask);
182  tgt.AddMat(1.0, src);
183 
184  patch_summands[c/num_input_fmaps] += 1;
185  }
186  }
187  out_fmap_cnt++;
188  }
189  }
190 
191  // divide diff by #summands (compensate for patches used in more pools)
192  for (int i = 0; i < fmap_x_len_; i++) {
193  for (int32 j = 0; j < fmap_y_len_; j++) {
194  int32 c = i * fmap_y_len_ + j;
195  CuSubMatrix<BaseFloat> tgt(in_diff->ColRange(c * num_input_fmaps, num_input_fmaps));
196  KALDI_ASSERT(patch_summands[c] > 0); // patch at least in one pool
197  tgt.Scale(1.0 / patch_summands[c]);
198  }
199  }
200  }
