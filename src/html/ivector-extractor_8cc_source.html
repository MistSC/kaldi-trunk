43 // This function basically inverts the input and puts it in the output, but it's
44 // smart numerically. It uses the prior knowledge that the "inverse_floor" can
45 // have no eigenvalues less than one, so it applies that floor (in double
46 // precision) before inverting. This avoids certain numerical problems that can
47 // otherwise occur.
48 // static
49 void IvectorExtractor::InvertWithFlooring(const SpMatrix<double> &inverse_var,
50  SpMatrix<double> *var) {
51  SpMatrix<double> dbl_var(inverse_var);
52  int32 dim = inverse_var.NumRows();
53  Vector<double> s(dim);
54  Matrix<double> P(dim, dim);
55  // Solve the symmetric eigenvalue problem, inverse_var = P diag(s) P^T.
56  inverse_var.Eig(&s, &P);
57  s.ApplyFloor(1.0);
58  s.InvertElements();
59  var->AddMat2Vec(1.0, P, kNoTrans, s, 0.0);
60 }

63 void IvectorExtractor::GetIvectorDistribution(
64  const IvectorExtractorUtteranceStats &utt_stats,
65  VectorBase<double> *mean,
66  SpMatrix<double> *var) const {
67  if (!IvectorDependentWeights()) {
68  Vector<double> linear(IvectorDim());
69  SpMatrix<double> quadratic(IvectorDim());
70  GetIvectorDistMean(utt_stats, &linear, &quadratic);
71  GetIvectorDistPrior(utt_stats, &linear, &quadratic);
72  if (var != NULL) {
73  var->CopyFromSp(quadratic);
74  var->Invert(); // now it's a variance.
75 
76  // mean of distribution = quadratic^{-1} * linear...
77  mean->AddSpVec(1.0, *var, linear, 0.0);
78  } else {
79  quadratic.Invert();
80  mean->AddSpVec(1.0, quadratic, linear, 0.0);
81  }
82  } else {
83  Vector<double> linear(IvectorDim());
84  SpMatrix<double> quadratic(IvectorDim());
85  GetIvectorDistMean(utt_stats, &linear, &quadratic);
86  GetIvectorDistPrior(utt_stats, &linear, &quadratic);
87  // At this point, "linear" and "quadratic" contain
88  // the mean and prior-related terms, and we avoid
89  // recomputing those.
90 
91  Vector<double> cur_mean(IvectorDim());
92 
93  SpMatrix<double> quadratic_inv(IvectorDim());
94  InvertWithFlooring(quadratic, &quadratic_inv);
95  cur_mean.AddSpVec(1.0, quadratic_inv, linear, 0.0);
96 
97  KALDI_VLOG(3) << "Trace of quadratic is " << quadratic.Trace()
98  << ", condition is " << quadratic.Cond();
99  KALDI_VLOG(3) << "Trace of quadratic_inv is " << quadratic_inv.Trace()
100  << ", condition is " << quadratic_inv.Cond();
101 
102  // The loop is finding successively better approximation points
103  // for the quadratic expansion of the weights.
104  int32 num_iters = 4;
105  double change_threshold = 0.1; // If the iVector changes by less than
106  // this (in 2-norm), we abort early.
107  for (int32 iter = 0; iter < num_iters; iter++) {
108  if (GetVerboseLevel() >= 3) {
109  KALDI_VLOG(3) << "Auxf on iter " << iter << " is "
110  << GetAuxf(utt_stats, cur_mean, &quadratic_inv);
111  int32 show_dim = 5;
112  if (show_dim > cur_mean.Dim()) show_dim = cur_mean.Dim();
113  KALDI_VLOG(3) << "Current distribution mean is "
114  << cur_mean.Range(0, show_dim) << "... "
115  << ", var trace is " << quadratic_inv.Trace();
116  }
117  Vector<double> this_linear(linear);
118  SpMatrix<double> this_quadratic(quadratic);
119  GetIvectorDistWeight(utt_stats, cur_mean,
120  &this_linear, &this_quadratic);
121  InvertWithFlooring(this_quadratic, &quadratic_inv);
122  Vector<double> mean_diff(cur_mean);
123  cur_mean.AddSpVec(1.0, quadratic_inv, this_linear, 0.0);
124  mean_diff.AddVec(-1.0, cur_mean);
125  double change = mean_diff.Norm(2.0);
126  KALDI_VLOG(2) << "On iter " << iter << ", iVector changed by " << change;
127  if (change < change_threshold)
128  break;
129  }
130  mean->CopyFromVec(cur_mean);
131  if (var != NULL)
132  var->CopyFromSp(quadratic_inv);
133  }
134 }

137 IvectorExtractor::IvectorExtractor(
138  const IvectorExtractorOptions &opts,
139  const FullGmm &fgmm) {
140  KALDI_ASSERT(opts.ivector_dim > 0);
141  Sigma_inv_.resize(fgmm.NumGauss());
142  for (int32 i = 0; i < fgmm.NumGauss(); i++) {
143  const SpMatrix<BaseFloat> &inv_var = fgmm.inv_covars()[i];
144  Sigma_inv_[i].Resize(inv_var.NumRows());
145  Sigma_inv_[i].CopyFromSp(inv_var);
146  }
147  Matrix<double> gmm_means;
148  fgmm.GetMeans(&gmm_means);
149  KALDI_ASSERT(!Sigma_inv_.empty());
150  int32 feature_dim = Sigma_inv_[0].NumRows(),
151  num_gauss = Sigma_inv_.size();
152 
153  prior_offset_ = 100.0; // hardwired for now. Must be nonzero.
154  gmm_means.Scale(1.0 / prior_offset_);
155 
156  M_.resize(num_gauss);
157  for (int32 i = 0; i < num_gauss; i++) {
158  M_[i].Resize(feature_dim, opts.ivector_dim);
159  M_[i].SetRandn();
160  M_[i].CopyColFromVec(gmm_means.Row(i), 0);
161  }
162  if (opts.use_weights) { // will regress the log-weights on the iVector.
163  w_.Resize(num_gauss, opts.ivector_dim);
164  } else {
165  w_vec_.Resize(fgmm.NumGauss());
166  w_vec_.CopyFromVec(fgmm.weights());
167  }
168  ComputeDerivedVars();
169 }

182 void IvectorExtractor::ComputeDerivedVars() {
183  KALDI_LOG << "Computing derived variables for iVector extractor";
184  gconsts_.Resize(NumGauss());
185  for (int32 i = 0; i < NumGauss(); i++) {
186  double var_logdet = -Sigma_inv_[i].LogPosDefDet();
187  gconsts_(i) = -0.5 * (var_logdet + FeatDim() * M_LOG_2PI);
188  // the gconsts don't contain any weight-related terms.
189  }
190  U_.Resize(NumGauss(), IvectorDim() * (IvectorDim() + 1) / 2);
191  Sigma_inv_M_.resize(NumGauss());

193  // Note, we could have used RunMultiThreaded for this and similar tasks we
194  // have here, but we found that we don't get as complete CPU utilization as we
195  // could because some tasks finish before others.
196  {
197  TaskSequencerConfig sequencer_opts;
198  sequencer_opts.num_threads = g_num_threads;
199  TaskSequencer<IvectorExtractorComputeDerivedVarsClass> sequencer(
200  sequencer_opts);
201  for (int32 i = 0; i < NumGauss(); i++)
202  sequencer.Run(new IvectorExtractorComputeDerivedVarsClass(this, i));
203  }
204  KALDI_LOG << "Done.";
205 }

221 void IvectorExtractor::GetIvectorDistWeight(
222  const IvectorExtractorUtteranceStats &utt_stats,
223  const VectorBase<double> &mean,
224  VectorBase<double> *linear,
225  SpMatrix<double> *quadratic) const {
226  // If there is no w_, then weights do not depend on the iVector
227  // and the weights contribute nothing to the distribution.
228  if (!IvectorDependentWeights())
229  return;
230 
231  Vector<double> logw_unnorm(NumGauss());
232  logw_unnorm.AddMatVec(1.0, w_, kNoTrans, mean, 0.0);
233 
234  Vector<double> w(logw_unnorm);
235  w.ApplySoftMax(); // now w is the weights.
236 
237  // See eq.58 in SGMM paper
238  // http://www.sciencedirect.com/science/article/pii/S088523081000063X
239  // linear_coeff(i) = \gamma_{jmi} - \gamma_{jm} \hat{w}_{jmi} + \max(\gamma_{jmi}, \gamma_{jm} \hat{w}_{jmi} \hat{\w}_i \v_{jm}
240  // here \v_{jm} corresponds to the iVector. Ignore the j,m indices.
241  Vector<double> linear_coeff(NumGauss());
242  Vector<double> quadratic_coeff(NumGauss());
243  double gamma = utt_stats.gamma_.Sum();
244  for (int32 i = 0; i < NumGauss(); i++) {
245  double gamma_i = utt_stats.gamma_(i);
246  double max_term = std::max(gamma_i, gamma * w(i));
247  linear_coeff(i) = gamma_i - gamma * w(i) + max_term * logw_unnorm(i);
248  quadratic_coeff(i) = max_term;
249  }
250  linear->AddMatVec(1.0, w_, kTrans, linear_coeff, 1.0);
251 
252  // *quadratic += \sum_i quadratic_coeff(i) w_i w_i^T, where w_i is
253  // i'th row of w_.
254  quadratic->AddMat2Vec(1.0, w_, kTrans, quadratic_coeff, 1.0);
255 }

285 double IvectorExtractor::GetAcousticAuxfWeight(
286  const IvectorExtractorUtteranceStats &utt_stats,
287  const VectorBase<double> &mean,
288  const SpMatrix<double> *var) const {
289  if (!IvectorDependentWeights()) { // Not using the weight-projection matrices.
290  Vector<double> log_w_vec(w_vec_);
291  log_w_vec.ApplyLog();
292  return VecVec(log_w_vec, utt_stats.gamma_);
293  } else {
294  Vector<double> w(NumGauss());
295  w.AddMatVec(1.0, w_, kNoTrans, mean, 0.0); // now w is unnormalized
296  // log-weights.
297 
298  double lse = w.LogSumExp();
299  w.Add(-lse); // Normalize so log-weights sum to one.
300 
301  // "ans" below is the point-value of the weight auxf, without
302  // considering the variance. At the moment, "w" contains
303  // the normalized log weights.
304  double ans = VecVec(w, utt_stats.gamma_);
305 
306  w.ApplyExp(); // now w is the weights.
307 
308  if (var == NULL) {
309  return ans;
310  } else {
311  // Below, "Jacobian" will be the derivative d(log_w) / d(ivector)
312  // = (I - w w^T) W, where W (w_ in the code) is the projection matrix
313  // from iVector space to unnormalized log-weights, and w is the normalized
314  // weight values at the current point.
315  Matrix<double> Jacobian(w_);
316  Vector<double> WTw(IvectorDim()); // W^T w
317  WTw.AddMatVec(1.0, w_, kTrans, w, 0.0);
318  Jacobian.AddVecVec(1.0, w, WTw); // Jacobian += (w (W^T w)^T = w^T w W)
319 
320  // the matrix S is the negated 2nd derivative of the objf w.r.t. the iVector \x.
321  SpMatrix<double> S(IvectorDim());
322  S.AddMat2Vec(1.0, Jacobian, kTrans, Vector<double>(utt_stats.gamma_), 0.0);
323  ans += -0.5 * TraceSpSp(S, *var);
324  return ans;
325  }
326  }
327 }

344 /*
345  Get the prior-related part of the auxiliary function. Suppose
346  the ivector is x, the prior distribution is p(x), the likelihood
347  of the data given x (itself an auxiliary function) is q(x) and
348  the prior is r(x).
349 
350  In the case where we just have a point ivector x and no p(x),
351  the prior-related term we return will just be q(x).
352 
353  If we have a distribution over x, we define an auxiliary function
354  t(x) = \int p(x) log(q(x)r(x) / p(x)) dx.
355  We separate this into data-dependent and prior parts, where the prior
356  part is
357  \int p(x) log(q(x) / p(x)) dx
358  (which this function returns), and the acoustic part is
359  \int p(x) log(r(x)) dx.
360  Note that the fact that we divide by p(x) in the prior part and not
361  the acoustic part is a bit arbitrary, at least the way we have written
362  it down here; it doesn't matter where we do it, but this way seems
363  more natural.
364 */
365 double IvectorExtractor::GetPriorAuxf(
366  const VectorBase<double> &mean,
367  const SpMatrix<double> *var) const {
368  KALDI_ASSERT(mean.Dim() == IvectorDim());
369 
370  Vector<double> offset(mean);
371  offset(0) -= prior_offset_; // The mean of the prior distribution
372  // may only be nonzero in the first dimension. Now, "offset" is the
373  // offset of ivector from the prior's mean.
374 
375 
376  if (var == NULL) {
377  // The log-determinant of the variance of the prior distribution is one,
378  // since it's the unit matrix.
379  return -0.5 * (VecVec(offset, offset) + IvectorDim()*M_LOG_2PI);
380  } else {
381  // The mean-related part of the answer will be
382  // -0.5 * (VecVec(offset, offset), just like above.
383  // The variance-related part will be
384  // \int p(x) . -0.5 (x^T I x - x^T var^{-1} x + logdet(I) - logdet(var)) dx
385  // and using the fact that x is distributed with variance "var", this is:
386  //= \int p(x) . -0.5 (x^T I x - x^T var^{-1} x + logdet(I) - logdet(var)) dx
387  // = -0.5 ( trace(var I) - trace(var^{-1} var) + 0.0 - logdet(var))
388  // = -0.5 ( trace(var) - dim(var) - logdet(var))
389 
390 
391  KALDI_ASSERT(var->NumRows() == IvectorDim());
392  return -0.5 * (VecVec(offset, offset) + var->Trace() -
393  IvectorDim() - var->LogPosDefDet());
394  }
395 }

419 /* This is the part of the auxf that involves the data
420  means, and we also involve the gconsts here.
421  Let \m_i be the observed data mean vector for the
422  i'th Gaussian (which we get from the utterance-specific
423  stats). Let \mu_i(\x) be the mean of the i'th Gaussian,
424  written as a function of the iVector \x.
425 
426  \int_\x p(\x) (\sum_i \gamma_i -0.5 (\mu(\x) - \m_i)^T \Sigma_i^{-1} (\mu(\x) - \m_i)) d\x
427 
428  To compute this integral we'll first write out the summation as a function of \x.
429 
430  \sum_i -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
431  + \gamma_i \mu(\x)^T \Sigma_i^{-1} \m_i
432  -0.5 \gamma_i \mu(\x)^T \Sigma_i^{-1} \m_i
433  = \sum_i -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
434  + \gamma_i \x^T \M_i^T \Sigma_i^{-1} \m_i
435  -0.5 \gamma_i \x^T \M_i^T \Sigma_i^{-1} \x
436  = K + \x^T \a - 0.5 \x^T \B \x,
437  where K = \sum_i -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
438  \a = \sum_i \gamma_i \M_i^T \Sigma_i^{-1} \m_i
439  \B = \sum_i \gamma_i \U_i (where \U_i = \M_i^T \Sigma_i^{-1} \M_i
440  has been stored previously).
441  Note: the matrix M in the stats actually contains \gamma_i \m_i,
442  i.e. the mean times the count, so we need to modify the definitions
443  above accordingly.
444 */
445 double IvectorExtractor::GetAcousticAuxfMean(
446  const IvectorExtractorUtteranceStats &utt_stats,
447  const VectorBase<double> &mean,
448  const SpMatrix<double> *var) const {
449  double K = 0.0;
450  Vector<double> a(IvectorDim()), temp(FeatDim());
451 
452  int32 I = NumGauss();
453  for (int32 i = 0; i < I; i++) {
454  double gamma = utt_stats.gamma_(i);
455  if (gamma != 0.0) {
456  Vector<double> x(utt_stats.X_.Row(i)); // == \gamma(i) \m_i
457  temp.AddSpVec(1.0 / gamma, Sigma_inv_[i], x, 0.0);
458  // now temp = Sigma_i^{-1} \m_i.
459  // next line: K += -0.5 \gamma_i \m_i^T \Sigma_i^{-1} \m_i
460  K += -0.5 * VecVec(x, temp);
461  // next line: a += \gamma_i \M_i^T \Sigma_i^{-1} \m_i
462  a.AddMatVec(gamma, M_[i], kTrans, temp, 1.0);
463  }
464  }
465  SpMatrix<double> B(IvectorDim());
466  SubVector<double> B_vec(B.Data(), IvectorDim()*(IvectorDim()+1)/2);
467  B_vec.AddMatVec(1.0, U_, kTrans, Vector<double>(utt_stats.gamma_), 0.0);
468 
469  double ans = K + VecVec(mean, a) - 0.5 * VecSpVec(mean, B, mean);
470  if (var != NULL)
471  ans -= 0.5 * TraceSpSp(*var, B);
472  return ans;
473 }

482 double IvectorExtractor::GetAcousticAuxfVariance(
483  const IvectorExtractorUtteranceStats &utt_stats) const {
484  if (utt_stats.S_.empty()) {
485  // we did not store the variance, so assume it's as predicted
486  // by the model itself.
487  // for each Gaussian i, we have a term -0.5 * gamma(i) * trace(Sigma[i] * Sigma[i]^{-1})
488  // = -0.5 * gamma(i) * FeatDim().
489  return -0.5 * utt_stats.gamma_.Sum() * FeatDim();
490  } else {
491  int32 I = NumGauss();
492  double ans = 0.0;
493  for (int32 i = 0; i < I; i++) {
494  double gamma = utt_stats.gamma_(i);
495  if (gamma != 0.0) {
496  SpMatrix<double> var(utt_stats.S_[i]);
497  var.Scale(1.0 / gamma);
498  Vector<double> mean(utt_stats.X_.Row(i));
499  mean.Scale(1.0 / gamma);
500  var.AddVec2(-1.0, mean); // get centered covariance..
501  ans += -0.5 * gamma * TraceSpSp(var, Sigma_inv_[i]);
502  }
503  }
504  return ans;
505  }

509  double new_prior_offset) {

511  Tinv.Invert();
512  // w <-- w Tinv. (construct temporary copy with Matrix<double>(w))
513  if (IvectorDependentWeights())
514  w_.AddMatMat(1.0, Matrix<double>(w_), kNoTrans, Tinv, kNoTrans, 0.0);

519  prior_offset_ = new_prior_offset;

522 void OnlineIvectorEstimationStats::AccStats(
523  const IvectorExtractor &extractor,
524  const VectorBase<BaseFloat> &feature,
525  const std::vector<std::pair<int32, BaseFloat> > &gauss_post) {
526  KALDI_ASSERT(extractor.IvectorDim() == this->IvectorDim());
527  KALDI_ASSERT(!extractor.IvectorDependentWeights());
528 
529  Vector<double> feature_dbl(feature);
530  double tot_weight = 0.0;
531  int32 ivector_dim = this->IvectorDim(),
532  quadratic_term_dim = (ivector_dim * (ivector_dim + 1)) / 2;
533  SubVector<double> quadratic_term_vec(quadratic_term_.Data(),
534  quadratic_term_dim);
535 
536  for (size_t idx = 0; idx < gauss_post.size(); idx++) {
537  int32 g = gauss_post[idx].first;
538  double weight = gauss_post[idx].second;
539  // allow negative weights; it's needed in the online iVector extraction
540  // with speech-silence detection based on decoder traceback (we subtract
541  // stuff we previously added if the traceback changes).
542  if (weight == 0.0)
543  continue;
544  linear_term_.AddMatVec(weight, extractor.Sigma_inv_M_[g], kTrans,
545  feature_dbl, 1.0);
546  SubVector<double> U_g(extractor.U_, g);
547  quadratic_term_vec.AddVec(weight, U_g);
548  tot_weight += weight;
549  }
550  if (max_count_ > 0.0) {
551  // see comments in header RE max_count for explanation. It relates to
552  // prior scaling when the count exceeds max_count_
553  double old_num_frames = num_frames_,
554  new_num_frames = num_frames_ + tot_weight;
555  double old_prior_scale = std::max(old_num_frames, max_count_) / max_count_,
556  new_prior_scale = std::max(new_num_frames, max_count_) / max_count_;
557  // The prior_scales are the inverses of the scales we would put on the stats
558  // if we were implementing this by scaling the stats. Instead we
559  // scale the prior term.
560  double prior_scale_change = new_prior_scale - old_prior_scale;
561  if (prior_scale_change != 0.0) {
562  linear_term_(0) += prior_offset_ * prior_scale_change;
563  quadratic_term_.AddToDiag(prior_scale_change);
564  }
565  }
566 
567  num_frames_ += tot_weight;
568 }

570 void OnlineIvectorEstimationStats::Scale(double scale) {
571  KALDI_ASSERT(scale >= 0.0 && scale <= 1.0);
572  double old_num_frames = num_frames_;
573  num_frames_ *= scale;
574  quadratic_term_.Scale(scale);
575  linear_term_.Scale(scale);
576 
577  // Scale back up the prior term, by adding in whatever we scaled down.
578  if (max_count_ == 0.0) {
579  linear_term_(0) += prior_offset_ * (1.0 - scale);
580  quadratic_term_.AddToDiag(1.0 - scale);
581  } else {
582  double new_num_frames = num_frames_;
583  double old_prior_scale =
584  scale * std::max(old_num_frames, max_count_) / max_count_,
585  new_prior_scale = std::max(new_num_frames, max_count_) / max_count_;
586  // old_prior_scale is the scale the prior term currently has in the stats,
587  // i.e. the previous scale times "scale" as we just scaled the stats.
588  // new_prior_scale is the scale we want the prior term to have.
589  linear_term_(0) += prior_offset_ * (new_prior_scale - old_prior_scale);
590  quadratic_term_.AddToDiag(new_prior_scale - old_prior_scale);
591  }
592 }

875  // cleared and they may write to it.

900 // This function commits stats for a single sample of the ivector,
901 // to update the weight projection w_.
902 void IvectorExtractorStats::CommitStatsForWPoint(
903  const IvectorExtractor &extractor,
904  const IvectorExtractorUtteranceStats &utt_stats,
905  const VectorBase<double> &ivector,
906  double weight) {
907  int32 num_gauss = extractor.NumGauss();
908  // Compare this function with GetIvectorDistWeight(), from which it
909  // was derived.
910  Vector<double> logw_unnorm(num_gauss);
911  logw_unnorm.AddMatVec(1.0, extractor.w_, kNoTrans, ivector, 0.0);
912 
913  Vector<double> w(logw_unnorm);
914  w.ApplySoftMax(); // now w is the weights.
915 
916  Vector<double> linear_coeff(num_gauss);
917  Vector<double> quadratic_coeff(num_gauss);
918  double gamma = utt_stats.gamma_.Sum();
919  for (int32 i = 0; i < num_gauss; i++) {
920  double gamma_i = utt_stats.gamma_(i);
921  double max_term = std::max(gamma_i, gamma * w(i));
922  linear_coeff(i) = gamma_i - gamma * w(i) + max_term * logw_unnorm(i);
923  quadratic_coeff(i) = max_term;
924  }
925  weight_stats_lock_.Lock();
926  G_.AddVecVec(weight, linear_coeff, Vector<double>(ivector));
927 
928  int32 ivector_dim = extractor.IvectorDim();
929  SpMatrix<double> outer_prod(ivector_dim);
930  outer_prod.AddVec2(1.0, ivector);
931  SubVector<double> outer_prod_vec(outer_prod.Data(),
932  ivector_dim * (ivector_dim + 1) / 2);
933  Q_.AddVecVec(weight, quadratic_coeff, outer_prod_vec);
934  weight_stats_lock_.Unlock();
935 }

937 void IvectorExtractorStats::CommitStatsForW(
938  const IvectorExtractor &extractor,
939  const IvectorExtractorUtteranceStats &utt_stats,
940  const VectorBase<double> &ivec_mean,
941  const SpMatrix<double> &ivec_var) {
942  KALDI_ASSERT(config_.num_samples_for_weights > 1);
943 
944  Matrix<double> rand(config_.num_samples_for_weights, extractor.IvectorDim());
945  rand.SetRandn();
946  TpMatrix<double> ivec_stddev(extractor.IvectorDim());
947  ivec_stddev.Cholesky(ivec_var);
948  Matrix<double> ivecs(config_.num_samples_for_weights, extractor.IvectorDim());
949  ivecs.AddMatTp(1.0, rand, kNoTrans, ivec_stddev, kTrans, 0.0);
950  // Now make the ivecs zero-mean
951  Vector<double> avg_ivec(extractor.IvectorDim());
952  avg_ivec.AddRowSumMat(1.0 / config_.num_samples_for_weights, ivecs);
953  ivecs.AddVecToRows(-1.0, avg_ivec);
954  // Correct the variance for what we just did, so the expected
955  // variance still has the correct value.
956  ivecs.Scale(sqrt(config_.num_samples_for_weights / (config_.num_samples_for_weights - 1.0)));
957  // Add the mean of the distribution to "ivecs".
958  ivecs.AddVecToRows(1.0, ivec_mean);
959  // "ivecs" is now a sample from the iVector distribution.
960  for (int32 samp = 0; samp < config_.num_samples_for_weights; samp++)
961  CommitStatsForWPoint(extractor, utt_stats,
962  ivecs.Row(samp),
963  1.0 / config_.num_samples_for_weights);
964 }

1218 double IvectorExtractorStats::UpdateProjection(
1219  const IvectorExtractorEstimationOptions &opts,
1220  int32 i,
1221  IvectorExtractor *extractor) const {
1222  int32 I = extractor->NumGauss(), S = extractor->IvectorDim();
1223  KALDI_ASSERT(i >= 0 && i < I);
1224  /*
1225  For Gaussian index i, maximize the auxiliary function
1226  Q_i(x) = tr(M_i^T Sigma_i^{-1} Y_i) - 0.5 tr(Sigma_i^{-1} M_i R_i M_i^T)
1227  */
1228  if (gamma_(i) < opts.gaussian_min_count) {
1229  KALDI_WARN << "Skipping Gaussian index " << i << " because count "
1230  << gamma_(i) << " is below min-count.";
1231  return 0.0;
1232  }
1233  SpMatrix<double> R(S, kUndefined), SigmaInv(extractor->Sigma_inv_[i]);

1235  SubVector<double> R_sp(R.Data(), S * (S+1) / 2);
1236  R_sp.CopyFromVec(R_vec); // copy to SpMatrix's memory.
1237 
1238  Matrix<double> M(extractor->M_[i]);
1239  SolverOptions solver_opts;
1240  solver_opts.name = "M";
1241  solver_opts.diagonal_precondition = true;
1242  double impr = SolveQuadraticMatrixProblem(R, Y_[i], SigmaInv, solver_opts, &M),
1243  gamma = gamma_(i);
1244  if (i < 4) {
1245  KALDI_VLOG(1) << "Objf impr for M for Gaussian index " << i << " is "
1246  << (impr / gamma) << " per frame over " << gamma << " frames.";
1247  }
1248  extractor->M_[i].CopyFromMat(M);
1249  return impr;

1252 void IvectorExtractorStats::GetOrthogonalIvectorTransform(
1253  const SubMatrix<double> &T,
1254  IvectorExtractor *extractor,
1255  Matrix<double> *A) const {

1257  int32 ivector_dim = extractor->IvectorDim(),
1258  num_gauss = extractor->NumGauss();

1260 
1261  // Each row of extractor->U_ is an SpMatrix. We can compute the weighted
1262  // avg of these rows in a SubVector that updates the data of the SpMatrix
1263  // Uavg.
1264  SpMatrix<double> Uavg(ivector_dim), Vavg(ivector_dim - 1);

1269  uavg_vec.AddMatVec(1.0/num_gauss, extractor->U_, kTrans, w_uniform, 0.0);
1270  } else {
1271  uavg_vec.AddMatVec(1.0, extractor->U_, kTrans, extractor->w_vec_, 0.0);
1272  }
1273 
1274  Matrix<double> Tinv(T);
1275  Tinv.Invert();
1276  Matrix<double> Vavg_temp(Vavg), Uavg_temp(Uavg);
1277 
1278  Vavg_temp.AddMatMatMat(1.0, Tinv, kTrans, SubMatrix<double>(Uavg_temp,
1279  1, ivector_dim-1, 1, ivector_dim-1),
1280  kNoTrans, Tinv, kNoTrans, 0.0);
1281  Vavg.CopyFromMat(Vavg_temp);
1282 
1283  Vector<double> s(ivector_dim-1);
1284  Matrix<double> P(ivector_dim-1, ivector_dim-1);
1285  Vavg.Eig(&s, &P);
1286  SortSvd(&s, &P);
1287  A->Resize(P.NumCols(), P.NumRows());
1288  A->SetZero();
1289  A->AddMat(1.0, P, kTrans);
1290  KALDI_LOG << "Eigenvalues of Vavg: " << s;
1291 }

1336 double IvectorExtractorStats::UpdateVariances(
1337  const IvectorExtractorEstimationOptions &opts,
1338  IvectorExtractor *extractor) const {
1339  int32 num_gauss = extractor->NumGauss(),
1340  feat_dim = extractor->FeatDim(),
1341  ivector_dim = extractor->IvectorDim();
1342  KALDI_ASSERT(!S_.empty());
1343  double tot_objf_impr = 0.0;
1344 
1345  // "raw_variances" will be the variances directly from
1346  // the stats, without any flooring.
1347  std::vector<SpMatrix<double> > raw_variances(num_gauss);
1348  SpMatrix<double> var_floor(feat_dim);
1349  double var_floor_count = 0.0;
1350 
1351  for (int32 i = 0; i < num_gauss; i++) {
1352  if (gamma_(i) < opts.gaussian_min_count) continue; // warned in UpdateProjections
1353  SpMatrix<double> &S(raw_variances[i]);
1354  S = S_[i]; // Set it to the raw scatter statistics.
1355 
1356  // The equations for estimating the variance are similar to
1357  // those used in SGMMs. We need to convert it to a centered
1358  // covariance, and for this we can use a combination of other
1359  // stats and the model parameters.
1360 
1361  Matrix<double> M(extractor->M_[i]);
1362  // Y * M^T.
1363  Matrix<double> YM(feat_dim, feat_dim);
1364  YM.AddMatMat(1.0, Y_[i], kNoTrans, M, kTrans, 0.0);
1365  Matrix<double> YMMY(YM, kTrans);
1366  YMMY.AddMat(1.0, YM);
1367  // Now, YMMY = Y * M^T + M * Y^T. This is a kind of cross-term
1368  // between the mean and the data, which we subtract.
1369  SpMatrix<double> YMMY_sp(YMMY, kTakeMeanAndCheck);
1370  S.AddSp(-1.0, YMMY_sp);
1371 
1372  // Add in a mean-squared term.
1373  SpMatrix<double> R(ivector_dim); // will be scatter of iVectors, weighted
1374  // by count for this Gaussian.
1375  SubVector<double> R_vec(R.Data(),
1376  ivector_dim * (ivector_dim + 1) / 2);
1377  R_vec.CopyFromVec(R_.Row(i)); //
1378 
1379  S.AddMat2Sp(1.0, M, kNoTrans, R, 1.0);
1380 
1381  var_floor.AddSp(1.0, S);
1382  var_floor_count += gamma_(i);
1383  S.Scale(1.0 / gamma_(i));
1384  }
1385  KALDI_ASSERT(var_floor_count > 0.0);
1386  KALDI_ASSERT(opts.variance_floor_factor > 0.0 &&
1387  opts.variance_floor_factor <= 1.0);
1388 
1389  var_floor.Scale(opts.variance_floor_factor / var_floor_count);
1390 
1391  int32 tot_num_floored = 0;
1392  for (int32 i = 0; i < num_gauss; i++) {
1393  SpMatrix<double> &S(raw_variances[i]); // un-floored variance.
1394  if (S.NumRows() == 0) continue; // due to low count.
1395  SpMatrix<double> floored_var(S);
1396  SpMatrix<double> old_inv_var(extractor->Sigma_inv_[i]);
1397 
1398  int32 num_floored = floored_var.ApplyFloor(var_floor);
1399  tot_num_floored += num_floored;
1400  if (num_floored > 0)
1401  KALDI_LOG << "For Gaussian index " << i << ", floored "
1402  << num_floored << " eigenvalues of variance.";
1403  // this objf is per frame;
1404  double old_objf = -0.5 * (TraceSpSp(S, old_inv_var) -
1405  old_inv_var.LogPosDefDet());
1406 
1407  SpMatrix<double> new_inv_var(floored_var);
1408  new_inv_var.Invert();
1409 
1410  double new_objf = -0.5 * (TraceSpSp(S, new_inv_var) -
1411  new_inv_var.LogPosDefDet());
1412  if (i < 4) {
1413  KALDI_VLOG(1) << "Objf impr/frame for variance for Gaussian index "
1414  << i << " was " << (new_objf - old_objf);
1415  }
1416  tot_objf_impr += gamma_(i) * (new_objf - old_objf);
1417  extractor->Sigma_inv_[i].CopyFromSp(new_inv_var);
1418  }
1419  double floored_percent = tot_num_floored * 100.0 / (num_gauss * feat_dim);
1420  KALDI_LOG << "Floored " << floored_percent << "% of all Gaussian eigenvalues";
1421 
1422  KALDI_LOG << "Overall objf impr/frame for variances was "
1423  << (tot_objf_impr / gamma_.Sum()) << " over "
1424  << gamma_.Sum() << " frames.";
1425  return tot_objf_impr / gamma_.Sum();
1426 }

1501 double IvectorExtractorStats::PriorDiagnostics(double old_prior_offset) const {
1502  // The iVectors had a centered covariance "covar"; we want to figure out
1503  // the objective-function change from rescaling. It's as if we were
1504  // formerly modeling "covar" with the unit matrix, and we're now modeling
1505  // it with "covar" itself. This is ignoring flooring issues. Of course,
1506  // we implement it through rescaling the space, but it has the same effect.
1507  // We also need to take into account that before the rescaling etc., the
1508  // old mean might have been wrong.
1509 
1510  int32 ivector_dim = ivector_sum_.Dim();
1511  Vector<double> sum(ivector_sum_);
1512  sum.Scale(1.0 / num_ivectors_);
1513  SpMatrix<double> covar(ivector_scatter_);
1514  covar.Scale(1.0 / num_ivectors_);
1515  covar.AddVec2(-1.0, sum); // Get the centered covariance.
1516 
1517  // Now work out the offset from the old prior's mean.
1518  Vector<double> mean_offset(sum);
1519  mean_offset(0) -= old_prior_offset;
1520 
1521  SpMatrix<double> old_covar(covar); // the covariance around the old mean.
1522  old_covar.AddVec2(1.0, mean_offset);
1523  // old likelihood = -0.5 * (Trace(I old_covar) + logdet(I) + [ignored])
1524  double old_like = -0.5 * old_covar.Trace();
1525  // new likelihood is if we updated the variance to equal "covar"... this isn't
1526  // how we did it (we use rescaling of the ivectors) but it has the same
1527  // effect. -0.5 * (Trace(covar^{-1} covar) + logdet(covar))
1528  double new_like = -0.5 * (ivector_dim + covar.LogPosDefDet()),
1529  like_change = new_like - old_like,
1530  like_change_per_frame = like_change * num_ivectors_ / gamma_.Sum();
1531 
1532  KALDI_LOG << "Overall auxf improvement from prior is " << like_change_per_frame
1533  << " per frame, or " << like_change << " per iVector.";
1534  return like_change_per_frame; // we'll be adding this to other per-frame
1535  // quantities.
1536 }

1554  covar.Eig(&s, &P);
1555  KALDI_LOG << "Eigenvalues of iVector covariance range from "
1556  << s.Min() << " to " << s.Max();

1558  if (num_floored > 0)

1560  << "of iVectors.";

1563  { // set T to a transformation that makes covar unit
1564  // (modulo floored eigenvalues).
1565  Vector<double> scales(s);

1570  Tproj.AddMat2Sp(1.0, T, kNoTrans, covar, 0.0);
1571  KALDI_ASSERT(Tproj.IsUnit(1.0e-06));
1572  }
1573  }

1576  sum_proj.AddMatVec(1.0, T, kNoTrans, sum, 0.0);
1577 
1578  KALDI_ASSERT(sum_proj.Norm(2.0) != 0.0);
1579

1581  // but also that sends "sum" to a multiple of the vector e0 = [ 1 0 0 0 .. ].
1582  // We'll do this by a transform that follows T, of the form
1583  // (I - 2 a a^T), where a is unit. [i.e. a Householder reflection].
1584  // Firstly, let x equal sum_proj normalized to unit length.
1585  // We'll let a = alpha x + beta e0, for suitable coefficients alpha and beta,
1586  // To project sum_proj (or equivalenty, x) to a multiple of e0, we'll need that
1587  // the x term in
1588  // (I - 2(alpha x + beta e0)(alpha x + beta e0) x
1589  // equals zero., i.e. 1 - 2 alpha (alpha x^T x + beta e0^T x) == 0,
1590  // (1 - 2 alpha^2 - 2 alpha beta x0) = 0
1591  // To ensure that a is unit, we require that
1592  // (alpha x + beta e0).(alpha x + beta e0) = 1, i.e.
1593  // alpha^2 + beta^2 + 2 alpha beta x0 = 1
1594  // at wolframalpha.com,
1595  // Solve[ {a^2 + b^2 + 2 a b x = 1}, {1 - 2 a^2 - 2 a b x = 0}, {a, b} ]
1596  // gives different solutions, but the one that keeps the offset positive
1597  // after projection seems to be:
1598  // alpha = 1/(sqrt(2)sqrt(1 - x0)), beta = -alpha
1599 
1600  Matrix<double> U(ivector_dim, ivector_dim);
1601  U.SetUnit();
1602  Vector<double> x(sum_proj);
1603  x.Scale(1.0 / x.Norm(2.0));
1604  double x0 = x(0), alpha, beta;
1605  alpha = 1.0 / (M_SQRT2 * sqrt(1.0 - x0));
1606  beta = -alpha;
1607  Vector<double> a(x);
1608  a.Scale(alpha);
1609  a(0) += beta;
1610  U.AddVecVec(-2.0, a, a);
1611 
1612  Matrix<double> V(ivector_dim, ivector_dim);
1613  V.AddMatMat(1.0, U, kNoTrans, T, kNoTrans, 0.0);
1614 
1615  // Optionally replace transform V with V' such that V' makes the
1616  // covariance unit and additionally diagonalizes the quadratic
1617  // term.
1618  if (opts.diagonalize) {
1619 
1620  SubMatrix<double> Vsub(V, 1, V.NumRows()-1, 0, V.NumCols());
1621  Matrix<double> Vtemp(SubMatrix<double>(V, 1, V.NumRows()-1,
1622  0, V.NumCols())),
1623  A;
1624  GetOrthogonalIvectorTransform(SubMatrix<double>(Vtemp, 0,
1625  Vtemp.NumRows(), 1, Vtemp.NumCols()-1),
1626  extractor, &A);
1627 
1628  // It is necessary to exclude the first row of V in this transformation
1629  // so that the sum_vproj has the form [ x 0 0 0 .. ], where x > 0.
1630  Vsub.AddMatMat(1.0, A, kNoTrans, Vtemp, kNoTrans, 0.0);
1631  }
1632 
1633  if (num_floored == 0) { // a check..
1634  SpMatrix<double> Vproj(ivector_dim);
1635  Vproj.AddMat2Sp(1.0, V, kNoTrans, covar, 0.0);
1636  KALDI_ASSERT(Vproj.IsUnit(1.0e-04));
1637  }
1638 
1639 
1640  Vector<double> sum_vproj(ivector_dim);
1641  sum_vproj.AddMatVec(1.0, V, kNoTrans, sum, 0.0);
1642  // Make sure sum_vproj is of the form [ x 0 0 0 .. ] with x > 0.
1643  // (the x > 0 part isn't really necessary, it's just nice to know.)
1644  KALDI_ASSERT(ApproxEqual(sum_vproj(0), sum_vproj.Norm(2.0)));
1645 
1646  double ans = PriorDiagnostics(extractor->prior_offset_);
1647 
1648  extractor->TransformIvectors(V, sum_vproj(0));
1649 
1650  return ans;
