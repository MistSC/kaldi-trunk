38 // this struct contains the attributes for a single command. see class
39 // ComputationVariables for the meaning of a variable. note, variables may be
40 // both read and written, e.g. for operations that do += or that write to only
41 // some elements of a variable (we can think of these as, for purposes of
42 // analysis, reading the remaining elements and writing them back... a more
43 // exact analysis would have to define the variables on a more fine-grained
44 // level).
45 struct CommandAttributes {
46  // All of the vector variables below are made sorted and uniq by
47  // ComputeCommandAttributes.
48 
49  // variables read
50  std::vector<int32> variables_read;
51  // variables written
52  std::vector<int32> variables_written;
53 
54  // sub-matrices read (i.e. the submatrix appears in the command directly)
55  std::vector<int32> submatrices_read;
56  // sub-matrices written (i.e. the submatrix appears in the command directly)

58 
59  // matrices read

61  // matrices written

64  // true if this command has side effects e.g. on the model (such as
65  // Backprop on an updatable component, or StoreStats).
66  bool has_side_effects;
67  CommandAttributes(): has_side_effects(false) { }
68 };

179  // Indexed first by matrix-index and then a list, this gives us all the split
180  // points at which column ranges start and end. For instance, if the 3'rd
181  // matrix has 20 columns and is split into ranges 0:9 and 10:19,
