27 // returns true if two are string are equal except for what looks like it might
28 // be a difference last digit of a floating point number, e.g. accept
29 // 1.234 to be the same as 1.235. Not very rigorous.
30 static bool StringsApproxEqual(const std::string &a,
31  const std::string &b) {
32  if (a == b || a.size() != b.size())
33  return true;
34  size_t size = a.size();
35  for (size_t pos = 0; pos < size; pos++) {
36  if (a[pos] != b[pos]) {
37  if (!isdigit(a[pos]) || !isdigit(b[pos]))
38  goto fail;
39  // if it's not the last digit in the string, goto fail
40  if (pos + 1 != size && isdigit(a[pos+1]))
41  goto fail;
42  size_t pos2;
43  for (pos2 = pos - 1; pos2 > 0; pos2--) {
44  if (a[pos2] == '.') break; // we accept this difference: we went backwards and found a '.'
45  if (!isdigit(a[pos2])) // we reject this difference: we went back and
46  // found non-digit before '.' -> not floating
47  // point.
48  goto fail;
49  }
50  if (pos2 == 0)
51  goto fail;
52  }
53  }
54  return true;
55 fail:
56  KALDI_WARN << "Info strings differ: '" << a << "' vs. '" << b << "'.";
57  return false;
58 }

94 void TestNnetComponentVectorizeUnVectorize(Component *c) {
95  if (!(c->Properties() & kUpdatableComponent))
96  return;
97  UpdatableComponent *uc = dynamic_cast<UpdatableComponent*>(c);
98  KALDI_ASSERT(uc != NULL);
99  UpdatableComponent *uc2 = dynamic_cast<UpdatableComponent*>(uc->Copy());
100  uc2->SetZero(false);
101  Vector<BaseFloat> params(uc2->NumParameters());
102  uc2->Vectorize(&params);
103  KALDI_ASSERT(params.Min()==0.0 && params.Sum()==0.0);
104  uc->Vectorize(&params);
105  uc2->UnVectorize(params);
106  KALDI_ASSERT(StringsApproxEqual(uc2->Info(), uc->Info()));
107  BaseFloat x = uc2->DotProduct(*uc2), y = uc->DotProduct(*uc),
108  z = uc2->DotProduct(*uc);
109  KALDI_ASSERT(ApproxEqual(x, y) && ApproxEqual(y, z));
110  Vector<BaseFloat> params2(uc2->NumParameters());
111  uc2->Vectorize(&params2);
112  for(int i = 0; i < params.Dim(); i++)
113  KALDI_ASSERT(params(i) == params2(i));
114  delete uc2;

126 void TestNnetComponentUpdatable(Component *c) {
127  if (!(c->Properties() & kUpdatableComponent))
128  return;
129  UpdatableComponent *uc = dynamic_cast<UpdatableComponent*>(c);
130  if (uc == NULL) {
131  KALDI_ASSERT(!(c->Properties() & kUpdatableComponent) &&
132  "Component returns updatable flag but does not inherit "
133  "from UpdatableComponent");
134  return;
135  }
136  if(!(uc->Properties() & kUpdatableComponent)){
137  // testing that if it declares itself as non-updatable,
138  // Scale() and Add() and SetZero() have no effect.
139  KALDI_ASSERT(uc->NumParameters() == 0);
140  KALDI_ASSERT(uc->DotProduct(*uc) == 0);
141  UpdatableComponent *uc2 = dynamic_cast<UpdatableComponent*>(uc->Copy());
142  uc2->Scale(7.0);
143  uc2->Add(3.0, *uc);
144  KALDI_ASSERT(StringsApproxEqual(uc2->Info(), uc->Info()));
145  uc->SetZero(false);
146  KALDI_ASSERT(StringsApproxEqual(uc2->Info(), uc->Info()));
147  delete uc2;
148  } else {
149  KALDI_ASSERT(uc->NumParameters() != 0);
150  UpdatableComponent *uc2 = dynamic_cast<UpdatableComponent*>(uc->Copy()),
151  *uc3 = dynamic_cast<UpdatableComponent*>(uc->Copy());
152 
153  // testing some expected invariances of scale and add.
154  uc2->Scale(5.0);
155  uc2->Add(3.0, *uc3);
156  uc3->Scale(8.0);
157  // now they should both be scaled to 8 times the original component.
158  if (!StringsApproxEqual(uc2->Info(), uc3->Info())) {
159  KALDI_ERR << "Expected info strings to be equal: '"
160  << uc2->Info() << "' vs. '" << uc3->Info() << "'";
161  }
162  // testing that scaling by 0.5 works the same whether
163  // done on the vectorized paramters or via Scale().
164  Vector<BaseFloat> vec2(uc->NumParameters());
165  uc2->Vectorize(&vec2);
166  vec2.Scale(0.5);
167  uc2->UnVectorize(vec2);
168  uc3->Scale(0.5);
169  KALDI_ASSERT(uc2->Info() == uc3->Info());
170 
171  // testing that SetZero() works the same whether done on the vectorized
172  // paramters or via SetZero(), and that unvectorizing something that's been
173  // zeroed gives us zero parameters.
174  uc2->Vectorize(&vec2);
175  vec2.SetZero();
176  uc2->UnVectorize(vec2);
177  uc3->SetZero(false);
178  uc3->Vectorize(&vec2);
179  KALDI_ASSERT(uc2->Info() == uc3->Info() && VecVec(vec2, vec2) == 0.0);
180 
181  delete uc2;
182  delete uc3;
183  }

189  int32 properties = c.Properties();

192  c_copy = c.Copy(); // This will test backprop with an updatable component.
193  if (RandInt(0, 1) == 0 &&
194  (properties & kLinearInParameters)) {
195  c_copy_scaled = c.Copy(); // This will test backprop with an updatable component.
196  c_copy_scaled->Scale(0.5);
197  }
198  int32 input_dim = c.InputDim(),
199  output_dim = c.OutputDim(),
200  num_rows = RandInt(1, 100);
201  CuMatrix<BaseFloat> input_data(num_rows, input_dim);
202  input_data.SetRandn();
203  CuMatrix<BaseFloat>
204  input_data_scaled(input_data),
205  output_data1(num_rows, output_dim),
206  output_data2(num_rows, output_dim),
207  output_data3(input_data),
208  output_data4(num_rows, output_dim),
209  output_data5(num_rows, output_dim);
210  output_data2.Add(1.0);
211  input_data_scaled.Scale(2.0);
212 
213  if ((properties & kPropagateAdds) && (properties & kPropagateInPlace)) {
214  KALDI_ERR << "kPropagateAdds and kPropagateInPlace flags are incompatible.";
215  }
216 
217  c.Propagate(NULL, input_data, &output_data1);
218  c.Propagate(NULL, input_data, &output_data2);
219  if (properties & kPropagateInPlace) {
220  c.Propagate(NULL, output_data3, &output_data3);
221  if (!output_data1.ApproxEqual(output_data3)) {
222  KALDI_ERR << "Test of kPropagateInPlace flag for component of type "
223  << c.Type() << " failed.";
224  }
225  }
226  if (properties & kPropagateAdds)
227  output_data2.Add(-1.0); // remove the offset
228  AssertEqual(output_data1, output_data2);
229 
230  if (c_copy_scaled) {
231  c_copy_scaled->Propagate(NULL, input_data, &output_data4);
232  output_data4.Scale(2.0); // we scaled the parameters by 0.5 above, and the
233  // output is supposed to be linear in the parameter value.
234  AssertEqual(output_data1, output_data4);
235  }
236  if (properties & kLinearInInput) {
237  c.Propagate(NULL, input_data_scaled, &output_data5);
238  output_data5.Scale(0.5);
239  AssertEqual(output_data1, output_data5);
240  }
241

244  output_deriv.SetRandn();
245  CuMatrix<BaseFloat> input_deriv1(num_rows, input_dim),
246  input_deriv2(num_rows, input_dim),

248  input_deriv2.Add(1.0);

250 
251  // test with input_deriv1 that's zero

280  delete c_copy;
281  delete c_copy_scaled;

284 bool TestSimpleComponentDataDerivative(const Component &c,
285  BaseFloat perturb_delta) {
286  int32 input_dim = c.InputDim(),
287  output_dim = c.OutputDim(),
288  num_rows = RandInt(1, 100);
289  int32 properties = c.Properties();
290  CuMatrix<BaseFloat> input_data(num_rows, input_dim),
291  output_data(num_rows, output_dim),
292  output_deriv(num_rows, output_dim);
293  input_data.SetRandn();
294  output_deriv.SetRandn();
295 
296  c.Propagate(NULL, input_data, &output_data);
297 
298  CuMatrix<BaseFloat> input_deriv(num_rows, input_dim), empty_mat;
299  c.Backprop("foobar", NULL,
300  ((properties & kBackpropNeedsInput) ? input_data : empty_mat),
301  ((properties & kBackpropNeedsOutput) ? output_data : empty_mat),
302  output_deriv, NULL, &input_deriv);
303 
304  int32 test_dim = 3;
305  BaseFloat original_objf = TraceMatMat(output_deriv, output_data, kTrans);
306  Vector<BaseFloat> measured_objf_change(test_dim),
307  predicted_objf_change(test_dim);
308  for (int32 i = 0; i < test_dim; i++) {
309  CuMatrix<BaseFloat> perturbed_input_data(num_rows, input_dim),
310  perturbed_output_data(num_rows, output_dim);
311  perturbed_input_data.SetRandn();
312  perturbed_input_data.Scale(perturb_delta);
313  // at this point, perturbed_input_data contains the offset at the input data.
314  predicted_objf_change(i) = TraceMatMat(perturbed_input_data, input_deriv,
315  kTrans);
316  perturbed_input_data.AddMat(1.0, input_data);
317  c.Propagate(NULL, perturbed_input_data, &perturbed_output_data);
318  measured_objf_change(i) = TraceMatMat(output_deriv, perturbed_output_data,
319  kTrans) - original_objf;
320  }
321  KALDI_LOG << "Predicted objf-change = " << predicted_objf_change;
322  KALDI_LOG << "Measured objf-change = " << measured_objf_change;
323  BaseFloat threshold = 0.1;
324  bool ans = ApproxEqual(predicted_objf_change, measured_objf_change, threshold);
325  if (!ans)
326  KALDI_WARN << "Data-derivative test failed, component-type="
327  << c.Type() << ", input-dim=" << input_dim
328  << ", output-dim=" << output_dim;
329  if (c.Type() == "NormalizeComponent" && input_dim == 1) {
330  // derivatives are mathematically zero, but the measured and predicted
331  // objf have different roundoff and the relative differences are large.
332  // this is not unexpected.
333  KALDI_LOG << "Accepting deriv differences since it is NormalizeComponent "
334  << "with dim=1.";
335  return true;
336  }
337  return ans;
338 }

341 // if test_derivative == false then the test only tests that the update
342 // direction is downhill. if true, then we measure the actual model-derivative
343 // and check that it's accurate.
344 // returns true on success, false on test failure.
345 bool TestSimpleComponentModelDerivative(const Component &c,
346  BaseFloat perturb_delta,
347  bool test_derivative) {
348  int32 input_dim = c.InputDim(),
349  output_dim = c.OutputDim(),
350  num_rows = RandInt(1, 100);
351  int32 properties = c.Properties();
352  if ((properties & kUpdatableComponent) == 0) {
353  // nothing to test.
354  return true;
355  }
356 
357  CuMatrix<BaseFloat> input_data(num_rows, input_dim),
358  output_data(num_rows, output_dim),
359  output_deriv(num_rows, output_dim);
360  input_data.SetRandn();
361  output_deriv.SetRandn();
362 
363  c.Propagate(NULL, input_data, &output_data);
364 
365  BaseFloat original_objf = TraceMatMat(output_deriv, output_data, kTrans);
366 
367  Component *c_copy = c.Copy();
368 
369  const UpdatableComponent *uc = dynamic_cast<const UpdatableComponent*>(&c);
370  UpdatableComponent *uc_copy = dynamic_cast<UpdatableComponent*>(c_copy);
371  KALDI_ASSERT(uc != NULL && uc_copy != NULL);
372  if (test_derivative) {
373  bool is_gradient = true;
374  uc_copy->SetZero(is_gradient);
375  }
376 
377  CuMatrix<BaseFloat> input_deriv(num_rows, input_dim), empty_mat;
378  c.Backprop("foobar", NULL,
379  ((properties & kBackpropNeedsInput) ? input_data : empty_mat),
380  ((properties & kBackpropNeedsOutput) ? output_data : empty_mat),
381  output_deriv, c_copy,
382  (RandInt(0, 1) == 0 ? &input_deriv : NULL));
383 
384  if (!test_derivative) { // Just testing that the model update is downhill.
385  CuMatrix<BaseFloat> new_output_data(num_rows, output_dim);
386  c_copy->Propagate(NULL, input_data, &new_output_data);
387 
388  BaseFloat new_objf = TraceMatMat(output_deriv, new_output_data, kTrans);
389  bool ans = (new_objf > original_objf);
390  if (!ans) {
391  KALDI_WARN << "After update, new objf is not better than the original objf: "
392  << new_objf << " <= " << original_objf;
393  }
394  delete c_copy;
395  return ans;
396  } else {
397  // check that the model derivative is accurate.
398  int32 test_dim = 3;
399 
400  Vector<BaseFloat> measured_objf_change(test_dim),
401  predicted_objf_change(test_dim);
402  for (int32 i = 0; i < test_dim; i++) {
403  CuMatrix<BaseFloat> perturbed_output_data(num_rows, output_dim);
404  Component *c_perturbed = c.Copy();
405  UpdatableComponent *uc_perturbed =
406  dynamic_cast<UpdatableComponent*>(c_perturbed);
407  KALDI_ASSERT(uc_perturbed != NULL);
408  uc_perturbed->PerturbParams(perturb_delta);
409 
410  predicted_objf_change(i) = uc_copy->DotProduct(*uc_perturbed) -
411  uc_copy->DotProduct(*uc);
412  c_perturbed->Propagate(NULL, input_data, &perturbed_output_data);
413  measured_objf_change(i) = TraceMatMat(output_deriv, perturbed_output_data,
414  kTrans) - original_objf;
415  delete c_perturbed;
416  }
417  KALDI_LOG << "Predicted objf-change = " << predicted_objf_change;
418  KALDI_LOG << "Measured objf-change = " << measured_objf_change;
419  BaseFloat threshold = 0.1;
420 
421  bool ans = ApproxEqual(predicted_objf_change, measured_objf_change,
422  threshold);
423  if (!ans)
424  KALDI_WARN << "Model-derivative test failed, component-type="
425  << c.Type() << ", input-dim=" << input_dim
426  << ", output-dim=" << output_dim;
427  delete c_copy;
428  return ans;
429  }
