58 template<typename T>
59 inline bool IsSorted(const std::vector<T> &vec) {
60  typename std::vector<T>::const_iterator iter = vec.begin(), end = vec.end();
61  if (iter == end) return true;
62  while (1) {
63  typename std::vector<T>::const_iterator next_iter = iter;
64  ++next_iter;
65  if (next_iter == end) return true; // end of loop and nothing out of order
66  if (*next_iter < *iter) return false;
67  iter = next_iter;
68  }
69 }

99  // copies members of s into v, in sorted order from lowest to highest
100  // (because the set was in sorted order).

230 struct VectorHasher { // hashing function for vector<Int>.
231  size_t operator()(const std::vector<Int> &x) const {
232  size_t ans = 0;
233  typename std::vector<Int>::const_iterator iter = x.begin(), end = x.end();
234  for (; iter != end; ++iter) {
235  ans *= kPrime;
236  ans += *iter;
237  }
238  return ans;
239  }

241  KALDI_ASSERT_IS_INTEGER_TYPE(Int);
242  }
243  private:
244  static const int kPrime = 7853;

301 template<typename I, typename F>
302 inline void MergePairVectorSumming(std::vector<std::pair<I, F> > *vec) {
303  KALDI_ASSERT_IS_INTEGER_TYPE(I);
304  CompareFirstMemberOfPair<I, F> c;
305  std::sort(vec->begin(), vec->end(), c); // sort on 1st element.
306  typename std::vector<std::pair<I, F> >::iterator out = vec->begin(),
307  in = vec->begin(), end = vec->end();
308  // special case: while there is nothing to be changed, skip over
309  // initial input (avoids unnecessary copying).
310  while (in + 1 < end && in[0].first != in[1].first && in[0].second != 0.0) {
311  in++;
312  out++;
313  }
314  while (in < end) {
315  // We reach this point only at the first element of
316  // each stretch of identical .first elements.
317  *out = *in;
318  ++in;
319  while (in < end && in->first == out->first) {
320  out->second += in->second; // this is the merge operation.
321  ++in;
322  }
323  if (out->second != static_cast<F>(0)) // Don't keep zero elements.
324  out++;
325  }
326  vec->erase(out, end);
327 }
