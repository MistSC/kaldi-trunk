221 inline void Component::Propagate(const CuMatrixBase<BaseFloat> &in,
222  CuMatrix<BaseFloat> *out) {
223  // Check the dims
224  if (input_dim_ != in.NumCols()) {
225  KALDI_ERR << "Non-matching dims! " << TypeToMarker(GetType())
226  << " input-dim : " << input_dim_ << " data : " << in.NumCols();
227  }
228  // Allocate target buffer
229  out->Resize(in.NumRows(), output_dim_, kSetZero); // reset
230  // Call the propagation implementation of the component
231  PropagateFnc(in, out);
232 }

235 inline void Component::Backpropagate(const CuMatrixBase<BaseFloat> &in,
236  const CuMatrixBase<BaseFloat> &out,
237  const CuMatrixBase<BaseFloat> &out_diff,
238  CuMatrix<BaseFloat> *in_diff) {
239  // Check the dims
240  if (output_dim_ != out_diff.NumCols()) {
241  KALDI_ERR << "Non-matching output dims, component:" << output_dim_
242  << " data:" << out_diff.NumCols();
243  }
244 
245  // Target buffer NULL : backpropagate only through components with nested nnets.
246  if (in_diff == NULL) {
247  if (GetType() == kParallelComponent ||
248  GetType() == kSentenceAveragingComponent) {
249  BackpropagateFnc(in, out, out_diff, NULL);
250  } else {
251  return;
252  }
253  } else {
254  // Allocate target buffer
255  in_diff->Resize(out_diff.NumRows(), input_dim_, kSetZero); // reset
256  // Asserts on the dims
257  KALDI_ASSERT((in.NumRows() == out.NumRows()) &&
258  (in.NumRows() == out_diff.NumRows()) &&
259  (in.NumRows() == in_diff->NumRows()));
260  KALDI_ASSERT(in.NumCols() == in_diff->NumCols());
261  KALDI_ASSERT(out.NumCols() == out_diff.NumCols());
262  // Call the backprop implementation of the component
263  BackpropagateFnc(in, out, out_diff, in_diff);
264  }
265 }
