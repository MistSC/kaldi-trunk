26 Mfcc::Mfcc(const MfccOptions &opts)
27  : opts_(opts), feature_window_function_(opts.frame_opts), srfft_(NULL) {
28  int32 num_bins = opts.mel_opts.num_bins;
29  Matrix<BaseFloat> dct_matrix(num_bins, num_bins);
30  ComputeDctMatrix(&dct_matrix);
31  // Note that we include zeroth dct in either case. If using the
32  // energy we replace this with the energy. This means a different
33  // ordering of features than HTK.
34  SubMatrix<BaseFloat> dct_rows(dct_matrix, 0, opts.num_ceps, 0, num_bins);
35  dct_matrix_.Resize(opts.num_ceps, num_bins);
36  dct_matrix_.CopyFromMat(dct_rows); // subset of rows.
37  if (opts.cepstral_lifter != 0.0) {
38  lifter_coeffs_.Resize(opts.num_ceps);
39  ComputeLifterCoeffs(opts.cepstral_lifter, &lifter_coeffs_);
40  }
41  if (opts.energy_floor > 0.0)
42  log_energy_floor_ = Log(opts.energy_floor);
43 
44  int32 padded_window_size = opts.frame_opts.PaddedWindowSize();
45  if ((padded_window_size & (padded_window_size-1)) == 0) // Is a power of two...
46  srfft_ = new SplitRadixRealFft<BaseFloat>(padded_window_size);
47 
48  // We'll definitely need the filterbanks info for VTLN warping factor 1.0.
49  // [note: this call caches it.] The reason we call this here is to
50  // improve the efficiency of the "const" version of Compute().
51  GetMelBanks(1.0);
52 }

116 void Mfcc::ComputeInternal(const VectorBase<BaseFloat> &wave,
117  const MelBanks &mel_banks,
118  Matrix<BaseFloat> *output,
119  Vector<BaseFloat> *wave_remainder) const {
120  KALDI_ASSERT(output != NULL);
121  int32 rows_out = NumFrames(wave.Dim(), opts_.frame_opts),
122  cols_out = opts_.num_ceps;
123  if (rows_out == 0) {
124  output->Resize(0, 0);
125  if (wave_remainder != NULL)
126  *wave_remainder = wave;
127  return;
128  }
129  output->Resize(rows_out, cols_out);
130  if (wave_remainder != NULL)
131  ExtractWaveformRemainder(wave, opts_.frame_opts, wave_remainder);
132  Vector<BaseFloat> window; // windowed waveform.
133  Vector<BaseFloat> mel_energies;
134  std::vector<BaseFloat> temp_buffer; // used by srfft.
135  for (int32 r = 0; r < rows_out; r++) { // r is frame index..
136  BaseFloat log_energy;
137  ExtractWindow(wave, r, opts_.frame_opts, feature_window_function_, &window,
138  (opts_.use_energy && opts_.raw_energy ? &log_energy : NULL));
139 
140  if (opts_.use_energy && !opts_.raw_energy)
141  log_energy = Log(std::max(VecVec(window, window),
142  std::numeric_limits<BaseFloat>::min()));
143 
144  if (srfft_ != NULL) // Compute FFT using the split-radix algorithm.
145  srfft_->Compute(window.Data(), true, &temp_buffer);
146  else // An alternative algorithm that works for non-powers-of-two.
147  RealFft(&window, true);
148 
149  // Convert the FFT into a power spectrum.
150  ComputePowerSpectrum(&window);
151  SubVector<BaseFloat> power_spectrum(window, 0, window.Dim()/2 + 1);
152 
153  mel_banks.Compute(power_spectrum, &mel_energies);
154 
155  // avoid log of zero (which should be prevented anyway by dithering).
156  mel_energies.ApplyFloor(std::numeric_limits<BaseFloat>::min());
157  mel_energies.ApplyLog(); // take the log.
158 
159  SubVector<BaseFloat> this_mfcc(output->Row(r));
160 
161  // this_mfcc = dct_matrix_ * mel_energies [which now have log]
162  this_mfcc.AddMatVec(1.0, dct_matrix_, kNoTrans, mel_energies, 0.0);
163 
164  if (opts_.cepstral_lifter != 0.0)
165  this_mfcc.MulElements(lifter_coeffs_);
166 
167  if (opts_.use_energy) {
168  if (opts_.energy_floor > 0.0 && log_energy < log_energy_floor_)
169  log_energy = log_energy_floor_;
170  this_mfcc(0) = log_energy;
171  }
172 
173  if (opts_.htk_compat) {
174  BaseFloat energy = this_mfcc(0);
175  for (int32 i = 0; i < opts_.num_ceps-1; i++)
176  this_mfcc(i) = this_mfcc(i+1);
177  if (!opts_.use_energy)
178  energy *= M_SQRT2; // scale on C0 (actually removing scale
179  // we previously added that's part of one common definition of
180  // cosine transform.)
181  this_mfcc(opts_.num_ceps-1) = energy;
182  }
183  }
184 }
