110 // There is an extended comment within this file, referencing a paper by Ioffe, that
111 // may clarify what this function is doing.
112 double Plda::LogLikelihoodRatio(
113  const VectorBase<double> &transformed_train_ivector,
114  int32 n, // number of training utterances.
115  const VectorBase<double> &transformed_test_ivector) {
116  int32 dim = Dim();
117  double loglike_given_class, loglike_without_class;
118  { // work out loglike_given_class.
119  // "mean" will be the mean of the distribution if it comes from the
120  // training example. The mean is \frac{n \Psi}{n \Psi + I} \bar{u}^g
121  // "variance" will be the variance of that distribution, equal to
122  // I + \frac{\Psi}{n\Psi + I}.
123  Vector<double> mean(dim, kUndefined);
124  Vector<double> variance(dim, kUndefined);
125  for (int32 i = 0; i < dim; i++) {
126  mean(i) = n * psi_(i) / (n * psi_(i) + 1.0) * transformed_train_ivector(i);
127  variance(i) = 1.0 + psi_(i) / (n * psi_(i) + 1.0);
128  }
129  double logdet = variance.SumLog();
130  Vector<double> sqdiff(transformed_test_ivector);
131  sqdiff.AddVec(-1.0, mean);
132  sqdiff.ApplyPow(2.0);
133  variance.InvertElements();
134  loglike_given_class = -0.5 * (logdet + M_LOG_2PI * dim +
135  VecVec(sqdiff, variance));
136  }
137  { // work out loglike_without_class. Here the mean is zero and the variance
138  // is I + \Psi.
139  Vector<double> sqdiff(transformed_test_ivector); // there is no offset.
140  sqdiff.ApplyPow(2.0);
141  Vector<double> variance(psi_);
142  variance.Add(1.0); // I + \Psi.
143  double logdet = variance.SumLog();
144  variance.InvertElements();
145  loglike_without_class = -0.5 * (logdet + M_LOG_2PI * dim +
146  VecVec(sqdiff, variance));
147  }
148  double loglike_ratio = loglike_given_class - loglike_without_class;
149  return loglike_ratio;
150 }

178 void PldaStats::AddSamples(double weight,
179  const Matrix<double> &group) {
180  if (dim_ == 0) {
181  Init(group.NumCols());
182  } else {
183  KALDI_ASSERT(dim_ == group.NumCols());
184  }
185  int32 n = group.NumRows(); // number of examples for this class
186  Vector<double> *mean = new Vector<double>(dim_);
187  mean->AddRowSumMat(1.0 / n, group);
188 
189  offset_scatter_.AddMat2(weight, group, kTrans, 1.0);
190  // the following statement has the same effect as if we
191  // had first subtracted the mean from each element of
192  // the group before the statement above.
193  offset_scatter_.AddVec2(-n * weight, *mean);
194 
195  class_info_.push_back(ClassInfo(weight, mean, n));
196 
197  num_classes_ ++;
198  num_examples_ += n;
199  class_weight_ += weight;
200  example_weight_ += weight * n;
201 
202  sum_.AddVec(weight, *mean);
203 }

237 double PldaEstimator::ComputeObjfPart1() const {
238  // Returns the part of the objf relating to offsets from the class means.
239  // within_class_count equals the sum over the classes, of the weight of that
240  // class (normally 1) times (1 - #examples) of that class, which equals the
241  // rank of the covariance we're modeling. We imagine that we're modeling (1 -
242  // #examples) separate samples, each with the within-class covariance.. the
243  // argument is a little complicated and involves an orthogonal complement of a
244  // matrix whose first row computes the mean.
245 
246  double within_class_count = stats_.example_weight_ - stats_.class_weight_,
247  within_logdet, det_sign;
248  SpMatrix<double> inv_within_var(within_var_);
249  inv_within_var.Invert(&within_logdet, &det_sign);
250  KALDI_ASSERT(det_sign == 1 && "Within-class covariance is singular");
251 
252  double objf = -0.5 * (within_class_count * (within_logdet + M_LOG_2PI * Dim())
253  + TraceSpSp(inv_within_var, stats_.offset_scatter_));
254  return objf;
255 }

257 double PldaEstimator::ComputeObjfPart2() const {
258  double tot_objf = 0.0;
259 
260  int32 n = -1; // the number of examples for the current class
261  SpMatrix<double> combined_inv_var(Dim());
262  // combined_inv_var = (between_var_ + within_var_ / n)^{-1}
263  double combined_var_logdet;
264 
265  for (size_t i = 0; i < stats_.class_info_.size(); i++) {
266  const ClassInfo &info = stats_.class_info_[i];
267  if (info.num_examples != n) {
268  n = info.num_examples;
269  // variance of mean of n examples is between-class + 1/n * within-class
270  combined_inv_var.CopyFromSp(between_var_);
271  combined_inv_var.AddSp(1.0 / n, within_var_);
272  combined_inv_var.Invert(&combined_var_logdet);
273  }
274  Vector<double> mean (*(info.mean));
275  mean.AddVec(-1.0 / stats_.class_weight_, stats_.sum_);
276  tot_objf += info.weight * -0.5 * (combined_var_logdet + M_LOG_2PI * Dim()
277  + VecSpVec(mean, combined_inv_var, mean));
278  }
279  return tot_objf;
280 }

282 double PldaEstimator::ComputeObjf() const {
283  double ans1 = ComputeObjfPart1(),
284  ans2 = ComputeObjfPart2(),

362 void PldaEstimator::GetStatsFromClassMeans() {
363  SpMatrix<double> between_var_inv(between_var_);
364  between_var_inv.Invert();
365  SpMatrix<double> within_var_inv(within_var_);
366  within_var_inv.Invert();
367  // mixed_var will equal (between_var^{-1} + n within_var^{-1})^{-1}.
368  SpMatrix<double> mixed_var(Dim());
369  int32 n = -1; // the current number of examples for the class.
370 
371  for (size_t i = 0; i < stats_.class_info_.size(); i++) {
372  const ClassInfo &info = stats_.class_info_[i];
373  double weight = info.weight;
374  if (info.num_examples != n) {
375  n = info.num_examples;
376  mixed_var.CopyFromSp(between_var_inv);
377  mixed_var.AddSp(n, within_var_inv);
378  mixed_var.Invert();
379  }
380  Vector<double> m = *(info.mean); // the mean for this class.
381  m.AddVec(-1.0 / stats_.class_weight_, stats_.sum_); // remove global mean
382  Vector<double> temp(Dim()); // n within_var^{-1} m
383  temp.AddSpVec(n, within_var_inv, m, 0.0);
384  Vector<double> w(Dim()); // w, as defined in the comment.
385  w.AddSpVec(1.0, mixed_var, temp, 0.0);
386  Vector<double> m_w(m); // m - w
387  m_w.AddVec(-1.0, w);
388  between_var_stats_.AddSp(weight, mixed_var);
389  between_var_stats_.AddVec2(weight, w);
390  between_var_count_ += weight;
391  within_var_stats_.AddSp(weight * n, mixed_var);
392  within_var_stats_.AddVec2(weight * n, m_w);
393  within_var_count_ += weight;
394  }
395 }

428 template<class Real>
429 static void ComputeNormalizingTransform(const SpMatrix<Real> &covar,
430  MatrixBase<Real> *proj) {
431  int32 dim = covar.NumRows();
432  TpMatrix<Real> C(dim); // Cholesky of covar, covar = C C^T
433  C.Cholesky(covar);
434  C.Invert(); // The matrix that makes covar unit is C^{-1}, because
435  // C^{-1} covar C^{-T} = C^{-1} C C^T C^{-T} = I.
436  proj->CopyFromTp(C, kNoTrans); // set "proj" to C^{-1}.
437 }

440 void PldaEstimator::GetOutput(Plda *plda) {
441  plda->mean_ = stats_.sum_;
442  plda->mean_.Scale(1.0 / stats_.class_weight_);
443  KALDI_LOG << "Norm of mean of iVector distribution is "
444  << plda->mean_.Norm(2.0);
445 
446  Matrix<double> transform1(Dim(), Dim());
447  ComputeNormalizingTransform(within_var_, &transform1);
448  // now transform is a matrix that if we project with it,
449  // within_var_ becomes unit.
450 
451  // between_var_proj is between_var after projecting with transform1.
452  SpMatrix<double> between_var_proj(Dim());
453  between_var_proj.AddMat2Sp(1.0, transform1, kNoTrans, between_var_, 0.0);
454 
455  Matrix<double> U(Dim(), Dim());
456  Vector<double> s(Dim());
457  // Do symmetric eigenvalue decomposition between_var_proj = U diag(s) U^T,
458  // where U is orthogonal.
459  between_var_proj.Eig(&s, &U);
460 
461  KALDI_ASSERT(s.Min() >= 0.0);
462  int32 n = s.ApplyFloor(0.0);
463  if (n > 0) {
464  KALDI_WARN << "Floored " << n << " eigenvalues of between-class "
465  << "variance to zero.";
466  }
467  // Sort from greatest to smallest eigenvalue.
468  SortSvd(&s, &U);
469 
470  // The transform U^T will make between_var_proj diagonal with value s
471  // (i.e. U^T U diag(s) U U^T = diag(s)). The final transform that
472  // makes within_var_ unit and between_var_ diagonal is U^T transform1,
473  // i.e. first transform1 and then U^T.
474 
475  plda->transform_.Resize(Dim(), Dim());
476  plda->transform_.AddMatMat(1.0, U, kTrans, transform1, kNoTrans, 0.0);
477  plda->psi_ = s;
478 
479  KALDI_LOG << "Diagonal of between-class variance in normalized space is " << s;
480 
481  if (GetVerboseLevel() >= 2) { // at higher verbose levels, do a self-test
482  // (just tests that this function does what it
483  // should).
484  SpMatrix<double> tmp_within(Dim());
485  tmp_within.AddMat2Sp(1.0, plda->transform_, kNoTrans, within_var_, 0.0);
486  KALDI_ASSERT(tmp_within.IsUnit(0.0001));
487  SpMatrix<double> tmp_between(Dim());
488  tmp_between.AddMat2Sp(1.0, plda->transform_, kNoTrans, between_var_, 0.0);
489  KALDI_ASSERT(tmp_between.IsDiagonal(0.0001));
490  Vector<double> psi(Dim());
491  psi.CopyDiagFromSp(tmp_between);
492  AssertEqual(psi, plda->psi_);
493  }
494  plda->ComputeDerivedVars();
495 }

515 void PldaUnsupervisedAdaptor::UpdatePlda(const PldaUnsupervisedAdaptorConfig &config,
516  Plda *plda) const {
517  KALDI_ASSERT(tot_weight_ > 0.0);
518  int32 dim = mean_stats_.Dim();
519  KALDI_ASSERT(dim == plda->Dim());
520  Vector<double> mean(mean_stats_);
521  mean.Scale(1.0 / tot_weight_);
522  SpMatrix<double> variance(variance_stats_);
523  variance.Scale(1.0 / tot_weight_);
524  variance.AddVec2(-1.0, mean); // Make it the uncentered variance.
525 
526  // mean_diff of the adaptation data from the training data. We optionally add
527  // this to our total covariance matrix
528  Vector<double> mean_diff(mean);
529  mean_diff.AddVec(-1.0, plda->mean_);
530  KALDI_ASSERT(config.mean_diff_scale >= 0.0);
531  variance.AddVec2(config.mean_diff_scale, mean_diff);
532 
533  // update the plda's mean data-member with our adaptation-data mean.
534  plda->mean_.CopyFromVec(mean);
535 
536 
537  // transform_model_ is a row-scaled version of plda->transform_ that
538  // transforms into the space where the total covariance is 1.0. Because
539  // plda->transform_ transforms into a space where the within-class covar is
540  // 1.0 and the the between-class covar is diag(plda->psi_), we need to scale
541  // each dimension i by 1.0 / sqrt(1.0 + plda->psi_(i))
542 
543  Matrix<double> transform_mod(plda->transform_);
544  for (int32 i = 0; i < dim; i++)
545  transform_mod.Row(i).Scale(1.0 / sqrt(1.0 + plda->psi_(i)));
546 
547  // project the variance of the adaptation set into this space where
548  // the total covariance is unit.
549  SpMatrix<double> variance_proj(dim);
550  variance_proj.AddMat2Sp(1.0, transform_mod, kNoTrans,
551  variance, 0.0);
552 
553  // Do eigenvalue decomposition of variance_proj; this will tell us the
554  // directions in which the adaptation-data covariance is more than
555  // the training-data covariance.
556  Matrix<double> P(dim, dim);
557  Vector<double> s(dim);
558  variance_proj.Eig(&s, &P);
559  SortSvd(&s, &P);
560  KALDI_LOG << "Eigenvalues of adaptation-data total-covariance in space where "
561  << "training-data total-covariance is unit, is: " << s;
562 
563  // W, B are the (within,between)-class covars in the space transformed by
564  // transform_mod.
565  SpMatrix<double> W(dim), B(dim);
566  for (int32 i = 0; i < dim; i++) {
567  W(i, i) = 1.0 / (1.0 + plda->psi_(i)),
568  B(i, i) = plda->psi_(i) / (1.0 + plda->psi_(i));
569  }
570 
571  // OK, so variance_proj (projected by transform_mod) is P diag(s) P^T.
572  // Suppose that after transform_mod we project by P^T. Then the adaptation-data's
573  // variance would be P^T P diag(s) P^T P = diag(s), and the PLDA model's
574  // within class variance would be P^T W P and its between-class variance would be
575  // P^T B P. We'd still have that W+B = I in this space.
576  // First let's compute these projected variances... we call the "proj2" because
577  // it's after the data has been projected twice (actually, transformed, as there is no
578  // dimension loss), by transform_mod and then P^T.
579 
580  SpMatrix<double> Wproj2(dim), Bproj2(dim);
581  Wproj2.AddMat2Sp(1.0, P, kTrans, W, 0.0);
582  Bproj2.AddMat2Sp(1.0, P, kTrans, B, 0.0);
583 
584  Matrix<double> Ptrans(P, kTrans);
585 
586  SpMatrix<double> Wproj2mod(Wproj2), Bproj2mod(Bproj2);
587 
588  for (int32 i = 0; i < dim; i++) {
589  // For this eigenvalue, compute the within-class covar projected with this direction,
590  // and the same for between.
591  BaseFloat within = Wproj2(i, i),
592  between = Bproj2(i, i);
593  KALDI_LOG << "For " << i << "'th eigenvalue, value is " << s(i)
594  << ", within-class covar in this direction is " << within
595  << ", between-class is " << between;
596  if (s(i) > 1.0) {
597  double excess_eig = s(i) - 1.0;
598  double excess_within_covar = excess_eig * config.within_covar_scale,
599  excess_between_covar = excess_eig * config.between_covar_scale;
600  Wproj2mod(i, i) += excess_within_covar;
601  Bproj2mod(i, i) += excess_between_covar;
602  } /*
603  Below I was considering a method like below, to try to scale up
604  the dimensions that had less variance than expected in our sample..
605  this didn't help, and actually when I set that power to +0.2 instead
606  of -0.5 it gave me an improvement on sre08. But I'm not sure
607  about this.. it just doesn't seem right.
608  else {
609  BaseFloat scale = pow(std::max(1.0e-10, s(i)), -0.5);
610  BaseFloat max_scale = 10.0; // I'll make this configurable later.
611  scale = std::min(scale, max_scale);
612  Ptrans.Row(i).Scale(scale);
613  } */
614  }
615 
616  // combined transform "transform_mod" and then P^T that takes us to the space
617  // where {W,B}proj2{,mod} are.
618  Matrix<double> combined_trans(dim, dim);
619  combined_trans.AddMatMat(1.0, Ptrans, kNoTrans,
620  transform_mod, kNoTrans, 0.0);
621  Matrix<double> combined_trans_inv(combined_trans); // ... and its inverse.
622  combined_trans_inv.Invert();
623 
624  // Wmod and Bmod are as Wproj2 and Bproj2 but taken back into the original
625  // iVector space.
626  SpMatrix<double> Wmod(dim), Bmod(dim);
627  Wmod.AddMat2Sp(1.0, combined_trans_inv, kNoTrans, Wproj2mod, 0.0);
628  Bmod.AddMat2Sp(1.0, combined_trans_inv, kNoTrans, Bproj2mod, 0.0);
629 
630  TpMatrix<double> C(dim);
631  // Do Cholesky Wmod = C C^T. Now if we use C^{-1} as a transform, we have
632  // C^{-1} W C^{-T} = I, so it makes the within-class covar unit.
633  C.Cholesky(Wmod);
634  TpMatrix<double> Cinv(C);
635  Cinv.Invert();
636 
637  // Bmod_proj is Bmod projected by Cinv.
638  SpMatrix<double> Bmod_proj(dim);
639  Bmod_proj.AddTp2Sp(1.0, Cinv, kNoTrans, Bmod, 0.0);
640  Vector<double> psi_new(dim);
641  Matrix<double> Q(dim, dim);
642  // Do symmetric eigenvalue decomposition of Bmod_proj, so
643  // Bmod_proj = Q diag(psi_new) Q^T
644  Bmod_proj.Eig(&psi_new, &Q);
645  SortSvd(&psi_new, &Q);
646  // This means that if we use Q^T as a transform, then Q^T Bmod_proj Q =
647  // diag(psi_new), hence Q^T diagonalizes Bmod_proj (while leaving the
648  // within-covar unit).
649  // The final transform we want, that projects from our original
650  // space to our newly normalized space, is:
651  // first Cinv, then Q^T, i.e. the
652  // matrix Q^T Cinv.
653  Matrix<double> final_transform(dim, dim);
654  final_transform.AddMatTp(1.0, Q, kTrans, Cinv, kNoTrans, 0.0);
655 
656  KALDI_LOG << "Old diagonal of between-class covar was: "
657  << plda->psi_ << ", new diagonal is "
658  << psi_new;
659  plda->transform_.CopyFromMat(final_transform);
660  plda->psi_.CopyFromVec(psi_new);
661 }
