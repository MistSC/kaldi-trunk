30 void DistributeComponent::ComputeInputIndexAndBlock(const Index &output_index,
31  Index *input_index,
32  int32 *block) const {
33  int32 num_blocks = input_dim_ / output_dim_;
34  *input_index = output_index;
35  int32 output_x = output_index.x, input_x;
36  if (output_x >= 0) {
37  input_x = output_x / num_blocks;
38  } else {
39  input_x = (output_x - num_blocks + 1) / num_blocks;
40  }
41  input_index->x = input_x;
42  if (block)
43  *block = output_x - (input_x * num_blocks);
44 }

90 // virtual
91 ComponentPrecomputedIndexes* DistributeComponent::PrecomputeIndexes(
92  const MiscComputationInfo &, // misc_info
93  const std::vector<Index> &input_indexes,
94  const std::vector<Index> &output_indexes,
95  bool) const { // the bool is 'need_backprop'- unused.
96  unordered_map<Index, int32, IndexHasher> index_to_input_dim;
97  int32 num_input_indexes = input_indexes.size(),
98  num_output_indexes = output_indexes.size();
99  for (int32 i = 0; i < num_input_indexes; i++)
100  index_to_input_dim[input_indexes[i]] = i;
101  DistributeComponentPrecomputedIndexes *ans = new
102  DistributeComponentPrecomputedIndexes;
103  ans->pairs.resize(output_indexes.size());
104 
105  int32 num_blocks = input_dim_ / output_dim_,
106  block_size = input_dim_ / num_blocks;
107 
108  for (int32 i = 0; i < num_output_indexes; i++) {
109  Index input_index;
110  int32 block_index;
111  ComputeInputIndexAndBlock(output_indexes[i], &input_index, &block_index);
112  unordered_map<Index, int32, IndexHasher>::iterator iter =
113  index_to_input_dim.find(input_index);
114  if (iter == index_to_input_dim.end())
115  KALDI_ERR << "Input index not found (code error)";
116  int32 input_row = iter->second;
117  ans->pairs[i] = std::pair<int32,int32>(input_row, block_index * block_size);
118  }
119  return ans;
120 }

123 void DistributeComponent::ComputeInputPointers(
124  const ComponentPrecomputedIndexes *indexes_in,
125  const CuMatrixBase<BaseFloat> &in,
126  int32 num_output_rows,
127  std::vector<const BaseFloat*> *input_pointers) const {
128  const DistributeComponentPrecomputedIndexes *indexes =
129  dynamic_cast<const DistributeComponentPrecomputedIndexes*>(indexes_in);
130  KALDI_ASSERT(indexes != NULL && "Invalid pointer type");
131  KALDI_ASSERT(num_output_rows == static_cast<int32>(indexes->pairs.size()));
132  input_pointers->resize(num_output_rows);
133 
134  const BaseFloat *input_data = in.Data();
135  int32 input_stride = in.Stride();
136  const BaseFloat **input_pointers_data = &((*input_pointers)[0]);
137  const std::pair<int32, int32> *pairs_data = &(indexes->pairs[0]);
138  for (int32 i = 0; i < num_output_rows; i++) {
139  input_pointers_data[i] = input_data +
140  pairs_data[i].first * input_stride +
141  pairs_data[i].second;
142  }
143 }

169 void DistributeComponent::Propagate(const ComponentPrecomputedIndexes *indexes,
170  const CuMatrixBase<BaseFloat> &in,
171  CuMatrixBase<BaseFloat> *out) const {
172  KALDI_ASSERT(indexes != NULL &&
173  in.NumCols() == input_dim_ && out->NumCols() == output_dim_);
174  int32 num_output_rows = out->NumRows();
175  std::vector<const BaseFloat*> input_pointers;
176  ComputeInputPointers(indexes, in, num_output_rows, &input_pointers);
177  CuArray<const BaseFloat*> input_pointers_cuda(input_pointers);
178  out->CopyRows(input_pointers_cuda);
179 }

181 // virtual
182 void DistributeComponent::Backprop(const std::string &debug_info,
183  const ComponentPrecomputedIndexes *indexes,
184  const CuMatrixBase<BaseFloat> &, // in_value,
185  const CuMatrixBase<BaseFloat> &, // out_value
186  const CuMatrixBase<BaseFloat> &out_deriv,
187  Component *, // to_update,
188  CuMatrixBase<BaseFloat> *in_deriv) const {
189  if (in_deriv == NULL) return;
190 
191  int32 num_blocks = input_dim_ / output_dim_,
192  num_output_rows = out_deriv.NumRows();
193  if (num_output_rows != in_deriv->NumRows() * num_blocks) {
194  // there could be some 'gaps', i.e. some input values that are not ever
195  // referred to. So we need to zero the input. This would't happen in the
196  // setups I plan to use this for.
197  in_deriv->SetZero();
198  }
199 
200  std::vector<BaseFloat*> input_pointers;
201  ComputeInputPointers(indexes, num_output_rows, in_deriv, &input_pointers);
202  CuArray<BaseFloat*> input_pointers_cuda(input_pointers);
203  out_deriv.CopyToRows(input_pointers_cuda);
204 }
