41 template<typename Real>
42 void CuRand<Real>::SeedBuffer(MatrixIndexT state_size, uint32 **tgt) {
43 #if HAVE_CUDA == 1
44  CuDevice &device = CuDevice::Instantiate();
45  if (device.Enabled()) {
46  if (*tgt != NULL) {
47  device.Free(*tgt);
48  *tgt = NULL;
49  }
50  if (state_size == 0) return; // Nothing to do.
51  std::vector<uint32> temp_rand_data(state_size);
52  for(MatrixIndexT i = 0; i < state_size; i++)
53  temp_rand_data[i] = RandInt(128, RAND_MAX);
54  int32 state_size_in_bytes = state_size * sizeof(uint32);
55  *tgt = static_cast<uint32*>(device.Malloc(state_size_in_bytes));
56  CU_SAFE_CALL(cudaMemcpy(*tgt, &(temp_rand_data[0]),
57  state_size_in_bytes, cudaMemcpyHostToDevice));
58  }
59 #endif
60 }

144 template<typename Real> void CuRand<Real>::BinarizeProbs(const CuMatrix<Real> &probs, CuMatrix<Real> *states) {
145 #if HAVE_CUDA == 1
146  if (CuDevice::Instantiate().Enabled()) {
147  Timer tim;
148 
149  // optionally re-seed the inner state
150  // (this is done in host, for good performance it is better to avoid re-seeding)
151  int32 tgt_size = probs.num_rows_ * probs.stride_;
152  if (tgt_size != state_size_) SeedGpu(tgt_size);
153 
154  // prepare the output matrix
155  if (states != &probs)
156  states->Resize(probs.num_rows_, probs.num_cols_, kUndefined);
157  // prepare the temporary matrix of uniform random numbers (0,1)
158  tmp_.Resize(probs.num_rows_, probs.num_cols_, kUndefined);
159  RandUniform(&tmp_);
160 
161  // use the uniform random numbers to compute discrete 0/1 states
162  dim3 dimBlock(CU2DBLOCK, CU2DBLOCK);
163  dim3 dimGrid(n_blocks(states->num_cols_, CU2DBLOCK), n_blocks(states->num_rows_, CU2DBLOCK));
164 
165  cuda_binarize_probs(dimGrid, dimBlock, states->data_, probs.data_, tmp_.data_, states->Dim());
166  CU_SAFE_CALL(cudaGetLastError());
167 
168  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
169  } else
170 #endif
171  {
172  for(int32 r=0; r<states->num_rows_; r++) {
173  for(int32 c=0; c<states->num_cols_; c++) {
174  states->Mat()(r, c) = ((kaldi::RandUniform() < probs.Mat()(r, c))? 1 : 0 );
175  }
176  }
177  }
178 }
