38 template <typename T>
39 inline void CountCorrectFramesWeighted(const CuArray<T> &v1,
40  const CuArray<T> &v2,
41  const CuVectorBase<BaseFloat> &weights,
42  double *correct) {
43  KALDI_ASSERT(v1.Dim() == v2.Dim());
44  KALDI_ASSERT(v1.Dim() == weights.Dim());
45  int32 dim = v1.Dim();
46  // Get GPU data to host,
47  std::vector<T> v1_h(dim), v2_h(dim);
48  v1.CopyToVec(&v1_h);
49  v2.CopyToVec(&v2_h);
50  Vector<BaseFloat> w(dim);
51  weights.CopyToVec(&w);
52  // Get correct frame count (weighted),
53  double corr = 0.0;
54  for (int32 i=0; i<dim; i++) {
55  corr += w(i) * (v1_h[i] == v2_h[i] ? 1.0 : 0.0);
56  }
57  // Return,
58  (*correct) = corr;
59 }

62 void Xent::Eval(const VectorBase<BaseFloat> &frame_weights,
63  const CuMatrixBase<BaseFloat> &net_out,
64  const CuMatrixBase<BaseFloat> &targets,
65  CuMatrix<BaseFloat> *diff) {
66  // check inputs,
67  KALDI_ASSERT(net_out.NumCols() == targets.NumCols());
68  KALDI_ASSERT(net_out.NumRows() == targets.NumRows());
69  KALDI_ASSERT(net_out.NumRows() == frame_weights.Dim());
70 
71  KALDI_ASSERT(KALDI_ISFINITE(frame_weights.Sum()));
72  KALDI_ASSERT(KALDI_ISFINITE(net_out.Sum()));
73  KALDI_ASSERT(KALDI_ISFINITE(targets.Sum()));
74 
75  // get frame_weights to GPU,
76  frame_weights_ = frame_weights;
77 
78  // There may be frames for which the sum of targets is zero.
79  // This happens in multi-lingual training when the frame
80  // has target class in the softmax of another language.
81  // We 'switch-off' such frames by masking the 'frame_weights_',
82  target_sum_.Resize(targets.NumRows());
83  target_sum_.AddColSumMat(1.0, targets, 0.0);
84  frame_weights_.MulElements(target_sum_);
85 
86  // get the number of frames after the masking,
87  double num_frames = frame_weights_.Sum();
88  KALDI_ASSERT(num_frames >= 0.0);
89 
90  // compute derivative wrt. activations of last layer of neurons,
91  *diff = net_out;
92  diff->AddMat(-1.0, targets);
93  diff->MulRowsVec(frame_weights_); // weighting,
94 
95  // evaluate the frame-level classification,
96  double correct;
97  net_out.FindRowMaxId(&max_id_out_); // find max in nn-output
98  targets.FindRowMaxId(&max_id_tgt_); // find max in targets
99  CountCorrectFramesWeighted(max_id_out_, max_id_tgt_, frame_weights_, &correct);
100 
101  // calculate cross_entropy (in GPU),
102  xentropy_aux_ = net_out; // y
103  xentropy_aux_.Add(1e-20); // avoid log(0)
104  xentropy_aux_.ApplyLog(); // log(y)
105  xentropy_aux_.MulElements(targets); // t*log(y)
106  xentropy_aux_.MulRowsVec(frame_weights_); // w*t*log(y)
107  double cross_entropy = -xentropy_aux_.Sum();
108 
109  // caluculate entropy (in GPU),
110  entropy_aux_ = targets; // t
111  entropy_aux_.Add(1e-20); // avoid log(0)
112  entropy_aux_.ApplyLog(); // log(t)
113  entropy_aux_.MulElements(targets); // t*log(t)
114  entropy_aux_.MulRowsVec(frame_weights_); // w*t*log(t)
115  double entropy = -entropy_aux_.Sum();
116 
117  KALDI_ASSERT(KALDI_ISFINITE(cross_entropy));
118  KALDI_ASSERT(KALDI_ISFINITE(entropy));
119 
120  loss_ += cross_entropy;
121  entropy_ += entropy;
122  correct_ += correct;
123  frames_ += num_frames;
124 
125  // progressive loss reporting
126  {
127  static const int32 progress_step = 3600*100; // 1h
128  frames_progress_ += num_frames;
129  loss_progress_ += cross_entropy;
130  entropy_progress_ += entropy;
131  if (frames_progress_ > progress_step) {
132  KALDI_VLOG(1) << "ProgressLoss[last "
133  << static_cast<int>(frames_progress_/100/3600) << "h of "
134  << static_cast<int>(frames_/100/3600) << "h]: "
135  << (loss_progress_-entropy_progress_)/frames_progress_ << " (Xent)";
136  // store
137  loss_vec_.push_back((loss_progress_-entropy_progress_)/frames_progress_);
138  // reset
139  frames_progress_ = 0;
140  loss_progress_ = 0.0;
141  entropy_progress_ = 0.0;
142  }
143  }
144 }

183 void Mse::Eval(const VectorBase<BaseFloat> &frame_weights,
184  const CuMatrixBase<BaseFloat>& net_out,
185  const CuMatrixBase<BaseFloat>& target,
186  CuMatrix<BaseFloat>* diff) {
187  // check inputs,
188  KALDI_ASSERT(net_out.NumCols() == target.NumCols());
189  KALDI_ASSERT(net_out.NumRows() == target.NumRows());
190  KALDI_ASSERT(net_out.NumRows() == frame_weights.Dim());
191 
192  KALDI_ASSERT(KALDI_ISFINITE(frame_weights.Sum()));
193  KALDI_ASSERT(KALDI_ISFINITE(net_out.Sum()));
194  KALDI_ASSERT(KALDI_ISFINITE(target.Sum()));
195 
196  int32 num_frames = frame_weights.Sum();
197  KALDI_ASSERT(num_frames >= 0.0);
198 
199  // get frame_weights to GPU,
200  frame_weights_ = frame_weights;
201 
202  //compute derivative w.r.t. neural nerwork outputs
203  *diff = net_out; // y
204  diff->AddMat(-1.0,target); // (y - t)
205  diff->MulRowsVec(frame_weights_); // weighting,
206 
207  // Compute MeanSquareError loss of mini-batch
208  diff_pow_2_ = *diff;
209  diff_pow_2_.MulElements(diff_pow_2_); // (y - t)^2
210  diff_pow_2_.MulRowsVec(frame_weights_); // w*(y - t)^2
211  double mean_square_error = 0.5 * diff_pow_2_.Sum(); // sum the matrix,
212 
213  KALDI_ASSERT(KALDI_ISFINITE(mean_square_error));
214 
215  // accumulate
216  loss_ += mean_square_error;
217  frames_ += num_frames;
218 
219  // progressive loss reporting
220  {
221  static const int32 progress_step = 3600*100; // 1h
222  frames_progress_ += num_frames;
223  loss_progress_ += mean_square_error;
224  if (frames_progress_ > progress_step) {
225  KALDI_VLOG(1) << "ProgressLoss[last "
226  << static_cast<int>(frames_progress_/100/3600) << "h of "
227  << static_cast<int>(frames_/100/3600) << "h]: "
228  << loss_progress_/frames_progress_ << " (Mse)";
229  // store
230  loss_vec_.push_back(loss_progress_/frames_progress_);
231  // reset
232  frames_progress_ = 0;
233  loss_progress_ = 0.0;
234  }
235  }
236 }

272 void MultiTaskLoss::InitFromString(const std::string& s) {
273  std::vector<std::string> v;
274  SplitStringToVector(s, ",:" /* delimiter */, false, &v);
275 
276  KALDI_ASSERT((v.size()-1) % 3 == 0); // triplets,
277  KALDI_ASSERT(v[0] == "multitask"); // header,
278 
279  // parse the definition of multitask loss,
280  std::vector<std::string>::iterator it(v.begin()+1); // skip header,
281  for ( ; it != v.end(); ++it) {
282  // type,
283  if (*it == "xent") {
284  loss_vec_.push_back(new Xent());
285  } else if (*it == "mse") {
286  loss_vec_.push_back(new Mse());
287  } else {
288  KALDI_ERR << "Unknown objective function code : " << *it;
289  }
290  ++it;
291  // dim,
292  int32 dim;
293  if (!ConvertStringToInteger(*it, &dim)) {
294  KALDI_ERR << "Cannot convert 'dim' " << *it << " to integer!";
295  }
296  loss_dim_.push_back(dim);
297  ++it;
298  // weight,
299  BaseFloat weight;
300  if (!ConvertStringToReal(*it, &weight)) {
301  KALDI_ERR << "Cannot convert 'weight' " << *it << " to integer!";
302  }
303  KALDI_ASSERT(weight >= 0.0);
304  loss_weights_.push_back(weight);
305  }
306 
307  // build vector with starting-point offsets,
308  loss_dim_offset_.resize(loss_dim_.size()+1, 0); // 1st zero stays,
309  for (int32 i = 1; i <= loss_dim_.size(); i++) {
310  loss_dim_offset_[i] = loss_dim_offset_[i-1] + loss_dim_[i-1];
311  }
312 
313  // sanity check,
314  KALDI_ASSERT(loss_vec_.size() > 0);
315  KALDI_ASSERT(loss_vec_.size() == loss_dim_.size());
316  KALDI_ASSERT(loss_vec_.size() == loss_weights_.size());
317 }

319 void MultiTaskLoss::Eval(const VectorBase<BaseFloat> &frame_weights,
320  const CuMatrixBase<BaseFloat>& net_out,
321  const Posterior& post,
322  CuMatrix<BaseFloat>* diff) {
323  int32 num_frames = net_out.NumRows(),
324  num_output = net_out.NumCols();
325  KALDI_ASSERT(num_frames == post.size());
326  KALDI_ASSERT(num_output == loss_dim_offset_.back()); // sum of loss-dims,
327 
328  // convert posterior to matrix,
329  PosteriorToMatrix(post, num_output, &tgt_mat_);
330 
331  // allocate diff matrix,
332  diff->Resize(num_frames, num_output);
333 
334  // call the vector of loss functions,
335  CuMatrix<BaseFloat> diff_aux;
336  for (int32 i = 0; i < loss_vec_.size(); i++) {
337  loss_vec_[i]->Eval(frame_weights,
338  net_out.ColRange(loss_dim_offset_[i], loss_dim_[i]),
339  tgt_mat_.ColRange(loss_dim_offset_[i], loss_dim_[i]),
340  &diff_aux);
341  // Scale the gradients,
342  diff_aux.Scale(loss_weights_[i]);
343  // Copy to diff,
344  diff->ColRange(loss_dim_offset_[i], loss_dim_[i]).CopyFromMat(diff_aux);
345  }
346 }

374 BaseFloat MultiTaskLoss::AvgLoss() {
375  BaseFloat ans(0.0);
376  for (int32 i = 0; i < loss_vec_.size(); i++) {
377  BaseFloat val = loss_weights_[i] * loss_vec_[i]->AvgLoss();

381  }
382  ans += val;
383  }
