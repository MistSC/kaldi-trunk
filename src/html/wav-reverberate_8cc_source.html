27 /*
28  This function is to repeatedly concatenate signal1 by itself
29  to match the length of signal2 and add the two signals together.
30 */
31 void AddVectorsOfUnequalLength(const Vector<BaseFloat> &signal1, Vector<BaseFloat> *signal2) {
32  for (int32 po = 0; po < signal2->Dim(); po += signal1.Dim()) {
33  int32 block_length = signal1.Dim();
34  if (signal2->Dim() - po < block_length) block_length = signal2->Dim() - po;
35  signal2->Range(po, block_length).AddVec(1.0, signal1.Range(0, block_length));
36  }
37 }

43 /*
44  Early reverberation component of the signal is composed of reflections
45  within 0.05 seconds of the direct path signal (assumed to be the peak of
46  the room impulse response). This function returns the energy in
47  this early reverberation component of the signal.
48  The input parameters to this function are the room impulse response, the signal
49  and their sampling frequency respectively.
50 */
51 BaseFloat ComputeEarlyReverbEnergy(const Vector<BaseFloat> &rir, const Vector<BaseFloat> &signal,
52  BaseFloat samp_freq) {
53  int32 peak_index = 0;
54  rir.Max(&peak_index);
55  KALDI_VLOG(1) << "peak index is " << peak_index;
56 
57  const float sec_before_peak = 0.001;
58  const float sec_after_peak = 0.05;
59  int32 early_rir_start_index = peak_index - sec_before_peak * samp_freq;
60  int32 early_rir_end_index = peak_index + sec_after_peak * samp_freq;
61  if (early_rir_start_index < 0) early_rir_start_index = 0;
62  if (early_rir_end_index > rir.Dim()) early_rir_end_index = rir.Dim();
63 
64  int32 duration = early_rir_end_index - early_rir_start_index;
65  Vector<BaseFloat> early_rir(rir.Range(early_rir_start_index, duration));
66  Vector<BaseFloat> early_reverb(signal);
67  FFTbasedBlockConvolveSignals(early_rir, &early_reverb);
68 
69  // compute the energy
70  return VecVec(early_reverb, early_reverb) / early_reverb.Dim();
71 }

73 /*
74  This is the core function to do reverberation and noise addition
75  on the given signal. The noise will be scaled before the addition
76  to match the given signal-to-noise ratio (SNR) and it will also concatenate
77  itself repeatedly to match the length of the signal.
78  The input parameters to this function are the room impulse response,
79  the sampling frequency, the SNR(dB), the noise and the signal respectively.
80 */
81 void DoReverberation(const Vector<BaseFloat> &rir, BaseFloat samp_freq,
82  BaseFloat snr_db, Vector<BaseFloat> *noise,
83  Vector<BaseFloat> *signal) {
84  if (noise->Dim()) {
85  float input_power = ComputeEarlyReverbEnergy(rir, *signal, samp_freq);
86  float noise_power = VecVec(*noise, *noise) / noise->Dim();
87  float scale_factor = sqrt(pow(10, -snr_db / 10) * input_power / noise_power);
88  noise->Scale(scale_factor);
89  KALDI_VLOG(1) << "Noise signal is being scaled with " << scale_factor
90  << " to generate output with SNR " << snr_db << "db\n";
91  }
92 
93  FFTbasedBlockConvolveSignals(rir, signal);
94 
95  if (noise->Dim() > 0) {
96  AddVectorsOfUnequalLength(*noise, signal);
97  }
98 }

106  "Corrupts the wave files supplied via input pipe with the specified\n"

124 
125  po.Register("multi-channel-output", &multi_channel_output,
126  "Specifies if the output should be multi-channel or not");
127  po.Register("input-wave-channel", &input_channel,
128  "Specifies the channel to be used from input as only a "
129  "single channel will be used to generate reverberated output");
130  po.Register("rir-channel", &rir_channel,
131  "Specifies the channel of the room impulse response, "
132  "it will only be used when multi-channel-output is false");
133  po.Register("noise-channel", &noise_channel,
134  "Specifies the channel of the noise file, "
135  "it will only be used when multi-channel-output is false");
136  po.Register("noise-file", &noise_file,
137  "File with additive noise");
138  po.Register("snr-db", &snr_db,
139  "Desired SNR(dB) of the output");
140  po.Register("normalize-output", &normalize_output,
141  "If true, then after reverberating and "
142  "possibly adding noise, scale so that the signal "
143  "energy is the same as the original input signal.");
144  po.Register("volume", &volume,
145  "If nonzero, a scaling factor on the signal that is applied "
146  "after reverberating and possibly adding noise. "
147  "If you set this option to a nonzero value, it will be as"
148  "if you had also specified --normalize-output=false.");
149 
150  po.Read(argc, argv);
151  if (po.NumArgs() != 3) {
152  po.PrintUsage();
153  exit(1);
154  }
155 
156  if (multi_channel_output) {
157  if (rir_channel != 0 || noise_channel != 0)
158  KALDI_WARN << "options for --rir-channel and --noise-channel"
159  "are ignored as --multi-channel-output is true.";
160  }
161 
162  std::string input_wave_file = po.GetArg(1);
163  std::string rir_file = po.GetArg(2);
164  std::string output_wave_file = po.GetArg(3);
165 
166  WaveData input_wave;
167  {
168  Input ki(input_wave_file);
169  input_wave.Read(ki.Stream());
170  }
171

173  BaseFloat samp_freq_input = input_wave.SampFreq();
174  int32 num_samp_input = input_matrix.NumCols(), // #samples in the input

176  KALDI_VLOG(1) << "sampling frequency of input: " << samp_freq_input
177  << " #samples: " << num_samp_input
178  << " #channel: " << num_input_channel;
179  KALDI_ASSERT(input_channel < num_input_channel);
180 
181  WaveData rir_wave;
182  {
183  Input ki(rir_file);
184  rir_wave.Read(ki.Stream());
185  }
186  const Matrix<BaseFloat> &rir_matrix = rir_wave.Data();
187  BaseFloat samp_freq_rir = rir_wave.SampFreq();
188  int32 num_samp_rir = rir_matrix.NumCols(),
189  num_rir_channel = rir_matrix.NumRows();
190  KALDI_VLOG(1) << "sampling frequency of rir: " << samp_freq_rir
191  << " #samples: " << num_samp_rir
192  << " #channel: " << num_rir_channel;
193  if (!multi_channel_output) {
194  KALDI_ASSERT(rir_channel < num_rir_channel);
195  }
196 
197  Matrix<BaseFloat> noise_matrix;
198  if (!noise_file.empty()) {
199  WaveData noise_wave;
200  {
201  Input ki(noise_file);
202  noise_wave.Read(ki.Stream());
203  }
204  noise_matrix = noise_wave.Data();
205  BaseFloat samp_freq_noise = noise_wave.SampFreq();
206  int32 num_samp_noise = noise_matrix.NumCols(),
207  num_noise_channel = noise_matrix.NumRows();
208  KALDI_VLOG(1) << "sampling frequency of noise: " << samp_freq_noise
209  << " #samples: " << num_samp_noise
210  << " #channel: " << num_noise_channel;
211  if (multi_channel_output) {
212  KALDI_ASSERT(num_rir_channel == num_noise_channel);
213  } else {
214  KALDI_ASSERT(noise_channel < num_noise_channel);
215  }
216  }
217 
218  int32 num_output_channels = (multi_channel_output ? num_rir_channel : 1);
219  Matrix<BaseFloat> out_matrix(num_output_channels, num_samp_input);
220 
221  for (int32 output_channel = 0; output_channel < num_output_channels; output_channel++) {
222  Vector<BaseFloat> input(num_samp_input);
223  input.CopyRowFromMat(input_matrix, input_channel);
224  float power_before_reverb = VecVec(input, input) / input.Dim();
225 
226  int32 this_rir_channel = (multi_channel_output ? output_channel : rir_channel);
227  Vector<BaseFloat> rir(num_samp_rir);
228  rir.CopyRowFromMat(rir_matrix, this_rir_channel);
229  rir.Scale(1.0 / (1 << 15));
230 
231  Vector<BaseFloat> noise(0);
232  if (!noise_file.empty()) {
233  noise.Resize(noise_matrix.NumCols());
234  int32 this_noise_channel = (multi_channel_output ? output_channel : noise_channel);
235  noise.CopyRowFromMat(noise_matrix, this_noise_channel);
236  }
237 
238  DoReverberation(rir, samp_freq_rir, snr_db, &noise, &input);
239 
240  float power_after_reverb = VecVec(input, input) / input.Dim();
241 
242  if (volume > 0)
243  input.Scale(volume);
244  else if (normalize_output)
245  input.Scale(sqrt(power_before_reverb / power_after_reverb));
246 
247  out_matrix.CopyRowFromVec(input, output_channel);
248  }
249 
250  WaveData out_wave(samp_freq_input, out_matrix);
251  Output ko(output_wave_file, false);
252  out_wave.Write(ko.Stream());
253 
254  return 0;
255  } catch(const std::exception &e) {
256  std::cerr << e.what();
257  return -1;
258  }
259 }
