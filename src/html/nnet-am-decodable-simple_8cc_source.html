89 void NnetDecodableBase::EnsureFrameIsComputed(int32 subsampled_frame) {
90  KALDI_ASSERT(subsampled_frame >= 0 &&
91  subsampled_frame < num_subsampled_frames_);
92  int32 feature_dim = feats_.NumCols(),
93  ivector_dim = GetIvectorDim(),
94  nnet_input_dim = nnet_.InputDim("input"),
95  nnet_ivector_dim = std::max<int32>(0, nnet_.InputDim("ivector"));
96  if (feature_dim != nnet_input_dim)
97  KALDI_ERR << "Neural net expects 'input' features with dimension "
98  << nnet_input_dim << " but you provided "
99  << feature_dim;
100  if (ivector_dim != std::max<int32>(0, nnet_.InputDim("ivector")))
101  KALDI_ERR << "Neural net expects 'ivector' features with dimension "
102  << nnet_ivector_dim << " but you provided " << ivector_dim;
103 
104  int32 current_subsampled_frames_computed = current_log_post_.NumRows(),
105  current_subsampled_offset = current_log_post_subsampled_offset_;
106  KALDI_ASSERT(subsampled_frame < current_subsampled_offset ||
107  subsampled_frame >= current_subsampled_offset +
108  current_subsampled_frames_computed);
109 
110  // all subsampled frames pertain to the output of the network,
111  // they are output frames divided by opts_.frame_subsampling_factor.
112  int32 subsampling_factor = opts_.frame_subsampling_factor,
113  subsampled_frames_per_chunk = opts_.frames_per_chunk / subsampling_factor,
114  start_subsampled_frame = subsampled_frame,
115  num_subsampled_frames = std::min<int32>(num_subsampled_frames_ -
116  start_subsampled_frame,
117  subsampled_frames_per_chunk),
118  last_subsampled_frame = start_subsampled_frame + num_subsampled_frames - 1;
119  KALDI_ASSERT(num_subsampled_frames > 0);
120  // the output-frame numbers are the subsampled-frame numbers
121  int32 first_output_frame = start_subsampled_frame * subsampling_factor,
122  last_output_frame = last_subsampled_frame * subsampling_factor;
123 
124  KALDI_ASSERT(opts_.extra_left_context >= 0 && opts_.extra_right_context >= 0);
125  int32 left_context = nnet_left_context_ + opts_.extra_left_context,
126  right_context = nnet_right_context_ + opts_.extra_right_context;
127  int32 first_input_frame = first_output_frame - left_context,
128  last_input_frame = last_output_frame + right_context,
129  num_input_frames = last_input_frame + 1 - first_input_frame;
130  Vector<BaseFloat> ivector;
131  GetCurrentIvector(first_output_frame,
132  last_output_frame - first_output_frame,
133  &ivector);
134 
135  Matrix<BaseFloat> input_feats;
136  if (first_input_frame >= 0 &&
137  last_input_frame < feats_.NumRows()) {
138  SubMatrix<BaseFloat> input_feats(feats_.RowRange(first_input_frame,
139  num_input_frames));
140  DoNnetComputation(first_input_frame, input_feats, ivector,
141  first_output_frame, num_subsampled_frames);
142  } else {
143  Matrix<BaseFloat> feats_block(num_input_frames, feats_.NumCols());
144  int32 tot_input_feats = feats_.NumRows();
145  for (int32 i = 0; i < num_input_frames; i++) {
146  SubVector<BaseFloat> dest(feats_block, i);
147  int32 t = i + first_input_frame;
148  if (t < 0) t = 0;
149  if (t >= tot_input_feats) t = tot_input_feats - 1;
150  const SubVector<BaseFloat> src(feats_, t);
151  dest.CopyFromVec(src);
152  }
153  DoNnetComputation(first_input_frame, feats_block, ivector,
154  first_output_frame, num_subsampled_frames);
155  }
156 }

170 void NnetDecodableBase::GetCurrentIvector(int32 output_t_start,
171  int32 num_output_frames,
172  Vector<BaseFloat> *ivector) {
173  if (ivector_ != NULL) {
174  *ivector = *ivector_;
175  return;
176  } else if (online_ivector_feats_ == NULL) {
177  return;
178  }
179  KALDI_ASSERT(online_ivector_period_ > 0);
180  // frame_to_search is the frame that we want to get the most recent iVector
181  // for. We choose a point near the middle of the current window, the concept
182  // being that this is the fairest comparison to nnet2. Obviously we could do
183  // better by always taking the last frame's iVector, but decoding with
184  // 'online' ivectors is only really a mechanism to simulate online operation.
185  int32 frame_to_search = output_t_start + num_output_frames / 2;
186  int32 ivector_frame = frame_to_search / online_ivector_period_;
187  KALDI_ASSERT(ivector_frame >= 0);
188  if (ivector_frame >= online_ivector_feats_->NumRows()) {
189  int32 margin = ivector_frame - (online_ivector_feats_->NumRows() - 1);
190  if (margin * online_ivector_period_ > 50) {
191  // Half a second seems like too long to be explainable as edge effects.
192  KALDI_ERR << "Could not get iVector for frame " << frame_to_search
193  << ", only available till frame "
194  << online_ivector_feats_->NumRows()
195  << " * ivector-period=" << online_ivector_period_
196  << " (mismatched --ivector-period?)";
197  }
198  ivector_frame = online_ivector_feats_->NumRows() - 1;
199  }
200  *ivector = online_ivector_feats_->Row(ivector_frame);
201 }

204 void NnetDecodableBase::DoNnetComputation(
205  int32 input_t_start,
206  const MatrixBase<BaseFloat> &input_feats,
207  const VectorBase<BaseFloat> &ivector,
208  int32 output_t_start,
209  int32 num_subsampled_frames) {
210  ComputationRequest request;
211  request.need_model_derivative = false;
212  request.store_component_stats = false;
213 
214  bool shift_time = true; // shift the 'input' and 'output' to a consistent
215  // time, to take advantage of caching in the compiler.
216  // An optimization.
217  int32 time_offset = (shift_time ? -output_t_start : 0);
218 
219  // First add the regular features-- named "input".
220  request.inputs.reserve(2);
221  request.inputs.push_back(
222  IoSpecification("input", time_offset + input_t_start,
223  time_offset + input_t_start + input_feats.NumRows()));
224  if (ivector.Dim() != 0) {
225  std::vector<Index> indexes;
226  indexes.push_back(Index(0, 0, 0));
227  request.inputs.push_back(IoSpecification("ivector", indexes));
228  }
229  IoSpecification output_spec;
230  output_spec.name = "output";
231  output_spec.has_deriv = false;
232  int32 subsample = opts_.frame_subsampling_factor;
233  output_spec.indexes.resize(num_subsampled_frames);
234  // leave n and x values at 0 (the constructor sets these).
235  for (int32 i = 0; i < num_subsampled_frames; i++)
236  output_spec.indexes[i].t = time_offset + output_t_start + i * subsample;
237  request.outputs.resize(1);
238  request.outputs[0].Swap(&output_spec);
239 
240  const NnetComputation *computation = compiler_.Compile(request);
241  Nnet *nnet_to_update = NULL; // we're not doing any update.
242  NnetComputer computer(opts_.compute_config, *computation,
243  nnet_, nnet_to_update);
244 
245  CuMatrix<BaseFloat> input_feats_cu(input_feats);
246  computer.AcceptInput("input", &input_feats_cu);
247  CuMatrix<BaseFloat> ivector_feats_cu;
248  if (ivector.Dim() > 0) {
249  ivector_feats_cu.Resize(1, ivector.Dim());
250  ivector_feats_cu.Row(0).CopyFromVec(ivector);
251  computer.AcceptInput("ivector", &ivector_feats_cu);
252  }
253  computer.Forward();
254  CuMatrix<BaseFloat> cu_output;
255  computer.GetOutputDestructive("output", &cu_output);
256  // subtract log-prior (divide by prior)
257  if (log_priors_.Dim() != 0)
258  cu_output.AddVecToRows(-1.0, log_priors_);
259  // apply the acoustic scale
260  cu_output.Scale(opts_.acoustic_scale);
261  current_log_post_.Resize(0, 0);
262  // the following statement just swaps the pointers if we're not using a GPU.
263  cu_output.Swap(&current_log_post_);
264  current_log_post_subsampled_offset_ = output_t_start / subsample;
265 }

267 void NnetDecodableBase::CheckAndFixConfigs() {
268  static bool warned_modulus = false,
269  warned_subsampling = false;
270  int32 nnet_modulus = nnet_.Modulus();
271  if (opts_.frame_subsampling_factor < 1 ||
272  opts_.frames_per_chunk < 1)
273  KALDI_ERR << "--frame-subsampling-factor and --frames-per-chunk must be > 0";
274  if (opts_.frames_per_chunk % opts_.frame_subsampling_factor != 0) {
275  int32 f = opts_.frame_subsampling_factor,
276  frames_per_chunk = f * ((opts_.frames_per_chunk + f - 1) / f);
277  if (!warned_subsampling) {
278  warned_subsampling = true;
279  KALDI_LOG << "Increasing --frames-per-chunk from "
280  << opts_.frames_per_chunk << " to "
281  << frames_per_chunk << " to make it a multiple of "
282  << "--frame-subsampling-factor="
283  << opts_.frame_subsampling_factor;
284  }
285  opts_.frames_per_chunk = frames_per_chunk;
286  }
287  if (opts_.frames_per_chunk % nnet_modulus != 0 && !warned_modulus) {
288  warned_modulus = true;
289  KALDI_WARN << "It may be more efficient to set the --frames-per-chunk "
290  << "(currently " << opts_.frames_per_chunk << " to a "
291  << "multiple of the network's shift-invariance modulus "
292  << nnet_modulus;
293  }
294 }
