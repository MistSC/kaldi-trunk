26 void SgmmClusterable::Accumulate(
27  const SgmmPerFrameDerivedVars &per_frame_vars,
28  int32 j, // state index in original SGMM.
29  BaseFloat weight) {
30  Matrix<BaseFloat> post;
31  KALDI_ASSERT(weight >= 0.0); // Doesn't make sense to use negative weights here.
32  // Compute Gaussian-level posteriors.
33  // Note: "post" is indexed by Gaussian-selection index.
34  sgmm_.ComponentPosteriors(per_frame_vars, j, &post);
35  if (weight != 1.0) post.Scale(weight);
36  const std::vector<int32> &gselect = per_frame_vars.gselect;
37  for (int32 ki = 0; ki < gselect.size(); ki++) {
38  int32 i = gselect[ki];
39  BaseFloat gamma = 0.0; // Sum the weight over all the vectors (index m) in
40  // the state. In sensible cases there should be just one vector per state
41  // at the point where we do this, though.
42  for (int32 m = 0; m < post.NumCols(); m++) gamma += post(ki, m);
43  gamma_(i) += gamma;
44  y_.AddVec(gamma, per_frame_vars.zti.Row(ki));
45  }
46  // Invalidate my_H_, if present, since it's not efficient to
47  // keep it updated during accumulation.
48  if (my_H_.NumRows() != 0)
49  my_H_.Resize(0);
50 }

52 BaseFloat SgmmClusterable::Objf() const {
53  // Objective function consists of the expected log-likelihood of
54  // a weight (assuming we estimate the weights directly as parameters
55  // instead of the whole subspace thing on the weights), plus
56  // the auxiliary function improvement we would get from estimating
57  // the state vector v_j starting from zero. Note: zero is an
58  // arbitrary starting point-- we could use any value as long as
59  // we were consistent.
60  KALDI_ASSERT(static_cast<int32>(H_.size()) == sgmm_.NumGauss());
61  if (my_H_.NumRows() == 0.0) {
62  SgmmClusterable *s = static_cast<SgmmClusterable*>(this->Copy()); // will
63  // set up my_H_, which we need.
64  BaseFloat ans = s->Objf();
65  delete s;
66  return ans;
67  }
68  double ans = 0.0;
69  double tot_gamma = gamma_.Sum(), tot_gamma2 = 0.0;
70  if (tot_gamma == 0.0) return 0.0;
71  int32 I = gamma_.Dim();
72 
73  for (int32 i = 0; i < I; i++) {
74  double gamma = gamma_(i);
75  if (gamma > 0.0) { // Note: should not be negative-- if it is, due to
76  double prob = gamma / tot_gamma;
77  if (prob > 0.0) { // Note: prob could be zero due to underflow-- this
78  // happened! [we can get tiny values due to floating-point roundoff
79  // while subtracting clusterable objects].
80  ans += gamma * Log(gamma / tot_gamma);
81  }
82  }
83  tot_gamma2 += gamma;
84  }
85  if (tot_gamma2 == 0.0)
86  return 0.0; // No positive elements... maybe small negative ones were from
87  // round off.
88 
89  // objf improvement is y^T H^{-1} y.
90  // We'll try to compute this using Cholesky, first, which is more
91  // efficient; if this fails or appears to lead to big values,
92  // we'll back off to a more efficient SVD-based implementation.
93  try {
94  TpMatrix<double> C(my_H_.NumRows());
95  C.Cholesky(my_H_);
96  C.Invert();
97  for (int32 i = 0; i < C.NumRows(); i++)
98  if (fabs(C(i, i)) > 100.0) {
99  KALDI_VLOG(3) << "Condion-number probably bad: element is "
100  << C(i, i);
101  throw std::runtime_error("Bad condition number"); // back off to SVD.
102  }
103  // Note: assuming things are well preconditioned, the elements
104  // C(i,i) should be of the rough magnitude 1/sqrt(count).
105  Vector<double> yC(C.NumRows());
106  // Note: if we decompose H = C C^T, then the line below
107  // does yC = C^{-1} y. Note: we are computing the inner
108  // product y^T H^{-1} y. H^{-1} = C^{-T} C^{-1}, so
109  // y^T H^{-1} y = y^T C^{-T} C^{-1} y = yC^T yC.
110  yC.AddTpVec(1.0, C, kNoTrans, y_, 0.0);
111  ans += 0.5 * VecVec(yC, yC);
112  } catch (...) { // Choleksy threw, or we detected bad condition.
113  // we'll do this using an SVD-based implementation that will
114  // deal with non-invertible matrices.
115  KALDI_VLOG(3) << "Backing off to SVD-based objective computation.";
116  Vector<double> v(y_.Dim()); // Initialized automatically to zero.
117  ans += SolveQuadraticProblem(my_H_, y_, SolverOptions(), &v); // The objective function
118  // change from estimating this vector.
119  }
120  return ans;
121 }

129 void SgmmClusterable::Add(const Clusterable &other_in) {
130  const SgmmClusterable *other =
131  static_cast<const SgmmClusterable*>(&other_in);
132  gamma_.AddVec(1.0, other->gamma_);
133  y_.AddVec(1.0, other->y_);
134  if (!H_.empty()) { // we need to compute my_H_.
135  if (my_H_.NumRows() != 0 && other->my_H_.NumRows() != 0)
136  my_H_.AddSp(1.0, other->my_H_);
137  else {
138  my_H_.Resize(0);
139  ComputeH();
140  }
141  }
142 }

164  SgmmClusterable *ans = new SgmmClusterable(sgmm_, H_);
165  ans->gamma_.CopyFromVec(gamma_);
166  ans->y_.CopyFromVec(y_);
167  if (!H_.empty()) {
168  if (my_H_.NumRows() == 0.0) ans->ComputeH();

170  ans->my_H_.Resize(my_H_.NumRows());
171  ans->my_H_.CopyFromSp(my_H_);

198 bool AccumulateSgmmTreeStats(const TransitionModel &trans_model,
199  const AmSgmm &am_sgmm,
200  const std::vector<SpMatrix<double> > &H,
201  int N, // context window size.
202  int P, // central position.
203  const std::vector<int32> &ci_phones, // must be sorted
204  const std::vector<int32> &alignment,
205  const std::vector<std::vector<int32> > &gselect,
206  const SgmmPerSpkDerivedVars &per_spk_vars,
207  const Matrix<BaseFloat> &features,
208  std::map<EventType, SgmmClusterable*> *stats) {
209  KALDI_ASSERT(IsSortedAndUniq(ci_phones));
210  std::vector<std::vector<int32> > split_alignment;
211  bool ans = SplitToPhones(trans_model, alignment, &split_alignment);
212  if (!ans) {
213  KALDI_WARN << "AccumulateTreeStats: bad alignment.";
214  return false;
215  }
216  int t = 0;
217  SgmmPerFrameDerivedVars per_frame_vars;
218 
219  KALDI_ASSERT(features.NumRows() == static_cast<int32>(alignment.size())
220  && alignment.size() == gselect.size());
221  for (int i = -N; i < static_cast<int>(split_alignment.size()); i++) {
222  // consider window starting at i, only if i+P is within
223  // list of phones.
224  if (i + P >= 0 && i + P < static_cast<int>(split_alignment.size())) {
225  int32 central_phone = trans_model.TransitionIdToPhone(split_alignment[i+P][0]);
226  bool is_ctx_dep = ! std::binary_search(ci_phones.begin(),
227  ci_phones.end(),
228  central_phone);
229  EventType evec;
230  for (int j = 0; j < N; j++) {
231  int phone;
232  if (i + j >= 0 && i + j < static_cast<int>(split_alignment.size()))
233  phone = trans_model.TransitionIdToPhone(split_alignment[i+j][0]);
234  else
235  phone = 0; // ContextDependency class uses 0 to mean "out of window".
236 
237  if (is_ctx_dep || j == P)
238  evec.push_back(std::make_pair(static_cast<EventKeyType>(j), static_cast<EventValueType>(phone)));
239  }
240  for (int j = 0; j < static_cast<int>(split_alignment[i+P].size());j++) {
241  // for central phone of this window...
242  EventType evec_more(evec);
243  int32 pdf_id = trans_model.TransitionIdToPdf(split_alignment[i+P][j]),
244  pdf_class = trans_model.TransitionIdToPdfClass(split_alignment[i+P][j]);
245  // pdf_id represents the acoustic state in the current model.
246  // pdf_class will normally by 0, 1 or 2 for a 3-state HMM.
247 
248  std::pair<EventKeyType, EventValueType> pr(kPdfClass, pdf_class);
249  evec_more.push_back(pr);
250  std::sort(evec_more.begin(), evec_more.end()); // these must be sorted!
251  if (stats->count(evec_more) == 0)
252  (*stats)[evec_more] = new SgmmClusterable(am_sgmm, H);
253 
254  am_sgmm.ComputePerFrameVars(features.Row(t), gselect[t], per_spk_vars, 0.0,
255  &per_frame_vars);
256  BaseFloat weight = 1.0; // weight is one, since we have alignment.
257  (*stats)[evec_more]->Accumulate(per_frame_vars, pdf_id, weight);
258  t++;
259  }
260  }
261  }
262  KALDI_ASSERT(t == static_cast<int>(alignment.size()));
263  return true;
264 }

266 void SgmmClusterable::ComputeH() {
267  // We're computing my_H_, as a weighted sum of H_, with gamma_ as the
268  // weights.
269  KALDI_ASSERT(!H_.empty() && my_H_.NumRows() == 0); // Invalid to call this if H_ empty,
270  // or my_H_ already set up.
271  my_H_.Resize(H_[0].NumRows()); // will initialize to zero.
272  KALDI_ASSERT(static_cast<int32>(H_.size()) == gamma_.Dim());
273  for (int32 i = 0; i < gamma_.Dim(); i++) {
274  double gamma = gamma_(i);
275  if (gamma > 0.0) my_H_.AddSp(gamma, H_[i]);
276  }
277 }
