61 void DecodeUtteranceLatticeFasterClass::operator () () {
62  // Decoding and lattice determinization happens here.
63  computed_ = true; // Just means this function was called-- a check on the
64  // calling code.
65  success_ = true;
66  using fst::VectorFst;
67  if (!decoder_->Decode(decodable_)) {
68  KALDI_WARN << "Failed to decode file " << utt_;
69  success_ = false;
70  }
71  if (!decoder_->ReachedFinal()) {
72  if (allow_partial_) {
73  KALDI_WARN << "Outputting partial output for utterance " << utt_
74  << " since no final-state reached\n";
75  partial_ = true;
76  } else {
77  KALDI_WARN << "Not producing output for utterance " << utt_
78  << " since no final-state reached and "
79  << "--allow-partial=false.\n";
80  success_ = false;
81  }
82  }
83  if (!success_) return;
84 
85  // Get lattice, and do determinization if requested.
86  lat_ = new Lattice;
87  decoder_->GetRawLattice(lat_);
88  if (lat_->NumStates() == 0)
89  KALDI_ERR << "Unexpected problem getting lattice for utterance " << utt_;
90  fst::Connect(lat_);
91  if (determinize_) {
92  clat_ = new CompactLattice;
93  if (!DeterminizeLatticePhonePrunedWrapper(
94  *trans_model_,
95  lat_,
96  decoder_->GetOptions().lattice_beam,
97  clat_,
98  decoder_->GetOptions().det_opts))
99  KALDI_WARN << "Determinization finished earlier than the beam for "
100  << "utterance " << utt_;
101  delete lat_;
102  lat_ = NULL;
103  // We'll write the lattice without acoustic scaling.
104  if (acoustic_scale_ != 0.0)
105  fst::ScaleLattice(fst::AcousticLatticeScale(1.0 / acoustic_scale_), clat_);
106  } else {
107  // We'll write the lattice without acoustic scaling.
108  if (acoustic_scale_ != 0.0)
109  fst::ScaleLattice(fst::AcousticLatticeScale(1.0 / acoustic_scale_), lat_);
110  }
111 }

113 DecodeUtteranceLatticeFasterClass::~DecodeUtteranceLatticeFasterClass() {
114  if (!computed_)
115  KALDI_ERR << "Destructor called without operator (), error in calling code.";
116 
117  if (!success_) {
118  if (num_err_ != NULL) (*num_err_)++;
119  } else { // successful decode.
120  // Getting the one-best output is lightweight enough that we can do it in
121  // the destructor (easier than adding more variables to the class, and
122  // will rarely slow down the main thread.)
123  double likelihood;
124  LatticeWeight weight;
125  int32 num_frames;
126  { // First do some stuff with word-level traceback...
127  // This is basically for diagnostics.
128  fst::VectorFst<LatticeArc> decoded;
129  decoder_->GetBestPath(&decoded);
130  if (decoded.NumStates() == 0) {
131  // Shouldn't really reach this point as already checked success.
132  KALDI_ERR << "Failed to get traceback for utterance " << utt_;
133  }
134  std::vector<int32> alignment;
135  std::vector<int32> words;
136  GetLinearSymbolSequence(decoded, &alignment, &words, &weight);
137  num_frames = alignment.size();
138  if (words_writer_->IsOpen())
139  words_writer_->Write(utt_, words);
140  if (alignments_writer_->IsOpen())
141  alignments_writer_->Write(utt_, alignment);
142  if (word_syms_ != NULL) {
143  std::cerr << utt_ << ' ';
144  for (size_t i = 0; i < words.size(); i++) {
145  std::string s = word_syms_->Find(words[i]);
146  if (s == "")
147  KALDI_ERR << "Word-id " << words[i] << " not in symbol table.";
148  std::cerr << s << ' ';
149  }
150  std::cerr << '\n';
151  }
152  likelihood = -(weight.Value1() + weight.Value2());
153  }
154 
155  // Ouptut the lattices.
156  if (determinize_) { // CompactLattice output.
157  KALDI_ASSERT(compact_lattice_writer_ != NULL && clat_ != NULL);
158  if (clat_->NumStates() == 0) {
159  KALDI_WARN << "Empty lattice for utterance " << utt_;
160  } else {
161  compact_lattice_writer_->Write(utt_, *clat_);
162  }
163  delete clat_;
164  clat_ = NULL;
165  } else {
166  KALDI_ASSERT(lattice_writer_ != NULL && lat_ != NULL);
167  if (lat_->NumStates() == 0) {
168  KALDI_WARN << "Empty lattice for utterance " << utt_;
169  } else {
170  lattice_writer_->Write(utt_, *lat_);
171  }
172  delete lat_;
173  lat_ = NULL;
174  }
175 
176  // Print out logging information.
177  KALDI_LOG << "Log-like per frame for utterance " << utt_ << " is "
178  << (likelihood / num_frames) << " over "
179  << num_frames << " frames.";
180  KALDI_VLOG(2) << "Cost for utterance " << utt_ << " is "
181  << weight.Value1() << " + " << weight.Value2();
182 
183  // Now output the various diagnostic variables.
184  if (like_sum_ != NULL) *like_sum_ += likelihood;
185  if (frame_sum_ != NULL) *frame_sum_ += num_frames;
186  if (num_done_ != NULL) (*num_done_)++;
187  if (partial_ && num_partial_ != NULL) (*num_partial_)++;
188  }
189  // We were given ownership of these two objects that were passed in in
190  // the initializer.
191  delete decoder_;
192  delete decodable_;
193 }

393 void ModifyGraphForCarefulAlignment(
394  fst::VectorFst<fst::StdArc> *fst) {
395  typedef fst::StdArc Arc;
396  typedef Arc::StateId StateId;
397  typedef Arc::Label Label;
398  typedef Arc::Weight Weight;
399  StateId num_states = fst->NumStates();
400  if (num_states == 0) {
401  KALDI_WARN << "Empty FST input.";
402  return;
403  }
404  Weight zero = Weight::Zero();
405  // fst_rhs will be the right hand side of the Concat operation.
406  fst::VectorFst<fst::StdArc> fst_rhs(*fst);
407  // first remove the final-probs from fst_rhs.
408  for (StateId state = 0; state < num_states; state++)
409  fst_rhs.SetFinal(state, zero);
410  StateId pre_initial = fst_rhs.AddState();
411  Arc to_initial(0, 0, Weight::One(), fst_rhs.Start());
412  fst_rhs.AddArc(pre_initial, to_initial);
413  fst_rhs.SetStart(pre_initial);
414  // make the pre_initial state final with probability one;
415  // this is equivalent to keeping the final-probs of the first
416  // FST when we do concat (otherwise they would get deleted).
417  fst_rhs.SetFinal(pre_initial, Weight::One());
418  fst::VectorFst<fst::StdArc> fst_concat;
419  fst::Concat(fst, fst_rhs);
420 }

423 void AlignUtteranceWrapper(
424  const AlignConfig &config,
425  const std::string &utt,
426  BaseFloat acoustic_scale, // affects scores written to scores_writer, if
427  // present
428  fst::VectorFst<fst::StdArc> *fst, // non-const in case config.careful ==
429  // true.
430  DecodableInterface *decodable, // not const but is really an input.
431  Int32VectorWriter *alignment_writer,
432  BaseFloatWriter *scores_writer,
433  int32 *num_done,
434  int32 *num_error,
435  int32 *num_retried,
436  double *tot_like,
437  int64 *frame_count) {
438 
439  if ((config.retry_beam != 0 && config.retry_beam <= config.beam) ||
440  config.beam <= 0.0) {
441  KALDI_ERR << "Beams do not make sense: beam " << config.beam
442  << ", retry-beam " << config.retry_beam;
443  }
444 
445  if (fst->Start() == fst::kNoStateId) {
446  KALDI_WARN << "Empty decoding graph for " << utt;
447  if (num_error != NULL) (*num_error)++;
448  return;
449  }
450 
451 
452  fst::StdArc::Label special_symbol = 0;
453  if (config.careful)
454  ModifyGraphForCarefulAlignment(fst);
455 
456  FasterDecoderOptions decode_opts;
457  decode_opts.beam = config.beam;
458 
459  FasterDecoder decoder(*fst, decode_opts);
460  decoder.Decode(decodable);
461 
462  bool ans = decoder.ReachedFinal(); // consider only final states.
463 
464  if (!ans && config.retry_beam != 0.0) {
465  if (num_retried != NULL) (*num_retried)++;
466  KALDI_WARN << "Retrying utterance " << utt << " with beam "
467  << config.retry_beam;
468  decode_opts.beam = config.retry_beam;
469  decoder.SetOptions(decode_opts);
470  decoder.Decode(decodable);
471  ans = decoder.ReachedFinal();
472  }
473 
474  if (!ans) { // Still did not reach final state.
475  KALDI_WARN << "Did not successfully decode file " << utt << ", len = "
476  << decodable->NumFramesReady();
477  if (num_error != NULL) (*num_error)++;
478  return;
479  }
480 
481  fst::VectorFst<LatticeArc> decoded; // linear FST.
482  decoder.GetBestPath(&decoded);
483  if (decoded.NumStates() == 0) {
484  KALDI_WARN << "Error getting best path from decoder (likely a bug)";
485  if (num_error != NULL) (*num_error)++;
486  return;
487  }
488 
489  std::vector<int32> alignment;
490  std::vector<int32> words;
491  LatticeWeight weight;
492 
493  GetLinearSymbolSequence(decoded, &alignment, &words, &weight);
494  BaseFloat like = -(weight.Value1()+weight.Value2()) / acoustic_scale;
495 
496  if (num_done != NULL) (*num_done)++;
497  if (tot_like != NULL) (*tot_like) += like;
498  if (frame_count != NULL) (*frame_count) += decodable->NumFramesReady();
499 
500  if (alignment_writer != NULL && alignment_writer->IsOpen())
501  alignment_writer->Write(utt, alignment);
502 
503  if (scores_writer != NULL && scores_writer->IsOpen())
504  scores_writer->Write(utt, -(weight.Value1()+weight.Value2()));
505 }
