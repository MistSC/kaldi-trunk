27 void TransitionModel::ComputeTriples(const ContextDependencyInterface &ctx_dep) {
28  const std::vector<int32> &phones = topo_.GetPhones();
29  std::vector<std::vector<std::pair<int32, int32> > > pdf_info;
30  KALDI_ASSERT(!phones.empty());
31  std::vector<int32> num_pdf_classes( 1 + *std::max_element(phones.begin(), phones.end()), -1);
32  for (size_t i = 0; i < phones.size(); i++)
33  num_pdf_classes[phones[i]] = topo_.NumPdfClasses(phones[i]);
34  ctx_dep.GetPdfInfo(phones, num_pdf_classes, &pdf_info);
35  // pdf_info is list indexed by pdf of which (phone, pdf_class) it
36  // can correspond to.
37 
38  std::map<std::pair<int32, int32>, std::vector<int32> > to_hmm_state_list;
39  // to_hmm_state_list is a map from (phone, pdf_class) to the list
40  // of hmm-states in the HMM for that phone that that (phone, pdf-class)
41  // can correspond to.
42  for (size_t i = 0; i < phones.size(); i++) { // setting up to_hmm_state_list.
43  int32 phone = phones[i];
44  const HmmTopology::TopologyEntry &entry = topo_.TopologyForPhone(phone);
45  for (int32 j = 0; j < static_cast<int32>(entry.size()); j++) { // for each state...
46  int32 pdf_class = entry[j].pdf_class;
47  if (pdf_class != kNoPdf) {
48  to_hmm_state_list[std::make_pair(phone, pdf_class)].push_back(j);
49  }
50  }
51  }
52  for (int32 pdf = 0; pdf < static_cast<int32>(pdf_info.size()); pdf++) {
53  for (size_t j = 0; j < pdf_info[pdf].size(); j++) {
54  int32 phone = pdf_info[pdf][j].first,
55  pdf_class = pdf_info[pdf][j].second;
56  const std::vector<int32> &state_vec = to_hmm_state_list[std::make_pair(phone, pdf_class)];
57  KALDI_ASSERT(!state_vec.empty());
58  // state_vec is a list of the possible HMM-states that emit this
59  // pdf_class.
60  for (size_t k = 0; k < state_vec.size(); k++) {
61  int32 hmm_state = state_vec[k];
62  triples_.push_back(Triple(phone, hmm_state, pdf));
63  }
64  }
65  }
66 
67  // now triples_ is populated with all possible triples of (phone, hmm_state, pdf).
68  std::sort(triples_.begin(), triples_.end()); // sort to enable reverse lookup.
69  // this sorting defines the transition-ids.
70 }

72 void TransitionModel::ComputeDerived() {
73  state2id_.resize(triples_.size()+2); // indexed by transition-state, which
74  // is one based, but also an entry for one past end of list.
75 
76  int32 cur_transition_id = 1;
77  num_pdfs_ = 0;
78  for (int32 tstate = 1;
79  tstate <= static_cast<int32>(triples_.size()+1); // not a typo.
80  tstate++) {
81  state2id_[tstate] = cur_transition_id;
82  if (static_cast<size_t>(tstate) <= triples_.size()) {
83  int32 phone = triples_[tstate-1].phone,
84  hmm_state = triples_[tstate-1].hmm_state,
85  pdf = triples_[tstate-1].pdf;
86  num_pdfs_ = std::max(num_pdfs_, 1+pdf);
87  const HmmTopology::HmmState &state = topo_.TopologyForPhone(phone)[hmm_state];
88  int32 my_num_ids = static_cast<int32>(state.transitions.size());
89  cur_transition_id += my_num_ids; // # trans out of this state.
90  }
91  }
92 
93  id2state_.resize(cur_transition_id); // cur_transition_id is #transition-ids+1.
94  for (int32 tstate = 1; tstate <= static_cast<int32>(triples_.size()); tstate++)
95  for (int32 tid = state2id_[tstate]; tid < state2id_[tstate+1]; tid++)
96  id2state_[tid] = tstate;
97 
98 }
99 void TransitionModel::InitializeProbs() {
100  log_probs_.Resize(NumTransitionIds()+1); // one-based array, zeroth element empty.
101  for (int32 trans_id = 1; trans_id <= NumTransitionIds(); trans_id++) {
102  int32 trans_state = id2state_[trans_id];
103  int32 trans_index = trans_id - state2id_[trans_state];
104  const Triple &triple = triples_[trans_state-1];
105  const HmmTopology::TopologyEntry &entry = topo_.TopologyForPhone(triple.phone);
106  KALDI_ASSERT(static_cast<size_t>(triple.hmm_state) < entry.size());
107  BaseFloat prob = entry[triple.hmm_state].transitions[trans_index].second;
108  if (prob <= 0.0)
109  KALDI_ERR << "TransitionModel::InitializeProbs, zero "
110  "probability [should remove that entry in the topology]";
111  if (prob > 1.0)
112  KALDI_WARN << "TransitionModel::InitializeProbs, prob greater than one.";
113  log_probs_(trans_id) = Log(prob);
114  }
115  ComputeDerivedOfProbs();
116 }

118 void TransitionModel::Check() const {
119  KALDI_ASSERT(NumTransitionIds() != 0 && NumTransitionStates() != 0);
120  {
121  int32 sum = 0;
122  for (int32 ts = 1; ts <= NumTransitionStates(); ts++) sum += NumTransitionIndices(ts);
123  KALDI_ASSERT(sum == NumTransitionIds());
124  }
125  for (int32 tid = 1; tid <= NumTransitionIds(); tid++) {
126  int32 tstate = TransitionIdToTransitionState(tid),
127  index = TransitionIdToTransitionIndex(tid);
128  KALDI_ASSERT(tstate > 0 && tstate <=NumTransitionStates() && index >= 0);
129  KALDI_ASSERT(tid == PairToTransitionId(tstate, index));
130  int32 phone = TransitionStateToPhone(tstate),
131  hmm_state = TransitionStateToHmmState(tstate),
132  pdf = TransitionStateToPdf(tstate);
133  KALDI_ASSERT(tstate == TripleToTransitionState(phone, hmm_state, pdf));
134  KALDI_ASSERT(log_probs_(tid) <= 0.0 && log_probs_(tid) - log_probs_(tid) == 0.0);
135  // checking finite and non-positive (and not out-of-bounds).
136  }
137 }

148 int32 TransitionModel::TripleToTransitionState(int32 phone, int32 hmm_state, int32 pdf) const {
149  Triple triple(phone, hmm_state, pdf);
150  // Note: if this ever gets too expensive, which is unlikely, we can refactor
151  // this code to sort first on pdf, and then index on pdf, so those
152  // that have the same pdf are in a contiguous range.
153  std::vector<Triple>::const_iterator iter =
154  std::lower_bound(triples_.begin(), triples_.end(), triple);
155  if (iter == triples_.end() || !(*iter == triple)) {
156  KALDI_ERR << "TransitionModel::TripleToTransitionState, triple not found."
157  << " (incompatible tree and model?)";
158  }
159  // triples_ is indexed by transition_state-1, so add one.
160  return static_cast<int32>((iter - triples_.begin())) + 1;
161 }

240 int32 TransitionModel::SelfLoopOf(int32 trans_state) const { // returns the self-loop transition-id,
241  KALDI_ASSERT(static_cast<size_t>(trans_state-1) < triples_.size());
242  const Triple &triple = triples_[trans_state-1];
243  // or zero if does not exist.
244  int32 phone = triple.phone, hmm_state = triple.hmm_state;
245  const HmmTopology::TopologyEntry &entry = topo_.TopologyForPhone(phone);
246  KALDI_ASSERT(static_cast<size_t>(hmm_state) < entry.size());
247  for (int32 trans_index = 0;
248  trans_index < static_cast<int32>(entry[hmm_state].transitions.size());
249  trans_index++)
250  if (entry[hmm_state].transitions[trans_index].first == hmm_state)
251  return PairToTransitionId(trans_state, trans_index);
252  return 0; // invalid transition id.

255 void TransitionModel::ComputeDerivedOfProbs() {
256  non_self_loop_log_probs_.Resize(NumTransitionStates()+1); // this array indexed
257  // by transition-state with nothing in zeroth element.
258  for (int32 tstate = 1; tstate <= NumTransitionStates(); tstate++) {
259  int32 tid = SelfLoopOf(tstate);
260  if (tid == 0) { // no self-loop
261  non_self_loop_log_probs_(tstate) = 0.0; // log(1.0)
262  } else {
263  BaseFloat self_loop_prob = Exp(GetTransitionLogProb(tid)),
264  non_self_loop_prob = 1.0 - self_loop_prob;
265  if (non_self_loop_prob <= 0.0) {
266  KALDI_WARN << "ComputeDerivedOfProbs(): non-self-loop prob is " << non_self_loop_prob;
267  non_self_loop_prob = 1.0e-10; // just so we can continue...
268  }
269  non_self_loop_log_probs_(tstate) = Log(non_self_loop_prob); // will be negative.
270  }
271  }
272 }

339 // stats are counts/weights, indexed by transition-id.
340 void TransitionModel::MleUpdate(const Vector<double> &stats,
341  const MleTransitionUpdateConfig &cfg,
342  BaseFloat *objf_impr_out,
343  BaseFloat *count_out) {
344  if (cfg.share_for_pdfs) {
345  MleUpdateShared(stats, cfg, objf_impr_out, count_out);
346  return;
347  }
348  BaseFloat count_sum = 0.0, objf_impr_sum = 0.0;
349  int32 num_skipped = 0, num_floored = 0;
350  KALDI_ASSERT(stats.Dim() == NumTransitionIds()+1);
351  for (int32 tstate = 1; tstate <= NumTransitionStates(); tstate++) {
352  int32 n = NumTransitionIndices(tstate);
353  KALDI_ASSERT(n>=1);
354  if (n > 1) { // no point updating if only one transition...
355  Vector<double> counts(n);
356  for (int32 tidx = 0; tidx < n; tidx++) {
357  int32 tid = PairToTransitionId(tstate, tidx);
358  counts(tidx) = stats(tid);
359  }
360  double tstate_tot = counts.Sum();
361  count_sum += tstate_tot;
362  if (tstate_tot < cfg.mincount) { num_skipped++; }
363  else {
364  Vector<BaseFloat> old_probs(n), new_probs(n);
365  for (int32 tidx = 0; tidx < n; tidx++) {
366  int32 tid = PairToTransitionId(tstate, tidx);
367  old_probs(tidx) = new_probs(tidx) = GetTransitionProb(tid);
368  }
369  for (int32 tidx = 0; tidx < n; tidx++)
370  new_probs(tidx) = counts(tidx) / tstate_tot;
371  for (int32 i = 0; i < 3; i++) { // keep flooring+renormalizing for 3 times..
372  new_probs.Scale(1.0 / new_probs.Sum());
373  for (int32 tidx = 0; tidx < n; tidx++)
374  new_probs(tidx) = std::max(new_probs(tidx), cfg.floor);
375  }
376  // Compute objf change
377  for (int32 tidx = 0; tidx < n; tidx++) {
378  if (new_probs(tidx) == cfg.floor) num_floored++;
379  double objf_change = counts(tidx) * (Log(new_probs(tidx))
380  - Log(old_probs(tidx)));
381  objf_impr_sum += objf_change;
382  }
383  // Commit updated values.
384  for (int32 tidx = 0; tidx < n; tidx++) {
385  int32 tid = PairToTransitionId(tstate, tidx);
386  log_probs_(tid) = Log(new_probs(tidx));
387  if (log_probs_(tid) - log_probs_(tid) != 0.0)
388  KALDI_ERR << "Log probs is inf or NaN: error in update or bad stats?";
389  }
390  }
391  }
392  }
393  KALDI_LOG << "TransitionModel::Update, objf change is "
394  << (objf_impr_sum / count_sum) << " per frame over " << count_sum
395  << " frames. ";
396  KALDI_LOG << num_floored << " probabilities floored, " << num_skipped
397  << " out of " << NumTransitionStates() << " transition-states "
398  "skipped due to insuffient data (it is normal to have some skipped.)";
399  if (objf_impr_out) *objf_impr_out = objf_impr_sum;
400  if (count_out) *count_out = count_sum;
401  ComputeDerivedOfProbs();
402 }

406 void TransitionModel::MapUpdate(const Vector<double> &stats,
407  const MapTransitionUpdateConfig &cfg,
408  BaseFloat *objf_impr_out,
409  BaseFloat *count_out) {
410  KALDI_ASSERT(cfg.tau > 0.0);
411  if (cfg.share_for_pdfs) {
412  MapUpdateShared(stats, cfg, objf_impr_out, count_out);
413  return;
414  }
415  BaseFloat count_sum = 0.0, objf_impr_sum = 0.0;
416  KALDI_ASSERT(stats.Dim() == NumTransitionIds()+1);
417  for (int32 tstate = 1; tstate <= NumTransitionStates(); tstate++) {
418  int32 n = NumTransitionIndices(tstate);
419  KALDI_ASSERT(n>=1);
420  if (n > 1) { // no point updating if only one transition...
421  Vector<double> counts(n);
422  for (int32 tidx = 0; tidx < n; tidx++) {
423  int32 tid = PairToTransitionId(tstate, tidx);
424  counts(tidx) = stats(tid);
425  }
426  double tstate_tot = counts.Sum();
427  count_sum += tstate_tot;
428  Vector<BaseFloat> old_probs(n), new_probs(n);
429  for (int32 tidx = 0; tidx < n; tidx++) {
430  int32 tid = PairToTransitionId(tstate, tidx);
431  old_probs(tidx) = new_probs(tidx) = GetTransitionProb(tid);
432  }
433  for (int32 tidx = 0; tidx < n; tidx++)
434  new_probs(tidx) = (counts(tidx) + cfg.tau * old_probs(tidx)) /
435  (cfg.tau + tstate_tot);
436  // Compute objf change
437  for (int32 tidx = 0; tidx < n; tidx++) {
438  double objf_change = counts(tidx) * (Log(new_probs(tidx))
439  - Log(old_probs(tidx)));
440  objf_impr_sum += objf_change;
441  }
442  // Commit updated values.
443  for (int32 tidx = 0; tidx < n; tidx++) {
444  int32 tid = PairToTransitionId(tstate, tidx);
445  log_probs_(tid) = Log(new_probs(tidx));
446  if (log_probs_(tid) - log_probs_(tid) != 0.0)
447  KALDI_ERR << "Log probs is inf or NaN: error in update or bad stats?";
448  }
449  }
450  }
451  KALDI_LOG << "Objf change is " << (objf_impr_sum / count_sum)
452  << " per frame over " << count_sum
453  << " frames.";
454  if (objf_impr_out) *objf_impr_out = objf_impr_sum;
455  if (count_out) *count_out = count_sum;
456  ComputeDerivedOfProbs();
457 }

464 void TransitionModel::MleUpdateShared(const Vector<double> &stats,
465  const MleTransitionUpdateConfig &cfg,
466  BaseFloat *objf_impr_out,
467  BaseFloat *count_out) {
468  KALDI_ASSERT(cfg.share_for_pdfs);
469 
470  BaseFloat count_sum = 0.0, objf_impr_sum = 0.0;
471  int32 num_skipped = 0, num_floored = 0;
472  KALDI_ASSERT(stats.Dim() == NumTransitionIds()+1);
473  std::map<int32, std::set<int32> > pdf_to_tstate;
474 
475  for (int32 tstate = 1; tstate <= NumTransitionStates(); tstate++) {
476  int32 pdf = TransitionStateToPdf(tstate);
477  pdf_to_tstate[pdf].insert(tstate);
478  }
479  std::map<int32, std::set<int32> >::iterator map_iter;
480  for (map_iter = pdf_to_tstate.begin();
481  map_iter != pdf_to_tstate.end();
482  ++map_iter) {
483  // map_iter->first is pdf-id... not needed.
484  const std::set<int32> &tstates = map_iter->second;
485  KALDI_ASSERT(!tstates.empty());
486  int32 one_tstate = *(tstates.begin());
487  int32 n = NumTransitionIndices(one_tstate);
488  KALDI_ASSERT(n >= 1);
489  if (n > 1) { // Only update if >1 transition...
490  Vector<double> counts(n);
491  for (std::set<int32>::const_iterator iter = tstates.begin();
492  iter != tstates.end();
493  ++iter) {
494  int32 tstate = *iter;
495  if (NumTransitionIndices(tstate) != n)
496  KALDI_ERR << "Mismatch in #transition indices: you cannot "
497  "use the --share-for-pdfs option with this topology "
498  "and sharing scheme.";
499  for (int32 tidx = 0; tidx < n; tidx++) {
500  int32 tid = PairToTransitionId(tstate, tidx);
501  counts(tidx) += stats(tid);
502  }
503  }
504  double pdf_tot = counts.Sum();
505  count_sum += pdf_tot;
506  if (pdf_tot < cfg.mincount) { num_skipped++; }
507  else {
508  // Note: when calculating objf improvement, we
509  // assume we previously had the same tying scheme so
510  // we can get the params from one_tstate and they're valid
511  // for all.
512  Vector<BaseFloat> old_probs(n), new_probs(n);
513  for (int32 tidx = 0; tidx < n; tidx++) {
514  int32 tid = PairToTransitionId(one_tstate, tidx);
515  old_probs(tidx) = new_probs(tidx) = GetTransitionProb(tid);
516  }
517  for (int32 tidx = 0; tidx < n; tidx++)
518  new_probs(tidx) = counts(tidx) / pdf_tot;
519  for (int32 i = 0; i < 3; i++) { // keep flooring+renormalizing for 3 times..
520  new_probs.Scale(1.0 / new_probs.Sum());
521  for (int32 tidx = 0; tidx < n; tidx++)
522  new_probs(tidx) = std::max(new_probs(tidx), cfg.floor);
523  }
524  // Compute objf change
525  for (int32 tidx = 0; tidx < n; tidx++) {
526  if (new_probs(tidx) == cfg.floor) num_floored++;
527  double objf_change = counts(tidx) * (Log(new_probs(tidx))
528  - Log(old_probs(tidx)));
529  objf_impr_sum += objf_change;
530  }
531  // Commit updated values.
532  for (std::set<int32>::const_iterator iter = tstates.begin();
533  iter != tstates.end();
534  ++iter) {
535  int32 tstate = *iter;
536  for (int32 tidx = 0; tidx < n; tidx++) {
537  int32 tid = PairToTransitionId(tstate, tidx);
538  log_probs_(tid) = Log(new_probs(tidx));
539  if (log_probs_(tid) - log_probs_(tid) != 0.0)
540  KALDI_ERR << "Log probs is inf or NaN: error in update or bad stats?";
541  }
542  }
543  }
544  }
545  }
546  KALDI_LOG << "Objf change is " << (objf_impr_sum / count_sum)
547  << " per frame over " << count_sum << " frames; "
548  << num_floored << " probabilities floored, "
549  << num_skipped << " pdf-ids skipped due to insuffient data.";
550  if (objf_impr_out) *objf_impr_out = objf_impr_sum;
551  if (count_out) *count_out = count_sum;
552  ComputeDerivedOfProbs();
553 }

559 void TransitionModel::MapUpdateShared(const Vector<double> &stats,
560  const MapTransitionUpdateConfig &cfg,
561  BaseFloat *objf_impr_out,
562  BaseFloat *count_out) {
563  KALDI_ASSERT(cfg.share_for_pdfs);
564 
565  BaseFloat count_sum = 0.0, objf_impr_sum = 0.0;
566  KALDI_ASSERT(stats.Dim() == NumTransitionIds()+1);
567  std::map<int32, std::set<int32> > pdf_to_tstate;
568 
569  for (int32 tstate = 1; tstate <= NumTransitionStates(); tstate++) {
570  int32 pdf = TransitionStateToPdf(tstate);
571  pdf_to_tstate[pdf].insert(tstate);
572  }
573  std::map<int32, std::set<int32> >::iterator map_iter;
574  for (map_iter = pdf_to_tstate.begin();
575  map_iter != pdf_to_tstate.end();
576  ++map_iter) {
577  // map_iter->first is pdf-id... not needed.
578  const std::set<int32> &tstates = map_iter->second;
579  KALDI_ASSERT(!tstates.empty());
580  int32 one_tstate = *(tstates.begin());
581  int32 n = NumTransitionIndices(one_tstate);
582  KALDI_ASSERT(n >= 1);
583  if (n > 1) { // Only update if >1 transition...
584  Vector<double> counts(n);
585  for (std::set<int32>::const_iterator iter = tstates.begin();
586  iter != tstates.end();
587  ++iter) {
588  int32 tstate = *iter;
589  if (NumTransitionIndices(tstate) != n)
590  KALDI_ERR << "Mismatch in #transition indices: you cannot "
591  "use the --share-for-pdfs option with this topology "
592  "and sharing scheme.";
593  for (int32 tidx = 0; tidx < n; tidx++) {
594  int32 tid = PairToTransitionId(tstate, tidx);
595  counts(tidx) += stats(tid);
596  }
597  }
598  double pdf_tot = counts.Sum();
599  count_sum += pdf_tot;
600 
601  // Note: when calculating objf improvement, we
602  // assume we previously had the same tying scheme so
603  // we can get the params from one_tstate and they're valid
604  // for all.
605  Vector<BaseFloat> old_probs(n), new_probs(n);
606  for (int32 tidx = 0; tidx < n; tidx++) {
607  int32 tid = PairToTransitionId(one_tstate, tidx);
608  old_probs(tidx) = new_probs(tidx) = GetTransitionProb(tid);
609  }
610  for (int32 tidx = 0; tidx < n; tidx++)
611  new_probs(tidx) = (counts(tidx) + old_probs(tidx) * cfg.tau) /
612  (pdf_tot + cfg.tau);
613  // Compute objf change
614  for (int32 tidx = 0; tidx < n; tidx++) {
615  double objf_change = counts(tidx) * (Log(new_probs(tidx))
616  - Log(old_probs(tidx)));
617  objf_impr_sum += objf_change;
618  }
619  // Commit updated values.
620  for (std::set<int32>::const_iterator iter = tstates.begin();
621  iter != tstates.end();
622  ++iter) {
623  int32 tstate = *iter;
624  for (int32 tidx = 0; tidx < n; tidx++) {
625  int32 tid = PairToTransitionId(tstate, tidx);
626  log_probs_(tid) = Log(new_probs(tidx));
627  if (log_probs_(tid) - log_probs_(tid) != 0.0)
628  KALDI_ERR << "Log probs is inf or NaN: error in update or bad stats?";
629  }
630  }
631  }
632  }
633  KALDI_LOG << "Objf change is " << (objf_impr_sum / count_sum)
634  << " per frame over " << count_sum
635  << " frames.";
636  if (objf_impr_out) *objf_impr_out = objf_impr_sum;
637  if (count_out) *count_out = count_sum;
638  ComputeDerivedOfProbs();
639 }

712  trans_model.TransitionStateToPdf(tstate))
713  && !std::binary_search(phones.begin(), phones.end(),
714  trans_model.TransitionStateToPhone(tstate)))
715  return false;
716  return true;
