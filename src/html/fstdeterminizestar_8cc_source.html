62 int main(int argc, char *argv[]) {
63  try {
64  using namespace kaldi;
65  using namespace fst;
66  using kaldi::int32;
67 
68  const char *usage =
69  "Removes epsilons and determinizes in one step\n"
70  "\n"
71  "Usage: fstdeterminizestar [in.fst [out.fst] ]\n"
72  "\n"
73  "See also: fstdeterminizelog, lattice-determinize\n";
74 
75  float delta = kDelta;
76  int max_states = -1;
77  bool use_log = false;
78  ParseOptions po(usage);
79  po.Register("use-log", &use_log, "Determinize in log semiring.");
80  po.Register("delta", &delta, "Delta value used to determine equivalence of weights.");
81  po.Register("max-states", &max_states, "Maximum number of states in determinized FST before it will abort.");
82  po.Read(argc, argv);
83 
84  if (po.NumArgs() > 2) {
85  po.PrintUsage();
86  exit(1);
87  }
88 
89  std::string fst_in_str = po.GetOptArg(1),
90  fst_out_str = po.GetOptArg(2);
91 
92  // This enables us to get traceback info from determinization that is
93  // not seeming to terminate.
94 #ifndef _MSC_VER
95  signal(SIGUSR1, signal_handler);
96 #endif
97  if (ClassifyRspecifier(fst_in_str, NULL, NULL) == kNoRspecifier) {
98  // Normal case: just files.
99  VectorFst<StdArc> *fst = ReadFstKaldi(fst_in_str);
100 
101  ArcSort(fst, ILabelCompare<StdArc>()); // improves speed.
102  if (use_log) {
103  DeterminizeStarInLog(fst, delta, &debug_location, max_states);
104  } else {
105  VectorFst<StdArc> det_fst;
106  DeterminizeStar(*fst, &det_fst, delta, &debug_location, max_states);
107  *fst = det_fst; // will do shallow copy and then det_fst goes
108  // out of scope anyway.
109  }
110  WriteFstKaldi(*fst, fst_out_str);
111  delete fst;
112  } else { // Dealing with archives.
113  SequentialTableReader<VectorFstHolder> fst_reader(fst_in_str);
114  TableWriter<VectorFstHolder> fst_writer(fst_out_str);
115  for (; !fst_reader.Done(); fst_reader.Next()) {
116  std::string key = fst_reader.Key();
117  VectorFst<StdArc> fst(fst_reader.Value());
118  fst_reader.FreeCurrent();
119  ArcSort(&fst, ILabelCompare<StdArc>()); // improves speed.
120  try {
121  if (use_log) {
122  DeterminizeStarInLog(&fst, delta, &debug_location, max_states);
123  } else {
124  VectorFst<StdArc> det_fst;
125  DeterminizeStar(fst, &det_fst, delta, &debug_location, max_states);
126  fst = det_fst; // will do shallow copy and then det_fst goes out
127  // of scope anyway.
128  }
129  fst_writer.Write(key, fst);
130  } catch (const std::runtime_error e) {
131  KALDI_WARN << "Error during determinization for key " << key;
132  }
133  }
134  }
135  return 0;
136  } catch(const std::exception &e) {
137  std::cerr << e.what();
138  return -1;
139  }
140 }
