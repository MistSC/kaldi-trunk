113  b[i]->SetRandn();

118  C.push_back(new CuSubMatrix<Real>(*(c[i]), 0, c[i]->NumRows(), 0,
119  c[i]->NumCols()));
120  }
121  BaseFloat time_in_secs = 0.025;
122  Timer tim;
123  int32 iter = 0;

125  AddMatMatBatched(static_cast<Real>(1.0), C, A, kNoTrans, B, kNoTrans,

127  }
128  for (int32 i = 0; i< batchCount; i++) {
129  delete a[i]; delete b[i]; delete c[i];
130  delete A[i]; delete B[i]; delete C[i];

132 
133  BaseFloat fdim = dim;

181 template<typename Real>
182 static void TestCuMatrixCompObjfAndDeriv(int32 dim) {
183  BaseFloat time_in_secs = 0.025;
184  // Previously tested for larger dims, but test was slow.
185 
186  int32 n_r = dim, n_c = dim + Rand() % 5;
187 
188  CuMatrix<Real> A(n_r, n_c), B(n_r, n_c);
189  B.SetRandn();
190  B.Add(1.0);
191  B.ApplyFloor(1.0e-10);
192 
193  std::vector<MatrixElement<Real> > labels;
194  for(int i = 0; i < n_r; i++) {
195  for(int j = 0; j < n_c; j++) {
196  // have approximately one weight per row of the matrix.
197  if (Rand() % n_c == 0) {
198  A(i, j) = RandUniform();
199  MatrixElement<Real> t = {i, j, A(i, j)};
200  labels.push_back(t);
201  }
202  }
203  }
204  CuMatrix<Real> C(n_r, n_c);
205 
206  int iter = 0;
207  Timer tim;
208  Real a = 0.0, b = 0.0;
209  for (;tim.Elapsed() < time_in_secs; iter++)
210  C.CompObjfAndDeriv(labels, B, &a, &b);
211 
212  BaseFloat gflops = (n_r * n_c * iter) / (tim.Elapsed() * 1.0e+09);
213  KALDI_LOG << "For CuMatrix::CompObjfAndDeriv" << NameOf<Real>() << ", for dim = "
214  << dim << ", speed was " << gflops << " gigaflops.";
215 
216 
217  // do it one more time for correctness test.
218  C.SetZero();
219  C.CompObjfAndDeriv(labels, B, &a, &b);
220 
221  KALDI_ASSERT(ApproxEqual(b, A.Sum()));
222 
223  // repeat the real test.
224  Real sum2; // sum(i, j) A(i, j) log(B(i, j));
225  {
226  CuMatrix<Real> Bcopy(B);
227  Bcopy.ApplyLog();
228  sum2 = TraceMatMat(Bcopy, A, kTrans);
229  }
230 
231  KALDI_ASSERT(ApproxEqual(a, sum2));
232 
233  B.InvertElements();
234  A.MulElements(B); // each element of A is now A(i, j) / B(i, j);
235  KALDI_ASSERT(ApproxEqual(A, C));
236 
237 
238 }
