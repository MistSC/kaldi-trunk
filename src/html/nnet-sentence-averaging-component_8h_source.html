81  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
82  // get the average row-vector,
83  average_row_.Resize(InputDim());
84  average_row_.AddRowSumMat(1.0/in.NumRows(), in, 0.0);
85  // copy it on the output,
86  out->AddVecToRows(1.0, average_row_, 0.0);
87  }

89  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
90  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
91  // While averaging, a single frame from input influenced all frames on the output,
92  // hence the derivative w.r.t. single input frame is a sum of the output derivatives.
93  // (while scaling by averaging constant 1/Nframes).
94  //
95  // In fact all the input frames influenced all the output frames,
96  // so the derivarive is the same for all the input frames.
97  //
98  // getting the average output diff,
99  average_diff_.Resize(OutputDim());
100  average_diff_.AddRowSumMat(1.0/out_diff.NumRows(), out_diff, 0.0);
101  // copy the derivative into the input diff, (applying gradient-boost!!)
102  in_diff->AddVecToRows(gradient_boost_, average_diff_, 0.0);
103  }
104 
105  private:
106  CuVector<BaseFloat> average_row_;
107  CuVector<BaseFloat> average_diff_;
108  BaseFloat gradient_boost_;
109 };

163  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
164  // Get NN output
165  CuMatrix<BaseFloat> out_nnet;
166  nnet_.Propagate(in, &out_nnet);
167  // Get the average row (averaging over the time axis):
168  // averaging corresponds to extraction of constant vector code for single sentence,
169  int32 num_inputs = in.NumCols(),
170  nnet_outputs = nnet_.OutputDim(),
171  num_frames = out_nnet.NumRows();
172 
173  CuVector<BaseFloat> average_row(nnet_outputs);
174  average_row.AddRowSumMat(1.0/num_frames, out_nnet, 0.0);
175  // Forwarding sentence codes along with input features
176  out->ColRange(0,nnet_outputs).AddVecToRows(1.0, average_row, 0.0);
177  out->ColRange(nnet_outputs,num_inputs).CopyFromMat(in);
178  }

188  void Update(const CuMatrixBase<BaseFloat> &input, const CuMatrixBase<BaseFloat> &diff) {
189 
190  int32 nnet_outputs = nnet_.OutputDim(),
191  num_frames = diff.NumRows();
192  // Passing the derivative into the nested network. The loss derivative is averaged:
193  // single frame from nested network influenced all the frames in the main network,
194  // so to get the derivative w.r.t. single frame from nested network we sum derivatives
195  // of all frames from main network (and scale by 1/Nframes constant).
196  //
197  // In fact all the frames from nested network influenced all the input frames to main nnet,
198  // so the loss derivarive w.r.t. nested network output is same for all frames in sentence.
199  CuVector<BaseFloat> average_diff(nnet_outputs);
200  average_diff.AddRowSumMat(1.0/num_frames, diff.ColRange(0,nnet_outputs), 0.0);
201  CuMatrix<BaseFloat> nnet_out_diff(num_frames, nnet_outputs);
202  nnet_out_diff.AddVecToRows(1.0, average_diff, 0.0);
203  //
204  nnet_.Backpropagate(nnet_out_diff, NULL);
205  }
