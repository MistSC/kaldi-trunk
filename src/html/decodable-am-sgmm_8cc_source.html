27 BaseFloat DecodableAmSgmm::LogLikelihoodZeroBased(int32 frame, int32 pdf_id) {
28  KALDI_ASSERT(frame >= 0 && frame < NumFramesReady());
29  KALDI_ASSERT(pdf_id >= 0 && pdf_id < NumIndices());
30 
31  if (log_like_cache_[pdf_id].hit_time == frame) {
32  return log_like_cache_[pdf_id].log_like; // return cached value, if found
33  }
34 
35  const VectorBase<BaseFloat> &data = feature_matrix_.Row(frame);
36  // check if everything is in order
37  if (acoustic_model_.FeatureDim() != data.Dim()) {
38  KALDI_ERR << "Dim mismatch: data dim = " << data.Dim()
39  << "vs. model dim = " << acoustic_model_.FeatureDim();
40  }
41 
42  if (frame != previous_frame_) { // Per-frame precomputation for SGMM.
43  if (gselect_all_.empty())
44  acoustic_model_.GaussianSelection(sgmm_config_, data, &gselect_);
45  else {
46  KALDI_ASSERT(frame < gselect_all_.size());
47  gselect_ = gselect_all_[frame];
48  }
49  acoustic_model_.ComputePerFrameVars(data, gselect_, spk_,
50  0.0 /*FMLLR logdet*/, &per_frame_vars_);
51  previous_frame_ = frame;
52  }
53 
54  BaseFloat loglike = acoustic_model_.LogLikelihood(per_frame_vars_, pdf_id,
55  log_prune_);
56  if (KALDI_ISNAN(loglike) || KALDI_ISINF(loglike))
57  KALDI_ERR << "Invalid answer (overflow or invalid variances/features?)";
58  log_like_cache_[pdf_id].log_like = loglike;
59  log_like_cache_[pdf_id].hit_time = frame;
60  return loglike;
61 }
