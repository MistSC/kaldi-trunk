32 void MleAmSgmmGlobalAccs::ResizeAccumulators(const AmSgmm &model,
33  SgmmUpdateFlagsType flags) {
34  num_gaussians_ = model.NumGauss();
35  feature_dim_ = model.FeatureDim();
36  phn_space_dim_ = model.PhoneSpaceDim();
37  spk_space_dim_ = model.SpkSpaceDim();
38 
39  if (flags & (kSgmmPhoneProjections | kSgmmCovarianceMatrix)) {
40  Y_.resize(num_gaussians_);
41  Q_.resize(num_gaussians_);
42  for (int32 i = 0; i < num_gaussians_; ++i) {
43  Y_[i].Resize(feature_dim_, phn_space_dim_, kSetZero);
44  Q_[i].Resize(phn_space_dim_, kSetZero);
45  }
46  } else {
47  Y_.clear();
48  Q_.clear();
49  }
50 
51  if (flags & kSgmmCovarianceMatrix) {
52  S_.resize(num_gaussians_);
53  S_means_.resize(num_gaussians_);
54  for (int32 i = 0; i < num_gaussians_; i++) {
55  S_[i].Resize(feature_dim_, kSetZero);
56  S_means_[i].Resize(feature_dim_, kSetZero);
57  }
58  } else {
59  S_.clear();
60  }
61 
62  if (flags & kSgmmSpeakerProjections) {
63  if (spk_space_dim_ == 0) {
64  KALDI_ERR << "Cannot set up accumulators for speaker projections "
65  << "because speaker subspace has not been set up";
66  }
67  Z_.resize(num_gaussians_);
68  R_.resize(num_gaussians_);
69  for (int32 i = 0; i < num_gaussians_; ++i) {
70  Z_[i].Resize(feature_dim_, spk_space_dim_, kSetZero);
71  R_[i].Resize(spk_space_dim_, kSetZero);
72  }
73  } else {
74  Z_.clear();
75  R_.clear();
76  }
77 
78  gamma_i_.Resize(num_gaussians_, kSetZero);
79 }

102 void MleAmSgmmGlobalAccs::AddAccumulators(const AmSgmm &model,
103  const MleAmSgmmAccs &accs,
104  SgmmUpdateFlagsType flags) {
105  total_frames_ += accs.total_frames_;
106  total_like_ += accs.total_like_;
107  for (int32 i = 0; i < num_gaussians_; ++i) {
108  if (flags & (kSgmmPhoneProjections | kSgmmCovarianceMatrix)) {
109  Y_[i].AddMat(1.0, accs.Y_[i], kNoTrans);
110  }
111  if (flags & kSgmmSpeakerProjections) {
112  Z_[i].AddMat(1.0, accs.Z_[i], kNoTrans);
113  R_[i].AddSp(1.0, accs.R_[i]);
114  }
115  if (flags & kSgmmCovarianceMatrix)
116  S_[i].AddSp(1.0, accs.S_[i]);
117  }
118 
119  // gamma_i
120  for (int32 j = 0; j < model.NumPdfs(); ++j) {
121  for (int32 m = 0; m < model.NumSubstates(j); ++m) {
122  gamma_i_.AddVec(1.0, accs.gamma_[j].Row(m));
123  }
124  }
125 
126  // Compute the Q_i quantities (Eq. 64).
127  if (flags & kSgmmPhoneProjections) {
128  for (int32 i = 0; i < num_gaussians_; ++i) {
129  for (int32 j = 0; j < accs.num_states_; ++j) {
130  const Matrix<BaseFloat> &state_vec(model.StateVectors(j));
131  for (int32 m = 0; m < model.NumSubstates(j); ++m) {
132  if (accs.gamma_[j](m, i) > 0.0) {
133  Q_[i].AddVec2(static_cast<BaseFloat>(accs.gamma_[j](m, i)),
134  state_vec.Row(m));
135  }
136  }
137  }
138  }
139  }
140 
141  // Compute the S_i^{(means)} quantities (Eq. 74).
142  if (flags & kSgmmCovarianceMatrix) {
143  Matrix<double> YM_MY(feature_dim_, feature_dim_);
144  SpMatrix<double> tmp_S_means(feature_dim_);
145  Vector<BaseFloat> mu_jmi(feature_dim_);
146  for (int32 i = 0; i < num_gaussians_; ++i) {
147  // YM_MY = - (Y_{i} M_{i}^T)
148  Matrix<double> M(model.GetPhoneProjection(i));
149  YM_MY.AddMatMat(-1.0, accs.Y_[i], kNoTrans, M, kTrans, 0.0);
150  // Add its own transpose: YM_MY = - (Y_{i} M_{i}^T + M_{i} Y_{i}^T)
151  {
152  Matrix<double> M(YM_MY, kTrans);
153  YM_MY.AddMat(1.0, M);
154  }
155  tmp_S_means.CopyFromMat(YM_MY); // Sigma_{i} = -(YM' + MY')
156 
157  for (int32 j = 0; j < accs.num_states_; ++j) {
158  for (int32 m = 0; m < model.NumSubstates(j); ++m) {
159  // Sigma_{i} += gamma_{jmi} * mu_{jmi}*mu_{jmi}^T
160  model.GetSubstateMean(j, m, i, &mu_jmi);
161  tmp_S_means.AddVec2(static_cast<BaseFloat>(accs.gamma_[j](m, i)), mu_jmi);
162  }
163  }
164  S_means_[i].AddSp(1.0, tmp_S_means);
165  KALDI_ASSERT(1.0 / S_means_[i](0, 0) != 0.0);
166  }
167  }
168 }

190 void MleAmSgmmUpdaterMulti::Update(const std::vector<MleAmSgmmAccs*> &accs,
191  const std::vector<AmSgmm*> &models,
192  SgmmUpdateFlagsType flags) {
193  KALDI_ASSERT((flags & (kSgmmPhoneVectors | kSgmmPhoneProjections |
194  kSgmmPhoneWeightProjections | kSgmmCovarianceMatrix |
195  kSgmmSubstateWeights | kSgmmSpeakerProjections)) != 0);
196  if (accs.size() != models.size()) {
197  KALDI_ERR << "Found " << accs.size() << " accs and " << models.size()
198  << " models. Must have same number of models and accs.";
199  }
200 
201  SgmmUpdateFlagsType global_flags = (flags & (kSgmmPhoneProjections |
202  kSgmmPhoneWeightProjections |
203  kSgmmSpeakerProjections |
204  kSgmmCovarianceMatrix));
205  SgmmUpdateFlagsType state_spec_flags = (flags & ~global_flags);
206  MleAmSgmmGlobalAccs glob_accs;
207  BaseFloat tot_impr = 0.0;
208  int32 num_models = models.size();
209 
210  std::vector< SpMatrix<double> > H;
211  if (update_options_.renormalize_V)
212  models[0]->ComputeH(&H);
213 
214  if (global_flags != 0) { // expected operating case
215  glob_accs.ResizeAccumulators(*models[0], global_flags);
216  for (int32 i = 0; i < num_models; ++i) {
217  glob_accs.AddAccumulators(*models[i], *accs[i], global_flags);
218  }
219  UpdateGlobals(glob_accs, global_flags);
220 
221  // Weight projection needs access to all models
222  if (global_flags & kSgmmPhoneWeightProjections) {
223  if (update_options_.use_sequential_weight_update)
224  KALDI_ERR << "Sequential weight update not implemented, using parallel";
225 // tot_impr += UpdateWSequential(accs, model);
226 // } else {
227  tot_impr += UpdateWParallel(accs, models);
228 // }
229  }
230  } else { // Shouldn't be using this class without updating global params
231  KALDI_WARN << "Using MleAmSgmmUpdaterMulti class without updating global "
232  << " parameters.";
233  }
234 
235  // Update the state-specific parameters: phone vectors & substate weights
236  if (state_spec_flags != 0) {
237  MleAmSgmmOptions state_spec_opts = update_options_;
238  state_spec_opts.renormalize_V = false;
239  state_spec_opts.renormalize_N = false;
240 
241  MleAmSgmmUpdater sgmm_updater(state_spec_opts);
242  for (int32 i = 0; i < num_models; ++i)
243  tot_impr += sgmm_updater.Update(*accs[i], models[i], state_spec_flags);
244  }
245 
246 
247  if (update_options_.renormalize_V && (global_flags != 0)) {
248  SpMatrix<double> H_sm;
249  this->ComputeSmoothingTerms(glob_accs, H, &H_sm);
250  RenormalizeV(H_sm, models);
251  }
252 
253  KALDI_LOG << "**Total auxf improvement, combining all parameters, over "
254  << "all model is " << tot_impr << " per frame.";
255 
256  // The following is just for diagnostics
257  double total_frames = 0, total_like = 0;
258  for (int32 i = 0; i < num_models; ++i) {
259  total_frames += accs[i]->TotalFrames();
260  total_like += accs[i]->TotalLike();
261  }
262  KALDI_LOG << "***Total data likelihood, over all models, is "
263  << (total_like/total_frames) << " over " << total_frames
264  << " frames.";
265 
266  // Now, copy the global parameters to the models
267  for (int32 i = 0; i < num_models; ++i) {
268  if ((flags & kSgmmPhoneProjections) || update_options_.renormalize_V)
269  models[i]->M_ = global_M_;
270  if (flags & kSgmmCovarianceMatrix)
271  models[i]->SigmaInv_ = global_SigmaInv_;
272  if ((flags & kSgmmSpeakerProjections) || update_options_.renormalize_N)
273  models[i]->N_ = global_N_;
274  if ((flags & kSgmmPhoneWeightProjections) || update_options_.renormalize_V)
275  models[i]->w_ = global_w_;
276  models[i]->ComputeNormalizers(); // So that the models are ready to use.
277  }
278 }

280 // Compute H^{(sm)}, the "smoothing" matrices.
281 void MleAmSgmmUpdaterMulti::ComputeSmoothingTerms(
282  const MleAmSgmmGlobalAccs &accs,
283  const std::vector< SpMatrix<double> > &H,
284  SpMatrix<double> *H_sm) const {
285  KALDI_ASSERT(H_sm != NULL);
286  H_sm->Resize(PhoneSpaceDim());
287 
288  double sum = 0.0;
289  for (int32 i = 0; i < NumGauss(); ++i) {
290  if (accs.gamma_i_(i) > 0) {
291  H_sm->AddSp(accs.gamma_i_(i), H[i]);
292  sum += accs.gamma_i_(i);
293  }
294  }
295 
296  if (sum == 0.0) {
297  KALDI_WARN << "Sum of counts is zero. Smoothing matrix set to unit";
298  H_sm->SetUnit(); // arbitrary non-singular matrix
299  } else {
300  H_sm->Scale(1.0 / sum);
301  int32 tmp = H_sm->LimitCondDouble(update_options_.max_cond_H_sm);
302  if (tmp > 0) {
303  KALDI_WARN << "Limited " << tmp << " eigenvalues of H_sm.";
304  }
305  }

308 double MleAmSgmmUpdaterMulti::UpdateM(const MleAmSgmmGlobalAccs &accs) {
309  double totcount = 0.0, tot_like_impr = 0.0;
310  for (int32 i = 0; i < accs.num_gaussians_; ++i) {
311  if (accs.gamma_i_(i) < accs.feature_dim_) {
312  KALDI_WARN << "For component " << i << ": not updating M due to very "
313  << "small count (=" << accs.gamma_i_(i) << ").";
314  continue;
315  }
316 
317 
318  SolverOptions opts;
319  opts.name = "M";
320  opts.K = update_options_.max_cond;
321  opts.eps = update_options_.epsilon;
322 
323  Matrix<double> Mi(global_M_[i]);
324  double impr =
325  SolveQuadraticMatrixProblem(accs.Q_[i], accs.Y_[i],
326  SpMatrix<double>(global_SigmaInv_[i]),
327  opts, &Mi);
328  global_M_[i].CopyFromMat(Mi);
329 
330  if (i % 50 == 0) {
331  KALDI_VLOG(2) << "Objf impr for projection M for i = " << i << ", is "
332  << (impr/(accs.gamma_i_(i) + 1.0e-20)) << " over "
333  << accs.gamma_i_(i) << " frames";
334  }
335  totcount += accs.gamma_i_(i);
336  tot_like_impr += impr;
337  }
338  tot_like_impr /= (totcount + 1.0e-20);
339  KALDI_LOG << "Overall objective function improvement for model projections "
340  << "M is " << tot_like_impr << " over " << totcount << " frames";
341  return tot_like_impr;
342 }

344 double MleAmSgmmUpdaterMulti::UpdateN(const MleAmSgmmGlobalAccs &accs) {
345  double totcount = 0.0, tot_like_impr = 0.0;
346  if (accs.spk_space_dim_ == 0 || accs.R_.size() == 0 || accs.Z_.size() == 0) {
347  KALDI_ERR << "Speaker subspace dim is zero or no stats accumulated";
348  }
349 
350  for (int32 i = 0; i < accs.num_gaussians_; ++i) {
351  if (accs.gamma_i_(i) < 2 * accs.spk_space_dim_) {
352  KALDI_WARN << "Not updating speaker basis for i = " << (i)
353  << " because count is too small " << (accs.gamma_i_(i));
354  continue;
355  }
356 
357  SolverOptions opts;
358  opts.name = "N";
359  opts.K = update_options_.max_cond;
360  opts.eps = update_options_.epsilon;
361 
362  Matrix<double> Ni(global_N_[i]);
363  double impr =
364  SolveQuadraticMatrixProblem(accs.R_[i], accs.Z_[i],
365  SpMatrix<double>(global_SigmaInv_[i]),
366  opts, &Ni);
367  global_N_[i].CopyFromMat(Ni);
368  if (i < 10) {
369  KALDI_LOG << "Objf impr for spk projection N for i = " << (i)
370  << ", is " << (impr / (accs.gamma_i_(i) + 1.0e-20)) << " over "
371  << (accs.gamma_i_(i)) << " frames";
372  }
373  totcount += accs.gamma_i_(i);
374  tot_like_impr += impr;
375  }
376 
377  tot_like_impr /= (totcount+1.0e-20);
378  KALDI_LOG << "**Overall objf impr for N is " << tot_like_impr << " over "
379  << totcount << " frames";
380  return tot_like_impr;
381 }

384 double MleAmSgmmUpdaterMulti::UpdateVars(const MleAmSgmmGlobalAccs &accs) {
385  SpMatrix<double> Sigma_i(accs.feature_dim_), Sigma_i_ml(accs.feature_dim_);
386  double tot_objf_impr = 0.0, tot_t = 0.0;
387  SpMatrix<double> covfloor(accs.feature_dim_);
388  Vector<double> objf_improv(accs.num_gaussians_);
389 
390  // First pass over all (shared) Gaussian components to calculate the
391  // ML estimate of the covariances, and the total covariance for flooring.
392  for (int32 i = 0; i < accs.num_gaussians_; ++i) {
393  // Eq. (75): Sigma_{i}^{ml} = 1/gamma_{i} [S_{i} + S_{i}^{(means)} - ...
394  // Y_{i} M_{i}^T - M_{i} Y_{i}^T]
395  // Note the S_means_ already contains the Y_{i} M_{i}^T terms.
396  Sigma_i_ml.CopyFromSp(accs.S_means_[i]);
397  Sigma_i_ml.AddSp(1.0, accs.S_[i]);
398  covfloor.AddSp(1.0, Sigma_i_ml);
399  // inverting small values e.g. 4.41745328e-40 seems to generate inf,
400  // although would be fixed up later.
401  if (accs.gamma_i_(i) > 1.0e-20) {
402  Sigma_i_ml.Scale(1 / (accs.gamma_i_(i) + 1.0e-20));
403  } else {
404  Sigma_i_ml.SetUnit();
405  }
406  KALDI_ASSERT(1.0 / Sigma_i_ml(0, 0) != 0.0);
407  // Eq. (76): Compute the objective function with the old parameter values
408  objf_improv(i) = global_SigmaInv_[i].LogPosDefDet() -
409  TraceSpSp(SpMatrix<double>(global_SigmaInv_[i]), Sigma_i_ml);
410 
411  global_SigmaInv_[i].CopyFromSp(Sigma_i_ml); // inverted in the next loop.
412  }
413 
414  // Compute the covariance floor.
415  if (accs.gamma_i_.Sum() == 0) { // If no count, use identity.
416  KALDI_WARN << "Updating variances: zero counts. Setting floor to unit.";
417  covfloor.SetUnit();
418  } else { // else, use the global average covariance.
419  covfloor.Scale(update_options_.cov_floor / accs.gamma_i_.Sum());
420  int32 tmp;
421  if ((tmp = covfloor.LimitCondDouble(update_options_.max_cond)) != 0) {
422  KALDI_WARN << "Covariance flooring matrix is poorly conditioned. Fixed "
423  << "up " << (tmp) << " eigenvalues.";
424  }
425  }
426 
427  if (update_options_.cov_diag_ratio > 1000) {
428  KALDI_LOG << "Assuming you want to build a diagonal system since "
429  << "cov_diag_ratio is large: making diagonal covFloor.";
430  for (int32 i = 0; i < covfloor.NumRows(); i++)
431  for (int32 j = 0; j < i; j++)
432  covfloor(i, j) = 0.0;
433  }
434 
435  // Second pass over all (shared) Gaussian components to calculate the
436  // floored estimate of the covariances, and update the model.
437  for (int32 i = 0; i < accs.num_gaussians_; ++i) {
438  Sigma_i.CopyFromSp(global_SigmaInv_[i]);
439  Sigma_i_ml.CopyFromSp(Sigma_i);
440  // In case of insufficient counts, make the covariance matrix diagonal.
441  // cov_diag_ratio is 2 by default, set to very large to always get diag-cov
442  if (accs.gamma_i_(i) < update_options_.cov_diag_ratio * accs.feature_dim_) {
443  KALDI_WARN << "For Gaussian component " << i << ": Too low count "
444  << accs.gamma_i_(i) << " for covariance matrix estimation. "
445  << "Setting to diagonal";
446  for (int32 d = 0; d < accs.feature_dim_; d++)
447  for (int32 e = 0; e < d; e++)
448  Sigma_i(d, e) = 0.0; // SpMatrix, can only set lower traingular part
449 
450  int floored = Sigma_i.ApplyFloor(covfloor);
451  if (floored > 0) {
452  KALDI_WARN << "For Gaussian component " << i << ": Floored " << floored
453  << " covariance eigenvalues.";
454  }
455  global_SigmaInv_[i].CopyFromSp(Sigma_i);
456  global_SigmaInv_[i].InvertDouble();
457  } else { // Updating the full covariance matrix.
458  try {
459  int floored = Sigma_i.ApplyFloor(covfloor);
460  if (floored > 0) {
461  KALDI_WARN << "For Gaussian component " << i << ": Floored "
462  << floored << " covariance eigenvalues.";
463  }
464  global_SigmaInv_[i].CopyFromSp(Sigma_i);
465  global_SigmaInv_[i].InvertDouble();
466 
467  objf_improv(i) += Sigma_i.LogPosDefDet() +
468  TraceSpSp(SpMatrix<double>(global_SigmaInv_[i]), Sigma_i_ml);
469  objf_improv(i) *= (-0.5 * accs.gamma_i_(i)); // Eq. (76)
470 
471  tot_objf_impr += objf_improv(i);
472  tot_t += accs.gamma_i_(i);
473  if (i < 5) {
474  KALDI_VLOG(2) << "objf impr from variance update =" << objf_improv(i)
475  / (accs.gamma_i_(i) + 1.0e-20) << " over " << (accs.gamma_i_(i))
476  << " frames for i = " << (i);
477  }
478  } catch(...) {
479  KALDI_WARN << "Updating within-class covariance matrix i = " << (i)
480  << ", numerical problem";
481  // This is a catch-all thing in case of unanticipated errors, but
482  // flooring should prevent this occurring for the most part.
483  global_SigmaInv_[i].SetUnit(); // Set to unit.
484  }
485  }
486  }
487  KALDI_LOG << "**Overall objf impr for variance update = "
488  << (tot_objf_impr / (tot_t+ 1.0e-20))
489  << " over " << (tot_t) << " frames";
490  return tot_objf_impr / (tot_t + 1.0e-20);
491 }

494 // The parallel weight update, in the paper.
495 double MleAmSgmmUpdaterMulti::UpdateWParallel(
496  const std::vector<MleAmSgmmAccs*> &accs,
497  const std::vector<AmSgmm*> &models) {
498  KALDI_LOG << "Updating weight projections";
499 
500  int32 phn_dim = models[0]->PhoneSpaceDim(),
501  num_gauss = models[0]->NumGauss(),
502  num_models = models.size();
503  SpMatrix<double> v_vT(phn_dim);
504  // tot_like_{after, before} are totals over multiple iterations,
505  // not valid likelihoods. but difference is valid (when divided by tot_count).
506  double tot_predicted_like_impr = 0.0, tot_like_before = 0.0,
507  tot_like_after = 0.0, tot_count = 0.0;
508 
509  Vector<double> w_jm(num_gauss);
510  Matrix<double> g_i(num_gauss, phn_dim);
511  std::vector< SpMatrix<double> > F_i(num_gauss);
512 
513  Matrix<double> w(global_w_);
514  for (int iter = 0; iter < update_options_.weight_projections_iters; iter++) {
515  for (int32 i = 0; i < num_gauss; ++i) {
516  F_i[i].Resize(phn_dim, kSetZero);
517  }
518  double k_like_before = 0.0, k_count = 0.0;
519  g_i.SetZero();
520 
521  // Unlike in the report the inner most loop is over Gaussians, where
522  // per-gaussian statistics are accumulated. This is more memory demanding
523  // but more computationally efficient, as outer product v_{jvm} v_{jvm}^T
524  // is computed only once for all gaussians.
525 
526  for (int32 mdl_idx = 0; mdl_idx < num_models; ++mdl_idx) {
527  std::vector< Matrix<double> > gamma(accs[mdl_idx]->GetOccs());
528  for (int32 j = 0; j < models[mdl_idx]->NumPdfs(); j++) {
529  for (int32 m = 0; m < models[mdl_idx]->NumSubstates(j); m++) {
530  double gamma_jm = gamma[j].Row(m).Sum();
531  k_count += gamma_jm;
532 
533  // w_jm = softmax([w_{k1}^T ... w_{kD}^T] * v_{jkm}) eq.(7)
534  w_jm.AddMatVec(1.0, w, kNoTrans,
535  Vector<double>(models[mdl_idx]->v_[j].Row(m)), 0.0);
536  w_jm.Add((-1.0) * w_jm.LogSumExp());
537  k_like_before += VecVec(w_jm, gamma[j].Row(m));
538  w_jm.ApplyExp();
539  v_vT.SetZero();
540  // v_vT := v_{jkm} v_{jkm}^T
541  v_vT.AddVec2(1.0, models[mdl_idx]->v_[j].Row(m));
542 
543  for (int32 i = 0; i < num_gauss; i++) {
544  // Suggestion: g_jkm can be computed more efficiently
545  // using the Vector/Matrix routines for all i at once
546  // linear term around cur value.
547  double linear_term = gamma[j](m, i) - gamma_jm * w_jm(i);
548  double quadratic_term = std::max(gamma[j](m, i), gamma_jm * w_jm(i));
549  g_i.Row(i).AddVec(linear_term, models[mdl_idx]->v_[j].Row(m));
550  // Now I am calling this F_i in the document. [dan]
551  F_i[i].AddSp(quadratic_term, v_vT);
552  }
553  } // loop over substates
554  } // loop over states
555  } // loop over model/acc pairs
556 
557  Matrix<double> w_orig(w);
558  double k_predicted_like_impr = 0.0, k_like_after = 0.0;
559  double min_step = 0.001, step_size;
560 
561  SolverOptions opts;
562  opts.name = "w";
563  opts.K = update_options_.max_cond;
564  opts.eps = update_options_.epsilon;
565 
566  for (step_size = 1.0; step_size >= min_step; step_size /= 2) {
567  k_predicted_like_impr = 0.0;
568  k_like_after = 0.0;
569 
570  for (int32 i = 0; i < num_gauss; i++) {
571  // auxf is formulated in terms of change in w.
572  Vector<double> delta_w(phn_dim);
573  // returns objf impr with step_size = 1,
574  // but it may not be 1 so we recalculate it.
575  SolveQuadraticProblem(F_i[i], g_i.Row(i), opts, &delta_w);
576 
577  delta_w.Scale(step_size);
578  double predicted_impr = VecVec(delta_w, g_i.Row(i)) -
579  0.5 * VecSpVec(delta_w, F_i[i], delta_w);
580 
581  // should never be negative because
582  // we checked inside SolveQuadraticProblem.
583  KALDI_ASSERT(predicted_impr >= -1.0e-05);
584 
585  if (i < 10) {
586  KALDI_LOG << "Predicted objf impr for w (not per frame), iter = " <<
587  (iter) << ", i = " << (i) << " is " << (predicted_impr);
588  }
589  k_predicted_like_impr += predicted_impr;
590  w.Row(i).AddVec(1.0, delta_w);
591  }
592 
593  for (int32 mdl_idx = 0; mdl_idx < num_models; ++mdl_idx) {
594  std::vector< Matrix<double> > gamma(accs[mdl_idx]->GetOccs());
595  for (int32 j = 0; j < models[mdl_idx]->NumPdfs(); j++) {
596  for (int32 m = 0; m < models[mdl_idx]->NumSubstates(j); m++) {
597  w_jm.AddMatVec(1.0, w, kNoTrans,
598  Vector<double>(models[mdl_idx]->v_[j].Row(m)), 0.0);
599  w_jm.Add((-1.0) * w_jm.LogSumExp());
600  k_like_after += VecVec(w_jm, gamma[j].Row(m));
601  }
602  }
603  }
604  KALDI_VLOG(2) << "For iteration " << (iter) << ", updating w gives "
605  << "predicted per-frame like impr "
606  << (k_predicted_like_impr / k_count) << ", actual "
607  << ((k_like_after - k_like_before) / k_count) << ", over "
608  << (k_count) << " frames";
609  if (k_like_after < k_like_before) {
610  w.CopyFromMat(w_orig); // Undo what we computed.
611  if (fabs(k_like_after - k_like_before) / k_count < 1.0e-05) {
612  k_like_after = k_like_before;
613  KALDI_WARN << "Not updating weights as not increasing auxf and "
614  << "probably due to numerical issues (since small change).";
615  break;
616  } else {
617  KALDI_WARN << "Halving step size for weights as likelihood did "
618  << "not increase";
619  }
620  } else {
621  break;
622  }
623  }
624  if (step_size < min_step) {
625  // Undo any step as we have no confidence that this is right.
626  w.CopyFromMat(w_orig);
627  } else {
628  if (iter == 0) {
629  tot_count += k_count;
630  }
631  tot_predicted_like_impr += k_predicted_like_impr;
632  tot_like_after += k_like_after;
633  tot_like_before += k_like_before;
634  }
635  }
636 
637  global_w_.CopyFromMat(w);
638 
639  tot_predicted_like_impr /= tot_count;
640  tot_like_after = (tot_like_after - tot_like_before) / tot_count;
641  KALDI_LOG << "**Overall objf impr for w is " << tot_predicted_like_impr
642  << ", actual " << tot_like_after << ", over "
643  << tot_count << " frames";
644  return tot_like_after;
645 }

647 void MleAmSgmmUpdaterMulti::RenormalizeV(const SpMatrix<double> &H_sm,
648  const vector<AmSgmm*> &models) {
649  int32 phn_dim = PhoneSpaceDim(),
650  feat_dim = FeatureDim(),
651  num_models = models.size();
652  SpMatrix<double> Sigma(phn_dim);
653  int32 count = 0;
654  for (int32 mdl = 0; mdl < num_models; ++mdl) {
655  for (int32 j = 0; j < models[mdl]->NumPdfs(); ++j) {
656  for (int32 m = 0; m < models[mdl]->NumSubstates(j); ++m) {
657  count++;
658  Sigma.AddVec2(static_cast<BaseFloat>(1.0), models[mdl]->v_[j].Row(m));
659  }
660  }
661  }
662  Sigma.Scale(1.0 / count);
663  int32 fixed_eigs = Sigma.LimitCondDouble(update_options_.max_cond);
664  if (fixed_eigs != 0) {
665  KALDI_WARN << "Scatter of vectors v is poorly conditioned. Fixed up "
666  << fixed_eigs << " eigenvalues.";
667  }
668  KALDI_LOG << "Eigenvalues of scatter of vectors v is : ";
669  Sigma.PrintEigs("Sigma");
670  if (!Sigma.IsPosDef()) {
671  KALDI_LOG << "Not renormalizing v because scatter is not positive definite"
672  << " -- maybe first iter?";
673  return;
674  }
675 
676  // Want to make variance of v unit and H_sm (like precision matrix) diagonal.
677  TpMatrix<double> L(phn_dim);
678  L.Cholesky(Sigma);
679  TpMatrix<double> LInv(L);
680  LInv.Invert();
681 
682  Matrix<double> tmpL(phn_dim, phn_dim);
683  tmpL.CopyFromTp(L);
684 
685  SpMatrix<double> H_sm_proj(phn_dim);
686  H_sm_proj.AddMat2Sp(1.0, tmpL, kTrans, H_sm, 0.0);
687  // H_sm_proj := L^{T} * H_sm * L.
688  // This is right because we would transform the vectors themselves
689  // by L^{-1}, and H_sm is like the inverse of the vectors,
690  // so it's {L^{-1}}^{-T} = L^T.
691 
692  Matrix<double> U(phn_dim, phn_dim);
693  Vector<double> eigs(phn_dim);
694  H_sm_proj.SymPosSemiDefEig(&eigs, &U, 1.0); // 1.0 means no checking +ve def -> faster
695  KALDI_LOG << "Note on the next diagnostic: the first number is generally not "
696  << "that meaningful as it relates to the static offset";
697  H_sm_proj.PrintEigs("H_sm_proj (Significance of dims in vector space.. note)");
698 
699  // Transform on vectors is U^T L^{-1}.
700  // Why? Because transform on H_sm is T =U^T L^T
701  // and we want T^{-T} by normal rules of vector/covector and we
702  // have (U^T L^T)^{-T} = (L U)^{-1} = U^T L^{-1}.
703  Matrix<double> Trans(phn_dim, phn_dim); // T^{-T}
704  Matrix<double> tmpLInv(phn_dim, phn_dim);
705  tmpLInv.CopyFromTp(LInv);
706  Trans.AddMatMat(1.0, U, kTrans, tmpLInv, kNoTrans, 0.0);
707  Matrix<double> TransInv(Trans);
708  TransInv.Invert(); // T in above...
709 
710 #ifdef KALDI_PARANOID
711  {
712  SpMatrix<double> H_sm_tmp(phn_dim);
713  H_sm_tmp.AddMat2Sp(1.0, TransInv, kTrans, H_sm, 0.0);
714  KALDI_ASSERT(H_sm_tmp.IsDiagonal(0.1));
715  }
716  {
717  SpMatrix<double> Sigma_tmp(phn_dim);
718  Sigma_tmp.AddMat2Sp(1.0, Trans, kNoTrans, Sigma, 0.0);
719  KALDI_ASSERT(Sigma_tmp.IsUnit(0.1));
720  }
721 #endif
722 
723  for (int32 mdl = 0; mdl < num_models; ++mdl) {
724  for (int32 j = 0; j < models[mdl]->NumPdfs(); ++j) {
725  for (int32 m = 0; m < models[mdl]->NumSubstates(j); ++m) {
726  Vector<double> tmp(phn_dim);
727  tmp.AddMatVec(1.0, Trans, kNoTrans, Vector<double>(models[mdl]->v_[j].Row(m)), 0.0);
728  models[mdl]->v_[j].Row(m).CopyFromVec(tmp);
729  }
730  }
731  }
732  for (int32 i = 0; i < NumGauss(); ++i) {
733  Vector<double> tmp(phn_dim);
734  tmp.AddMatVec(1.0, TransInv, kTrans, Vector<double>(global_w_.Row(i)), 0.0);
735  global_w_.Row(i).CopyFromVec(tmp);
736 
737  Matrix<double> tmpM(feat_dim, phn_dim);
738  // Multiplying on right not left so must not transpose TransInv.
739  tmpM.AddMatMat(1.0, Matrix<double>(global_M_[i]), kNoTrans,
740  TransInv, kNoTrans, 0.0);
741  global_M_[i].CopyFromMat(tmpM);
742  }
743  KALDI_LOG << "Renormalized subspace.";
744 }
