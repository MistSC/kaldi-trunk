47 template<class C>
48 void OnlineGenericBaseFeature<C>::AcceptWaveform(BaseFloat sampling_rate,
49  const VectorBase<BaseFloat> &waveform) {
50  if (waveform.Dim() == 0) {
51  return; // Nothing to do.
52  }
53  if (input_finished_) {
54  KALDI_ERR << "AcceptWaveform called after InputFinished() was called.";
55  }
56  if (sampling_rate != sampling_frequency_) {
57  KALDI_ERR << "Sampling frequency mismatch, expected "
58  << sampling_frequency_ << ", got " << sampling_rate;
59  }
60 
61  Vector<BaseFloat> appended_wave;
62 
63  const VectorBase<BaseFloat> &wave_to_use = (waveform_remainder_.Dim() != 0 ?
64  appended_wave : waveform);
65  if (waveform_remainder_.Dim() != 0) {
66  appended_wave.Resize(waveform_remainder_.Dim() +
67  waveform.Dim());
68  appended_wave.Range(0, waveform_remainder_.Dim()).CopyFromVec(
69  waveform_remainder_);
70  appended_wave.Range(waveform_remainder_.Dim(),
71  waveform.Dim()).CopyFromVec(waveform);
72  }
73  waveform_remainder_.Resize(0);
74 
75  Matrix<BaseFloat> feats;
76  BaseFloat vtln_warp = 1.0; // We don't support VTLN warping in this wrapper.
77  mfcc_or_plp_.Compute(wave_to_use, vtln_warp, &feats, &waveform_remainder_);
78 
79  if (feats.NumRows() == 0) {
80  // Presumably we got a very small waveform and could output no whole
81  // features. The waveform will have been appended to waveform_remainder_.
82  return;
83  }
84  int32 new_num_frames = num_frames_ + feats.NumRows();
85  BaseFloat increase_ratio = 1.5; // This is a tradeoff between memory and
86  // compute; it's the factor by which we
87  // increase the memory used each time.
88  if (new_num_frames > features_.NumRows()) {
89  int32 new_num_rows = std::max<int32>(new_num_frames,
90  features_.NumRows() * increase_ratio);
91  // Increase the size of the features_ matrix and copy over any existing
92  // data.
93  features_.Resize(new_num_rows, Dim(), kCopyData);
94  }
95  features_.Range(num_frames_, feats.NumRows(), 0, Dim()).CopyFromMat(feats);
96  num_frames_ = new_num_frames;
97 }

152 void OnlineCmvn::GetMostRecentCachedFrame(int32 frame,
153  int32 *cached_frame,
154  Matrix<double> *stats) {
155  KALDI_ASSERT(frame >= 0);
156  InitRingBufferIfNeeded();
157  // look for a cached frame on a previous frame as close as possible in time
158  // to "frame". Return if we get one.
159  for (int32 t = frame; t >= 0 && t >= frame - opts_.ring_buffer_size; t--) {
160  if (t % opts_.modulus == 0) {
161  // if this frame should be cached in cached_stats_modulo_, then
162  // we'll look there, and we won't go back any further in time.
163  break;
164  }
165  int32 index = t % opts_.ring_buffer_size;
166  if (cached_stats_ring_[index].first == t) {
167  *cached_frame = t;
168  *stats = cached_stats_ring_[index].second;
169  return;
170  }
171  }
172  int32 n = frame / opts_.modulus;
173  if (n >= cached_stats_modulo_.size()) {
174  if (cached_stats_modulo_.size() == 0) {
175  *cached_frame = -1;
176  stats->Resize(2, this->Dim() + 1);
177  return;
178  } else {
179  n = static_cast<int32>(cached_stats_modulo_.size() - 1);
180  }
181  }
182  *cached_frame = n * opts_.modulus;
183  KALDI_ASSERT(cached_stats_modulo_[n] != NULL);
184  *stats = *(cached_stats_modulo_[n]);
185 }

196 void OnlineCmvn::CacheFrame(int32 frame, const Matrix<double> &stats) {
197  KALDI_ASSERT(frame >= 0);
198  if (frame % opts_.modulus == 0) { // store in cached_stats_modulo_.
199  int32 n = frame / opts_.modulus;
200  if (n >= cached_stats_modulo_.size()) {
201  // The following assert is a limitation on in what order you can call
202  // CacheFrame. Fortunately the calling code always calls it in sequence,
203  // which it has to because you need a previous frame to compute the
204  // current one.
205  KALDI_ASSERT(n == cached_stats_modulo_.size());
206  cached_stats_modulo_.push_back(new Matrix<double>(stats));
207  } else {
208  KALDI_WARN << "Did not expect to reach this part of code.";
209  // do what seems right, but we shouldn't get here.
210  cached_stats_modulo_[n]->CopyFromMat(stats);
211  }
212  } else { // store in the ring buffer.
213  InitRingBufferIfNeeded();
214  if (!cached_stats_ring_.empty()) {
215  int32 index = frame % cached_stats_ring_.size();
216  cached_stats_ring_[index].first = frame;
217  cached_stats_ring_[index].second.CopyFromMat(stats);
218  }
219  }
220 }

228 void OnlineCmvn::ComputeStatsForFrame(int32 frame,
229  MatrixBase<double> *stats_out) {
230  KALDI_ASSERT(frame >= 0 && frame < src_->NumFramesReady());
231 
232  int32 dim = this->Dim(), cur_frame;
233  Matrix<double> stats(2, dim + 1);
234  GetMostRecentCachedFrame(frame, &cur_frame, &stats);
235 
236  Vector<BaseFloat> feats(dim);
237  Vector<double> feats_dbl(dim);
238  while (cur_frame < frame) {
239  cur_frame++;
240  src_->GetFrame(cur_frame, &feats);
241  feats_dbl.CopyFromVec(feats);
242  stats.Row(0).Range(0, dim).AddVec(1.0, feats_dbl);
243  stats.Row(1).Range(0, dim).AddVec2(1.0, feats_dbl);
244  stats(0, dim) += 1.0;
245  // it's a sliding buffer; a frame at the back may be
246  // leaving the buffer so we have to subtract that.
247  int32 prev_frame = cur_frame - opts_.cmn_window;
248  if (prev_frame >= 0) {
249  // we need to subtract frame prev_f from the stats.
250  src_->GetFrame(prev_frame, &feats);
251  feats_dbl.CopyFromVec(feats);
252  stats.Row(0).Range(0, dim).AddVec(-1.0, feats_dbl);
253  stats.Row(1).Range(0, dim).AddVec2(-1.0, feats_dbl);
254  stats(0, dim) -= 1.0;
255  }
256  CacheFrame(cur_frame, stats);
257  }
258  stats_out->CopyFromMat(stats);
259 }

262 // static
263 void OnlineCmvn::SmoothOnlineCmvnStats(const MatrixBase<double> &speaker_stats,
264  const MatrixBase<double> &global_stats,
265  const OnlineCmvnOptions &opts,
266  MatrixBase<double> *stats) {
267  int32 dim = stats->NumCols() - 1;
268  double cur_count = (*stats)(0, dim);
269  // If count exceeded cmn_window it would be an error in how "window_stats"
270  // was accumulated.
271  KALDI_ASSERT(cur_count <= 1.001 * opts.cmn_window);
272  if (cur_count >= opts.cmn_window) return;
273  if (speaker_stats.NumRows() != 0) { // if we have speaker stats..
274  double count_from_speaker = opts.cmn_window - cur_count,
275  speaker_count = speaker_stats(0, dim);
276  if (count_from_speaker > opts.speaker_frames)
277  count_from_speaker = opts.speaker_frames;
278  if (count_from_speaker > speaker_count)
279  count_from_speaker = speaker_count;
280  if (count_from_speaker > 0.0)
281  stats->AddMat(count_from_speaker / speaker_count,
282  speaker_stats);
283  cur_count = (*stats)(0, dim);
284  }
285  if (cur_count >= opts.cmn_window) return;
286  if (global_stats.NumRows() != 0) {
287  double count_from_global = opts.cmn_window - cur_count,
288  global_count = global_stats(0, dim);
289  KALDI_ASSERT(global_count > 0.0);
290  if (count_from_global > opts.global_frames)
291  count_from_global = opts.global_frames;
292  if (count_from_global > 0.0)
293  stats->AddMat(count_from_global / global_count,
294  global_stats);
295  } else {
296  KALDI_ERR << "Global CMN stats are required";
297  }
298 }

300 void OnlineCmvn::GetFrame(int32 frame,
301  VectorBase<BaseFloat> *feat) {
302  src_->GetFrame(frame, feat);
303  KALDI_ASSERT(feat->Dim() == this->Dim());
304  int32 dim = feat->Dim();
305  Matrix<double> stats(2, dim + 1);
306  if (frozen_state_.NumRows() != 0) { // the CMVN state has been frozen.
307  stats.CopyFromMat(frozen_state_);
308  } else {
309  // first get the raw CMVN stats (this involves caching..)
310  this->ComputeStatsForFrame(frame, &stats);
311  // now smooth them.
312  SmoothOnlineCmvnStats(orig_state_.speaker_cmvn_stats,
313  orig_state_.global_cmvn_stats,
314  opts_,
315  &stats);
316  }
317 
318  if (!skip_dims_.empty())
319  FakeStatsForSomeDims(skip_dims_, &stats);
320 
321  // call the function ApplyCmvn declared in ../transform/cmvn.h, which
322  // requires a matrix.
323  Matrix<BaseFloat> feat_mat(1, dim);
324  feat_mat.Row(0).CopyFromVec(*feat);
325  // the function ApplyCmvn takes a matrix, so form a one-row matrix to give it.
326  if (opts_.normalize_mean)
327  ApplyCmvn(stats, opts_.normalize_variance, &feat_mat);
328  else
329  KALDI_ASSERT(!opts_.normalize_variance);
330  feat->CopyFromVec(feat_mat.Row(0));
331 }

383 void OnlineSpliceFrames::GetFrame(int32 frame, VectorBase<BaseFloat> *feat) {
384  KALDI_ASSERT(left_context_ >= 0 && right_context_ >= 0);
385  KALDI_ASSERT(frame >= 0 && frame < NumFramesReady());
386  int32 dim_in = src_->Dim();
387  KALDI_ASSERT(feat->Dim() == dim_in * (1 + left_context_ + right_context_));
388  int32 T = src_->NumFramesReady();
389  for (int32 t2 = frame - left_context_; t2 <= frame + right_context_; t2++) {
390  int32 t2_limited = t2;
391  if (t2_limited < 0) t2_limited = 0;
392  if (t2_limited >= T) t2_limited = T - 1;
393  int32 n = t2 - (frame - left_context_); // 0 for left-most frame,
394  // increases to the right.
395  SubVector<BaseFloat> part(*feat, n * dim_in, dim_in);
396  src_->GetFrame(t2_limited, &part);

441 void OnlineDeltaFeature::GetFrame(int32 frame,
442  VectorBase<BaseFloat> *feat) {
443  KALDI_ASSERT(frame >= 0 && frame < NumFramesReady());
444  KALDI_ASSERT(feat->Dim() == Dim());
445  // We'll produce a temporary matrix containing the features we want to
446  // compute deltas on, but truncated to the necessary context.
447  int32 context = opts_.order * opts_.window;
448  int32 left_frame = frame - context,
449  right_frame = frame + context,
450  src_frames_ready = src_->NumFramesReady();
451  if (left_frame < 0) left_frame = 0;
452  if (right_frame >= src_frames_ready)
453  right_frame = src_frames_ready - 1;
454  KALDI_ASSERT(right_frame >= left_frame);
455  int32 temp_num_frames = right_frame + 1 - left_frame,
456  src_dim = src_->Dim();
457  Matrix<BaseFloat> temp_src(temp_num_frames, src_dim);
458  for (int32 t = left_frame; t <= right_frame; t++) {
459  SubVector<BaseFloat> temp_row(temp_src, t - left_frame);
460  src_->GetFrame(t, &temp_row);
461  }
462  int32 temp_t = frame - left_frame; // temp_t is the offset of frame "frame"
463  // within temp_src
464  delta_features_.Process(temp_src, temp_t, feat);
465 }
