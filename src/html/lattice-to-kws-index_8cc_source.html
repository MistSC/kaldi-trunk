38  const char *usage =
39  "Create an inverted index of the given lattices. The output index is in the T*T*T\n"
40  "semiring. For details for the semiring, please refer to Dogan Can and Muran Saraclar's"
41  "lattice indexing paper."
42  "\n"
43  "Usage: lattice-to-kws-index [options] utter-symtab-rspecifier lattice-rspecifier index-wspecifier\n"
44  " e.g.: lattice-to-kws-index ark:utter.symtab ark:1.lats ark:global.idx\n";

46  ParseOptions po(usage);

49  bool strict = true;
50  bool allow_partial = true;
51  BaseFloat max_states_scale = 4;
52  po.Register("max-silence-frames", &max_silence_frames, "Maximum #frames for"
53  " silence arc.");
54  po.Register("strict", &strict, "Setting --strict=false will cause successful "
55  "termination even if we processed no lattices.");
56  po.Register("max-states-scale", &max_states_scale, "Number of states in the"
57  " original lattice times this scale is the number of states "
58  "allowed when optimizing the index. Negative number means no "
59  "limit on the number of states.");
60  po.Register("allow-partial", &allow_partial, "Allow partial output if fails"
61  " to determinize, otherwise skip determinization if it fails.");
62 
63  po.Read(argc, argv);
64 
65  if (po.NumArgs() < 3 || po.NumArgs() > 4) {
66  po.PrintUsage();
67  exit(1);
68  }

70  std::string usymtab_rspecifier = po.GetOptArg(1),
71  lats_rspecifier = po.GetArg(2),
72  index_wspecifier = po.GetOptArg(3);
73 
74  // We use RandomAccessInt32Reader to read the utterance symtab table.
75  RandomAccessInt32Reader usymtab_reader(usymtab_rspecifier);

77  // We read the lattice in as CompactLattice; We need the CompactLattice
78  // structure for the rest of the work

80  TableWriter< fst::VectorFstTplHolder<KwsLexicographicArc> > index_writer(index_wspecifier);

87  for (; !clat_reader.Done(); clat_reader.Next()) {
88  std::string key = clat_reader.Key();
89  CompactLattice clat = clat_reader.Value();
90  clat_reader.FreeCurrent();
91  KALDI_LOG << "Processing lattice " << key;
92 
93  if (max_states_scale > 0) {
94  max_states = static_cast<int32>(
95  max_states_scale * static_cast<BaseFloat>(clat.NumStates()));
96  }
97 
98  // Check if we have the corresponding utterance id.
99  if (!usymtab_reader.HasKey(key)) {
100  KALDI_WARN << "Cannot find utterance id for " << key;
101  n_fail++;
102  continue;
103  }
104 
105  // Topologically sort the lattice, if not already sorted.
106  uint64 props = clat.Properties(fst::kFstProperties, false);
107  if (!(props & fst::kTopSorted)) {
108  if (fst::TopSort(&clat) == false) {
109  KALDI_WARN << "Cycles detected in lattice " << key;
110  n_fail++;
111  continue;
112  }
113  }
114 
115  // Get the alignments
116  vector<int32> state_times;
117  CompactLatticeStateTimes(clat, &state_times);
118 
119  // Cluster the arcs in the CompactLattice, write the cluster_id on the
120  // output label side.
121  // ClusterLattice() corresponds to the second part of the preprocessing in
122  // Dogan and Murat's paper -- clustering. Note that we do the first part
123  // of preprocessing (the weight pushing step) later when generating the
124  // factor transducer.
125  KALDI_VLOG(1) << "Arc clustering...";
126  bool success = false;
127  success = ClusterLattice(&clat, state_times);
128  if (!success) {
129  KALDI_WARN << "State id's and alignments do not match for lattice " << key;
130  n_fail++;
131  continue;
132  }
133 
134  // The next part is something new, not in the Dogan and Can paper. It is
135  // necessary because we have epsilon arcs, due to silences, in our
136  // lattices. We modify the factor transducer, while maintaining
137  // equivalence, to ensure that states don't have both epsilon *and*
138  // non-epsilon arcs entering them. (and the same, with "entering"
139  // replaced with "leaving"). Later we will find out which states have
140  // non-epsilon arcs leaving/entering them and use it to be more selective
141  // in adding arcs to connect them with the initial/final states. The goal
142  // here is to disallow silences at the beginning or ending of a keyword
143  // occurrence.
144  if (true) {
145  EnsureEpsilonProperty(&clat);
146  fst::TopSort(&clat);
147  // We have to recompute the state times because they will have changed.
148  CompactLatticeStateTimes(clat, &state_times);
149  }
150 
151  // Generate factor transducer
152  // CreateFactorTransducer() corresponds to the "Factor Generation" part of
153  // Dogan and Murat's paper. But we also move the weight pushing step to
154  // this function as we have to compute the alphas and betas anyway.
155  KALDI_VLOG(1) << "Generating factor transducer...";
156  KwsProductFst factor_transducer;
157  int32 utterance_id = usymtab_reader.Value(key);
158  success = CreateFactorTransducer(clat, state_times, utterance_id, &factor_transducer);
159  if (!success) {
160  KALDI_WARN << "Cannot generate factor transducer for lattice " << key;
161  n_fail++;
162  }
163 
164  MaybeDoSanityCheck(factor_transducer);
165 
166  // Remove long silence arc
167  // We add the filtering step in our implementation. This is because gap
168  // between two successive words in a query term should be less than 0.5s
169  KALDI_VLOG(1) << "Removing long silence...";
170  RemoveLongSilences(max_silence_frames, state_times, &factor_transducer);
171 
172  MaybeDoSanityCheck(factor_transducer);
173 
174  // Do factor merging, and return a transducer in T*T*T semiring. This step
175  // corresponds to the "Factor Merging" part in Dogan and Murat's paper.
176  KALDI_VLOG(1) << "Merging factors...";
177  KwsLexicographicFst index_transducer;
178  DoFactorMerging(&factor_transducer, &index_transducer);
179 
180  MaybeDoSanityCheck(index_transducer);
181 
182  // Do factor disambiguation. It corresponds to the "Factor Disambiguation"
183  // step in Dogan and Murat's paper.
184  KALDI_VLOG(1) << "Doing factor disambiguation...";
185  DoFactorDisambiguation(&index_transducer);
186 
187  MaybeDoSanityCheck(index_transducer);
188 
189  // Optimize the above factor transducer. It corresponds to the
190  // "Optimization" step in the paper.
191  KALDI_VLOG(1) << "Optimizing factor transducer...";
192  OptimizeFactorTransducer(&index_transducer, max_states, allow_partial);
193 
194  MaybeDoSanityCheck(index_transducer);
195 
196  // Write result
197  index_writer.Write(key, index_transducer);
198 
199  n_done++;
200  }
201 
202  KALDI_LOG << "Done " << n_done << " lattices, failed for " << n_fail;
203  if (strict == true)
204  return (n_done != 0 ? 0 : 1);
205  else
206  return 0;
207  } catch(const std::exception &e) {
208  std::cerr << e.what();
209  return -1;
210  }
