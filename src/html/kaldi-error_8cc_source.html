47 // Given a filename like "/a/b/c/d/e/f.cc", GetShortFileName
48 // returns "e/f.cc". Does not currently work if backslash is
49 // the filename separator.
50 const char *GetShortFileName(const char *filename) {
51  const char *last_slash = strrchr(filename, '/');
52  if (!last_slash) {
53  return filename;
54  } else {
55  while (last_slash > filename && last_slash[-1] != '/')
56  last_slash--;
57  return last_slash;
58  }
59 }

63 // The function name looks like a macro: it's a macro if we don't have ccxxabi.h
64 inline void KALDI_APPEND_POSSIBLY_DEMANGLED_STRING(const char *to_append,
65  std::string *ans) {
66  // at input the string "to_append" looks like:
67  // ./kaldi-error-test(_ZN5kaldi13UnitTestErrorEv+0xb) [0x804965d]
68  // We want to extract the name e.g. '_ZN5kaldi13UnitTestErrorEv",
69  // demangle it and return it.
70  int32 status;
71  const char *paren = strchr(to_append, '(');
72  const char *plus = (paren ? strchr(paren, '+') : NULL);
73  if (!plus) { // did not find the '(' or did not find the '+'
74  // This is a soft failure in case we did not get what we expected.
75  ans->append(to_append);
76  return;
77  }
78  std::string stripped(paren+1, plus-(paren+1)); // the bit between ( and +.
79 
80  char *demangled_name = abi::__cxa_demangle(stripped.c_str(), 0, 0, &status);
81 
82  // if status != 0 it is an error (demangling failure), but not all names seem
83  // to demangle, so we don't check it.
84 
85  if (demangled_name != NULL) {
86  ans->append(demangled_name);
87  free(demangled_name);
88  } else {
89  ans->append(to_append); // add the original string.
90  }
91 }
92 #else // defined(HAVE_CXXABI_H) && defined(HAVE_EXECINFO_H)
93 #define KALDI_APPEND_POSSIBLY_DEMANGLED_STRING(to_append, ans) \
94  ans->append(to_append)
95 #endif // defined(HAVE_CXXABI_H) && defined(HAVE_EXECINFO_H)

98 std::string KaldiGetStackTrace() {
99 #define KALDI_MAX_TRACE_SIZE 50
100 #define KALDI_MAX_TRACE_PRINT 20 // must be even.
101  std::string ans;
102  void *array[KALDI_MAX_TRACE_SIZE];
103  size_t size = backtrace(array, KALDI_MAX_TRACE_SIZE);
104  char **strings = backtrace_symbols(array, size);
105  if (size <= KALDI_MAX_TRACE_PRINT) {
106  for (size_t i = 0; i < size; i++) {
107  KALDI_APPEND_POSSIBLY_DEMANGLED_STRING(strings[i], &ans);
108  ans += "\n";
109  }
110  } else { // print out first+last (e.g.) 5.
111  for (size_t i = 0; i < KALDI_MAX_TRACE_PRINT/2; i++) {
112  KALDI_APPEND_POSSIBLY_DEMANGLED_STRING(strings[i], &ans);
113  ans += "\n";
114  }
115  ans += ".\n.\n.\n";
116  for (size_t i = size - KALDI_MAX_TRACE_PRINT/2; i < size; i++) {
117  KALDI_APPEND_POSSIBLY_DEMANGLED_STRING(strings[i], &ans);
118  ans += "\n";
119  }
120  if (size == KALDI_MAX_TRACE_SIZE)
121  ans += ".\n.\n.\n"; // stack was too long, probably a bug.
122  }
123  free(strings); // it's all in one big malloc()ed block.
124 
125 
126 #ifdef HAVE_CXXABI_H // demangle the name, if possible.
127 #endif // HAVE_CXXABI_H
128  return ans;
129 }

132 void KaldiAssertFailure_(const char *func, const char *file,
133  int32 line, const char *cond_str) {
134  std::ostringstream ss;
135  ss << "KALDI_ASSERT: at " << GetProgramName() << func << ':'
136  << GetShortFileName(file)
137  << ':' << line << ", failed: " << cond_str << '\n';
138 #ifdef HAVE_EXECINFO_H
139  ss << "Stack trace is:\n" << KaldiGetStackTrace();
140 #endif
141  std::cerr << ss.str();
142  std::cerr.flush();
143  // We used to call abort() here, but switch to throwing an exception
144  // (like KALDI_ERR) because it's easier to deal with in multi-threaded
145  // code.
146  throw std::runtime_error(ss.str());
147 }

190 KaldiErrorMessage::~KaldiErrorMessage() KALDI_NOEXCEPT(false) {
191  std::string str = ss_.str();
192  while (!str.empty() && str[str.length() - 1] == '\n')
193  str.resize(str.length() - 1);
194 
195  // (1) Print the message to stderr.
196  fprintf(stderr, "%s\n", str.c_str());
197  // (2) Throw an exception with the message, plus traceback info if available.
198  if (!std::uncaught_exception()) {
199 #ifdef HAVE_EXECINFO_H
200  throw std::runtime_error(str + "\n\n[stack trace: ]\n" +
201  KaldiGetStackTrace() + "\n");
202 #else
203  throw std::runtime_error(str);
204 #endif
205  } else {
206  abort(); // This may be temporary...
207  }
208 }
