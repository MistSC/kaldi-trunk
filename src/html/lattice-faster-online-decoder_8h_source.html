201  Token *next_tok; // the next token [or NULL if represents final-state]
202  Label ilabel; // ilabel on link.

216  // Token is what's resident in a particular state at a particular time.
217  // In this decoder a Token actually contains *forward* links.
218  // When first created, a Token just has the (total) cost. We add forward
219  // links from it when we process the next frame.
220  struct Token {
221  BaseFloat tot_cost; // would equal weight.Value()... cost up to this point.
222  BaseFloat extra_cost; // >= 0. After calling PruneForwardLinks, this equals
223  // the minimum difference between the cost of the best path, and the cost of
224  // this is on, and the cost of the absolute best path, under the assumption
225  // that any of the currently active states at the decoding front may
226  // eventually succeed (e.g. if you were to take the currently active states
227  // one by one and compute this difference, and then take the minimum).
228 
229  ForwardLink *links; // Head of singly linked list of ForwardLinks
230 
231  Token *next; // Next in list of tokens for this frame.
232 
233  Token *backpointer; // best preceding Token (could be on this frame or a
234  // previous frame). This is only required for an
235  // efficient GetBestPath function, it plays no part in
236  // the lattice generation (the "links" list is what
237  // stores the forward links, for that).
238 
239  inline Token(BaseFloat tot_cost, BaseFloat extra_cost, ForwardLink *links,
240  Token *next, Token *backpointer):
241  tot_cost(tot_cost), extra_cost(extra_cost), links(links), next(next),
242  backpointer(backpointer) { }
243  inline void DeleteForwardLinks() {
244  ForwardLink *l = links, *m;
245  while (l != NULL) {
246  m = l->next;
247  delete l;
248  l = m;
249  }
250  links = NULL;
251  }
252  };

293  // This function computes the final-costs for tokens active on the final
294  // frame. It outputs to final-costs, if non-NULL, a map from the Token*
295  // pointer to the final-prob of the corresponding state, for all Tokens
296  // that correspond to states that have final-probs. This map will be
297  // empty if there were no final-probs. It outputs to
298  // final_relative_cost, if non-NULL, the difference between the best
299  // forward-cost including the final-prob cost, and the best forward-cost
300  // without including the final-prob cost (this will usually be positive), or
301  // infinity if there were no final-probs. [c.f. FinalRelativeCost(), which
302  // outputs this quanitity]. It outputs to final_best_cost, if
303  // non-NULL, the lowest for any token t active on the final frame, of
304  // forward-cost[t] + final-cost[t], where final-cost[t] is the final-cost in
305  // the graph of the state corresponding to token t, or the best of
306  // forward-cost[t] if there were no final-probs active on the final frame.
307  // You cannot call this after FinalizeDecoding() has been called; in that
308  // case you should get the answer from class-member variables.
309  void ComputeFinalCosts(unordered_map<Token*, BaseFloat> *final_costs,
