28 inline CuSubMatrix<Real>::CuSubMatrix(const CuMatrixBase<Real> &mat,
29  const MatrixIndexT row_offset,
30  const MatrixIndexT num_rows,
31  const MatrixIndexT col_offset,
32  const MatrixIndexT num_cols) {
33  if (num_rows == 0 || num_cols == 0) {
34  KALDI_ASSERT(num_rows == 0 && num_cols == 0);
35  // Everything will have been set to zero in CuMastrixBase's default
36  // initializer, so nothing to do.
37  } else {
38  KALDI_ASSERT(row_offset >= 0 && col_offset >= 0 &&
39  row_offset + num_rows <= mat.num_rows_ &&
40  col_offset + num_cols <= mat.num_cols_);
41  this->data_ = mat.data_ + (row_offset * mat.stride_) + col_offset;
42  this->num_cols_ = num_cols;
43  this->num_rows_ = num_rows;
44  this->stride_ = mat.stride_;
45  }
46 }

66 inline CuSubMatrix<Real>::CuSubMatrix(const CuMatrixBase<Real> &mat,
67  const MatrixIndexT batch_index,
68  const MatrixIndexT num_rows,
69  const MatrixIndexT num_cols)
70 {
71  if (num_rows == 0 || num_cols == 0)
72  {
73  KALDI_ASSERT(num_rows == 0 && num_cols == 0);
74  // Everything will have been set to zero in CuMastrixBase's default
75  // initializer, so nothing to do.
76  }
77  else
78  {
79  KALDI_ASSERT(num_cols > 0 && num_rows > 0 &&
80  batch_index >= 0 &&
81  batch_index <= mat.num_rows_ &&
82  num_cols * num_rows == mat.num_cols_ &&
83  num_cols * num_rows <= mat.stride_);
84  this->data_ = mat.data_ + batch_index * mat.stride_;
85  this->num_cols_ = num_cols;
86  this->num_rows_ = num_rows;
87  // stride: memory distance between each rows
88  // stride >= cols
89  this->stride_ = num_cols;
90  }
91 }
