60 void NnetComputeProb::Compute(const NnetExample &eg) {
61  bool need_model_derivative = config_.compute_deriv,
62  store_component_stats = false;
63  ComputationRequest request;
64  GetComputationRequest(nnet_, eg, need_model_derivative,
65  store_component_stats,
66  &request);
67  const NnetComputation *computation = compiler_.Compile(request);
68  NnetComputer computer(config_.compute_config, *computation,
69  nnet_, deriv_nnet_);
70  // give the inputs to the computer object.
71  computer.AcceptInputs(nnet_, eg.io);
72  computer.Forward();
73  this->ProcessOutputs(eg, &computer);
74  if (config_.compute_deriv)
75  computer.Backward();
76 }

78 void NnetComputeProb::ProcessOutputs(const NnetExample &eg,
79  NnetComputer *computer) {
80  std::vector<NnetIo>::const_iterator iter = eg.io.begin(),
81  end = eg.io.end();
82  for (; iter != end; ++iter) {
83  const NnetIo &io = *iter;
84  int32 node_index = nnet_.GetNodeIndex(io.name);
85  if (node_index < 0)
86  KALDI_ERR << "Network has no output named " << io.name;
87  ObjectiveType obj_type = nnet_.GetNode(node_index).u.objective_type;
88  if (nnet_.IsOutputNode(node_index)) {
89  const CuMatrixBase<BaseFloat> &output = computer->GetOutput(io.name);
90  if (output.NumCols() != io.features.NumCols()) {
91  KALDI_ERR << "Nnet versus example output dimension (num-classes) "
92  << "mismatch for '" << io.name << "': " << output.NumCols()
93  << " (nnet) vs. " << io.features.NumCols() << " (egs)\n";
94  }
95  {
96  BaseFloat tot_weight, tot_objf;
97  bool supply_deriv = config_.compute_deriv;
98  ComputeObjectiveFunction(io.features, obj_type, io.name,
99  supply_deriv, computer,
100  &tot_weight, &tot_objf);
101  SimpleObjectiveInfo &totals = objf_info_[io.name];
102  totals.tot_weight += tot_weight;
103  totals.tot_objective += tot_objf;
104  }
105  if (obj_type == kLinear && config_.compute_accuracy) {
106  BaseFloat tot_weight, tot_accuracy;
107  ComputeAccuracy(io.features, output,
108  &tot_weight, &tot_accuracy);
109  SimpleObjectiveInfo &totals = accuracy_info_[io.name];
110  totals.tot_weight += tot_weight;
111  totals.tot_objective += tot_accuracy;
112  }
113  num_minibatches_processed_++;
114  }
115  }
116 }

118 bool NnetComputeProb::PrintTotalStats() const {
119  bool ans = false;
120  unordered_map<std::string, SimpleObjectiveInfo, StringHasher>::const_iterator
121  iter, end;
122  { // First print regular objectives
123  iter = objf_info_.begin();
124  end = objf_info_.end();
125  for (; iter != end; ++iter) {
126  const std::string &name = iter->first;
127  int32 node_index = nnet_.GetNodeIndex(name);
128  KALDI_ASSERT(node_index >= 0);
129  ObjectiveType obj_type = nnet_.GetNode(node_index).u.objective_type;
130  const SimpleObjectiveInfo &info = iter->second;
131  KALDI_LOG << "Overall "
132  << (obj_type == kLinear ? "log-likelihood" : "objective")
133  << " for '" << name << "' is "
134  << (info.tot_objective / info.tot_weight) << " per frame"
135  << ", over " << info.tot_weight << " frames.";
136  if (info.tot_weight > 0)
137  ans = true;
138  }
139  }
140  { // now print accuracies.
141  iter = accuracy_info_.begin();
142  end = accuracy_info_.end();
143  for (; iter != end; ++iter) {
144  const std::string &name = iter->first;
145  const SimpleObjectiveInfo &info = iter->second;
146  KALDI_LOG << "Overall accuracy for '" << name << "' is "
147  << (info.tot_objective / info.tot_weight) << " per frame"
148  << ", over " << info.tot_weight << " frames.";
149  // don't bother changing ans; the loop over the regular objective should
150  // already have set it to true if we got any data.
151  }
152  }
153  return ans;
154 }

156 void ComputeAccuracy(const GeneralMatrix &supervision,
157  const CuMatrixBase<BaseFloat> &nnet_output,
158  BaseFloat *tot_weight_out,
159  BaseFloat *tot_accuracy_out) {
160  int32 num_rows = nnet_output.NumRows(),
161  num_cols = nnet_output.NumCols();
162  KALDI_ASSERT(supervision.NumRows() == num_rows &&
163  supervision.NumCols() == num_cols);
164 
165  CuArray<int32> best_index(num_rows);
166  nnet_output.FindRowMaxId(&best_index);
167  std::vector<int32> best_index_cpu;
168  // wasteful copy, but doesn't dominate.
169  best_index.CopyToVec(&best_index_cpu);
170 
171 
172  double tot_weight = 0.0,
173  tot_accuracy = 0.0;
174 
175  // note: we expect that in most cases where this code is called,
176  // supervision.Type() will be kSparseMatrix.
177  switch (supervision.Type()) {
178  case kCompressedMatrix: {
179  Matrix<BaseFloat> mat;
180  supervision.GetMatrix(&mat);
181  for (int32 r = 0; r < num_rows; r++) {
182  SubVector<BaseFloat> vec(mat, r);
183  BaseFloat row_sum = vec.Sum();
184  KALDI_ASSERT(row_sum >= 0.0);
185  int32 best_index;
186  vec.Max(&best_index); // discard max value.
187  tot_weight += row_sum;
188  if (best_index == best_index_cpu[r])
189  tot_accuracy += row_sum;
190  }
191  break;
192 
193  }
194  case kFullMatrix: {
195  const Matrix<BaseFloat> &mat = supervision.GetFullMatrix();
196  for (int32 r = 0; r < num_rows; r++) {
197  SubVector<BaseFloat> vec(mat, r);
198  BaseFloat row_sum = vec.Sum();
199  KALDI_ASSERT(row_sum >= 0.0);
200  int32 best_index;
201  vec.Max(&best_index); // discard max value.
202  tot_weight += row_sum;
203  if (best_index == best_index_cpu[r])
204  tot_accuracy += row_sum;
205  }
206  break;
207  }
208  case kSparseMatrix: {
209  const SparseMatrix<BaseFloat> &smat = supervision.GetSparseMatrix();
210  for (int32 r = 0; r < num_rows; r++) {
211  const SparseVector<BaseFloat> &row = smat.Row(r);
212  BaseFloat row_sum = row.Sum();
213  int32 best_index;
214  row.Max(&best_index);
215  KALDI_ASSERT(best_index < num_cols);
216  tot_weight += row_sum;
217  if (best_index == best_index_cpu[r])
218  tot_accuracy += row_sum;
219  }
220  break;
221  }
222  default: KALDI_ERR << "Bad general-matrix type.";
223  }
224  *tot_weight_out = tot_weight;
225  *tot_accuracy_out = tot_accuracy;
226 }
