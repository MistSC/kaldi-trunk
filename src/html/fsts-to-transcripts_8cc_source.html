36  "output-symbol sequence (if --output-side=true), or input-symbol sequence "
37  "otherwise.\n"
38  "\n"
39  "Usage: fsts-to-transcripts [options] <fsts-rspecifier> <transcriptions-wspecifier>\n"

46  po.Register("output-side", &output_side, "If true, extract the symbols on "
47  "the output side of the FSTs, else the input side.");
48 
49  po.Read(argc, argv);
50 
51  if (po.NumArgs() < 2 || po.NumArgs() > 3) {
52  po.PrintUsage();
53  exit(1);
54  }
55 
56  std::string fst_rspecifier = po.GetArg(1),
57  transcript_wspecifier = po.GetOptArg(2);
58 
59 
60  SequentialTableReader<VectorFstHolder> fst_reader(fst_rspecifier);
61  Int32VectorWriter transcript_writer(transcript_wspecifier);
62 
63  int32 n_done = 0, n_err = 0;
64  for (; !fst_reader.Done(); fst_reader.Next()) {
65  std::string key = fst_reader.Key();
66  const VectorFst<StdArc> &fst = fst_reader.Value();
67 
68 
69  VectorFst<StdArc> shortest_path;
70  ShortestPath(fst, &shortest_path); // the OpenFst algorithm ShortestPath.
71 
72  if (shortest_path.NumStates() == 0) {
73  KALDI_WARN << "Input FST (after shortest path) was empty. Producing no "
74  << "output for key " << key;
75  n_err++;
76  continue;
77  }
78 
79  std::vector<int32> transcript;
80  bool ans;
81  if (output_side) ans = fst::GetLinearSymbolSequence<StdArc, int32>(
82  shortest_path, NULL, &transcript, NULL);
83  else ans = fst::GetLinearSymbolSequence<StdArc, int32>(
84  shortest_path, &transcript, NULL, NULL);
85  if (!ans) {
86  KALDI_ERR << "GetLinearSymbolSequence returned false (code error);";
87  }
88  transcript_writer.Write(key, transcript);
89  n_done++;
90  }
91 
92  KALDI_LOG << "Converted " << n_done << " FSTs, " << n_err << " with errors";
93  return (n_done != 0 ? 0 : 1);
94  } catch(const std::exception &e) {
95  std::cerr << e.what();
96  return -1;
97  }
98 }
