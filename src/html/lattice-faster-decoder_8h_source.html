208  // ForwardLinks are the links from a token to a token on the next frame.
209  // or sometimes on the current frame (for input-epsilon links).

211  struct ForwardLink {
212  Token *next_tok; // the next token [or NULL if represents final-state]
213  Label ilabel; // ilabel on link.
214  Label olabel; // olabel on link.
215  BaseFloat graph_cost; // graph cost of traversing link (contains LM, etc.)
216  BaseFloat acoustic_cost; // acoustic cost (pre-scaled) of traversing link
217  ForwardLink *next; // next in singly-linked list of forward links from a
218  // token.
219  inline ForwardLink(Token *next_tok, Label ilabel, Label olabel,
220  BaseFloat graph_cost, BaseFloat acoustic_cost,
221  ForwardLink *next):
222  next_tok(next_tok), ilabel(ilabel), olabel(olabel),
223  graph_cost(graph_cost), acoustic_cost(acoustic_cost),
224  next(next) { }
225  };

227  // Token is what's resident in a particular state at a particular time.
228  // In this decoder a Token actually contains *forward* links.
229  // When first created, a Token just has the (total) cost. We add forward
230  // links from it when we process the next frame.
231  struct Token {
232  BaseFloat tot_cost; // would equal weight.Value()... cost up to this point.
233  BaseFloat extra_cost; // >= 0. After calling PruneForwardLinks, this equals
234  // the minimum difference between the cost of the best path, and the cost of
235  // this is on, and the cost of the absolute best path, under the assumption
236  // that any of the currently active states at the decoding front may
237  // eventually succeed (e.g. if you were to take the currently active states
238  // one by one and compute this difference, and then take the minimum).
239 
240  ForwardLink *links; // Head of singly linked list of ForwardLinks
241 
242  Token *next; // Next in list of tokens for this frame.
243 
244  inline Token(BaseFloat tot_cost, BaseFloat extra_cost, ForwardLink *links,
245  Token *next):
246  tot_cost(tot_cost), extra_cost(extra_cost), links(links), next(next) { }
247  inline void DeleteForwardLinks() {
248  ForwardLink *l = links, *m;
249  while (l != NULL) {
250  m = l->next;
251  delete l;
252  l = m;
253  }
254  links = NULL;
255  }
256  };

296  // This function computes the final-costs for tokens active on the final
297  // frame. It outputs to final-costs, if non-NULL, a map from the Token*
298  // pointer to the final-prob of the corresponding state, for all Tokens
299  // that correspond to states that have final-probs. This map will be
300  // empty if there were no final-probs. It outputs to
301  // final_relative_cost, if non-NULL, the difference between the best
302  // forward-cost including the final-prob cost, and the best forward-cost
303  // without including the final-prob cost (this will usually be positive), or
304  // infinity if there were no final-probs. [c.f. FinalRelativeCost(), which
305  // outputs this quanitity]. It outputs to final_best_cost, if
306  // non-NULL, the lowest for any token t active on the final frame, of
307  // forward-cost[t] + final-cost[t], where final-cost[t] is the final-cost in
308  // the graph of the state corresponding to token t, or the best of
309  // forward-cost[t] if there were no final-probs active on the final frame.
310  // You cannot call this after FinalizeDecoding() has been called; in that
311  // case you should get the answer from class-member variables.
312  void ComputeFinalCosts(unordered_map<Token*, BaseFloat> *final_costs,
