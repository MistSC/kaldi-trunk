27 /*
28  This class is responsible for computing a Fisher matrix which is a kind of
29  scatter of gradients on subsets; it's used for preconditioning the update in
30  class FastNnetCombiner. */
31 class FisherComputationClass: public MultiThreadable {
32  public:
33  FisherComputationClass(const Nnet &nnet,
34  const std::vector<Nnet> &nnets,
35  const std::vector<NnetExample> &egs,
36  int32 minibatch_size,
37  SpMatrix<double> *scatter):
38  nnet_(nnet), nnets_(nnets), egs_(egs), minibatch_size_(minibatch_size),
39  scatter_ptr_(scatter) { } // This initializer is only used to create a
40  // temporary version of the object; the next initializer is used to
41  // create the separate versions for the parallel jobs.
42 
43  FisherComputationClass(const FisherComputationClass &other):
44  nnet_(other.nnet_), nnets_(other.nnets_), egs_(other.egs_),
45  minibatch_size_(other.minibatch_size_), scatter_ptr_(other.scatter_ptr_) {
46  scatter_.Resize(nnets_.size() * nnet_.NumUpdatableComponents()); }
47 
48  void operator () () {
49  // b is the "minibatch id."
50  int32 num_egs = static_cast<int32>(egs_.size());
51  Nnet nnet_gradient(nnet_);
52  for (int32 b = 0; b * minibatch_size_ < num_egs; b++) {
53  if (b % num_threads_ != thread_id_)
54  continue; // We're not responsible for this minibatch.
55  int32 offset = b * minibatch_size_,
56  length = std::min(minibatch_size_,
57  num_egs - offset);
58  bool is_gradient = true;
59  nnet_gradient.SetZero(is_gradient);
60  std::vector<NnetExample> minibatch(egs_.begin() + offset,
61  egs_.begin() + offset + length);
62  DoBackprop(nnet_, minibatch, &nnet_gradient);
63  Vector<double> gradient(nnets_.size() * nnet_.NumUpdatableComponents());
64  int32 i = 0;
65  for (int32 n = 0; n < static_cast<int32>(nnets_.size()); n++) {
66  for (int32 c = 0; c < nnet_.NumComponents(); c++) {
67  const UpdatableComponent *uc = dynamic_cast<const UpdatableComponent*>(
68  &(nnet_gradient.GetComponent(c))),
69  *uc_other = dynamic_cast<const UpdatableComponent*>(
70  &(nnets_[n].GetComponent(c)));
71  if (uc != NULL) {
72  gradient(i) = uc->DotProduct(*uc_other);
73  i++;
74  }
75  }
76  }
77  KALDI_ASSERT(i == gradient.Dim());
78  scatter_.AddVec2(1.0, gradient);
79  }
80  }
81  ~FisherComputationClass() {
82  if (scatter_.NumRows() != 0) {
83  if (scatter_ptr_->NumRows() == 0)
84  scatter_ptr_->Resize(scatter_.NumRows());
85  scatter_ptr_->AddSp(1.0, scatter_);
86  }
87  }
88 
89  private:
90  const Nnet &nnet_; // point at which we compute the parameter gradients.
91  const std::vector<Nnet> &nnets_; // The dot-product of each of these with the parameter gradients,
92  // are the actual gradients that go into "scatter".
93  const std::vector<NnetExample> &egs_;
94  int32 minibatch_size_; // equals config --fisher-minbatch-size e.g. 64 (smaller than
95  // regular minibatch size.)
96  SpMatrix<double> *scatter_ptr_;
97  SpMatrix<double> scatter_; // Local accumulation of the scatter.
98 };

149  << regularizer_objf << " = " << objf;

201  const std::vector<Nnet> &nnets,

203  int32 num_nnets = nnets.size();
204  KALDI_ASSERT(num_nnets >= 1);

207 
208 
209  *dest = nnets[0];
210  SubVector<double> scale_params0(scale_params, 0, num_uc);
211  dest->ScaleComponents(Vector<BaseFloat>(scale_params0));
212  for (int32 n = 1; n < num_nnets; n++) {
213  SubVector<double> scale_params_n(scale_params, n * num_uc, num_uc);
214  dest->AddNnet(Vector<BaseFloat>(scale_params_n), nnets[n]);
215  }

219 void FastNnetCombiner::ComputePreconditioner() {
220  SpMatrix<double> F; // Fisher matrix.
221  Nnet nnet;
222  ComputeCurrentNnet(&nnet); // will be at initial value of neural net.
223 
224  { // This block does the multi-threaded computation.
225  // The next line just initializes an "example" object.
226  FisherComputationClass fc(nnet, nnets_, egs_,
227  config_.fisher_minibatch_size,
228  &F);
229 
230  // Setting num_threads to zero if config_.num_threads == 1
231  // is a signal to the MultiThreader class to run without creating
232  // any extra threads in this case; it helps support GPUs.
233  int32 num_threads = config_.num_threads == 1 ? 0 : config_.num_threads;
234  // The work gets done in the initializer and destructor of
235  // the class below.
236  MultiThreader<FisherComputationClass> m(num_threads, fc);
237  }
238 
239  // The scale of F is irrelevant but it might be quite
240  // large at this point, so we just normalize it.
241  KALDI_ASSERT(F.Trace() > 0);
242  F.Scale(F.NumRows() / F.Trace()); // same scale as unit matrix.
243  // Make zero diagonal elements of F non-zero. Relates to updatable
244  // components that have no effect, e.g. MixtureProbComponents that have
245  // no real free parameters.
246  KALDI_ASSERT(config_.fisher_floor > 0.0);
247  for (int32 i = 0; i < F.NumRows(); i++)
248  F(i, i) = std::max<BaseFloat>(F(i, i), config_.fisher_floor);
249  // We next smooth the diagonal elements of F by a small amount.
250  // This is mainly necessary in case the number of minibatches is
251  // smaller than the dimension of F; we want to ensure F is full rank.
252  for (int32 i = 0; i < F.NumRows(); i++)
253  F(i, i) *= (1.0 + config_.alpha);
254 
255  C_.Resize(F.NumRows());
256  C_.Cholesky(F);
257  C_inv_ = C_;
258  C_inv_.Invert();
259 
260  // Transform the params_ data-member to be in the preconditioned space.
261  Vector<double> raw_params(params_);
262  params_.AddTpVec(1.0, C_, kTrans, raw_params, 0.0);
263 }

267 void FastNnetCombiner::GetInitialParams() {
268  int32 initial_model = config_.initial_model,
269  num_nnets = static_cast<int32>(nnets_.size());
270  if (initial_model > num_nnets)
271  initial_model = num_nnets;
272  if (initial_model < 0)
273  initial_model = GetInitialModel(egs_, nnets_);
274 
275  KALDI_ASSERT(initial_model >= 0 && initial_model <= num_nnets);
276  int32 num_uc = nnets_[0].NumUpdatableComponents();
277 
278  Vector<double> raw_params(num_uc * num_nnets); // parameters in
279  // non-preconditioned space.
280  if (initial_model < num_nnets) {
281  KALDI_LOG << "Initializing with neural net with index " << initial_model;
282  // At this point we're using the best of the individual neural nets.
283  raw_params.Set(0.0);
284 
285  // Set the block of parameters corresponding to the "best" of the
286  // source neural nets to
287  SubVector<double> best_block(raw_params, num_uc * initial_model, num_uc);
288  best_block.Set(1.0);
289  } else { // initial_model == num_nnets
290  KALDI_LOG << "Initializing with all neural nets averaged.";
291  raw_params.Set(1.0 / num_nnets);
292  }
293  KALDI_ASSERT(C_.NumRows() == 0); // Assume this not set up yet.
294  params_ = raw_params; // this is in non-preconditioned space.
295 }

298 double FastNnetCombiner::ComputeObjfAndGradient(
299  Vector<double> *gradient,
300  double *regularizer_objf_ptr) {
301  Nnet nnet;
302  ComputeCurrentNnet(&nnet); // compute it at the value "params_".
303 
304  Nnet nnet_gradient(nnet);
305  bool is_gradient = true;
306  nnet_gradient.SetZero(is_gradient);
307  double tot_weight = 0.0;
308  double objf = DoBackpropParallel(nnet, config_.minibatch_size, config_.num_threads,
309  egs_, &tot_weight, &nnet_gradient) / egs_.size();
310 
311  // raw_gradient is gradient in non-preconditioned space.
312  Vector<double> raw_gradient(params_.Dim());
313 
314  double regularizer_objf = 0.0; // sum of -0.5 * config_.regularizer * params-squared.
315  int32 i = 0; // index into raw_gradient
316  int32 num_nnets = nnets_.size();
317  for (int32 n = 0; n < num_nnets; n++) {
318  for (int32 j = 0; j < nnet.NumComponents(); j++) {
319  const UpdatableComponent *uc =
320  dynamic_cast<const UpdatableComponent*>(&(nnets_[n].GetComponent(j))),
321  *uc_gradient =
322  dynamic_cast<const UpdatableComponent*>(&(nnet_gradient.GetComponent(j))),
323  *uc_params =
324  dynamic_cast<const UpdatableComponent*>(&(nnet.GetComponent(j)));
325  if (uc != NULL) {
326  double gradient = uc->DotProduct(*uc_gradient) / tot_weight;
327  // "gradient" is the derivative of the objective function w.r.t. this
328  // element of the parameters (i.e. this weight, which gets applied to
329  // the j'th component of the n'th source neural net).
330  if (config_.regularizer != 0.0) {
331  gradient -= config_.regularizer * uc->DotProduct(*uc_params);
332  if (n == 0) // only add this once...
333  regularizer_objf +=
334  -0.5 * config_.regularizer * uc_params->DotProduct(*uc_params);
335  }
336  raw_gradient(i) = gradient;
337  i++;
338  }
339  }
340  }
341  if (config_.regularizer != 0.0) {
342  KALDI_VLOG(2) << "Objf is " << objf << " + regularizer " << regularizer_objf
343  << " = " << (objf + regularizer_objf) << ", raw gradient is "
344  << raw_gradient;
345  } else {
346  KALDI_VLOG(2) << "Objf is " << objf << ", raw gradient is " << raw_gradient;
347  }
348  KALDI_ASSERT(i == raw_gradient.Dim());
349  // \hat{g} = C^{-1} g.
350  gradient->AddTpVec(1.0, C_inv_, kNoTrans, raw_gradient, 0.0);
351  *regularizer_objf_ptr = regularizer_objf;
352  return objf + regularizer_objf;
353 }

355 void FastNnetCombiner::ComputeCurrentNnet(
356  Nnet *dest, bool debug) {
357  int32 num_nnets = nnets_.size();
358  KALDI_ASSERT(num_nnets >= 1);
359  KALDI_ASSERT(params_.Dim() == num_nnets * nnets_[0].NumUpdatableComponents());
360  Vector<double> raw_params(params_.Dim()); // Weights in non-preconditioned space:
361  // p = C^{-T} \hat{p}. Here, raw_params is p, params_, is \hat{p}.
362 
363  if (C_inv_.NumRows() > 0)
364  raw_params.AddTpVec(1.0, C_inv_, kTrans, params_, 0.0);
365  else
366  raw_params = params_; // C not set up yet: interpret params_ as raw parameters.
367 
368  if (debug) {
369  Matrix<double> params_mat(num_nnets,
370  nnets_[0].NumUpdatableComponents());
371  params_mat.CopyRowsFromVec(raw_params);
372  KALDI_LOG << "Scale parameters are " << params_mat;
373  }
374  CombineNnets(raw_params, nnets_, dest);
375 }

380 int32 FastNnetCombiner::GetInitialModel(
381  const std::vector<NnetExample> &validation_set,
382  const std::vector<Nnet> &nnets) const {
383  int32 num_nnets = static_cast<int32>(nnets.size());
384  KALDI_ASSERT(!nnets.empty());
385  int32 best_n = -1;
386  double best_objf;
387  Vector<double> objfs(nnets.size());
388  for (int32 n = 0; n < num_nnets; n++) {
389  double num_frames;
390  double objf = ComputeNnetObjfParallel(nnets[n], config_.minibatch_size,
391  config_.num_threads, validation_set,
392  &num_frames);
393  KALDI_ASSERT(num_frames != 0);
394  objf /= num_frames;
395 
396  if (n == 0 || objf > best_objf) {
397  best_objf = objf;
398  best_n = n;
399  }
400  objfs(n) = objf;
401  }
402  KALDI_LOG << "Objective functions for the source neural nets are " << objfs;
403 
404  int32 num_uc = nnets[0].NumUpdatableComponents();
405 
406  if (num_nnets > 1) { // Now try a version where all the neural nets have the
407  // same weight. Don't do this if num_nnets == 1 as
408  // it would be a waste of time (identical to n == 0).
409  Vector<double> scale_params(num_uc * num_nnets);
410  scale_params.Set(1.0 / num_nnets);
411  Nnet average_nnet;
412  CombineNnets(scale_params, nnets, &average_nnet);
413  double num_frames;
414  double objf = ComputeNnetObjfParallel(average_nnet, config_.minibatch_size,
415  config_.num_threads, validation_set,
416  &num_frames);
417  objf /= num_frames;
418  KALDI_LOG << "Objf with all neural nets averaged is " << objf;
419  if (objf > best_objf) {
420  return num_nnets;
421  } else {
422  return best_n;
423  }
424  } else {
425  return best_n;
426  }
427 }
