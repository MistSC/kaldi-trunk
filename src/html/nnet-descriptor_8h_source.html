86 // A ForwardingDescriptor describes how we copy data from another NetworkNode,
87 // or from multiple other NetworkNodes. In the simplest case this can just be
88 // equivalent to giving the name of another NetworkNode, but we also support
89 // things like time-offsets, selecting depending on the index from multiple
90 // different inputs, and things like that.
91 //
92 // note: nodes of type kOutput (i.e. output nodes of the network) cannot appear
93 // as inputs in any descriptor. This is to simplify compilation.
94 class ForwardingDescriptor {
95  public:
96  // Given an Index that's requested at the output of this descriptor, maps it
97  // to a (node_index, Index) pair that says where we are to get the data from.
98  virtual Cindex MapToInput(const Index &output) const = 0;
99 
100  // Return the feature dimension.
101  virtual int32 Dim(const Nnet &nnet) const = 0;
102 
103  virtual ForwardingDescriptor *Copy() const = 0;
104 
111  virtual int32 Modulus() const { return 1; }
112 
113  // Write to string that will be one line of a config-file-like format. The
114  // opposite of Parse.
115  virtual void WriteConfig(std::ostream &os,
116  const std::vector<std::string> &node_names) const = 0;
117 
119  // that this descriptor may access.
120  virtual void GetNodeDependencies(std::vector<int32> *node_indexes) const = 0;
121 
122  virtual ~ForwardingDescriptor() { }
123  ForwardingDescriptor() { }

435  // input. Assumes the input has already been tokenized into an array of
436  // strings by DescriptorTokenize(); it moves the begin-pointer "next_token" to
437  // account for token that it consumes. Prints warning and returns false on
438  // error (including if there was junk after the last token). The input tokens
