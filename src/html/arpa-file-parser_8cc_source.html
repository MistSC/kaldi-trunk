39 void ArpaFileParser::Read(std::istream &is, bool binary) {
40  if (binary) {
41  KALDI_ERR << "binary-mode reading is not implemented for ArpaFileParser";
42  }
43 
44  // Argument sanity checks.
45  if (options_.bos_symbol <= 0 || options_.eos_symbol <= 0 ||
46  options_.bos_symbol == options_.eos_symbol)
47  KALDI_ERR << "BOS and EOS symbols are required, must not be epsilons, and "
48  << "differ from each other. Given:"
49  << " BOS=" << options_.bos_symbol
50  << " EOS=" << options_.eos_symbol;
51  if (symbols_ != NULL &&
52  options_.oov_handling == ArpaParseOptions::kReplaceWithUnk &&
53  (options_.unk_symbol <= 0 ||
54  options_.unk_symbol == options_.bos_symbol ||
55  options_.unk_symbol == options_.eos_symbol))
56  KALDI_ERR << "When symbol table is given and OOV mode is kReplaceWithUnk, "
57  << "UNK symbol is required, must not be epsilon, and "
58  << "differ from both BOS and EOS symbols. Given:"
59  << " UNK=" << options_.unk_symbol
60  << " BOS=" << options_.bos_symbol
61  << " EOS=" << options_.eos_symbol;
62  if (symbols_ != NULL && symbols_->Find(options_.bos_symbol).empty())
63  KALDI_ERR << "BOS symbol must exist in symbol table";
64  if (symbols_ != NULL && symbols_->Find(options_.eos_symbol).empty())
65  KALDI_ERR << "EOS symbol must exist in symbol table";
66  if (symbols_ != NULL && options_.unk_symbol > 0 &&
67  symbols_->Find(options_.unk_symbol).empty())
68  KALDI_ERR << "UNK symbol must exist in symbol table";
69 
70  ngram_counts_.clear();
71  line_number_ = 0;
72 
73 #define PARSE_ERR (KALDI_ERR << "in line " << line_number_ << ": ")
74 
75  // Give derived class an opportunity to prepare its state.
76  ReadStarted();
77 
78  std::string line;
79 
80  // Processes "\data\" section.
81  bool keyword_found = false;
82  while (++line_number_, getline(is, line) && !is.eof()) {
83  if (line.empty()) continue;
84 
85  // The section keywords starts with backslash. We terminate the while loop
86  // if a new section is found.
87  if (line[0] == '\\') {
88  if (!keyword_found && line == "\\data\\") {
89  KALDI_LOG << "Reading \\data\\ section.";
90  keyword_found = true;
91  continue;
92  }
93  break;
94  }
95 
96  if (!keyword_found) continue;
97 
98  // Enters "\data\" section, and looks for patterns like "ngram 1=1000",
99  // which means there are 1000 unigrams.
100  std::size_t equal_symbol_pos = line.find("=");
101  if (equal_symbol_pos != std::string::npos)
102  line.replace(equal_symbol_pos, 1, " = "); // Inserts spaces around "="
103  std::vector<std::string> col;
104  SplitStringToVector(line, " \t", true, &col);
105  if (col.size() == 4 && col[0] == "ngram" && col[2] == "=") {
106  int32 order, ngram_count;
107  if (!ConvertStringToInteger(col[1], &order) ||
108  !ConvertStringToInteger(col[3], &ngram_count)) {
109  PARSE_ERR << "Cannot parse ngram count '" << line << "'.";
110  }
111  if (ngram_counts_.size() <= order) {
112  ngram_counts_.resize(order);
113  }
114  ngram_counts_[order - 1] = ngram_count;
115  } else {
116  KALDI_WARN << "Uninterpretable line in \\data\\ section: " << line;
117  }
118  }
119 
120  if (ngram_counts_.size() == 0)
121  PARSE_ERR << "\\data\\ section missing or empty.";
122 
123  // Signal that grammar order and n-gram counts are known.
124  HeaderAvailable();
125 
126  NGram ngram;
127  ngram.words.reserve(ngram_counts_.size());
128 
129  // Processes "\N-grams:" section.
130  for (int32 cur_order = 1; cur_order <= ngram_counts_.size(); ++cur_order) {
131  // Skips n-grams with zero count.
132  if (ngram_counts_[cur_order - 1] == 0) {
133  KALDI_WARN << "Zero ngram count in ngram order " << cur_order
134  << "(look for 'ngram " << cur_order << "=0' in the \\data\\ "
135  << " section). There is possibly a problem with the file.";
136  continue;
137  }
138 
139  // Must be looking at a \k-grams: directive at this point.
140  std::ostringstream keyword;
141  keyword << "\\" << cur_order << "-grams:";
142  if (line != keyword.str()) {
143  PARSE_ERR << "Invalid directive '" << line << "', "
144  << "expecting '" << keyword.str() << "'.";
145  }
146  KALDI_LOG << "Reading " << line << " section.";
147 
148  int32 ngram_count = 0;
149  while (++line_number_, getline(is, line) && !is.eof()) {
150  if (line.empty()) continue;
151  if (line[0] == '\\') break;
152 
153  std::vector<std::string> col;
154  SplitStringToVector(line, " \t", true, &col);
155 
156  if (col.size() < 1 + cur_order ||
157  col.size() > 2 + cur_order ||
158  (cur_order == ngram_counts_.size() && col.size() != 1 + cur_order)) {
159  PARSE_ERR << "Invalid n-gram line '" << line << "'";
160  }
161  ++ngram_count;
162 
163  // Parse out n-gram logprob and, if present, backoff weight.
164  if (!ConvertStringToReal(col[0], &ngram.logprob)) {
165  PARSE_ERR << "Invalid n-gram logprob '" << col[0] << "'.";
166  }
167  ngram.backoff = 0.0;
168  if (col.size() > cur_order + 1) {
169  if (!ConvertStringToReal(col[cur_order + 1], &ngram.backoff))
170  PARSE_ERR << "Invalid backoff weight '" << col[cur_order + 1] << "'.";
171  }
172  // Convert to natural log unless the option is set not to.
173  if (!options_.use_log10) {
174  ngram.logprob *= M_LN10;
175  ngram.backoff *= M_LN10;
176  }
177 
178  ngram.words.resize(cur_order);
179  bool skip_ngram = false;
180  for (int32 index = 0; !skip_ngram && index < cur_order; ++index) {
181  int32 word;
182  if (symbols_) {
183  // Symbol table provided, so symbol labels are expected.
184  if (options_.oov_handling == ArpaParseOptions::kAddToSymbols) {
185  word = symbols_->AddSymbol(col[1 + index]);
186  } else {
187  word = symbols_->Find(col[1 + index]);
188  if (word == fst::SymbolTable::kNoSymbol) {
189  switch(options_.oov_handling) {
190  case ArpaParseOptions::kReplaceWithUnk:
191  word = options_.unk_symbol;
192  break;
193  case ArpaParseOptions::kSkipNGram:
194  skip_ngram = true;
195  break;
196  default:
197  PARSE_ERR << "Word '" << col[1 + index]
198  << "' not in symbol table.";
199  }
200  }
201  }
202  } else {
203  // Symbols not provided, LM file should contain integers.
204  if (!ConvertStringToInteger(col[1 + index], &word) || word < 0) {
205  PARSE_ERR << "invalid symbol '" << col[1 + index] << "'";
206  }
207  }
208  // Whichever way we got it, an epsilon is invalid.
209  if (word == 0) {
210  PARSE_ERR << "Epsilon symbol '" << col[1 + index]
211  << "' is illegal in ARPA LM.";
212  }
213  ngram.words[index] = word;
214  }
215  if (!skip_ngram) {
216  ConsumeNGram(ngram);
217  }
218  }
219  if (ngram_count > ngram_counts_[cur_order - 1]) {
220  PARSE_ERR << "Header said there would be " << ngram_counts_[cur_order]
221  << " n-grams of order " << cur_order << ", but we saw "
222  << ngram_count;
223  }
224  }
225 
226  if (line != "\\end\\") {
227  PARSE_ERR << "Invalid or unexpected directive line '" << line << "', "
228  << "expected \\end\\.";
229  }
230 
231  ReadComplete();
232 
233 #undef PARSE_ERR
234 }
