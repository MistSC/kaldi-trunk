63 template<class I>
64 bool SplitStringToIntegers(const std::string &full,
65  const char *delim,
66  bool omit_empty_strings, // typically false [but
67  // should probably be true
68  // if "delim" is spaces].
69  std::vector<I> *out) {
70  KALDI_ASSERT(out != NULL);
71  KALDI_ASSERT_IS_INTEGER_TYPE(I);
72  if (*(full.c_str()) == '\0') {
73  out->clear();
74  return true;
75  }
76  std::vector<std::string> split;
77  SplitStringToVector(full, delim, omit_empty_strings, &split);
78  out->resize(split.size());
79  for (size_t i = 0; i < split.size(); i++) {
80  const char *this_str = split[i].c_str();
81  char *end = NULL;
82  int64 j = 0;
83  j = KALDI_STRTOLL(this_str, &end);
84  if (end == this_str || *end != '\0') {
85  out->clear();
86  return false;
87  } else {
88  I jI = static_cast<I>(j);
89  if (static_cast<int64>(jI) != j) {
90  // output type cannot fit this integer.
91  out->clear();
92  return false;
93  }
94  (*out)[i] = jI;
95  }
96  }
97  return true;
98 }

113 template<class Int>
114 bool ConvertStringToInteger(const std::string &str,
115  Int *out) {
116  KALDI_ASSERT_IS_INTEGER_TYPE(Int);
117  const char *this_str = str.c_str();
118  char *end = NULL;
119  errno = 0;
120  int64 i = KALDI_STRTOLL(this_str, &end);
121  if (end != this_str)
122  while (isspace(*end)) end++;
123  if (end == this_str || *end != '\0' || errno != 0)
124  return false;
125  Int iInt = static_cast<Int>(i);
126  if (static_cast<int64>(iInt) != i ||
127  (i < 0 && !std::numeric_limits<Int>::is_signed)) {
128  return false;
129  }
130  *out = iInt;
131  return true;
132 }
