37 struct MfccOptions {
38  FrameExtractionOptions frame_opts;
39  MelBanksOptions mel_opts;
40  int32 num_ceps; // e.g. 13: num cepstral coeffs, counting zero.
41  bool use_energy; // use energy; else C0
42  BaseFloat energy_floor;
43  bool raw_energy; // If true, compute energy before preemphasis and windowing
44  BaseFloat cepstral_lifter; // Scaling factor on cepstra for HTK compatibility.
45  // if 0.0, no liftering is done.
46  bool htk_compat; // if true, put energy/C0 last and introduce a factor of
47  // sqrt(2) on C0 to be the same as HTK.
48 
49  MfccOptions() : mel_opts(23),
50  // defaults the #mel-banks to 23 for the MFCC computations.
51  // this seems to be common for 16khz-sampled data,
52  // but for 8khz-sampled data, 15 may be better.
53  num_ceps(13),
54  use_energy(true),
55  energy_floor(0.0), // not in log scale: a small value e.g. 1.0e-10
56  raw_energy(true),
57  cepstral_lifter(22.0),
58  htk_compat(false) {}
