51 bool ReadScriptFile(std::istream &is,
52  bool warn,
53  std::vector<std::pair<std::string, std::string> >
54  *script_out) {
55  KALDI_ASSERT(script_out != NULL);
56  std::string line;
57  int line_number = 0;
58  while (getline(is, line)) {
59  line_number++;
60  const char *c = line.c_str();
61  if (*c == '\0') {
62  if (warn)
63  KALDI_WARN << "Empty " << line_number << "'th line in script file";
64  return false; // Empty line so invalid scp file format..
65  }
66 
67  std::string key, rest;
68  SplitStringOnFirstSpace(line, &key, &rest);
69 
70  if (key.empty() || rest.empty()) {
71  if (warn)
72  KALDI_WARN << "Invalid " << line_number << "'th line in script file"
73  <<":\"" << line << '"';
74  return false;
75  }
76  // Not using push_back because who knows how many temp. variables
77  // used there.
78  script_out->resize(script_out->size()+1);
79  script_out->back().first = key;
80  script_out->back().second = rest;
81  }
82  return true;
83 }

85 bool WriteScriptFile(std::ostream &os,
86  const std::vector<std::pair<std::string, std::string> >
87  &script) {
88  if (!os.good()) {
89  KALDI_WARN << "WriteScriptFile: attempting to write to invalid stream.";
90  return false;
91  }
92  std::vector<std::pair<std::string, std::string> >::const_iterator iter;
93  for (iter = script.begin(); iter != script.end(); ++iter) {
94  if (!IsToken(iter->first)) {
95  KALDI_WARN << "WriteScriptFile: using invalid token \"" << iter->first <<
96  '"';
97  return false;
98  }
99  if (iter->second.find('\n') != std::string::npos ||
100  (iter->second.length() != 0 &&
101  (isspace(iter->second[0]) ||
102  isspace(iter->second[iter->second.length()-1])))) {
103  // second part contains newline or leading or trailing space.
104  KALDI_WARN << "WriteScriptFile: attempting to write invalid line \"" <<
105  iter->second << '"';
106  return false;
107  }
108  os << iter->first << ' ' << iter->second << '\n';
109  }
110  if (!os.good()) {
111  KALDI_WARN << "WriteScriptFile: stream in error state.";
112  return false;
113  }
114  return true;
115 }

137 WspecifierType ClassifyWspecifier(const std::string &wspecifier,
138  std::string *archive_wxfilename,
139  std::string *script_wxfilename,
140  WspecifierOptions *opts) {
141  // Examples:
142  // ark,t:wxfilename -> kArchiveWspecifier
143  // ark,b:wxfilename -> kArchiveWspecifier
144  // scp,t:rxfilename -> kScriptWspecifier
145  // scp,t:rxfilename -> kScriptWspecifier
146  // ark,scp,t:filename, wxfilename -> kBothWspecifier
147  // ark,scp:filename, wxfilename -> kBothWspecifier
148  // Note we can include the flush option (f) or no-flush (nf)
149  // anywhere: e.g.
150  // ark,scp,f:filename, wxfilename -> kBothWspecifier
151  // or:
152  // scp,t,nf:rxfilename -> kScriptWspecifier
153 
154  if (archive_wxfilename) archive_wxfilename->clear();
155  if (script_wxfilename) script_wxfilename->clear();
156 
157  size_t pos = wspecifier.find(':');
158  if (pos == std::string::npos) return kNoWspecifier;
159  if (isspace(*(wspecifier.rbegin()))) return kNoWspecifier; // Trailing space
160  // disallowed.
161 
162  std::string before_colon(wspecifier, 0, pos), after_colon(wspecifier, pos+1);
163 
164  std::vector<std::string> split_first_part; // Split part before ':' on ', '.
165  SplitStringToVector(before_colon, ", ", false, &split_first_part); // false==
166  // don't omit empty strings between commas.
167 
168  WspecifierType ws = kNoWspecifier;
169 
170  if (opts != NULL)
171  *opts = WspecifierOptions(); // Make sure all the defaults are as in the
172  // default constructor of the options class.
173 
174  for (size_t i = 0; i < split_first_part.size(); i++) {
175  const std::string &str = split_first_part[i]; // e.g. "b", "t", "f", "ark",
176  // "scp".
177  const char *c = str.c_str();
178  if (!strcmp(c, "b")) {
179  if (opts) opts->binary = true;
180  } else if (!strcmp(c, "f")) {
181  if (opts) opts->flush = true;
182  } else if (!strcmp(c, "nf")) {
183  if (opts) opts->flush = false;
184  } else if (!strcmp(c, "t")) {
185  if (opts) opts->binary = false;
186  } else if (!strcmp(c, "p")) {
187  if (opts) opts->permissive = true;
188  } else if (!strcmp(c, "ark")) {
189  if (ws == kNoWspecifier) ws = kArchiveWspecifier;
190  else
191  return kNoWspecifier; // We do not allow "scp, ark", only "ark,
192  // scp".
193  } else if (!strcmp(c, "scp")) {
194  if (ws == kNoWspecifier) ws = kScriptWspecifier;
195  else if (ws == kArchiveWspecifier) ws = kBothWspecifier;
196  else
197  return kNoWspecifier; // repeated "scp" option: invalid.
198  } else {
199  return kNoWspecifier; // Could not interpret this option.
200  }
201  }
202 
203  switch (ws) {
204  case kArchiveWspecifier:
205  if (archive_wxfilename)
206  *archive_wxfilename = after_colon;
207  break;
208  case kScriptWspecifier:
209  if (script_wxfilename)
210  *script_wxfilename = after_colon;
211  break;
212  case kBothWspecifier:
213  pos = after_colon.find(','); // first comma.
214  if (pos == std::string::npos) return kNoWspecifier;
215  if (archive_wxfilename)
216  *archive_wxfilename = std::string(after_colon, 0, pos);
217  if (script_wxfilename)
218  *script_wxfilename = std::string(after_colon, pos+1);
219  break;
220  case kNoWspecifier: default: break;
221  }
222  return ws;
223 }

227 RspecifierType ClassifyRspecifier(const std::string &rspecifier,
228  std::string *wxfilename,
229  RspecifierOptions *opts) {
230  // Examples
231  // ark:rxfilename -> kArchiveRspecifier
232  // scp:rxfilename -> kScriptRspecifier
233  //
234  // We also allow the meaningless prefixes b, and t,
235  // plus the options o (once), no (not-once),
236  // s (sorted) and ns (not-sorted), p (permissive)
237  // and np (not-permissive).
238  // so the following would be valid:
239  //
240  // f, o, b, np, ark:rxfilename -> kArchiveRspecifier
241  //
242  // Examples:
243  //
244  // b, ark:rxfilename -> kArchiveRspecifier
245  // t, ark:rxfilename -> kArchiveRspecifier
246  // b, scp:rxfilename -> kScriptRspecifier
247  // t, no, s, scp:rxfilename -> kScriptRspecifier
248  // t, ns, scp:rxfilename -> kScriptRspecifier
249 
250  // Improperly formed Rspecifiers will be classified as kNoRspecifier.
251 
252  if (wxfilename) wxfilename->clear();
253 
254  if (opts != NULL)
255  *opts = RspecifierOptions(); // Make sure all the defaults are as in the
256  // default constructor of the options class.
257 
258  size_t pos = rspecifier.find(':');
259  if (pos == std::string::npos) return kNoRspecifier;
260 
261  if (isspace(*(rspecifier.rbegin()))) return kNoRspecifier; // Trailing space
262  // disallowed.
263 
264  std::string before_colon(rspecifier, 0, pos),
265  after_colon(rspecifier, pos+1);
266 
267  std::vector<std::string> split_first_part; // Split part before ':' on ', '.
268  SplitStringToVector(before_colon, ", ", false, &split_first_part); // false==
269  // don't omit empty strings between commas.
270 
271  RspecifierType rs = kNoRspecifier;
272 
273  for (size_t i = 0; i < split_first_part.size(); i++) {
274  const std::string &str = split_first_part[i]; // e.g. "b", "t", "f", "ark",
275  // "scp".
276  const char *c = str.c_str();
277  if (!strcmp(c, "b")); // Ignore this option. It's so we can use the same
278  // specifiers for rspecifiers and wspecifiers.
279  else if (!strcmp(c, "t")); // Ignore this option too.
280  else if (!strcmp(c, "o")) {
281  if (opts) opts->once = true;
282  } else if (!strcmp(c, "no")) {
283  if (opts) opts->once = false;
284  } else if (!strcmp(c, "p")) {
285  if (opts) opts->permissive = true;
286  } else if (!strcmp(c, "np")) {
287  if (opts) opts->permissive = false;
288  } else if (!strcmp(c, "s")) {
289  if (opts) opts->sorted = true;
290  } else if (!strcmp(c, "ns")) {
291  if (opts) opts->sorted = false;
292  } else if (!strcmp(c, "cs")) {
293  if (opts) opts->called_sorted = true;
294  } else if (!strcmp(c, "ncs")) {
295  if (opts) opts->called_sorted = false;
296  } else if (!strcmp(c, "ark")) {
297  if (rs == kNoRspecifier) rs = kArchiveRspecifier;
298  else
299  return kNoRspecifier; // Repeated or combined ark and scp options
300  // invalid.
301  } else if (!strcmp(c, "scp")) {
302  if (rs == kNoRspecifier) rs = kScriptRspecifier;
303  else
304  return kNoRspecifier; // Repeated or combined ark and scp options
305  // invalid.
306  } else {
307  return kNoRspecifier; // Could not interpret this option.
308  }
309  }
310  if ((rs == kArchiveRspecifier || rs == kScriptRspecifier)
311  && wxfilename != NULL)
312  *wxfilename = after_colon;
313  return rs;
314 }
