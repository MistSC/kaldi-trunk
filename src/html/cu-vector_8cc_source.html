260 template<typename Real>
261 Real CuVectorBase<Real>::Sum() const {
262  if (dim_ == 0)
263  return 0.0;
264 #if HAVE_CUDA == 1
265  if (CuDevice::Instantiate().Enabled()) {
266  Timer tim;
267  int max_threads = 2048;
268  // This is the smallest block of consecutive vector elements, which
269  // its sum will save at the partial vector.
270  int block_size = (dim_ + max_threads - 1) / max_threads;
271  if (block_size > 3) {
272  int dimBlock(CU1DBLOCK);
273  int dimGrid(n_blocks(max_threads, CU1DBLOCK));
274  CuVector<Real> g(dimGrid);
275  cuda_pvec_sum(dimGrid, dimBlock, data_, g.Data(), dim_, block_size);
276  CU_SAFE_CALL(cudaGetLastError());
277  Vector<Real> tmp(dimGrid);
278  g.CopyToVec(&tmp);
279  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
280  return tmp.Sum();
281  } else {
282  if (dim_ == 0) return 0.0;
283  CuVector<Real> tmp(1, kUndefined);
284  int dimBlock(CU1DBLOCK);
285  int dimGrid = 1; // only 1 block here. we have loops in each thread.
286  cuda_vec_sum(dimGrid, dimBlock, data_, tmp.Data(), dim_, 1);
287  CU_SAFE_CALL(cudaGetLastError());
288  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
289  return tmp(0);
290  }
291  } else
292 #endif
293  {
294  return Vec().Sum();
295  }

537 template<typename Real>
538 void CuVectorBase<Real>::AddDiagMatMat(
539  Real alpha,
540  const CuMatrixBase<Real> &M, MatrixTransposeType transM,
541  const CuMatrixBase<Real> &N, MatrixTransposeType transN,
542  Real beta) {
543 #if HAVE_CUDA == 1
544  if (CuDevice::Instantiate().Enabled()) {
545  Timer tim;
546  MatrixIndexT dim = this->dim_,
547  M_col_dim = (transM == kTrans ? M.NumRows() : M.NumCols()),
548  N_row_dim = (transN == kTrans ? N.NumCols() : N.NumRows());
549  KALDI_ASSERT(M_col_dim == N_row_dim); // this is the dimension we sum over
550  MatrixIndexT M_row_stride = M.Stride(), M_col_stride = 1;
551  if (transM == kTrans) std::swap(M_row_stride, M_col_stride);
552  MatrixIndexT N_row_stride = N.Stride(), N_col_stride = 1;
553  if (transN == kTrans) std::swap(N_row_stride, N_col_stride);
554  if (dim_ == 0) return;
555 
556  // This kernel can take a variable grid dimension, it makes use
557  // of the extra threads by partitioning each vector-vector dot
558  // product into multiple pieces.
559  int dimBlock(CU1DBLOCK);
560  int dimGrid(n_blocks(dim,CU1DBLOCK));
561  int threads_per_element = 1;
562  // dimGridLimit may be any power of two between 1 and 256 inclusive; it was
563  // determined empirically based on speed tests.
564  int dimGridLimit = (transM == kNoTrans && transN == kTrans ? 2048 :
565  (transM == kTrans && transN == kNoTrans ? 16 : 32));
566 
567 
568  while (M_col_dim > 10 * threads_per_element &&
569  dimGrid < dimGridLimit && threads_per_element < 256) {
570  threads_per_element *= 2;
571  dimGrid = n_blocks(dim * threads_per_element, CU1DBLOCK);
572  }
573 
574  cuda_add_diag_mat_mat(dimGrid, dimBlock, alpha, data_, dim,
575  M.Data(), M_col_dim, M_row_stride, M_col_stride,
576  N.Data(), N_row_stride, N_col_stride,
577  threads_per_element, beta);
578  CU_SAFE_CALL(cudaGetLastError());
579  CuDevice::Instantiate().AccuProfile(__func__, tim.Elapsed());
580  } else
581 #endif
582  {
583  Vec().AddDiagMatMat(alpha, M.Mat(), transM, N.Mat(), transN, beta);
584  }
585 }

840 template<typename Real>
841 void CuVector<Real>::Resize(MatrixIndexT dim, MatrixResizeType t) {
842  KALDI_ASSERT(t == kSetZero || t == kUndefined); // Others not implemented
843  // yet.
844  if (this->dim_ == dim) {
845  this->SetZero();

866 template<typename Real>
867 void CuVector<Real>::Swap(Vector<Real> *vec) {
868 #if HAVE_CUDA == 1
869  if (CuDevice::Instantiate().Enabled()) {
870  if (this->dim_ == 0) {
871  if (vec->dim_ != 0) {
872  // *this is empty, but vec is nonempty.
873  Resize(vec->dim_, kUndefined);
874  this->CopyFromVec(*vec);
875  vec->Resize(0);
876  }
877  // else both are empty.
878  } else { // *this is nonempty.
879  if (vec->dim_ != 0) {
880  // Both *this and *vec are nonempty. Recurse to simpler cases.
881  // this could be done more efficiently in the case where
882  // the size does not change.
883  Vector<Real> temp;
884  this->Swap(&temp); // now temp is full, *this is empty.
885  vec->Swap(&temp); // now vec has data from *this, temp has
886  // data from vec.
887  Swap(vec); // copy data in vec to *this, which is now empty.
888  } else { // *this is full but *vec is empty.
889  vec->Resize(this->dim_, kUndefined);
890  this->CopyToVec(vec);
891  this->Destroy();
892  }
893  }
894  } else
895 #endif
896  {
897  std::swap(vec->data_, this->data_);
898  std::swap(vec->dim_, this->dim_);
899  }
900 }
