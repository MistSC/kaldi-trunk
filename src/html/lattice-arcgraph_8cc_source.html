31 /* does the same as ConvertFstToLattice, but
32  we encode HCLG state and transition into the output labels
33  (which is what we later use as HCLG arc graph)
34  and we replace transition-ids with transition states to make it independent
35  of whether self-loops occur before or after the normal transitions,
36  which is necessary, since the order was time reversed */
37 void ConvertFstToArcLattice(fst::StdVectorFst *net, kaldi::Lattice *lat,
38  std::vector<std::pair<int32,int32> > *arc_map,
39  kaldi::TransitionModel *tmodel, bool keep_weights) {
40  int32 num_arcs = 0; // count to reserve the size of arc_map
41  for(int32 i = 0; i < net->NumStates(); i++) {
42  lat->AddState();
43  num_arcs += net->NumArcs(i);
44  }
45  arc_map->reserve(num_arcs);
46  lat->SetStart(net->Start());
47  for(fst::StateIterator<fst::StdVectorFst> siter(*net); !siter.Done(); siter.Next()) {
48  StateId s = siter.Value();
49  Weight w_final = net->Final(s);
50  if (w_final != Weight::Zero()) { // final state
51  kaldi::LatticeWeight new_weight(kaldi::LatticeWeight::One());
52  if (keep_weights) new_weight.SetValue1(w_final.Value());
53  lat->SetFinal(s, new_weight);
54  }
55  kaldi::int32 arc_id = 0;
56  for (fst::MutableArcIterator<fst::StdVectorFst> aiter(net, s); !aiter.Done(); aiter.Next()) {
57  fst::StdArc arc = aiter.Value();
58  kaldi::LatticeWeight new_weight(kaldi::LatticeWeight::One());
59  if (keep_weights) new_weight.SetValue1(arc.weight.Value());
60  Label ilabel = arc.ilabel; // transition-id
61  // transition state is independent of self-loop order
62  if (ilabel > 0) ilabel = tmodel->TransitionIdToTransitionState(ilabel);
63  arc_map->push_back(std::make_pair(s, arc_id));
64  Label olabel = arc_map->size(); // new labels encode the unique index
65  kaldi::LatticeArc new_arc(ilabel, olabel, new_weight, arc.nextstate);
66  lat->AddArc(s, new_arc);
67  arc_id ++;
68  }
69  }
70 }

72 /* Discards the ilabel on an arc, and replaces the (ilabel, olabel) with a pair
73  obtained from the "arc_map" vector, indexed with the original olabel. At
74  input the (ilabel, olabel) of the graph are (transition-ids, indexes into
75  arc_map), and when this function is done they are (states in HCLG,
76  indexes into the lists of arcs leaving those states). */
77 void DecodeGraphSymbols(const std::vector<std::pair<int32,int32> > &arc_map,
78  fst::StdVectorFst *net) {
79  // maps symbols back state/arc pairs
80  for(fst::StateIterator<fst::StdVectorFst> siter(*net);
81  !siter.Done(); siter.Next()) {
82  StateId s = siter.Value();
83  for (fst::MutableArcIterator<fst::StdVectorFst> aiter(net, s);
84  !aiter.Done(); aiter.Next()) {
85  fst::StdArc arc = aiter.Value();
86  Label ilabel = arc_map[arc.ilabel-1].first; // state
87  Label olabel = arc_map[arc.ilabel-1].second; // arc
88  fst::StdArc new_arc(ilabel, olabel, arc.weight, arc.nextstate);
89  aiter.SetValue(new_arc);
90  }
91  }
92 }

94 void MapTransitionIdsToTransitionStates(kaldi::CompactLattice *lat,
95  kaldi::TransitionModel *tmodel, bool keep_weights) {
96  // maps transition-ids to transition states and removes weights
97  for(fst::StateIterator<kaldi::CompactLattice> siter(*lat);
98  !siter.Done(); siter.Next()) {
99  StateId s = siter.Value();
100 
101  kaldi::CompactLatticeWeight w_final = lat->Final(s);
102  if (w_final.Weight() != kaldi::LatticeWeight::Zero()) { // final state
103  std::vector<kaldi::int32> syms = w_final.String();
104  // map all transition-ids to transition-states
105  for(std::vector<kaldi::int32>::iterator it = syms.begin();
106  it != syms.end(); ++it) {
107  *it = tmodel->TransitionIdToTransitionState(*it);
108  }
109  kaldi::LatticeWeight new_w(w_final.Weight());
110  if (!keep_weights) new_w = kaldi::LatticeWeight::One();
111  kaldi::CompactLatticeWeight newwgt(new_w, syms);
112  lat->SetFinal(s, newwgt);
113  }
114 
115  // go through all states
116  for(fst::MutableArcIterator<kaldi::CompactLattice> aiter(lat, s);
117  !aiter.Done(); aiter.Next()) {
118  kaldi::CompactLatticeArc arc = aiter.Value();
119  kaldi::CompactLatticeWeight w = arc.weight;
120  std::vector<kaldi::int32> syms = w.String();
121 
122  // map all transition-ids to transition-states
123  for(std::vector<kaldi::int32>::iterator it = syms.begin();
124  it != syms.end(); ++it) {
125  *it = tmodel->TransitionIdToTransitionState(*it);
126  }
127  kaldi::LatticeWeight new_w(w.Weight());
128  if (!keep_weights) new_w = kaldi::LatticeWeight::One();
129  kaldi::CompactLatticeWeight newwgt(new_w, syms);
130  arc.weight = newwgt;
131  aiter.SetValue(arc);
132  }
133  }

136 int main(int argc, char *argv[]) {
137  try {
138  using namespace kaldi;
139  typedef kaldi::int32 int32;
140  typedef kaldi::int64 int64;
141  using fst::SymbolTable;
142  using fst::VectorFst;
143  using fst::StdArc;
144 
145  std::string lattice_wspecifier;
146  std::string fst_out_filename;
147 
148  const char *usage =
149  "Compose decoding graph with given lattices to obtain active-arc lattices.\n"
150  "Usage: lattice-arcgraph [options] <model-file> "
151  "<decoding-graph-fst> <lattice-rspecifier> <arcs-wspecifier>\n"
152  " e.g.: lattice-arcgraph final.mdl HCLG.fst ark:in.lats ark:out.arcs\n";
153 
154  ParseOptions po(usage);
155  po.Register("write-lattices", &lattice_wspecifier,
156  "Write intermediate lattices to archive.");
157 
158  bool reverse = true;
159  po.Register("reverse", &reverse, "Reverse input lattices in time");
160 
161  po.Read(argc, argv);
162 
163  if (po.NumArgs() != 4) {
164  po.PrintUsage();
165  exit(1);
166  }
167 
168  std::string
169  model_filename = po.GetArg(1),
170  fst_in_filename = po.GetArg(2),
171  lats_rspecifier = po.GetArg(3),
172  arcs_wspecifier = po.GetArg(4);
173 
174  // options for lattice determinization
175  fst::DeterminizeLatticeOptions lat_opts;
176  lat_opts.max_mem = 200000000; // 200 MB
177  lat_opts.max_loop = 500000;
178  lat_opts.delta = fst::kDelta;
179 
180  // load transition model as begin of model file
181  TransitionModel trans_model;
182  ReadKaldiObject(model_filename, &trans_model);
183 
184  // read decoding graph and convert to transition-state acceptor lattice
185  fst::StdVectorFst *hclg = fst::ReadFstKaldi(fst_in_filename);
186  Lattice graph;
187  // encode HCLG state and transition into the output labels
188  // and replace transition-ids with transition states
189  std::vector<std::pair<int32,int32> > arc_map; // maps state/arc to symbol-id
190  ConvertFstToArcLattice(hclg, &graph, &arc_map, &trans_model, true); // keep weights
191  fst::ArcSort(&graph, fst::ILabelCompare<kaldi::LatticeArc>());
192 
193  TableWriter<fst::VectorFstHolder> arcs_writer(arcs_wspecifier);
194  LatticeWriter lat_writer;
195  if (lattice_wspecifier != "") lat_writer.Open(lattice_wspecifier);
196 
197  // convert all lattices and compose with acceptor HCLG
198  int32 n_done = 0, n_error = 0;
199  kaldi::SequentialCompactLatticeReader lat_reader(lats_rspecifier);
200  for (; !lat_reader.Done(); lat_reader.Next()) {
201  std::string key = lat_reader.Key();
202  CompactLattice lat = lat_reader.Value();
203  lat_reader.FreeCurrent();
204 
205  // map transition-ids to self-loop independent and set weights to zero
206  MapTransitionIdsToTransitionStates(&lat, &trans_model, false);
207 
208  // convert from CompactLattice to Lattice
209  Lattice lat_mapped;
210  ConvertLattice(lat, &lat_mapped);
211  fst::Project(&lat_mapped, fst::PROJECT_INPUT); // keep only transition states
212 
213  Lattice lat_composed;
214  if(reverse) {
215  // reverse lattice in time
216  Lattice lat_reverse;
217  fst::Reverse(lat_mapped, &lat_reverse);
218  RemoveEpsLocal(&lat_reverse);
219  // compose
220  KALDI_LOG << "compose with lattice " << key;
221  Compose(lat_reverse, graph, &lat_composed); // composed FST contains HCLG arcs
222  } else {
223  KALDI_LOG << "compose with lattice " << key;
224  Compose(lat_mapped, graph, &lat_composed); // composed FST contains HCLG arcs
225  }
226  if (lattice_wspecifier != "") lat_writer.Write(key, lat_composed);
227 
228  CompactLattice clat_determinized;
229  if (DeterminizeLattice(lat_composed, &clat_determinized, lat_opts, NULL)) {
230  // now we can forget about the weights
231  ScaleLattice(fst::LatticeScale(0.0, 0.0), &clat_determinized);
232  Lattice lat_det;
233  ConvertLattice(clat_determinized, &lat_det);
234 
235  // convert to StdFst, remove word labels
236  fst::VectorFst<fst::StdArc> fst_det;
237  fst::LatticeToStdMapper<BaseFloat> mapper;
238  Map(lat_det, &fst_det, mapper);
239  Project(&fst_det, fst::PROJECT_INPUT);
240  ArcSort(&fst_det, fst::ILabelCompare<fst::StdArc>()); //to improve speed
241 
242  // determinize again on the arc_ids
243  fst::VectorFst<fst::StdArc> fst_final;
244  bool debug_location = false;
245  DeterminizeStar(fst_det, &fst_final, fst::kDelta, &debug_location, -1);
246  DecodeGraphSymbols(arc_map, &fst_final);
247  ArcSort(&fst_final, fst::OLabelCompare<fst::StdArc>());
248  // the decoder expects the arc numbers to be sorted
249  arcs_writer.Write(key, fst_final);
250 
251  KALDI_LOG << key << " finished";
252  n_done++;
253  } else {
254  n_error++; // will have already printed warning.
255  }
256  }
257  if (lattice_wspecifier != "") lat_writer.Close();
258 
259  KALDI_LOG << "Done converting " << n_done << " lattices, failed: " << n_error;
260  return (n_done != 0 ? 0 : 1);
261  } catch(const std::exception &e) {
262  std::cerr << e.what();
263  return -1;
264  }
265 }
