35 // sums a bunch of archives to produce one archive
36 int32 TypeOneUsage(const ParseOptions &po) {
37  int32 num_args = po.NumArgs();
38  std::string vector_in_fn1 = po.GetArg(1),
39  vector_out_fn = po.GetArg(num_args);
40 
41  // Output vector
42  BaseFloatVectorWriter vector_writer(vector_out_fn);
43 
44  // Input vectors
45  SequentialBaseFloatVectorReader vector_reader1(vector_in_fn1);
46  std::vector<RandomAccessBaseFloatVectorReader*> vector_readers(num_args-2,
47  static_cast<RandomAccessBaseFloatVectorReader*>(NULL));
48  std::vector<std::string> vector_in_fns(num_args-2);
49  for (int32 i = 2; i < num_args; ++i) {
50  vector_readers[i-2] = new RandomAccessBaseFloatVectorReader(po.GetArg(i));
51  vector_in_fns[i-2] = po.GetArg(i);
52  }
53 
54  int32 n_utts = 0, n_total_vectors = 0,
55  n_success = 0, n_missing = 0, n_other_errors = 0;
56 
57  for (; !vector_reader1.Done(); vector_reader1.Next()) {
58  std::string key = vector_reader1.Key();
59  Vector<BaseFloat> vector1 = vector_reader1.Value();
60  vector_reader1.FreeCurrent();
61  n_utts++;
62  n_total_vectors++;
63 
64  Vector<BaseFloat> vector_out(vector1);
65 
66  for (int32 i = 0; i < num_args-2; ++i) {
67  if (vector_readers[i]->HasKey(key)) {
68  Vector<BaseFloat> vector2 = vector_readers[i]->Value(key);
69  n_total_vectors++;
70  if (vector2.Dim() == vector_out.Dim()) {
71  vector_out.AddVec(1.0, vector2);
72  } else {
73  KALDI_WARN << "Dimension mismatch for utterance " << key
74  << " : " << vector2.Dim() << " for "
75  << "system " << (i + 2) << ", rspecifier: "
76  << vector_in_fns[i] << " vs " << vector_out.Dim()
77  << " primary vector, rspecifier:" << vector_in_fn1;
78  n_other_errors++;
79  }
80  } else {
81  KALDI_WARN << "No vector found for utterance " << key << " for "
82  << "system " << (i + 2) << ", rspecifier: "
83  << vector_in_fns[i];
84  n_missing++;
85  }
86  }
87 
88  vector_writer.Write(key, vector_out);
89  n_success++;
90  }
91 
92  KALDI_LOG << "Processed " << n_utts << " utterances: with a total of "
93  << n_total_vectors << " vectors across " << (num_args-1)
94  << " different systems";
95  KALDI_LOG << "Produced output for " << n_success << " utterances; "
96  << n_missing << " total missing vectors";
97 
98  DeletePointers(&vector_readers);
99 
100  return (n_success != 0 && n_missing < (n_success - n_missing)) ? 0 : 1;
101 }

103 int32 TypeTwoUsage(const ParseOptions &po,
104  bool binary) {
105  KALDI_ASSERT(po.NumArgs() == 2);
106  KALDI_ASSERT(ClassifyRspecifier(po.GetArg(1), NULL, NULL) != kNoRspecifier &&
107  "vector-sum: first argument must be an rspecifier");
108  // if next assert fails it would be bug in the code as otherwise we shouldn't
109  // be called.
110  KALDI_ASSERT(ClassifyWspecifier(po.GetArg(2), NULL, NULL, NULL) ==
111  kNoWspecifier);
112 
113  SequentialBaseFloatVectorReader vec_reader(po.GetArg(1));
114 
115  Vector<double> sum;
116 
117  int32 num_done = 0, num_err = 0;
118 
119  for (; !vec_reader.Done(); vec_reader.Next()) {
120  const Vector<BaseFloat> &vec = vec_reader.Value();
121  if (vec.Dim() == 0) {
122  KALDI_WARN << "Zero vector input for key " << vec_reader.Key();
123  num_err++;
124  } else {
125  if (sum.Dim() == 0) sum.Resize(vec.Dim());
126  if (sum.Dim() != vec.Dim()) {
127  KALDI_WARN << "Dimension mismatch for key " << vec_reader.Key()
128  << ": " << vec.Dim() << " vs. " << sum.Dim();
129  num_err++;
130  } else {
131  sum.AddVec(1.0, vec);
132  num_done++;
133  }
134  }
135  }
136 
137  Vector<BaseFloat> sum_float(sum);
138  WriteKaldiObject(sum_float, po.GetArg(2), binary);
139 
140  KALDI_LOG << "Summed " << num_done << " vectors, "
141  << num_err << " with errors; wrote sum to "
142  << PrintableWxfilename(po.GetArg(2));
143  return (num_done > 0 && num_err < num_done) ? 0 : 1;
144 }

146 // sum a bunch of single files to produce a single file [including
147 // extended filenames, of course]
148 int32 TypeThreeUsage(const ParseOptions &po,
149  bool binary) {
150  KALDI_ASSERT(po.NumArgs() >= 2);
151  for (int32 i = 1; i < po.NumArgs(); i++) {
152  if (ClassifyRspecifier(po.GetArg(i), NULL, NULL) != kNoRspecifier) {
153  KALDI_ERR << "Wrong usage (type 3): if first and last arguments are not "
154  << "tables, the intermediate arguments must not be tables.";
155  }
156  }
157  if (ClassifyWspecifier(po.GetArg(po.NumArgs()), NULL, NULL, NULL) !=
158  kNoWspecifier) {
159  KALDI_ERR << "Wrong usage (type 3): if first and last arguments are not "
160  << "tables, the intermediate arguments must not be tables.";
161  }
162 
163  bool add = true;
164  Vector<BaseFloat> vec;
165  for (int32 i = 1; i < po.NumArgs(); i++) {
166  bool binary_in;
167  Input ki(po.GetArg(i), &binary_in);
168  // this Read function will throw if there is a size mismatch.
169  vec.Read(ki.Stream(), binary_in, add);
170  }
171  WriteKaldiObject(vec, po.GetArg(po.NumArgs()), binary);
172  KALDI_LOG << "Summed " << (po.NumArgs() - 1) << " vectors; "
173  << "wrote sum to " << PrintableWxfilename(po.GetArg(po.NumArgs()));
174  return 0;
175 }

181 int main(int argc, char *argv[]) {
182  try {
183  using namespace kaldi;
184 
185  const char *usage =
186  "Add vectors (e.g. weights, transition-accs; speaker vectors)\n"
187  "If you need to scale the inputs, use vector-scale on the inputs\n"
188  "\n"
189  "Type one usage:\n"
190  " vector-sum [options] <vector-in-rspecifier1> [<vector-in-rspecifier2>"
191  " <vector-in-rspecifier3> ...] <vector-out-wspecifier>\n"
192  " e.g.: vector-sum ark:1.weights ark:2.weights ark:combine.weights\n"
193  "Type two usage (sums a single table input to produce a single output):\n"
194  " vector-sum [options] <vector-in-rspecifier> <vector-out-wxfilename>\n"
195  " e.g.: vector-sum --binary=false vecs.ark sum.vec\n"
196  "Type three usage (sums single-file inputs to produce a single output):\n"
197  " vector-sum [options] <vector-in-rxfilename1> <vector-in-rxfilename2> ..."
198  " <vector-out-wxfilename>\n"
199  " e.g.: vector-sum --binary=false 1.vec 2.vec 3.vec sum.vec\n"
200  "See also: copy-vector, dot-weights\n";
201 
202  bool binary;
203 
204  ParseOptions po(usage);
205 
206  po.Register("binary", &binary, "If true, write output as binary (only "
207  "relevant for usage types two or three");
208 
209  po.Read(argc, argv);
210 
211  int32 N = po.NumArgs(), exit_status;
212 
213  if (po.NumArgs() >= 2 &&
214  ClassifyWspecifier(po.GetArg(N), NULL, NULL, NULL) != kNoWspecifier) {
215  // output to table.
216  exit_status = TypeOneUsage(po);
217  } else if (po.NumArgs() == 2 &&
218  ClassifyRspecifier(po.GetArg(1), NULL, NULL) != kNoRspecifier &&
219  ClassifyWspecifier(po.GetArg(N), NULL, NULL, NULL) ==
220  kNoWspecifier) {
221  // input from a single table, output not to table.
222  exit_status = TypeTwoUsage(po, binary);
223  } else if (po.NumArgs() >= 2 &&
224  ClassifyRspecifier(po.GetArg(1), NULL, NULL) == kNoRspecifier &&
225  ClassifyWspecifier(po.GetArg(N), NULL, NULL, NULL) ==
226  kNoWspecifier) {
227  // summing flat files.
228  exit_status = TypeThreeUsage(po, binary);
229  } else {
230  po.PrintUsage();
231  exit(1);
232  }
233  return exit_status;
234  } catch(const std::exception &e) {
235  std::cerr << e.what();
236  return -1;
237  }
