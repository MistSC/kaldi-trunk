39 void HmmTopology::Read(std::istream &is, bool binary) {
40  ExpectToken(is, binary, "<Topology>");
41  if (!binary) { // Text-mode read, different "human-readable" format.
42  phones_.clear();
43  phone2idx_.clear();
44  entries_.clear();
45  std::string token;
46  while ( ! (is >> token).fail() ) {
47  if (token == "</Topology>") { break; } // finished parsing.
48  else if (token != "<TopologyEntry>") {
49  KALDI_ERR << "Reading HmmTopology object, expected </Topology> or <TopologyEntry>, got "<<token;
50  } else {
51  ExpectToken(is, binary, "<ForPhones>");
52  std::vector<int32> phones;
53  std::string s;
54  while (1) {
55  is >> s;
56  if (is.fail()) KALDI_ERR << "Reading HmmTopology object, unexpected end of file while expecting phones.";
57  if (s == "</ForPhones>") break;
58  else {
59  int32 phone;
60  if (!ConvertStringToInteger(s, &phone))
61  KALDI_ERR << "Reading HmmTopology object, expected "
62  << "integer, got instead " << s;
63  phones.push_back(phone);
64  }
65  }
66 
67  std::vector<HmmState> this_entry;
68  std::string token;
69  ReadToken(is, binary, &token);
70  while (token != "</TopologyEntry>") {
71  if (token != "<State>")
72  KALDI_ERR << "Expected </TopologyEntry> or <State>, got instead "<<token;
73  int32 state;
74  ReadBasicType(is, binary, &state);
75  if (state != static_cast<int32>(this_entry.size()))
76  KALDI_ERR << "States are expected to be in order from zero, expected "
77  << this_entry.size() << ", got " << state;
78  ReadToken(is, binary, &token);
79  int32 pdf_class = kNoPdf; // -1 by default, means no pdf.
80  if (token == "<PdfClass>") {
81  ReadBasicType(is, binary, &pdf_class);
82  ReadToken(is, binary, &token);
83  }
84  this_entry.push_back(HmmState(pdf_class));
85  while (token == "<Transition>") {
86  int32 dst_state;
87  BaseFloat trans_prob;
88  ReadBasicType(is, binary, &dst_state);
89  ReadBasicType(is, binary, &trans_prob);
90  this_entry.back().transitions.push_back(std::make_pair(dst_state, trans_prob));
91  ReadToken(is, binary, &token);
92  }
93  if(token == "<Final>") // TODO: remove this clause after a while.
94  KALDI_ERR << "You are trying to read old-format topology with new Kaldi.";
95  if (token != "</State>")
96  KALDI_ERR << "Reading HmmTopology, unexpected token "<<token;
97  ReadToken(is, binary, &token);
98  }
99  int32 my_index = entries_.size();
100  entries_.push_back(this_entry);
101 
102  for (size_t i = 0; i < phones.size(); i++) {
103  int32 phone = phones[i];
104  if (static_cast<int32>(phone2idx_.size()) <= phone)
105  phone2idx_.resize(phone+1, -1); // -1 is invalid index.
106  KALDI_ASSERT(phone > 0);
107  if (phone2idx_[phone] != -1)
108  KALDI_ERR << "Phone with index "<<(i)<<" appears in multiple topology entries.";
109  phone2idx_[phone] = my_index;
110  phones_.push_back(phone);
111  }
112  }
113  }
114  std::sort(phones_.begin(), phones_.end());
115  KALDI_ASSERT(IsSortedAndUniq(phones_));
116  } else { // binary I/O, just read member objects directly from disk.
117  ReadIntegerVector(is, binary, &phones_);
118  ReadIntegerVector(is, binary, &phone2idx_);
119  int32 sz;
120  ReadBasicType(is, binary, &sz);
121  entries_.resize(sz);
122  for (int32 i = 0; i < sz; i++) {
123  int32 thist_sz;
124  ReadBasicType(is, binary, &thist_sz);
125  entries_[i].resize(thist_sz);
126  for (int32 j = 0 ; j < thist_sz; j++) {
127  ReadBasicType(is, binary, &(entries_[i][j].pdf_class));
128  int32 thiss_sz;
129  ReadBasicType(is, binary, &thiss_sz);
130  entries_[i][j].transitions.resize(thiss_sz);
131  for (int32 k = 0; k < thiss_sz; k++) {
132  ReadBasicType(is, binary, &(entries_[i][j].transitions[k].first));
133  ReadBasicType(is, binary, &(entries_[i][j].transitions[k].second));
134  }
135  }
136  }
137  ExpectToken(is, binary, "</Topology>");
138  }
139  Check(); // Will throw if not ok.
140 }

143 void HmmTopology::Write(std::ostream &os, bool binary) const {
144  WriteToken(os, binary, "<Topology>");
145  if (!binary) { // Text-mode write.
146  os << "\n";
147  for (int32 i = 0; i < static_cast<int32> (entries_.size()); i++) {
148  WriteToken(os, binary, "<TopologyEntry>");
149  os << "\n";
150  WriteToken(os, binary, "<ForPhones>");
151  os << "\n";
152  for (size_t j = 0; j < phone2idx_.size(); j++) {
153  if (phone2idx_[j] == i)
154  os << j << " ";
155  }
156  os << "\n";
157  WriteToken(os, binary, "</ForPhones>");
158  os << "\n";
159  for (size_t j = 0; j < entries_[i].size(); j++) {
160  WriteToken(os, binary, "<State>");
161  WriteBasicType(os, binary, static_cast<int32>(j));
162  if (entries_[i][j].pdf_class != kNoPdf) {
163  WriteToken(os, binary, "<PdfClass>");
164  WriteBasicType(os, binary, entries_[i][j].pdf_class);
165  }
166  for (size_t k = 0; k < entries_[i][j].transitions.size(); k++) {
167  WriteToken(os, binary, "<Transition>");
168  WriteBasicType(os, binary, entries_[i][j].transitions[k].first);
169  WriteBasicType(os, binary, entries_[i][j].transitions[k].second);
170  }
171  WriteToken(os, binary, "</State>");
172  os << "\n";
173  }
174  WriteToken(os, binary, "</TopologyEntry>");
175  os << "\n";
176  }
177  } else {
178  WriteIntegerVector(os, binary, phones_);
179  WriteIntegerVector(os, binary, phone2idx_);
180  WriteBasicType(os, binary, static_cast<int32>(entries_.size()));
181  for (size_t i = 0; i < entries_.size(); i++) {
182  WriteBasicType(os, binary, static_cast<int32>(entries_[i].size()));
183  for (size_t j = 0; j < entries_[i].size(); j++) {
184  WriteBasicType(os, binary, entries_[i][j].pdf_class);
185  WriteBasicType(os, binary, static_cast<int32>(entries_[i][j].transitions.size()));
186  for (size_t k = 0; k < entries_[i][j].transitions.size(); k++) {
187  WriteBasicType(os, binary, entries_[i][j].transitions[k].first);
188  WriteBasicType(os, binary, entries_[i][j].transitions[k].second);
189  }
190  }
191  }
192  }
193  WriteToken(os, binary, "</Topology>");
194  if (!binary) os << "\n";
195 }

197 void HmmTopology::Check() {
198  if (entries_.empty() || phones_.empty() || phone2idx_.empty())
199  KALDI_ERR << "HmmTopology::Check(), empty object.";
200  std::vector<bool> is_seen(entries_.size(), false);
201  for (size_t i = 0; i < phones_.size(); i++) {
202  int32 phone = phones_[i];
203  if (static_cast<size_t>(phone) >= phone2idx_.size() ||
204  static_cast<size_t>(phone2idx_[phone]) >= entries_.size())
205  KALDI_ERR << "HmmTopology::Check(), phone has no valid index.";
206  is_seen[phone2idx_[phone]] = true;
207  }
208  for (size_t i = 0; i < entries_.size(); i++) {
209  if (!is_seen[i])
210  KALDI_ERR << "HmmTopoloy::Check(), entry with no corresponding phones.";
211  int32 num_states = static_cast<int32>(entries_[i].size());
212  if (num_states <= 1)
213  KALDI_ERR << "HmmTopology::Check(), cannot only have one state (i.e., must "
214  "have at least one emitting state).";
215  if (!entries_[i][num_states-1].transitions.empty())
216  KALDI_ERR << "HmmTopology::Check(), last state must have no transitions.";
217  // not sure how necessary this next stipulation is.
218  if (entries_[i][num_states-1].pdf_class != kNoPdf)
219  KALDI_ERR << "HmmTopology::Check(), last state must not be emitting.";
220 
221  std::vector<bool> has_trans_in(num_states, false);
222  std::vector<int32> seen_pdf_classes;
223 
224  for (int32 j = 0; j < num_states; j++) { // j is the state-id.
225  BaseFloat tot_prob = 0.0;
226  if (entries_[i][j].pdf_class != kNoPdf)
227  seen_pdf_classes.push_back(entries_[i][j].pdf_class);
228  std::set<int32> seen_transition;
229  for (int32 k = 0;
230  static_cast<size_t>(k) < entries_[i][j].transitions.size();
231  k++) {
232  tot_prob += entries_[i][j].transitions[k].second;
233  if (entries_[i][j].transitions[k].second <= 0.0)
234  KALDI_ERR << "HmmTopology::Check(), negative or zero transition prob.";
235  int32 dst_state = entries_[i][j].transitions[k].first;
236  // The commented code in the next few lines disallows a completely
237  // skippable phone, as this would cause to stop working some mechanisms
238  // that are being built, which enable the creation of phone-level lattices
239  // and rescoring these with a different lexicon and LM.
240  if (dst_state == num_states-1 // && j != 0
241  && entries_[i][j].pdf_class == kNoPdf)
242  KALDI_ERR << "We do not allow any state to be "
243  "nonemitting and have a transition to the final-state (this would "
244  "stop the SplitToPhones function from identifying the last state "
245  "of a phone.";
246  if (dst_state < 0 || dst_state >= num_states)
247  KALDI_ERR << "HmmTopology::Check(), invalid dest state " << (dst_state);
248  if (seen_transition.count(dst_state) != 0)
249  KALDI_ERR << "HmmTopology::Check(), duplicate transition found.";
250  if (dst_state == k) { // self_loop...
251  KALDI_ASSERT(entries_[i][j].pdf_class != kNoPdf && "Nonemitting states cannot have self-loops.");
252  }
253  seen_transition.insert(dst_state);
254  has_trans_in[dst_state] = true;
255  }
256  if (j+1 < num_states) {
257  KALDI_ASSERT(tot_prob > 0.0 && "Non-final state must have transitions out."
258  "(with nonzero probability)");
259  if (fabs(tot_prob - 1.0) > 0.01)
260  KALDI_WARN << "Total probability for state " << j <<
261  " in topology entry is " << tot_prob;
262  } else
263  KALDI_ASSERT(tot_prob == 0.0);
264  }
265  // make sure all but start state have input transitions.
266  for (int32 j = 1; j < num_states; j++)
267  if (!has_trans_in[j])
268  KALDI_ERR << "HmmTopology::Check, state "<<(j)<<" has no input transitions.";
269  SortAndUniq(&seen_pdf_classes);
270  if (seen_pdf_classes.front() != 0 ||
271  seen_pdf_classes.back() != static_cast<int32>(seen_pdf_classes.size()) - 1) {
272  KALDI_ERR << "HmmTopology::Check(), pdf_classes are expected to be "
273  "contiguous and start from zero.";
274  }
275  }
276 }

294 int32 HmmTopology::MinLength(int32 phone) const {
295  const TopologyEntry &entry = TopologyForPhone(phone);
296  // min_length[state] gives the minimum length for sequences up to and
297  // including that state.
298  std::vector<int32> min_length(entry.size(),
299  std::numeric_limits<int32>::max());
300  KALDI_ASSERT(!entry.empty());
301 
302  min_length[0] = (entry[0].pdf_class == -1 ? 0 : 1);
303  int32 num_states = min_length.size();
304  bool changed = true;
305  while (changed) {
306  changed = false;
307  for (int32 s = 0; s < num_states; s++) {
308  const HmmState &this_state = entry[s];
309  std::vector<std::pair<int32, BaseFloat> >::const_iterator
310  iter = this_state.transitions.begin(),
311  end = this_state.transitions.end();
312  for (; iter != end; ++iter) {
313  int32 next_state = iter->first;
314  KALDI_ASSERT(next_state < num_states);
315  int32 next_state_min_length = min_length[s] +
316  (entry[next_state].pdf_class == -1 ? 0 : 1);
317  if (next_state_min_length < min_length[next_state]) {
318  min_length[next_state] = next_state_min_length;
319  if (next_state < s)
320  changed = true;
321  // the test of 'next_state < s' is an optimization for speed.
322  }
323  }
324  }
325  }
326  KALDI_ASSERT(min_length.back() != std::numeric_limits<int32>::max());
327  // the last state is the final-state.
328  return min_length.back();
329 }
