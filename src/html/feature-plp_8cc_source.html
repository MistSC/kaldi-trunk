160 void Plp::ComputeInternal(const VectorBase<BaseFloat> &wave,
161  const MelBanks &mel_banks,
162  const Vector<BaseFloat> &equal_loudness,
163  Matrix<BaseFloat> *output,
164  Vector<BaseFloat> *wave_remainder) const {
165  KALDI_ASSERT(output != NULL);
166  int32 rows_out = NumFrames(wave.Dim(), opts_.frame_opts),
167  cols_out = opts_.num_ceps;
168  if (rows_out == 0) {
169  output->Resize(0, 0);
170  if (wave_remainder != NULL)
171  *wave_remainder = wave;
172  return;
173  }
174  output->Resize(rows_out, cols_out);
175  if (wave_remainder != NULL)
176  ExtractWaveformRemainder(wave, opts_.frame_opts, wave_remainder);
177  Vector<BaseFloat> window; // windowed waveform.
178  int32 num_mel_bins = opts_.mel_opts.num_bins;
179  Vector<BaseFloat> mel_energies(num_mel_bins);
180  Vector<BaseFloat> mel_energies_duplicated(num_mel_bins+2);
181  Vector<BaseFloat> autocorr_coeffs(opts_.lpc_order+1);
182  Vector<BaseFloat> lpc_coeffs(opts_.lpc_order);
183  Vector<BaseFloat> raw_cepstrum(opts_.lpc_order); // not including C0,
184  // and size may differ from final size.
185  Vector<BaseFloat> final_cepstrum(opts_.num_ceps);
186  std::vector<BaseFloat> temp_buffer; // used by srfft.
187 
188  KALDI_ASSERT(opts_.num_ceps <= opts_.lpc_order+1); // our num-ceps includes C0.
189  for (int32 r = 0; r < rows_out; r++) { // r is frame index..
190  BaseFloat log_energy;
191  ExtractWindow(wave, r, opts_.frame_opts,
192  feature_window_function_, &window,
193  (opts_.use_energy && opts_.raw_energy ? &log_energy : NULL));
194 
195  if (opts_.use_energy && !opts_.raw_energy)
196  log_energy = Log(std::max(VecVec(window, window),
197  std::numeric_limits<BaseFloat>::min()));
198 
199  if (srfft_ != NULL) // Compute FFT using split-radix algorithm.
200  srfft_->Compute(window.Data(), true, &temp_buffer);
201  else // An alternative algorithm that works for non-powers-of-two.
202  RealFft(&window, true);
203 
204  // Convert the FFT into a power spectrum.
205  ComputePowerSpectrum(&window); // elements 0 ... window.Dim()/2
206 
207  SubVector<BaseFloat> power_spectrum(window, 0, window.Dim()/2 + 1);
208 
209  mel_banks.Compute(power_spectrum, &mel_energies);
210 
211  mel_energies.MulElements(equal_loudness);
212 
213  mel_energies.ApplyPow(opts_.compress_factor);
214 
215  // duplicate first and last elements.
216  {
217  SubVector<BaseFloat> v(mel_energies_duplicated, 1, num_mel_bins);
218  v.CopyFromVec(mel_energies);
219  }
220  mel_energies_duplicated(0) = mel_energies(0);
221  mel_energies_duplicated(num_mel_bins+1) = mel_energies(num_mel_bins-1);
222 
223  autocorr_coeffs.AddMatVec(1.0, idft_bases_, kNoTrans,
224  mel_energies_duplicated, 0.0);
225 
226  BaseFloat energy = ComputeLpc(autocorr_coeffs, &lpc_coeffs);
227 
228  energy = std::max(energy,
229  std::numeric_limits<BaseFloat>::min());
230 
231  Lpc2Cepstrum(opts_.lpc_order, lpc_coeffs.Data(), raw_cepstrum.Data());
232  {
233  SubVector<BaseFloat> dst(final_cepstrum, 1, opts_.num_ceps-1);
234  SubVector<BaseFloat> src(raw_cepstrum, 0, opts_.num_ceps-1);
235  dst.CopyFromVec(src);
236  final_cepstrum(0) = energy;
237  }
238 
239  if (opts_.cepstral_lifter != 0.0)
240  final_cepstrum.MulElements(lifter_coeffs_);
241 
242  if (opts_.cepstral_scale != 1.0)
243  final_cepstrum.Scale(opts_.cepstral_scale);
244 
245  if (opts_.use_energy) {
246  if (opts_.energy_floor > 0.0 && log_energy < log_energy_floor_)
247  log_energy = log_energy_floor_;
248  final_cepstrum(0) = log_energy;
249  }
250 
251  if (opts_.htk_compat) {
252  BaseFloat energy = final_cepstrum(0);
253  for (int32 i = 0; i < opts_.num_ceps-1; i++)
254  final_cepstrum(i) = final_cepstrum(i+1);
255  // if (!opts_.use_energy)
256  // energy *= M_SQRT2; // scale on C0 (actually removing scale
257  // we previously added that's part of one common definition of
258  // cosine transform.)
259  final_cepstrum(opts_.num_ceps-1) = energy;
260  }
261 
262  output->Row(r).CopyFromVec(final_cepstrum);
263  // std::cout << "FIN" << final_cepstrum;
264  }
265 }
