51 bool Semaphore::TryWait() {
52  int32 ret = 0;
53  bool try_wait_succeeded = false;
54  ret |= pthread_mutex_lock(&mutex_);
55  if (counter_ > 0) {
56  counter_--;
57  try_wait_succeeded = true;
58  }
59  ret |= pthread_mutex_unlock(&mutex_);
60  if (ret != 0) {
61  KALDI_ERR << "Error in pthreads";
62  }

68 void Semaphore::Wait() {
69  int32 ret = 0;
