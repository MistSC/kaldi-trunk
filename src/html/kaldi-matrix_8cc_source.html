33 void MatrixBase<Real>::Invert(Real *log_det, Real *det_sign,
34  bool inverse_needed) {
35  KALDI_ASSERT(num_rows_ == num_cols_);
36  if (num_rows_ == 0) {
37  if (det_sign) *det_sign = 1;
38  if (log_det) *log_det = 0.0;
39  return;
40  }
41 #ifndef HAVE_ATLAS
42  KaldiBlasInt *pivot = new KaldiBlasInt[num_rows_];
43  KaldiBlasInt M = num_rows_;
44  KaldiBlasInt N = num_cols_;
45  KaldiBlasInt LDA = stride_;
46  KaldiBlasInt result = -1;
47  KaldiBlasInt l_work = std::max<KaldiBlasInt>(1, N);
48  Real *p_work;
49  void *temp;
50  if ((p_work = static_cast<Real*>(
51  KALDI_MEMALIGN(16, sizeof(Real)*l_work, &temp))) == NULL)
52  throw std::bad_alloc();
53 
54  clapack_Xgetrf2(&M, &N, data_, &LDA, pivot, &result);
55  const int pivot_offset = 1;
56 #else
57  int *pivot = new int[num_rows_];
58  int result;
59  clapack_Xgetrf(num_rows_, num_cols_, data_, stride_, pivot, &result);
60  const int pivot_offset = 0;
61 #endif
62  KALDI_ASSERT(result >= 0 && "Call to CLAPACK sgetrf_ or ATLAS clapack_sgetrf "
63  "called with wrong arguments");
64  if (result > 0) {
65  if (inverse_needed) {
66  KALDI_ERR << "Cannot invert: matrix is singular";
67  } else {
68  if (log_det) *log_det = -std::numeric_limits<Real>::infinity();
69  if (det_sign) *det_sign = 0;
70  delete[] pivot;
71  return;
72  }
73  }
74  if (det_sign != NULL) {
75  int sign = 1;
76  for (MatrixIndexT i = 0; i < num_rows_; i++)
77  if (pivot[i] != static_cast<int>(i) + pivot_offset) sign *= -1;
78  *det_sign = sign;
79  }
80  if (log_det != NULL || det_sign != NULL) { // Compute log determinant.
81  if (log_det != NULL) *log_det = 0.0;
82  Real prod = 1.0;
83  for (MatrixIndexT i = 0; i < num_rows_; i++) {
84  prod *= (*this)(i, i);
85  if (i == num_rows_ - 1 || std::fabs(prod) < 1.0e-10 ||
86  std::fabs(prod) > 1.0e+10) {
87  if (log_det != NULL) *log_det += Log(std::fabs(prod));
88  if (det_sign != NULL) *det_sign *= (prod > 0 ? 1.0 : -1.0);
89  prod = 1.0;
90  }
91  }
92  }
93 #ifndef HAVE_ATLAS
94  if (inverse_needed) clapack_Xgetri2(&M, data_, &LDA, pivot, p_work, &l_work,
95  &result);
96  delete[] pivot;
97  KALDI_MEMALIGN_FREE(p_work);
98 #else
99  if (inverse_needed)
100  clapack_Xgetri(num_rows_, data_, stride_, pivot, &result);
101  delete [] pivot;
102 #endif
103  KALDI_ASSERT(result == 0 && "Call to CLAPACK sgetri_ or ATLAS clapack_sgetri "
104  "called with wrong arguments");
105 }

345 template<typename Real>
346 void MatrixBase<Real>::AddMat(const Real alpha, const MatrixBase<Real>& A,
347  MatrixTransposeType transA) {
348  if (&A == this) {
349  if (transA == kNoTrans) {
350  Scale(alpha + 1.0);
351  } else {
352  KALDI_ASSERT(num_rows_ == num_cols_ && "AddMat: adding to self (transposed): not symmetric.");
353  Real *data = data_;
354  if (alpha == 1.0) { // common case-- handle separately.
355  for (MatrixIndexT row = 0; row < num_rows_; row++) {
356  for (MatrixIndexT col = 0; col < row; col++) {
357  Real *lower = data + (row * stride_) + col, *upper = data + (col
358  * stride_) + row;
359  Real sum = *lower + *upper;
360  *lower = *upper = sum;
361  }
362  *(data + (row * stride_) + row) *= 2.0; // diagonal.
363  }
364  } else {
365  for (MatrixIndexT row = 0; row < num_rows_; row++) {
366  for (MatrixIndexT col = 0; col < row; col++) {
367  Real *lower = data + (row * stride_) + col, *upper = data + (col
368  * stride_) + row;
369  Real lower_tmp = *lower;
370  *lower += alpha * *upper;
371  *upper += alpha * lower_tmp;
372  }
373  *(data + (row * stride_) + row) *= (1.0 + alpha); // diagonal.
374  }
375  }
376  }
377  } else {
378  int aStride = (int) A.stride_, stride = stride_;
379  Real *adata = A.data_, *data = data_;
380  if (transA == kNoTrans) {
381  KALDI_ASSERT(A.num_rows_ == num_rows_ && A.num_cols_ == num_cols_);
382  if (num_rows_ == 0) return;
383  for (MatrixIndexT row = 0; row < num_rows_; row++, adata += aStride,
384  data += stride) {
385  cblas_Xaxpy(num_cols_, alpha, adata, 1, data, 1);
386  }
387  } else {
388  KALDI_ASSERT(A.num_cols_ == num_rows_ && A.num_rows_ == num_cols_);
389  if (num_rows_ == 0) return;
390  for (MatrixIndexT row = 0; row < num_rows_; row++, adata++, data += stride)
391  cblas_Xaxpy(num_cols_, alpha, adata, aStride, data, 1);
392  }
393  }

691 template<typename Real>
692 void Matrix<Real>::Resize(const MatrixIndexT rows,
693  const MatrixIndexT cols,
694  MatrixResizeType resize_type) {
695  // the next block uses recursion to handle what we have to do if
696  // resize_type == kCopyData.
697  if (resize_type == kCopyData) {
698  if (this->data_ == NULL || rows == 0) resize_type = kSetZero; // nothing to copy.
699  else if (rows == this->num_rows_ && cols == this->num_cols_) { return; } // nothing to do.
700  else {
701  // set tmp to a matrix of the desired size; if new matrix
702  // is bigger in some dimension, zero it.
703  MatrixResizeType new_resize_type =
704  (rows > this->num_rows_ || cols > this->num_cols_) ? kSetZero : kUndefined;
705  Matrix<Real> tmp(rows, cols, new_resize_type);
706  MatrixIndexT rows_min = std::min(rows, this->num_rows_),
707  cols_min = std::min(cols, this->num_cols_);
708  tmp.Range(0, rows_min, 0, cols_min).
709  CopyFromMat(this->Range(0, rows_min, 0, cols_min));
710  tmp.Swap(this);
711  // and now let tmp go out of scope, deleting what was in *this.
712  return;
713  }
714  }
715  // At this point, resize_type == kSetZero or kUndefined.
716 
717  if (MatrixBase<Real>::data_ != NULL) {
718  if (rows == MatrixBase<Real>::num_rows_
719  && cols == MatrixBase<Real>::num_cols_) {
720  if (resize_type == kSetZero)
721  this->SetZero();
722  return;
723  }
724  else
725  Destroy();
726  }
727  Init(rows, cols);
728  if (resize_type == kSetZero) MatrixBase<Real>::SetZero();
729 }

1907 {

1909 
1910  KALDI_ASSERT(num_rows_ == num_cols_);
1911  KALDI_ASSERT(IsSymmetric() && "SymPosSemiDefEig: expecting input to be symmetrical.");

1928  SpMatrix<Real> tmpThis(D);
1929  tmpThis.AddMat2(1.0, tmpU, kNoTrans, 0.0);
1930  Matrix<Real> tmpThisFull(tmpThis);
1931  float new_norm = tmpThisFull.FrobeniusNorm();
1932  float old_norm = (*this).FrobeniusNorm();
1933  tmpThisFull.AddMat(-1.0, (*this));
1934

1937  if (std::abs(new_norm-old_norm) > old_norm*check_thresh || diff_norm > old_norm*check_thresh) {
