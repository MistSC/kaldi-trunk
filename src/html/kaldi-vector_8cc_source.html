188 template<typename Real>
189 void Vector<Real>::Resize(const MatrixIndexT dim, MatrixResizeType resize_type) {
190 
191  // the next block uses recursion to handle what we have to do if
192  // resize_type == kCopyData.
193  if (resize_type == kCopyData) {
194  if (this->data_ == NULL || dim == 0) resize_type = kSetZero; // nothing to copy.
195  else if (this->dim_ == dim) { return; } // nothing to do.
196  else {
197  // set tmp to a vector of the desired size.
198  Vector<Real> tmp(dim, kUndefined);
199  if (dim > this->dim_) {
200  memcpy(tmp.data_, this->data_, sizeof(Real)*this->dim_);
201  memset(tmp.data_+this->dim_, 0, sizeof(Real)*(dim-this->dim_));
202  } else {
203  memcpy(tmp.data_, this->data_, sizeof(Real)*dim);
204  }
205  tmp.Swap(this);
206  // and now let tmp go out of scope, deleting what was in *this.
207  return;
208  }
209  }
210  // At this point, resize_type == kSetZero or kUndefined.
211 
212  if (this->data_ != NULL) {
213  if (this->dim_ == dim) {
214  if (resize_type == kSetZero) this->SetZero();
215  return;
216  } else {
217  Destroy();
218  }
219  }
220  Init(dim);
221  if (resize_type == kSetZero) this->SetZero();
222 }

506 // Computes the p-th norm. Throws exception if could not.
507 template<typename Real>
508 Real VectorBase<Real>::Norm(Real p) const {
509  KALDI_ASSERT(p >= 0.0);
510  Real sum = 0.0;
511  if (p == 0.0) {
512  for (MatrixIndexT i = 0; i < dim_; i++)
513  if (data_[i] != 0.0) sum += 1.0;
514  return sum;
515  } else if (p == 1.0) {
516  for (MatrixIndexT i = 0; i < dim_; i++)
517  sum += std::abs(data_[i]);
518  return sum;
519  } else if (p == 2.0) {
520  for (MatrixIndexT i = 0; i < dim_; i++)
521  sum += data_[i] * data_[i];
522  return std::sqrt(sum);
523  } else {
524  Real tmp;
525  bool ok = true;
526  for (MatrixIndexT i = 0; i < dim_; i++) {
527  tmp = pow(std::abs(data_[i]), p);
528  if (tmp == HUGE_VAL) // HUGE_VAL is what pow returns on error.
529  ok = false;
530  sum += tmp;
531  }
532  tmp = pow(sum, static_cast<Real>(1.0/p));
533  KALDI_ASSERT(tmp != HUGE_VAL); // should not happen here.
534  if (ok) {
535  return tmp;
536  } else {
537  Real maximum = this->Max(), minimum = this->Min(),
538  max_abs = std::max(maximum, -minimum);
539  KALDI_ASSERT(max_abs > 0); // Or should not have reached here.
540  Vector<Real> tmp(*this);
541  tmp.Scale(1.0 / max_abs);
542  return tmp.Norm(p) * max_abs;
543  }
544  }

1084  Vector<Real> tmp(this->Dim());
1085  tmp.Read(is, binary, false); // read without adding.
1086  if (this->Dim() == 0) this->Resize(tmp.Dim());
1087  if (this->Dim() != tmp.Dim()) {
1088  KALDI_ERR << "Vector<Real>::Read, adding but dimensions mismatch "
1089  << this->Dim() << " vs. " << tmp.Dim();
1090  }
1091  this->AddVec(1.0, tmp);
1092  return;
1093  } // now assume add == false.
1094 
1095  std::ostringstream specific_error;
1096  MatrixIndexT pos_at_start = is.tellg();
1097 
1098  if (binary) {
1099  int peekval = Peek(is, binary);
1100  const char *my_token = (sizeof(Real) == 4 ? "FV" : "DV");
1101  char other_token_start = (sizeof(Real) == 4 ? 'D' : 'F');
1102  if (peekval == other_token_start) { // need to instantiate the other type to read it.
1103  typedef typename OtherReal<Real>::Real OtherType; // if Real == float, OtherType == double, and vice versa.
1104  Vector<OtherType> other(this->Dim());
1105  other.Read(is, binary, false); // add is false at this point.
1106  if (this->Dim() != other.Dim()) this->Resize(other.Dim());
1107  this->CopyFromVec(other);
1108  return;
1109  }
1110  std::string token;
1111  ReadToken(is, binary, &token);
1112  if (token != my_token) {
1113  specific_error << ": Expected token " << my_token << ", got " << token;
1114  goto bad;
1115  }
1116  int32 size;
1117  ReadBasicType(is, binary, &size); // throws on error.
1118  if ((MatrixIndexT)size != this->Dim()) this->Resize(size);
1119  if (size > 0)
1120  is.read(reinterpret_cast<char*>(this->data_), sizeof(Real)*size);
1121  if (is.fail()) {
1122  specific_error << "Error reading vector data (binary mode); truncated "
1123  "stream? (size = " << size << ")";
1124  goto bad;
1125  }
1126  return;
1127  } else { // Text mode reading; format is " [ 1.1 2.0 3.4 ]\n"
1128  std::string s;
1129  is >> s;
1130  // if ((s.compare("DV") == 0) || (s.compare("FV") == 0)) { // Back compatibility.
1131  // is >> s; // get dimension
1132  // is >> s; // get "["
1133  // }
1134  if (is.fail()) { specific_error << "EOF while trying to read vector."; goto bad; }
1135  if (s.compare("[]") == 0) { Resize(0); return; } // tolerate this variant.
1136  if (s.compare("[")) { specific_error << "Expected \"[\" but got " << s; goto bad; }
1137  std::vector<Real> data;
1138  while (1) {
1139  int i = is.peek();
1140  if (i == '-' || (i >= '0' && i <= '9')) { // common cases first.
1141  Real r;
1142  is >> r;
1143  if (is.fail()) { specific_error << "Failed to read number."; goto bad; }
1144  if (! std::isspace(is.peek()) && is.peek() != ']') {
1145  specific_error << "Expected whitespace after number."; goto bad;
1146  }
1147  data.push_back(r);
1148  // But don't eat whitespace... we want to check that it's not newlines
1149  // which would be valid only for a matrix.
1150  } else if (i == ' ' || i == '\t') {
1151  is.get();
1152  } else if (i == ']') {
1153  is.get(); // eat the ']'
1154  this->Resize(data.size());
1155  for (size_t j = 0; j < data.size(); j++)
1156  this->data_[j] = data[j];
1157  i = is.peek();
1158  if (static_cast<char>(i) == '\r') {
1159  is.get();
1160  is.get(); // get \r\n (must eat what we wrote)
1161  } else if (static_cast<char>(i) == '\n') { is.get(); } // get \n (must eat what we wrote)
1162  if (is.fail()) {
1163  KALDI_WARN << "After end of vector data, read error.";
1164  // we got the data we needed, so just warn for this error.
1165  }
1166  return; // success.
1167  } else if (i == -1) {
1168  specific_error << "EOF while reading vector data.";
1169  goto bad;
1170  } else if (i == '\n' || i == '\r') {
1171  specific_error << "Newline found while reading vector (maybe it's a matrix?)";
