144 
145  private:
146  double weight_; // sum of weights of the source vectors. Never negative.
147  Vector<double> stats_; // Equals the weighted sum of the source vectors.
148  double sumsq_; // Equals the sum over all sources, of weight_ * vec.vec,
149  // where vec = stats_ / weight_. Used in computing
150  // the objective function.
