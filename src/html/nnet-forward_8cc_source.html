176  << "Subtracting log-prior on 'probability-like' data in range [0..1] "
177  << "(Did you forget --no-softmax=true or --apply-log=true ?)";
178  }
179  pdf_prior.SubtractOnLogpost(&nnet_out);
180  }
181 
182  // download from GPU,

184  nnet_out.CopyToMat(&nnet_out_host);
185

187  if (time_shift > 0) {

189  nnet_out_host = tmp.RowRange(time_shift, tmp.NumRows() - time_shift);
190  }
191 
192  // write,
193  if (!KALDI_ISFINITE(nnet_out_host.Sum())) { // check there's no nan/inf,
194  KALDI_ERR << "NaN or inf found in final output nn-output for " << utt;
195  }
196  feature_writer.Write(feature_reader.Key(), nnet_out_host);
197 
198  // progress log
199  if (num_done % 100 == 0) {
200  time_now = time.Elapsed();
201  KALDI_VLOG(1) << "After " << num_done << " utterances: time elapsed = "
202  << time_now/60 << " min; processed " << tot_t/time_now
203  << " frames per second.";
204  }
205  num_done++;
206  tot_t += mat.NumRows();
207  }
208 
209  // final message
210  KALDI_LOG << "Done " << num_done << " files"
211  << " in " << time.Elapsed()/60 << "min,"
212  << " (fps " << tot_t/time.Elapsed() << ")";
213 
214 #if HAVE_CUDA==1
215  if (kaldi::g_kaldi_verbose_level >= 1) {
216  CuDevice::Instantiate().PrintProfile();
217  }
218 #endif
219 
220  if (num_done == 0) return -1;
221  return 0;
222  } catch(const std::exception &e) {
223  std::cerr << e.what();

225  }
