38  void AccStats(const NnetExample &eg) {
39  ComputationRequest request;
40  bool need_backprop = false, store_stats = false;
41  GetComputationRequest(nnet_, eg, need_backprop, store_stats, &request);
42  const NnetComputation &computation = *(compiler_.Compile(request));
43  NnetComputeOptions options;
44  if (GetVerboseLevel() >= 3)
45  options.debug = true;
46  NnetComputer computer(options, computation, nnet_, NULL);
47 
48  computer.AcceptInputs(nnet_, eg.io);
49  computer.Forward();
50  const CuMatrixBase<BaseFloat> &nnet_output = computer.GetOutput("output");
51  AccStatsFromOutput(eg, nnet_output);
52  }

54  void WriteStats(const std::string &stats_wxfilename, bool binary) {
55  if (lda_stats_.TotCount() == 0) {
56  KALDI_ERR << "Accumulated no stats.";
57  } else {
58  WriteKaldiObject(lda_stats_, stats_wxfilename, binary);
59  KALDI_LOG << "Accumulated stats, soft frame count = "
60  << lda_stats_.TotCount() << ". Wrote to "
61  << stats_wxfilename;
62  }
63  }
64  private:
65  void AccStatsFromOutput(const NnetExample &eg,
66  const CuMatrixBase<BaseFloat> &nnet_output) {
67  BaseFloat rand_prune = rand_prune_;
68  const NnetIo *output_supervision = NULL;
69  for (size_t i = 0; i < eg.io.size(); i++)
70  if (eg.io[i].name == "output")
71  output_supervision = &(eg.io[i]);
72  KALDI_ASSERT(output_supervision != NULL && "no output in eg named 'output'");
73  int32 num_rows = output_supervision->features.NumRows(),
74  num_pdfs = output_supervision->features.NumCols();
75  KALDI_ASSERT(num_rows == nnet_output.NumRows());
76  if (lda_stats_.Dim() == 0)
77  lda_stats_.Init(num_pdfs, nnet_output.NumCols());
78  if (output_supervision->features.Type() != kSparseMatrix)
79  KALDI_ERR << "Expected supervision in eg to contain sparse matrices.";
80  const SparseMatrix<BaseFloat> &smat =
81  output_supervision->features.GetSparseMatrix();
82  for (int32 r = 0; r < num_rows; r++) {
83  // the following, transferring row by row to CPU, would be wasteful
84  // if we actually were using a GPU, but we don't anticipate doing this
85  // in this program.
86  CuSubVector<BaseFloat> cu_row(nnet_output, r);
87  // "row" is actually just a redudant copy, since we're likely on CPU,
88  // but we're about to do an outer product, so this doesn't dominate.
89  Vector<BaseFloat> row(cu_row);
90 
91  const SparseVector<BaseFloat> &post(smat.Row(r));
92  const std::pair<MatrixIndexT, BaseFloat> *post_data = post.Data(),
93  *post_end = post_data + post.NumElements();
94  for (; post_data != post_end; ++post_data) {
95  MatrixIndexT pdf = post_data->first;
96  BaseFloat weight = post_data->second;
97  BaseFloat pruned_weight = RandPrune(weight, rand_prune);
98  if (pruned_weight != 0.0)
99  lda_stats_.Accumulate(row, pdf, pruned_weight);
100  }
101  }
102  }

114 int main(int argc, char *argv[]) {
115  try {
116  using namespace kaldi;
117  using namespace kaldi::nnet3;
118  typedef kaldi::int32 int32;
119  typedef kaldi::int64 int64;
120 
121  const char *usage =
122  "Accumulate statistics in the same format as acc-lda (i.e. stats for\n"
123  "estimation of LDA and similar types of transform), starting from nnet\n"
124  "training examples. This program puts the features through the network,\n"
125  "and the network output will be the features; the supervision in the\n"
126  "training examples is used for the class labels. Used in obtaining\n"
127  "feature transforms that help nnet training work better.\n"
128  "\n"
129  "Usage: nnet3-acc-lda-stats [options] <raw-nnet-in> <training-examples-in> <lda-stats-out>\n"
130  "e.g.:\n"
131  "nnet3-acc-lda-stats 0.raw ark:1.egs 1.acc\n"
132  "See also: nnet-get-feature-transform\n";
133 
134  bool binary_write = true;
135  BaseFloat rand_prune = 0.0;
136 
137  ParseOptions po(usage);
138  po.Register("binary", &binary_write, "Write output in binary mode");
139  po.Register("rand-prune", &rand_prune,
140  "Randomized pruning threshold for posteriors");
141 
142  po.Read(argc, argv);
143 
144  if (po.NumArgs() != 3) {
145  po.PrintUsage();
146  exit(1);
147  }
148 
149  std::string nnet_rxfilename = po.GetArg(1),
150  examples_rspecifier = po.GetArg(2),
151  lda_accs_wxfilename = po.GetArg(3);
152 
153  Nnet nnet;
154  ReadKaldiObject(nnet_rxfilename, &nnet);
155 
156  NnetLdaStatsAccumulator accumulator(rand_prune, nnet);
157 
158  int64 num_egs = 0;
159 
160  SequentialNnetExampleReader example_reader(examples_rspecifier);
161  for (; !example_reader.Done(); example_reader.Next(), num_egs++)
162  accumulator.AccStats(example_reader.Value());
163 
164  KALDI_LOG << "Processed " << num_egs << " examples.";
165  // the next command will die if we accumulated no stats.
166  accumulator.WriteStats(lda_accs_wxfilename, binary_write);
167 
168  return 0;
169  } catch(const std::exception &e) {
170  std::cerr << e.what() << '\n';
171  return -1;
172  }
173 }
