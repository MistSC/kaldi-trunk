30 void GetWeights(const std::string &weights_str,
31  int32 num_inputs,
32  vector<BaseFloat> *weights) {
33  KALDI_ASSERT(num_inputs >= 1);
34  if (!weights_str.empty()) {
35  SplitStringToFloats(weights_str, ":", true, weights);
36  if (weights->size() != num_inputs) {
37  KALDI_ERR << "--weights option must be a colon-separated list "
38  << "with " << num_inputs << " elements, got: "
39  << weights_str;
40  }
41  } else {
42  for (int32 i = 0; i < num_inputs; i++)
43  weights->push_back(1.0 / num_inputs);
44  }
45  // normalize the weights to sum to one.
46  float weight_sum = 0.0;
47  for (int32 i = 0; i < num_inputs; i++)
48  weight_sum += (*weights)[i];
49  for (int32 i = 0; i < num_inputs; i++)
50  (*weights)[i] = (*weights)[i] / weight_sum;
51  if (fabs(weight_sum - 1.0) > 0.01) {
52  KALDI_WARN << "Normalizing weights to sum to one, sum was " << weight_sum;
53  }
54 }

58 std::vector<bool> GetSkipLayers(const std::string &skip_layers_str,
59  const int32 first_layer_idx,
60  const int32 last_layer_idx) {
61 
62  std::vector<bool> skip_layers(last_layer_idx, false);
63 
64  if (skip_layers_str.empty()) {
65  return skip_layers;
66  }
67 
68  std::vector<int> layer_indices;
69  bool ret = SplitStringToIntegers(skip_layers_str, ":", true, &layer_indices);
70  if (!ret) {
71  KALDI_ERR << "Cannot parse the skip layers specifier. It should be"
72  << "colon-separated list of integers";
73  }
74 
75  int min_elem = std::numeric_limits<int>().max(),
76  max_elem = std::numeric_limits<int>().min();
77 
78  std::vector<int>::iterator it;
79  for ( it = layer_indices.begin(); it != layer_indices.end(); ++it ) {
80  if ( *it < 0 )
81  *it = last_layer_idx + *it; // convert the negative indices to
82  // correct indices -- -1 would be the
83  // last one, -2 the one before the last
84  // and so on.
85  if (*it > max_elem)
86  max_elem = *it;
87 
88  if (*it < min_elem)
89  min_elem = *it;
90  }
91 
92  if (max_elem >= last_layer_idx) {
93  KALDI_ERR << "--skip-layers option has to be a colon-separated list"
94  << "of indices which are supposed to be skipped.\n"
95  << "Maximum expected index: " << last_layer_idx
96  << " got: " << max_elem ;
97  }
98  if (min_elem < first_layer_idx) {
99  KALDI_ERR << "--skip-layers option has to be a colon-separated list"
100  << "of indices which are supposed to be skipped.\n"
101  << "Minimum expected index: " << first_layer_idx
102  << " got: " << min_elem ;
103  }
104 
105  for ( it = layer_indices.begin(); it != layer_indices.end(); ++it ) {
106  skip_layers[*it] = true;
107  }
108  return skip_layers;
109 }

112 int main(int argc, char *argv[]) {
113  try {
114  using namespace kaldi;
115  using namespace kaldi::nnet2;
116  typedef kaldi::int32 int32;
117  typedef kaldi::int64 int64;
118 
119  const char *usage =
120  "This program averages (or sums, if --sum=true) the parameters over a\n"
121  "number of neural nets. If you supply the option --skip-last-layer=true,\n"
122  "the parameters of the last updatable layer are copied from <model1> instead\n"
123  "of being averaged (useful in multi-language scenarios).\n"
124  "The --weights option can be used to weight each model differently.\n"
125  "\n"
126  "Usage: nnet-am-average [options] <model1> <model2> ... <modelN> <model-out>\n"
127  "\n"
128  "e.g.:\n"
129  " nnet-am-average 1.1.nnet 1.2.nnet 1.3.nnet 2.nnet\n";

136  po.Register("binary", &binary_write, "Write output in binary mode");
137  string weights_str;
138  bool skip_last_layer = false;
139  string skip_layers_str;
140  po.Register("weights", &weights_str, "Colon-separated list of weights, one "
141  "for each input model. These will be normalized to sum to one.");
142  po.Register("skip-last-layer", &skip_last_layer, "If true, averaging of "
143  "the last updatable layer is skipped (result comes from model1)");
144  po.Register("skip-layers", &skip_layers_str, "Colon-separated list of "
145  "indices of the layers that should be skipped during averaging."
146  "Be careful: this parameter uses an absolute indexing of "
147  "layers, i.e. iterates over all components, not over updatable "
148  "ones only.");
149 
150  po.Read(argc, argv);
151 
152  if (po.NumArgs() < 2) {
153  po.PrintUsage();
154  exit(1);
155  }
156 
157  std::string

159  nnet_wxfilename = po.GetArg(po.NumArgs());
160 
161  TransitionModel trans_model1;
162  AmNnet am_nnet1;
163  {
164  bool binary_read;

168  }
169 
170  int32 num_inputs = po.NumArgs() - 1;
171

173  GetWeights(weights_str, num_inputs, &model_weights);
174 
175  int32 c_begin = 0,
176  c_end = (skip_last_layer ?
177  am_nnet1.GetNnet().LastUpdatableComponent() :
178  am_nnet1.GetNnet().NumComponents());
179  KALDI_ASSERT(c_end != -1 && "Network has no updatable components.");
180 
181  int32 last_layer_idx = am_nnet1.GetNnet().NumComponents();
182  vector<bool> skip_layers = GetSkipLayers(skip_layers_str,
183  0,
184  last_layer_idx);
185 
186  // scale the components - except the last layer, if skip_last_layer == true.
187  for (int32 c = c_begin; c < c_end; c++) {
188  if (skip_layers[c]) {
189  KALDI_VLOG(2) << "Not averaging layer " << c << " (as requested)";
190  continue;
191  }
192  bool updated = false;
193  UpdatableComponent *uc =
194  dynamic_cast<UpdatableComponent*>(&(am_nnet1.GetNnet().GetComponent(c)));
195  if (uc != NULL) {
196  KALDI_VLOG(2) << "Averaging layer " << c << " (UpdatableComponent)";
197  uc->Scale(model_weights[0]);
198  updated = true;
199  }
200  NonlinearComponent *nc =
201  dynamic_cast<NonlinearComponent*>(&(am_nnet1.GetNnet().GetComponent(c)));
202  if (nc != NULL) {
203  KALDI_VLOG(2) << "Averaging layer " << c << " (NonlinearComponent)";
204  nc->Scale(model_weights[0]);
205  updated = true;
206  }
207  if (! updated) {
208  KALDI_VLOG(2) << "Not averaging layer " << c
209  << " (unscalable component)";
210  }
211  }
212 
213  for (int32 i = 2; i <= num_inputs; i++) {
214  bool binary_read;
215  Input ki(po.GetArg(i), &binary_read);
216  TransitionModel trans_model;
217  trans_model.Read(ki.Stream(), binary_read);
218  AmNnet am_nnet;
219  am_nnet.Read(ki.Stream(), binary_read);
220 
221  for (int32 c = c_begin; c < c_end; c++) {
222  if (skip_layers[c]) continue;
223 
224  UpdatableComponent *uc_average =
225  dynamic_cast<UpdatableComponent*>(&(am_nnet1.GetNnet().GetComponent(c)));
226  const UpdatableComponent *uc_this =
227  dynamic_cast<const UpdatableComponent*>(&(am_nnet.GetNnet().GetComponent(c)));
228  if (uc_average != NULL) {
229  KALDI_ASSERT(uc_this != NULL &&
230  "Networks must have the same structure.");
231  uc_average->Add(model_weights[i-1], *uc_this);
232  }
233 
234  NonlinearComponent *nc_average =
235  dynamic_cast<NonlinearComponent*>(&(am_nnet1.GetNnet().GetComponent(c)));
236  const NonlinearComponent *nc_this =
237  dynamic_cast<const NonlinearComponent*>(&(am_nnet.GetNnet().GetComponent(c)));
238  if (nc_average != NULL) {
239  KALDI_ASSERT(nc_this != NULL &&
240  "Networks must have the same structure.");
241  nc_average->Add(model_weights[i-1], *nc_this);
242  }
243  }
244  }
245 
246  {
247  Output ko(nnet_wxfilename, binary_write);
248  trans_model1.Write(ko.Stream(), binary_write);
249  am_nnet1.Write(ko.Stream(), binary_write);
250  }
251 
252  KALDI_LOG << "Averaged parameters of " << num_inputs
253  << " neural nets, and wrote to " << nnet_wxfilename;
254  return 0; // it will throw an exception if there are any problems.
255  } catch(const std::exception &e) {
256  std::cerr << e.what() << '\n';
257  return -1;
258  }
259 }
