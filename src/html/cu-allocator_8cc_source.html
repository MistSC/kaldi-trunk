49 static inline size_t IntegerLog2(size_t i) {
50  size_t ans = 0;
51  while (i > 256) {
52  i >>= 8;
53  ans += 8;
54  }
55  while (i > 16) {
56  i >>= 4;
57  ans += 4;
58  }
59  while (i > 1) {
60  i >>= 1;
61  ans++;
62  }
63  return ans;
64 }

75 void* CuMemoryAllocator::MallocPitchInternal(size_t row_bytes,
76  size_t num_rows,
77  size_t *pitch) {
78  num_system_allocations_++;
79  void *ans;
80  cudaError_t e;
81  for (int32 i = 0; i <= 2; i++) {
82  if (num_rows != 1) {
83  Timer tim;
84  e = cudaMallocPitch(&ans, pitch, row_bytes, num_rows);
85  tot_time_taken_in_cuda_malloc_pitch_ += tim.Elapsed();
86  } else {
87  Timer tim;
88  // we might save a little time this way.
89  e = cudaMalloc(&ans, row_bytes);
90  tot_time_taken_in_cuda_malloc_ += tim.Elapsed();
91  *pitch = row_bytes;
92  }
93  if (e != cudaSuccess) {
94  PrintMemoryUsage();
95  // On the first 2 out of the 3 iters, try freeing memory.
96  if (i <= 1) {
97  KALDI_WARN << "Allocation of " << row_bytes << " x "
98  << num_rows << " region failed: freeing some memory and "
99  << "trying again. ";
100  BaseFloat new_memory_factor = 1.1;
101  if (opts_.memory_factor > new_memory_factor) {
102  KALDI_LOG << "To avoid future problems like this, changing "
103  << "memory_factor from " << opts_.memory_factor << " to "
104  << new_memory_factor;
105  opts_.memory_factor = new_memory_factor;
106  }
107  size_t memory_cached = MemoryCached(),
108  memory_requested = row_bytes * num_rows,
109  memory_to_free = std::max<size_t>(memory_cached / 2,
110  std::min<size_t>(memory_cached,
111  memory_requested));
112  FreeSomeCachedMemory(memory_to_free);
113  } else {
114  KALDI_ERR << "Cannot allocate the requested memory ("
115  << row_bytes << " x " << num_rows << " = "
116  << row_bytes * num_rows << " bytes)";
117  }
118  cudaGetLastError(); // Clear the error state.
119  } else {
120  break;
121  }
122  }
123  return ans;
124 }

155 void* CuMemoryAllocator::MallocPitch(size_t row_bytes,
156  size_t num_rows,
157  size_t *pitch) {
158  Timer tim;
159  t_++;
160  num_user_allocations_++;
161  size_t requested_bytes = row_bytes * num_rows;
162  if (cur_bytes_used_ + requested_bytes > max_bytes_used_)
163  max_bytes_used_ = cur_bytes_used_ + requested_bytes;
164  MruCache &cache = GetCacheForSize(requested_bytes);
165  MemoryRequest request(row_bytes, num_rows);
166  CachedMemoryElement output;
167  if (cache.Lookup(request, &output)) {
168  // we have cached memory with this value.
169  void *ans = output.pointer;
170  *pitch = output.pitch;
171  used_map_[ans] = UsedMemoryElement(row_bytes, num_rows, output.pitch);
172  cur_bytes_used_ += requested_bytes;
173  tot_time_taken_in_malloc_pitch_ += tim.Elapsed();
174  return ans;
175  } else {
176  // note: it's important that we already updated max_bytes_used_.
177  size_t next_bytes_allocated = cur_bytes_allocated_ + requested_bytes,
178  max_bytes_to_allocate =
179  static_cast<size_t>(opts_.memory_factor * max_bytes_used_);
180  ssize_t bytes_overflow = next_bytes_allocated - max_bytes_to_allocate;
181  if (bytes_overflow > 0) {
182  // The amount we would have allocated, after fulfilling this request,
183  // would exceed our limits (we don't allow ourselves to allocate more than
184  // memory_factor times the maximum amount of memory the user ever owns
185  // during the lifetime of the program). So free some memory.
186  KALDI_ASSERT(bytes_overflow <= MemoryCached()); // sanity check.
187  FreeSomeCachedMemory(static_cast<size_t>(bytes_overflow));
188  KALDI_ASSERT(cur_bytes_allocated_ + requested_bytes <=
189  max_bytes_to_allocate);
190  }
191  void *ans = MallocPitchInternal(row_bytes, num_rows, pitch);
192  cur_bytes_allocated_ += requested_bytes;
193  if (cur_bytes_allocated_ > max_bytes_allocated_)
194  max_bytes_allocated_ = cur_bytes_allocated_;
195  used_map_[ans] = UsedMemoryElement(row_bytes, num_rows, *pitch);
196  cur_bytes_used_ += requested_bytes;
197  tot_time_taken_in_malloc_pitch_ += tim.Elapsed();
198  return ans;
199  }

202 void CuMemoryAllocator::FreeSomeCachedMemory(size_t bytes_to_free_in) {
203  Timer tim;
204  // the next few lines are responsible for increasing the amount of memory we
205  // are going to free, in case the user requested an amount that's very tiny
206  // compared with the total amount of memory ever used. This helps us
207  // to amortize the cost of visiting all of the buckets inside this code.
208  // (there are only 40 buckets so it's not so big, but we're being careful.
209  size_t bytes_cached = cur_bytes_allocated_ - cur_bytes_used_,
210  min_to_free = static_cast<size_t>(max_bytes_used_ * opts_.delete_factor);
211  size_t bytes_to_free = std::min(bytes_cached,
212  std::max(bytes_to_free_in, min_to_free)),
213  bytes_freed = 0;
214 
215  size_t num_caches = caches_.size(),
216  t = t_;
217  // size_factor contains the approximate (power-of-two) size of the pointers
218  // that each cache's pointers contain. The 'cost' of keeping any given pointer,
219  // we declare to be the time since we last used it multiplied by the size
220  // of the memory in the pointer.
221  std::vector<BaseFloat> size_factor(num_caches);
222  for (size_t i = 0, j=1; i < num_caches; i++, j *= 2)
223  size_factor[i] = j;
224 
225  std::priority_queue<std::pair<BaseFloat,int32> > queue;
226  // Set up the queue.
227  for (int32 i = 0; i < num_caches; i++) {
228  const MruCache &cache = caches_[i];
229  size_t cache_t = cache.LeastRecentTime();
230  if (cache_t > 0) { // t == 0 means the cache is empty.
231  size_t interval = t - cache_t;
232  BaseFloat cost = size_factor[i] * interval;
233  KALDI_ASSERT(interval > 0);
234  queue.push(std::pair<BaseFloat,int32>(cost, i));
235  }
236  }
237  while (bytes_freed < bytes_to_free) {
238  // If the following fails it means I made some kind of bookkeeping error,
239  // and most likely we are trying to free more memory than we really have
240  // cached.
241  KALDI_ASSERT(!queue.empty() && "Code error.");
242  std::pair<BaseFloat, int32> p = queue.top();
243  int32 cache_index = p.second;
244  MruCache &cache = caches_[cache_index];
245  queue.pop();
246  if (queue.empty()) {
247  while (bytes_freed < bytes_to_free) {
248  bytes_freed += cache.RemoveLeastRecentlyUsed();
249  }
250  } else {
251  BaseFloat next_worst_cost = queue.top().first;
252  while (1) {
253  bytes_freed += cache.RemoveLeastRecentlyUsed();
254  if (bytes_freed >= bytes_to_free)
255  break;
256  size_t least_recent_time = cache.LeastRecentTime();
257  if (least_recent_time == 0) // this cache is now empty
258  break;
259  size_t interval = t - least_recent_time;
260  KALDI_ASSERT(interval > 0);
261  BaseFloat cost = size_factor[cache_index] * interval;
262  if (cost < next_worst_cost) {
263  // There is another bucket that has worse cost than this,
264  // so stop processing this bucket-- but first put it
265  // back in the queue.
266  queue.push(std::pair<BaseFloat, int32>(cost, cache_index));
267  break;
268  }
269  }
270  }
271  }
272  KALDI_ASSERT(bytes_freed <= cur_bytes_allocated_);
273  cur_bytes_allocated_ -= bytes_freed;
274  tot_time_taken_in_cuda_free_ += tim.Elapsed();
275 }

277 void CuMemoryAllocator::Free(void *ptr) {
278  t_++;
279  unordered_map<void*, UsedMemoryElement, PointerHasher>::iterator iter =
280  used_map_.find(ptr);
281  if (iter == used_map_.end()) {
282  KALDI_ERR << "Attempt to free CUDA memory pointer that was not allocated: "
283  << ptr;
284  }
285  const UsedMemoryElement &elem = iter->second;
286  size_t num_bytes = elem.row_bytes * elem.num_rows;
287 
288  cur_bytes_used_ -= num_bytes;
289  MruCache &cache = GetCacheForSize(num_bytes);
290 
291  cache.Insert(MemoryRequest(elem.row_bytes, elem.num_rows),
292  CachedMemoryElement(ptr, t_, elem.pitch));
293  used_map_.erase(iter);
294 }

296 size_t CuMemoryAllocator::MruCache::LeastRecentTime() const {
297  if (list_.empty()) {
298  KALDI_PARANOID_ASSERT(map_.empty());
299  return 0;
300  } else {
301  const MemoryRequest &mr = list_.front();
302  MapType::const_iterator iter = map_.find(mr);
303  KALDI_ASSERT(iter != map_.end());
304  const MapValueType &queue = iter->second;
305  KALDI_ASSERT(!queue.empty());
306  return queue.front().first.t;
307  }

310 bool CuMemoryAllocator::MruCache::Lookup(const MemoryRequest &request,
311  CachedMemoryElement *output) {
312  MapType::iterator iter = map_.find(request);
313  if (iter == map_.end())
314  return false;
315  MapValueType &q = iter->second;
316  KALDI_ASSERT(!q.empty());
317  // use q.back() as we want to return the most recently used one if there
318  // is a choice. We believe this will give better caching behavior.
319  *output = q.back().first;
320  list_.erase(q.back().second);
321  q.pop_back();
322  if (q.empty())
323  map_.erase(request);
324  return true;
325 }

335 size_t CuMemoryAllocator::MruCache::RemoveLeastRecentlyUsed() {
336  // Remove least-recently-used element from cache.
337  KALDI_ASSERT(!list_.empty());
338  MemoryRequest request = list_.front();
339  MapType::iterator iter = map_.find(request);
340  KALDI_ASSERT(iter != map_.end());
341  MapValueType &queue = iter->second;
342  KALDI_ASSERT(!queue.empty());
343  // least recently used elements are at the front of the queue.
344  std::pair<CachedMemoryElement, ListIterType> &p = queue.front();
345  KALDI_ASSERT(p.second == list_.begin());
346  CU_SAFE_CALL(cudaFree(p.first.pointer));
347  queue.pop_front();
348  if (queue.empty())
349  map_.erase(request);
350  list_.pop_front();
351  return request.first * request.second;
352 }
