101 void TestableArpaFileParser::ConsumeNGram(const NGram& ngram) {
102  KALDI_ASSERT(header_available_);
103  KALDI_ASSERT(!read_complete_);
104  KALDI_ASSERT(ngram.words.size() <= NgramCounts().size());
105  KALDI_ASSERT(ngram.words.size() >= last_order_);
106  last_order_ = ngram.words.size();
107 
108  NGramTestData entry = { 0 };
109  entry.line_number = LineNumber();
110  entry.logprob = ngram.logprob;
111  entry.backoff = ngram.backoff;
112  std::copy(ngram.words.begin(), ngram.words.end(), entry.words);
113  ngrams_.push_back(entry);
114 }

123 bool CompareNgrams(const NGramTestData& actual,
124  const NGramTestData& expected) {
125  if (actual.line_number != expected.line_number
126  || !std::equal(actual.words, actual.words + kMaxOrder,
127  expected.words)
128  || !ApproxEqual(actual.logprob, expected.logprob)
129  || !ApproxEqual(actual.backoff, expected.backoff)) {
130  KALDI_WARN << "Actual n-gram [" << actual
131  << "] differs from expected [" << expected << "]";
132  return false;
133  }
134  return true;
135 }
