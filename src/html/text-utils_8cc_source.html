61 void SplitStringToVector(const std::string &full, const char *delim,
62  bool omit_empty_strings,
63  std::vector<std::string> *out) {
64  size_t start = 0, found = 0, end = full.size();
65  out->clear();
66  while (found != std::string::npos) {
67  found = full.find_first_of(delim, start);
68  // start != end condition is for when the delimiter is at the end
69  if (!omit_empty_strings || (found != start && start != end))
70  out->push_back(full.substr(start, found - start));
71  start = found + 1;
72  }
73 }

103 bool IsToken(const std::string &token) {
104  size_t l = token.length();
105  if (l == 0) return false;
106  for (size_t i = 0; i < l; i++) {
107  unsigned char c = token[i];
108  if ((!isprint(c) || isspace(c)) && (isascii(c) || c == (unsigned char)255))
109  return false;
110  // The "&& (isascii(c) || c == 255)" was added so that we won't reject
111  // non-ASCII characters such as French characters with accents [except for
112  // 255 which is "nbsp", a form of space].
113  }
114  return true;
115 }

118 void SplitStringOnFirstSpace(const std::string &str,
119  std::string *first,
120  std::string *rest) {
121  const char *white_chars = " \t\n\r\f\v";
122  typedef std::string::size_type I;
123  const I npos = std::string::npos;
124  I first_nonwhite = str.find_first_not_of(white_chars);
125  if (first_nonwhite == npos) {
126  first->clear();
127  rest->clear();
128  return;
129  }
130  // next_white is first whitespace after first nonwhitespace.
131  I next_white = str.find_first_of(white_chars, first_nonwhite);
132 
133  if (next_white == npos) { // no more whitespace...
134  *first = std::string(str, first_nonwhite);
135  rest->clear();
136  return;
137  }
138  I next_nonwhite = str.find_first_not_of(white_chars, next_white);
139  if (next_nonwhite == npos) {
140  *first = std::string(str, first_nonwhite, next_white-first_nonwhite);
141  rest->clear();
142  return;
143  }
144 
145  I last_nonwhite = str.find_last_not_of(white_chars);
146  KALDI_ASSERT(last_nonwhite != npos); // or coding error.
147 
148  *first = std::string(str, first_nonwhite, next_white-first_nonwhite);
149  *rest = std::string(str, next_nonwhite, last_nonwhite+1-next_nonwhite);
150 }

152 bool IsLine(const std::string &line) {
153  if (line.find('\n') != std::string::npos) return false;
154  if (line.empty()) return true;
155  if (isspace(*(line.begin()))) return false;
156  if (isspace(*(line.rbegin()))) return false;
157  std::string::const_iterator iter = line.begin(), end = line.end();
158  for (; iter != end; iter++)
159  if (!isprint(*iter)) return false;
160  return true;
161 }

173 inline bool is_nan_text(const std::string &in, const std::string &prefix) {
174  if (in.size() < prefix.size())
175  return false;
176 
177  if (stricmp(in, prefix))
178  return true;
179 
180  for (int i = 0; i < prefix.size(); ++i)
181  if (tolower(in[i]) != tolower(prefix[i]))
182  return false;
183 
184  for (int i = prefix.size(); i < in.size(); ++i)
185  if (!isalpha(in[i]) && (in[i] != '_'))
186  return false;
187 
188  return true;
189 }

191 template<typename T>
192 bool convert_special_number(const std::string &str, T *out) {
193  if (stricmp(str, "infinity") || stricmp(str, "inf") ||
194  starts_with(str, "1.#INF")) {
195  *out = std::numeric_limits<T>::infinity();
196  return true;
197  } else if (stricmp(str, "-infinity") || stricmp(str, "-inf") ||
198  starts_with(str, "-1.#INF")) {
199  *out = -std::numeric_limits<T>::infinity();
200  return true;
201  } else if (is_nan_text(str, "nan") || starts_with(str, "1.#QNAN")) {
202  *out = std::numeric_limits<T>::quiet_NaN();
203  return true;
204  } else if (is_nan_text(str, "-nan") || starts_with(str, "-1.#QNAN")) {
205  *out = -std::numeric_limits<T>::quiet_NaN();
206  return true;
207  }
208  return false;
209 }

212  double *out) {
213  const char *this_str = str.c_str();
214  char *end = NULL;
215  errno = 0;
216

225  if (end != this_str)
226  while (isspace(*end)) end++;

228  return false;
