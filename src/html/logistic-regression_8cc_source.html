27 void LogisticRegression::Train(const Matrix<BaseFloat> &xs,
28  const std::vector<int32> &ys,
29  const LogisticRegressionConfig &conf) {
30 
31  int32 xs_num_rows = xs.NumRows(), xs_num_cols = xs.NumCols(),
32  num_ys = ys.size();
33  KALDI_ASSERT(xs_num_rows == num_ys);
34 
35  // Adding on extra column for each x to handle the prior.
36  Matrix<BaseFloat> xs_with_prior(xs_num_rows, xs_num_cols + 1);
37  SubMatrix<BaseFloat> sub_xs(xs_with_prior, 0, xs_num_rows, 0, xs_num_cols);
38  sub_xs.CopyFromMat(xs);
39 
40  int32 num_classes = *std::max_element(ys.begin(), ys.end()) + 1;
41  weights_.Resize(num_classes, xs_num_cols + 1);
42  Matrix<BaseFloat> xw(xs_num_rows, num_classes);
43 
44  // Adding on extra column for each x to handle the prior.
45  for (int32 i = 0; i < xs_num_rows; i++) {
46  xs_with_prior(i, xs_num_cols) = 1.0;
47  }
48 
49  // At the beginning of training we have no mixture components,
50  // therefore class_ is the "identity" mapping, that is
51  // class_[i] = i.
52  for (int32 i = 0; i < num_classes; i++) {
53  class_.push_back(i);
54  }
55 
56  weights_.SetZero();
57  TrainParameters(xs_with_prior, ys, conf, &xw);
58  KALDI_LOG <<
59  "Finished training parameters without mixture components." << std::endl;
60 
61  // If we are using mixture components, we add those components
62  // in MixUp and retrain with the extra weights.
63  if (conf.mix_up > num_classes) {
64  MixUp(ys, num_classes, conf);
65  Matrix<BaseFloat> xw(xs_num_rows, weights_.NumRows());
66  TrainParameters(xs_with_prior, ys, conf, &xw);
67  KALDI_LOG <<
68  "Finished training mixture components." << std::endl;
69  }
70 }

73 void LogisticRegression::MixUp(const std::vector<int32> &ys,
74  const int32 &num_classes,
75  const LogisticRegressionConfig &conf) {
76 
77  Vector<BaseFloat> counts(num_classes);
78  for (int32 i = 0; i < ys.size(); i++) {
79  counts(ys[i]) += 1.0;
80  }
81 
82  // TODO: Figure out what min_count should be
83  int32 min_count = 1;
84  std::vector<int32> targets;
85  GetSplitTargets(counts, conf.mix_up, conf.power, min_count, &targets);
86  int32 new_dim = std::accumulate(targets.begin(), targets.end(),
87  static_cast<int32>(0));
88 
89  KALDI_LOG << "Target number mixture components was " << conf.mix_up
90  << ". Training " << new_dim << " mixture components. "
91  << std::endl;
92 
93  int32 old_dim = weights_.NumRows(),
94  num_components = old_dim,
95  num_feats = weights_.NumCols();
96 
97  Matrix<BaseFloat> old_weights(weights_);
98  weights_.Resize(new_dim, num_feats);
99  SubMatrix<BaseFloat> sub_weights(weights_, 0, num_classes, 0, num_feats);
100  // We need to retain the original weights
101  sub_weights.CopyFromMat(old_weights);
102  class_.resize(new_dim);
103  // For each class i
104  for (int32 i = 0; i < targets.size(); i++) {
105  int32 mixes = targets[i];
106  // We start at j = 1 since one copy of the components already
107  // exists in weights_.
108  for (int32 j = 1; j < mixes; j++) {
109  int32 offset = num_components;
110  weights_.Row(offset).CopyRowFromMat(weights_, i);
111  Vector<BaseFloat> noise(num_feats);
112  noise.SetRandn();
113  weights_.Row(offset).AddVec(1.0e-05, noise);
114  class_[offset] = i; // The class i maps to the row at offset
115  num_components += 1;
116  }
117  }
118 }

120 void LogisticRegression::TrainParameters(const Matrix<BaseFloat> &xs,
121  const std::vector<int32> &ys, const LogisticRegressionConfig &conf,
122  Matrix<BaseFloat> *xw) {
123  int32 max_steps = conf.max_steps;
124  BaseFloat normalizer = conf.normalizer;
125  LbfgsOptions lbfgs_opts;
126  lbfgs_opts.minimize = false;
127  // Get initial w vector
128  Vector<BaseFloat> init_w(weights_.NumRows() * weights_.NumCols());
129  init_w.CopyRowsFromMat(weights_);
130  OptimizeLbfgs<BaseFloat> lbfgs(init_w, lbfgs_opts);
131 
132  for (int32 step = 0; step < max_steps; step++) {
133  DoStep(xs, xw, ys, &lbfgs, normalizer);
134  }
135 
136  Vector<BaseFloat> best_w(lbfgs.GetValue());
137  weights_.CopyRowsFromVec(best_w);
138 }

140 void LogisticRegression::GetLogPosteriors(const Matrix<BaseFloat> &xs,
141  Matrix<BaseFloat> *log_posteriors) {
142  int32 xs_num_rows = xs.NumRows(),
143  xs_num_cols = xs.NumCols(),
144  num_mixes = weights_.NumRows();
145 
146  int32 num_classes = *std::max_element(class_.begin(), class_.end()) + 1;
147 
148  log_posteriors->Resize(xs_num_rows, num_classes);
149  Matrix<BaseFloat> xw(xs_num_rows, num_mixes);
150 
151  Matrix<BaseFloat> xs_with_prior(xs_num_rows, xs_num_cols + 1);
152  SubMatrix<BaseFloat> sub_xs(xs_with_prior, 0, xs_num_rows, 0, xs_num_cols);
153  sub_xs.CopyFromMat(xs);
154  // Adding on extra column for each x to handle the prior.
155  for (int32 i = 0; i < xs_num_rows; i++) {
156  xs_with_prior(i, xs_num_cols) = 1.0;
157  }
158  xw.AddMatMat(1.0, xs_with_prior, kNoTrans, weights_,
159  kTrans, 0.0);
160 
161  log_posteriors->Set(-std::numeric_limits<BaseFloat>::infinity());
162 
163  // i is the training example
164  for (int32 i = 0; i < xs_num_rows; i++) {
165  for (int32 j = 0; j < num_mixes; j++) {
166  int32 k = class_[j];
167  (*log_posteriors)(i,k) = LogAdd((*log_posteriors)(i,k), xw(i, j));
168  }
169  // Normalize the row.
170  log_posteriors->Row(i).Add(-xw.Row(i).LogSumExp());
171  }
172 }

199 BaseFloat LogisticRegression::DoStep(const Matrix<BaseFloat> &xs,
200  Matrix<BaseFloat> *xw,
201  const std::vector<int32> &ys, OptimizeLbfgs<BaseFloat> *lbfgs,
202  BaseFloat normalizer) {
203  Matrix<BaseFloat> gradient(weights_.NumRows(), weights_.NumCols());
204  // Vector form of the above matrix
205  Vector<BaseFloat> grad_vec(weights_.NumRows() * weights_.NumCols());
206 
207  // Calculate XW.T. The rows correspond to the x
208  // training examples and the columns to the class labels.
209  xw->AddMatMat(1.0, xs, kNoTrans, weights_, kTrans, 0.0);
210 
211  // Calculate both the gradient and the objective function.
212  BaseFloat objf = GetObjfAndGrad(xs, ys, *xw, &gradient, normalizer);
213 
214  // Convert gradient (a matrix) into a vector of size
215  // gradient.NumCols * gradient.NumRows.
216  grad_vec.CopyRowsFromMat(gradient);
217 
218  // Compute next step in L-BFGS.
219  lbfgs->DoStep(objf, grad_vec);
220 
221  // Update weights
222  Vector<BaseFloat> new_w(lbfgs->GetProposedValue());
223  weights_.CopyRowsFromVec(new_w);
224  KALDI_LOG << "Objective function is " << objf;
225  return objf;
226 }

228 BaseFloat LogisticRegression::GetObjfAndGrad(
229  const Matrix<BaseFloat> &xs,
230  const std::vector<int32> &ys, const Matrix<BaseFloat> &xw,
231  Matrix<BaseFloat> *grad, BaseFloat normalizer) {
232  BaseFloat raw_objf = 0.0;
233  int32 num_classes = *std::max_element(ys.begin(), ys.end()) + 1;
234  std::vector< std::vector<int32> > class_to_cols(num_classes, std::vector<int32>());
235  for (int32 i = 0; i < class_.size(); i++) {
236  class_to_cols[class_[i]].push_back(i);
237  }
238  // For each training example class
239  for (int32 i = 0; i < ys.size(); i++) {
240  Vector<BaseFloat> row(xw.NumCols());
241  row.CopyFromVec(xw.Row(i));
242  row.ApplySoftMax();
243  // Identify the rows of weights_ (which are a set of columns in wx)
244  // which correspond to class ys[i]
245  const std::vector<int32> &cols = class_to_cols[ys[i]];
246  SubVector<BaseFloat> x = xs.Row(i);
247  BaseFloat class_sum = 0.0;
248  for (int32 j = 0; j < cols.size(); j++) {
249  class_sum += row(cols[j]);
250  }
251  if (class_sum < 1.0e-20) class_sum = 1.0e-20;
252  raw_objf += Log(class_sum);
253  // Iterate over weights for each component. If there are no
254  // mixtures each row corresponds to a class.
255  for (int32 k = 0; k < weights_.NumRows(); k++) {
256  // p(y = k | x_i) where k is a component.
257  BaseFloat p = row(k);
258  if (class_[k] == ys[i]) {
259  // If the classes aren't split into mixture components
260  // then p/class_sum = 1.0.
261  grad->Row(k).AddVec(p/class_sum - p, x);
262  } else {
263  grad->Row(k).AddVec(-1.0 * p, x);
264  }
265  }
266  }
267  // Scale and add regularization term.
268  grad->Scale(1.0/ys.size());
269  grad->AddMat(-1.0 * normalizer, weights_);
270  raw_objf /= ys.size();
271  BaseFloat regularizer = - 0.5 * normalizer
272  * TraceMatMat(weights_, weights_, kTrans);
273  KALDI_VLOG(2) << "Objf is " << raw_objf << " + " << regularizer
274  << " = " << (raw_objf + regularizer);
275  return raw_objf + regularizer;
276 }
