117  // requested element of each output was computable. "computable" will have

128  void Prune();

130  // This enum says for each cindex_id, whether we can compute it from the given
131  // inputs or not. Note that there may be situations where before adding
132  // dependencies of a particular cindex_id we realize that we won't be able to
133  // use this cindex_id (i.e. it may be computable but it's not used) because
134  // its usable_count is zero, and in those cases we change the status to
135  // kWillNotCompute even though the cindex-id may be computable- for most
136  // purposes this status is treated the same as kNotComputable.
137  enum ComputableInfo {
138  kUnknown = 0,
139  kComputable = 1,
140  kNotComputable = 2,
141  kWillNotCompute = 3
142  };
143  private:
144  // This function, called from ExplainWhyNotComputable(), prints to "os"
145  // a human-readable form of a given cindex_id, that looks like
146  // some_network_node(n, t, x), e.g. "final_logsoftmax(0, -4, 0)".
147  void PrintCindexId(std::ostream &os, int32 cindex_id) const;
148 
149  // This function, typically to be called just before dying, prints logging
150  // information to explain why the given cindex_id is not computable.
151  void ExplainWhyNotComputable(int32 cindex_id) const;

184  // To be called when this cindex_id has just been newly added to graph_, this
185  // function adds various initial variables associated with it, to *this.
186  // is_input should be set to true if this cindex-id is being added as an input

195  void IncrementUsableCount(int32 cindex_id);
196 
197  // decrement the "usable" value of this cindex_id.
198  void DecrementUsableCount(int32 cindex_id);
199 
200  // This function, called from Prune(), modifies the members of
201  // graph_->dependencies-- it removes those cindexes that are not used in the
202  // computation for the current cindex_id. This will only do something
203  // interesting in cases where there are optional dependencies.
204  // It also clears the dependencies of those cindexes that are not computable.
205  void PruneDependencies(int32 cindex_id);
206 
207  // This function, called from Prune(), computes an array "required", with an
208  // element for each cindex_id that says whether it is required to compute the
209  // requested outputs. This is similar in function to the "usable_count_"
210  // array, but it's more exact because it's computed after we have done
211  // PruneDependencies() to remove unused dependencies, so it will only say
212  // something is required if it is really accessed in the computation.
213  // We'll later use this to remove unnecessary cindexes.
214  void ComputeRequiredArray(std::vector<bool> *required) const;
215 
216  // this function, to be called from Compute(), does some sanity checks to
217  // verify that the internal state is consistent.
218  void Check() const;

221  const ComputationRequest &request_;
222  ComputationGraph *graph_;
223 
224  // this is the transpose of graph_->dependencies; it tells us
225  // for each cindex_id, which other cindex_ids depend on it.
226  std::vector<std::vector<int32> > depend_on_this_;
227 
228  // this vector, indexed by cindex_id, contains our information about whether
229  // each cindex_id is computable; it's ComputableInfo, cast to char.
230  std::vector<char> computable_info_;
231 
232  // this is a queue of cindex_ids that we need to re-compute whether they are
233  // computable or not (because either they are new and haven't had dependencies
234  // added, or their dependencies' computable status has changed since we last
235  // computed their computable_ value).
236  std::deque<int32> computable_queue_;
237  // this vector tells us whether a cindex_id is in computable_queued_; it
238  // stops us from adding things twice.
239  std::vector<bool> computable_queued_;
240 
241  // usable_count_[i] for a cindex_id i is defined as 1 if i is a requested
242  // output, and otherwise as the number of other cindex_ids j such that
243  // computable_info_[j] is not kNotComputable AND usable_count_[j] > 0 AND i is
244  // a member of graph->dependencies[j]. A cindex_id is termed "usable"
245  // (meaning it could potentially participate in the computation of the output)
246  // if its usable_count_ is > 0. This quantity is designed to be easy to keep
247  // updated as we add cindex_ids.
248  std::vector<int32> usable_count_;
249 
250  // current_distance_ >= 0 is the distance to the output, of the cindex_ids in
251  // current_queue_;
252  int32 current_distance_;
253  // the cindex_ids in current_queue_ are at distance "current_distance" to the
254  // output and have not yet had their dependencies processed.
255  std::vector<int32> current_queue_;
256  // the cindex_ids in next_queue_ are at distance current_distance + 1 to the
257  // output and have not yet had their dependencies processed.
258  std::vector<int32> next_queue_;
259 };
