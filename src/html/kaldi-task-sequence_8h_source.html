75 // C should have an operator () taking no arguments, that does some kind
76 // of computation, and a destructor that produces some kind of output (the
77 // destructors will be run sequentially in the same order Run as called.
78 template<class C>
79 class TaskSequencer {
80  public:
81  TaskSequencer(const TaskSequencerConfig &config):
82  threads_avail_(config.num_threads),
83  tot_threads_avail_(config.num_threads_total > 0 ? config.num_threads_total :
84  config.num_threads + 20),
85  thread_list_(NULL) {
86  KALDI_ASSERT((config.num_threads_total <= 0 ||
87  config.num_threads_total >= config.num_threads) &&
88  "num-threads-total, if specified, must be >= num-threads");
89  }

93  void Run(C *c) {
94  threads_avail_.Wait(); // wait till we have a thread for computation free.
95  tot_threads_avail_.Wait(); // this ensures we don't have too many threads
96  // waiting on I/O, and consume too much memory.
97 
98  // put the new RunTaskArgsList object at head of the singly
99  // linked list thread_list_.
100  thread_list_ = new RunTaskArgsList(this, c, thread_list_);
101  int32 ret;
102  if ((ret=pthread_create(&(thread_list_->thread),
103  NULL, // default attributes
104  TaskSequencer<C>::RunTask,
105  static_cast<void*>(thread_list_)))) {
106  const char *c = strerror(ret);
107  KALDI_ERR << "Error creating thread, errno was: " << (c ? c : "[NULL]");
108  }
109  }
110 
111  void Wait() { // You call this at the end if it's more convenient
112  // than waiting for the destructor. It waits for all tasks to finish.
113  if (thread_list_ != NULL) {
114  int ret = pthread_join(thread_list_->thread, NULL);
115  if (ret != 0) {
116  const char *c = strerror(ret);
117  KALDI_ERR << "Error joining thread, errno was: " << (c ? c : "[NULL]");
118  }
119  KALDI_ASSERT(thread_list_->tail == NULL); // thread would not
120  // have exited without setting tail to NULL.
121  delete thread_list_;
122  thread_list_ = NULL;
123  }
124  }

127  ~TaskSequencer() {
128  Wait();
129  }

131  struct RunTaskArgsList {
132  TaskSequencer *me; // Think of this as a "this" pointer.
133  C *c; // Clist element of the task we're expected
134  pthread_t thread;
135  RunTaskArgsList *tail;
136  RunTaskArgsList(TaskSequencer *me, C *c, RunTaskArgsList *tail):
137  me(me), c(c), tail(tail) {}
138  };
139  // This static function gets run in the threads that we create.
140  static void* RunTask(void *input) {
141  RunTaskArgsList *args = static_cast<RunTaskArgsList*>(input);
142 
143  // (1) run the job.
144  (*(args->c))(); // call operator () on args->c, which does the computation.
145  args->me->threads_avail_.Signal(); // Signal that the compute-intensive
146  // part of the thread is done (we want to run no more than
147  // config_.num_threads of these.)
148 
149  // (2) we want to destroy the object "c" now, by deleting it. But for
150  // correct sequencing (this is the whole point of this class, it
151  // is intended to ensure the output of the program is in correct order),
152  // we first wait till the previous thread, whose details will be in "tail",
153  // is finished.
154  if (args->tail != NULL) {
155  int ret = pthread_join(args->tail->thread, NULL);
156  if (ret != 0) {
157  const char *c = strerror(ret);
158  KALDI_ERR << "Error joining thread, errno was: " << (c ? c : "[NULL]");
159  }
160  }
161 
162  delete args->c; // delete the object "c". This may cause some output,
163  // e.g. to a stream. We don't need to worry about concurrent access to
164  // the output stream, because each thread waits for the previous thread
165  // to be done, before doing this. So there is no risk of concurrent
166  // access.
167  args->c = NULL;
168 
169  if (args->tail != NULL) {
170  KALDI_ASSERT(args->tail->tail == NULL); // Because we already
171  // did pthread_join on args->tail->thread, which means that
172  // thread was done, and before it exited, it would have
173  // deleted and set to NULL its tail (which is the next line of code).
174  delete args->tail;
175  args->tail = NULL;
176  }
177  // At this point we are exiting from the thread. Signal the
178  // "tot_threads_avail_" semaphore which is used to limit the total number of threads that are alive, including
179  // not onlhy those that are in active computation in c->operator (), but those
180  // that are waiting on I/O or other threads.
181  args->me->tot_threads_avail_.Signal();
182  // .. and exit the thread, by returning.
183  return NULL;
184  }
185 
186  Semaphore threads_avail_; // Initialized to the number of threads we are
187  // supposed to run with; the function Run() waits on this.
188 
189  Semaphore tot_threads_avail_; // We use this semaphore to ensure we don't
190  // consume too much memory...
191  RunTaskArgsList *thread_list_;
192 
193 };
