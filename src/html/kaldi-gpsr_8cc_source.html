37 double GpsrObjective(const SpMatrix<double> &H, const Vector<double> &c,
38  const Vector<double> &u, const Vector<double> &v) {
39  KALDI_ASSERT(u.Dim() == v.Dim() && u.Dim() > 0);
40  KALDI_ASSERT(c.Dim() == 2 * u.Dim());
41  KALDI_VLOG(2) << "u dim = " << u.Dim() << ", v dim = " << v.Dim()
42  << ", c dim = " << c.Dim();
43 
44  MatrixIndexT dim = u.Dim();
45  Vector<double> H_x(dim), x(dim);
46  // x = u - v, where u_i = (x_i)_+; v_i = (-x_i)_+; and (x)_+ = max{0,x}
47  x.CopyFromVec(u);
48  x.AddVec(-1.0, v);
49 
50  // Calculate c^T z = c^T [u^T v^T]^T
51  double objf = VecVec(c.Range(0, dim), u);
52  objf += VecVec(c.Range(dim, dim), v);
53 
54  // Now, calculate the quadratic term: z^T B z = (u-v)^T H (u-v) = x^T H x
55  H_x.AddSpVec(1.0, H, x, 0.0);
56  objf += 0.5 * VecVec(x, H_x);
57  return objf;
58 }

62 void GpsrGradient(const SpMatrix<double> &H, const Vector<double> &c,
63  const Vector<double> &u, const Vector<double> &v,
64  Vector<double> *grad_u, Vector<double> *grad_v) {
65  KALDI_ASSERT(u.Dim() == v.Dim() && u.Dim() > 0);
66  KALDI_ASSERT(u.Dim() == grad_u->Dim() && v.Dim() == grad_v->Dim());
67  KALDI_ASSERT(c.Dim() == 2 * u.Dim());
68  KALDI_VLOG(2) << "u dim = " << u.Dim() << ", v dim = " << v.Dim()
69  << ", c dim = " << c.Dim();
70 
71  MatrixIndexT dim = u.Dim();
72  Vector<double> H_x(dim), x(dim);
73  // x = u - v, where u_i = (x_i)_+; v_i = (-x_i)_+; and (x)_+ = max{0,x}
74  x.CopyFromVec(u);
75  x.AddVec(-1.0, v);
76  // To calculate B z = [ H (u-v); -H (u-v) ] = [ H x; -H x ], we only need H x
77  H_x.AddSpVec(1.0, H, x, 0.0);
78  grad_u->CopyFromVec(c.Range(0, dim));
79  grad_u->AddVec(1.0, H_x);
80  grad_v->CopyFromVec(c.Range(dim, dim));
81  grad_v->AddVec(-1.0, H_x);
82 }

91 double GpsrBasicAlpha(const SpMatrix<double> &H, const Vector<double> &u,
92  const Vector<double> &v, const Vector<double> &grad_u,
93  const Vector<double> &grad_v) {
94  KALDI_ASSERT(H.NumRows() == grad_u.Dim() && grad_u.Dim() == grad_v.Dim() &&
95  grad_u.Dim() > 0);
96  KALDI_VLOG(2) << "grad_u dim = " << grad_u.Dim() << ", grad_v dim = "
97  << grad_v.Dim() << ", H rows = " << H.NumRows();
98  MatrixIndexT dim = grad_u.Dim();
99 
100  // Find the projection of the gradient on the nonnegative orthant, or, more
101  // precisely, the projection s.t. the next iterate will be in the orthant.
102  Vector<double> proj_grad_u(dim);
103  Vector<double> proj_grad_v(dim);
104  for (MatrixIndexT i = 0; i < dim; i++) {
105  proj_grad_u(i) = (u(i) > 0 || grad_u(i) < 0)? grad_u(i) : 0;
106  proj_grad_v(i) = (v(i) > 0 || grad_v(i) < 0)? grad_v(i) : 0;
107  }
108 
109  // The numerator: g^T g = g_u^T g_u + g_v^T g_v
110  double alpha = VecVec(proj_grad_u, proj_grad_u);
111  alpha += VecVec(proj_grad_v, proj_grad_v);
112 
113  // The denominator: g^T B g = (g_u - g_v)^T H (g_u - g_v)
114  Vector<double> diff_g(proj_grad_u);
115  diff_g.AddVec(-1.0, proj_grad_v);
116  Vector<double> H_diff_g(dim);
117  H_diff_g.AddSpVec(1.0, H, diff_g, 0.0);
118  alpha /= (VecVec(diff_g, H_diff_g) + DBL_EPSILON);
119  return alpha;
120 }

133 // This removes the L1 penalty term, and uses conjugate gradient to solve the
134 // resulting quadratic problem while keeping the zero elements fixed at 0.
135 double Debias(const GpsrConfig &opts, const SpMatrix<double> &H,
136  const Vector<double> &g, Vector<double> *x) {
137  KALDI_ASSERT(H.NumRows() == g.Dim() && g.Dim() == x->Dim() && x->Dim() != 0);
138 // KALDI_ASSERT(H.IsPosDef() &&
139 // "Must have positive definite matrix for conjugate gradient.");
140  MatrixIndexT dim = x->Dim();
141 
142  Vector<double> x_bias(*x);
143  Vector<double> nonzero_indices(dim);
144  // Initialize the index of non-zero elements in x
145  for (MatrixIndexT i = 0; i < dim; i++)
146  nonzero_indices(i) = (x_bias(i) == 0)? 0.0 : 1.0;
147 
148  Vector<double> residual(dim);
149  Vector<double> conj_direction(dim);
150  Vector<double> resid_change(dim);
151  double alpha_cg; // CG step size for iterate: x <- x + \alpha p
152  double beta_cg; // CG step size for conj. direction: p <- \beta p - r
153  double resid_prod, resid_prod_new; // inner product of residual vectors
154 
155  // Calculate the initial residual: r = H x_0 - g
156  residual.AddSpVec(1.0, H, x_bias, 0.0);
157  residual.AddVec(-1.0, g);
158  residual.MulElements(nonzero_indices); // only change non-zero elements of x
159 
160  conj_direction.CopyFromVec(residual);
161  conj_direction.Scale(-1.0); // Initial conjugate direction p = -r
162  resid_prod = VecVec(residual, residual);
163 
164  // set the convergence threshold for residual
165  double tol_debias = opts.stop_thresh_debias * VecVec(residual, residual);
166 
167  for (int32 iter = 0; iter < opts.max_iters_debias; iter++) {
168  resid_change.AddSpVec(1.0, H, conj_direction, 0.0);
169  resid_change.MulElements(nonzero_indices); // only change non-zero elements
170 
171  alpha_cg = resid_prod / VecVec(conj_direction, resid_change);
172  x_bias.AddVec(alpha_cg, conj_direction);
173  residual.AddVec(alpha_cg, resid_change);
174 
175  resid_prod_new = VecVec(residual, residual);
176  beta_cg = resid_prod_new / resid_prod;
177  conj_direction.Scale(beta_cg);
178  conj_direction.AddVec(-1.0, residual);
179  resid_prod = resid_prod_new;
180 
181  if (resid_prod < tol_debias) {
182  KALDI_VLOG(1) << "iter=" << iter << "\t residual =" << resid_prod
183  << "\t tol_debias=" << tol_debias;
184  break;
185  }
186  } // end CG iters
187 
188  x->CopyFromVec(x_bias);
189  return resid_prod;
190 }

192 template<>
193 double GpsrBasic(const GpsrConfig &opts, const SpMatrix<double> &H,
194  const Vector<double> &g, Vector<double> *x,
195  const char *debug_str) {
196  KALDI_ASSERT(H.NumRows() == g.Dim() && g.Dim() == x->Dim() && x->Dim() != 0);
197  MatrixIndexT dim = x->Dim();
198  if (H.IsZero(0.0)) {
199  KALDI_WARN << "Zero quadratic term in GPSR for " << debug_str
200  << ": leaving it unchanged.";
201  return 0.0;
202  }
203 
204  // initialize the positive (u) and negative (v) parts of x, s.t. x = u - v
205  Vector<double> u(dim, kSetZero);
206  Vector<double> v(dim, kSetZero);
207  for (MatrixIndexT i = 0; i < dim; i++) {
208  if ((*x)(i) > 0) {
209  u(i) = (*x)(i);
210  } else {
211  v(i) = -(*x)(i);
212  }
213  }
214 
215  double tau = opts.gpsr_tau; // May be modified later.
216  Vector<double> c(2*dim);
217  GpsrCalcLinearCoeff(tau, g, &c);
218 
219  double objf_ori = GpsrObjective(H, c, u, v); // the obj. function at start
220  KALDI_VLOG(2) << "GPSR for " << debug_str << ": tau = " << tau
221  << ";\t objf = " << objf_ori;
222 
223  Vector<double> grad_u(dim);
224  Vector<double> grad_v(dim);
225  Vector<double> delta_u(dim);
226  Vector<double> delta_v(dim);
227  Vector<double> u_new(dim);
228  Vector<double> v_new(dim);
229  double objf_old, objf_new, num_zeros;
230  bool keep_going = true;
231 
232  for (int32 iter = 0; keep_going; iter++) {
233  objf_old = GpsrObjective(H, c, u, v);
234  GpsrGradient(H, c, u, v, &grad_u, &grad_v);
235  double alpha = GpsrBasicAlpha(H, u, v, grad_u, grad_v);
236  if (alpha < opts.alpha_min) alpha = opts.alpha_min;
237  if (alpha > opts.alpha_max) alpha = opts.alpha_max;
238 
239  // This is the backtracking line search part:
240  for (int32 k = 0; k < opts.max_iters_backtrak; k++) {
241  // Calculate the potential new iterate: [z_k - \alpha_k \grad F(z_k)]_+
242  u_new.CopyFromVec(u);
243  u_new.AddVec(-alpha, grad_u);
244  u_new.ApplyFloor(0.0);
245  v_new.CopyFromVec(v);
246  v_new.AddVec(-alpha, grad_v);
247  v_new.ApplyFloor(0.0);
248 
249  delta_u.CopyFromVec(u_new);
250  delta_v.CopyFromVec(v_new);
251  delta_u.AddVec(-1.0, u);
252  delta_v.AddVec(-1.0, v);
253 
254  double delta_objf_apx = opts.gpsr_mu * (VecVec(grad_u, delta_u) +
255  VecVec(grad_v, delta_v));
256  objf_new = GpsrObjective(H, c, u_new, v_new);
257  double delta_objf_real = objf_new - objf_old;
258 
259  KALDI_VLOG(2) << "GPSR for " << debug_str << ": iter " << iter
260  << "; tau = " << tau << ";\t objf = " << objf_new
261  << ";\t alpha = " << alpha << ";\t delta_apx = "
262  << delta_objf_apx << ";\t delta_real = " << delta_objf_real;
263 
264  if (delta_objf_real < delta_objf_apx + DBL_EPSILON)
265  break;
266  else
267  alpha *= opts.gpsr_beta;
268 
269  if (k == opts.max_iters_backtrak - 1) { // Stop further optimization
270  KALDI_WARN << "Backtracking line search did not decrease objective.";
271  u_new.CopyFromVec(u);
272  u_new.ApplyFloor(0.0);
273  v_new.CopyFromVec(v);
274  v_new.ApplyFloor(0.0);
275  delta_u.SetZero();
276  delta_v.SetZero();
277  }
278  } // end of backtracking line search
279 
280  x->CopyFromVec(u_new);
281  x->AddVec(-1.0, v_new);
282 
283  num_zeros = 0;
284  for (MatrixIndexT i = 0; i < dim; i++)
285  if ((*x)(i) == 0)
286  num_zeros++;
287 
288  // ad hoc way to modify tau, if the solution is too sparse
289  if ((num_zeros / static_cast<double>(dim)) > opts.max_sparsity) {
290  std::ostringstream msg;
291  msg << num_zeros << " out of " << dim << " dimensions set to 0. "
292  << "Changing tau from " << tau;
293  tau *= opts.tau_reduction;
294  GpsrCalcLinearCoeff(tau, g, &c); // Recalculate c with new tau
295  double tmp_objf = GpsrObjective(H, c, u, v);
296  msg << " to " << tau << ".\n\tStarting objective function changed from "
297  << objf_ori << " to " << tmp_objf << ".";
298  KALDI_LOG << "GPSR for " << debug_str << ": " << msg.str();
299  iter = 0;
300  keep_going = true;
301  continue;
302  }
303 
304  u.CopyFromVec(u_new);
305  v.CopyFromVec(v_new);
306  double delta = (delta_u.Norm(2.0) + delta_v.Norm(2.0)) / x->Norm(2.0);
307  KALDI_VLOG(1) << "GPSR for " << debug_str << ": iter " << iter
308  << ", objf = " << objf_new << ", delta = " << delta;
309 
310  keep_going = (iter < opts.max_iters) && (delta > opts.stop_thresh);
311 
312  KALDI_VLOG(3) << "GPSR for " << debug_str << ": iter " << iter
313  << ", objf = " << objf_new << ", value = " << x;
314  }
315 
316  if (num_zeros != 0) {
317  KALDI_LOG << "GPSR for " << debug_str << ": number of 0's = " << num_zeros
318  << " out of " << dim << " dimensions.";
319  }
320 
321  if (opts.debias && num_zeros != 0) {
322  double residual = Debias(opts, H, g, x);
323  KALDI_LOG << "Debiasing: new residual = " << residual;
324  }
325  return objf_new - objf_ori;
326 }

343  const Vector<double> &g, Vector<double> *x,
344  const char *debug_str) {
345  KALDI_ASSERT(H.NumRows() == g.Dim() && g.Dim() == x->Dim() && x->Dim() != 0);
346  MatrixIndexT dim = x->Dim();
347  if (H.IsZero(0.0)) {
348  KALDI_WARN << "Zero quadratic term in GPSR for " << debug_str
349  << ": leaving it unchanged.";
350  return 0.0;
351  }
352 
353  // initialize the positive (u) and negative (v) parts of x, s.t. x = u - v
354  Vector<double> u(dim, kSetZero);
355  Vector<double> v(dim, kSetZero);
356  for (MatrixIndexT i = 0; i < dim; i++) {
357  if ((*x)(i) > 0) {
358  u(i) = (*x)(i);
359  } else {
360  v(i) = -(*x)(i);
361  }
362  }
363 
364  double tau = opts.gpsr_tau; // May be modified later.
365  Vector<double> c(2*dim);
366  GpsrCalcLinearCoeff(tau, g, &c);
367 
368  double objf_ori = GpsrObjective(H, c, u, v); // the obj. function at start
369  KALDI_VLOG(2) << "GPSR for " << debug_str << ": tau = " << tau
370  << ";\t objf = " << objf_ori;
371 
372  Vector<double> grad_u(dim);
373  Vector<double> grad_v(dim);
374  Vector<double> delta_u(dim);
375  Vector<double> delta_v(dim);
376  Vector<double> delta_x(dim);
377  Vector<double> H_delta_x(dim);
378  Vector<double> u_new(dim);
379  Vector<double> v_new(dim);
380  double objf_old, objf_new, num_zeros;
381  bool keep_going = true;
382  double alpha = 1.0;
383 
384  for (int32 iter = 0; keep_going; iter++) {
385  objf_old = GpsrObjective(H, c, u, v);
386  GpsrGradient(H, c, u, v, &grad_u, &grad_v);
387 
388  // Calculate the new step: [z_k - \alpha_k \grad F(z_k)]_+ - z_k
389  delta_u.CopyFromVec(u);
390  delta_u.AddVec(-alpha, grad_u);
391  delta_u.ApplyFloor(0.0);
392  delta_u.AddVec(-1.0, u);
393  delta_v.CopyFromVec(v);
394  delta_v.AddVec(-alpha, grad_v);
395  delta_v.ApplyFloor(0.0);
396  delta_v.AddVec(-1.0, v);
397

399  delta_x.AddVec(-1.0, delta_v);
400  H_delta_x.AddSpVec(1.0, H, delta_x, 0.0);
401  double dx_H_dx = VecVec(delta_x, H_delta_x);
402 
403  double lambda = -(VecVec(delta_u, grad_u) + VecVec(delta_v, grad_v))
404  / (dx_H_dx + DBL_EPSILON); // step length
405  if (lambda < 0)
406  KALDI_WARN << "lambda is less than zero";
407  if (lambda > 1.0) lambda = 1.0;
408 
409  //update alpha
410  alpha = (VecVec(delta_u, delta_u) + VecVec(delta_v, delta_v))
411  / (dx_H_dx + DBL_EPSILON);
412  if (dx_H_dx <= 0) {
413  KALDI_WARN << "nonpositive curvature detected";
414  alpha = opts.alpha_max;
415  }
416  else if (alpha < opts.alpha_min)
417  alpha = opts.alpha_min;
418  else if (alpha > opts.alpha_max) alpha = opts.alpha_max;
419 
420  u_new.CopyFromVec(delta_u);
421  u_new.Scale(lambda);
422  v_new.CopyFromVec(delta_v);
423  v_new.Scale(lambda);
424  u_new.AddVec(1.0, u);
425  v_new.AddVec(1.0, v);
426 
427  objf_new = GpsrObjective(H, c, u_new, v_new);
428  double delta_objf = objf_old - objf_new;
429  KALDI_VLOG(2) << "GPSR for " << debug_str << ": iter " << iter
430  << "; tau = " << tau << ";\t objf = " << objf_new
431  << ";\t alpha = " << alpha << ";\t delta_real = "
432  << delta_objf;
433 
434  u.CopyFromVec(u_new);
435  v.CopyFromVec(v_new);
436  x->CopyFromVec(u);
437  x->AddVec(-1.0, v);
438 
439  num_zeros = 0;
440  for (MatrixIndexT i = 0; i < dim; i++)
441  if ((*x)(i) == 0)
442  num_zeros++;
443 
444  // ad hoc way to modify tau, if the solution is too sparse
445  if ((num_zeros / static_cast<double>(dim)) > opts.max_sparsity) {
446  std::ostringstream msg;
447  msg << num_zeros << " out of " << dim << " dimensions set to 0. "
448  << "Changing tau from " << tau;
449  tau *= 0.9;
450  GpsrCalcLinearCoeff(tau, g, &c); // Recalculate c with new tau
451  double tmp_objf = GpsrObjective(H, c, u, v);
452  msg << " to " << tau << ".\n\tStarting objective function changed from "
453  << objf_ori << " to " << tmp_objf << ".";
454  KALDI_LOG << "GPSR for " << debug_str << ": " << msg.str();
455  iter = 0;

459

461  KALDI_VLOG(1) << "GPSR for " << debug_str << ": iter " << iter
462  << ", objf = " << objf_new << ", delta = " << delta;
463 
464  keep_going = (iter < opts.max_iters) && (delta > opts.stop_thresh);

467  << ", objf = " << objf_new << ", value = " << x;
468  }
469
