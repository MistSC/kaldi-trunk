29 bool PrintProxyFstPath(const VectorFst<StdArc> &proxy,
30  vector<vector<StdArc::Label> > *path,
31  vector<StdArc::Weight> *weight,
32  StdArc::StateId cur_state,
33  vector<StdArc::Label> cur_path,
34  StdArc::Weight cur_weight) {
35  if (proxy.Final(cur_state) != StdArc::Weight::Zero()) {
36  // Assumes only final state has non-zero weight.
37  cur_weight = Times(proxy.Final(cur_state), cur_weight);
38  path->push_back(cur_path);
39  weight->push_back(cur_weight);
40  return true;
41  }
42 
43  for (ArcIterator<StdFst> aiter(proxy, cur_state);
44  !aiter.Done(); aiter.Next()) {
45  const StdArc &arc = aiter.Value();
46  StdArc::Weight temp_weight = Times(arc.weight, cur_weight);
47  cur_path.push_back(arc.ilabel);
48  PrintProxyFstPath(proxy, path, weight,
49  arc.nextstate, cur_path, temp_weight);
50  cur_path.pop_back();
51  }
52 
53  return true;
54 }
55 }

66  const char *usage =
67  "Convert the keywords into in-vocabulary words using the given phone\n"
68  "level edit distance fst (E.fst). The large lexicon (L2.fst) and\n"
69  "inverted small lexicon (L1'.fst) are also expected to be present. We\n"
70  "actually use the composed FST L2xE.fst to be more efficient. Ideally\n"
71  "we should have used L2xExL1'.fst but this is quite computationally\n"
72  "expensive at command level. Keywords.int is in the transcription\n"
73  "format. If kwlist-wspecifier is given, the program also prints out\n"
74  "the proxy fst in a format where each line is \"kwid weight proxy\".\n"
75  "\n"
76  "Usage: generate-proxy-keywords [options] <L2xE.fst> <L1'.fst> \\\n"
77  " <keyword-rspecifier> <proxy-wspecifier> [kwlist-wspecifier] \n"
78  " e.g.: generate-proxy-keywords L2xE.fst L1'.fst ark:keywords.int \\\n"
79  " ark:proxy.fsts [ark,t:proxy.kwlist.txt]\n";
80 
81  ParseOptions po(usage);
82 
83  int32 max_states = 100000;
84  int32 phone_nbest = 50;
85  int32 proxy_nbest = 100;
86  double phone_beam = 5;
87  double proxy_beam = 5;
88  po.Register("phone-nbest", &phone_nbest, "Prune KxL2xE transducer to only "
89  "contain top n phone sequences, -1 means all sequences.");
90  po.Register("proxy-nbest", &proxy_nbest, "Prune KxL2xExL1' transducer to "
91  "only contain top n proxy keywords, -1 means all proxies.");
92  po.Register("phone-beam", &phone_beam, "Prune KxL2xE transducer to the "
93  "given beam, -1 means no prune.");
94  po.Register("proxy-beam", &proxy_beam, "Prune KxL2xExL1' transducer to the "
95  "given beam, -1 means no prune.");
96  po.Register("max-states", &max_states, "Prune kxL2xExL1' transducer to the "
97  "given number of states, 0 means no prune.");
98 
99  po.Read(argc, argv);
100 
101  // Checks input options.
102  if (phone_nbest != -1 && phone_nbest <= 0) {
103  KALDI_ERR << "--phone-nbest must either be -1 or positive.";
104  exit(1);
105  }
106  if (proxy_nbest != -1 && proxy_nbest <= 0) {
107  KALDI_ERR << "--proxy-nbest must either be -1 or positive.";
108  exit(1);
109  }
110  if (phone_beam != -1 && phone_beam < 0) {
111  KALDI_ERR << "--phone-beam must either be -1 or non-negative.";
112  exit(1);
113  }
114  if (proxy_beam != -1 && proxy_beam <=0) {
115  KALDI_ERR << "--proxy-beam must either be -1 or non-negative.";
116  exit(1);
117  }
118 
119  if (po.NumArgs() < 4 || po.NumArgs() > 5) {
120  po.PrintUsage();
121  exit(1);
122  }
123 
124  std::string L2xE_filename = po.GetArg(1),
125  L1_filename = po.GetArg(2),
126  keyword_rspecifier = po.GetArg(3),
127  proxy_wspecifier = po.GetArg(4),
128  kwlist_wspecifier = (po.NumArgs() == 5) ? po.GetArg(5) : "";
129 
130  VectorFst<StdArc> *L2xE = ReadFstKaldi(L2xE_filename);
131  VectorFst<StdArc> *L1 = ReadFstKaldi(L1_filename);
132  SequentialInt32VectorReader keyword_reader(keyword_rspecifier);
133  TableWriter<VectorFstHolder> proxy_writer(proxy_wspecifier);
134  TableWriter<BasicVectorHolder<double> > kwlist_writer(kwlist_wspecifier);
135 
136  // Processing the keywords.
137  int32 n_done = 0;
138  for (; !keyword_reader.Done(); keyword_reader.Next()) {
139  std::string key = keyword_reader.Key();
140  std::vector<int32> keyword = keyword_reader.Value();
141  keyword_reader.FreeCurrent();
142 
143  KALDI_LOG << "Processing " << key;
144 
145  VectorFst<StdArc> proxy;
146  VectorFst<StdArc> tmp_proxy;
147  MakeLinearAcceptor(keyword, &proxy);
148 
149  // Composing K and L2xE. We assume L2xE is ilabel sorted.
150  KALDI_VLOG(1) << "Compose(K, L2xE)";
151  ArcSort(&proxy, OLabelCompare<StdArc>());
152  Compose(proxy, *L2xE, &tmp_proxy);
153 
154  // Processing KxL2xE.
155  KALDI_VLOG(1) << "Project(KxL2xE, PROJECT_OUTPUT)";
156  Project(&tmp_proxy, PROJECT_OUTPUT);
157  if (phone_beam >= 0) {
158  KALDI_VLOG(1) << "Prune(KxL2xE, " << phone_beam << ")";
159  Prune(&tmp_proxy, phone_beam);
160  }
161  if (phone_nbest > 0) {
162  KALDI_VLOG(1) << "ShortestPath(KxL2xE, " << phone_nbest << ")";
163  RmEpsilon(&tmp_proxy);
164  ShortestPath(tmp_proxy, &proxy, phone_nbest, true, true);
165  tmp_proxy.DeleteStates(); // Not needed for now.
166  KALDI_VLOG(1) << "Determinize(KxL2xE)";
167  Determinize(proxy, &tmp_proxy);
168  proxy.DeleteStates(); // Not needed for now.
169  }
170  KALDI_VLOG(1) << "ArcSort(KxL2xE, OLabel)";
171  proxy = tmp_proxy;
172  tmp_proxy.DeleteStates(); // Not needed for now.
173  ArcSort(&proxy, OLabelCompare<StdArc>());
174 
175 
176  // Processing KxL2xExL1'.
177  RmEpsilon(&proxy);
178  ArcSort(&proxy, OLabelCompare<StdArc>());
179  if (proxy_beam >= 0) {
180  // We only use the delayed FST when pruning is requested, because we do
181  // the optimization in pruning.
182  // Composing KxL2xE and L1'. We assume L1' is ilabel sorted.
183  KALDI_VLOG(1) << "Compose(KxL2xE, L1')";
184  ComposeFst<StdArc> lazy_compose(proxy, *L1);
185  proxy.DeleteStates();
186 
187  KALDI_VLOG(1) << "Project(KxL2xExL1', PROJECT_OUTPUT)";
188  ProjectFst<StdArc> lazy_project(lazy_compose, PROJECT_OUTPUT);
189 
190  // This will likely be the most time consuming part, we use a special
191  // pruning algorithm where we don't expand the full FST.
192  KALDI_VLOG(1) << "Prune(KxL2xExL1', " << proxy_beam << ")";
193  PruneSpecial(lazy_project, &tmp_proxy, proxy_beam, max_states);
194  } else {
195  // If no pruning is requested, we do the normal composition.
196  KALDI_VLOG(1) << "Compose(KxL2xE, L1')";
197  Compose(proxy, *L1, &tmp_proxy);
198  proxy.DeleteStates();
199 
200  KALDI_VLOG(1) << "Project(KxL2xExL1', PROJECT_OUTPUT)";
201  Project(&tmp_proxy, PROJECT_OUTPUT);
202  }
203  if (proxy_nbest > 0) {
204  KALDI_VLOG(1) << "ShortestPath(KxL2xExL1', " << proxy_nbest << ")";
205  proxy = tmp_proxy;
206  tmp_proxy.DeleteStates(); // Not needed for now.
207  RmEpsilon(&proxy);
208  ShortestPath(proxy, &tmp_proxy, proxy_nbest, true, true);
209  proxy.DeleteStates(); // Not needed for now.
210  }
211  KALDI_VLOG(1) << "RmEpsilon(KxL2xExL1')";
212  RmEpsilon(&tmp_proxy);
213  KALDI_VLOG(1) << "Determinize(KxL2xExL1')";
214  Determinize(tmp_proxy, &proxy);
215  tmp_proxy.DeleteStates();
216  KALDI_VLOG(1) << "ArcSort(KxL2xExL1', OLabel)";
217  ArcSort(&proxy, fst::OLabelCompare<StdArc>());
218 
219  // Write the proxy FST.
220  proxy_writer.Write(key, proxy);
221 
222  // Print the proxy FST with each line looks like "kwid weight proxy"
223  if (po.NumArgs() == 5) {
224  if (proxy.Properties(kAcyclic, true) == 0) {
225  KALDI_WARN << "Proxy FST has cycles, skip printing paths for " << key;
226  } else {
227  vector<vector<StdArc::Label> > path;
228  vector<StdArc::Weight> weight;
229  PrintProxyFstPath(proxy, &path, &weight, proxy.Start(),
230  vector<StdArc::Label>(), StdArc::Weight::One());
231  KALDI_ASSERT(path.size() == weight.size());
232  for (int32 i = 0; i < path.size(); i++) {
233  vector<double> kwlist;
234  kwlist.push_back(static_cast<double>(weight[i].Value()));
235  for (int32 j = 0; j < path[i].size(); j++) {
236  kwlist.push_back(static_cast<double>(path[i][j]));
237  }
238  kwlist_writer.Write(key, kwlist);
239  }
240  }
241  }
242 
243  n_done++;
244  }
245 
246  delete L1;
247  delete L2xE;
248  KALDI_LOG << "Done " << n_done << " keywords";
249  return (n_done != 0 ? 0 : 1);
250  } catch(const std::exception &e) {
251  std::cerr << e.what();
252  return -1;
253  }
254 }
