284 // This generates a single config corresponding to an LSTM.
285 // based on the equations in
286 // Sak et. al. "LSTM based RNN architectures for LVCSR", 2014
287 // We name the components based on the following equations (Eqs 7-15 in paper)
288 // i(t) = S(Wix * x(t) + Wir * r(t-1) + Wic * c(t-1) + bi)
289 // f(t) = S(Wfx * x(t) + Wfr * r(t-1) + Wfc * c(t-1) + bf)
290 // c(t) = {f(t) .* c(t-1)} + {i(t) .* g(Wcx * x(t) + Wcr * r(t-1) + bc)}
291 // o(t) = S(Wox * x(t) + Wor * r(t-1) + Woc * c(t) + bo)
292 // m(t) = o(t) .* h(c(t))
293 // r(t) = Wrm * m(t)
294 // p(t) = Wpm * m(t)
295 // y(t) = Wyr * r(t) + Wyp * p(t) + by
296 // where S : sigmoid
297 // matrix with feed-forward connections
298 // from the input x(t)
299 // W*x = [Wix^T, Wfx^T, Wcx^T, Wox^T]^T
300 
301 // matrix with recurrent (feed-back) connections
302 // from the output projection
303 // W*r = [Wir^T, Wfr^T, Wcr^T, Wor^T]^T
304 
305 // matrix to generate r(t) and p(t)
306 // m(t)
307 // W*m = [Wrm^T, Wpm^T]^T
308 // matrix to generate y(t)
309 // Wy* = [Wyr^T, Wyp^T]
310 
311 // Diagonal matrices with recurrent connections and feed-forward connections
312 // from the cell output c(t) since these can be both recurrent and
313 // feed-forward we dont combine the matrices
314 // Wic, Wfc, Woc

778 void ComputeExampleComputationRequestSimple(
779  const Nnet &nnet,
780  ComputationRequest *request,
781  std::vector<Matrix<BaseFloat> > *inputs) {
782  KALDI_ASSERT(IsSimpleNnet(nnet));
783 
784  int32 left_context, right_context;
785  ComputeSimpleNnetContext(nnet, &left_context, &right_context);
786 
787  int32 num_output_frames = 1 + Rand() % 10,
788  output_start_frame = Rand() % 10,
789  num_examples = 1 + Rand() % 10,
790  output_end_frame = output_start_frame + num_output_frames,
791  input_start_frame = output_start_frame - left_context - (Rand() % 3),
792  input_end_frame = output_end_frame + right_context + (Rand() % 3),
793  n_offset = Rand() % 2;
794  bool need_deriv = (Rand() % 2 == 0);
795 
796  request->inputs.clear();
797  request->outputs.clear();
798  inputs->clear();
799 
800  std::vector<Index> input_indexes, ivector_indexes, output_indexes;
801  for (int32 n = n_offset; n < n_offset + num_examples; n++) {
802  for (int32 t = input_start_frame; t < input_end_frame; t++)
803  input_indexes.push_back(Index(n, t, 0));
804  for (int32 t = output_start_frame; t < output_end_frame; t++)
805  output_indexes.push_back(Index(n, t, 0));
806  ivector_indexes.push_back(Index(n, 0, 0));
807  }
808  request->outputs.push_back(IoSpecification("output", output_indexes));
809  if (need_deriv || (Rand() % 3 == 0))
810  request->outputs.back().has_deriv = true;
811  request->inputs.push_back(IoSpecification("input", input_indexes));
812  if (need_deriv && (Rand() % 2 == 0))
813  request->inputs.back().has_deriv = true;
814  int32 input_dim = nnet.InputDim("input");
815  KALDI_ASSERT(input_dim > 0);
816  inputs->push_back(
817  Matrix<BaseFloat>((input_end_frame - input_start_frame) * num_examples,
818  input_dim));
819  inputs->back().SetRandn();
820  int32 ivector_dim = nnet.InputDim("ivector"); // may not exist.
821  if (ivector_dim != -1) {
822  request->inputs.push_back(IoSpecification("ivector", ivector_indexes));
823  inputs->push_back(Matrix<BaseFloat>(num_examples, ivector_dim));
824  inputs->back().SetRandn();
825  if (need_deriv && (Rand() % 2 == 0))
826  request->inputs.back().has_deriv = true;
827  }
828  if (Rand() % 2 == 0)
829  request->need_model_derivative = need_deriv;
830  if (Rand() % 2 == 0)
831  request->store_component_stats = true;
832 }
