173  // ForwardLinks are the links from a token to a token on the next frame.
174  // or sometimes on the current frame (for input-epsilon links).
175  struct ForwardLink {
176  Token *next_tok; // the next token [or NULL if represents final-state]
177  Label ilabel; // ilabel on link.
178  Label olabel; // olabel on link.
179  BaseFloat graph_cost; // graph cost of traversing link (contains LM, etc.)
180  BaseFloat acoustic_cost; // acoustic cost (pre-scaled) of traversing link
181  ForwardLink *next; // next in singly-linked list of forward links from a
182  // token.
183  inline ForwardLink(Token *next_tok, Label ilabel, Label olabel,
184  BaseFloat graph_cost, BaseFloat acoustic_cost,
185  ForwardLink *next):
186  next_tok(next_tok), ilabel(ilabel), olabel(olabel),
187  graph_cost(graph_cost), acoustic_cost(acoustic_cost),
188  next(next) { }
189  };

191  // Token is what's resident in a particular state at a particular time.
192  // In this decoder a Token actually contains *forward* links.
193  // When first created, a Token just has the (total) cost. We add forward
194  // links to it when we process the next frame.
195  struct Token {
196  BaseFloat tot_cost; // would equal weight.Value()... cost up to this point.
197  BaseFloat extra_cost; // >= 0. After calling PruneForwardLinks, this equals
198  // the minimum difference between the cost of the best path, and the cost of
199  // this is on, and the cost of the absolute best path, under the assumption
200  // that any of the currently active states at the decoding front may
201  // eventually succeed (e.g. if you were to take the currently active states
202  // one by one and compute this difference, and then take the minimum).
203 
204  ForwardLink *links; // Head of singly linked list of ForwardLinks
205 
206  Token *next; // Next in list of tokens for this frame.
207 
208  StateId lat_state; // current state in graph arc lattice from first pass decoding
209  // lat_state == fst::kNoStateId means that this token is not tracked
210 
211  inline Token(BaseFloat tot_cost, BaseFloat extra_cost, ForwardLink *links,
212  Token *next, StateId lat_state): tot_cost(tot_cost), extra_cost(extra_cost),
213  links(links), next(next), lat_state(lat_state) { }
214  inline void DeleteForwardLinks() {
215  ForwardLink *l = links, *m;
216  while (l != NULL) {
217  m = l->next;
218  delete l;
219  l = m;
220  }
221  links = NULL;
222  }
223  };
