86 const char* Component::TypeToMarker(ComponentType t) {
87  int32 N=sizeof(kMarkerMap)/sizeof(kMarkerMap[0]);
88  for(int i=0; i<N; i++) {
89  if (kMarkerMap[i].key == t) return kMarkerMap[i].value;
90  }
91  KALDI_ERR << "Unknown type" << t;
92  return NULL;
93 }

95 Component::ComponentType Component::MarkerToType(const std::string &s) {
96  std::string s_lowercase(s);
97  std::transform(s.begin(), s.end(), s_lowercase.begin(), ::tolower); // lc
98  int32 N=sizeof(kMarkerMap)/sizeof(kMarkerMap[0]);
99  for(int i=0; i<N; i++) {
100  std::string m(kMarkerMap[i].value);
101  std::string m_lowercase(m);
102  std::transform(m.begin(), m.end(), m_lowercase.begin(), ::tolower);
103  if (s_lowercase == m_lowercase) return kMarkerMap[i].key;
104  }
105  KALDI_ERR << "Unknown marker : '" << s << "'";
106  return kUnknown;
107 }

208 Component* Component::Init(const std::string &conf_line) {
209  std::istringstream is(conf_line);
210  std::string component_type_string;
211  int32 input_dim, output_dim;
212 
213  // initialize component w/o internal data
214  ReadToken(is, false, &component_type_string);
215  ComponentType component_type = MarkerToType(component_type_string);
216  ExpectToken(is, false, "<InputDim>");
217  ReadBasicType(is, false, &input_dim);
218  ExpectToken(is, false, "<OutputDim>");
219  ReadBasicType(is, false, &output_dim);
220  Component *ans = NewComponentOfType(component_type, input_dim, output_dim);
221 
222  // initialize internal data with the remaining part of config line
223  ans->InitData(is);
224 
225  return ans;
226 }

229 Component* Component::Read(std::istream &is, bool binary) {
230  int32 dim_out, dim_in;
231  std::string token;
232 
233  int first_char = Peek(is, binary);
234  if (first_char == EOF) return NULL;
235 
236  ReadToken(is, binary, &token);
237  // Skip optional initial token
238  if(token == "<Nnet>") {
239  ReadToken(is, binary, &token); // Next token is a Component
240  }
241  // Finish reading when optional terminal token appears
242  if(token == "</Nnet>") {
243  return NULL;
244  }
245 
246  ReadBasicType(is, binary, &dim_out);
247  ReadBasicType(is, binary, &dim_in);
248 
249  Component *ans = NewComponentOfType(MarkerToType(token), dim_in, dim_out);
250  ans->ReadData(is, binary);
251  return ans;
252 }
