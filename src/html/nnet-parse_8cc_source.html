31 bool ConfigLine::ParseLine(const std::string &line) {
32  if (line.size() == 0) return false; // Empty line
33 
34  size_t pos = 0, size = line.size();
35  while (pos < size) {
36  if (isspace(line[pos])) {
37  pos++;
38  continue;
39  }
40  // OK, at this point we know that we are pointing at nonspace.
41  size_t next_equals_sign = line.find_first_of("=", pos);
42  if (next_equals_sign == pos || next_equals_sign == std::string::npos) {
43  // we're looking for something like 'key=value'. If there is no equals sign,
44  // or it's not preceded by something, it's a parsing failure.
45  return false;
46  }
47  std::string key(line, pos, next_equals_sign - pos);
48  if (!IsValidName(key)) return false;
49 
50  // handle any quotes. we support key='blah blah' or key="foo bar".
51  // no escaping is supported.
52  if (line[next_equals_sign+1] == '\'' || line[next_equals_sign+1] == '"') {
53  char my_quote = line[next_equals_sign+1];
54  size_t next_quote = line.find_first_of(my_quote, next_equals_sign + 2);
55  if (next_quote == std::string::npos) { // no matching quote was found.
56  KALDI_WARN << "No matching quote for " << my_quote << " in config line '"
57  << line << "'";
58  return false;
59  } else {
60  std::string value(line, next_equals_sign + 2,
61  next_quote - next_equals_sign - 2);
62  data_.insert(std::make_pair(key, std::make_pair(value, false)));
63  pos = next_quote + 1;
64  continue;
65  }
66  } else {
67  // we want to be able to parse something like "... input=Offset(a, -1) foo=bar":
68  // in general, config values with spaces in them, even without quoting.
69 
70  size_t next_next_equals_sign = line.find_first_of("=", next_equals_sign + 1),
71  terminating_space = size;
72 
73  if (next_next_equals_sign != std::string::npos) { // found a later equals sign.
74  size_t preceding_space = line.find_last_of(" \t", next_next_equals_sign);
75  if (preceding_space != std::string::npos &&
76  preceding_space > next_equals_sign)
77  terminating_space = preceding_space;
78  }
79  while (isspace(line[terminating_space - 1]) && terminating_space > 0)
80  terminating_space--;
81 
82  std::string value(line, next_equals_sign + 1,
83  terminating_space - (next_equals_sign + 1));
84  data_.insert(std::make_pair(key, std::make_pair(value, false)));
85  pos = terminating_space;
86  }
87  }
88  whole_line_ = line;
89  return !data_.empty();
90 }

220  int32 *param) {

233  for (size_t j = 0; j < split_string.size(); j++) {
234  if (j != i) {
235  if (!string->empty()) *string += " ";
236  *string += split_string[j];
237  }
238  }
239  return true;
240  }
241  }

355 bool DescriptorTokenize(const std::string &input,
356  std::vector<std::string> *tokens) {
357  KALDI_ASSERT(tokens != NULL);
358  size_t start = input.find_first_not_of(" \t"), size = input.size();
359  tokens->clear();
360  while (start < size) {
361  KALDI_ASSERT(!isspace(input[start]));
362  if (input[start] == '(' || input[start] == ')' || input[start] == ',') {
363  tokens->push_back(std::string(input, start, 1));
364  start = input.find_first_not_of(" \t", start + 1);
365  } else {
366  size_t found = input.find_first_of(" \t(),", start);
367  KALDI_ASSERT(found != start);
368  if (found == std::string::npos) {
369  std::string str(input, start, input.size() - start);
370  int32 tmp;
371  if (!IsValidName(str) && !ConvertStringToInteger(str, &tmp)) {
372  KALDI_WARN << "Could not parse line " << ErrorContext(std::string(input, start));
373  return false;
374  }
375  tokens->push_back(str);
376  break;
377  } else {
378  if (input[found] == '(' || input[found] == ')' || input[found] == ',') {
379  std::string str(input, start, found - start);
380  int32 tmp;
381  if (!IsValidName(str) && !ConvertStringToInteger(str, &tmp)) {
382  KALDI_WARN << "Could not parse line " << ErrorContext(std::string(input, start));
383  return false;
384  }
385  tokens->push_back(str);
386  start = found;
387  } else {
388  std::string str(input, start, found - start);
389  int32 tmp;
390  if (!IsValidName(str) && !ConvertStringToInteger(str, &tmp)) {
391  KALDI_WARN << "Could not parse line " << ErrorContext(std::string(input, start));
392  return false;
393  }
394  tokens->push_back(str);
395  start = input.find_first_not_of(" \t", found);
396  }
397  }
398  }
399  }
400  return true;
401 }

445 static void PrintFloatSuccinctly(std::ostream &os, BaseFloat f) {
446  if (fabs(f) < 10000.0 && fabs(f) >= 10.0) {

460 // Returns a string that summarizes a vector fairly succintly, for
461 // printing stats in info lines.
462 std::string SummarizeVector(const Vector<BaseFloat> &vec) {
463  std::ostringstream os;
464  if (vec.Dim() < 10) {
465  os << "[ ";
466  for (int32 i = 0; i < vec.Dim(); i++) {
467  PrintFloatSuccinctly(os, vec(i));
468  os << ' ';
469  }
470  os << "]";
471  } else {
472  // print out mean and standard deviation, and some selected values.
473  BaseFloat mean = vec.Sum() / vec.Dim(),
474  stddev = sqrt(VecVec(vec, vec) / vec.Dim() - mean * mean);
475 
476  std::string percentiles_str = "0,1,2,5 10,20,50,80,90 95,98,99,100";
477  std::vector<int32> percentiles;
478  bool ans = SplitStringToIntegers(percentiles_str, ", ", false,
479  &percentiles);
480  KALDI_ASSERT(ans);
481  os << "[percentiles(" << percentiles_str << ")=(";
482  Vector<BaseFloat> vec_sorted(vec);
483  std::sort(vec_sorted.Data(), vec_sorted.Data() + vec_sorted.Dim());
484  int32 n = vec.Dim() - 1;
485  for (size_t i = 0; i < percentiles.size(); i++) {
486  int32 percentile = percentiles[i];
487  BaseFloat value = vec_sorted((n * percentile) / 100);
488  PrintFloatSuccinctly(os, value);
489  if (i + 1 < percentiles.size())
490  os << (i == 3 || i == 8 ? ' ' : ',');
491  }
492  os << std::setprecision(3);
493  os << "), mean=" << mean << ", stddev=" << stddev << "]";
494  }
495  return os.str();
496 }
