31  const char *usage =
32  "Accumulate statistics for decision tree training.\n"
33  "This version accumulates statistics in the form of state-specific "
34  "SGMM stats; you need to use the program sgmm-build-tree to build "
35  "the tree (and sgmm-sum-tree-accs to sum the stats).\n"
36  "Usage: sgmm-acc-tree-stats [options] sgmm-model-in features-rspecifier "
37  "alignments-rspecifier [tree-accs-out]\n"
38  "e.g.: sgmm-acc-tree-stats --ci-phones=48:49 1.mdl scp:train.scp ark:1.ali 1.tacc\n";

45  SgmmGselectConfig sgmm_opts;
46  po.Register("binary", &binary, "Write output in binary mode");
47  po.Register("gselect", &gselect_rspecifier, "Precomputed Gaussian indices (rspecifier)");
48  po.Register("spk-vecs", &spkvecs_rspecifier, "Speaker vectors (rspecifier)");

50  "rspecifier for utterance to speaker map");

52  "indices of context-independent phones.");

54  po.Register("central-position", &P,
55  "Central context-window position (zero-based)");
56  sgmm_opts.Register(&po);

71  if (ci_phones_str != "") {

83  {

86  trans_model.Read(ki.Stream(), binary);

95  RandomAccessInt32VectorVectorReader gselect_reader(gselect_rspecifier);
96  RandomAccessBaseFloatVectorReaderMapped spkvecs_reader(spkvecs_rspecifier,
97  utt2spk_rspecifier);

99  std::map<EventType, SgmmClusterable*> tree_stats;

104  std::string utt = feature_reader.Key();
105  if (!alignment_reader.HasKey(utt)) {
106  num_err++;
107  } else {
108  const Matrix<BaseFloat> &mat = feature_reader.Value();
109  const std::vector<int32> &alignment = alignment_reader.Value(utt);
110 
111  if (!gselect_reader.HasKey(utt) ||
112 
113  gselect_reader.Value(utt).size() != mat.NumRows()) {
114  KALDI_WARN << "No gselect information for utterance " << utt
115  << " (or wrong size)";
116  num_err++;
117  continue;
118  }
119 
120  const std::vector<std::vector<int32> > &gselect =
121  gselect_reader.Value(utt);
122 
123  if (alignment.size() != mat.NumRows()) {
124  KALDI_WARN << "Alignments has wrong size "<< (alignment.size())<<" vs. "<< (mat.NumRows());
125  num_err++;
126  continue;
127  }
128

130  if (spkvecs_reader.IsOpen()) {
131  if (spkvecs_reader.HasKey(utt)) {
132  spk_vars.v_s = spkvecs_reader.Value(utt);
133  am_sgmm.ComputePerSpkDerivedVars(&spk_vars);
134  } else {
135  KALDI_WARN << "Cannot find speaker vector for " << utt;
136  }
137  } // else spk_vars is "empty"
138 
139 
140  // The work gets done here.
141  if (!AccumulateSgmmTreeStats(trans_model,
142  am_sgmm,
143  H,
144  N, P,
145  ci_phones,
146  alignment,
147  gselect,
148  spk_vars,
149  mat,
150  &tree_stats)) {
151  num_err++;
152  } else {
153  num_done++;
154  if (num_done % 1000 == 0)
155  KALDI_LOG << "Processed " << num_done << " utterances.";
156  }
157  }
158  }
159 
160  BuildTreeStatsType stats; // Converting from a map to a vector of pairs.
161 
162  for (std::map<EventType, SgmmClusterable*>::const_iterator iter = tree_stats.begin();
163  iter != tree_stats.end();
164  iter++ ) {
165  stats.push_back(std::make_pair(iter->first, static_cast<Clusterable*>(iter->second)));
166  }
167  tree_stats.clear();
168 
169  {
170  Output ko(accs_wxfilename, binary);
171  WriteBuildTreeStats(ko.Stream(), binary, stats);
172  }
173  KALDI_LOG << "Accumulated stats for " << num_done << " files, "
174  << num_err << " failed.";
175  KALDI_LOG << "Number of separate stats (context-dependent states) is "
176  << stats.size();
177  DeleteBuildTreeStats(&stats);
178  return (num_done != 0 ? 0 : 1);
179  } catch(const std::exception &e) {
180  std::cerr << e.what();
181  return -1;
182  }
183 }
