33 void InitAndOutputSgmm(const HmmTopology &topo,
34  const AmSgmm &am_sgmm,
35  const ContextDependency &ctx_dep,
36  const std::vector<SpMatrix<double> > &H,
37  const BuildTreeStatsType &stats,
38  const std::string &sgmm_wxfilename,
39  bool binary) {
40  int32 num_pdfs = ctx_dep.NumPdfs();
41  AmSgmm am_sgmm_out;
42  am_sgmm_out.CopyGlobalsInitVecs(am_sgmm, am_sgmm.PhoneSpaceDim(),
43  am_sgmm.SpkSpaceDim(), num_pdfs);
44  MleAmSgmmOptions opts; // Use default options; we can change this later
45  // if we need to use any non-default options.
46  MleAmSgmmUpdater updater(opts);
47 
48  std::vector<BuildTreeStatsType> split_stats;
49  SplitStatsByMap(stats, ctx_dep.ToPdfMap(), &split_stats);
50  // Make sure each leaf has stats.
51  for (size_t i = 0; i < split_stats.size(); i++)
52  KALDI_ASSERT(! split_stats[i].empty() && "Tree has leaves with no stats."
53  " Modify your roots file as necessary to fix this.");
54  std::vector<Clusterable*> summed_stats;
55  SumStatsVec(split_stats, &summed_stats);
56 
57  std::vector<SgmmClusterable*> &summed_sgmm_stats =
58  *(reinterpret_cast<std::vector<SgmmClusterable*>*> (&summed_stats));
59 
60  for (int32 iter = 0; iter < 5; iter++) { // Update for
61  // several iterations; we're starting from zero so we won't
62  // converge exactly on the first iteration.
63  updater.UpdatePhoneVectorsCheckedFromClusterable(summed_sgmm_stats,
64  H,
65  &am_sgmm_out);
66  }
67  DeletePointers(&summed_stats);
68 
69  TransitionModel trans_model_out(ctx_dep, topo);
70  {
71  Output ko(sgmm_wxfilename, binary);
72  am_sgmm_out.ComputeNormalizers();
73  trans_model_out.Write(ko.Stream(), binary);
74  am_sgmm_out.Write(ko.Stream(), binary, kSgmmWriteAll);
75  }
76 }

81 int main(int argc, char *argv[]) {
82  try {

86  const char *usage =
87  "Initialize an SGMM from a previously built SGMM, a tree, \n"

95  po.Read(argc, argv);
96 
97  if (po.NumArgs() != 4) {
98  po.PrintUsage();
99  exit(1);
100  }
101 
102  std::string sgmm_in_filename = po.GetArg(1),
103  tree_in_filename = po.GetArg(2),
104  tree_stats_filename = po.GetArg(3),
105  sgmm_out_filename = po.GetArg(4);
106 
107  AmSgmm am_sgmm;
108  TransitionModel trans_model;
109  {
110  bool binary;
111  Input ki(sgmm_in_filename, &binary);
112  trans_model.Read(ki.Stream(), binary);
113  am_sgmm.Read(ki.Stream(), binary);
114  }
115 
116  const HmmTopology &topo = trans_model.GetTopo();
117  std::vector<SpMatrix<double> > H;
118  am_sgmm.ComputeH(&H);
119 
120  ContextDependency ctx_dep;
121  {
122  bool binary_in;
123  Input ki(tree_in_filename.c_str(), &binary_in);
124  ctx_dep.Read(ki.Stream(), binary_in);
125  }
126 
127  BuildTreeStatsType stats;
128  {
129  bool binary_in;
130  SgmmClusterable sc(am_sgmm, H); // dummy stats needed to provide
131  // type info, and access to am_sgmm and H.
132  Input ki(tree_stats_filename, &binary_in);
133  ReadBuildTreeStats(ki.Stream(), binary_in, sc, &stats);
134  }
135  KALDI_LOG << "Number of separate statistics is " << stats.size();
136 
137  InitAndOutputSgmm(topo, am_sgmm, ctx_dep, H, stats,
138  sgmm_out_filename, binary);
139 
140  KALDI_LOG << "Written model to " << sgmm_out_filename;
141  } catch(const std::exception &e) {
142  std::cerr << e.what() << '\n';
143  return -1;
144  }
