340 template<typename Real>
341 inline void AmSgmm::GetSubstateMean(int32 j, int32 m, int32 i,
342  VectorBase<Real> *mean_out) const {
343  KALDI_ASSERT(mean_out != NULL);
344  KALDI_ASSERT(j < NumPdfs() && m < NumSubstates(j) && i < NumGauss());
345  KALDI_ASSERT(mean_out->Dim() == FeatureDim());
346  Vector<BaseFloat> mean_tmp(FeatureDim());
347  mean_tmp.AddMatVec(1.0, M_[i], kNoTrans, v_[j].Row(m), 0.0);
348  mean_out->CopyFromVec(mean_tmp);
349 }
