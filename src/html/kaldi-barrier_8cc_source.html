80 int32 Barrier::Wait() {
81  if (threshold_ == 0)
82  KALDI_ERR << "Cannot wait when ``threshold'' value was not set";
83 
84  if (pthread_mutex_lock(&mutex_) != 0)
85  KALDI_ERR << "Cannot lock pthread mutex";
86 
87  int32 cycle = cycle_; // memorize which cycle we're in
88 
89  int32 ret;
90  if (--counter_ == 0) { // this is the last thread
91  cycle_ = !cycle_;
92  counter_ = threshold_;
93  // wake up.
94  if (pthread_cond_broadcast(&cv_) != 0) {
95  KALDI_ERR << "Error on pthred_cond_broadcast";
96  }
97  /*
98  * The last incoming thread will return -1, the others 0
99  */
100  ret = -1;
101 
102  } else {
103  /*
104  * Wait with thread cancellation disabled, barrier waiting
105  * should not be a cancellation point.
106  */
107  int32 cancel, tmp;
108  pthread_setcancelstate(PTHREAD_CANCEL_DISABLE, &cancel);
109 
110  /*
111  * Wait until the barrier's cycle_ changes, like this we know
112  * that the last thread arrived and a brodcast was sent.
113  */
114  while (cycle == cycle_) {
115  // go to sleep
116  if (pthread_cond_wait(&cv_, &mutex_) != 0) {
117  KALDI_ERR << "Error on pthread_cond_wait";
118  }
119  }
120 
121  // Restore the thread's cancel state
122 
123  pthread_setcancelstate(cancel, &tmp);
124 
125  /*
126  * Indicate that it is not the last thread
127  */
128  ret = 0;
129  }
130 
131  if (pthread_mutex_unlock (&mutex_) != 0) {
132  KALDI_ERR << "Error unlcoking pthread mutex";
133  }
134 
135  return ret;
136 }
