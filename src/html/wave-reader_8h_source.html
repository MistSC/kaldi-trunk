111 class WaveHolder {
112  public:

114 
115  static bool Write(std::ostream &os, bool binary, const T &t) {
116  // We don't write the binary-mode header here [always binary].
117  if (!binary)
118  KALDI_ERR << "Wave data can only be written in binary mode.";
119  try {
120  t.Write(os); // throws exception on failure.
121  return true;
122  } catch (const std::exception &e) {
123  KALDI_WARN << "Exception caught in WaveHolder object (writing).";
124  if (!IsKaldiError(e.what())) { std::cerr << e.what(); }
125  return false; // write failure.
126  }
127  }
