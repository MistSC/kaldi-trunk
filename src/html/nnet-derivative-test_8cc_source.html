31 void ComputeMinAndMaxTimes(const std::vector<Index> &indexes,
32  int32 *min_t,
33  int32 *max_t) {
34  KALDI_ASSERT(!indexes.empty());
35  *min_t = indexes[0].t;
36  *max_t = *min_t;
37  for (int32 n = 1; n < static_cast<int32>(indexes.size()); n++) {
38  *min_t = std::min(*min_t, indexes[n].t);
39  *max_t = std::max(*max_t, indexes[n].t);
40  }
41 }

44 // This function is called if you want to set min_deriv_time and max_deriv_time.
45 // It works out some meaningful values to set, based on the config.
46 void SetDerivTimesOptions(const ComputationRequest &request,
47  NnetOptimizeOptions *opt_config) {
48  int32 min_t, max_t;
49  KALDI_ASSERT(request.inputs[0].name == "input");
50  const std::vector<Index> &input_indexes = request.inputs[0].indexes;
51  ComputeMinAndMaxTimes(input_indexes, &min_t, &max_t);
52 
53  int32 orig_min_t = min_t, orig_max_t = max_t;
54  int t_length = max_t + 1 - min_t;
55  KALDI_ASSERT(t_length > 0);
56  if (t_length == 1)
57  return;
58  if (RandInt(0, 2) == 0) {
59  // remove as much as 4 frames from the left (but don't remove everything).
60  min_t += std::min(4, RandInt(0, t_length - 1));
61  opt_config->min_deriv_time = min_t;
62  t_length = max_t + 1 - min_t;
63  KALDI_ASSERT(t_length > 0);
64  }
65  if (RandInt(0, 2) == 0) {
66  max_t -= std::min(4, RandInt(0, t_length - 1));
67  opt_config->max_deriv_time = max_t;
68  t_length = max_t + 1 - min_t;
69  KALDI_ASSERT(t_length > 0);
70  }
71  if (RandInt(0, 4) == 0) {
72  // ensure that all derivs will be pruned away;
73  // this tests more code.
74  min_t = orig_min_t - 10;
75  max_t = min_t + 1;
76  }
77 
78  int32 output_min_t, output_max_t;
79  KALDI_ASSERT(request.outputs[0].name == "output");
80  ComputeMinAndMaxTimes(request.outputs[0].indexes,
81  &output_min_t, &output_max_t);
82 
83  KALDI_LOG << "ComputationRequest has output (min,max) = (" << output_min_t
84  << ',' << output_max_t << "), input (min,max) = (" << orig_min_t
85  << ',' << orig_max_t << "), limiting deriv times to ("
86  << opt_config->min_deriv_time << ','
87  << opt_config->max_deriv_time;
88 }

90 // This test makes sure that the model-derivatives are correct.
91 void UnitTestNnetModelDerivatives() {
92  int32 num_tries = 20, num_success = 0, num_fail = 0;
93  for (int32 n = 0; n < num_tries; n++) {
94  struct NnetGenerationOptions gen_config;
95  //gen_config.allow_nonlinearity = false;
96  //gen_config.allow_recursion = false;
97  //gen_config.allow_final_nonlinearity = true;
98  bool allow_optimization = true;
99  bool limit_deriv_times = (RandInt(0, 2) == 0);
100 
101  std::vector<std::string> configs;
102  GenerateConfigSequence(gen_config, &configs);
103  Nnet nnet;
104  for (size_t j = 0; j < configs.size(); j++) {
105  KALDI_LOG << "Input config[" << j << "] is: " << configs[j];
106  std::istringstream is(configs[j]);
107  nnet.ReadConfig(is);
108  }
109 
110  ComputationRequest request;
111  std::vector<Matrix<BaseFloat> > inputs;
112  ComputeExampleComputationRequestSimple(nnet, &request, &inputs);
113 
114  // make sure that a model-derivative is requested, and an output-derivative
115  // is supplied.
116  request.need_model_derivative = true;
117  request.outputs[0].has_deriv = true;
118  // whether input-derivatives are required or not does not matter,
119  // so leave it as it is in that regard.
120 
121  NnetComputation computation;
122  Compiler compiler(request, nnet);
123 
124  CompilerOptions opts;
125  compiler.CreateComputation(opts, &computation);
126  {
127  std::ostringstream os;
128  computation.Print(os, nnet);
129  KALDI_LOG << "Generated computation is: " << os.str();
130  }
131  CheckComputationOptions check_config;
132  // we can do the rewrite check since it's before optimization.
133  check_config.check_rewrite = true;
134  ComputationChecker checker(check_config, nnet, computation);
135  checker.Check();
136 
137  if (RandInt(0, 3) != 0 && allow_optimization) {
138  NnetOptimizeOptions opt_config;
139  if (limit_deriv_times)
140  SetDerivTimesOptions(request, &opt_config);
141 
142  Optimize(opt_config, nnet, request, &computation);
143  std::ostringstream os;
144  computation.Print(os, nnet);
145  KALDI_LOG << "Optimized computation is: " << os.str();
146  check_config.check_rewrite = false;
147  ComputationChecker checker_opt(check_config, nnet, computation);
148  checker_opt.Check();
149  }
150 
151  NnetComputeOptions compute_opts;
152  if (RandInt(0, 1) == 0)
153  compute_opts.debug = true;
154  computation.ComputeCudaIndexes();
155 
156 
157  Nnet nnet_deriv(nnet);
158  bool is_gradient = true;
159  SetZero(is_gradient, &nnet_deriv); // forces "simple" update and unit
160  // learning rate.
161 
162  int32 num_directions = 4; // must be >= 1. Best if it's >1, will reduce
163  // the probability of random failures.
164 
165  // the order of these vectors is:
166  // [ un-perturbed, perturbed-1, perturbed-2, perturbed-3 ].
167  std::vector<BaseFloat> measured_objf(num_directions + 1, 0.0),
168  predicted_objf_change(num_directions + 1, 0.0);
169  BaseFloat delta = 5.0e-04;
170 
171  // output_deriv is the derivative of the objective function w.r.t. the
172  // (single) output. We make the objf a linear function of the output and
173  // just set the output_deriv to be a random matrix, which defines the
174  // objective function.
175  CuMatrix<BaseFloat> output_deriv;
176  output_deriv.Resize(request.outputs[0].indexes.size(),
177  nnet.OutputDim("output"));
178  output_deriv.SetRandn();
179 
180  // pass 0 is the forward pass with the un-perturbed model.
181  // Other passes are with various differently-perturbed versions of
182  // the model.
183  for (int32 pass = 0; pass <= num_directions; pass++) {
184  Nnet nnet_copy(nnet);
185  if (pass > 0)
186  PerturbParams(delta, &nnet_copy);
187 
188  NnetComputer computer(compute_opts,
189  computation,
190  nnet_copy,
191  (pass == 0 ? &nnet_deriv : &nnet_copy));
192 
193 
194  // provide the input to the computation.
195  for (size_t i = 0; i < request.inputs.size(); i++) {
196  CuMatrix<BaseFloat> temp(inputs[i]);
197  computer.AcceptInput(request.inputs[i].name, &temp);
198  }
199 
200  KALDI_LOG << "Running forward computation";
201  computer.Forward();
202 
203  const CuMatrixBase<BaseFloat> &output(computer.GetOutput("output"));
204  KALDI_LOG << "Output sum for pass " << pass << " is " << output.Sum();
205  BaseFloat objf = TraceMatMat(output, output_deriv, kTrans);
206  measured_objf[pass] = objf;
207 
208  if (pass == 0) {
209  // we need to do the backward computation (to get the model derivative)
210  CuMatrix<BaseFloat> temp(output_deriv);
211  computer.AcceptOutputDeriv("output", &temp);
212  KALDI_LOG << "Running backward computation";
213  computer.Backward();
214  } else {
215  // work out the predicted objf-change as dot-product of deriv and
216  // parameter-change. The expression below can be interpreted as
217  // DotProduct(nnet_copy - nnet, nnet_deriv).
218  predicted_objf_change[pass] = DotProduct(nnet_copy, nnet_deriv) -
219  DotProduct(nnet, nnet_deriv);
220  }
221  }
222 
223  Vector<BaseFloat> predicted_objf_change_vec(num_directions),
224  measured_objf_change_vec(num_directions);
225  for (int32 d = 0; d < num_directions; d++) {
226  BaseFloat predicted_change = predicted_objf_change[d+1],
227  measured_change = measured_objf[d+1] - measured_objf[0];
228  predicted_objf_change_vec(d) = predicted_change;
229  measured_objf_change_vec(d) = measured_change;
230  }
231  KALDI_LOG << "Vector of predicted objf-change is: "
232  << predicted_objf_change_vec;
233  KALDI_LOG << "Vector of measured objf-change is: "
234  << measured_objf_change_vec;
235  BaseFloat delta_thresh = 0.05;
236  if (limit_deriv_times) {
237  KALDI_LOG << "Not checking that predicted/measured changes matched "
238  << "because we limited times of derivatives.";
239  } else {
240  if (!ApproxEqual(predicted_objf_change_vec,
241  measured_objf_change_vec, delta_thresh)) {
242  KALDI_WARN << "Predicted and measured objf-changes differ too much.";
243  num_fail++;
244  } else {
245  num_success++;
246  }
247  }
248  }
249  KALDI_LOG << "Model-derivative check succeeded for " << num_success << " out of "
250  << (num_fail + num_success) << " tries.";
251  int32 num_checked = num_fail + num_success;
252  if (num_success < num_checked - (2 + num_checked / 5))
253  KALDI_ERR << "Failed too many times.";
254 }

257 // This test makes sure that the input-derivatives are correct.
258 void UnitTestNnetInputDerivatives() {
259  int32 num_tries = 20, num_success = 0;
260  for (int32 n = 0; n < num_tries; n++) {
261  struct NnetGenerationOptions gen_config;
262  //gen_config.allow_nonlinearity = false;
263  //gen_config.allow_recursion = false;
264  //gen_config.allow_final_nonlinearity = true;
265  bool allow_optimization = true;
266 
267  std::vector<std::string> configs;
268  GenerateConfigSequence(gen_config, &configs);
269  Nnet nnet;
270  for (size_t j = 0; j < configs.size(); j++) {
271  KALDI_LOG << "Input config[" << j << "] is: " << configs[j];
272  std::istringstream is(configs[j]);
273  nnet.ReadConfig(is);
274  }
275 
276  ComputationRequest request;
277  std::vector<Matrix<BaseFloat> > inputs;
278  ComputeExampleComputationRequestSimple(nnet, &request, &inputs);
279 
280  // make sure that all inputs and outputs have derivatives requested/provided,
281  // and that the model-update (need_model_derivative) is not requested.
282  request.need_model_derivative = false;
283  for (int32 i = 0; i < request.inputs.size(); i++)
284  request.inputs[i].has_deriv = true;
285  request.outputs[0].has_deriv = true;
286 
287  NnetComputation computation;
288  Compiler compiler(request, nnet);
289 
290  CompilerOptions opts;
291  compiler.CreateComputation(opts, &computation);
292  {
293  std::ostringstream os;
294  computation.Print(os, nnet);
295  KALDI_LOG << "Generated computation is: " << os.str();
296  }
297  CheckComputationOptions check_config;
298  // we can do the rewrite check since it's before optimization.
299  check_config.check_rewrite = true;
300  ComputationChecker checker(check_config, nnet, computation);
301  checker.Check();
302 
303  if (RandInt(0, 3) != 0 && allow_optimization) {
304  NnetOptimizeOptions opt_config;
305  // opt_config.initialize_undefined = false; // temp
306  Optimize(opt_config, nnet, request, &computation);
307  std::ostringstream os;
308  computation.Print(os, nnet);
309  KALDI_LOG << "Optimized computation is: " << os.str();
310  }
311 
312  NnetComputeOptions compute_opts;
313  if (RandInt(0, 1) == 0)
314  compute_opts.debug = true;
315  computation.ComputeCudaIndexes();
316 
317  // the only reason we might need to provide the &nnet parameter is if the
318  // StoreStats() operation had been requested. We made sure no model update
319  // is being performed.
320  NnetComputer computer(compute_opts,
321  computation,
322  nnet,
323  &nnet);
324 
325  int32 num_directions = 3; // must be >= 1. Best if it's >1, will reduce
326  // the probability of random failures.
327 
328  // the order of these vectors is:
329  // [ un-perturbed, perturbed-1, perturbed-2, perturbed-3, un-perturbed ].
330  // we compute un-perturbed twice to double-check the model did not change.
331  std::vector<BaseFloat> measured_objf(num_directions + 2, 0.0),
332  predicted_objf_change(num_directions + 2, 0.0);
333  BaseFloat delta = 1.0e-03;
334 
335  // output_deriv is the derivative of the objective function w.r.t. the
336  // (single) output. We make the objf a linear function of the output and
337  // just set the output_deriv to be a random matrix, which defines the
338  // objective function.
339  CuMatrix<BaseFloat> output_deriv;
340  output_deriv.Resize(request.outputs[0].indexes.size(),
341  nnet.OutputDim("output"));
342  output_deriv.SetRandn();
343 
344  std::vector<CuMatrix<BaseFloat> > delta_inputs(inputs.size());
345  std::vector<CuMatrix<BaseFloat> > input_derivs(inputs.size());
346 
347  // pass 0 is the forward pass with the un-perturbed features; so is
348  // pass num_directions + 1.
349  // Other passes are with various differently-perturbed versions of
350  // the features.
351  for (int32 pass = 0; pass <= num_directions + 1; pass++) {
352  // provide the input to the computations.
353  for (size_t i = 0; i < request.inputs.size(); i++) {
354  CuMatrix<BaseFloat> temp(inputs[i]);
355  if (pass > 0 && pass <= num_directions) { // Perturb the input randomly.
356  delta_inputs[i].Resize(inputs[i].NumRows(), inputs[i].NumCols());
357  delta_inputs[i].SetRandn();
358  delta_inputs[i].Scale(delta);
359  // if there are >1 inputs, sometimes set the delta for input 0 to
360  // zero. might sometimes give more accurate test of error in iVector
361  // derivative computation.
362  if (i == 0 && request.inputs.size() > 1 && RandInt(0, 1) == 0)
363  delta_inputs[i].SetZero();
364  temp.AddMat(1.0, delta_inputs[i]);
365  predicted_objf_change[pass] += TraceMatMat(input_derivs[i],
366  delta_inputs[i], kTrans);
367  }
368  computer.AcceptInput(request.inputs[i].name, &temp);
369  }
370 
371  KALDI_LOG << "Running forward computation";
372  computer.Forward();
373 
374  const CuMatrixBase<BaseFloat> &output(computer.GetOutput("output"));
375  KALDI_LOG << "Output sum for pass " << pass << " is " << output.Sum();
376  BaseFloat objf = TraceMatMat(output, output_deriv, kTrans);
377  measured_objf[pass] = objf;
378 
379  if (pass == 0) {
380  // We need to compute the input derivatives.
381  CuMatrix<BaseFloat> temp(output_deriv);
382  computer.AcceptOutputDeriv("output", &temp);
383  KALDI_LOG << "Running backward computation";
384  computer.Backward();
385  for (size_t i = 0; i < request.inputs.size(); i++) {
386  input_derivs[i] = computer.GetInputDeriv(request.inputs[i].name);
387  KALDI_LOG << "Input-deriv norm for '" << request.inputs[i].name
388  << "' is " << input_derivs[i].FrobeniusNorm();
389  }
390  }
391  }
392  KALDI_ASSERT(ApproxEqual(measured_objf[0],
393  measured_objf[num_directions + 1]));
394 
395  Vector<BaseFloat> predicted_objf_change_vec(num_directions),
396  measured_objf_change_vec(num_directions);
397  for (int32 d = 0; d < num_directions; d++) {
398  BaseFloat predicted_change = predicted_objf_change[d+1],
399  measured_change = measured_objf[d+1] - measured_objf[0];
400  predicted_objf_change_vec(d) = predicted_change;
401  measured_objf_change_vec(d) = measured_change;
402  }
403  KALDI_LOG << "Vector of predicted objf-change is: "
404  << predicted_objf_change_vec;
405  KALDI_LOG << "Vector of measured objf-change is: "
406  << measured_objf_change_vec;
407  BaseFloat delta_thresh = 0.1;
408  if (!ApproxEqual(predicted_objf_change_vec,
409  measured_objf_change_vec, delta_thresh)) {
410  KALDI_WARN << "Predicted and measured objf-changes differ too much.";
411  } else {
412  num_success++;
413  }
414  }
415  KALDI_LOG << "Input-derivative check succeeded for " << num_success << " out of "
416  << num_tries << " tries.";
417  if (num_success < num_tries - (2 + num_tries / 5))
418  KALDI_ERR << "Failed too many times.";
419 }
