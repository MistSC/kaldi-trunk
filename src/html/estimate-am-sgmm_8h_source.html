425 // This class, used in multi-core implementation of the updates of the "w_i"
426 // quantities, was previously in estimate-am-sgmm.cc, but is being moved to the
427 // header so it can be used in estimate-am-sgmm-ebw.cc. It is responsible for
428 // computing, in parallel, the F_i and g_i quantities used in the updates of
429 // w_i.
430 class UpdateWParallelClass: public MultiThreadable {
431  public:
432  UpdateWParallelClass(const MleAmSgmmAccs &accs,
433  const AmSgmm &model,
434  const Matrix<double> &w,
435  Matrix<double> *F_i,
436  Matrix<double> *g_i,
437  double *tot_like):
438  accs_(accs), model_(model), w_(w),
439  F_i_ptr_(F_i), g_i_ptr_(g_i), tot_like_ptr_(tot_like) {
440  tot_like_ = 0.0;
441  F_i_.Resize(F_i->NumRows(), F_i->NumCols());
442  g_i_.Resize(g_i->NumRows(), g_i->NumCols());
443  }
444 
445  ~UpdateWParallelClass() {
446  F_i_ptr_->AddMat(1.0, F_i_, kNoTrans);
447  g_i_ptr_->AddMat(1.0, g_i_, kNoTrans);
448  *tot_like_ptr_ += tot_like_;
449  }
450 
451  inline void operator() () {
452  // Note: give them local copy of the sums we're computing,
453  // which will be propagated to the total sums in the destructor.
454  MleAmSgmmUpdater::UpdateWParallelGetStats(accs_, model_, w_,
455  &F_i_, &g_i_, &tot_like_,
456  num_threads_, thread_id_);
457  }
458  private:
459  // MleAmSgmmUpdater *updater_;
460  const MleAmSgmmAccs &accs_;
461  const AmSgmm &model_;
462  const Matrix<double> &w_;
463  Matrix<double> *F_i_ptr_;
464  Matrix<double> *g_i_ptr_;
465  Matrix<double> F_i_;
466  Matrix<double> g_i_;
467  double *tot_like_ptr_;
468  double tot_like_;
469 };
