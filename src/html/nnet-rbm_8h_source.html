100  void InitData(std::istream &is) {
101  // define options
102  std::string vis_type;
103  std::string hid_type;
104  float vis_bias_mean = 0.0, vis_bias_range = 0.0,
105  hid_bias_mean = 0.0, hid_bias_range = 0.0,
106  param_stddev = 0.1;
107  std::string vis_bias_cmvn_file; // initialize biases to logit(p_active)
108  // parse config
109  std::string token;
110  while (!is.eof()) {
111  ReadToken(is, false, &token);
112  if (token == "<VisibleType>") ReadToken(is, false, &vis_type);
113  else if (token == "<HiddenType>") ReadToken(is, false, &hid_type);
114  else if (token == "<VisibleBiasMean>") ReadBasicType(is, false, &vis_bias_mean);
115  else if (token == "<VisibleBiasRange>") ReadBasicType(is, false, &vis_bias_range);
116  else if (token == "<HiddenBiasMean>") ReadBasicType(is, false, &hid_bias_mean);
117  else if (token == "<HiddenBiasRange>") ReadBasicType(is, false, &hid_bias_range);
118  else if (token == "<ParamStddev>") ReadBasicType(is, false, &param_stddev);
119  else if (token == "<VisibleBiasCmvnFilename>") ReadToken(is, false, &vis_bias_cmvn_file);
120  else KALDI_ERR << "Unknown token " << token << " Typo in config?";
121  is >> std::ws; // eat-up whitespace
122  }
123 
124  //
125  // initialize
126  //
127  if (vis_type == "bern" || vis_type == "Bernoulli") vis_type_ = RbmBase::Bernoulli;
128  else if (vis_type == "gauss" || vis_type == "Gaussian") vis_type_ = RbmBase::Gaussian;
129  else KALDI_ERR << "Wrong <VisibleType>" << vis_type;
130  //
131  if (hid_type == "bern" || hid_type == "Bernoulli") hid_type_ = RbmBase::Bernoulli;
132  else if (hid_type == "gauss" || hid_type == "Gaussian") hid_type_ = RbmBase::Gaussian;
133  else KALDI_ERR << "Wrong <HiddenType>" << hid_type;
134  // visible-hidden connections
135  Matrix<BaseFloat> mat(output_dim_, input_dim_);
136  for (int32 r=0; r<output_dim_; r++) {
137  for (int32 c=0; c<input_dim_; c++) {
138  mat(r,c) = param_stddev * RandGauss(); // 0-mean Gauss with given std_dev
139  }
140  }
141  vis_hid_ = mat;
142  // hidden-bias
143  Vector<BaseFloat> vec(output_dim_);
144  for (int32 i=0; i<output_dim_; i++) {
145  // +/- 1/2*bias_range from bias_mean:
146  vec(i) = hid_bias_mean + (RandUniform() - 0.5) * hid_bias_range;
147  }
148  hid_bias_ = vec;
149  // visible-bias
150  if (vis_bias_cmvn_file == "") {
151  Vector<BaseFloat> vec2(input_dim_);
152  for (int32 i=0; i<input_dim_; i++) {
153  // +/- 1/2*bias_range from bias_mean:
154  vec2(i) = vis_bias_mean + (RandUniform() - 0.5) * vis_bias_range;
155  }
156  vis_bias_ = vec2;
157  } else {
158  KALDI_LOG << "Initializing from <VisibleBiasCmvnFilename> " << vis_bias_cmvn_file;
159  Nnet cmvn;
160  cmvn.Read(vis_bias_cmvn_file);
161  // getting probablity that neuron fires:
162  Vector<BaseFloat> p(dynamic_cast<AddShift&>(cmvn.GetComponent(0)).GetShiftVec());
163  p.Scale(-1.0);
164  // compute logit:
165  Vector<BaseFloat> logit_p(p.Dim());
166  for(int32 d = 0; d < p.Dim(); d++) {
167  if(p(d) < 0.0001) p(d) = 0.0001;
168  if(p(d) > 0.9999) p(d) = 0.9999;
169  logit_p(d) = Log(p(d)) - Log(1.0 - p(d));
170  }
171  vis_bias_ = logit_p;
172  KALDI_ASSERT(vis_bias_.Dim() == InputDim());
173  }
174  //
175  }

233  // RBM training API
234  void Reconstruct(const CuMatrixBase<BaseFloat> &hid_state, CuMatrix<BaseFloat> *vis_probs) {
235  // check the dim
236  if (output_dim_ != hid_state.NumCols()) {
237  KALDI_ERR << "Nonmatching dims, component:" << output_dim_ << " data:" << hid_state.NumCols();
238  }
239  // optionally allocate buffer
240  if (input_dim_ != vis_probs->NumCols() || hid_state.NumRows() != vis_probs->NumRows()) {
241  vis_probs->Resize(hid_state.NumRows(), input_dim_);
242  }
243 
244  // pre-fill with bias
245  vis_probs->AddVecToRows(1.0, vis_bias_, 0.0);
246  // multiply by weights
247  vis_probs->AddMatMat(1.0, hid_state, kNoTrans, vis_hid_, kNoTrans, 1.0);
248  // optionally apply sigmoid
249  if (vis_type_ == RbmBase::Bernoulli) {
250  vis_probs->Sigmoid(*vis_probs);
251  }

254  void RbmUpdate(const CuMatrixBase<BaseFloat> &pos_vis, const CuMatrixBase<BaseFloat> &pos_hid, const CuMatrixBase<BaseFloat> &neg_vis, const CuMatrixBase<BaseFloat> &neg_hid) {
255  // dims
256  KALDI_ASSERT(pos_vis.NumRows() == pos_hid.NumRows() &&
257  pos_vis.NumRows() == neg_vis.NumRows() &&
258  pos_vis.NumRows() == neg_hid.NumRows() &&
259  pos_vis.NumCols() == neg_vis.NumCols() &&
260  pos_hid.NumCols() == neg_hid.NumCols() &&
261  pos_vis.NumCols() == input_dim_ &&
262  pos_hid.NumCols() == output_dim_);
263 
264  // lazy initialization of buffers
265  if ( vis_hid_corr_.NumRows() != vis_hid_.NumRows() ||
266  vis_hid_corr_.NumCols() != vis_hid_.NumCols() ||
267  vis_bias_corr_.Dim() != vis_bias_.Dim() ||
268  hid_bias_corr_.Dim() != hid_bias_.Dim() ){
269  vis_hid_corr_.Resize(vis_hid_.NumRows(),vis_hid_.NumCols(),kSetZero);
270  vis_bias_corr_.Resize(vis_bias_.Dim(), kSetZero);
271  hid_bias_corr_.Resize(hid_bias_.Dim(), kSetZero);
272  }
273 
274  // ANTI-WEIGHT-EXPLOSION PROTECTION (Gaussian-Bernoulli RBM)
275  // in the following section we detect that the weights in Gaussian-Bernoulli RBM
276  // are about to explode. The weight explosion is caused by large variance of the
277  // reconstructed data, which causes a feed-back loop that keeps increasing the weights.
278  //
279  // To avoid explosion, the variance of the visible-data and reconstructed-data
280  // should be about the same. The model is particularly sensitive at the very
281  // beginning of the CD-1 training.
282  //
283  // We compute variance of a)input mini-batch b)reconstruction.
284  // When the ratio b)/a) is larger than 2, we:
285  // 1. scale down the weights and biases by b)/a) (for next mini-batch b)/a) gets 1.0)
286  // 2. shrink learning rate by 0.9x
287  // 3. reset the momentum buffer
288  //
289  // Also a warning message is put to log. Note that in later stage
290  // the learning-rate returns to its original value.
291  //
292  // An alternative approach is to use smaller values in weight-matrix initialization.
293  //
294  if (vis_type_ == RbmBase::Gaussian) {
295  // check the data have no nan/inf:
296  CheckNanInf(pos_vis,"pos_vis");
297  CheckNanInf(pos_hid,"pos_hid");
298  CheckNanInf(neg_vis,"neg_vis");
299  CheckNanInf(neg_hid,"pos_hid");
300 
301  // get standard deviations of pos_vis and neg_vis:
302  BaseFloat pos_vis_std = ComputeStdDev(pos_vis);
303  BaseFloat neg_vis_std = ComputeStdDev(neg_vis);
304 
305  // monitor the standard deviation mismatch : data vs. reconstruction
306  if (pos_vis_std * 2 < neg_vis_std) {
307  // 1) scale-down the weights and biases
308  BaseFloat scale = pos_vis_std / neg_vis_std;
309  vis_hid_.Scale(scale);
310  vis_bias_.Scale(scale);
311  hid_bias_.Scale(scale);
312  // 2) reduce the learning rate
313  rbm_opts_.learn_rate *= 0.9;
314  // 3) reset the momentum buffers
315  vis_hid_corr_.SetZero();
316  vis_bias_corr_.SetZero();
317  hid_bias_corr_.SetZero();
318 
319  KALDI_WARN << "Mismatch between pos_vis and neg_vis variances, "
320  << "danger of weight explosion. a) Reducing weights with scale " << scale
321  << " b) Lowering learning rate to " << rbm_opts_.learn_rate
322  << " [pos_vis_std:" << pos_vis_std
323  << ",neg_vis_std:" << neg_vis_std << "]";
324  return; /* i.e. don't update weights with current stats, as the update would be too BIG */
325  }
326  }
327  //
328  // End of weight-explosion check
329 
330 
331  // We use these training hyper-parameters
332  //
333  const BaseFloat lr = rbm_opts_.learn_rate;
334  const BaseFloat mmt = rbm_opts_.momentum;
335  const BaseFloat l2 = rbm_opts_.l2_penalty;
336 
337  // UPDATE vishid matrix
338  //
339  // vishidinc = momentum*vishidinc + ...
340  // epsilonw*( (posprods-negprods)/numcases - weightcost*vishid);
341  //
342  // vishidinc[t] = -(epsilonw/numcases)*negprods + momentum*vishidinc[t-1]
343  // +(epsilonw/numcases)*posprods
344  // -(epsilonw*weightcost)*vishid[t-1]
345  //
346  BaseFloat N = static_cast<BaseFloat>(pos_vis.NumRows());
347  vis_hid_corr_.AddMatMat(-lr/N, neg_hid, kTrans, neg_vis, kNoTrans, mmt);
348  vis_hid_corr_.AddMatMat(+lr/N, pos_hid, kTrans, pos_vis, kNoTrans, 1.0);
349  vis_hid_corr_.AddMat(-lr*l2, vis_hid_);
350  vis_hid_.AddMat(1.0, vis_hid_corr_);
351 
352  // UPDATE visbias vector
353  //
354  // visbiasinc = momentum*visbiasinc + (epsilonvb/numcases)*(posvisact-negvisact);
355  //
356  vis_bias_corr_.AddRowSumMat(-lr/N, neg_vis, mmt);
357  vis_bias_corr_.AddRowSumMat(+lr/N, pos_vis, 1.0);
358  vis_bias_.AddVec(1.0, vis_bias_corr_, 1.0);
359 
360  // UPDATE hidbias vector
361  //
362  // hidbiasinc = momentum*hidbiasinc + (epsilonhb/numcases)*(poshidact-neghidact);
363  //
364  hid_bias_corr_.AddRowSumMat(-lr/N, neg_hid, mmt);
365  hid_bias_corr_.AddRowSumMat(+lr/N, pos_hid, 1.0);
366  hid_bias_.AddVec(1.0, hid_bias_corr_, 1.0);
367  }
