41 void Spectrogram::Compute(const VectorBase<BaseFloat> &wave,
42  Matrix<BaseFloat> *output,
43  Vector<BaseFloat> *wave_remainder) {
44  KALDI_ASSERT(output != NULL);
45 
46  // Get dimensions of output features
47  int32 rows_out = NumFrames(wave.Dim(), opts_.frame_opts);
48  int32 cols_out = opts_.frame_opts.PaddedWindowSize()/2 +1;
49  if (rows_out == 0)
50  KALDI_ERR << "No frames fit in file (#samples is " << wave.Dim() << ")";
51  // Prepare the output buffer
52  output->Resize(rows_out, cols_out);
53 
54  // Optionally extract the remainder for further processing
55  if (wave_remainder != NULL)
56  ExtractWaveformRemainder(wave, opts_.frame_opts, wave_remainder);
57 
58  // Buffers
59  Vector<BaseFloat> window; // windowed waveform.
60  BaseFloat log_energy;
61 
62  // Compute all the freames, r is frame index..
63  for (int32 r = 0; r < rows_out; r++) {
64  // Cut the window, apply window function
65  ExtractWindow(wave, r, opts_.frame_opts, feature_window_function_,
66  &window, (opts_.raw_energy ? &log_energy : NULL));
67 
68  // Compute energy after window function (not the raw one)
69  if (!opts_.raw_energy)
70  log_energy = Log(std::max(VecVec(window, window),
71  std::numeric_limits<BaseFloat>::min()));
72 
73  if (srfft_ != NULL) // Compute FFT using split-radix algorithm.
74  srfft_->Compute(window.Data(), true);
75  else // An alternative algorithm that works for non-powers-of-two
76  RealFft(&window, true);
77 
78  // Convert the FFT into a power spectrum.
79  ComputePowerSpectrum(&window);
80  SubVector<BaseFloat> power_spectrum(window, 0, window.Dim()/2 + 1);
81 
82  power_spectrum.ApplyFloor(std::numeric_limits<BaseFloat>::min());
83  power_spectrum.ApplyLog();
84 
85  // Output buffers
86  SubVector<BaseFloat> this_output(output->Row(r));
87  this_output.CopyFromVec(power_spectrum);
88  if (opts_.energy_floor > 0.0 && log_energy < log_energy_floor_) {
89  log_energy = log_energy_floor_;
90  }
91  this_output(0) = log_energy;
92  }
93 }
