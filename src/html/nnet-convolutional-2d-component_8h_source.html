89  // parse config

250  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
251  // useful dims
252  int32 num_input_fmaps = input_dim_ / (fmap_x_len_ * fmap_y_len_);
253  // int32 inp_fmap_size = fmap_x_len_ * fmap_y_len_;
254  int32 out_fmap_x_len = (fmap_x_len_ - filt_x_len_)/filt_x_step_ + 1;
255  int32 out_fmap_y_len = (fmap_y_len_ - filt_y_len_)/filt_y_step_ + 1;
256  int32 out_fmap_size = out_fmap_x_len*out_fmap_y_len;
257  int32 num_output_fmaps = output_dim_ / (out_fmap_x_len * out_fmap_y_len);
258  // this is total num_filters,
259  // so each input_fmap has size num_filters/num_input_fmaps
260  int32 num_filters = filters_.NumRows();
261  KALDI_ASSERT(num_filters == num_output_fmaps);
262  // int32 filter_size = filt_x_len_*filt_y_len_;
263  int32 num_frames = in.NumRows();
264 
265  // we will need the buffers
266  if (vectorized_feature_patches_.size() == 0) {
267  vectorized_feature_patches_.resize(out_fmap_size);
268  feature_patch_diffs_.resize(out_fmap_size);
269  }
270 
271  for (int32 p = 0; p < out_fmap_size; p++) {
272  vectorized_feature_patches_[p].Resize(num_frames, filters_.NumCols());
273  }
274 
275  // Checked for num_input_fmaps=1, check for num_inp_fmaps>1
276  int32 out_fmap_cnt = 0;
277  for (int32 m = 0; m < fmap_x_len_-filt_x_len_+1; m = m+filt_x_step_) {
278  for (int32 n = 0; n < fmap_y_len_-filt_y_len_+1; n = n+filt_y_step_) {
279  std::vector<int32> column_mask;
280  int32 st = 0;
281  if (connect_fmap_ == 1) {
282  st = (m * fmap_y_len_ + n) * num_input_fmaps;
283  } else {
284  st = m * fmap_y_len_ * num_input_fmaps + n;
285  }
286 
287  for (int32 i = 0; i < filt_x_len_; i++) {
288  for (int32 j = 0; j < filt_y_len_*num_input_fmaps; j++) {
289  int32 c = 0;
290  if (connect_fmap_ == 1) {
291  c = st + i * (num_input_fmaps*fmap_y_len_) + j;
292  } else {
293  c = st + i * (num_input_fmaps * fmap_y_len_)
294  + (j / num_input_fmaps)
295  + (j % num_input_fmaps) * fmap_y_len_;
296  }
297  column_mask.push_back(c);
298  }
299  }
300  CuArray<int32> cu_column_mask(column_mask);
301  vectorized_feature_patches_[out_fmap_cnt].CopyCols(in, cu_column_mask);
302  out_fmap_cnt++;
303  }
304  }
305 
306  for (int32 p = 0; p < out_fmap_size; p++) {
307  CuSubMatrix<BaseFloat> tgt(out->ColRange(p*num_filters, num_filters));
308  tgt.AddVecToRows(1.0, bias_, 0.0);
309  tgt.AddMatMat(1.0, vectorized_feature_patches_[p], kNoTrans, filters_, kTrans, 1.0);
310  }
311  }

314  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
315  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
316  // useful dims
317  int32 num_input_fmaps = input_dim_ / (fmap_x_len_ * fmap_y_len_);
318 
319  int32 out_fmap_x_len = (fmap_x_len_ - filt_x_len_)/filt_x_step_ + 1;
320  int32 out_fmap_y_len = (fmap_y_len_ - filt_y_len_)/filt_y_step_ + 1;
321  int32 out_fmap_size = out_fmap_x_len * out_fmap_y_len;
322  int32 num_output_fmaps = output_dim_ / (out_fmap_x_len * out_fmap_y_len);
323  // this is total num_filters,
324  // so each input_fmap has num_filters/num_input_fmaps
325  int32 num_filters = filters_.NumRows();
326  KALDI_ASSERT(num_filters == num_output_fmaps);
327  // int32 filter_size = filt_x_len_*filt_y_len_;
328  int32 num_frames = in.NumRows();
329 
330  for (int32 p = 0; p < out_fmap_size; p++) {
331  feature_patch_diffs_[p].Resize(num_frames, filters_.NumCols(), kSetZero);
332  CuSubMatrix<BaseFloat> out_diff_patch(out_diff.ColRange(p*num_filters, num_filters));
333  feature_patch_diffs_[p].AddMatMat(1.0, out_diff_patch, kNoTrans, filters_, kNoTrans, 0.0);
334  }
335 
336  // compute in_diff_summands_ once
337  if (in_diff_summands_.Dim() == 0) {
338  in_diff_summands_.Resize(in_diff->NumCols(), kSetZero);
339  for (int32 m = 0; m < fmap_x_len_-filt_x_len_+1; m = m+filt_x_step_) {
340  for (int32 n = 0; n < fmap_y_len_-filt_y_len_+1; n = n+filt_y_step_) {
341  int32 st = 0;
342  if (connect_fmap_ == 1) {
343  st = (m * fmap_y_len_ + n) * num_input_fmaps;
344  } else {
345  st = m * fmap_y_len_ * num_input_fmaps + n;
346  }
347  for (int32 i = 0; i < filt_x_len_; i++) {
348  for (int32 j = 0; j < filt_y_len_*num_input_fmaps; j++) {
349  int32 c = 0;
350  if (connect_fmap_ == 1) {
351  c = st + i * (num_input_fmaps * fmap_y_len_) + j;
352  } else {
353  c = st + i * (num_input_fmaps * fmap_y_len_)
354  + (j / num_input_fmaps)
355  + (j % num_input_fmaps) * fmap_y_len_;
356  }
357  // add 1.0
358  in_diff_summands_.Range(c, 1).Add(1.0);
359  }
360  }
361  }
362  }
363  in_diff_summands_.InvertElements();
364  }
365 
366  int32 out_fmap_cnt = 0;
367 
368  for (int32 m = 0; m < fmap_x_len_-filt_x_len_+1; m = m+filt_x_step_) {
369  for (int32 n = 0; n< fmap_y_len_-filt_y_len_+1; n = n+filt_y_step_) {
370  int32 st = 0;
371  if (connect_fmap_ == 1) {
372  st = (m * fmap_y_len_ + n) * num_input_fmaps;
373  } else {
374  st = m * fmap_y_len_ * num_input_fmaps + n;
375  }
376 
377  for (int32 i = 0; i < filt_x_len_; i++) {
378  for (int32 j = 0; j < filt_y_len_*num_input_fmaps; j++) {
379  int32 c = 0;
380  if (connect_fmap_ == 1) {
381  c = st + i *(num_input_fmaps*fmap_y_len_)+j;
382  } else {
383  c = st + i * (num_input_fmaps * fmap_y_len_)
384  + (j / num_input_fmaps)
385  + (j % num_input_fmaps) * fmap_y_len_;
386  }
387  CuSubMatrix<BaseFloat> src(feature_patch_diffs_[out_fmap_cnt].ColRange(i*filt_y_len_*num_input_fmaps+j, 1)); // from which col
388  CuSubMatrix<BaseFloat> tgt(in_diff->ColRange(c, 1)); // to which col?
389  tgt.AddMat(1.0, src);
390  }
391  }
392  out_fmap_cnt++;
393  }
394  }
395  // compensate for summands
396  in_diff->MulColsVec(in_diff_summands_);
397  }

400  void Update(const CuMatrixBase<BaseFloat> &input, const CuMatrixBase<BaseFloat> &diff) {
401  // useful dims
402  int32 out_fmap_x_len = (fmap_x_len_ - filt_x_len_)/filt_x_step_ + 1;
403  int32 out_fmap_y_len = (fmap_y_len_ - filt_y_len_)/filt_y_step_ + 1;
404  int32 out_fmap_size = out_fmap_x_len*out_fmap_y_len;
405  int32 num_output_fmaps = output_dim_ / (out_fmap_x_len * out_fmap_y_len);
406  int32 num_filters = filters_.NumRows(); // this is total num_filters, so each input_fmap has num_filters/num_input_fmaps
407  KALDI_ASSERT(num_filters == num_output_fmaps);
408 
409  // we use following hyperparameters from the option class
410  const BaseFloat lr = opts_.learn_rate;
411  /* NOT NOW:
412  const BaseFloat mmt = opts_.momentum;
413  const BaseFloat l2 = opts_.l2_penalty;
414  const BaseFloat l1 = opts_.l1_penalty;
415  */
416 
417 
418  //
419  // calculate the gradient
420  //
421  filters_grad_.Resize(filters_.NumRows(), filters_.NumCols(), kSetZero);
422  bias_grad_.Resize(filters_.NumRows());
423 
424  for (int32 p = 0; p < out_fmap_size; p++) {
425  CuSubMatrix<BaseFloat> diff_patch(diff.ColRange(p*num_filters, num_filters));
426  filters_grad_.AddMatMat(1.0, diff_patch, kTrans, vectorized_feature_patches_[p], kNoTrans, 1.0);
427  bias_grad_.AddRowSumMat(1.0, diff_patch, 1.0);
428  }
429 
430  // scale
431  filters_grad_.Scale(1.0/out_fmap_size);
432  bias_grad_.Scale(1.0/out_fmap_size);
433 
434  //
435  // update
436  //
437  filters_.AddMat(-lr*learn_rate_coef_, filters_grad_);
438  bias_.AddVec(-lr*bias_learn_rate_coef_, bias_grad_);
439  //
440  }
