30 int main(int argc, char *argv[]) {
31  using namespace kaldi;
32  try {
33  const char *usage =
34  "Accumulate stats for SGMM training.\n"
35  "Usage: sgmm-acc-stats [options] <model-in> <feature-rspecifier> "
36  "<posteriors-rspecifier> <stats-out>\n"
37  "e.g.: sgmm-acc-stats 1.mdl 1.ali scp:train.scp 'ark:ali-to-post 1.ali ark:-|' 1.acc\n";
38 
39  ParseOptions po(usage);
40  bool binary = true;
41  std::string gselect_rspecifier, spkvecs_rspecifier, utt2spk_rspecifier;
42  std::string update_flags_str = "vMNwcSt";
43  BaseFloat rand_prune = 1.0e-05;
44  SgmmGselectConfig sgmm_opts;
45  po.Register("binary", &binary, "Write output in binary mode");
46  po.Register("gselect", &gselect_rspecifier, "Precomputed Gaussian indices (rspecifier)");
47  po.Register("spk-vecs", &spkvecs_rspecifier, "Speaker vectors (rspecifier)");
48  po.Register("utt2spk", &utt2spk_rspecifier,
49  "rspecifier for utterance to speaker map");
50  po.Register("rand-prune", &rand_prune, "Pruning threshold for posteriors");
51  po.Register("update-flags", &update_flags_str, "Which SGMM parameters to accumulate "
52  "stats for: subset of vMNwcS.");
53  sgmm_opts.Register(&po);
54 
55  po.Read(argc, argv);
56 
57  kaldi::SgmmUpdateFlagsType acc_flags = StringToSgmmUpdateFlags(update_flags_str);
58 
59  if (po.NumArgs() != 4) {
60  po.PrintUsage();
61  exit(1);
62  }
63 
64  std::string model_filename = po.GetArg(1),
65  feature_rspecifier = po.GetArg(2),
66  posteriors_rspecifier = po.GetArg(3),
67  accs_wxfilename = po.GetArg(4);
68 
69  using namespace kaldi;
70  typedef kaldi::int32 int32;
71 
72  // Initialize the readers before the model, as the model can
73  // be large, and we don't want to call fork() after reading it if
74  // virtual memory may be low.
75  SequentialBaseFloatMatrixReader feature_reader(feature_rspecifier);
76  RandomAccessPosteriorReader posteriors_reader(posteriors_rspecifier);
77  RandomAccessInt32VectorVectorReader gselect_reader(gselect_rspecifier);
78  RandomAccessBaseFloatVectorReaderMapped spkvecs_reader(spkvecs_rspecifier,
79  utt2spk_rspecifier);
80 
81  AmSgmm am_sgmm;
82  TransitionModel trans_model;
83  {
84  bool binary;
85  Input ki(model_filename, &binary);
86  trans_model.Read(ki.Stream(), binary);
87  am_sgmm.Read(ki.Stream(), binary);
88  }
89 
90  Vector<double> transition_accs;
91  if (acc_flags & kaldi::kSgmmTransitions)
92  trans_model.InitStats(&transition_accs);
93  MleAmSgmmAccs sgmm_accs(rand_prune);
94  sgmm_accs.ResizeAccumulators(am_sgmm, acc_flags);
95 
96  double tot_like = 0.0;
97  double tot_t = 0;
98 
99  kaldi::SgmmPerFrameDerivedVars per_frame_vars;
100 
101  int32 num_done = 0, num_no_posterior = 0, num_other_error = 0;
102  for (; !feature_reader.Done(); feature_reader.Next()) {
103  std::string utt = feature_reader.Key();
104  if (!posteriors_reader.HasKey(utt)) {
105  num_no_posterior++;
106  } else {
107  const Matrix<BaseFloat> &mat = feature_reader.Value();
108  const Posterior &posterior = posteriors_reader.Value(utt);
109 
110  bool have_gselect = !gselect_rspecifier.empty()
111  && gselect_reader.HasKey(utt)
112  && gselect_reader.Value(utt).size() == mat.NumRows();
113  if (!gselect_rspecifier.empty() && !have_gselect)
114  KALDI_WARN << "No Gaussian-selection info available for utterance "
115  << utt << " (or wrong size)";
116  std::vector<std::vector<int32> > empty_gselect;
117  const std::vector<std::vector<int32> > *gselect =
118  (have_gselect ? &gselect_reader.Value(utt) : &empty_gselect);
119 
120  if (posterior.size() != mat.NumRows()) {
121  KALDI_WARN << "Alignments has wrong size "<< (posterior.size()) <<
122  " vs. "<< (mat.NumRows());
123  num_other_error++;
124  continue;
125  }
126 
127  SgmmPerSpkDerivedVars spk_vars;
128  if (spkvecs_reader.IsOpen()) {
129  if (spkvecs_reader.HasKey(utt)) {
130  spk_vars.v_s = spkvecs_reader.Value(utt);
131  am_sgmm.ComputePerSpkDerivedVars(&spk_vars);
132  } else {
133  KALDI_WARN << "Cannot find speaker vector for " << utt;
134  num_other_error++;
135  continue;
136  }
137  } // else spk_vars is "empty"
138 
139  num_done++;
140  BaseFloat tot_like_this_file = 0.0, tot_weight = 0.0;
141 
142  Posterior pdf_posterior;
143  ConvertPosteriorToPdfs(trans_model, posterior, &pdf_posterior);
144  for (size_t i = 0; i < posterior.size(); i++) {
145  if (posterior[i].empty())
146  continue;
147  std::vector<int32> this_gselect;
148  if (!gselect->empty()) this_gselect = (*gselect)[i];
149  else am_sgmm.GaussianSelection(sgmm_opts, mat.Row(i), &this_gselect);
150  am_sgmm.ComputePerFrameVars(mat.Row(i), this_gselect, spk_vars, 0.0,
151  &per_frame_vars);
152 
153  // Accumulates for SGMM.
154  for (size_t j = 0; j < pdf_posterior[i].size(); j++) {
155  int32 pdf_id = pdf_posterior[i][j].first;
156  BaseFloat weight = pdf_posterior[i][j].second;
157  tot_like_this_file += sgmm_accs.Accumulate(am_sgmm, per_frame_vars,
158  spk_vars.v_s, pdf_id,
159  weight, acc_flags)
160  * weight;
161  tot_weight += weight;
162  }
163 
164  // Accumulates for transitions.
165  for (size_t j = 0; j < posterior[i].size(); j++) {
166  if (acc_flags & kaldi::kSgmmTransitions) {
167  int32 tid = posterior[i][j].first;
168  BaseFloat weight = posterior[i][j].second;
169  trans_model.Accumulate(weight, tid, &transition_accs);
170  }
171  }
172  }
173 
174  sgmm_accs.CommitStatsForSpk(am_sgmm, spk_vars.v_s); // no harm doing it per utterance.
175 
176  KALDI_VLOG(2) << "Average like for this file is "
177  << (tot_like_this_file/tot_weight) << " over "
178  << tot_weight <<" frames.";
179  tot_like += tot_like_this_file;
180  tot_t += tot_weight;
181  if (num_done % 50 == 0) {
182  KALDI_LOG << "Processed " << num_done << " utterances; for utterance "
183  << utt << " avg. like is "
184  << (tot_like_this_file/tot_weight)
185  << " over " << tot_weight <<" frames.";
186  }
187  }
188  }
189  KALDI_LOG << "Overall like per frame (Gaussian only) = "
190  << (tot_like/tot_t) << " over " << tot_t << " frames.";
191 
192  KALDI_LOG << "Done " << num_done << " files, " << num_no_posterior
193  << " with no posteriors, " << num_other_error
194  << " with other errors.";
195 
196  {
197  Output ko(accs_wxfilename, binary);
198  // TODO(arnab): Ideally, we shouldn't be writing transition accs if not
199  // asked for, but that will complicate reading later. To be fixed?
200  transition_accs.Write(ko.Stream(), binary);
201  sgmm_accs.Write(ko.Stream(), binary);
202  }
203  KALDI_LOG << "Written accs.";
204  return (num_done != 0 ? 0 : 1);
205  } catch(const std::exception &e) {
206  std::cerr << e.what();
207  return -1;
208  }
209 }
