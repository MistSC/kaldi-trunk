29  bool input, bool *is_new) {
30  typedef unordered_map<Cindex, int32, CindexHasher> map_type;
31  int32 new_index = cindexes.size(); // we'll add this if we don't find it.
32  std::pair<map_type::iterator, bool> p = cindex_to_cindex_id_.insert(
33  std::pair<Cindex, int32>(cindex, new_index));

57 void ComputationGraph::Renumber(const std::vector<bool> &keep) {
58  int32 num_cindex_ids = cindexes.size();
59  KALDI_ASSERT(keep.size() == num_cindex_ids);
60  ComputationGraph temp_graph;
61  std::vector<int32> old2new(num_cindex_ids, -1), new2old;
62  new2old.reserve(num_cindex_ids);
63  for (int32 j = 0; j < num_cindex_ids; j++) {
64  if (keep[j]) {
65  old2new[j] = new2old.size();
66  new2old.push_back(j);
67  }
68  }
69  int32 new_num_cindex_ids = new2old.size();
70  if (new_num_cindex_ids == num_cindex_ids) {
71  // this is an optimization for when we are not deleting any
72  // cindex-ids.
73  return;
74  }
75  temp_graph.cindexes.resize(new_num_cindex_ids);
76  temp_graph.is_input.resize(new_num_cindex_ids);
77  temp_graph.dependencies.resize(new_num_cindex_ids);
78  for (int32 c = 0; c < new_num_cindex_ids; c++) {
79  int32 d = new2old[c];
80  temp_graph.cindexes[c] = cindexes[d];
81  temp_graph.is_input[c] = is_input[d];
82  temp_graph.dependencies[c].reserve(dependencies[d].size());
83  std::vector<int32>::const_iterator
84  iter = dependencies[d].begin(), end = dependencies[d].end();
85  for (; iter != end; ++iter) {
86  int32 old_dep = *iter, new_dep = old2new[old_dep];
87  if (new_dep != -1)
88  temp_graph.dependencies[c].push_back(new_dep);
89  else
90  KALDI_ERR << "Dependency on nonexistent cindex-id";
91  }
92  }
93 
94  // at this point, rather than setting up cindex_to_cindex_id_ on the temporary
95  // graph, we copy cindexes, is_input and dependencies to this graph, and then
96  // set up cindex_to_cindex_id_ locally.
97  cindexes.swap(temp_graph.cindexes);
98  is_input.swap(temp_graph.is_input);
99  dependencies.swap(temp_graph.dependencies);
100  cindex_to_cindex_id_.clear();
101  for (int32 c = 0; c < new_num_cindex_ids; c++)
102  cindex_to_cindex_id_[cindexes[c]] = c;
103 }

338 // this function limits the dependencies of cindex_id "cindex_id" to just those
339 // which are actually used in computing it. It also clears the dependencies
340 // of those cindexes that are not computable.
341 void ComputationGraphBuilder::PruneDependencies(int32 cindex_id) {
342  ComputableInfo c = static_cast<ComputableInfo>(computable_info_[cindex_id]);
343  // by the time this is called, there should be no cindexes with unknown state.
344  KALDI_ASSERT(c != kUnknown);
345  if (c == kNotComputable || c == kWillNotCompute) {
346  // if something is not computable, there is no point
347  // keeping around its dependencies.
348  graph_->dependencies[cindex_id].clear();
349  return;
350  }
351  KALDI_ASSERT(c == kComputable);
352  const Cindex &cindex = graph_->cindexes[cindex_id];
353  int32 node_id = cindex.first;
354  const Index &index = cindex.second;
355  const NetworkNode &node = nnet_.GetNode(node_id);
356 
357  std::vector<int32> &dependencies = graph_->dependencies[cindex_id];
358  std::sort(dependencies.begin(), dependencies.end());
359  std::vector<int32> used_cindex_ids;
360 
361  switch (node.node_type) {
362  case kDescriptor: {
363  const Descriptor &desc = node.descriptor;
364  bool dont_care = false; // there should be no kUnknown, and we check this
365  CindexSet cindex_set(*graph_, computable_info_, dont_care);
366  std::vector<Cindex> used_cindexes;
367  bool ans = desc.IsComputable(index, cindex_set, &used_cindexes);
368  // If the next assert fails it could be a failure in the assumption that
369  // making more inputs available will never change something from not being
370  // computable to being computable; or it could be a bug elsewhere.
371  KALDI_ASSERT(ans);
372  std::vector<int32> &dependencies = graph_->dependencies[cindex_id];
373  std::sort(dependencies.begin(), dependencies.end());
374  size_t size = used_cindexes.size();
375  used_cindex_ids.resize(size);
376  for (size_t i = 0; i < size; i++) {
377  int32 dep_cindex_id = graph_->GetCindexId(used_cindexes[i]);
378  KALDI_ASSERT(dep_cindex_id != -1);
379  used_cindex_ids[i] = dep_cindex_id;
380  KALDI_ASSERT(std::binary_search(dependencies.begin(),
381  dependencies.end(),
382  dep_cindex_id));
383  }
384  break;
385  }
386  case kComponent: {
387  const Component *c = nnet_.GetComponent(node.u.component_index);
388  bool dont_care = false; // there should be no kUnknown, and we check this
389  IndexSet index_set(*graph_, computable_info_, node_id, dont_care);
390  std::vector<Index> used_indexes;
391  bool ans = c->IsComputable(request_.misc_info, index, index_set,
392  &used_indexes);
393  // If the next assert fails it could be a failure in the assumption that
394  // making more inputs available will never change something from not being
395  // computable to being computable; or it could be a bug elsewhere.
396  KALDI_ASSERT(ans);
397  size_t size = used_indexes.size();
398  used_cindex_ids.resize(size);
399  for (size_t i = 0; i < size; i++) {
400  Cindex dep_cindex(node_id - 1, used_indexes[i]);
401  int32 dep_cindex_id = graph_->GetCindexId(dep_cindex);
402  KALDI_ASSERT(dep_cindex_id != -1);
403  used_cindex_ids[i] = dep_cindex_id;
404  KALDI_ASSERT(std::binary_search(dependencies.begin(),
405  dependencies.end(),
406  dep_cindex_id));
407  }
408  break;
409  }
410  case kDimRange:
411  KALDI_ASSERT(dependencies.size() == 1);
412  // there should be exactly one dependency and it is required, not
413  // optional, so there is nothing to do here. Return.
414  return;
415  case kInput:
416  KALDI_ASSERT(dependencies.empty());
417  // there is nothing to do; return.
418  return;
419  default:
420  KALDI_ERR << "Invalid node type";
421  }
422  SortAndUniq(&used_cindex_ids);
423 
424  // the next statement modifies the graph.
425  dependencies.swap(used_cindex_ids);
426 }

428 void ComputationGraphBuilder::Compute() {
429  KALDI_ASSERT(current_distance_ == -1 && "Compute() called twice?");
430  AddInputs();
431  AddOutputs(); // sets current_distance_ to 0.
432  // max_distance for debugging, to detect infinite recursion.
433  int32 max_distance = 10000;
434  while (current_distance_ < max_distance) {
435  BuildGraphOneIter();
436  // only check rarely if we're running at low verbose level.
437  if (GetVerboseLevel() >= 3 || RandInt(1, (current_distance_ + 1)) == 1)
438  Check();
439  // TODO: come up with a scheme to delay when we call
440  // UpdateAllComputableInfo().
441  UpdateAllComputableInfo();

443  break;
444  }

446  KALDI_ERR << "Loop detected while building computation graph (bad "
447  << "network topology?)";
448  Check();

452 void ComputationGraphBuilder::Check() const {
453  int32 num_cindex_ids = graph_->cindexes.size();
454  for (int32 cindex_id = 0; cindex_id < num_cindex_ids;
455  cindex_id += 1 + RandInt(0, num_cindex_ids / 100)) {
456  { // check depend_on_this.
457  std::vector<int32> depend_on_this = depend_on_this_[cindex_id];
458  int32 size = depend_on_this.size();
459  std::sort(depend_on_this.begin(), depend_on_this.end());
460  KALDI_ASSERT(IsSortedAndUniq(depend_on_this));
461  for (size_t j = 0; j < size; j++) {
462  int32 other_cindex_id = depend_on_this[j];
463  // make sure appears in appropriate dependencies array.
464  const std::vector<int32> &dep = graph_->dependencies[other_cindex_id];
465  KALDI_ASSERT(std::count(dep.begin(), dep.end(), cindex_id) == 1);
466  }
467  }
468  { // check dependencies.
469  std::vector<int32> dependencies = graph_->dependencies[cindex_id];
470  int32 size = dependencies.size();
471  std::sort(dependencies.begin(), dependencies.end());
472  KALDI_ASSERT(IsSortedAndUniq(dependencies));
473  for (size_t j = 0; j < size; j++) {
474  int32 dep_cindex_id = dependencies[j];
475  // make sure appears in appropriate depend_on_this_ array.
476  const std::vector<int32> &dep = depend_on_this_[dep_cindex_id];
477  KALDI_ASSERT(std::count(dep.begin(), dep.end(), cindex_id) == 1);
478  }
479  }
480  { // check usable_count_.
481  int32 node_index = graph_->cindexes[cindex_id].first;
482  int32 usable_count = usable_count_[cindex_id],
483  usable_count_recomputed = nnet_.IsOutputNode(node_index) ? 1 : 0;
484  std::vector<int32> depend_on_this = depend_on_this_[cindex_id];
485  int32 size = depend_on_this.size();
486  for (size_t j = 0; j < size; j++) {
487  int32 other_cindex_id = depend_on_this[j];
488  if (usable_count_[other_cindex_id] != 0 &&
489  computable_info_[other_cindex_id] != kNotComputable)
490  usable_count_recomputed++;
491  }
492  KALDI_ASSERT(usable_count == usable_count_recomputed);
493  }
494  // check computable_info_. note: this will not be accurate
495  // if the cindex_id is still queued to have dependencies added
496  // (in cur_queue_ or next_queue_).
497  if (computable_queue_.empty()) {
498  ComputationGraphBuilder::ComputableInfo c =
499  ComputeComputableInfo(cindex_id);
500  // the status doesn't have to be correct if it's kWillNotCompute,
501  // because these are cindex-ids that we chose not to compute
502  // because we determined they would not be useful, and
503  // ComputeComputableInfo() will never return this value.
504  if (c != computable_info_[cindex_id] &&
505  computable_info_[cindex_id] != kWillNotCompute) {
506  int32 count_cur = std::count(current_queue_.begin(),
507  current_queue_.end(), cindex_id),
508  count_next = std::count(next_queue_.begin(),
509  next_queue_.end(), cindex_id);
510  // if it wasn't queued, then something is wrong.
511  if (count_cur + count_next == 0)
512  KALDI_ERR << "Mismatch in computable status";
513  }
514  }
515  // check computable_queued_.
516  // note, the following checks might be a bit slow.
517  if (computable_queued_[cindex_id]) {
518  KALDI_ASSERT(std::count(computable_queue_.begin(),
519  computable_queue_.end(),
520  cindex_id) == 1);
521  } else {
522  KALDI_ASSERT(std::count(computable_queue_.begin(),
523  computable_queue_.end(),
524  cindex_id) == 0);
525  }
526  }

529 void ComputationGraphBuilder::Prune() {
530  int32 num_cindex_ids = graph_->cindexes.size();
531  // Prune the dependencies to just those that are used (to remove
532  // optional dependencies that don't end up getting used).
533 
534  for (int32 cindex_id = 0; cindex_id < num_cindex_ids; cindex_id++)
535  PruneDependencies(cindex_id);
536  depend_on_this_.clear(); // not valid any more after pruning dependencies.
537  std::vector<bool> required;
538  ComputeRequiredArray(&required);
539 
540  std::vector<bool> keep(num_cindex_ids, false);
541  for (int32 c = 0; c < num_cindex_ids; c++) {
542  if (required[c] || graph_->is_input[c]) {
543  KALDI_ASSERT(computable_info_[c] == kComputable &&
544  "You are calling Prune when not everything is computable.");
545  keep[c] = true;
546  }
547  }
548  graph_->Renumber(keep);
549  // The following variables will not be valid any more after the renumbering,
550  // so clear them.
551  computable_info_.clear();
552  computable_queue_.clear();
553  usable_count_.clear();

556 // Add cindex_ids that this cindex_id depends on.
557 void ComputationGraphBuilder::AddDependencies(int32 cindex_id) {
558  if (static_cast<int32>(graph_->dependencies.size()) <= cindex_id) {
559  graph_->dependencies.resize(2 * cindex_id + 1);
560  }
561 
562  Cindex cindex = graph_->cindexes[cindex_id];
563 
564  // find the dependencies of this cindex.
565  int32 node_index = cindex.first;
566  const Index &index = cindex.second;
567  const NetworkNode &node = nnet_.GetNode(node_index);
568 
569  std::vector<Cindex> input_cindexes;
570 
571  // the following switch statement sets up "input_cindexes".
572  switch (node.node_type) {
573  case kDescriptor: {
574  // desc describes how this node obtains its input from other nodes.
575  const Descriptor &desc = node.descriptor;
576  desc.GetDependencies(index, &input_cindexes);
577  break;
578  }
579  case kComponent: {
580  int32 c = node.u.component_index;
581  const Component *component = nnet_.GetComponent(c);
582  std::vector<Index> input_indexes;
583  component->GetInputIndexes(request_.misc_info, index,
584  &input_indexes);
585  input_cindexes.resize(input_indexes.size());
586  for (size_t i = 0; i < input_indexes.size(); i++) {
587  input_cindexes[i].first = node_index - 1; // preceding node
588  input_cindexes[i].second = input_indexes[i];
589  }
590  break;
591  }
592  case kDimRange: {
593  input_cindexes.resize(1);
594  input_cindexes[0] = Cindex(node.u.node_index, index);
595  break;
596  }
597  case kInput:
598  break; // There will be no dependencies.
599  default:
600  KALDI_ERR << "Invalid node type";
601  }
602 
603  int32 num_dependencies = input_cindexes.size();
604  // this "reserve" statement is to make sure the reference
605  // we declare below does not become invalid in the loop below
606  // (the call to graph_->GetCindexId() could add up to
607  // num_dependencies elements to the graph_->dependencies array
608  // and we want to avoid allocation).
609  // the RoundUpToNearestPowerOfTwo is for efficiency, to
610  // avoid too-frequent resizes.
611  graph_->dependencies.reserve(RoundUpToNearestPowerOfTwo(
612  graph_->dependencies.size() + num_dependencies));
613  std::vector<int32> &this_dep = graph_->dependencies[cindex_id];
614 
615  this_dep.resize(num_dependencies);
616  for (size_t i = 0; i < num_dependencies; i++) {
617  bool is_input = false, is_new;
618  int32 dep_cindex_id = graph_->GetCindexId(input_cindexes[i],
619  is_input, &is_new);
620  this_dep[i] = dep_cindex_id;
621  if (is_new)
622  AddCindexId(dep_cindex_id, false, false);
623  // we will keep dependent's usable_count_ up to date below
624  }
625  // remove duplicates of dependencies.
626  SortAndUniq(&this_dep);
627  // set up the "depend_on_this_" array.
628  std::vector<int32>::const_iterator iter = this_dep.begin(),
629  end = this_dep.end();
630 
631  // Populate the "depend_on_this_" array, and append the
632  // usable_count_ of things we depend on (see the definition
633  // of this quantity next to where it is declared).
634  // Note: before calling AddDependencies() we verified the following:
635  // computable_info_[cindex_id] == kUnknown
636  // and
637  // usable_count_[cindex_id] != 0
638  // which ensures that the conditions to increment the dependency's
639  // usable_count_ are satisfied.
640  for (; iter != end; ++iter) {
641  int32 dep_cindex_id = *iter;
642  depend_on_this_[dep_cindex_id].push_back(cindex_id);
643  IncrementUsableCount(dep_cindex_id);
644  }
645 
646  // Now that we've added the dependencies, we can put this into
647  // the computable_queue_ to assess whether it's computable
648  KALDI_ASSERT(computable_info_[cindex_id] == kUnknown &&
649  !computable_queued_[cindex_id]);
650  // we think it'll be faster in the next line to do push_front instead of
651  // push_back; either one would be correct.
652  computable_queue_.push_front(cindex_id);
653  computable_queued_[cindex_id] = true;
654 }

657 ComputationGraphBuilder::ComputableInfo
658 ComputationGraphBuilder::ComputeComputableInfo(int32 cindex_id)
659  const {
660  const Cindex &cindex = graph_->cindexes[cindex_id];
661  int32 node_id = cindex.first;
662  const Index &index = cindex.second;
663  const NetworkNode &node = nnet_.GetNode(node_id);
664  switch (node.node_type) {
665  case kDescriptor: {
666  const Descriptor &desc = node.descriptor;
667  {
668  CindexSet cindex_set(*graph_, computable_info_, false);
669  if (desc.IsComputable(index, cindex_set, NULL)) {
670  // it's computable even without counting kUnknown inputs as computable
671  // [treat_unknown_as_computable = false] -> definitely computable.
672  return kComputable;
673  }
674  }
675  CindexSet cindex_set2(*graph_, computable_info_, true);
676  if (!desc.IsComputable(index, cindex_set2, NULL)) {
677  // it's not computable even when counting kUnknown inputs as
678  // computable [treat_unknown_as_computable = true] -> definitely not
679  // computable.
680  return kNotComputable;
681  }
682  return kUnknown;
683  }
684  case kComponent: {
685  const Component *c = nnet_.GetComponent(node.u.component_index);
686  const int32 input_node_id = node_id - 1;
687  {
688  IndexSet index_set(*graph_, computable_info_, input_node_id, false);
689  if (c->IsComputable(request_.misc_info, index, index_set, NULL)) {
690  // it's computable even without counting kUnknown inputs as computable
691  // [treat_unknown_as_computable = false] -> definitely computable.
692  return kComputable;
693  }
694  }
695  IndexSet index_set2(*graph_, computable_info_, input_node_id, true);
696  if (!c->IsComputable(request_.misc_info, index, index_set2, NULL)) {
697  // it's not computable even when counting kUnknown inputs as computable
698  // [treat_unknown_as_computable = true] -> definitely not computable.
699  return kNotComputable;
700  }
701  return kUnknown;
702  }
703  case kDimRange: {
704  Cindex input_cindex(node.u.node_index, index);
705  int32 input_cindex_id = graph_->GetCindexId(input_cindex);
706  if (input_cindex_id != -1)
707  return ComputableInfo(computable_info_[input_cindex_id]);
708  else
709  return kUnknown;
710  }
711  case kInput: {
712  // cindexes for input nodes that are part of the computation request will
713  // have graph_->is_input[cindex_id] == true; others will have
714  // graph_->is_input[cindex_id] == true.
715  return graph_->is_input[cindex_id] ? kComputable : kNotComputable;
716  }
717  default:
718  KALDI_ERR << "Invalid node type.";
719  return kUnknown; // suppress compiler warning.
720  }

748 void ComputationGraphBuilder::UpdateComputableInfo(int32 cindex_id) {
749  // if the current computable_info_ for cindex_id value is not kUnknown, this
750  // cindex_id should not have been in the queue.
751  KALDI_ASSERT(static_cast<size_t>(cindex_id) < computable_info_.size());

753  KALDI_ASSERT(output == kUnknown);
754 
755  output = static_cast<char>(ComputeComputableInfo(cindex_id));
756 
757  if (output != kUnknown) {

759  // status is currently kUnknown might now change, so if they are not in the
760  // computable queue, put them there.
761  std::vector<int32>::const_iterator iter = depend_on_this_[cindex_id].begin(),
762  end = depend_on_this_[cindex_id].end();
763  for (; iter != end; ++iter) {

766  !computable_queued_[other_cindex_id]) {
767  computable_queue_.push_back(other_cindex_id);
768  computable_queued_[other_cindex_id] = true;

770  }

774  // the declaration), this means that we must decrement the
775  // usable_count_ of all cindex_ids that we depend on.

778  end = graph_->dependencies[cindex_id].end();
779  for (; iter != end; ++iter) {
780  int32 dep_cindex_id = *iter;
781  DecrementUsableCount(dep_cindex_id);

783  }

1002  const Nnet &nnet,

1004  std::vector<int32> *cindex_id_to_epoch,
1005  std::vector<std::vector<int32 > > *epochs,
1006  std::vector<bool> *epoch_is_trivial) {
1007

1009  // us what order to compute them in... but we may need to compute
1010  // a finer ordering at the cindex_id level in cases like RNNs.

1012  ComputeNnetComputationEpochs(nnet, &node_to_epoch);

1014  std::ostringstream os;

1024  int32 num_nodes = nnet.NumNodes(),
1025  num_cindex_ids = graph.cindexes.size(),
1026  num_epoch_indexes = 1 + *std::max_element(node_to_epoch.begin(),
1027  node_to_epoch.end());

1033  std::vector<int32> epoch_to_num_nodes(num_epoch_indexes, 0);
1034  for (int32 n = 0; n < num_nodes; n++)
1035  epoch_to_num_nodes[node_to_epoch[n]]++;
1036 
1037  epoch_is_trivial->resize(num_epoch_indexes);
1038  for (int32 o = 0; o < num_epoch_indexes; o++) {
1039  KALDI_ASSERT(o == 0 || epoch_to_num_nodes[o] > 0);
1040  (*epoch_is_trivial)[o] = (epoch_to_num_nodes[o] <= 1);
1041  }
1042 
1043  cindex_id_to_epoch->resize(num_cindex_ids);
1044  epochs->resize(num_epoch_indexes);
1045  for (int32 cindex_id = 0; cindex_id < num_cindex_ids; cindex_id++) {
1046  int32 node_index = graph.cindexes[cindex_id].first,
1047  epoch_index = (graph.is_input[cindex_id] ? 0 :
1048  node_to_epoch[node_index]);
1049  (*cindex_id_to_epoch)[cindex_id] = epoch_index;
1050  (*epochs)[epoch_index].push_back(cindex_id);
1051  }
1052 }

1058 void ComputeComputationGraph(const Nnet &nnet,
1059  const ComputationRequest &request,
1060  ComputationGraph *graph) {
1061  using namespace computation_graph;
1062  // make sure graph is empty at the start.
1063  KALDI_ASSERT(graph->cindexes.empty());
1064 
1065  AddInputToGraph(request, nnet, graph);
1066  AddOutputToGraph(request, nnet, graph);
1067 
1068  // queue of cindex_ids to process.
1069  std::vector<int32> queue(graph->cindexes.size());
1070  for (int32 i = 0; i < graph->cindexes.size(); i++)
1071  queue.push_back(i);
1072 
1073  while (!queue.empty()) {
1074  int32 cindex_id = queue.back();
1075  queue.pop_back();
1076  if (static_cast<int32>(graph->dependencies.size()) <= cindex_id)
1077  graph->dependencies.resize(cindex_id + 1);
1078 
1079  if (graph->is_input[cindex_id])
1080  continue;
1081  Cindex cindex = graph->cindexes[cindex_id];
1082 
1083  // find the dependencies of this cindex.
1084  int32 n = cindex.first;
1085  const Index &index = cindex.second;
1086  const NetworkNode &node = nnet.GetNode(n);
1087 
1088  std::vector<Cindex> input_cindexes;
1089 
1090  // the following switch statement sets up "input_cindexes".
1091  switch (node.node_type) {
1092  case kDescriptor: {
1093  // desc describes how this node obtains its input from other nodes.
1094  const Descriptor &desc = node.descriptor;
1095  desc.GetDependencies(index, &input_cindexes);
1096  break;
1097  }
1098  case kComponent: {
1099  int32 c = node.u.component_index;
1100  const Component *component = nnet.GetComponent(c);
1101  std::vector<Index> input_indexes;
1102  component->GetInputIndexes(request.misc_info, index,
1103  &input_indexes);
1104  // each Component node should be preceded by a node that describes its
1105  // input, of type kDescriptor
1106  KALDI_ASSERT(nnet.GetNode(n-1).node_type ==
1107  kDescriptor);
1108 
1109  input_cindexes.resize(input_indexes.size());
1110  for (size_t i = 0; i < input_indexes.size(); i++) {
1111  input_cindexes[i].first = n - 1; // preceding node.
1112  input_cindexes[i].second = input_indexes[i];
1113  }
1114  break;
1115  }
1116  case kDimRange: {
1117  input_cindexes.resize(1);
1118  input_cindexes[0] = Cindex(node.u.node_index, index);
1119  break;
1120  }

1153 // this function is called from ComputeComputationPhases; it handles the part of
1154 // the computation from one epoch (this code was broken out to avoid that
1155 // function being super-long)
1156 // dependencies_subset is the same as graph.dependencies, but only including
1157 // that subset of the dependencies that belong to this same epoch
1158 // (i.e. dependencies within the same strongly connected component of the
1159 // graph on nodes). depend_on_subset is the tranpose of the graph
1160 // represented by dependencies_subset.
1161 // phase_index maps cindex_id to phase_index (or -1 if not assigned yet).
1162 static inline void ComputeComputationPhasesForEpoch(
1163  const Nnet &nnet,
1164  const ComputationGraph &graph,
1165  const std::vector<std::vector<int32> > &dependencies_subset,
1166  const std::vector<std::vector<int32> > &depend_on_subset,
1167  const std::vector<int32> &this_epoch,
1168  bool epoch_is_trivial,
1169  std::vector<int32> *phase_index,
1170  std::vector<std::vector<int32> > *phases) {
1171  std::vector<int32> this_phase, next_phase_candidates;
1172 
1173  if (this_epoch.empty())
1174  return;
1175 
1176  if (epoch_is_trivial) { // an optimization
1177  this_phase = this_epoch;
1178  } else {
1179  // Start out with all elements of this epoch that have no
1180  // dependencies within the same epoch
1181  std::vector<int32>::const_iterator iter = this_epoch.begin(),
1182  end = this_epoch.end();
1183  for (; iter != end; ++iter) {
1184  int32 cindex_id = *iter;
1185  if (dependencies_subset[cindex_id].empty())
1186  this_phase.push_back(cindex_id);
1187  }
1188  }
1189 
1190  // if the next assert fails, the graph at the level of cindex_ids is not acyclic.
1191  KALDI_ASSERT(!this_phase.empty() &&
1192  "Trying to process computation with cycles");
1193 
1194  while (!this_phase.empty()) {
1195  phases->push_back(this_phase);
1196  // The next if-statement is an optimization: if for this epoch index
1197  // there is just one node, there will be just one order-index for this
1198  // epoch, so we can skip the rest of this loop. Note: if
1199  // epoch == 0, even if there is just one node, cindex_ids from
1200  // multiple nodes may be put here because of the rule that cindex_ids which
1201  // are inputs always get epoch 0. But it's still true that they
1202  // will have no dependencies, so we can still skip the code below.
1203  if (epoch_is_trivial)
1204  return;
1205 
1206  int32 cur_phase_index = phases->size() - 1;
1207 
1208  // next_phases_candidates is a list of cindexes that we should check
1209  // whether they are computable now, because one of the things they depend
1210  // on just became computable.
1211  next_phase_candidates.clear();
1212  int32 size = this_phase.size();
1213  for (int32 i = 0; i < size; i++) {
1214  int32 c = this_phase[i]; // c is a cindex_id with phase cur_phase_index.
1215  (*phase_index)[c] = cur_phase_index;
1216  std::vector<int32>::const_iterator iter = depend_on_subset[c].begin(),
1217  end = depend_on_subset[c].end();
1218  for (; iter != end; ++iter) {
1219  int32 d = *iter; // cindex_id that depends on c.
1220  next_phase_candidates.push_back(d);
1221  }
1222  }
1223  SortAndUniq(&next_phase_candidates);
1224  this_phase.clear();
1225  // now check the candidates that might be in the next phase, and put any
1226  // members that we are currently able to compute into "this_phase".
1227  std::vector<int32>::const_iterator iter = next_phase_candidates.begin(),
1228  end = next_phase_candidates.end();
1229  for (; iter != end; ++iter) {
1230  int32 c = *iter;
1231  std::vector<int32>::const_iterator
1232  dep_iter = dependencies_subset[c].begin(),
1233  dep_end = dependencies_subset[c].end();
1234  for (; dep_iter != dep_end; ++dep_iter) {
1235  int32 d = *dep_iter; // d is cindex_id that c depends on.
1236  if ((*phase_index)[d] < 0) // we can't compute c yet as something we depend
1237  break; // on is not yet computed.
1238  }
1239  if (dep_iter == dep_end) {
1240  // we reached the end and did not break -> all dependencies satisfied
1241  this_phase.push_back(c);
1242  }
1243  }
1244  if (!next_phase_candidates.empty() && this_phase.empty()) {
1245  // this should have been caught earlier so likely a code error rather than
1246  // a problem with user input.
1247  KALDI_ERR << "Possibly some cindexes were not reachable (code error?)";
1248  }
1249  }
1250 }

1707  const ComputationRequest &request,
1708  const ComputationGraph &graph,
1709  std::vector<std::vector<int32> > *steps) {
1710

1712  std::vector<int32> &cindex_ids = (*steps)[step];
1713  if (cindex_ids.empty()) continue;
1714  int32 cindex_id = cindex_ids.front();

1716  const NetworkNode &node = nnet.GetNode(node_index);
1717  if (node.node_type != kComponent ||
1718  graph.is_input[cindex_id])
1719  continue; // nothing to do if an input, or if not a Component.
1720

1722  const Component *component = nnet.GetComponent(c);
1723  if (!(component->Properties() & kReordersIndexes))
1724  continue; // nothing to do if it doesn't modify indexes.
1725  KALDI_ASSERT(step > 0); // or should have continued already.
1726 
1727  // preceding step will be Cindexes from the input Descriptor.
1728  std::vector<int32> &input_cindex_ids = (*steps)[step - 1];
1729 
1730  int32 size = cindex_ids.size(), input_size = input_cindex_ids.size();
1731  std::vector<Index> indexes(size), input_indexes(input_size);
1732 
1733  for (int32 i = 0; i < size; i++)
1734  indexes[i] = graph.cindexes[cindex_ids[i]].second;
1735  for (int32 i = 0; i < input_size; i++)
1736  input_indexes[i] = graph.cindexes[input_cindex_ids[i]].second;
1737 
1738  component->ReorderIndexes(&input_indexes, &indexes);
1739  // size should not change.
1740  KALDI_ASSERT(input_indexes.size() == input_size && indexes.size() == size);
1741 
1742  if (size > 0) {
1743  int32 node_index = graph.cindexes[cindex_ids.front()].first;
1744  for (int32 i = 0; i < size; i++) {
1745  Cindex cindex(node_index, indexes[i]);
1746  cindex_ids[i] = graph.GetCindexId(cindex);
1747  }
1748  }
1749  if (input_size > 0) {

1751  for (int32 i = 0; i < input_size; i++) {
1752  Cindex cindex(input_node_index, input_indexes[i]);
1753  input_cindex_ids[i] = graph.GetCindexId(cindex);
1754  }
1755  }
1756  // note: cindex_ids and input_cindex_ids are references, so we have
1757  // changed *steps by writing to them in the above two loops.
1758  }
