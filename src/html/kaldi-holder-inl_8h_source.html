38 // KaldiObjectHolder is valid only for Kaldi objects with
39 // copy constructors, default constructors, and "normal"
40 // Kaldi Write and Read functions. E.g. it works for

50  try {

54  KALDI_WARN << "Exception caught writing Table object: " << e.what();

72  // object.

139  " header\n";
140  return false;

143  int c;

159 
160  if (!is_binary) { // This is to catch errors, the class would work
161  // without it..

164  is.get();
165  }

168  << CharToString(is.peek()) << ", position " << is.tellg();
169  return false;
170  }
171  is.get(); // Consume the newline.
172  }
173  return true;
174  } catch(const std::exception &e) {
175  KALDI_WARN << "Exception caught reading Table object";
176  if (!IsKaldiError(e.what())) { std::cerr << e.what(); }
177  return false;
178  }
179  }

186  return t_;

208  static bool Write(std::ostream &os, bool binary, const T &t) {
209  InitKaldiOutputStream(os, binary); // Puts binary header if binary mode.
210  try {
211  if (binary) { // need to write the size, in binary mode.
212  KALDI_ASSERT(static_cast<size_t>(static_cast<int32>(t.size())) ==
213  t.size());
214  // Or this Write routine cannot handle such a large vector.
215  // use int32 because it's fixed size regardless of compilation.
216  // change to int64 (plus in Read function) if this becomes a problem.
217  WriteBasicType(os, binary, static_cast<int32>(t.size()));
218  for (typename std::vector<BasicType>::const_iterator iter = t.begin();
219  iter != t.end(); ++iter)
220  WriteBasicType(os, binary, *iter);
221 
222  } else {
223  for (typename std::vector<BasicType>::const_iterator iter = t.begin();
224  iter != t.end(); ++iter)
225  WriteBasicType(os, binary, *iter);
226  os << '\n'; // Makes output format more readable and
227  // easier to manipulate. In text mode, this function writes something
228  // like "1 2 3\n".
229  }
230  return os.good();
231  } catch(const std::exception &e) {
232  KALDI_WARN << "Exception caught writing Table object (BasicVector). ";
233  if (!IsKaldiError(e.what())) { std::cerr << e.what(); }
234  return false; // Write failure.
235  }
236  }

240  // Reads into the holder.
241  bool Read(std::istream &is) {
242  t_.clear();
243  bool is_binary;
244  if (!InitKaldiInputStream(is, &is_binary)) {
245  KALDI_WARN << "Reading Table object [integer type], failed reading binary"
246  " header\n";
247  return false;
248  }
249  if (!is_binary) {
250  // In text mode, we terminate with newline.
251  std::string line;
252  getline(is, line); // this will discard the \n, if present.
253  if (is.fail()) {
254  KALDI_WARN << "BasicVectorHolder::Read, error reading line " <<
255  (is.eof() ? "[eof]" : "");
256  return false; // probably eof. fail in any case.
257  }
258  std::istringstream line_is(line);
259  try {
260  while (1) {
261  line_is >> std::ws; // eat up whitespace.
262  if (line_is.eof()) break;
263  BasicType bt;
264  ReadBasicType(line_is, false, &bt);
265  t_.push_back(bt);
266  }
267  return true;
268  } catch(const std::exception &e) {
269  KALDI_WARN << "BasicVectorHolder::Read, could not interpret line: "
270  << line;
271  if (!IsKaldiError(e.what())) { std::cerr << e.what(); }
272  return false;
273  }
274  } else { // binary mode.
275  size_t filepos = is.tellg();
276  try {
277  int32 size;
278  ReadBasicType(is, true, &size);
279  t_.resize(size);
280  for (typename std::vector<BasicType>::iterator iter = t_.begin();
281  iter != t_.end();
282  ++iter) {
283  ReadBasicType(is, true, &(*iter));
284  }
285  return true;
286  } catch(...) {
287  KALDI_WARN << "BasicVectorHolder::Read, read error or unexpected data"
288  " at archive entry beginning at file position " << filepos;
289  return false;
290  }
291  }
292  }
293 
294  // Objects read/written with the Kaldi I/O functions always have the stream
295  // open in binary mode for reading.
296  static bool IsReadInBinary() { return true; }
297 
298  const T &Value() const { return t_; }

301  private:

728 // SphinxMatrixHolder can be used to read and write feature files in
729 // CMU Sphinx format. 13-dimensional big-endian features are assumed.
730 // The ultimate reference is SphinxBase's source code (for example see
731 // feat_s2mfc_read() in src/libsphinxbase/feat/feat.c).
732 // We can't fully automate the detection of machine/feature file endianess
733 // mismatch here, because for this Sphinx relies on comparing the feature
734 // file's size with the number recorded in its header. We are working with
735 // streams, however(what happens if this is a Kaldi archive?). This should
736 // be no problem, because the usage help of Sphinx' "wave2feat" for example
737 // says that Sphinx features are always big endian.
738 // Note: the kFeatDim defaults to 13, see forward declaration in kaldi-holder.h
739 template<int kFeatDim> class SphinxMatrixHolder {
740  public:
741  typedef Matrix<BaseFloat> T;
742 
743  SphinxMatrixHolder() {}
744 
745  void Clear() { feats_.Resize(0, 0); }
746 
747  // Writes Sphinx-format features
748  static bool Write(std::ostream &os, bool binary, const T &m) {
749  if (!binary) {
750  KALDI_WARN << "SphinxMatrixHolder can't write Sphinx features in text ";
751  return false;
752  }
753 
754  int32 size = m.NumRows() * m.NumCols();
755  if (MachineIsLittleEndian())
756  KALDI_SWAP4(size);
757  os.write(reinterpret_cast<char*> (&size), sizeof(size)); // write the
758  // header
759 
760  for (MatrixIndexT i = 0; i < m.NumRows(); i++) {
761  std::vector<float32> tmp(m.NumCols());
762  for (MatrixIndexT j = 0; j < m.NumCols(); j++) {
763  tmp[j] = static_cast<float32>(m(i, j));
764  if (MachineIsLittleEndian())
765  KALDI_SWAP4(tmp[j]);
766  }
767  os.write(reinterpret_cast<char*>(&(tmp[0])),
768  tmp.size() * 4);
769  }
770  return true;
771  }
772 
773  // Reads the features into a Kaldi Matrix
774  bool Read(std::istream &is) {
775  int32 nmfcc;
776 
777  is.read(reinterpret_cast<char*> (&nmfcc), sizeof(nmfcc));
778  if (MachineIsLittleEndian())
779  KALDI_SWAP4(nmfcc);
780  KALDI_VLOG(2) << "#feats: " << nmfcc;
781  int32 nfvec = nmfcc / kFeatDim;
782  if ((nmfcc % kFeatDim) != 0) {
783  KALDI_WARN << "Sphinx feature count is inconsistent with vector length ";
784  return false;
785  }
786 
787  feats_.Resize(nfvec, kFeatDim);
788  for (MatrixIndexT i = 0; i < feats_.NumRows(); i++) {
789  if (sizeof(BaseFloat) == sizeof(float32)) {
790  is.read(reinterpret_cast<char*> (feats_.RowData(i)),
791  kFeatDim * sizeof(float32));
792  if (!is.good()) {
793  KALDI_WARN << "Unexpected error/EOF while reading Sphinx features ";
794  return false;
795  }
796  if (MachineIsLittleEndian()) {
797  for (MatrixIndexT j = 0; j < kFeatDim; j++)
798  KALDI_SWAP4(feats_(i, j));
799  }
800  } else { // KALDI_DOUBLEPRECISION=1
801  float32 tmp[kFeatDim];
802  is.read(reinterpret_cast<char*> (tmp), sizeof(tmp));
803  if (!is.good()) {
804  KALDI_WARN << "Unexpected error/EOF while reading Sphinx features ";
805  return false;
806  }
807  for (MatrixIndexT j = 0; j < kFeatDim; j++) {
808  if (MachineIsLittleEndian())
809  KALDI_SWAP4(tmp[j]);
810  feats_(i, j) = static_cast<BaseFloat>(tmp[j]);
811  }
812  }
813  }
814 
815  return true;
816  }
817 
818  // Only read in binary
819  static bool IsReadInBinary() { return true; }
820 
821  const T &Value() const { return feats_; }
822 
823  private:
824  KALDI_DISALLOW_COPY_AND_ASSIGN(SphinxMatrixHolder);
825  T feats_;
826 };
