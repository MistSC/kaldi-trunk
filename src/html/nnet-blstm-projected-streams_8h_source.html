446  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
447  int DEBUG = 0;
448  int32 nstream_ = sequence_lengths_.size();
449  KALDI_ASSERT(in.NumRows() % nstream_ == 0);
450  int32 T = in.NumRows() / nstream_;
451  int32 S = nstream_;
452 
453  // 0:forward pass history, [1, T]:current sequence, T+1:dummy
454  // forward direction
455  f_propagate_buf_.Resize((T+2)*S, 7 * ncell_ + nrecur_, kSetZero);
456  // backward direction
457  b_propagate_buf_.Resize((T+2)*S, 7 * ncell_ + nrecur_, kSetZero);
458
