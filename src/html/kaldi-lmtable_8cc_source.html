90 void LmFstConverter::AddArcsForNgramProb(
91  int ngram_order, int max_ngram_order,
92  float logProb,
93  float logBow,
94  std::vector<string> &ngram,
95  fst::StdVectorFst *fst,
96  const string startSent,
97  const string endSent) {
98  fst::StdArc::StateId src, dst, dbo;
99  std::string curwrd = ngram[1];
100  if (curwrd == "<eps>") {
101  KALDI_ERR << "The word <eps> is not allowed as a word in an ARPA LM.";
102  }
103  int64 ilab, olab;
104  LmWeight prob = ConvertArpaLogProbToWeight(logProb);
105  LmWeight bow = ConvertArpaLogProbToWeight(logBow);
106  bool newSrc, newDbo, newDst = false;
107 
108  if (ngram_order >= 2) {
109  // General case works from N down to 2-grams
110  src = AddStateFromSymb(ngram, ngram_order, 2, fst, newSrc);
111  if (ngram_order != max_ngram_order) {
112  // add all intermediate levels from 2 to current
113  // last ones will be current backoff source and destination
114  for (int i = 2; i <= ngram_order; i++) {
115  dst = AddStateFromSymb(ngram, i, 1, fst, newDst);
116  dbo = AddStateFromSymb(ngram, i-1, 1, fst, newDbo);
117  backoff_state_[dst] = dbo;
118  }
119  } else {
120  // add all intermediate levels from 2 to current
121  // last ones will be current backoff source and destination
122  for (int i = 2; i <= ngram_order; i++) {
123  dst = AddStateFromSymb(ngram, i-1, 1, fst, newDst);
124  dbo = AddStateFromSymb(ngram, i-2, 1, fst, newDbo);
125  backoff_state_[dst] = dbo;
126  }
127  }
128  } else {
129  // special case for 1-grams: start from 0-gram
130  if (curwrd.compare(startSent) != 0) {
131  src = AddStateFromSymb(ngram, 0, 1, fst, newSrc);
132  } else {
133  // extra special case if in addition we are at beginning of sentence
134  // starts from initial state and has no cost
135  src = fst->Start();
136  prob = fst::StdArc::Weight::One();
137  }
138  dst = AddStateFromSymb(ngram, 1, 1, fst, newDst);
139  dbo = AddStateFromSymb(ngram, 0, 1, fst, newDbo);
140  backoff_state_[dst] = dbo;
141  }
142 
143  // state is final if last word is end of sentence
144  if (curwrd.compare(endSent) == 0) {
145  fst->SetFinal(dst, fst::StdArc::Weight::One());
146  }
147  // add labels to symbol tables
148  ilab = fst->MutableInputSymbols()->AddSymbol(curwrd);
149  olab = fst->MutableOutputSymbols()->AddSymbol(curwrd);
150 
151  // add arc with weight "prob" between source and destination states
152  // cerr << "n-gram prob, fstAddArc: src "<< src << " dst " << dst;
153  // cerr << " lab " << ilab << endl;
154  fst->AddArc(src, fst::StdArc(ilab, olab, prob, dst));
155 
156  // add backoffs to any newly created destination state
157  // but only if non-final
158  if (!IsFinal(fst, dst) && newDst && dbo != dst) {
159  ilab = olab = 0;
160  // cerr << "backoff, fstAddArc: src "<< src << " dst " << dst;
161  // cerr << " lab " << ilab << endl;
162  fst->AddArc(dst, fst::StdArc(ilab, olab, bow, dbo));
163  }
164 }

168 bool LmTable::ReadFstFromLmFile(std::istream &istrm,
169  fst::StdVectorFst *fst,
170  bool useNaturalOpt,
171  const string startSent,
172  const string endSent) {
173 #ifdef KALDI_PARANOID
174  KALDI_ASSERT(fst);
175  KALDI_ASSERT(fst->InputSymbols() && fst->OutputSymbols());
176 #endif
177 
178  conv_->UseNaturalLog(useNaturalOpt);
179 
180  // do not use state symbol table for word histories anymore
181  string inpline;
182  size_t pos1, pos2;
183  int ngram_order, max_ngram_order = 0;
184 
185  // process \data\ section
186 
187  while (getline(istrm, inpline) && !istrm.eof()) {
188  std::istringstream ss(inpline);
189  std::string token;
190  ss >> token >> std::ws;
191  if (token == "\\data\\" && ss.eof()) break;
192  }
193  if (istrm.eof()) {
194  KALDI_ERR << "\\data\\ token not found in arpa file.";
195  }
196 
197  std::vector<int> orders;
198  while (getline(istrm, inpline) && !istrm.eof()) {
199  // break out of loop if another section is found
200  if (inpline.find("-grams:") != string::npos) break;
201  if (inpline.find("\\end\\") != string::npos) break;
202 
203  // look for valid "ngram N = M" lines
204  pos1 = inpline.find("ngram");
205  pos2 = inpline.find("=");
206  if (pos1 == string::npos || pos2 == string::npos || pos2 <= pos1) {
207  continue; // not valid, continue looking
208  }
209  // found valid line
210  ngram_order = atoi(inpline.substr(pos1+5, pos2-(pos1+5)).c_str());
211  orders.push_back(ngram_order);
212  if (ngram_order > max_ngram_order) {
213  max_ngram_order = ngram_order;
214  }
215  }
216  if (max_ngram_order == 0) {
217  // reached end of loop without having found any n-gram
218  KALDI_ERR << "No ngrams found in specified file";
219  }
220 
221  for (int32 i = 0; i < orders.size(); i++) {
222  if (orders[i] != i+1)
223  KALDI_ERR << (i + 1) <<"-grams not specified in arpa file";
224  }
225 
226  // process "\N-grams:" sections, we may have already read a "\N-grams:" line
227  // if so, process it, otherwise get another line
228  while (inpline.find("-grams:") != string::npos
229  || (getline(istrm, inpline) && !istrm.eof()) ) {
230  // look for a valid "\N-grams:" section
231  pos1 = inpline.find("\\");
232  pos2 = inpline.find("-grams:");
233  if (pos1 == string::npos || pos2 == string::npos || pos2 <= pos1) {
234  continue; // not valid line, continue looking for one
235  }
236  // found, set current level
237  ngram_order = atoi(inpline.substr(pos1+1, pos2-(pos1+1)).c_str());
238  if (orders[0] != ngram_order)
239  KALDI_ERR << ngram_order << "-grams not specified in arpa header, or "
240  << "statistics of "<< orders[0] << "-grams not provided ? "
241  << "Check your arpa lm file.";
242  else {
243  cerr << "Processing " << ngram_order << "-grams" << endl;
244  orders.erase(orders.begin());
245  }
246 
247  // process individual n-grams
248  while (getline(istrm, inpline) && !istrm.eof()) {
249  // break out of inner loop if another section is found
250  if (!inpline.empty() && inpline[0] == '\\') {
251  if (inpline.find("-grams:") != string::npos) break;
252  if (inpline.find("\\end\\") != string::npos) break;
253  }
254  // parse ngram line: first field = prob, other fields = words,
255  // last field = backoff (optional)
256  std::vector<string> ngramString;
257  float prob, bow;
258 
259  // eat up space.
260  const char *cur_cstr = inpline.c_str();
261  while (*cur_cstr && isspace(*cur_cstr))
262  cur_cstr++;
263 
264  if (*cur_cstr == '\0') // Ignore empty lines.
265  continue;
266  char *next_cstr;
267  // found, parse probability from first field
268  prob = KALDI_STRTOF(cur_cstr, &next_cstr);
269  if (prob != prob || prob - prob != 0) {
270  KALDI_ERR << "nan or inf detected in LM file [parsing " << (ngram_order)
271  << "-grams]: " << inpline;
272  }
273  if (next_cstr == cur_cstr)
274  KALDI_ERR << "Bad line in LM file [parsing "<<(ngram_order)<<"-grams]: "<<inpline;
275  cur_cstr = next_cstr;
276  while (*cur_cstr && isspace(*cur_cstr))
277  cur_cstr++;
278 
279  // element 0 will be empty, element 1 will be the current word,
280  // element 2 will be the immediately preceding word, and so on.
281  // Apparently an IRSTLM convention.
282  ngramString.resize(ngram_order + 1);
283 
284  bool illegal_bos_or_eos = false;
285  for (int i = 0; i < ngram_order; i++) {
286  if (*cur_cstr == '\0')
287  KALDI_ERR << "Bad line in LM file [parsing "<<(ngram_order)<<"-grams]: "<<inpline;
288 
289  const char *end_cstr = strpbrk(cur_cstr, " \t\r");
290  std::string this_word;
291  if (end_cstr == NULL) {
292  this_word = std::string(cur_cstr);
293  cur_cstr += strlen(cur_cstr);
294  } else {
295  this_word = std::string(cur_cstr, end_cstr-cur_cstr);
296  cur_cstr = end_cstr;
297  while (*cur_cstr && isspace(*cur_cstr))
298  cur_cstr++;
299  }
300 
301  // Checks if <s> only appears at the beginning of the ngram, and if </s>
302  // only appears at the end of the ngram.
303  if ((ngram_order > 1 && i != 0 && this_word == "<s>") ||
304  (ngram_order > 1 && i != ngram_order - 1 && this_word == "</s>")) {
305  illegal_bos_or_eos = true;
306  break;
307  }
308 
309  ngramString[ngram_order - i].swap(this_word);
310  }
311  if (illegal_bos_or_eos) {
312  KALDI_WARN << "<s> is not at the beginning of the n-gram, or </s> is "
313  << "not at the end of the n-gram, skipping it: " << inpline;
314  continue;
315  }
316 
317  bow = 0;
318  if (ngram_order < max_ngram_order) {
319  // try converting anything left in the line to a backoff weight
320  if (*cur_cstr != '\0') {
321  char *end_cstr;
322  bow = KALDI_STRTOF(cur_cstr, &end_cstr);
323  if (bow != bow || bow - bow != 0) {
324  KALDI_ERR << "nan or inf detected in LM file [parsing " << (ngram_order)
325  << "-grams]: " << inpline;
326  }
327  if (end_cstr != cur_cstr) { // got something.
328  while (*end_cstr != '\0' && isspace(*end_cstr))
329  end_cstr++;
330  if (*end_cstr != '\0')
331  KALDI_ERR << "Junk " << (end_cstr) << " at end of line [parsing "
332  << (ngram_order) << "-grams]" << inpline;
333  } else {
334  KALDI_ERR << "Junk " << (cur_cstr) << " at end of line [parsing "
335  << (ngram_order) << "-grams]" << inpline;
336  }
337  }
338  }
339  conv_->AddArcsForNgramProb(ngram_order, max_ngram_order, prob, bow,
340  ngramString, fst,
341  startSent, endSent);
342  } // end of loop on individual n-gram lines
343  }
344  if (orders.size() > 0)
345  KALDI_ERR << orders[0] << "-grams specified in arpa header "
346  << "but no statistics provided to build FST";
347 
348  conv_->ConnectUnusedStates(fst);
349 
350  // not used anymore: delete pStateSymbs;
351 
352  // input and output symbol tables will be deleted by ~fst()
353  return true;
354 }

400 void LmTable::DumpContinue(ngram ng, int ngram_order, int elev,
401  table_entry_pos_t ipos, table_entry_pos_t epos,
402  fst::StdVectorFst *fst,
403  fst::SymbolTable *pStateSymbs,
404  const string startSent, const string endSent) {

406  ngram ing(ng.dict);
407  int ndsz = nodesize(ndt);
408 
409 #ifdef KALDI_PARANOID
410  KALDI_ASSERT(ng.size == ngram_order - 1);
411  KALDI_ASSERT(ipos >= 0 && epos <= cursize[ngram_order] && ipos < epos);
412  KALDI_ASSERT(pStateSymbs);
413 #endif
414 
415  ng.pushc(0);
416 
417  for (table_entry_pos_t i = ipos; i < epos; i++) {
418  *ng.wordp(1) = word(table[ngram_order] + (table_pos_t)i * ndsz);
419  float ipr = prob(table[ngram_order] + (table_pos_t)i * ndsz, ndt);
420  // int ipr = prob(table[ngram_order] + i * ndsz, ndt);
421  // skip pruned n-grams
422  if (isPruned && ipr == NOPROB) continue;
423 
424  if (ngram_order < elev) {
425  // get first and last successor position
426  table_entry_pos_t isucc = (i > 0 ? bound(table[ngram_order] +
427  (table_pos_t) (i-1) * ndsz,
428  ndt) : 0);
429  table_entry_pos_t esucc = bound(table[ngram_order] +
430  (table_pos_t) i * ndsz, ndt);
431  if (isucc < esucc) // there are successors!
432  DumpContinue(ng, ngram_order+1, elev, isucc, esucc,
433  fst, pStateSymbs, startSent, endSent);
434  // else
435  // cerr << "no successors for " << ng << "\n";
436  } else {
437  // cerr << i << " "; // this was just to count printed n-grams
438  // cerr << ipr <<"\t";
439  // cerr << (isQtable?ipr:*(float *)&ipr) <<"\t";
440

448

452  // words are inserted so position 1 is most recent word,

458  ngramString.insert(ngramString.begin(), "");

465  }
466  conv_->AddArcsForNgramProb(ngram_order, max_ngram_order, ipr, ibo,

469  }
470  }
