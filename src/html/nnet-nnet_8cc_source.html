67 void Nnet::Propagate(const CuMatrixBase<BaseFloat> &in, CuMatrix<BaseFloat> *out) {
68  KALDI_ASSERT(NULL != out);
69 
70  if (NumComponents() == 0) {
71  (*out) = in; // copy
72  return;
73  }
74 
75  // we need at least L+1 input buffers
76  KALDI_ASSERT((int32)propagate_buf_.size() >= NumComponents()+1);
77 
78  propagate_buf_[0].Resize(in.NumRows(), in.NumCols());
79  propagate_buf_[0].CopyFromMat(in);
80 
81  for(int32 i=0; i<(int32)components_.size(); i++) {
82  components_[i]->Propagate(propagate_buf_[i], &propagate_buf_[i+1]);
83  }
84 
85  (*out) = propagate_buf_[components_.size()];
86 }

89 void Nnet::Backpropagate(const CuMatrixBase<BaseFloat> &out_diff, CuMatrix<BaseFloat> *in_diff) {
90 
92  // Backpropagation
93  //
94 
95  // 0 layers
96  if (NumComponents() == 0) { (*in_diff) = out_diff; return; }
97 
98  KALDI_ASSERT((int32)propagate_buf_.size() == NumComponents()+1);
99  KALDI_ASSERT((int32)backpropagate_buf_.size() == NumComponents()+1);
100 
101  // copy out_diff to last buffer
102  backpropagate_buf_[NumComponents()] = out_diff;
103  // backpropagate using buffers
104  for (int32 i = NumComponents()-1; i >= 0; i--) {
105  components_[i]->Backpropagate(propagate_buf_[i], propagate_buf_[i+1],
106  backpropagate_buf_[i+1], &backpropagate_buf_[i]);
107  if (components_[i]->IsUpdatable()) {
108  UpdatableComponent *uc = dynamic_cast<UpdatableComponent*>(components_[i]);
109  uc->Update(propagate_buf_[i], backpropagate_buf_[i+1]);
110  }
111  }
112  // eventually export the derivative
113  if (NULL != in_diff) (*in_diff) = backpropagate_buf_[0];
114 
115  //
116  // End of Backpropagation
118 }

282 {
283  KALDI_ASSERT(NULL != out);
284 
285  KALDI_LOG << filename;
286 
287  if (NumComponents() == 0) {
288  out->Resize(in.NumRows(), in.NumCols());
289  out->CopyFromMat(in);
290  return;
291  }
292 
293  if (NumComponents() == 1) {
294  Output ot;
295  ot.Open(filename, binary, false);
296  WriteToken(ot.Stream(), binary, "<Layer0>");
297  in.Write(ot.Stream(), binary);
298  components_[0]->Propagate(in, out);
299  WriteToken(ot.Stream(), binary, "<Layer1>");
300  out->Write(ot.Stream(), binary);
301  ot.Close();
302  return;
303  }
304 
305  // we need at least 2 input buffers
306  KALDI_ASSERT(propagate_buf_.size() >= 2);
307 
308  // file to store data
309  std::string file = filename+"/features.txt";
310 
311  // propagate by using exactly 2 auxiliary buffers
312  Output ot;
313  int32 L = 0;
314  components_[L]->Propagate(in, &propagate_buf_[L%2]);
315  ot.Open(file, binary, false);
316  WriteToken(ot.Stream(), binary, "<Layer0>");
317  in.Write(ot.Stream(), binary);
318 
319  for(L++; L<=NumComponents()-2; L++) {
320  components_[L]->Propagate(propagate_buf_[(L-1)%2], &propagate_buf_[L%2]);
321  WriteToken(ot.Stream(), binary, "<Layer"+ToString<int32>(L)+">");
322  propagate_buf_[(L-1)%2].Write(ot.Stream(), binary);
323  }
324 
325  WriteToken(ot.Stream(), binary, "<Layer"+ToString<int32>(L)+">");
326  propagate_buf_[(L-1)%2].Write(ot.Stream(), binary);
327 
328  components_[L]->Propagate(propagate_buf_[(L-1)%2], out);
329 
330  WriteToken(ot.Stream(), binary, "<Layer"+ToString<int32>(L+1)+">");
331  out->Write(ot.Stream(), binary);
332  ot.Close();
333  // release the buffers we don't need anymore

423 void Nnet::GetWeights(Vector<BaseFloat>* wei_copy) const {
424  wei_copy->Resize(NumParams());
425  int32 pos = 0;
426  // copy the params
427  for(int32 n=0; n<components_.size(); n++) {
428  if(components_[n]->IsUpdatable()) {
429  switch(components_[n]->GetType()) {
430  case Component::kAffineTransform : {
431  // copy weight matrix row-by-row to the vector
432  Matrix<BaseFloat> mat(dynamic_cast<AffineTransform*>(components_[n])->GetLinearity());
433  int32 mat_size = mat.NumRows()*mat.NumCols();
434  wei_copy->Range(pos,mat_size).CopyRowsFromMat(mat);
435  pos += mat_size;
436  // append biases
437  Vector<BaseFloat> vec(dynamic_cast<AffineTransform*>(components_[n])->GetBias());
438  wei_copy->Range(pos,vec.Dim()).CopyFromVec(vec);
439  pos += vec.Dim();
440  } break;
441  default :
442  KALDI_ERR << "Unimplemented access to parameters "
443  << "of updatable component "
444  << Component::TypeToMarker(components_[n]->GetType());
445  }
446  }
447  }
448  KALDI_ASSERT(pos == NumParams());
449 }

452 void Nnet::SetWeights(const Vector<BaseFloat>& wei_src) {
453  KALDI_ASSERT(wei_src.Dim() == NumParams());
454  int32 pos = 0;
455  for(int32 n=0; n<components_.size(); n++) {
456  if(components_[n]->IsUpdatable()) {
457  switch(components_[n]->GetType()) {
458  case Component::kAffineTransform : {
459  // get the component
460  AffineTransform* aff_t = dynamic_cast<AffineTransform*>(components_[n]);
461  // we need weight matrix with original dimensions
462  Matrix<BaseFloat> mat(aff_t->GetLinearity());
463  int32 mat_size = mat.NumRows()*mat.NumCols();
464  mat.CopyRowsFromVec(wei_src.Range(pos,mat_size));
465  pos += mat_size;
466  // get the bias vector
467  Vector<BaseFloat> vec(aff_t->GetBias());
468  vec.CopyFromVec(wei_src.Range(pos,vec.Dim()));
469  pos += vec.Dim();
470  // assign to the component
471  aff_t->SetLinearity(CuMatrix<BaseFloat>(mat));
472  aff_t->SetBias(CuVector<BaseFloat>(vec));
473  } break;
474  default :
475  KALDI_ERR << "Unimplemented access to parameters "
476  << "of updatable component "
477  << Component::TypeToMarker(components_[n]->GetType());
478  }
479  }
480  }
481  KALDI_ASSERT(pos == NumParams());

485 void Nnet::GetGradient(Vector<BaseFloat>* grad_copy) const {
486  grad_copy->Resize(NumParams());
487  int32 pos = 0;
488  // copy the params
489  for(int32 n=0; n<components_.size(); n++) {
490  if(components_[n]->IsUpdatable()) {
491  switch(components_[n]->GetType()) {
492  case Component::kAffineTransform : {
493  // get the weights from CuMatrix to Matrix
494  const CuMatrixBase<BaseFloat>& cu_mat =
495  dynamic_cast<AffineTransform*>(components_[n])->GetLinearityCorr();
496  Matrix<BaseFloat> mat(cu_mat.NumRows(),cu_mat.NumCols());
497  cu_mat.CopyToMat(&mat);
498  // copy the the matrix row-by-row to the vector
499  int32 mat_size = mat.NumRows()*mat.NumCols();
500  grad_copy->Range(pos,mat_size).CopyRowsFromMat(mat);
501  pos += mat_size;
502  // get the biases from CuVector to Vector
503  const CuVector<BaseFloat>& cu_vec =
504  dynamic_cast<AffineTransform*>(components_[n])->GetBiasCorr();
505  Vector<BaseFloat> vec(cu_vec.Dim());
506  cu_vec.CopyToVec(&vec);
507  // append biases to the supervector
508  grad_copy->Range(pos,vec.Dim()).CopyFromVec(vec);
509  pos += vec.Dim();
510  } break;
511  default :
512  KALDI_ERR << "Unimplemented access to parameters "
513  << "of updatable component "
514  << Component::TypeToMarker(components_[n]->GetType());
515  }
516  }
517  }
518  KALDI_ASSERT(pos == NumParams());
519 }

564 void Nnet::Init(const std::string &file) {
565  Input in(file);
566  std::istream &is = in.Stream();
567  // do the initialization with config lines,
568  std::string conf_line, token;
569  while (!is.eof()) {
570  KALDI_ASSERT(is.good());
571  std::getline(is, conf_line); // get a line from config file,
572  if (conf_line == "") continue;
573  KALDI_VLOG(1) << conf_line;
574  std::istringstream(conf_line) >> std::ws >> token; // get 1st token,
575  if (token == "<NnetProto>" || token == "</NnetProto>") continue; // ignored tokens,
576  AppendComponent(Component::Init(conf_line+"\n"));
577  is >> std::ws;
578  }
579  // cleanup
580  in.Close();
581  Check();
582 }

704 void Nnet::Check() const {
705  // check we have correct number of buffers,
706  KALDI_ASSERT(propagate_buf_.size() == NumComponents()+1);
707  KALDI_ASSERT(backpropagate_buf_.size() == NumComponents()+1);
708  // check dims,
709  for (size_t i = 0; i + 1 < components_.size(); i++) {
710  KALDI_ASSERT(components_[i] != NULL);
711  int32 output_dim = components_[i]->OutputDim(),
712  next_input_dim = components_[i+1]->InputDim();
713  KALDI_ASSERT(output_dim == next_input_dim);
714  }
715  // check for nan/inf in network weights,
716  Vector<BaseFloat> weights;
717  GetParams(&weights);
718  BaseFloat sum = weights.Sum();
719  if(KALDI_ISINF(sum)) {
720  KALDI_ERR << "'inf' in network parameters (weight explosion, try lower learning rate?)";
721  }
722  if(KALDI_ISNAN(sum)) {
723  KALDI_ERR << "'nan' in network parameters (try lower learning rate?)";
724  }
725 }
