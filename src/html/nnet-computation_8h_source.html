63 // This defines one type of input that the network gets, or output that it will
64 // produce. For inputs, the name should correspond to an input or component
65 // node name in the nnet (components are allowed so context can be provided in
66 // recurrent setups); for outputs, the name should be an output node name in the
67 // Nnet. In the normal case there will just be one input and one output, and
68 // the indexes will vary only in the t index, with the others all identical.
69 struct IoSpecification {
70  std::string name;
71  std::vector<Index> indexes;
72  bool has_deriv; // For output nodes, true if a derivative w.r.t. that output
73  // will be supplied. For input nodes, true if the derivative

76

81  name(name), indexes(indexes), has_deriv(has_deriv) { }
