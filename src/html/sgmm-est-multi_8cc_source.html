29 int main(int argc, char *argv[]) {
30  using namespace kaldi;
31  // Memory for these will be freed in the catch block in case of exceptions.
32  std::vector<AmSgmm*> sgmms_in;
33  std::vector<MleAmSgmmAccs*> sgmm_accs_in;
34  std::vector<TransitionModel*> trans_models_in;
35 
36  try {
37  typedef kaldi::int32 int32;
38  const char *usage =
39  "Estimate multiple SGMM models from corresponding stats, such that the"
40  " global parameters\n(phone-, speaker-, and weight-projections and "
41  "covariances) are tied across models.\n"
42  "Usage: sgmm-est-multi [options] <model1> <stats1> <model1_out> <occs1_out> [<model2> "
43  "<stats2> <model2_out> <occs2_out> ...]\n";
44 
45  bool binary_write = true;
46  std::string update_flags_str = "vMNwcSt";
47  std::string write_flags_str = "gsnu";
48  kaldi::MleTransitionUpdateConfig tcfg;
49  kaldi::MleAmSgmmOptions sgmm_opts;
50  std::string split_substates = ""; // Space-seperated list of #substates
51  std::vector<int32> split_substates_int; // The above string split on space
52  int32 increase_phn_dim = 0;
53  int32 increase_spk_dim = 0;
54  bool remove_speaker_space = false;
55  BaseFloat perturb_factor = 0.01;
56  BaseFloat power = 0.2;
57  BaseFloat max_cond = 100;
58 
59  ParseOptions po(usage);
60  po.Register("binary", &binary_write, "Write output in binary mode");
61  // The split-substates option also takes a single integer: the same number
62  // of substates for all models.
63  po.Register("split-substates", &split_substates, "Space-separated string "
64  "with target number of substates for each model.");
65  po.Register("increase-phn-dim", &increase_phn_dim, "Increase phone-space "
66  "dimension as far as allowed towards this target.");
67  po.Register("increase-spk-dim", &increase_spk_dim, "Increase speaker-space "
68  "dimension as far as allowed towards this target.");
69  po.Register("remove-speaker-space", &remove_speaker_space,
70  "Remove speaker-specific projections N");
71  po.Register("power", &power, "Exponent for substate occupancies used while "
72  "splitting substates.");
73  po.Register("perturb-factor", &perturb_factor, "Perturbation factor for "
74  "state vectors while splitting substates.");
75  po.Register("max-cond-split", &max_cond, "Max condition number of smoothing "
76  "matrix used in substate splitting.");
77  po.Register("update-flags", &update_flags_str, "Which SGMM parameters to "
78  "update: subset of vMNwcSt.");
79  po.Register("write-flags", &write_flags_str, "Which SGMM parameters to "
80  "write: subset of gsnu");
81  tcfg.Register(&po);
82  sgmm_opts.Register(&po);
83 
84  po.Read(argc, argv);
85  if (po.NumArgs() <= 0 || (po.NumArgs() % 4 != 0)) {
86  po.PrintUsage();
87  exit(1);
88  }
89  // How many 4-tuples of model, stats, output model, output occs
90  int32 num_models = po.NumArgs()/4;
91  sgmms_in.resize(num_models, NULL);
92  sgmm_accs_in.resize(num_models, NULL);
93  trans_models_in.resize(num_models, NULL);
94 
95  if (!split_substates.empty()) {
96  SplitStringToIntegers(split_substates, " ", true /*omit empty strings*/,
97  &split_substates_int);
98  if (split_substates_int.size() == 1) { // Same #substates for all models
99  int32 tmp_int = split_substates_int[0];
100  split_substates_int.resize(num_models, tmp_int);
101  }
102  if (split_substates_int.size() != num_models) {
103  KALDI_ERR << "Found " << split_substates_int.size() << " splitting "
104  << "targets; expecting 1 or " << num_models;
105  }
106  }
107 
108  SgmmUpdateFlagsType update_flags = StringToSgmmUpdateFlags(update_flags_str);
109  SgmmWriteFlagsType write_flags = StringToSgmmWriteFlags(write_flags_str);
110 
111  std::vector<std::string> model_out_filenames(num_models);
112  std::vector<std::string> occs_out_filenames(num_models);
113  int32 phn_dim, spk_dim, num_gauss, feat_dim;
114 
115  for (int i = 0; i < num_models; ++i) {
116  std::string model_in_filename = po.GetArg(i*4+1),
117  stats_filename = po.GetArg(i*4+2);
118  model_out_filenames[i] = po.GetArg(i*4+3);
119  occs_out_filenames[i] = po.GetArg(i*4+4);
120 
121  AmSgmm *am_sgmm = new AmSgmm();
122  TransitionModel *trans_model = new TransitionModel();
123  {
124  bool binary;
125  Input ki(model_in_filename, &binary);
126  trans_model->Read(ki.Stream(), binary);
127  am_sgmm->Read(ki.Stream(), binary);
128  }
129  if (i == 0) {
130  phn_dim = am_sgmm->PhoneSpaceDim();
131  spk_dim = am_sgmm->SpkSpaceDim();
132  num_gauss = am_sgmm->NumGauss();
133  feat_dim = am_sgmm->FeatureDim();
134  } else {
135  if (am_sgmm->PhoneSpaceDim() != phn_dim) {
136  KALDI_ERR << "File '" << model_in_filename << "': mismatched "
137  << "phone-space dim: expecting " << phn_dim << ", found "
138  << am_sgmm->PhoneSpaceDim();
139  }
140  if (am_sgmm->SpkSpaceDim() != spk_dim) {
141  KALDI_ERR << "File '" << model_in_filename << "': mismatched "
142  << "speaker-space dim: expecting " << spk_dim << ", found "
143  << am_sgmm->SpkSpaceDim();
144  }
145  if (am_sgmm->NumGauss() != num_gauss) {
146  KALDI_ERR << "File '" << model_in_filename << "': mismatched UBM "
147  << "size: expecting " << num_gauss << ", found "
148  << am_sgmm->NumGauss();
149  }
150  if (am_sgmm->FeatureDim() != feat_dim) {
151  KALDI_ERR << "File '" << model_in_filename << "': mismatched feature "
152  << "dim: expecting " << feat_dim << ", found "
153  << am_sgmm->FeatureDim();
154  }
155  }
156  sgmms_in[i] = am_sgmm;
157  trans_models_in[i] = trans_model;
158 
159  Vector<double> transition_accs;
160  MleAmSgmmAccs *sgmm_accs = new MleAmSgmmAccs();
161  {
162  bool binary;
163  Input ki(stats_filename, &binary);
164  transition_accs.Read(ki.Stream(), binary);
165  sgmm_accs->Read(ki.Stream(), binary, false);
166  }
167  // Check consistency and print some diagnostics.
168  sgmm_accs->Check(*am_sgmm, true);
169  sgmm_accs_in[i] = sgmm_accs;
170 
171  if (update_flags & kSgmmTransitions) { // Update transition model.
172  BaseFloat objf_impr, count;
173  KALDI_LOG << "Updating transitions for model: " << model_in_filename;
174  trans_model->MleUpdate(transition_accs, tcfg, &objf_impr, &count);
175  KALDI_LOG << "Transition model update: average " << (objf_impr/count)
176  << " log-like improvement per frame over " << (count)
177  << " frames";
178  }
179  }
180 
181  { // Update all the SGMMs together.
182  kaldi::MleAmSgmmUpdaterMulti multi_sgmm_updater(*sgmms_in[0], sgmm_opts);
183  multi_sgmm_updater.Update(sgmm_accs_in, sgmms_in, update_flags);
184  }
185 
186  for (int i = 0; i < num_models; ++i) {
187  Vector<BaseFloat> state_occs;
188  sgmm_accs_in[i]->GetStateOccupancies(&state_occs);
189 
190  if (!split_substates.empty()) {
191  sgmms_in[i]->SplitSubstates(state_occs, split_substates_int[i], perturb_factor,
192  power, max_cond);
193  sgmms_in[i]->ComputeDerivedVars(); // recompute normalizers...
194  }
195 
196  {
197  kaldi::Output ko(occs_out_filenames[i], false /* no binary write */);
198  state_occs.Write(ko.Stream(), false /* no binary write */);
199  }
200 
201  if (increase_phn_dim != 0 || increase_spk_dim != 0) {
202  // Feature normalizing transform matrix used to initialize the new columns
203  // of the phonetic- or speaker-space projection matrices.
204  kaldi::Matrix<BaseFloat> norm_xform;
205  ComputeFeatureNormalizer(sgmms_in[i]->full_ubm(), &norm_xform);
206  if (increase_phn_dim != 0)
207  sgmms_in[i]->IncreasePhoneSpaceDim(increase_phn_dim, norm_xform);
208  if (increase_spk_dim != 0)
209  sgmms_in[i]->IncreaseSpkSpaceDim(increase_spk_dim, norm_xform);
210  }
211  if (remove_speaker_space) {
212  KALDI_LOG << "Removing speaker space (projections N_)";
213  sgmms_in[i]->RemoveSpeakerSpace();
214  }
215 
216  {
217  Output ko(model_out_filenames[i], binary_write);
218  trans_models_in[i]->Write(ko.Stream(), binary_write);
219  sgmms_in[i]->Write(ko.Stream(), binary_write, write_flags);
220  KALDI_LOG << "Written model to " << model_out_filenames[i];
221  }
222  }
223  return 0;
224  } catch(const std::exception& e) {
225  kaldi::DeletePointers(&sgmms_in);
226  kaldi::DeletePointers(&sgmm_accs_in);
227  kaldi::DeletePointers(&trans_models_in);
228  std::cerr << e.what();
229  return -1;
230  }
231 }
