59 
60  // all these "fn"'s are either rspecifiers or filenames.
61 
62  bool in_is_rspecifier =

65  out_is_wspecifier =

70  KALDI_ERR << "Cannot mix archives with regular files (copying matrices)";
71 
72  if (!in_is_rspecifier) {
73  Matrix<BaseFloat> mat;
74  ReadKaldiObject(matrix_in_fn, &mat);
75  if (scale != 1.0) mat.Scale(scale);
76  Output ko(matrix_out_fn, binary);
77  mat.Write(ko.Stream(), binary);
78  KALDI_LOG << "Copied matrix to " << matrix_out_fn;
79  return 0;

84  for (; !reader.Done(); reader.Next(), num_done++) {
85  if (scale != 1.0) {
86  Matrix<BaseFloat> mat(reader.Value());
87  mat.Scale(scale);
88  writer.Write(reader.Key(), mat);
89  } else {
90  writer.Write(reader.Key(), reader.Value());
91  }
92  }
93  KALDI_LOG << "Copied " << num_done << " matrices.";
94  return (num_done != 0 ? 0 : 1);
95  }
96  } catch(const std::exception &e) {
97  std::cerr << e.what();
