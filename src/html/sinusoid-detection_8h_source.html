172  // For each bin b such that we have valid "info" data for bins b, b+1 and b+2,
173  // does quadratic interpolation to find the maximum predicted energy in the
174  // range [b, b+2]. The location of the maximum predicted energy is output to
175  // "bin_out" and "offset_out", and the corresponding predicted energy is
176  // returned.
177  //
178  // Note: if there are two different frequencies with similar maximum energies
179  // (e.g. within a factor of probably around 1.2 or so), the fact that
180  // OptimizeFrequency only returns one maximum may potentially lead to the
181  // smaller maximum being output. We could have modified this to output
182  // multiple different maxima, which could have been more accurate in terms of
183  // being guaranteed to output the best maximum, but this probably wouldn't
184  // have a measurable impact on our application so we haven't bothered.
185  BaseFloat OptimizeFrequency(

228 struct MultiSinusoidDetectorConfig {
229 
230  // frame length in milliseconds
231  BaseFloat frame_length_ms;
232  // frame shift in milliseconds
233  BaseFloat frame_shift_ms;
234 
235  // Proportion of the total energy of the signal that the quieter of
236  // the two sinusoids must comprise, in order to be counted, if two
237  // sinusoids are detected.
238  BaseFloat two_freq_min_energy;
239 
240  // Proportion of the total energy of the signal that both sinusoids (if
241  // two are detected) must comprise, in order to be output.
242  BaseFloat two_freq_min_total_energy;
243 
244  // Proportion of the total energy of the signal that a single sinusoid
245  // must comprise, in order to be output, if we are considering
246  // reporting a single sinusoid. Note: detection of two sinusoids
247  // will take precedence over detection of a single sinusoid.
248  BaseFloat one_freq_min_energy;
249 
250  // Lower end of frequency range that we consider; frequencies outside
251  // this range are not candidates to appear in the detected output.
252  BaseFloat min_freq;
253  // Upper end of frequency range that we consider, see min_freq.
254  BaseFloat max_freq;
255 
256  // Frequency to which we subsample the signal before processing it.
257  // Must be integer because of how LinearResample code works.
258  int32 subsample_freq;
259 
260  // Filter cut-off frequency used in sub-sampling.
261  BaseFloat subsample_filter_cutoff;
262 
263  // the following is not critical and is not exported to the
264  // command line.
265  int32 subsample_filter_zeros;
266 
267  MultiSinusoidDetectorConfig():
268  frame_length_ms(20), frame_shift_ms(10),
269  two_freq_min_energy(0.2), two_freq_min_total_energy(0.6),
270  one_freq_min_energy(0.75), min_freq(300.0),
271  max_freq(1800.0), subsample_freq(4000),
272  subsample_filter_cutoff(1900.0), subsample_filter_zeros(5) {}

299  void Check() const {
300  KALDI_ASSERT(frame_length_ms > 0 && frame_length_ms >= frame_shift_ms &&
301  min_freq > 0 && max_freq > min_freq &&
302  subsample_filter_cutoff > max_freq &&
303  subsample_freq/2 > subsample_filter_cutoff &&
304  subsample_filter_zeros > 2 &&
305  subsample_filter_cutoff > 0.25 * subsample_freq &&
306  two_freq_min_total_energy > two_freq_min_energy &&
307  two_freq_min_energy <= 0.5 * two_freq_min_total_energy);
308  BaseFloat samples_per_frame_shift =
309  frame_shift_ms * 0.001 * subsample_freq;
310  // The following assert ensures that the frame-shift is an exact
311  // number of samples, so that the locations of the frames
312  // don't gradually drift out of sync.
313  KALDI_ASSERT(fabs(samples_per_frame_shift -
314  static_cast<int32>(samples_per_frame_shift)) <
315  0.001);
316 
317  }
