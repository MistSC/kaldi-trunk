20 // This program is for system combination using MBR decoding as described in:
21 // "Minimum Bayes Risk decoding and system combination based on a recursion for
22 // edit distance", Haihua Xu, Daniel Povey, Lidia Mangu and Jie Zhu, Computer
23 // Speech and Language, 2011. However, instead of averaging the posteriors, as
24 // described in the paper, this removes the total backward probability from the
25 // individual lattices being combined and outputs the union of them. The output
26 // should be used with lattice-mbr-decode (without any acoustic or LM scaling)
27 // or with lattice-to-ctm-conf with --decode-mbr=true (also without any scaling)
28 
29 // IMPORTANT CAVEAT: the total backward probability (which is a float) is
30 // removed from value1_ of arc weight. So graph scores are no longer correct
31 // but instead only the combined acoustic and graph scores are valid. So no
32 // acoustic or LM scaling should be done with the output of this program.

46 // This removes the total weight from a CompactLattice. Since the total backward
47 // score is in log likelihood domain, and the lattice weights are in negative
48 // log likelihood domain, the total weight is *added* to the weight of the final
49 // states. This is equivalent to dividing the probability of each path by the
50 // total probability over all paths. There is an additional weight to control
51 // the relative contribution of individual lattices, which can be used as an
52 // exponent (i.e. multiply the log scores) or as a multiplicative weight (i.e.
53 // log of the weight is subtracted from the total backward score).
54 bool CompactLatticeNormalize(CompactLattice *clat, BaseFloat weight,
55  bool exp_weights) {
56  if (weight <= 0.0) {
57  KALDI_WARN << "Weights must be positive; found: " << weight;
58  return false;
59  }
60 
61  if (clat->Properties(fst::kTopSorted, false) == 0) {
62  if (fst::TopSort(clat) == false) {
63  KALDI_WARN << "Cycles detected in lattice: cannot normalize.";
64  return false;
65  }
66  }
67 
68  // If exp_weights = true, multiply the log AM & LM scores.
69  if (exp_weights)
70  fst::ScaleLattice(fst::LatticeScale(weight, weight), clat);
71 
72  vector<double> beta;
73  if (!ComputeCompactLatticeBetas(*clat, &beta)) {
74  KALDI_WARN << "Failed to compute backward probabilities on lattice.";
75  return false;
76  }
77 
78  typedef CompactLattice::Arc::StateId StateId;
79  StateId start = clat->Start(); // Should be 0
80  BaseFloat total_backward_cost = beta[start];
81 
82  // If exp_weights = false, add to the log AM & LM scores.
83  if (!exp_weights)
84  total_backward_cost -= Log(weight);
85 
86  for (fst::StateIterator<CompactLattice> sit(*clat); !sit.Done(); sit.Next()) {
87  CompactLatticeWeight f = clat->Final(sit.Value());
88  LatticeWeight w = f.Weight();
89  w.SetValue1(w.Value1() + total_backward_cost);
90  f.SetWeight(w);
91  clat->SetFinal(sit.Value(), f);
92  }
93  return true;
94 }

96 // This is a wrapper for SplitStringToFloats, with added checks to make sure
97 // the weights are valid probabilities.
98 void SplitStringToWeights(const string &full, const char *delim,
99  vector<BaseFloat> *out) {
100  vector<BaseFloat> tmp;
101  SplitStringToFloats(full, delim, true /*omit empty strings*/, &tmp);
102  if (tmp.size() != out->size()) {
103  KALDI_WARN << "Expecting " << out->size() << " weights, found " << tmp.size()
104  << ": using uniform weights.";
105  return;
106  }
107  BaseFloat sum = 0;
108  for (vector<BaseFloat>::const_iterator itr = tmp.begin();
109  itr != tmp.end(); ++itr) {
110  if (*itr < 0.0) {
111  KALDI_WARN << "Cannot use negative weight: " << *itr << "; input string: "
112  << full << "\n\tUsing uniform weights.";
113  return;
114  }
115  sum += (*itr);
116  }
117  if (sum != 1.0) {
118  KALDI_WARN << "Weights sum to " << sum << " instead of 1: renormalizing";
119  for (vector<BaseFloat>::iterator itr = tmp.begin();
120  itr != tmp.end(); ++itr)
121  (*itr) /= sum;
122  }
123  out->swap(tmp);
124 }

129 int main(int argc, char *argv[]) {
130  try {
131  using namespace kaldi;
132  typedef kaldi::int32 int32;
133 
134  const char *usage =
135  "Combine lattices generated by different systems by removing the total\n"
136  "cost of all paths (backward cost) from individual lattices and doing\n"
137  "a union of the reweighted lattices. Note: the acoustic and LM scales\n"
138  "that this program applies are not removed before outputting the lattices.\n"
139  "Intended for use in system combination prior to MBR decoding, see comments\n"
140  "in code.\n"
141  "Usage: lattice-combine [options] lattice-rspecifier1 lattice-rspecifier2"
142  " [lattice-rspecifier3 ... ] lattice-wspecifier\n";
143 
144  ParseOptions po(usage);
145  BaseFloat acoustic_scale = 1.0, inv_acoustic_scale = 1.0, lm_scale = 1.0;
146  string weight_str;
147  bool exp_weights = false;
148  po.Register("acoustic-scale", &acoustic_scale, "Scaling factor for "
149  "acoustic likelihoods");
150  po.Register("inv-acoustic-scale", &inv_acoustic_scale, "An alternative way "
151  "of setting the acoustic scale: you can set its inverse.");
152  po.Register("lm-scale", &lm_scale, "Scaling factor for language model "
153  "probabilities");
154  po.Register("lat-weights", &weight_str, "Colon-separated list of weights "
155  "for each lattice that sum to 1");
156  // The lattice weights can be used either as exponents or as multiplicative
157  // weights, determined by the exp_weights flag.
158  po.Register("exp-weights", &exp_weights, "If true, lattice weights are "
159  "exponents, else they are multiplicative");
160  po.Read(argc, argv);
161 
162  KALDI_ASSERT(acoustic_scale == 1.0 || inv_acoustic_scale == 1.0);
163  if (inv_acoustic_scale != 1.0)
164  acoustic_scale = 1.0 / inv_acoustic_scale;
165 
166 
167  int32 num_args = po.NumArgs();
168  if (num_args < 3) {
169  po.PrintUsage();
170  exit(1);
171  }
172 
173  string lats_rspecifier1 = po.GetArg(1),
174  lats_wspecifier = po.GetArg(num_args);
175 
176  // Output lattice
177  CompactLatticeWriter clat_writer(lats_wspecifier);
178 
179  // Input lattices
180  SequentialCompactLatticeReader clat_reader1(lats_rspecifier1);
181  vector<RandomAccessCompactLatticeReader*> clat_reader_vec(
182  num_args-2, static_cast<RandomAccessCompactLatticeReader*>(NULL));
183  vector<string> clat_rspec_vec(num_args-2);
184  for (int32 i = 2; i < num_args; ++i) {
185  clat_reader_vec[i-2] = new RandomAccessCompactLatticeReader(po.GetArg(i));
186  clat_rspec_vec[i-2] = po.GetArg(i);
187  }
188 
189  vector<BaseFloat> lat_weights(num_args-1, 1.0/(num_args-1));
190  if (!weight_str.empty())
191  SplitStringToWeights(weight_str, ":", &lat_weights);
192 
193  int32 n_utts = 0, n_total_lats = 0, n_success = 0, n_missing = 0,
194  n_other_errors = 0;
195  vector< vector<double> > lat_scale = fst::LatticeScale(lm_scale,
196  acoustic_scale);
197 
198  for (; !clat_reader1.Done(); clat_reader1.Next()) {
199  std::string key = clat_reader1.Key();
200  CompactLattice clat1 = clat_reader1.Value();
201  clat_reader1.FreeCurrent();
202  n_utts++;
203  n_total_lats++;
204  fst::ScaleLattice(lat_scale, &clat1);
205  bool success = CompactLatticeNormalize(&clat1, lat_weights[0],
206  exp_weights);
207  if (!success) {
208  KALDI_WARN << "Could not normalize lattice for system 1, utterance: "
209  << key;
210  n_other_errors++;
211  continue;
212  }
213 
214  for (int32 i = 0; i < num_args-2; ++i) {
215  if (clat_reader_vec[i]->HasKey(key)) {
216  CompactLattice clat2 = clat_reader_vec[i]->Value(key);
217  n_total_lats++;
218  fst::ScaleLattice(lat_scale, &clat2);
219  success = CompactLatticeNormalize(&clat2, lat_weights[i+1], exp_weights);
220  if (!success) {
221  KALDI_WARN << "Could not normalize lattice for system "<< (i + 2)
222  << ", utterance: " << key;
223  n_other_errors++;
224  continue;
225  }
226  fst::Union(&clat1, clat2);
227  } else {
228  KALDI_WARN << "No lattice found for utterance " << key << " for "
229  << "system " << (i + 2) << ", rspecifier: "
230  << clat_rspec_vec[i];
231  n_missing++;
232  }
233  }
234 
235  clat_writer.Write(key, clat1);
236  n_success++;
237  }
238 
239  KALDI_LOG << "Processed " << n_utts << " utterances: with a total of "
240  << n_total_lats << " lattices across " << (num_args-1)
241  << " different systems";
242  KALDI_LOG << "Produced output for " << n_success << " utterances; "
243  << n_missing << " total missing lattices and " << n_other_errors
244  << " total lattices had errors in processing.";
245  DeletePointers(&clat_reader_vec);
246  // The success code we choose is that at least one lattice was output,
247  // and more lattices were "all there" than had at least one system missing.
248  return (n_success != 0 && n_missing < (n_success - n_missing) ? 0 : 1);
249  } catch(const std::exception &e) {
250  std::cerr << e.what();
251  return -1;
252  }
253 }
