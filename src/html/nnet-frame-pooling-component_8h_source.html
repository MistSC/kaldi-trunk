56  void InitData(std::istream &is) {
57  // temporary, for initialization,
58  std::vector<int32> pool_size;
59  std::vector<int32> central_offset;
60  Vector<BaseFloat> pool_weight;
61  float learn_rate_coef = 0.01;
62  // parse config
63  std::string token;
64  while (!is.eof()) {
65  ReadToken(is, false, &token);
66  if (token == "<FeatureDim>") ReadBasicType(is, false, &feature_dim_);
67  else if (token == "<CentralOffset>") ReadIntegerVector(is, false, &central_offset);
68  else if (token == "<PoolSize>") ReadIntegerVector(is, false, &pool_size);
69  else if (token == "<PoolWeight>") pool_weight.Read(is, false);
70  else if (token == "<LearnRateCoef>") ReadBasicType(is, false, &learn_rate_coef);
71  else if (token == "<Normalize>") ReadBasicType(is, false, &normalize_);
72  else KALDI_ERR << "Unknown token " << token << ", a typo in config?"
73  << " (FeatureDim|CentralOffset <vec>|PoolSize <vec>|LearnRateCoef|Normalize)";
74  is >> std::ws; // eat-up whitespace
75  }
76  // check inputs:
77  KALDI_ASSERT(feature_dim_ > 0);
78  KALDI_ASSERT(central_offset.size() > 0);
79  KALDI_ASSERT(central_offset.size() == pool_size.size());
80  // initialize:
81  int32 num_frames = InputDim() / feature_dim_;
82  int32 central_frame = (num_frames -1) / 2;
83  int32 num_pools = central_offset.size();
84  offset_.resize(num_pools);
85  weight_.resize(num_pools);
86  for (int32 p=0; p<num_pools; p++) {
87  offset_[p] = central_frame + central_offset[p] + std::min(0,pool_size[p]+1);
88  weight_[p].Resize(std::abs(pool_size[p]));
89  weight_[p].Set(1.0/std::abs(pool_size[p]));
90  }
91  learn_rate_coef_ = learn_rate_coef;
92  if (pool_weight.Dim() != 0) {
93  KALDI_LOG << "Initializing from pool-weight vector";
94  int32 num_weights = 0;
95  for (int32 p=0; p<num_pools; p++) {
96  weight_[p].CopyFromVec(pool_weight.Range(num_weights, weight_[p].Dim()));
97  num_weights += weight_[p].Dim();
98  }
99  KALDI_ASSERT(num_weights == pool_weight.Dim());
100  }
101  // check that offsets are within the splice we had,
102  for (int32 p=0; p<num_pools; p++) {
103  KALDI_ASSERT(offset_[p] >= 0);
104  KALDI_ASSERT(offset_[p] + weight_[p].Dim() <= num_frames);
105  }
106  }

112  void ReadData(std::istream &is, bool binary) {
113  // get the input dimension before splicing
114  ExpectToken(is, binary, "<FeatureDim>");
115  ReadBasicType(is, binary, &feature_dim_);
116  ExpectToken(is, binary, "<LearnRateCoef>");
117  ReadBasicType(is, binary, &learn_rate_coef_);
118  ExpectToken(is, binary, "<Normalize>");
119  ReadBasicType(is, binary, &normalize_);
120  // read the offsets w.r.t. central frame
121  ExpectToken(is, binary, "<FrameOffset>");
122  ReadIntegerVector(is, binary, &offset_);
123  // read the frame-weights
124  ExpectToken(is, binary, "<FrameWeight>");
125  int32 num_pools = offset_.size();
126  weight_.resize(num_pools);
127  for (int32 p=0; p<num_pools; p++) {
128  weight_[p].Read(is, binary);
129  }
130  //
131  // Sanity checks:
132  //
133  KALDI_ASSERT(input_dim_ % feature_dim_ == 0);
134  KALDI_ASSERT(output_dim_ % feature_dim_ == 0);
135  KALDI_ASSERT(output_dim_ / feature_dim_ == num_pools);
136  KALDI_ASSERT(offset_.size() == weight_.size());
137  // check the shifts don't exceed the splicing
138  int32 total_frame = InputDim() / feature_dim_;
139  for (int32 p=0; p<num_pools; p++) {
140  KALDI_ASSERT(offset_[p] >= 0);
141  KALDI_ASSERT(offset_[p] + (weight_[p].Dim()-1) < total_frame);
142  }
143  //
144  }

202  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
203  // check dims
204  KALDI_ASSERT(in.NumCols() % feature_dim_ == 0);
205  KALDI_ASSERT(out->NumCols() % feature_dim_ == 0);
206  // useful dims
207  int32 num_pools = offset_.size();
208  // compute the output pools
209  for (int32 p=0; p<num_pools; p++) {
210  CuSubMatrix<BaseFloat> tgt(out->ColRange(p*feature_dim_,feature_dim_));
211  tgt.SetZero(); // reset
212  for (int32 i=0; i<weight_[p].Dim(); i++) {
213  tgt.AddMat(weight_[p](i), in.ColRange((offset_[p]+i) * feature_dim_, feature_dim_));
214  }
215  }
216  }

224  void Update(const CuMatrixBase<BaseFloat> &input, const CuMatrixBase<BaseFloat> &diff) {
225  // useful dims
226  int32 num_pools = offset_.size();
227  // lazy init
228  if (weight_diff_.size() != num_pools) weight_diff_.resize(num_pools);
229  // get the derivatives
230  for (int32 p=0; p<num_pools; p++) {
231  weight_diff_[p].Resize(weight_[p].Dim(), kSetZero); // reset
232  for (int32 i=0; i<weight_[p].Dim(); i++) {
233  // multiply matrices element-wise, and sum to get the derivative
234  CuSubMatrix<BaseFloat> in_frame(input.ColRange((offset_[p]+i) * feature_dim_, feature_dim_));
235  CuSubMatrix<BaseFloat> diff_frame(diff.ColRange(p * feature_dim_, feature_dim_));
236  CuMatrix<BaseFloat> mul_elems(in_frame);
237  mul_elems.MulElements(diff_frame);
238  weight_diff_[p](i) = mul_elems.Sum();
239  }
240  }
241  // update
242  for (int32 p=0; p<num_pools; p++) {
243  weight_[p].AddVec(- learn_rate_coef_ * opts_.learn_rate, weight_diff_[p]);
244  }
245  // force to be positive, re-normalize the sum
246  if (normalize_) {
247  for (int32 p=0; p<num_pools; p++) {
248  weight_[p].ApplyFloor(0.0);
249  weight_[p].Scale(1.0/weight_[p].Sum());
250  }
251  }
252  }
