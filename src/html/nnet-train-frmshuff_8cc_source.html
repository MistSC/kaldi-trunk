94  using namespace kaldi::nnet1;

98 #if HAVE_CUDA==1
99  CuDevice::Instantiate().SelectGpuId(use_gpu);

103  if(feature_transform != "") {
104  nnet_transf.Read(feature_transform);
105  }
106 
107  Nnet nnet;
108  nnet.Read(model_filename);
109  nnet.SetTrainOptions(trn_opts);

112  nnet_transf.SetDropoutRetention(dropout_retention);
113  nnet.SetDropoutRetention(dropout_retention);
114  }

116  nnet_transf.SetDropoutRetention(1.0);
117  nnet.SetDropoutRetention(1.0);
118  }

121 
122  SequentialBaseFloatMatrixReader feature_reader(feature_rspecifier);
123  RandomAccessPosteriorReader targets_reader(targets_rspecifier);
124  RandomAccessBaseFloatVectorReader weights_reader;
125  if (frame_weights != "") {
126  weights_reader.Open(frame_weights);
127  }
128  RandomAccessBaseFloatReader utt_weights_reader;
129  if (utt_weights != "") {

135  PosteriorRandomizer targets_randomizer(rnd_opts);
136  VectorRandomizer weights_randomizer(rnd_opts);
137 
138  Xent xent;
139  Mse mse;
140 
141  MultiTaskLoss multitask;
142  if (0 == objective_function.compare(0,9,"multitask")) {
143  // objective_function contains something like :
144  // 'multitask,xent,2456,1.0,mse,440,0.001'
145  //
146  // the meaning is following:
147  // 'multitask,<type1>,<dim1>,<weight1>,...,<typeN>,<dimN>,<weightN>'
148  multitask.InitFromString(objective_function);
149  }

152 
153  Timer time;
154  KALDI_LOG << (crossvalidate?"CROSS-VALIDATION":"TRAINING") << " STARTED";
155 
156  int32 num_done = 0, num_no_tgt_mat = 0, num_other_error = 0;
157  while (!feature_reader.Done()) {
158 #if HAVE_CUDA==1
159  // check the GPU is not overheated
160  CuDevice::Instantiate().CheckGpuHealth();
161 #endif
162  // fill the randomizer
163  for ( ; !feature_reader.Done(); feature_reader.Next()) {
164  if (feature_randomizer.IsFull()) break; // suspend, keep utt for next loop
165  std::string utt = feature_reader.Key();
166  KALDI_VLOG(3) << "Reading " << utt;
167  // check that we have targets
168  if (!targets_reader.HasKey(utt)) {
169  KALDI_WARN << utt << ", missing targets";
170  num_no_tgt_mat++;
171  continue;
172  }
173  // check we have per-frame weights
174  if (frame_weights != "" && !weights_reader.HasKey(utt)) {
175  KALDI_WARN << utt << ", missing per-frame weights";
176  num_other_error++;
177  continue;
178  }
179  // check we have per-utterance weights
180  if (utt_weights != "" && !utt_weights_reader.HasKey(utt)) {
181  KALDI_WARN << utt << ", missing per-utterance weight";
182  num_other_error++;
183  continue;
184  }
185  // get feature / target pair
186  Matrix<BaseFloat> mat = feature_reader.Value();
187  Posterior targets = targets_reader.Value(utt);
188  // get per-frame weights
189  Vector<BaseFloat> weights;
190  if (frame_weights != "") {
191  weights = weights_reader.Value(utt);
192  } else { // all per-frame weights are 1.0
193  weights.Resize(mat.NumRows());
194  weights.Set(1.0);
195  }
196  // multiply with per-utterance weight,
197  if (utt_weights != "") {
198  BaseFloat w = utt_weights_reader.Value(utt);
199  KALDI_ASSERT(w >= 0.0);
200  if (w == 0.0) continue; // remove sentence from training,
201  weights.Scale(w);
202  }
203 
204  // correct small length mismatch ... or drop sentence
205  {
206  // add lengths to vector
207  std::vector<int32> lenght;
208  lenght.push_back(mat.NumRows());
209  lenght.push_back(targets.size());
210  lenght.push_back(weights.Dim());
211  // find min, max
212  int32 min = *std::min_element(lenght.begin(),lenght.end());
213  int32 max = *std::max_element(lenght.begin(),lenght.end());
214  // fix or drop ?
215  if (max - min < length_tolerance) {
216  if(mat.NumRows() != min) mat.Resize(min, mat.NumCols(), kCopyData);
217  if(targets.size() != min) targets.resize(min);
218  if(weights.Dim() != min) weights.Resize(min, kCopyData);
219  } else {
220  KALDI_WARN << utt << ", length mismatch of targets " << targets.size()
221  << " and features " << mat.NumRows();
222  num_other_error++;
223  continue;
224  }
225  }
226  // apply optional feature transform
227  nnet_transf.Feedforward(CuMatrix<BaseFloat>(mat), &feats_transf);
228 
229  // pass data to randomizers
230  KALDI_ASSERT(feats_transf.NumRows() == targets.size());
231  feature_randomizer.AddData(feats_transf);
232  targets_randomizer.AddData(targets);
233  weights_randomizer.AddData(weights);
234  num_done++;
235 
236  // report the speed
237  if (num_done % 5000 == 0) {
238  double time_now = time.Elapsed();
239  KALDI_VLOG(1) << "After " << num_done << " utterances: time elapsed = "
240  << time_now/60 << " min; processed " << total_frames/time_now
241  << " frames per second.";
242  }
243  }
244 
245  // randomize
246  if (!crossvalidate && randomize) {
247  const std::vector<int32>& mask = randomizer_mask.Generate(feature_randomizer.NumFrames());
248  feature_randomizer.Randomize(mask);
249  targets_randomizer.Randomize(mask);
250  weights_randomizer.Randomize(mask);
251  }
252 
253  // train with data from randomizers (using mini-batches)
254  for ( ; !feature_randomizer.Done(); feature_randomizer.Next(),
255  targets_randomizer.Next(),
256  weights_randomizer.Next()) {
257  // get block of feature/target pairs
258  const CuMatrixBase<BaseFloat>& nnet_in = feature_randomizer.Value();
259  const Posterior& nnet_tgt = targets_randomizer.Value();
260  const Vector<BaseFloat>& frm_weights = weights_randomizer.Value();
261 
262  // forward pass
263  nnet.Propagate(nnet_in, &nnet_out);
264 
265  // evaluate objective function we've chosen
266  if (objective_function == "xent") {
267  // gradients re-scaled by weights in Eval,
268  xent.Eval(frm_weights, nnet_out, nnet_tgt, &obj_diff);
269  } else if (objective_function == "mse") {
270  // gradients re-scaled by weights in Eval,
271  mse.Eval(frm_weights, nnet_out, nnet_tgt, &obj_diff);
272  } else if (0 == objective_function.compare(0,9,"multitask")) {
273  // gradients re-scaled by weights in Eval,
274  multitask.Eval(frm_weights, nnet_out, nnet_tgt, &obj_diff);
275  } else {
276  KALDI_ERR << "Unknown objective function code : " << objective_function;
277  }
278 
279  // backward pass
280  if (!crossvalidate) {
281  // backpropagate
282  nnet.Backpropagate(obj_diff, NULL);
283  }
284 
285  // 1st minibatch : show what happens in network
286  if (kaldi::g_kaldi_verbose_level >= 1 && total_frames == 0) { // vlog-1
287  KALDI_VLOG(1) << "### After " << total_frames << " frames,";
288  KALDI_VLOG(1) << nnet.InfoPropagate();
289  if (!crossvalidate) {
290  KALDI_VLOG(1) << nnet.InfoBackPropagate();
291  KALDI_VLOG(1) << nnet.InfoGradient();
292  }
293  }
294 
295  // monitor the NN training
296  if (kaldi::g_kaldi_verbose_level >= 2) { // vlog-2
297  if ((total_frames/25000) != ((total_frames+nnet_in.NumRows())/25000)) { // print every 25k frames
298  KALDI_VLOG(2) << "### After " << total_frames << " frames,";
299  KALDI_VLOG(2) << nnet.InfoPropagate();
300  if (!crossvalidate) {
301  KALDI_VLOG(2) << nnet.InfoGradient();
302  }
303  }
304  }
305 
306  total_frames += nnet_in.NumRows();
307  }
308  }
309 
310  // after last minibatch : show what happens in network
311  if (kaldi::g_kaldi_verbose_level >= 1) { // vlog-1
312  KALDI_VLOG(1) << "### After " << total_frames << " frames,";
313  KALDI_VLOG(1) << nnet.InfoPropagate();
314  if (!crossvalidate) {
315  KALDI_VLOG(1) << nnet.InfoBackPropagate();
316  KALDI_VLOG(1) << nnet.InfoGradient();
317  }
318  }
319 
320  if (!crossvalidate) {
321  nnet.Write(target_model_filename, binary);
322  }
323 
324  KALDI_LOG << "Done " << num_done << " files, " << num_no_tgt_mat
325  << " with no tgt_mats, " << num_other_error
326  << " with other errors. "
327  << "[" << (crossvalidate?"CROSS-VALIDATION":"TRAINING")
328  << ", " << (randomize?"RANDOMIZED":"NOT-RANDOMIZED")
329  << ", " << time.Elapsed()/60 << " min, fps" << total_frames/time.Elapsed()
330  << "]";
331 
332  if (objective_function == "xent") {
333  KALDI_LOG << xent.Report();
334  } else if (objective_function == "mse") {
335  KALDI_LOG << mse.Report();
336  } else if (0 == objective_function.compare(0,9,"multitask")) {
337  KALDI_LOG << multitask.Report();
338  } else {
339  KALDI_ERR << "Unknown objective function code : " << objective_function;
340  }
341 
342 #if HAVE_CUDA==1
343  CuDevice::Instantiate().PrintProfile();
344 #endif
345 
346  return 0;
347  } catch(const std::exception &e) {
348  std::cerr << e.what();
349  return -1;
350  }
