109 void WaveData::Read(std::istream &is) {
110  data_.Resize(0, 0); // clear the data.
111 
112  char tmp[5];
113  tmp[4] = '\0';
114  Read4ByteTag(is, &tmp[0]);
115  bool is_rifx = false;
116  if (!strcmp(tmp, "RIFX"))
117  is_rifx = true;
118  else if (strcmp(tmp, "RIFF"))
119  KALDI_ERR << "WaveData: expected RIFF or RIFX, got " << tmp;
120 
121 #ifdef __BIG_ENDIAN__
122  bool swap = !is_rifx;
123 #else
124  bool swap = is_rifx;
125 #endif
126 
127  uint32 riff_chunk_size = ReadUint32(is, swap);
128  Expect4ByteTag(is, "WAVE");
129 
130  uint32 riff_chunk_read = 0;
131  riff_chunk_read += 4; // WAVE included in riff_chunk_size.
132 
133  Expect4ByteTag(is, "fmt ");
134  uint32 subchunk1_size = ReadUint32(is, swap);
135  uint16 audio_format = ReadUint16(is, swap),
136  num_channels = ReadUint16(is, swap);
137  uint32 sample_rate = ReadUint32(is, swap),
138  byte_rate = ReadUint32(is, swap),
139  block_align = ReadUint16(is, swap),
140  bits_per_sample = ReadUint16(is, swap);
141  uint32 fmt_chunk_read = 16;
142 
143  if (audio_format == 1) {
144  if (subchunk1_size < 16) {
145  KALDI_ERR << "WaveData: expect PCM format data to have fmt chunk of at least size 16.";
146  }
147  } else if (audio_format == 0xFFFE) { // WAVE_FORMAT_EXTENSIBLE
148  uint16 extra_size = ReadUint16(is, swap);
149  if (subchunk1_size < 40 || extra_size < 22) {
150  KALDI_ERR << "WaveData: malformed WAVE_FORMAT_EXTENSIBLE format data.";
151  }
152  ReadUint16(is, swap); // Unused for PCM.
153  ReadUint32(is, swap); // Channel map: we do not care.
154  uint32 guid1 = ReadUint32(is, swap),
155  guid2 = ReadUint32(is, swap),
156  guid3 = ReadUint32(is, swap),
157  guid4 = ReadUint32(is, swap);
158  fmt_chunk_read = 40;
159 
160  // Support only KSDATAFORMAT_SUBTYPE_PCM for now. Interesting formats:
161  // ("00000001-0000-0010-8000-00aa00389b71", KSDATAFORMAT_SUBTYPE_PCM)
162  // ("00000003-0000-0010-8000-00aa00389b71", KSDATAFORMAT_SUBTYPE_IEEE_FLOAT)
163  // ("00000006-0000-0010-8000-00aa00389b71", KSDATAFORMAT_SUBTYPE_ALAW)
164  // ("00000007-0000-0010-8000-00aa00389b71", KSDATAFORMAT_SUBTYPE_MULAW)
165  if (guid1 != 0x00000001 || guid2 != 0x00100000 ||
166  guid3 != 0xAA000080 || guid4 != 0x719B3800) {
167  KALDI_ERR << "WaveData: unknown/unsupported WAVE_FORMAT_EXTENSIBLE format.";
168  }
169  } else {
170  KALDI_ERR << "WaveData: can read only PCM data, format id in file is: "
171  << audio_format;
172  }
173 
174  for (uint32 i = fmt_chunk_read; i < subchunk1_size; i++) is.get(); // use up extra data.
175 
176  if (num_channels <= 0)
177  KALDI_ERR << "WaveData: no channels present";
178  samp_freq_ = static_cast<BaseFloat>(sample_rate);
179  if (bits_per_sample != 8 && bits_per_sample != 16 && bits_per_sample != 32)
180  KALDI_ERR << "WaveData: bits_per_sample is " << bits_per_sample;
181  if (byte_rate != sample_rate * bits_per_sample/8 * num_channels)
182  KALDI_ERR << "Unexpected byte rate " << byte_rate << " vs. "
183  << sample_rate <<" * " << (bits_per_sample/8)
184  << " * " << num_channels;
185  if (block_align != num_channels * bits_per_sample/8)
186  KALDI_ERR << "Unexpected block_align: " << block_align << " vs. "
187  << num_channels << " * " << (bits_per_sample/8);
188 
189  riff_chunk_read += 8 + subchunk1_size;
190  // size of what we just read, 4 bytes for "fmt " + 4
191  // for subchunk1_size + subchunk1_size itself.
192 
193  // We support an optional "fact" chunk (which is useless but which
194  // we encountered), and then a single "data" chunk.
195 
196  char next_chunk_name[4];
197  Read4ByteTag(is, next_chunk_name);
198  riff_chunk_read += 4;
199 
200  // Skip any subchunks between "fmt" and "data". Usually there will
201  // be a single "fact" subchunk, but on Windows there can also be a
202  // "list" subchunk.
203  while (strncmp(next_chunk_name, "data", 4) != 0) {
204  // We will just ignore the data in these chunks.
205  uint32 chunk_sz = ReadUint32(is, swap);
206  if (chunk_sz != 4 && strncmp(next_chunk_name, "fact", 4) == 0)
207  KALDI_WARN << "Expected fact chunk to be 4 bytes long.";
208  for (uint32 i = 0; i < chunk_sz; i++)
209  is.get();
210  riff_chunk_read += 4 + chunk_sz; // for chunk_sz (4) + chunk contents (chunk-sz)
211 
212  // Now read the next chunk name.
213  Read4ByteTag(is, next_chunk_name);
214  riff_chunk_read += 4;
215  }
216 
217  if (strncmp(next_chunk_name, "data", 4))
218  KALDI_ERR << "WaveData: expected data chunk, got instead "
219  << next_chunk_name;
220 
221  uint32 data_chunk_size = ReadUint32(is, swap);
222  riff_chunk_read += 4;
223 
224  if (std::abs(static_cast<int64>(riff_chunk_read) +
225  static_cast<int64>(data_chunk_size) -
226  static_cast<int64>(riff_chunk_size)) > 1) {
227  // we allow the size to be off by one without warning, because there is a
228  // weirdness in the format of RIFF files that means that the input may
229  // sometimes be padded with 1 unused byte to make the total size even.
230  KALDI_WARN << "Expected " << riff_chunk_size << " bytes in RIFF chunk, but "
231  << "after first data block there will be " << riff_chunk_read
232  << " + " << data_chunk_size << " bytes "
233  << "(we do not support reading multiple data chunks).";
234  }
235 
236  std::vector<char*> data_pointer_vec;
237  std::vector<int> data_size_vec;
238  uint32 num_bytes_read = 0;
239  for (int32 remain_chunk_size = data_chunk_size; remain_chunk_size > 0;
240  remain_chunk_size -= kBlockSize) {
241  int32 this_block_size = remain_chunk_size;
242  if (kBlockSize < remain_chunk_size)
243  this_block_size = kBlockSize;
244  char *block_data_vec = new char[this_block_size];
245  is.read(block_data_vec, this_block_size);
246  num_bytes_read += is.gcount();
247  data_size_vec.push_back(is.gcount());
248  data_pointer_vec.push_back(block_data_vec);
249  if (num_bytes_read < this_block_size)
250  break;
251  }
252 
253  std::vector<char> chunk_data_vec(num_bytes_read);
254  uint32 data_address = 0;
255  for (int i = 0; i < data_pointer_vec.size(); i++) {
256  memcpy(&(chunk_data_vec[data_address]), data_pointer_vec[i],
257  data_size_vec[i]);
258  delete[] data_pointer_vec[i];
259  data_address += data_size_vec[i];
260  }
261 
262  char *data_ptr = &(chunk_data_vec[0]);
263  if (num_bytes_read == 0 && num_bytes_read != data_chunk_size) {
264  KALDI_ERR << "WaveData: failed to read data chunk (read no bytes)";
265  } else if (num_bytes_read != data_chunk_size) {
266  KALDI_ASSERT(num_bytes_read < data_chunk_size);
267  KALDI_WARN << "Read fewer bytes than specified in the header: "
268  << num_bytes_read << " < " << data_chunk_size;
269  }
270 
271  if (data_chunk_size == 0)
272  KALDI_ERR << "WaveData: empty file (no data)";
273 
274  uint32 num_samp = num_bytes_read / block_align;
275  data_.Resize(num_channels, num_samp);
276  for (uint32 i = 0; i < num_samp; i++) {
277  for (uint32 j = 0; j < num_channels; j++) {
278  switch (bits_per_sample) {
279  case 8:
280  data_(j, i) = *data_ptr;
281  data_ptr++;
282  break;
283  case 16:
284  {
285  int16 k = *reinterpret_cast<uint16*>(data_ptr);
286  if (swap)
287  KALDI_SWAP2(k);
288  data_(j, i) = k;
289  data_ptr += 2;
290  break;
291  }
292  case 32:
293  {
294  int32 k = *reinterpret_cast<uint32*>(data_ptr);
295  if (swap)
296  KALDI_SWAP4(k);
297  data_(j, i) = k;
298  data_ptr += 4;
299  break;
300  }
301  default:
302  KALDI_ERR << "bits per sample is " << bits_per_sample; // already checked this.
303  }
304  }
305  }
306 }

316 void WaveData::Write(std::ostream &os) const {
317  os << "RIFF";
318  if (data_.NumRows() == 0)
319  KALDI_ERR << "Error: attempting to write empty WAVE file";
320 
321  int32 num_chan = data_.NumRows(),
322  num_samp = data_.NumCols(),
323  bytes_per_samp = 2;
324 
325  int32 subchunk2size = (num_chan * num_samp * bytes_per_samp);
326  int32 chunk_size = 36 + subchunk2size;
327  WriteUint32(os, chunk_size);
328  os << "WAVE";
329  os << "fmt ";
330  WriteUint32(os, 16);
331  WriteUint16(os, 1);
332  WriteUint16(os, num_chan);
333  KALDI_ASSERT(samp_freq_ > 0);
334  WriteUint32(os, static_cast<int32>(samp_freq_));
335  WriteUint32(os, static_cast<int32>(samp_freq_) * num_chan * bytes_per_samp);
336  WriteUint16(os, num_chan * bytes_per_samp);
337  WriteUint16(os, 8 * bytes_per_samp);
338  os << "data";
339  WriteUint32(os, subchunk2size);
340 
341  const BaseFloat *data_ptr = data_.Data();
342  int32 stride = data_.Stride();
343 
344  int num_clipped = 0;
345  for (int32 i = 0; i < num_samp; i++) {
346  for (int32 j = 0; j < num_chan; j++) {
347  int32 elem = static_cast<int32>(trunc(data_ptr[j * stride + i]));
348  int16 elem_16 = static_cast<int16>(elem);
349  if (elem < std::numeric_limits<int16>::min()) {
350  elem_16 = std::numeric_limits<int16>::min();
351  ++num_clipped;
352  } else if (elem > std::numeric_limits<int16>::max()) {
353  elem_16 = std::numeric_limits<int16>::max();
354  ++num_clipped;
355  }
356 #ifdef __BIG_ENDIAN__
357  KALDI_SWAP2(elem_16);
358 #endif
359  os.write(reinterpret_cast<char*>(&elem_16), 2);
360  }
361  }
362  if (os.fail())
363  KALDI_ERR << "Error writing wave data to stream.";
364  if (num_clipped > 0)
365  KALDI_WARN << "WARNING: clipped " << num_clipped
366  << " samples out of total " << num_chan * num_samp
367  << ". Reduce volume?";
368 }
