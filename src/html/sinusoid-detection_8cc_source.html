29 // This function adds the given sinusoid to the signal, as:
30 // (*signal)(t) += amplitude * cos(2 pi freq/samp_freq t + phase).
31 void AddSinusoid(BaseFloat samp_freq,
32  const Sinusoid &sinusoid,
33  VectorBase<BaseFloat> *signal) {
34  // treat "factor" as a complex variable equal to exp(i * 2 pi freq / samp_freq); it's
35  // the factor by which we multiply on each frame.
36  BaseFloat factor_real = cos(M_2PI * sinusoid.freq / samp_freq),
37  factor_im = sin(M_2PI * sinusoid.freq / samp_freq);
38  BaseFloat *signal_data = signal->Data();
39  int32 dim = signal->Dim(), batch_size = 100;
40  // process frames in batches of size "batch_size", after which we recompute
41  // the starting point to prevent loss of accuracy due to drift.
42  for (int32 b = 0; b * batch_size < dim; b++) {
43  int32 t_offset = b * batch_size,
44  t_end = std::min(dim, t_offset + batch_size);
45  double phase = sinusoid.phase + M_2PI * t_offset * sinusoid.freq / samp_freq;
46  // treat x as a complex variable which initially is equal to amplitude * exp(i * phase),
47  // but which gets multiplied by "factor" on each frame.
48  BaseFloat x_real = sinusoid.amplitude * cos(phase),
49  x_im = sinusoid.amplitude * sin(phase);
50  for (int32 t = t_offset; t < t_end; t++) {
51  signal_data[t] += x_real;
52  ComplexMul(factor_real, factor_im, &x_real, &x_im); // x *= factor.
53  }
54  }
55 }

59 void SinusoidDetector::QuadraticMaximizeEqualSpaced(
60  BaseFloat y0, BaseFloat y1, BaseFloat y2,
61  BaseFloat *x_max, BaseFloat *y_max) {
62  // Let the function be y = a x^2 + b x + c, and
63  // suppose we have the values of y(0), y(1) and y(2).
64  // We have y0 = c, y1 = a + b + c, and y2 = 4a + 2b + c,
65  // so c = y0.
66  // Also, y2 - 2 y1 = 2a - c, so
67  // a = (y2 - 2 y1 + c) / 2, and
68  // b = y1 - a - c.
69  BaseFloat c = y0, a = y2 - 2 * y1 + c, b = y1 - a - c;
70  if (a >= 0) {
71  // The maximum of the function will occur at one of the end points.
72  if (y0 > y2) {
73  *x_max = 0;
74  *y_max = y0;
75  } else {
76  *x_max = 2;
77  *y_max = y2;
78  }
79  } else {
80  // derivative y' = 2a x + b. y' = 0 at x = -b / 2 a.
81  BaseFloat x = -b / (2.0 * a);
82  if (x <= 0.0) {
83  *x_max = 0;
84  *y_max = y0;
85  } else if (x >= 2.0) {
86  *x_max = 0;
87  *y_max = y2;
88  } else {
89  *x_max = x;
90  *y_max = a * x * x + b * x + c;
91  }
92  }
93 }

96 void SinusoidDetector::QuadraticMaximize(
97  BaseFloat x1, BaseFloat y0, BaseFloat y1, BaseFloat y2,
98  BaseFloat *x_max, BaseFloat *y_max) {
99  // Let the function be y = a x^2 + b x + c, and
100  // suppose we have the values of y(0), y(x1) and y(1),
101  // where 0 < x1 < 1.
102  // We have y0 = c, y1 = x1^2 a + x1 b + c, and y2 = a + b + c,
103  // so c = y0.
104  // Also, x1.y2 - y1 = a (x1 - x1^2) + (x1 - 1) c, so
105  // a = ( (x1 y2 - y1) - (x1 - 1) c) / (x1 - x1^2), and
106  // b = y2 - a - c.
107  BaseFloat c = y0,
108  a = (x1 * y2 - y1 - (x1 - 1.0) * c) / (x1 - x1*x1),
109  b = y2 - a - c;
110 
111  // TODO: remove these lines.
112  AssertEqual(y1, a * x1 * x1 + b * x1 + c);
113  AssertEqual(y2, a + b + c);
114 
115  if (a >= 0) {
116  // The maximum of the function will occur at one of the end points.
117  if (y0 > y2) {
118  *x_max = 0;
119  *y_max = y0;
120  } else {
121  *x_max = 1.0;
122  *y_max = y2;
123  }
124  } else {
125  // derivative y' = 2a x + b. y' = 0 at x = -b / 2 a.
126  BaseFloat x = -b / (2.0 * a);
127  if (x <= 0.0) {
128  *x_max = 0.0;
129  *y_max = y0;
130  } else if (x >= 1.0) {
131  *x_max = 1.0;
132  *y_max = y2;
133  } else {
134  *x_max = x;
135  *y_max = a * x * x + b * x + c;
136  }
137  }

141 BaseFloat SinusoidDetector::QuadraticInterpolate(
142  BaseFloat x1, BaseFloat y0, BaseFloat y1, BaseFloat y2,
143  BaseFloat x) {
144  // Let the function be y = a x^2 + b x + c, and
145  // suppose we have the values of y(0), y(x1) and y(1),
146  // where 0 < x1 < 1.
147  // We have y0 = c, y1 = x1^2 a + x1 b + c, and y2 = a + b + c,
148  // so c = y0.
149  // Also, x1.y2 - y1 = a (x1 - x1^2) + (x1 - 1) c, so
150  // a = ( (x1 y2 - y1) - (x1 - 1) c) / (x1 - x1^2), and
151  // b = y2 - a - c.
152  KALDI_ASSERT(x1 >= 0.0 && x1 <= 1.0);
153  if (x1 == 0.0) return y0;
154  else if (x1 == 1.0) return y2;
155 
156  BaseFloat c = y0,
157  a = (x1 * y2 - y1 - (x1 - 1.0) * c) / (x1 - x1*x1),
158  b = y2 - a - c;
159  return a * x * x + b * x + c;
160 }

248 BaseFloat SinusoidDetector::OptimizeFrequency(
249  const std::vector<InfoForBin> &info,
250  int32 *bin_out,
251  BaseFloat *offset_out) const {
252 
253  BaseFloat max_energy = 0.0;
254  *bin_out = -1;
255  int32 max_freq = num_samples_padded_ * 2;
256 
257  // For each bin, we consider the frequency range [bin, bin+1, bin+2],
258  // and if we have info for all those bins, do a quadratic interpolation to
259  // find the maximum within the range.
260  for (int32 bin = 0; bin + 2 <= max_freq; bin++) {
261  if (info[bin].valid && info[bin+1].valid && info[bin+2].valid) {
262  // First handle the left side of the bin.
263  BaseFloat best_x, best_y;
264  QuadraticMaximizeEqualSpaced(info[bin].energy, info[bin+1].energy,
265  info[bin+2].energy, &best_x, &best_y);
266  if (best_y > max_energy) {
267  max_energy = best_y;
268  if (best_x <= 1.0) {
269  *bin_out = bin;
270  *offset_out = best_x;
271  } else {
272  *bin_out = bin + 1;
273  *offset_out = best_x - 1;
274  }
275  }
276  }
277  }
278  return max_energy;
279 }

282 BaseFloat SinusoidDetector::DetectSinusoid(
283  BaseFloat min_energy,
284  const VectorBase<BaseFloat> &signal,
285  Sinusoid *sinusoid) {
286  if (signal(0) == 0.0 && signal.Norm(2.0) == 0.0)
287  return 0.0;
288  KALDI_ASSERT(signal.Dim() == num_samples_);
289  Vector<BaseFloat> fft(num_samples_padded_);
290  fft.Range(0, num_samples_).CopyFromVec(signal);
291  bool forward = true;
292  fft_.Compute(fft.Data(), forward);
293 
294  std::vector<InfoForBin> info;
295  ComputeCoarseInfo(fft, &info);
296  // we now have info for the "coarse" bins.
297 
298  // each element b of "bins" will be a multiple of 4: it's possible
299  // that the best frequency is in the range [b, b+4]
300  std::vector<int32> bins;
301  FindCandidateBins(min_energy, info, &bins);
302 
303  if (bins.empty())
304  return 0.0; // not enough energy in signal.
305 
306  for (size_t i = 0; i < bins.size(); i++) {
307  int32 bin = bins[i];
308  ComputeBinInfo(signal, bin, &(info[bin]));
309  }
310 
311  std::vector<int32> bins2;
312  FindCandidateBins2(min_energy, info, &bins2);
313 
314  for (size_t i = 0; i < bins2.size(); i++) {
315  int32 bin = bins2[i];
316  ComputeBinInfo(signal, bin, &(info[bin]));
317  }
318 
319  // compute energy for the predicted-optimum point, which will usually be
320  // between bins, with an offset.
321  int32 bin;
322  BaseFloat offset;
323 
324  BaseFloat opt_energy = OptimizeFrequency(info, &bin, &offset);
325 
326  if (opt_energy == 0.0)
327  return 0.0;
328 
329  BaseFloat max_freq = (bin + offset) * samp_freq_ / (num_samples_padded_ * 4);
330 
331  KALDI_VLOG(4) << "Best frequency based on interpolation is "
332  << max_freq << ", best energy is "
333  << opt_energy << ", bin is " << bin;
334 
335  OptimizedInfo final_info;
336 
337  FineOptimizeFrequency(signal, bin, offset, &info, &final_info);
338 
339  // the following while loop will rarely be accessed.
340  while (final_info.offset == 0.0 && bin > 0) {
341  bin--;
342  FineOptimizeFrequency(signal, bin, 1.0, &info, &final_info);
343  }
344 
345  // the following while loop will rarely be accessed.
346  while (final_info.offset == 1.0 && bin < num_samples_padded_ * 2) {
347  bin++;
348  FineOptimizeFrequency(signal, bin, 0.0, &info, &final_info);
349  }
350 
351  if (bin <= 1 || bin >= num_samples_padded_ * 2 - 2) {
352  // If we're in the lowest or next-to-lowest bin, or the highest or
353  // next-to-highest allowed bin (note, "bin" here is a range, and it can
354  // never have the value num_samples_padded_ * 2), we tend to get more
355  // estimation error than usual, so do another round of optimization.
356  FineOptimizeFrequency(signal, bin, final_info.offset, &info, &final_info);
357  }
358 
359  BaseFloat final_freq = (final_info.bin + final_info.offset) * samp_freq_ / (num_samples_padded_ * 4);
360  KALDI_VLOG(4) << "Final optimized info is: freq " << final_freq
361  << ", cos coeff " << final_info.cos_coeff << ", sin coeff "
362  << final_info.sin_coeff << ", energy " << final_info.energy;
363 
364  if (GetVerboseLevel() > 1)
365  SelfTest(signal, info, final_freq, final_info.energy);
366 
367  if (final_info.energy >= min_energy) {
368  sinusoid->amplitude = std::sqrt(final_info.cos_coeff * final_info.cos_coeff
369  + final_info.sin_coeff * final_info.sin_coeff);
370  sinusoid->freq = final_freq;
371  sinusoid->phase = -std::atan2(final_info.sin_coeff, final_info.cos_coeff);
372  KALDI_VLOG(4) << "Phase is " << sinusoid->phase << ", amplitude is "
373  << sinusoid->amplitude << ", freq is " << sinusoid->freq;
374  return final_info.energy;
375  } else {
376  return 0.0;
377  }
378 }

381 /*
382  This function computes, the original FFT bins, the amount of energy in
383  the signal that can be explained by a sinusoid at the corresponding frequency.
384 
385  Let f be the continuous-valued frequency.
386 
387  Define the vector C_f as
388  C_f = [ c_0, c_1 ... c_n ] where c_k = cos(2 pi k f / samp_freq). [obviously this notation depends on f].
389  and S_f the same thing with sin in place of cos.
390 
391  Let the signal, as a vector, be V.
392  We want to maximize the (positive) energy-difference:
393  ||V||^2 - || V - c C_f - s S_f ||^2
394  where c and s are the coefficients of C_f and S_f.
395  This quantity can be expanded as follows, where . means dot product.
396  \delta E = -c^2 C_f.C_f - s^2 S_f.S_f - 2 c s C_f.S_f + 2 c V.C_f + 2 s V.S_f.
397  which can be written as follows, where . means dot-product and ' means transpose:
398  \delta E = 2 [c s] v - [c s] M [c s]'
399  where M = [ C_f.C_f, C_f.S_f, C_f.S_f, S_f.S_f ],
400  and v = [V.C_f, V.S_f].
401  If M is invertible (i.e. for nonzero frequencies), this is maximized by
402  [c s] = M^-1 v
403  giving us the value.
404  \delta E = v' M^{-1} v.
405  We'll compute the inverse of M in advance, inside ComputeCoefficients(), using
406  the formula [a b;c d]^-1 = 1/(ad - bc) [d -b; -c a] For zero frequency and at the
407  Nyquist, M has the value [ a 0; 0 0 ], and we have the same type of expression
408  limited to the first dim of v, i.e. Minv = [ a^{-1} 0; 0 0 ], a kind of pseudo-inverse.
409  */

411 void SinusoidDetector::ComputeCoarseInfo(
412  const Vector<BaseFloat> &fft,
413  std::vector<InfoForBin> *info) const {
414  info->resize(num_samples_padded_ * 2 + 1); // 4 times resolution of FFT itself.
415 
416  const BaseFloat *fft_data = fft.Data();
417 
418  int32 num_bins = num_samples_padded_ / 2 + 1;
419  for (int32 k = 0; k < num_bins; k++) {
420  BaseFloat real, im;
421  if (k == 0) {
422  real = fft_data[0];
423  im = 0.0;
424  } else if (k == num_samples_padded_ / 2) {
425  real = fft_data[1];
426  im = 0.0;
427  } else {
428  real = fft_data[k * 2];
429  im = fft_data[k * 2 + 1];
430  }
431  // v1 and v2 are the two components of the vector v in the math above.
432  BaseFloat v1 = real, v2 = -im;
433  // Minv_'s row indexes correspond to frequencies with 4 times more
434  // resolution than the FFT bins.
435  const BaseFloat *Minv_data = Minv_.RowData(k * 4);
436  // The Matrix M^{-1} is of the form [a b; b d]
437  BaseFloat a = Minv_data[0], b = Minv_data[1], d = Minv_data[2];
438  // compute \delta E = v' M^{-1} v.
439  BaseFloat delta_e = v1 * v1 * a + v2 * v2 * d + 2 * v1 * v2 * b;
440  InfoForBin &this_info = (*info)[k * 4];
441  this_info.valid = true;
442  this_info.cos_dot = real;
443  this_info.sin_dot = -im;
444  this_info.energy = delta_e;
445  }
446 }

449 void SinusoidDetector::ComputeCoefficients() {
450  int32 num_samp = num_samples_;
451  int32 num_freq = num_samples_padded_ * 2 + 1;
452  cos_.Resize(num_freq, num_samp);
453  sin_.Resize(num_freq, num_samp);
454 
455  Vector<BaseFloat> cc(num_freq), cs(num_freq);
456  for (int32 k = 0; k < num_freq; k++) {
457  BaseFloat freq = k * samp_freq_ / (num_samples_padded_ * 4);
458  SubVector<BaseFloat> c(cos_, k), s(sin_, k);
459  CreateCosAndSin(samp_freq_, freq, &c, &s);
460  cc(k) = VecVec(c, c);
461  cs(k) = VecVec(c, s);
462  }
463 
464  M_.Resize(num_freq, 3, kUndefined);
465  Minv_.Resize(num_freq, 3, kUndefined);
466 
467  for (int32 k = 0; k < num_freq; k++) {
468  // Let the matrix M be [ a b; b d ]. [we don't write c because c == b].
469  // We want to compute Minv_.
470  BaseFloat a = cc(k), b = cs(k), d = num_samples_ - a;
471  M_(k, 0) = a;
472  M_(k, 1) = b;
473  M_(k, 2) = d;
474  if (k == 0 || k == num_freq - 1) {
475  // this is a special case; it's not really the inverse of M but it will
476 
477  // give us the expression we want; it's like an inverse in just one dimension.
478  Minv_(k, 0) = 1.0 / a;
479  Minv_(k, 1) = 0.0;
480  Minv_(k, 2) = 0.0;
481  } else {
482  BaseFloat inv_det = 1.0 / (a * d - b * b);
483  // check for NaN and inf.
484  KALDI_ASSERT(inv_det == inv_det && inv_det - inv_det == 0.0);
485  // use: [a b;c d]^-1 = 1/(ad - bc) [d -b; -c a], special case where c = b.
486  BaseFloat inv_a = d * inv_det, inv_b = -b * inv_det, inv_d = a * inv_det;
487  Minv_(k, 0) = inv_a;
488  Minv_(k, 1) = inv_b;
489  Minv_(k, 2) = inv_d;
490  }
491  }
492 }

495 // Does fine optimization of the frequency within this bin; returns the
496 // final energy, the optimized frequency, and the cos and sin coefficients.
497 void SinusoidDetector::FineOptimizeFrequency(
498  const VectorBase<BaseFloat> &signal,
499  int32 bin,
500  BaseFloat bin_offset,
501  std::vector<InfoForBin> *info_in,
502  OptimizedInfo *opt_info) const {
503  std::vector<InfoForBin> &info = *info_in;
504  if (!info[bin].valid) ComputeBinInfo(signal, bin, &(info[bin]));
505  if (!info[bin+1].valid) ComputeBinInfo(signal, bin+1, &(info[bin+1]));
506 
507  const BaseFloat epsilon = 0.02, delta = 0.001;
508 
509  // If the offset is very close to the edges of the bin, move it
510  // closer to the center. Otherwise we may have problems with the
511  // steps below. The initial offset is only used as a starting point
512  // anyway, so this won't affect the final value much.
513  if (bin_offset < epsilon)
514  bin_offset = epsilon;
515  if (bin_offset > 1.0 - epsilon)
516  bin_offset = 1.0 - epsilon;
517  KALDI_VLOG(4) << "Initial bin offset = " << bin_offset << ", bin = " << bin;
518 
519  // create cos and sin waves of the specified frequency.
520  BaseFloat freq = (bin + bin_offset) * samp_freq_ / (num_samples_padded_ * 4);
521  Vector<BaseFloat> c(num_samples_, kUndefined), s(num_samples_, kUndefined);
522  CreateCosAndSin(samp_freq_, freq, &c, &s);
523 
524  // these a, b and d values are the elements of the M matrix at this frequency
525  // "freq", i.e. the matrix M_f [ a b; b d ]. This will be invertible because
526  // we have ensured that the frequency is not too close to zero or the Nyquist.
527  BaseFloat a = VecVec(c, c), b = VecVec(c, s), d = num_samples_ - a;
528  BaseFloat inv_det = 1.0 / (a * d - b * b);
529  BaseFloat inv_a = d * inv_det, inv_b = -b * inv_det, inv_d = a * inv_det;
530 
531 
532  BaseFloat v1 = VecVec(c, signal), v2 = VecVec(s, signal);
533 
534  BaseFloat delta_e = v1 * v1 * inv_a + v2 * v2 * inv_d + 2 * v1 * v2 * inv_b;
535 
536  KALDI_VLOG(4) << "Actual energy-change at frequency " << freq << " is "
537  << delta_e;
538  // "freq" is frequency somewhere in the middle of the bin.
539 
540  BaseFloat final_offset, final_energy;
541  QuadraticMaximize(bin_offset, info[bin].energy, delta_e, info[bin+1].energy,
542  &final_offset, &final_energy);
543 
544  KALDI_VLOG(4) << "After further optimizing, offset was " << final_offset
545  << " giving freq "
546  << ((bin+final_offset) * samp_freq_ / (num_samples_padded_*4))
547  << ", with energy " << final_energy;
548 
549  // Use interpolation (using a quadratic function) to get the entries of the M matrix
550  // the the final, tuned frequency. Interpolation on M is better than M^{-1}, as its
551  // elements are much better behaved as the frequency varies.
552  const BaseFloat *M_left_data = M_.RowData(bin),
553  *M_right_data = M_.RowData(bin + 1);
554 
555  BaseFloat a_interp = QuadraticInterpolate(bin_offset, M_left_data[0], a, M_right_data[0],
556  final_offset);
557  BaseFloat b_interp = QuadraticInterpolate(bin_offset, M_left_data[1], b, M_right_data[1],
558  final_offset);
559  BaseFloat d_interp = QuadraticInterpolate(bin_offset, M_left_data[2], d, M_right_data[2],
560  final_offset);
561 
562  // Now get the inverse of the M matrix at the final point.
563  BaseFloat a_inv_interp, b_inv_interp, d_inv_interp;
564 
565  if ((bin == 0 && final_offset < delta) ||
566  (bin == num_samples_padded_ * 2 && final_offset > 1.0 - delta)) {
567  // If we're extremely close to zero or the Nyquist, we'll have trouble
568  // inverting M; just invert in the 1st dimension (only have a cos
569  // component).
570  a_inv_interp = 1.0 / a_interp;
571  b_inv_interp = 0.0;
572  d_inv_interp = 0.0;
573  } else {
574  BaseFloat inv_det = 1.0 / (a_interp * d_interp - b_interp * b_interp);
575  // check for NaN and inf.
576  KALDI_ASSERT(inv_det == inv_det && inv_det - inv_det == 0.0);
577  // use: [a b;c d]^-1 = 1/(ad - bc) [d -b; -c a], special case where c = b.
578  a_inv_interp = d_interp * inv_det;
579  b_inv_interp = -b_interp * inv_det;
580  d_inv_interp = a_interp * inv_det;
581  }
582 
583  BaseFloat v1_interp = QuadraticInterpolate(bin_offset, info[bin].cos_dot, v1,
584  info[bin+1].cos_dot, final_offset);
585  BaseFloat v2_interp = QuadraticInterpolate(bin_offset, info[bin].sin_dot, v2,
586  info[bin+1].sin_dot, final_offset);
587 
588  opt_info->bin = bin;
589  opt_info->offset = final_offset;
590  // Recompute the energy-reduction using the more accurate interpolated values of
591  // v1 and v2 (the dot-products of the cos and sin with the signal), and
592  // of M.
593  opt_info->energy = v1_interp * v1_interp * a_inv_interp +
594  v2_interp * v2_interp * d_inv_interp +
595  2 * v1_interp * v2_interp * b_inv_interp;
596  // Compute the coefficients of the cos and sin in the optimal sinusoid, as
597  // M^{-1} v.
598  opt_info->cos_coeff = a_inv_interp * v1_interp + b_inv_interp * v2_interp;
599  opt_info->sin_coeff = b_inv_interp * v1_interp + d_inv_interp * v2_interp;
600 }

652  int32 bin,
653  InfoForBin *info) const {
654  KALDI_ASSERT(!info->valid); // or wasted time.

748 void MultiSinusoidDetector::GetNextFrameOfSignal(Vector<BaseFloat> *frame) {
749  frame->Resize(samples_per_frame_subsampled_, kUndefined);
750 
751  int32 sample_offset = 0,
752  samples_needed = samples_per_frame_subsampled_;
753  while (samples_needed > 0 &&
754  !subsampled_signal_.empty()) {
755  Vector<BaseFloat> *src = subsampled_signal_.front();
756  int32 num_samples_avail = src->Dim() - samples_consumed_;
757  KALDI_ASSERT(num_samples_avail > 0);
758  int32 chunk_size = std::min(num_samples_avail,
759  samples_needed);
760  frame->Range(sample_offset, chunk_size).CopyFromVec(
761  src->Range(samples_consumed_, chunk_size));
762  sample_offset += chunk_size;
763  samples_needed -= chunk_size;
764  samples_consumed_ += chunk_size;
765  if (samples_consumed_ == src->Dim()) {
766  samples_consumed_ = 0;
767  delete src;
768  subsampled_signal_.pop_front();
769  }
770  }
771  if (samples_needed > 0) {
772  KALDI_ASSERT(waveform_finished_ && sample_offset > 0); // or code error.
773  frame->Range(sample_offset, samples_needed).SetZero();
774  }
775 }

778 void MultiSinusoidDetector::GetNextFrame(MultiSinusoidDetectorOutput *output) {
779  Vector<BaseFloat> frame;
780  GetNextFrameOfSignal(&frame);
781  // Mean subtraction
782  frame.Add(-1.0 * frame.Sum() / frame.Dim());
783  *output = MultiSinusoidDetectorOutput(); // reset to default.
784 
785  BaseFloat signal_energy = VecVec(frame, frame);
786  output->tot_energy = signal_energy / frame.Dim();
787  if (signal_energy == 0.0) return;
788 
789  // min_energy1 is the lowest energy we might care about.
790  BaseFloat min_energy1 = signal_energy *
791  std::min<BaseFloat>(config_.two_freq_min_total_energy * 0.5,
792  config_.one_freq_min_energy);
793 
794  Sinusoid sinusoid1;
795  BaseFloat energy1 = detector_.DetectSinusoid(min_energy1,
796  frame,
797  &sinusoid1);
798 
799  if (energy1 == 0.0) return; // Nothing detected.
800 
801  // we only care about the 2nd sinusoid if
802  // energy1 + energy2 >= signal_energy * two_freq_min_total_energy,
803  // and energy2 >= signal_energy * config.two_freq_min_energy.
804 
805  BaseFloat min_energy2 =
806  std::max(signal_energy * config_.two_freq_min_energy,
807  signal_energy * config_.two_freq_min_total_energy
808  - energy1);
809 
810  BaseFloat energy2;
811  Sinusoid sinusoid2;
812 
813  // If there is enough energy left in the signal that we could
814  // possibly detect a sinusoid of energy at least min_energy2...
815  if (min_energy2 <= signal_energy - energy1) {
816  sinusoid1.phase += M_PI; // reverse the phase.
817  AddSinusoid(config_.subsample_freq, sinusoid1, &frame);
818 
819 
820  energy2 = detector_.DetectSinusoid(min_energy2,
821  frame,
822  &sinusoid2);
823 
824  if (energy2 > energy1) {
825  // The following is just for our information, so we are aware
826  // when the sinusoid detection gives us the non-optimal sinusoid
827  // first.
828  BaseFloat factor = energy2 / energy1;
829  KALDI_VLOG(2) << "Second sinusoid greater than first by a factor of "
830  << factor << ". (This means sinusoid detection is not "
831  << " working ideally).";
832  }
833 
834  if (DetectedTwoFrequency(signal_energy,
835  sinusoid1, energy1,
836  sinusoid2, energy2,
837  output))
838  return;
839  } else {
840  energy2 = 0.0;
841  }
842  // We don't need the return status of the following; we just return anyway.
843  DetectedOneFrequency(signal_energy,
844  sinusoid1, energy1,
845  sinusoid2, energy2,
846  output);
847 }

850 bool MultiSinusoidDetector::DetectedTwoFrequency(
851  BaseFloat signal_energy,
852  const Sinusoid &sinusoid1,
853  BaseFloat energy1,
854  const Sinusoid &sinusoid2,
855  BaseFloat energy2,
856  MultiSinusoidDetectorOutput *output) {
857 
858  if (energy1 + energy2 >= signal_energy *
859  config_.two_freq_min_total_energy &&
860  std::min(energy1, energy2) >= signal_energy *
861  config_.two_freq_min_energy &&
862  std::min(sinusoid1.freq, sinusoid2.freq) >= config_.min_freq &&
863  std::max(sinusoid1.freq, sinusoid2.freq) <= config_.max_freq) {
864  output->freq1 = sinusoid1.freq;
865  output->energy1 = energy1 / signal_energy;
866  output->freq2 = sinusoid2.freq;
867  output->energy2 = energy2 / signal_energy;
868  if (output->freq1 > output->freq2) {
869  std::swap(output->freq1, output->freq2);
870  std::swap(output->energy1, output->energy2);
871  }
872  return true;
873  } else {
874  return false;
875  }

901  sinusoid2.freq <= config_.max_freq) {

903  output->energy1 = energy2 / signal_energy;

913 void DetectSinusoids(const VectorBase<BaseFloat> &signal,
914  MultiSinusoidDetector *detector,
915  Matrix<BaseFloat> *output) {
916  std::vector<MultiSinusoidDetectorOutput> output_vec;
917  detector->AcceptWaveform(signal);
918  detector->WaveformFinished();
919 
920  int32 safety_margin = 10, approx_num_frames = safety_margin +
921  (signal.Dim() / (detector->SamplingFrequency() *
922  detector->FrameShiftSecs()));
923  output_vec.reserve(approx_num_frames);
924  while (!detector->Done()) {
925  output_vec.resize(output_vec.size() + 1);
926  detector->GetNextFrame(&(output_vec.back()));
927  }
928  detector->Reset();
929  if (output_vec.empty()) {
930  output->Resize(0, 0);
931  } else {
932  output->Resize(output_vec.size(), 5, kUndefined);
933  for (int32 i = 0; i < output->NumRows(); i++) {
934  BaseFloat *row_data = output->RowData(i);
935  MultiSinusoidDetectorOutput &this_output = output_vec[i];
936  row_data[0] = this_output.tot_energy;
937  row_data[1] = this_output.freq1;
938  row_data[2] = this_output.energy1;
939  row_data[3] = this_output.freq2;
940  row_data[4] = this_output.energy2;
941  }
942  }
