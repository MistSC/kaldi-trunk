36 // Tests the Read() and Write() methods for the accumulators, in both binary
37 // and ASCII mode, as well as Check().
38 void TestSgmm2AccsIO(const AmSgmm2 &sgmm,
39  const kaldi::Matrix<BaseFloat> &feats) {
40  using namespace kaldi;
41  kaldi::SgmmUpdateFlagsType flags = kaldi::kSgmmAll & ~kSgmmSpeakerWeightProjections;
42  kaldi::Sgmm2PerFrameDerivedVars frame_vars;
43  kaldi::Sgmm2PerSpkDerivedVars empty;
44  frame_vars.Resize(sgmm.NumGauss(), sgmm.FeatureDim(),
45  sgmm.PhoneSpaceDim());
46  kaldi::Sgmm2GselectConfig sgmm_config;
47  sgmm_config.full_gmm_nbest = std::min(sgmm_config.full_gmm_nbest,
48  sgmm.NumGauss());
49  MleAmSgmm2Accs accs(sgmm, flags, true);
50  BaseFloat loglike = 0.0;
51 
52  for (int32 i = 0; i < feats.NumRows(); i++) {
53  std::vector<int32> gselect;
54  sgmm.GaussianSelection(sgmm_config, feats.Row(i), &gselect);
55  sgmm.ComputePerFrameVars(feats.Row(i), gselect, empty, &frame_vars);
56  loglike += accs.Accumulate(sgmm, frame_vars, 0, 1.0, &empty);
57  }
58  accs.CommitStatsForSpk(sgmm, empty);
59 
60  kaldi::MleAmSgmm2Options update_opts;
61  update_opts.check_v = (Rand()%2 == 0);
62  AmSgmm2 *sgmm1 = new AmSgmm2();
63  sgmm1->CopyFromSgmm2(sgmm, false, false);
64  kaldi::MleAmSgmm2Updater updater(update_opts);
65  updater.Update(accs, sgmm1, flags);
66  sgmm1->ComputeDerivedVars();
67  std::vector<int32> gselect;
68  Sgmm2LikelihoodCache like_cache(sgmm.NumGroups(), sgmm.NumPdfs());
69 
70  sgmm1->GaussianSelection(sgmm_config, feats.Row(0), &gselect);
71  sgmm1->ComputePerFrameVars(feats.Row(0), gselect, empty, &frame_vars);
72  BaseFloat loglike1 = sgmm1->LogLikelihood(frame_vars, 0, &like_cache, &empty);
73  delete sgmm1;
74 
75  // First, non-binary write
76  accs.Write(kaldi::Output("tmpf", false).Stream(), false);
77  bool binary_in;
78  MleAmSgmm2Accs *accs1 = new MleAmSgmm2Accs();
79  // Non-binary read
80  kaldi::Input ki1("tmpf", &binary_in);
81  accs1->Read(ki1.Stream(), binary_in, false);
82  accs1->Check(sgmm, true);
83  AmSgmm2 *sgmm2 = new AmSgmm2();
84  sgmm2->CopyFromSgmm2(sgmm, false, false);
85  updater.Update(*accs1, sgmm2, flags);
86  sgmm2->ComputeDerivedVars();
87  sgmm2->GaussianSelection(sgmm_config, feats.Row(0), &gselect);
88  sgmm2->ComputePerFrameVars(feats.Row(0), gselect, empty, &frame_vars);
89  Sgmm2LikelihoodCache like_cache2(sgmm2->NumGroups(), sgmm2->NumPdfs());
90  BaseFloat loglike2 = sgmm2->LogLikelihood(frame_vars, 0, &like_cache2, &empty);
91  kaldi::AssertEqual(loglike1, loglike2, 1e-4);
92  delete accs1;
93 
94  // Next, binary write
95  accs.Write(kaldi::Output("tmpfb", true).Stream(), true);
96  MleAmSgmm2Accs *accs2 = new MleAmSgmm2Accs();
97  // Binary read
98  kaldi::Input ki2("tmpfb", &binary_in);
99  accs2->Read(ki2.Stream(), binary_in, false);
100  accs2->Check(sgmm, true);
101  AmSgmm2 *sgmm3 = new AmSgmm2();
102  sgmm3->CopyFromSgmm2(sgmm, false, false);
103  updater.Update(*accs2, sgmm3, flags);
104  sgmm3->ComputeDerivedVars();
105  sgmm3->GaussianSelection(sgmm_config, feats.Row(0), &gselect);
106  sgmm3->ComputePerFrameVars(feats.Row(0), gselect, empty, &frame_vars);
107  Sgmm2LikelihoodCache like_cache3(sgmm3->NumGroups(), sgmm3->NumPdfs());
108  BaseFloat loglike3 = sgmm3->LogLikelihood(frame_vars, 0, &like_cache3, &empty);
109  kaldi::AssertEqual(loglike1, loglike3, 1e-6);
110 
111  // Testing the MAP update of M
112  update_opts.tau_map_M = 10;
113  update_opts.full_col_cov = (RandUniform() > 0.5)? true : false;
114  update_opts.full_row_cov = (RandUniform() > 0.5)? true : false;
115  kaldi::MleAmSgmm2Updater updater_map(update_opts);
116  sgmm3->CopyFromSgmm2(sgmm, false, false);
117  updater_map.Update(*accs2, sgmm3, flags);
118 
119  delete accs2;
120  delete sgmm2;
121  delete sgmm3;
122 
123  unlink("tmpf");
124  unlink("tmpfb");
125 }

127 void UnitTestEstimateSgmm2() {
128  int32 dim = 1 + kaldi::RandInt(0, 9); // random dimension of the gmm
129  int32 num_comp = 2 + kaldi::RandInt(0, 9); // random mixture size
130  kaldi::FullGmm full_gmm;
131  ut::InitRandFullGmm(dim, num_comp, &full_gmm);
132 
133  AmSgmm2 sgmm;
134  kaldi::Sgmm2GselectConfig config;
135  std::vector<int32> pdf2group;
136  pdf2group.push_back(0);
137  sgmm.InitializeFromFullGmm(full_gmm, pdf2group, dim+1, dim, false, 0.9); // TODO-- make this true!
138  sgmm.ComputeNormalizers();
139 
140  kaldi::Matrix<BaseFloat> feats;
141 
142  { // First, generate random means and variances
143  int32 num_feat_comp = num_comp + kaldi::RandInt(-num_comp/2, num_comp/2);
144  kaldi::Matrix<BaseFloat> means(num_feat_comp, dim),
145  vars(num_feat_comp, dim);
146  for (int32 m = 0; m < num_feat_comp; m++) {
147  for (int32 d= 0; d < dim; d++) {
148  means(m, d) = kaldi::RandGauss();
149  vars(m, d) = Exp(kaldi::RandGauss()) + 1e-2;
150  }
151  }
152  // Now generate random features with those means and variances.
153  feats.Resize(num_feat_comp * 200, dim);
154  for (int32 m = 0; m < num_feat_comp; m++) {
155  kaldi::SubMatrix<BaseFloat> tmp(feats, m*200, 200, 0, dim);
156  ut::RandDiagGaussFeatures(200, means.Row(m), vars.Row(m), &tmp);
157  }
158  }
159  sgmm.ComputeDerivedVars();
160  TestSgmm2AccsIO(sgmm, feats);
161 }
