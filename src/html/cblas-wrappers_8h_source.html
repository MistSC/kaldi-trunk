197 template<typename Real>
198 inline void Xgemv_sparsevec(MatrixTransposeType trans, MatrixIndexT num_rows,
199  MatrixIndexT num_cols, Real alpha, const Real *Mdata,
200  MatrixIndexT stride, const Real *xdata,
201  MatrixIndexT incX, Real beta, Real *ydata,
202  MatrixIndexT incY) {
203  if (trans == kNoTrans) {
204  if (beta != 1.0) cblas_Xscal(num_rows, beta, ydata, incY);
205  for (MatrixIndexT i = 0; i < num_cols; i++) {
206  Real x_i = xdata[i * incX];
207  if (x_i == 0.0) continue;
208  // Add to ydata, the i'th column of M, times alpha * x_i
209  cblas_Xaxpy(num_rows, x_i * alpha, Mdata + i, stride, ydata, incY);
210  }
211  } else {
212  if (beta != 1.0) cblas_Xscal(num_cols, beta, ydata, incY);
213  for (MatrixIndexT i = 0; i < num_rows; i++) {
214  Real x_i = xdata[i * incX];
215  if (x_i == 0.0) continue;
216  // Add to ydata, the i'th row of M, times alpha * x_i
217  cblas_Xaxpy(num_cols, x_i * alpha,
218  Mdata + (i * stride), 1, ydata, incY);
219  }
220  }
221 }

341 inline void mul_elements(
342  const MatrixIndexT dim,
343  const double *a,
344  double *b) { // does b *= a, elementwise.
345  double c1, c2, c3, c4;
346  MatrixIndexT i;
347  for (i = 0; i + 4 <= dim; i += 4) {
348  c1 = a[i] * b[i];
349  c2 = a[i+1] * b[i+1];
350  c3 = a[i+2] * b[i+2];
351  c4 = a[i+3] * b[i+3];
352  b[i] = c1;
353  b[i+1] = c2;
354  b[i+2] = c3;
355  b[i+3] = c4;
356  }
357  for (; i < dim; i++)
358  b[i] *= a[i];
359 }
