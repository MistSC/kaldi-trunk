95  bool Member() const {
96  // value1_ == value1_ tests for NaN.
97  // also test for no -inf, and either both or neither
98  // must be +inf, and
99  if (value1_ != value1_ || value2_ != value2_) return false; // NaN
100  if (value1_ == -numeric_limits<T>::infinity() ||
101  value2_ == -numeric_limits<T>::infinity()) return false; // -infty not allowed
102  if (value1_ == numeric_limits<T>::infinity() ||
103  value2_ == numeric_limits<T>::infinity()) {

111  LatticeWeightTpl Quantize(float delta = kDelta) const {
112  if (value1_+value2_ == -numeric_limits<T>::infinity()) {
113  return LatticeWeightTpl(-numeric_limits<T>::infinity(), -numeric_limits<T>::infinity());

244  function to apply to LexicographicWeight and the like, templated on

246  LexicographicWeight.

249 inline PairWeight<TropicalWeightTpl<FloatType>,

294 template<class FloatType>
295 inline int Compare (const LatticeWeightTpl<FloatType> &w1,
296  const LatticeWeightTpl<FloatType> &w2) {
297  FloatType f1 = w1.Value1() + w1.Value2(),
298  f2 = w2.Value1() + w2.Value2();
299  if (f1 < f2) { return 1; } // having smaller cost means you're larger
300  // in the semiring [higher probability]
301  else if (f1 > f2) { return -1; }
302  // mathematically we should be comparing (w1.value1_-w1.value2_ < w2.value1_-w2.value2_)
303  // in the next line, but add w1.value1_+w1.value2_ = w2.value1_+w2.value2_ to both sides and
304  // divide by two, and we get the simpler equivalent form w1.value1_ < w2.value1_.
305  else if (w1.Value1() < w2.Value1()) { return 1; }
306  else if (w1.Value1() > w2.Value1()) { return -1; }
307  else { return 0; }
308 }

548 // Compare is not part of the standard for weight types, but used internally for
549 // efficiency. The comparison here first compares the weight; if this is the
550 // same, it compares the string. The comparison on strings is: first compare
551 // the length, if this is the same, use lexicographical order. We can't just
552 // use the lexicographical order because this would destroy the distributive
553 // property of multiplication over addition, taking into account that addition
554 // uses Compare. The string element of "Compare" isn't super-important in
555 // practical terms; it's only needed to ensure that Plus always give consistent
556 // answers and is symmetric. It's essentially for tie-breaking, but we need to
557 // make sure all the semiring axioms are satisfied otherwise OpenFst might
558 // break.
559 
560 template<class WeightType, class IntType>
561 inline int Compare (const CompactLatticeWeightTpl<WeightType, IntType> &w1,
562  const CompactLatticeWeightTpl<WeightType, IntType> &w2) {
563  int c1 = Compare(w1.Weight(), w2.Weight());
564  if (c1 != 0) return c1;
565  int l1 = w1.String().size(), l2 = w2.String().size();
566  // Use opposite order on the string lengths, so that if the costs are the same,
567  // the shorter string wins.
568  if (l1 > l2) return -1;
569  else if (l1 < l2) return 1;
570  for(int i = 0; i < l1; i++) {
571  if (w1.String()[i] < w2.String()[i]) return -1;
572  else if (w1.String()[i] > w2.String()[i]) return 1;
573  }
574  return 0;
575 }

610 inline CompactLatticeWeightTpl<WeightType, IntType> Times(
611  const CompactLatticeWeightTpl<WeightType, IntType> &w1,
612  const CompactLatticeWeightTpl<WeightType, IntType> &w2) {
613  WeightType w = Times(w1.Weight(), w2.Weight());
614  if (w == WeightType::Zero()) {
615  return CompactLatticeWeightTpl<WeightType, IntType>::Zero();
616  // special case to ensure zero is unique
617  } else {

621  iter = std::copy(w1.String().begin(), w1.String().end(), iter); // returns end of first range.
622  std::copy(w2.String().begin(), w2.String().end(), iter);
623  return CompactLatticeWeightTpl<WeightType, IntType>(w, v);
624  }

627 template<class WeightType, class IntType>
628 inline CompactLatticeWeightTpl<WeightType, IntType> Divide(const CompactLatticeWeightTpl<WeightType, IntType> &w1,
629  const CompactLatticeWeightTpl<WeightType, IntType> &w2,
630  DivideType div = DIVIDE_ANY) {
631  if (w1.Weight() == WeightType::Zero()) {
632  if (w2.Weight() != WeightType::Zero()) {
633  return CompactLatticeWeightTpl<WeightType, IntType>::Zero();
634  } else {
635  std::cerr << "Division by zero [0/0] in CompactLatticeWeightTpl\n";
636  exit(1);
637  }
638  } else if (w2.Weight() == WeightType::Zero()) {
639  std::cerr << "Error: division by zero in CompactLatticeWeightTpl::Divide()";
640  exit(1);
641  }
642  WeightType w = Divide(w1.Weight(), w2.Weight());
643 
644  const vector<IntType> v1 = w1.String(), v2 = w2.String();
645  if (v2.size() > v1.size()) {
646  std::cerr << "Error in Divide (CompactLatticeWeightTpl): cannot divide, length mismatch.\n";
647  exit(1);
648  }
649  typename vector<IntType>::const_iterator v1b = v1.begin(),
650  v1e = v1.end(), v2b = v2.begin(), v2e = v2.end();
651  if (div == DIVIDE_LEFT) {
652  if (!std::equal(v2b, v2e, v1b)) { // v2 must be identical to first part of v1.
653  std::cerr << "Error in Divide (CompactLatticeWeighTpl): cannot divide, data mismatch.\n";
654  exit(1);
655  }
656  return CompactLatticeWeightTpl<WeightType, IntType>(
657  w, vector<IntType>(v1b+(v2e-v2b), v1e)); // return last part of v1.
658  } else if (div == DIVIDE_RIGHT) {
659  if (!std::equal(v2b, v2e, v1e-(v2e-v2b))) { // v2 must be identical to last part of v1.
660  std::cerr << "Error in Divide (CompactLatticeWeighTpl): cannot divide, data mismatch.\n";
661  exit(1);
662  }
663  return CompactLatticeWeightTpl<WeightType, IntType>(
664  w, vector<IntType>(v1b, v1e-(v2e-v2b))); // return first part of v1.
665 
666  } else {
667  std::cerr << "Cannot divide CompactLatticeWeightTpl with DIVIDE_ANY.\n";
668  exit(1);
669  }
670  return CompactLatticeWeightTpl<WeightType,IntType>::Zero(); // keep compiler happy.
671 }

687 inline istream &operator >>(istream &strm, CompactLatticeWeightTpl<WeightType, IntType> &w) {
688  std::string s;
689  strm >> s;
690  if (strm.fail()) {
691  return strm;
692  }

694  size_t pos = s.find_last_of(FLAGS_fst_weight_separator); // normally ","
695  if (pos == std::string::npos) {
696  strm.clear(std::ios::badbit);
697  return strm;
698  }
699  // get parts of str before and after the separator (default: ',');
700  std::string s1(s, 0, pos), s2(s, pos+1);
701  std::istringstream strm1(s1);
702  WeightType weight;
703  strm1 >> weight;
704  w.SetWeight(weight);
705  if (strm1.fail() || !strm1.eof()) {
706  strm.clear(std::ios::badbit);
707  return strm;
708  }
709  // read string part.
710  vector<IntType> string;
711  const char *c = s2.c_str();
712  while(*c != '\0') {
713  if (*c == kStringSeparator) // '_'
714  c++;
715  char *c2;
716  long int i = strtol(c, &c2, 10);
717  if (c2 == c || static_cast<long int>(static_cast<IntType>(i)) != i) {
718  strm.clear(std::ios::badbit);
719  return strm;
720  }
721  c = c2;
722  string.push_back(static_cast<IntType>(i));
723  }
724  w.SetString(string);
725  return strm;
726 }
