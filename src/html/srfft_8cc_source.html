48 template<typename Real>
49 void SplitRadixComplexFft<Real>::ComputeTables() {
50  MatrixIndexT imax, lg2, i, j;
51  MatrixIndexT m, m2, m4, m8, nel, n;
52  Real *cn, *spcn, *smcn, *c3n, *spc3n, *smc3n;
53  Real ang, c, s;
54 
55  lg2 = logn_ >> 1;
56  if (logn_ & 1) lg2++;
57  brseed_ = new MatrixIndexT[1 << lg2];
58  brseed_[0] = 0;
59  brseed_[1] = 1;
60  for (j = 2; j <= lg2; j++) {
61  imax = 1 << (j - 1);
62  for (i = 0; i < imax; i++) {
63  brseed_[i] <<= 1;
64  brseed_[i + imax] = brseed_[i] + 1;
65  }
66  }
67 
68  if (logn_ < 4) {
69  tab_ = NULL;
70  } else {
71  tab_ = new Real* [logn_-3];
72  for (i = logn_; i>=4 ; i--) {
73  /* Compute a few constants */
74  m = 1 << i; m2 = m / 2; m4 = m2 / 2; m8 = m4 /2;
75 
76  /* Allocate memory for tables */
77  nel = m4 - 2;
78 
79  tab_[i-4] = new Real[6*nel];
80 
81  /* Initialize pointers */
82  cn = tab_[i-4]; spcn = cn + nel; smcn = spcn + nel;
83  c3n = smcn + nel; spc3n = c3n + nel; smc3n = spc3n + nel;
84 
85  /* Compute tables */
86  for (n = 1; n < m4; n++) {
87  if (n == m8) continue;
88  ang = n * M_2PI / m;
89  c = std::cos(ang); s = std::sin(ang);
90  *cn++ = c; *spcn++ = - (s + c); *smcn++ = s - c;
91  ang = 3 * n * M_2PI / m;
92  c = std::cos(ang); s = std::sin(ang);
93  *c3n++ = c; *spc3n++ = - (s + c); *smc3n++ = s - c;
94  }
95  }
96  }
97 }

110 void SplitRadixComplexFft<Real>::Compute(Real *xr, Real *xi, bool forward) const {
111  if (!forward) { // reverse real and imaginary parts for complex FFT.
112  Real *tmp = xr;
113  xr = xi;
114  xi = tmp;
115  }
116  ComputeRecursive(xr, xi, logn_);
117  if (logn_ > 1) {
118  BitReversePermute(xr, logn_);
119  BitReversePermute(xi, logn_);
120  }

123 template<typename Real>
124 void SplitRadixComplexFft<Real>::Compute(Real *x, bool forward,
125  std::vector<Real> *temp_buffer) const {
126  KALDI_ASSERT(temp_buffer != NULL);
127  if (temp_buffer->size() != N_)
128  temp_buffer->resize(N_);
129  Real *temp_ptr = &((*temp_buffer)[0]);
130  for (MatrixIndexT i = 0; i < N_; i++) {
131  x[i] = x[i * 2]; // put the real part in the first half of x.
132  temp_ptr[i] = x[i * 2 + 1]; // put the imaginary part in temp_buffer.
133  }
134  // copy the imaginary part back to the second half of x.
135  memcpy(static_cast<void*>(x + N_),
136  static_cast<void*>(temp_ptr),
137  sizeof(Real) * N_);
138 
139  Compute(x, x + N_, forward);
140  // Now change the format back to interleaved.
141  memcpy(static_cast<void*>(temp_ptr),
142  static_cast<void*>(x + N_),
143  sizeof(Real) * N_);
144  for (MatrixIndexT i = N_-1; i > 0; i--) { // don't include 0,
145  // in case MatrixIndexT is unsigned, the loop would not terminate.
146  // Treat it as a special case.
147  x[i*2] = x[i];
148  x[i*2 + 1] = temp_ptr[i];
149  }
150  x[1] = temp_ptr[0]; // special case of i = 0.
151 }

158 template<typename Real>
159 void SplitRadixComplexFft<Real>::BitReversePermute(Real *x, MatrixIndexT logn) const {
160  MatrixIndexT i, j, lg2, n;
161  MatrixIndexT off, fj, gno, *brp;
162  Real tmp, *xp, *xq;
163 
164  lg2 = logn >> 1;
165  n = 1 << lg2;
166  if (logn & 1) lg2++;
167 
168  /* Unshuffling loop */
169  for (off = 1; off < n; off++) {
170  fj = n * brseed_[off]; i = off; j = fj;
171  tmp = x[i]; x[i] = x[j]; x[j] = tmp;
172  xp = &x[i];
173  brp = &(brseed_[1]);
174  for (gno = 1; gno < brseed_[off]; gno++) {
175  xp += n;
176  j = fj + *brp++;
177  xq = x + j;
178  tmp = *xp; *xp = *xq; *xq = tmp;
179  }
180  }
181 }

184 template<typename Real>
185 void SplitRadixComplexFft<Real>::ComputeRecursive(Real *xr, Real *xi, MatrixIndexT logn) const {
186 
187  MatrixIndexT m, m2, m4, m8, nel, n;
188  Real *xr1, *xr2, *xi1, *xi2;
189  Real *cn, *spcn, *smcn, *c3n, *spc3n, *smc3n;
190  Real tmp1, tmp2;
191  Real sqhalf = M_SQRT1_2;
192 
193  /* Check range of logn */
194  if (logn < 0)
195  KALDI_ERR << "Error: logn is out of bounds in SRFFT";
196 
197  /* Compute trivial cases */
198  if (logn < 3) {
199  if (logn == 2) { /* length m = 4 */
200  xr2 = xr + 2;
201  xi2 = xi + 2;
202  tmp1 = *xr + *xr2;
203  *xr2 = *xr - *xr2;
204  *xr = tmp1;
205  tmp1 = *xi + *xi2;
206  *xi2 = *xi - *xi2;
207  *xi = tmp1;
208  xr1 = xr + 1;
209  xi1 = xi + 1;
210  xr2++;
211  xi2++;
212  tmp1 = *xr1 + *xr2;
213  *xr2 = *xr1 - *xr2;
214  *xr1 = tmp1;
215  tmp1 = *xi1 + *xi2;
216  *xi2 = *xi1 - *xi2;
217  *xi1 = tmp1;
218  xr2 = xr + 1;
219  xi2 = xi + 1;
220  tmp1 = *xr + *xr2;
221  *xr2 = *xr - *xr2;
222  *xr = tmp1;
223  tmp1 = *xi + *xi2;
224  *xi2 = *xi - *xi2;
225  *xi = tmp1;
226  xr1 = xr + 2;
227  xi1 = xi + 2;
228  xr2 = xr + 3;
229  xi2 = xi + 3;
230  tmp1 = *xr1 + *xi2;
231  tmp2 = *xi1 + *xr2;
232  *xi1 = *xi1 - *xr2;
233  *xr2 = *xr1 - *xi2;
234  *xr1 = tmp1;
235  *xi2 = tmp2;
236  return;
237  }
238  else if (logn == 1) { /* length m = 2 */
239  xr2 = xr + 1;
240  xi2 = xi + 1;
241  tmp1 = *xr + *xr2;
242  *xr2 = *xr - *xr2;
243  *xr = tmp1;
244  tmp1 = *xi + *xi2;
245  *xi2 = *xi - *xi2;
246  *xi = tmp1;
247  return;
248  }
249  else if (logn == 0) return; /* length m = 1 */
250  }
251 
252  /* Compute a few constants */
253  m = 1 << logn; m2 = m / 2; m4 = m2 / 2; m8 = m4 /2;
254 
255 
256  /* Step 1 */
257  xr1 = xr; xr2 = xr1 + m2;
258  xi1 = xi; xi2 = xi1 + m2;
259  for (n = 0; n < m2; n++) {
260  tmp1 = *xr1 + *xr2;
261  *xr2 = *xr1 - *xr2;
262  xr2++;
263  *xr1++ = tmp1;
264  tmp2 = *xi1 + *xi2;
265  *xi2 = *xi1 - *xi2;
266  xi2++;
267  *xi1++ = tmp2;
268  }
269 
270  /* Step 2 */
271  xr1 = xr + m2; xr2 = xr1 + m4;
272  xi1 = xi + m2; xi2 = xi1 + m4;
273  for (n = 0; n < m4; n++) {
274  tmp1 = *xr1 + *xi2;
275  tmp2 = *xi1 + *xr2;
276  *xi1 = *xi1 - *xr2;
277  xi1++;
278  *xr2++ = *xr1 - *xi2;
279  *xr1++ = tmp1;
280  *xi2++ = tmp2;
281  // xr1++; xr2++; xi1++; xi2++;
282  }
283 
284  /* Steps 3 & 4 */
285  xr1 = xr + m2; xr2 = xr1 + m4;
286  xi1 = xi + m2; xi2 = xi1 + m4;
287  if (logn >= 4) {
288  nel = m4 - 2;
289  cn = tab_[logn-4]; spcn = cn + nel; smcn = spcn + nel;
290  c3n = smcn + nel; spc3n = c3n + nel; smc3n = spc3n + nel;
291  }
292  xr1++; xr2++; xi1++; xi2++;
293  // xr1++; xi1++;
294  for (n = 1; n < m4; n++) {
295  if (n == m8) {
296  tmp1 = sqhalf * (*xr1 + *xi1);
297  *xi1 = sqhalf * (*xi1 - *xr1);
298  *xr1 = tmp1;
299  tmp2 = sqhalf * (*xi2 - *xr2);
300  *xi2 = -sqhalf * (*xr2 + *xi2);
301  *xr2 = tmp2;
302  } else {
303  tmp2 = *cn++ * (*xr1 + *xi1);
304  tmp1 = *spcn++ * *xr1 + tmp2;
305  *xr1 = *smcn++ * *xi1 + tmp2;
306  *xi1 = tmp1;
307  tmp2 = *c3n++ * (*xr2 + *xi2);
308  tmp1 = *spc3n++ * *xr2 + tmp2;
309  *xr2 = *smc3n++ * *xi2 + tmp2;
310  *xi2 = tmp1;
311  }
312  xr1++; xr2++; xi1++; xi2++;
313  }
314 
315  /* Call ssrec again with half DFT length */
316  ComputeRecursive(xr, xi, logn-1);
317 
318  /* Call ssrec again twice with one quarter DFT length.
319  Constants have to be recomputed, because they are static! */
320  // m = 1 << logn; m2 = m / 2;
321  ComputeRecursive(xr + m2, xi + m2, logn - 2);
322  // m = 1 << logn;
323  m4 = 3 * (m / 4);
324  ComputeRecursive(xr + m4, xi + m4, logn - 2);
325 }

334 // This code is mostly the same as the RealFft function. It would be
335 // possible to replace it with more efficient code from Rico's book.
336 template<typename Real>
337 void SplitRadixRealFft<Real>::Compute(Real *data, bool forward,
338  std::vector<Real> *temp_buffer) const {
339  MatrixIndexT N = N_, N2 = N/2;
340  KALDI_ASSERT(N%2 == 0);
341  if (forward) // call to base class
342  SplitRadixComplexFft<Real>::Compute(data, true, temp_buffer);
343 
344  Real rootN_re, rootN_im; // exp(-2pi/N), forward; exp(2pi/N), backward
345  int forward_sign = forward ? -1 : 1;
346  ComplexImExp(static_cast<Real>(M_2PI/N *forward_sign), &rootN_re, &rootN_im);
347  Real kN_re = -forward_sign, kN_im = 0.0; // exp(-2pik/N), forward; exp(-2pik/N), backward
348  // kN starts out as 1.0 for forward algorithm but -1.0 for backward.
349  for (MatrixIndexT k = 1; 2*k <= N2; k++) {
350  ComplexMul(rootN_re, rootN_im, &kN_re, &kN_im);
351 
352  Real Ck_re, Ck_im, Dk_re, Dk_im;
353  // C_k = 1/2 (B_k + B_{N/2 - k}^*) :
354  Ck_re = 0.5 * (data[2*k] + data[N - 2*k]);
355  Ck_im = 0.5 * (data[2*k + 1] - data[N - 2*k + 1]);
356  // re(D_k)= 1/2 (im(B_k) + im(B_{N/2-k})):
357  Dk_re = 0.5 * (data[2*k + 1] + data[N - 2*k + 1]);
358  // im(D_k) = -1/2 (re(B_k) - re(B_{N/2-k}))
359  Dk_im =-0.5 * (data[2*k] - data[N - 2*k]);
360  // A_k = C_k + 1^(k/N) D_k:
361  data[2*k] = Ck_re; // A_k <-- C_k
362  data[2*k+1] = Ck_im;
363  // now A_k += D_k 1^(k/N)
364  ComplexAddProduct(Dk_re, Dk_im, kN_re, kN_im, &(data[2*k]), &(data[2*k+1]));
365 
366  MatrixIndexT kdash = N2 - k;
367  if (kdash != k) {
368  // Next we handle the index k' = N/2 - k. This is necessary
369  // to do now, to avoid invalidating data that we will later need.
370  // The quantities C_{k'} and D_{k'} are just the conjugates of C_k
371  // and D_k, so the equations are simple modifications of the above,
372  // replacing Ck_im and Dk_im with their negatives.
373  data[2*kdash] = Ck_re; // A_k' <-- C_k'
374  data[2*kdash+1] = -Ck_im;
375  // now A_k' += D_k' 1^(k'/N)
376  // We use 1^(k'/N) = 1^((N/2 - k) / N) = 1^(1/2) 1^(-k/N) = -1 * (1^(k/N))^*
377  // so it's the same as 1^(k/N) but with the real part negated.
378  ComplexAddProduct(Dk_re, -Dk_im, -kN_re, kN_im, &(data[2*kdash]), &(data[2*kdash+1]));
379  }
380  }
381 
382  { // Now handle k = 0.
383  // In simple terms: after the complex fft, data[0] becomes the sum of real
384  // parts input[0], input[2]... and data[1] becomes the sum of imaginary
385  // pats input[1], input[3]...
386  // "zeroth" [A_0] is just the sum of input[0]+input[1]+input[2]..
387  // and "n2th" [A_{N/2}] is input[0]-input[1]+input[2]... .
388  Real zeroth = data[0] + data[1],
389  n2th = data[0] - data[1];
390  data[0] = zeroth;
391  data[1] = n2th;
392  if (!forward) {
393  data[0] /= 2;
394  data[1] /= 2;
395  }
396  }
397  if (!forward) { // call to base class
398  SplitRadixComplexFft<Real>::Compute(data, false, temp_buffer);
399  for (MatrixIndexT i = 0; i < N; i++)
400  data[i] *= 2.0;
401  // This is so we get a factor of N increase, rather than N/2 which we would
402  // otherwise get from [ComplexFft, forward] + [ComplexFft, backward] in dimension N/2.
403  // It's for consistency with our normal FFT convensions.
404  }
405 }
