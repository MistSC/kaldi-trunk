102  inline Real operator() (MatrixIndexT r, MatrixIndexT c) const {
103  // if column is less than row, then swap these as matrix is stored
104  // as upper-triangular... only allowed for const matrix object.
105  if (static_cast<UnsignedMatrixIndexT>(c) >
106  static_cast<UnsignedMatrixIndexT>(r))
107  std::swap(c, r);
108  // c<=r now so don't have to check c.
109  KALDI_ASSERT(static_cast<UnsignedMatrixIndexT>(r) <
110  static_cast<UnsignedMatrixIndexT>(this->num_rows_));
111  return *(this->data_ + (r*(r+1)) / 2 + c);
112  // Duplicating code from PackedMatrix.h
113  }

330  // LimitCond:
331  // Limits the condition of symmetric positive semidefinite matrix to
332  // a specified value
333  // by flooring all eigenvalues to a positive number which is some multiple
334  // of the largest one (or zero if there are no positive eigenvalues).
335  // Takes the condition number we are willing to accept, and floors
336  // eigenvalues to the largest eigenvalue divided by this.
337  // Returns #eigs floored or already equal to the floor.
338  // Throws exception if input is not positive definite.
339  // returns #floored.
340  MatrixIndexT LimitCond(Real maxCond = 1.0e+5, bool invert = false);
341

344  SpMatrix<double> dmat(*this);
