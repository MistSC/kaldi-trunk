64 void NnetCombiner::AcceptNnet(const Nnet &nnet) {
65  KALDI_ASSERT(num_nnets_provided_ < num_real_input_nnets_ &&
66  "You called AcceptNnet too many times.");
67  int32 num_effective_nnets = nnet_params_.NumRows();
68  if (num_effective_nnets == num_real_input_nnets_) {
69  SubVector<BaseFloat> this_params(nnet_params_, num_nnets_provided_);
70  VectorizeNnet(nnet, &this_params);
71  tot_input_weighting_(num_nnets_provided_) += 1.0;
72  } else {
73  // this_index is a kind of warped index, mapping the range
74  // 0 ... num_real_inputs_nnets_ - 1 onto the range
75  // 0 ... num_effective_nnets - 1. View the index as falling in
76  // between two integer indexes and determining weighting factors.
77  // we could view this as triangular bins.
78  BaseFloat this_index = num_nnets_provided_ * (num_effective_nnets - 1)
79  / static_cast<BaseFloat>(num_real_input_nnets_ - 1);
80  int32 lower_index = std::floor(this_index),
81  upper_index = lower_index + 1;
82  BaseFloat remaining_part = this_index - lower_index,
83  lower_weight = 1.0 - remaining_part,
84  upper_weight = remaining_part;
85  KALDI_ASSERT(lower_index >= 0 && upper_index <= num_effective_nnets &&
86  lower_weight >= 0.0 && upper_weight >= 0.0 &&
87  lower_weight <= 1.0 && upper_weight <= 1.0);
88  Vector<BaseFloat> vec(nnet_params_.NumCols(), kUndefined);
89  VectorizeNnet(nnet, &vec);
90  nnet_params_.Row(lower_index).AddVec(lower_weight, vec);
91  tot_input_weighting_(lower_index) += lower_weight;
92  if (upper_index == num_effective_nnets) {
93  KALDI_ASSERT(upper_weight < 0.1);
94  } else {
95  nnet_params_.Row(upper_index).AddVec(upper_weight, vec);
96  tot_input_weighting_(upper_index) += upper_weight;
97  }
98  }
99  num_nnets_provided_++;
100 }

102 void NnetCombiner::FinishPreprocessingInput() {
103  KALDI_ASSERT(num_nnets_provided_ == num_real_input_nnets_ &&
104  "You did not call AcceptInput() enough times.");
105  int32 num_effective_nnets = nnet_params_.NumRows();
106  for (int32 i = 0; i < num_effective_nnets; i++) {
107  BaseFloat tot_weight = tot_input_weighting_(i);
108  KALDI_ASSERT(tot_weight > 0.0); // Or would be a coding error.
109  // Rescale so this row is like a weighted average instead of
110  // a weighted sum.
111  if (tot_weight != 1.0)
112  nnet_params_.Row(i).Scale(1.0 / tot_weight);
113  }
114 }

116 void NnetCombiner::Combine() {
117  FinishPreprocessingInput();
118 
119  if (!SelfTestDerivatives()) {
120  KALDI_LOG << "Self-testing model derivatives since parameter-derivatives "
121  "self-test failed.";
122  SelfTestModelDerivatives();
123  }
124 
125  int32 dim = ParameterDim();
126  LbfgsOptions lbfgs_options;
127  lbfgs_options.minimize = false; // We're maximizing.
128  lbfgs_options.m = dim; // Store the same number of vectors as the dimension
129  // itself, so this is BFGS.
130  lbfgs_options.first_step_impr = config_.initial_impr;
131 
132  Vector<BaseFloat> params(dim), deriv(dim);
133  BaseFloat objf, initial_objf;
134  GetInitialParameters(&params);
135 
136 
137  OptimizeLbfgs<BaseFloat> lbfgs(params, lbfgs_options);
138 
139  for (int32 i = 0; i < config_.num_iters; i++) {
140  params.CopyFromVec(lbfgs.GetProposedValue());
141  objf = ComputeObjfAndDerivFromParameters(params, &deriv);
142  KALDI_VLOG(2) << "Iteration " << i << " params = " << params
143  << ", objf = " << objf << ", deriv = " << deriv;
144  if (i == 0) initial_objf = objf;
145  lbfgs.DoStep(objf, deriv);
146  }
147 
148  KALDI_LOG << "Combining nnets, objective function changed from "
149  << initial_objf << " to " << objf;
150 
151  // must recompute nnet_ if "params" is not exactly equal to the
152  // final params that LB
153  Vector<BaseFloat> final_params(dim);
154  final_params.CopyFromVec(lbfgs.GetValue(&objf));
155  if (!params.ApproxEqual(final_params, 0.0)) {
156  // the following call makes sure that nnet_ corresponds to the parameters
157  // in "params".
158  ComputeObjfAndDerivFromParameters(final_params, &deriv);
159  }
160  PrintParams(final_params);
161 }

213 bool NnetCombiner::SelfTestDerivatives() {
214  int32 num_tests = 2; // more properly, this is the number of dimensions in a
215  // single test.
216  BaseFloat delta = 0.001;
217  int32 dim = ParameterDim();
218 
219  Vector<BaseFloat> params(dim), deriv(dim);
220  Vector<BaseFloat> predicted_changes(num_tests),
221  observed_changes(num_tests);
222 
223  GetInitialParameters(&params);
224  BaseFloat initial_objf = ComputeObjfAndDerivFromParameters(params,
225  &deriv);
226  for (int32 i = 0; i < num_tests; i++) {
227  Vector<BaseFloat> new_deriv(dim), offset(dim), new_params(params);
228  offset.SetRandn();
229  new_params.AddVec(delta, offset);
230  BaseFloat new_objf = ComputeObjfAndDerivFromParameters(new_params,
231  &new_deriv);
232  // for predicted changes, interpolate old and new derivs.
233  predicted_changes(i) =
234  0.5 * VecVec(new_params, deriv) - 0.5 * VecVec(params, deriv) +
235  0.5 * VecVec(new_params, new_deriv) - 0.5 * VecVec(params, new_deriv);
236  observed_changes(i) = new_objf - initial_objf;
237  }
238  BaseFloat threshold = 0.1;
239  KALDI_LOG << "predicted_changes = " << predicted_changes;
240  KALDI_LOG << "observed_changes = " << observed_changes;
241  if (!ApproxEqual(predicted_changes, observed_changes, threshold)) {
242  KALDI_WARN << "Derivatives self-test failed.";
243  return false;
244  } else {
245  return true;
246  }
247 }

250 void NnetCombiner::SelfTestModelDerivatives() {
251  int32 num_tests = 3; // more properly, this is the number of dimensions in a
252  // single test.
253  int32 dim = ParameterDim();
254 
255  Vector<BaseFloat> params(dim), deriv(dim);
256  Vector<BaseFloat> predicted_changes(num_tests),
257  observed_changes(num_tests);
258 
259  GetInitialParameters(&params);
260  Vector<BaseFloat> weights(WeightDim()), normalized_weights(WeightDim()),
261  nnet_params(NnetParameterDim(), kUndefined),
262  nnet_deriv(NnetParameterDim(), kUndefined);
263  GetWeights(params, &weights);
264  GetNormalizedWeights(weights, &normalized_weights);
265  GetNnetParameters(normalized_weights, &nnet_params);
266 
267  BaseFloat initial_objf = ComputeObjfAndDerivFromNnet(nnet_params,
268  &nnet_deriv);
269 
270  BaseFloat delta = 0.002 * std::sqrt(VecVec(nnet_params, nnet_params) /
271  NnetParameterDim());
272 
273 
274  for (int32 i = 0; i < num_tests; i++) {
275  Vector<BaseFloat> new_nnet_deriv(NnetParameterDim()),
276  offset(NnetParameterDim()), new_nnet_params(nnet_params);
277  offset.SetRandn();
278  new_nnet_params.AddVec(delta, offset);
279  BaseFloat new_objf = ComputeObjfAndDerivFromNnet(new_nnet_params,
280  &new_nnet_deriv);
281  // for predicted changes, interpolate old and new derivs.
282  predicted_changes(i) =
283  0.5 * VecVec(new_nnet_params, nnet_deriv) -
284  0.5 * VecVec(nnet_params, nnet_deriv) +
285  0.5 * VecVec(new_nnet_params, new_nnet_deriv) -
286  0.5 * VecVec(nnet_params, new_nnet_deriv);
287  observed_changes(i) = new_objf - initial_objf;
288  }
289  BaseFloat threshold = 0.1;
290  KALDI_LOG << "model-derivatives: predicted_changes = " << predicted_changes;
291  KALDI_LOG << "model-derivatives: observed_changes = " << observed_changes;
292  if (!ApproxEqual(predicted_changes, observed_changes, threshold))
293  KALDI_WARN << "Model derivatives self-test failed.";
294 }

317 void NnetCombiner::GetWeights(const VectorBase<BaseFloat> &params,
318  VectorBase<BaseFloat> *weights) const {
319  KALDI_ASSERT(weights->Dim() == WeightDim());
320  if (config_.separate_weights_per_component) {
321  weights->CopyFromVec(params);
322  } else {
323  int32 nc = NumUpdatableComponents();
324  // have one parameter per row of nnet_params_, and need to repeat
325  // the weight for the different components.
326  for (int32 n = 0; n < nnet_params_.NumRows(); n++) {
327  for (int32 c = 0; c < nc; c++)
328  (*weights)(n * nc + c) = params(n);
329  }
330  }
331  // we enforce positive weights by having the weights be the exponential of the
332  // corresponding parameters.
333  if (config_.enforce_positive_weights)
334  weights->ApplyExp();
335 }

338 void NnetCombiner::GetParamsDeriv(const VectorBase<BaseFloat> &weights,
339  const VectorBase<BaseFloat> &weights_deriv,
340  VectorBase<BaseFloat> *param_deriv) {
341  KALDI_ASSERT(weights.Dim() == WeightDim() &&
342  param_deriv->Dim() == ParameterDim());
343  Vector<BaseFloat> preexp_weights_deriv(weights_deriv);
344  if (config_.enforce_positive_weights) {
345  // to enforce positive weights we first compute weights (call these
346  // preexp_weights) and then take exponential. Note, d/dx exp(x) = exp(x).
347  // So the derivative w.r.t. the preexp_weights equals the derivative
348  // w.r.t. the weights, times the weights.
349  preexp_weights_deriv.MulElements(weights);
350  }
351  if (config_.separate_weights_per_component) {
352  param_deriv->CopyFromVec(preexp_weights_deriv);
353  } else {
354  int32 nc = NumUpdatableComponents();
355  param_deriv->SetZero();
356  for (int32 n = 0; n < nnet_params_.NumRows(); n++)
357  for (int32 c = 0; c < nc; c++)
358  (*param_deriv)(n) += preexp_weights_deriv(n * nc + c);
359  }
360 }

363 void NnetCombiner::GetNnetParameters(const Vector<BaseFloat> &weights,
364  VectorBase<BaseFloat> *nnet_params) const {
365  KALDI_ASSERT(nnet_params->Dim() == nnet_params_.NumCols());
366  nnet_params->SetZero();
367  int32 num_uc = NumUpdatableComponents(),
368  num_models = nnet_params_.NumRows();
369  for (int32 m = 0; m < num_models; m++) {
370  const SubVector<BaseFloat> src_params(nnet_params_, m);
371  int32 dim_offset = 0;
372  for (int32 c = 0; c < num_uc; c++) {
373  int32 index = m * num_uc + c;
374  BaseFloat weight = weights(index);
375  int32 dim = updatable_component_dims_[c];
376  const SubVector<BaseFloat> src_component_params(src_params, dim_offset,
377  dim);
378  SubVector<BaseFloat> dest_component_params(*nnet_params, dim_offset, dim);
379  dest_component_params.AddVec(weight, src_component_params);
380  dim_offset += dim;
381  }
382  KALDI_ASSERT(dim_offset == nnet_params_.NumCols());
383  }
384 }

412 double NnetCombiner::ComputeObjfAndDerivFromNnet(
413  VectorBase<BaseFloat> &nnet_params,
414  VectorBase<BaseFloat> *nnet_params_deriv) {
415  BaseFloat sum = nnet_params.Sum();
416  // inf/nan parameters->return -inf objective.
417  if (!(sum == sum && sum - sum == 0))
418  return -std::numeric_limits<double>::infinity();
419  // Set nnet to have these params.
420  UnVectorizeNnet(nnet_params, &nnet_);
421 
422  prob_computer_->Reset();
423  std::vector<NnetExample>::const_iterator iter = egs_.begin(),
424  end = egs_.end();
425  for (; iter != end; ++iter)
426  prob_computer_->Compute(*iter);
427  const SimpleObjectiveInfo *objf_info = prob_computer_->GetObjective("output");
428  if (objf_info == NULL)
429  KALDI_ERR << "Error getting objective info (unsuitable egs?)";
430  KALDI_ASSERT(objf_info->tot_weight > 0.0);
431  const Nnet &deriv = prob_computer_->GetDeriv();
432  VectorizeNnet(deriv, nnet_params_deriv);
433  // we prefer to deal with normalized objective functions.
434  nnet_params_deriv->Scale(1.0 / objf_info->tot_weight);
435  return objf_info->tot_objective / objf_info->tot_weight;
436 }

439 double NnetCombiner::ComputeObjfAndDerivFromParameters(
440  VectorBase<BaseFloat> &params,
441  VectorBase<BaseFloat> *params_deriv) {
442  Vector<BaseFloat> weights(WeightDim()), normalized_weights(WeightDim()),
443  nnet_params(NnetParameterDim(), kUndefined),
444  nnet_params_deriv(NnetParameterDim(), kUndefined),
445  normalized_weights_deriv(WeightDim()), weights_deriv(WeightDim());
446  GetWeights(params, &weights);
447  GetNormalizedWeights(weights, &normalized_weights);
448  GetNnetParameters(normalized_weights, &nnet_params);
449  double ans = ComputeObjfAndDerivFromNnet(nnet_params, &nnet_params_deriv);
450  if (ans != ans || ans - ans != 0) // NaN or inf
451  return ans; // No point computing derivative
452  GetWeightsDeriv(nnet_params_deriv, &normalized_weights_deriv);
453  GetUnnormalizedWeightsDeriv(weights, normalized_weights_deriv,
454  &weights_deriv);
455  GetParamsDeriv(weights, weights_deriv, params_deriv);
456  return ans;
457 }

460 // enforces the constraint that the weights for each component must sum to one.
461 void NnetCombiner::GetNormalizedWeights(
462  const VectorBase<BaseFloat> &unnorm_weights,
463  VectorBase<BaseFloat> *norm_weights) const {
464  if (!config_.enforce_sum_to_one) {
465  norm_weights->CopyFromVec(unnorm_weights);
466  return;
467  }
468  int32 num_uc = NumUpdatableComponents(),
469  num_models = nnet_params_.NumRows();
470  for (int32 c = 0; c < num_uc; c++) {
471  BaseFloat sum = 0.0;
472  for (int32 m = 0; m < num_models; m++) {
473  int32 index = m * num_uc + c;
474  sum += unnorm_weights(index);
475  }
476  BaseFloat inv_sum = 1.0 / sum; // if it's NaN then it's OK, we'll get NaN
477  // weights and eventually -inf objective.
478  for (int32 m = 0; m < num_models; m++) {
479  int32 index = m * num_uc + c;
480  (*norm_weights)(index) = unnorm_weights(index) * inv_sum;
481  }
482  }
483 }

485 void NnetCombiner::GetUnnormalizedWeightsDeriv(
486  const VectorBase<BaseFloat> &unnorm_weights,
487  const VectorBase<BaseFloat> &norm_weights_deriv,
488  VectorBase<BaseFloat> *unnorm_weights_deriv) {
489  if (!config_.enforce_sum_to_one) {
490  unnorm_weights_deriv->CopyFromVec(norm_weights_deriv);
491  return;
492  }
493  int32 num_uc = NumUpdatableComponents(),
494  num_models = nnet_params_.NumRows();
495  for (int32 c = 0; c < num_uc; c++) {
496  BaseFloat sum = 0.0;
497  for (int32 m = 0; m < num_models; m++) {
498  int32 index = m * num_uc + c;
499  sum += unnorm_weights(index);
500  }
501  BaseFloat inv_sum = 1.0 / sum;
502  BaseFloat inv_sum_deriv = 0.0;
503  for (int32 m = 0; m < num_models; m++) {
504  int32 index = m * num_uc + c;
505  // in the forward direction, we'd do:
506  // (*norm_weights)(index) = unnorm_weights(index) * inv_sum;
507  (*unnorm_weights_deriv)(index) = inv_sum * norm_weights_deriv(index);
508  inv_sum_deriv += norm_weights_deriv(index) * unnorm_weights(index);
509  }
510  // note: d/dx (1/x) = -1/x^2
511  BaseFloat sum_deriv = -1.0 * inv_sum_deriv * inv_sum * inv_sum;
512  for (int32 m = 0; m < num_models; m++) {
513  int32 index = m * num_uc + c;
514  (*unnorm_weights_deriv)(index) += sum_deriv;
515  }
516  }
517 }
