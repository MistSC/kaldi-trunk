168  while (1) {
169  // loop over all streams, check if any stream reaches the end of its utterance,
170  // if any, feed the exhausted stream with a new utterance, update book-keeping infos
171  for (int s = 0; s < num_stream; s++) {
172  // this stream still has valid frames

174  new_utt_flags[s] = 0;
175  continue;

177  // else, this stream exhausted, need new utterance

196  KALDI_WARN << key << ", length miss-match between feats and targets, skip";
197  num_other_error++;

203  keys[s] = key;

205  feat_transf.CopyToMat(&feats[s]);
206  targets[s] = target;
207  curt[s] = 0;
208  lent[s] = feats[s].NumRows();
209  new_utt_flags[s] = 1; // a new utterance feeded to this stream
210  feature_reader.Next();
211  break;
212  }
213  }
214 
215  // we are done if all streams are exhausted

217  for (int s = 0; s < num_stream; s++) {
218  if (curt[s] < lent[s]) done = 0; // this stream still contains valid data, not exhausted
219  }
220  if (done) break;

222  // fill a multi-stream bptt batch
223  // * frame_mask: 0 indicates padded frames, 1 indicates valid frames
224  // * target: padded to batch_size
225  // * feat: first shifted to achieve targets delay; then padded to batch_size
226  for (int t = 0; t < batch_size; t++) {
227  for (int s = 0; s < num_stream; s++) {
228  // frame_mask & targets padding
229  if (curt[s] < lent[s]) {
230  frame_mask(t * num_stream + s) = 1;
231  target[t * num_stream + s] = targets[s][curt[s]];
232  } else {
233  frame_mask(t * num_stream + s) = 0;
234  target[t * num_stream + s] = targets[s][lent[s]-1];
235  }
236  // feat shifting & padding
237  if (curt[s] + targets_delay < lent[s]) {
238  feat.Row(t * num_stream + s).CopyFromVec(feats[s].Row(curt[s]+targets_delay));
239  } else {
240  feat.Row(t * num_stream + s).CopyFromVec(feats[s].Row(lent[s]-1));
241  }
242 
243  curt[s]++;
244  }
245  }

248  nnet.ResetLstmStreams(new_utt_flags);
249

252
