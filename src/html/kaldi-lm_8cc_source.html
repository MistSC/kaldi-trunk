31 // add the string contained in inpline to the current transducer
32 // starting at initial state
33 LangModelFst::StateId LangModelFst::ReadTxtLine(const string &inpline) {
34  KALDI_ASSERT(pfst_);
35  KALDI_ASSERT(pfst_->InputSymbols());
36  KALDI_ASSERT(pfst_->OutputSymbols());
37 
38  StateId dst = pfst_->Start(), src = pfst_->Start();
39  // this will split on white spaces only
40  string curwrd; // Have a buffer string
41  std::stringstream ss(inpline); // Insert the string into a stream
42  while (ss >> curwrd) {
43  // add labels to symbol tables
44  int64 ilab = pfst_->MutableInputSymbols()->AddSymbol(curwrd);
45  int64 olab = pfst_->MutableOutputSymbols()->AddSymbol(curwrd);
46  dst = pfst_->AddState();
47  pfst_->AddArc(src, fst::StdArc(ilab, olab, 0, dst));
48  // cerr << " adding word " << curwrd << " from state " << src;
49  // cerr << " to state " << dst <<endl;
50  src = dst;
51  }
52  return dst;
53 }

55 // create a path in the FST for each line of the input stream
56 // fst must already be provided with symbol tables
57 void LangModelFst::ReadTxtString(std::istream &strm) {
58  string inpline;
59  StateId src, final = pfst_->AddState();
60 
61  while (getline(strm, inpline) && !strm.eof()) {
62  // cerr << "ReadTxtString: read line " << inpline << endl;
63  src = ReadTxtLine(inpline);
64  // add arc from last state produced to final state
65  pfst_->AddArc(src, fst::StdArc(0, 0, 0, final));
66  }
67  pfst_->SetFinal(final, fst::StdArc::Weight::One());
68 }
