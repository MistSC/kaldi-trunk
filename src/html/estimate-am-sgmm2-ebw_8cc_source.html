27 void EbwAmSgmm2Updater::Update(const MleAmSgmm2Accs &num_accs,
28  const MleAmSgmm2Accs &den_accs,
29  AmSgmm2 *model,
30  SgmmUpdateFlagsType flags,
31  BaseFloat *auxf_change_out,
32  BaseFloat *count_out) {
33 
34  // Various quantities need to be computed at the start, before we
35  // change any of the model parameters.
36  std::vector< SpMatrix<double> > Q_num, Q_den, H, S_means;
37 
38  if (flags & kSgmmPhoneProjections) {
39  MleAmSgmm2Updater::ComputeQ(num_accs, *model, &Q_num);
40  MleAmSgmm2Updater::ComputeQ(den_accs, *model, &Q_den);
41  }
42  if (flags & kSgmmCovarianceMatrix) { // compute the difference between
43  // the num and den S_means matrices... this is what we will need.
44  MleAmSgmm2Updater::ComputeSMeans(num_accs, *model, &S_means);
45  std::vector< SpMatrix<double> > S_means_tmp;
46  MleAmSgmm2Updater::ComputeSMeans(den_accs, *model, &S_means_tmp);
47  for (size_t i = 0; i < S_means.size(); i++)
48  S_means[i].AddSp(-1.0, S_means_tmp[i]);
49  }
50  if (flags & (kSgmmPhoneVectors | kSgmmPhoneWeightProjections))
51  model->ComputeH(&H);
52 
53  Vector<double> gamma_num(num_accs.num_gaussians_);
54  for (int32 j1 = 0; j1 < num_accs.num_groups_; j1++)
55  gamma_num.AddRowSumMat(1.0, num_accs.gamma_[j1]);
56  Vector<double> gamma_den(den_accs.num_gaussians_);
57  for (int32 j1 = 0; j1 < den_accs.num_groups_; j1++)
58  gamma_den.AddRowSumMat(1.0, den_accs.gamma_[j1]);
59 
60  BaseFloat tot_impr = 0.0;
61 
62  if (flags & kSgmmPhoneVectors)
63  tot_impr += UpdatePhoneVectors(num_accs, den_accs, H, model);
64 
65  if (flags & kSgmmPhoneProjections)
66  tot_impr += UpdateM(num_accs, den_accs, Q_num, Q_den,
67  gamma_num, gamma_den, model);
68 
69  if (flags & kSgmmPhoneWeightProjections)
70  tot_impr += UpdateW(num_accs, den_accs, gamma_num, gamma_den, model);
71 
72  if (flags & kSgmmSpeakerWeightProjections)
73  tot_impr += UpdateU(num_accs, den_accs, gamma_num, gamma_den, model);
74 
75  if (flags & kSgmmCovarianceMatrix)
76  tot_impr += UpdateVars(num_accs, den_accs,
77  gamma_num, gamma_den, S_means, model);
78 
79  if (flags & kSgmmSubstateWeights)
80  tot_impr += UpdateSubstateWeights(num_accs, den_accs, model);
81 
82  if (flags & kSgmmSpeakerProjections)
83  tot_impr += UpdateN(num_accs, den_accs, gamma_num, gamma_den, model);
84 
85 
86  if (auxf_change_out) *auxf_change_out = tot_impr * num_accs.total_frames_;
87  if (count_out) *count_out = num_accs.total_frames_;
88 
89  if (fabs(num_accs.total_frames_ - den_accs.total_frames_) >
90  0.01*(num_accs.total_frames_ + den_accs.total_frames_))
91  KALDI_WARN << "Num and den frame counts differ, "
92  << num_accs.total_frames_ << " vs. " << den_accs.total_frames_;
93 
94  BaseFloat like_diff = num_accs.total_like_ - den_accs.total_like_;
95 
96  KALDI_LOG << "***Averaged differenced likelihood per frame is "
97  << (like_diff/num_accs.total_frames_)
98  << " over " << (num_accs.total_frames_) << " frames.";
99  KALDI_LOG << "***Note: for this to be at all meaningful, if you use "
100  << "\"canceled\" stats you will have to renormalize this over "
101  << "the \"real\" frame count.";
102  KALDI_ASSERT(num_accs.total_frames_ > 0 && den_accs.total_frames_ > 0);
103 
104  model->ComputeNormalizers();
105 }

140 void EbwAmSgmm2Updater::ComputePhoneVecStats(
141  const MleAmSgmm2Accs &accs,
142  const AmSgmm2 &model,
143  const std::vector<SpMatrix<double> > &H,
144  int32 j1,
145  int32 m,
146  const Vector<double> &w_jm_in,
147  double gamma_jm,
148  Vector<double> *g_jm,
149  SpMatrix<double> *H_jm) {
150  Vector<double> w_jm(w_jm_in);
151  if (!accs.a_.empty() && accs.a_[j1](m, 0) != 0) { // [SSGMM]
152  w_jm.MulElements(accs.a_[j1].Row(m)); // multiply by "a" quantities..
153  w_jm.Scale(1.0 / w_jm.Sum()); // renormalize.
154  }
155  g_jm->CopyFromVec(accs.y_[j1].Row(m));
156  for (int32 i = 0; i < accs.num_gaussians_; i++) {
157  double gamma_jmi = accs.gamma_[j1](m, i);
158  double quadratic_term = std::max(gamma_jmi, gamma_jm * w_jm(i));
159  double scalar = gamma_jmi - gamma_jm * w_jm(i) + quadratic_term
160  * VecVec(model.w_.Row(i), model.v_[j1].Row(m));
161  g_jm->AddVec(scalar, model.w_.Row(i));
162  if (gamma_jmi != 0.0)
163  H_jm->AddSp(gamma_jmi, H[i]); // The most important term..
164  if (quadratic_term > 1.0e-10)
165  H_jm->AddVec2(static_cast<BaseFloat>(quadratic_term), model.w_.Row(i));
166  }
167 }

170 // Runs the phone vectors update for a subset of states (called
171 // multi-threaded).
172 void EbwAmSgmm2Updater::UpdatePhoneVectorsInternal(
173  const MleAmSgmm2Accs &num_accs,
174  const MleAmSgmm2Accs &den_accs,
175  const std::vector<SpMatrix<double> > &H,
176  AmSgmm2 *model,
177  double *auxf_impr,
178  int32 num_threads,
179  int32 thread_id) const {
180 
181  int32 block_size = (num_accs.num_groups_ + (num_threads-1)) / num_threads,
182  j1_start = block_size * thread_id,
183  j1_end = std::min(num_accs.num_groups_, j1_start + block_size);
184 
185  int32 S = num_accs.phn_space_dim_, I = num_accs.num_gaussians_;
186 
187  for (int32 j1 = j1_start; j1 < j1_end; j1++) {
188  double num_state_count = 0.0,
189  state_auxf_impr = 0.0;
190  Vector<double> w_jm(I);
191  for (int32 m = 0; m < model->NumSubstatesForGroup(j1); m++) {
192  double gamma_jm_num = num_accs.gamma_[j1].Row(m).Sum();
193  double gamma_jm_den = den_accs.gamma_[j1].Row(m).Sum();
194  num_state_count += gamma_jm_num;
195  Vector<double> g_jm_num(S); // computed using eq. 58 of SGMM paper [for numerator stats]
196  SpMatrix<double> H_jm_num(S); // computed using eq. 59 of SGMM paper [for numerator stats]
197  Vector<double> g_jm_den(S); // same, but for denominator stats.
198  SpMatrix<double> H_jm_den(S);
199 
200  // Compute the weights for this sub-state.
201  // w_jm = softmax([w_{k1}^T ... w_{kD}^T] * v_{jkm}) eq.(7)
202  w_jm.AddMatVec(1.0, Matrix<double>(model->w_), kNoTrans,
203  Vector<double>(model->v_[j1].Row(m)), 0.0);
204  w_jm.ApplySoftMax();
205  // Note: in the ML code, in the SSGMM case, at this point the w_jm would
206  // be modified with the "a" quantities to get the "\tilde{w}_{jm}" of the
207  // SSGMM techreport. But in this code, it gets done inside ComputePhoneVecStats.
208 
209  ComputePhoneVecStats(num_accs, *model, H, j1, m, w_jm, gamma_jm_num,
210  &g_jm_num, &H_jm_num);
211  ComputePhoneVecStats(den_accs, *model, H, j1, m, w_jm, gamma_jm_den,
212  &g_jm_den, &H_jm_den);
213 
214  Vector<double> v_jm(model->v_[j1].Row(m));
215  Vector<double> local_derivative(S); // difference of derivative of numerator
216  // and denominator objetive function.
217  local_derivative.AddVec(1.0, g_jm_num);
218  local_derivative.AddSpVec(-1.0, H_jm_num, v_jm, 1.0);
219  local_derivative.AddVec(-1.0, g_jm_den);
220  local_derivative.AddSpVec(-1.0 * -1.0, H_jm_den, v_jm, 1.0);
221 
222  SpMatrix<double> quadratic_term(H_jm_num);
223  quadratic_term.AddSp(1.0, H_jm_den);
224  double substate_count = 1.0e-10 + gamma_jm_num + gamma_jm_den;
225  quadratic_term.Scale( (substate_count + options_.tau_v) / substate_count);
226  quadratic_term.Scale(1.0 / (options_.lrate_v + 1.0e-10) );
227 
228  Vector<double> delta_v_jm(S);
229 
230  SolverOptions opts;
231  opts.name = "v";
232  opts.K = options_.max_cond;
233  opts.eps = options_.epsilon;
234 
235  double auxf_impr =
236  ((gamma_jm_num + gamma_jm_den == 0) ? 0.0 :
237  SolveQuadraticProblem(quadratic_term,
238  local_derivative,
239  opts, &delta_v_jm));
240 
241  v_jm.AddVec(1.0, delta_v_jm);
242  model->v_[j1].Row(m).CopyFromVec(v_jm);
243  state_auxf_impr += auxf_impr;
244  }
245 
246  *auxf_impr += state_auxf_impr;
247  if (j1 < 10 && thread_id == 0) {
248  KALDI_LOG << "Objf impr for group j = " << j1 << " is "
249  << (state_auxf_impr / (num_state_count + 1.0e-10))
250  << " over " << num_state_count << " frames";
251  }
252  }
253 }

255 double EbwAmSgmm2Updater::UpdatePhoneVectors(const MleAmSgmm2Accs &num_accs,
256  const MleAmSgmm2Accs &den_accs,
257  const vector< SpMatrix<double> > &H,
258  AmSgmm2 *model) const {
259  KALDI_LOG << "Updating phone vectors.";
260 
261  double count = 0.0, auxf_impr = 0.0;
262 
263  int32 J1 = num_accs.num_groups_;
264  for (int32 j1 = 0; j1 < J1; j1++) count += num_accs.gamma_[j1].Sum();
265 
266  EbwUpdatePhoneVectorsClass c(this, num_accs, den_accs, H, model, &auxf_impr);
267  RunMultiThreaded(c);
268 
269  auxf_impr /= count;
270 
271  KALDI_LOG << "**Overall auxf improvement for v is " << auxf_impr
272  << " over " << count << " frames";
273  return auxf_impr;
274 }

277 double EbwAmSgmm2Updater::UpdateM(const MleAmSgmm2Accs &num_accs,
278  const MleAmSgmm2Accs &den_accs,
279  const std::vector< SpMatrix<double> > &Q_num,
280  const std::vector< SpMatrix<double> > &Q_den,
281  const Vector<double> &gamma_num,
282  const Vector<double> &gamma_den,
283  AmSgmm2 *model) const {
284  int32 S = model->PhoneSpaceDim(),
285  D = model->FeatureDim(),
286  I = model->NumGauss();
287 
288  Vector<double> impr_vec(I);
289 
290  for (int32 i = 0; i < I; i++) {
291  double gamma_i_num = gamma_num(i), gamma_i_den = gamma_den(i);
292 
293  if (gamma_i_num + gamma_i_den == 0.0) {
294  KALDI_WARN << "Not updating phonetic basis for i = " << i
295  << " because count is zero. ";
296  continue;
297  }
298 
299  Matrix<double> Mi(model->M_[i]);
300  Matrix<double> L(D, S); // this is something like the Y quantity, which
301  // represents the linear term in the objf on M-- except that we make it the local
302  // derivative about the current value, instead of the derivative around zero.
303  // But it's not exactly the derivative w.r.t. M, due to the factor of Sigma_i.
304  // The auxiliary function is Q(x) = tr(M^T P Y) - 0.5 tr(P M Q M^T),
305  // where P is Y^{-1}. The quantity L we define here will be Y - M Q,
306  // and you can think of this as like the local derivative, except there is
307  // a term P in there.
308  L.AddMat(1.0, num_accs.Y_[i]);
309  L.AddMatSp(-1.0, Mi, kNoTrans, Q_num[i], 1.0);
310  L.AddMat(-1.0, den_accs.Y_[i]);
311  L.AddMatSp(-1.0*-1.0, Mi, kNoTrans, Q_den[i], 1.0);
312 
313  SpMatrix<double> Q(S); // This is a combination of the Q's for the numerator and denominator.
314  Q.AddSp(1.0, Q_num[i]);
315  Q.AddSp(1.0, Q_den[i]);
316 
317  double state_count = 1.0e-10 + gamma_i_num + gamma_i_den; // the count
318  // represented by the quadratic part of the stats.
319  Q.Scale( (state_count + options_.tau_M) / state_count );
320  Q.Scale( 1.0 / (options_.lrate_M + 1.0e-10) );
321 
322 
323  SolverOptions opts;
324  opts.name = "M";
325  opts.K = options_.max_cond;
326  opts.eps = options_.epsilon;
327 
328  Matrix<double> deltaM(D, S);
329  double impr =
330  SolveQuadraticMatrixProblem(Q, L,
331  SpMatrix<double>(model->SigmaInv_[i]),
332  opts, &deltaM);
333 
334  impr_vec(i) = impr;
335  Mi.AddMat(1.0, deltaM);
336  model->M_[i].CopyFromMat(Mi);
337  if (i < 10 || impr / state_count > 3.0) {
338  KALDI_VLOG(2) << "Objf impr for projection M for i = " << i << ", is "
339  << (impr/(gamma_i_num + 1.0e-20)) << " over " << gamma_i_num
340  << " frames";
341  }
342  }
343  BaseFloat tot_count = gamma_num.Sum(), tot_impr = impr_vec.Sum();
344 
345  tot_impr /= (tot_count + 1.0e-20);
346  KALDI_LOG << "Overall auxiliary function improvement for model projections "
347  << "M is " << tot_impr << " over " << tot_count << " frames";
348 
349  KALDI_VLOG(1) << "Updating M: num-count is " << gamma_num;
350  KALDI_VLOG(1) << "Updating M: den-count is " << gamma_den;
351  KALDI_VLOG(1) << "Updating M: objf-impr is " << impr_vec;
352 
353  return tot_impr;
354 }

357 // Note: we do just one iteration of the weight-projection update here. The
358 // weak-sense auxiliary functions used don't really make sense if we do it for
359 // multiple iterations. It would be possible to use a similar auxiliary
360 // function to the one on my (D. Povey)'s thesis for the Gaussian mixture
361 // weights, which would make sense for multiple iterations, but this would be a
362 // bit more complex to implement and probably would not give much improvement
363 // over this approach.
364 double EbwAmSgmm2Updater::UpdateW(const MleAmSgmm2Accs &num_accs,
365  const MleAmSgmm2Accs &den_accs,
366  const Vector<double> &gamma_num,
367  const Vector<double> &gamma_den,
368  AmSgmm2 *model) {
369  KALDI_LOG << "Updating weight projections";
370 
371  int32 I = num_accs.num_gaussians_, S = num_accs.phn_space_dim_;
372 
373  Matrix<double> g_i_num(I, S), g_i_den(I, S);
374 
375  // View F_i_{num,den} as vectors of SpMatrix [i.e. symmetric matrices,
376  // linearized into vectors]
377  Matrix<double> F_i_num(I, (S*(S+1))/2), F_i_den(I, (S*(S+1))/2);
378 
379  Vector<double> impr_vec(I);
380 
381  // Get the F_i and g_i quantities-- this is done in parallel (multi-core),
382  // using the same code we use in the ML update [except we get it for
383  // numerator and denominator separately.]
384  Matrix<double> w(model->w_);
385  {
386  std::vector<Matrix<double> > log_a_num;
387  if (model->HasSpeakerDependentWeights())
388  MleAmSgmm2Updater::ComputeLogA(num_accs, &log_a_num);
389  double garbage;
390  UpdateWClass c_num(num_accs, *model, w, log_a_num, &F_i_num, &g_i_num, &garbage);
391  RunMultiThreaded(c_num);
392  }
393  {
394  std::vector<Matrix<double> > log_a_den;
395  if (model->HasSpeakerDependentWeights())
396  MleAmSgmm2Updater::ComputeLogA(den_accs, &log_a_den);
397  double garbage;
398  UpdateWClass c_den(den_accs, *model, w, log_a_den, &F_i_den, &g_i_den, &garbage);
399  RunMultiThreaded(c_den);
400  }
401 
402  for (int32 i = 0; i < I; i++) {
403 
404  // auxf was originally formulated in terms of the change in w (i.e. the
405  // g quantities are the local derivatives), so there is less hassle than
406  // with some of the other updates, in changing it to be discriminative.
407  // we essentially just difference the linear terms and add the quadratic
408  // terms.
409 
410  Vector<double> derivative(g_i_num.Row(i));
411  derivative.AddVec(-1.0, g_i_den.Row(i));
412  // F_i_num quadratic_term is a bit like the negated 2nd derivative
413  // of the numerator stats-- actually it's not the actual 2nd deriv,
414  // but an upper bound on it.
415  SpMatrix<double> quadratic_term(S), tmp_F(S);
416  quadratic_term.CopyFromVec(F_i_num.Row(i));
417  tmp_F.CopyFromVec(F_i_den.Row(i)); // tmp_F is used for Vector->SpMatrix conversion.
418  quadratic_term.AddSp(1.0, tmp_F);
419 
420  double state_count = gamma_num(i) + gamma_den(i);
421 
422  quadratic_term.Scale((state_count + options_.tau_w) / (state_count + 1.0e-10));
423  quadratic_term.Scale(1.0 / (options_.lrate_w + 1.0e-10) );
424 
425  Vector<double> delta_w(S);
426 
427  SolverOptions opts;
428  opts.name = "w";
429  opts.K = options_.max_cond;
430  opts.eps = options_.epsilon;
431 
432  double objf_impr =
433  SolveQuadraticProblem(quadratic_term, derivative, opts, &delta_w);
434 
435  impr_vec(i) = objf_impr;
436  if (i < 10 || objf_impr / (gamma_num(i) + 1.0e-10) > 2.0) {
437  KALDI_LOG << "Predicted objf impr for w per frame is "
438  << (objf_impr / (gamma_num(i) + 1.0e-10))
439  << " over " << gamma_num(i) << " frames.";
440  }
441  model->w_.Row(i).AddVec(1.0, delta_w);
442  }
443  KALDI_VLOG(1) << "Updating w: numerator count is " << gamma_num;
444  KALDI_VLOG(1) << "Updating w: denominator count is " << gamma_den;
445  KALDI_VLOG(1) << "Updating w: objf-impr is " << impr_vec;
446 
447  double tot_num_count = gamma_num.Sum(), tot_impr = impr_vec.Sum();
448  tot_impr /= tot_num_count;
449 
450  KALDI_LOG << "**Overall objf impr for w per frame is "
451  << tot_impr << " over " << tot_num_count
452  << " frames.";
453  return tot_impr;
454 }

457 double EbwAmSgmm2Updater::UpdateU(const MleAmSgmm2Accs &num_accs,
458  const MleAmSgmm2Accs &den_accs,
459  const Vector<double> &gamma_num,
460  const Vector<double> &gamma_den,
461  AmSgmm2 *model) {
462  int32 T = num_accs.spk_space_dim_;
463  double tot_impr = 0.0;
464  for (int32 i = 0; i < num_accs.num_gaussians_; i++) {
465  if (gamma_num(i) < 200.0) {
466  KALDI_LOG << "Numerator count is small " << gamma_num(i) << " for gaussian "
467  << i << ", not updating u_i.";
468  continue;
469  }
470  Vector<double> u_i(model->u_.Row(i));
471  Vector<double> delta_u(T);
472  Vector<double> t(T); // derivative.
473  t.AddVec(1.0, num_accs.t_.Row(i));
474  t.AddVec(-1.0, den_accs.t_.Row(i));
475  SpMatrix<double> U(T); // quadratic term.
476  U.AddSp(1.0, num_accs.U_[i]);
477  U.AddSp(1.0, den_accs.U_[i]);
478 
479  double state_count = gamma_num(i) + gamma_den(i);
480  U.Scale((state_count + options_.tau_u) / (state_count + 1.0e-10));
481  U.Scale(1.0 / (options_.lrate_u + 1.0e-10) );
482 
483  SolverOptions opts;
484  opts.name = "u";
485  opts.K = options_.max_cond;
486  opts.eps = options_.epsilon;
487 
488  double impr = SolveQuadraticProblem(U, t, opts, &delta_u);
489  double impr_per_frame = impr / gamma_num(i);
490  if (impr_per_frame > options_.max_impr_u) {
491  KALDI_WARN << "Updating speaker weight projections u, for Gaussian index "
492  << i << ", impr/frame is " << impr_per_frame << " over "
493  << gamma_num(i) << " frames, scaling back to not exceed "
494  << options_.max_impr_u;
495  double scale = options_.max_impr_u / impr_per_frame;
496  impr *= scale;
497  delta_u.Scale(scale);
498  // Note: a linear scaling of "impr" with "scale" is not quite accurate
499  // in depicting how the quadratic auxiliary function varies as we change
500  // the scale on "delta", but this does not really matter-- the goal is
501  // to limit the auxiliary-function change to not be too large.
502  }
503  if (i < 10) {
504  KALDI_LOG << "Objf impr for spk weight-projection u for i = " << (i)
505  << ", is " << (impr / (gamma_num(i) + 1.0e-20)) << " over "
506  << gamma_num(i) << " frames";
507  }
508  u_i.AddVec(1.0, delta_u);
509  model->u_.Row(i).CopyFromVec(u_i);
510  tot_impr += impr;
511  }
512  KALDI_LOG << "**Overall objf impr for u is " << (tot_impr/gamma_num.Sum())
513  << ", over " << gamma_num.Sum() << " frames";
514  return tot_impr;
515 }

518 double EbwAmSgmm2Updater::UpdateN(const MleAmSgmm2Accs &num_accs,
519  const MleAmSgmm2Accs &den_accs,
520  const Vector<double> &gamma_num,
521  const Vector<double> &gamma_den,
522  AmSgmm2 *model) const {
523  if (num_accs.spk_space_dim_ == 0 || num_accs.R_.size() == 0 ||
524  num_accs.Z_.size() == 0) {
525  KALDI_ERR << "Speaker subspace dim is zero or no stats accumulated";
526  }
527 
528  int32 I = num_accs.num_gaussians_, D = num_accs.feature_dim_,
529  T = num_accs.spk_space_dim_;
530 
531  Vector<double> impr_vec(I);
532 
533  for (int32 i = 0; i < I; i++) {
534  double gamma_i_num = gamma_num(i), gamma_i_den = gamma_den(i);
535  if (gamma_i_num + gamma_i_den == 0.0) {
536  KALDI_WARN << "Not updating speaker basis for i = " << i
537  << " because count is zero. ";
538  continue;
539  }
540  Matrix<double> Ni(model->N_[i]);
541  // See comment near declaration of L in UpdateM(). This update is the
542  // same, but change M->N, Y->Z and Q->R.
543 
544  Matrix<double> L(D, T);
545  L.AddMat(1.0, num_accs.Z_[i]);
546  L.AddMatSp(-1.0, Ni, kNoTrans, num_accs.R_[i], 1.0);
547  L.AddMat(-1.0, den_accs.Z_[i]);
548  L.AddMatSp(-1.0*-1.0, Ni, kNoTrans, den_accs.R_[i], 1.0);
549 
550  SpMatrix<double> R(T); // combination of the numerator and denominator R's.
551  R.AddSp(1.0, num_accs.R_[i]);
552  R.AddSp(1.0, den_accs.R_[i]);
553 
554  double state_count = 1.0e-10 + gamma_i_num + gamma_i_den; // the count
555  // represented by the quadratic part of the stats.
556  R.Scale( (state_count + options_.tau_N) / state_count );
557  R.Scale( 1.0 / (options_.lrate_N + 1.0e-10) );
558 
559  Matrix<double> deltaN(D, T);
560 
561  SolverOptions opts;
562  opts.name = "N";
563  opts.K = options_.max_cond;
564  opts.eps = options_.epsilon;
565 
566  double impr =
567  SolveQuadraticMatrixProblem(R, L,
568  SpMatrix<double>(model->SigmaInv_[i]),
569  opts, &deltaN);
570  impr_vec(i) = impr;
571  Ni.AddMat(1.0, deltaN);
572  model->N_[i].CopyFromMat(Ni);
573  if (i < 10 || impr / (state_count+1.0e-20) > 3.0) {
574  KALDI_LOG << "Objf impr for spk projection N for i = " << (i)
575  << ", is " << (impr / (gamma_i_num + 1.0e-20)) << " over "
576  << gamma_i_num << " frames";
577  }
578  }
579 
580  KALDI_VLOG(1) << "Updating N: numerator count is " << gamma_num;
581  KALDI_VLOG(1) << "Updating N: denominator count is " << gamma_den;
582  KALDI_VLOG(1) << "Updating N: objf-impr is " << impr_vec;
583 
584  double tot_count = gamma_num.Sum(), tot_impr = impr_vec.Sum();
585  tot_impr /= (tot_count + 1.0e-20);
586  KALDI_LOG << "**Overall auxf impr for N is " << tot_impr
587  << " over " << tot_count << " frames";
588  return tot_impr;
589 }

591 double EbwAmSgmm2Updater::UpdateVars(const MleAmSgmm2Accs &num_accs,
592  const MleAmSgmm2Accs &den_accs,
593  const Vector<double> &gamma_num,
594  const Vector<double> &gamma_den,
595  const std::vector< SpMatrix<double> > &S_means,
596  AmSgmm2 *model) const {
597  // Note: S_means contains not only the quantity S_means in the paper,
598  // but also has a term - (Y_i M_i^T + M_i Y_i^T). Plus, it is differenced
599  // between numerator and denominator. We don't calculate it here,
600  // because it had to be computed with the original model, before we
601  // changed the M quantities.
602  int32 I = num_accs.num_gaussians_;
603  KALDI_ASSERT(S_means.size() == I);
604  Vector<double> impr_vec(I);
605 
606  for (int32 i = 0; i < I; i++) {
607  double num_count = gamma_num(i), den_count = gamma_den(i);
608 
609  SpMatrix<double> SigmaStats(S_means[i]);
610  SigmaStats.AddSp(1.0, num_accs.S_[i]);
611  SigmaStats.AddSp(-1.0, den_accs.S_[i]);
612  // SigmaStats now contain the stats for estimating Sigma (as in the main SGMM paper),
613  // differenced between num and den.
614  SpMatrix<double> SigmaInvOld(model->SigmaInv_[i]), SigmaOld(model->SigmaInv_[i]);
615  SigmaOld.Invert();
616  double count = num_count - den_count;
617  KALDI_ASSERT(options_.lrate_Sigma <= 1.0);
618  double inv_lrate = 1.0 / options_.lrate_Sigma;
619  // These formulas assure that the objective function behaves in
620  // a roughly symmetric way w.r.t. num and den counts.
621  double E_den = 1.0 + inv_lrate, E_num = inv_lrate - 1.0;
622 
623  double smoothing_count =
624  (options_.tau_Sigma * inv_lrate) + // multiply tau_Sigma by inverse-lrate
625  (E_den * den_count) + // for compatibility with other updates.
626  (E_num * num_count) +
627  1.0e-10;
628  SigmaStats.AddSp(smoothing_count, SigmaOld);
629  count += smoothing_count;
630  SigmaStats.Scale(1.0 / count);
631  SpMatrix<double> SigmaInv(SigmaStats); // before floor and ceiling. Currently sigma,
632  // not its inverse.
633  bool verbose = false;
634  int n_floor = SigmaInv.ApplyFloor(SigmaOld, options_.cov_min_value, verbose);
635  SigmaInv.Invert(); // make it inverse variance.
636  int n_ceiling = SigmaInv.ApplyFloor(SigmaInvOld, options_.cov_min_value, verbose);
637 
638  // this auxf_change.
639  double auxf_change = -0.5 * count *(TraceSpSp(SigmaInv, SigmaStats)
640  - TraceSpSp(SigmaInvOld, SigmaStats)
641  - SigmaInv.LogDet()
642  + SigmaInvOld.LogDet());
643 
644  model->SigmaInv_[i].CopyFromSp(SigmaInv);
645  impr_vec(i) = auxf_change;
646  if (i < 10 || auxf_change / (num_count+den_count+1.0e-10) > 2.0
647  || n_floor+n_ceiling > 0) {
648  KALDI_LOG << "Updating variance: Auxf change per frame for Gaussian "
649  << i << " is " << (auxf_change / num_count) << " over "
650  << num_count << " frames " << "(den count was " << den_count
651  << "), #floor,ceil was " << n_floor << ", " << n_ceiling;
652  }
653  }
654  KALDI_VLOG(1) << "Updating Sigma: numerator count is " << gamma_num;
655  KALDI_VLOG(1) << "Updating Sigma: denominator count is " << gamma_den;
656  KALDI_VLOG(1) << "Updating Sigma: objf-impr is " << impr_vec;
657 
658  double tot_count = gamma_num.Sum(), tot_impr = impr_vec.Sum();
659  tot_impr /= tot_count+1.0e-20;
660  KALDI_LOG << "**Overall auxf impr for Sigma is " << tot_impr
661  << " over " << tot_count << " frames";
662  return tot_impr;
663 }

666 double EbwAmSgmm2Updater::UpdateSubstateWeights(
667  const MleAmSgmm2Accs &num_accs,
668  const MleAmSgmm2Accs &den_accs,
669  AmSgmm2 *model) {
670  KALDI_LOG << "Updating substate mixture weights";
671 
672  double tot_count = 0.0, tot_impr = 0.0;
673  for (int32 j2 = 0; j2 < num_accs.num_pdfs_; j2++) {
674  int32 M = model->NumSubstatesForPdf(j2);
675  Vector<double> num_occs(M), den_occs(M),
676  orig_weights(model->c_[j2]), weights(model->c_[j2]);
677 
678  for (int32 m = 0; m < M; m++) {
679  num_occs(m) = num_accs.gamma_c_[j2](m)
680  + options_.tau_c * weights(m);
681  den_occs(m) = den_accs.gamma_c_[j2](m);
682  }
683 
684  if (weights.Dim() > 1) {
685  double begin_auxf = 0.0, end_auxf = 0.0;
686  for (int32 m = 0; m < M; m++) { // see eq. 4.32, Dan Povey's PhD thesis.
687  begin_auxf += num_occs(m) * log (weights(m))
688  - den_occs(m) * weights(m) / orig_weights(m);
689  }
690  for (int32 iter = 0; iter < 50; iter++) {
691  Vector<double> k_jm(M);
692  double max_m = 0.0;
693  for (int32 m = 0; m < M; m++)
694  max_m = std::max(max_m, den_occs(m)/orig_weights(m));
695  for (int32 m = 0; m < M; m++)
696  k_jm(m) = max_m - den_occs(m)/orig_weights(m);
697  for (int32 m = 0; m < M; m++)
698  weights(m) = num_occs(m) + k_jm(m)*weights(m);
699  weights.Scale(1.0 / weights.Sum());
700  }
701  for (int32 m = 0; m < M; m++)
702  weights(m) = std::max(weights(m),
703  static_cast<double>(options_.min_substate_weight));
704  weights.Scale(1.0 / weights.Sum()); // renormalize.
705 
706  for (int32 m = 0; m < M; m++) {
707  end_auxf += num_occs(m) * log (weights(m))
708  - den_occs(m) * weights(m) / orig_weights(m);
709  }
710  tot_impr += end_auxf - begin_auxf;
711  double this_impr = ((end_auxf - begin_auxf) / num_occs.Sum());
712  if (j2 < 10 || this_impr > 0.5) {
713  KALDI_LOG << "Updating substate weights: auxf impr for pdf " << j2
714  << " is " << this_impr << " per frame over " << num_occs.Sum()
715  << " frames (den count is " << den_occs.Sum() << ")";
716  }
717  }
718  model->c_[j2].CopyFromVec(weights);
719  tot_count += den_occs.Sum(); // Note: num and den occs should be the
720  // same, except num occs are smoothed, so this is what we want.
721  }
722 
723  tot_impr /= (tot_count + 1.0e-20);
724 
725  KALDI_LOG << "**Overall auxf impr for c is " << tot_impr
726  << " over " << tot_count << " frames";
727  return tot_impr;
728 }
