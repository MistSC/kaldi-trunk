137 // Note the similarity with SlidingWindowCmnOptions, but there
138 // are also differences. One which doesn't appear in the config
139 // itself, because it's a difference between the setups, is that
140 // in OnlineCmn, we carry over data from the previous utterance,
141 // or, if no previous utterance is available, from global stats,
142 // or, if previous utterances are available but the total amount
143 // of data is less than prev_frames, we pad with up to "global_frames"
144 // frames from the global stats.
145 struct OnlineCmvnOptions {
146  int32 cmn_window;
147  int32 speaker_frames; // must be <= cmn_window
148  int32 global_frames; // must be <= speaker_frames.
149  bool normalize_mean; // Must be true if normalize_variance==true.
150  bool normalize_variance;
151 
152  int32 modulus; // not configurable from command line, relates to how the
153  // class computes the cmvn internally. smaller->more
154  // time-efficient but less memory-efficient. Must be >= 1.
155  int32 ring_buffer_size; // not configurable from command line; size of ring
156  // buffer used for caching CMVN stats.
157  std::string skip_dims; // Colon-separated list of dimensions to skip normalization
158  // of, e.g. 13:14:15.
159 
160  OnlineCmvnOptions():
161  cmn_window(600),
162  speaker_frames(600),
163  global_frames(200),
164  normalize_mean(true),
165  normalize_variance(false),
166  modulus(20),
167  ring_buffer_size(20),
168  skip_dims("") { }
169 
170  void Check() {
171  KALDI_ASSERT(speaker_frames <= cmn_window && global_frames <= speaker_frames
172  && modulus > 0);
173  }
174 
175  void Register(ParseOptions *po) {
176  po->Register("cmn-window", &cmn_window, "Number of frames of sliding "
177  "context for cepstral mean normalization.");
178  po->Register("global-frames", &global_frames, "Number of frames of "
179  "global-average cepstral mean normalization stats to use for "
180  "first utterance of a speaker");
181  po->Register("speaker-frames", &speaker_frames, "Number of frames of "
182  "previous utterance(s) from this speaker to use in cepstral "
183  "mean normalization");
184  // we name the config string "norm-vars" for compatibility with
185  // ../featbin/apply-cmvn.cc
186  po->Register("norm-vars", &normalize_variance, "If true, do "
187  "cepstral variance normalization in addition to cepstral mean "
188  "normalization ");
189  po->Register("norm-mean", &normalize_mean, "If true, do mean normalization "
190  "(note: you cannot normalize the variance but not the mean)");
191  po->Register("skip-dims", &skip_dims, "Dimensions to skip normalization of "
192  "(colon-separated list of integers)");}
193 };
