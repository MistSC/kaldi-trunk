40 // For now we don't give the user a way to modify these from the command line.
41 struct CuAllocatorOptions {
42  // memory_factor is the total amount of (allocated + cached) memory that we
43  // allow to be held, relative to the max amount of memory the program has ever
44  // allocated. It will increase the amount of memory the program will
45  // potentially consume, by this factor.
46  BaseFloat memory_factor;
47 
48  // This is the minimum amount of memory that we will delete when we are forced
49  // to delete stuff, relative to the max amount of memory the program has ever
50  // allocated. This should be less than memory_factor - 1.0 and > 0. It
51  // shouldn't be too critical. The reason it exists is to avoid calling the
52  // cleanup code and only releasing very small amounts of memory, because there
53  // is a constant overhead proportional to the number of buckets.
54  BaseFloat delete_factor;
55 
56  CuAllocatorOptions(): memory_factor(1.5),
57  delete_factor(0.001) { }
58 
59  void Check() {
60  KALDI_ASSERT(delete_factor < memory_factor - 1.0 && delete_factor > 0.0);
61  }
62 };

139  struct MemoryRequestHasher {
140  // input is interpreted as (row_bytes, num_rows). row_bytes will always
141  // be a multiple of 4, and num_rows will frequently be a multiple of
142  // powers of 2 also. We need to shift right and add so that there will be
143  // some action in the lower-order bits.
144  size_t operator () (const std::pair<size_t,size_t> &p) const {
145  size_t temp = p.first + 1867 * p.second;
146  return temp + (temp >> 2) + (temp >> 8);
147  }
148  };
149 
150  MruCache() { }
151  // Define these to make inclusion in std::vector possible, but make them
152  // fail if called on anything but empty cache objects-- we never resize
153  // the vector of caches after initializing it.
154  MruCache &operator = (const MruCache &other);
155  MruCache(const MruCache &other);
156  private:
157  typedef std::list<MemoryRequest> ListType;
158  typedef std::list<MemoryRequest>::iterator ListIterType;
159  typedef std::deque<std::pair<CachedMemoryElement, ListIterType> > MapValueType;
160  typedef unordered_map<MemoryRequest, MapValueType,
161  MemoryRequestHasher> MapType;
162  // 'list_' contains MemoryRequests with the most recent on the back (where they are added),
163  // and least recent on the front (where they are removed by RemoveLeastRecentlyUsed, although
164  // they are also removed from random parts of the list by Lookup().
165  // There will in general be duplicates of MemoryRequests in the list, as

167  ListType list_;
168  // 'map_' maps from a MemoryRequest to a queue of (memory-element,

172  MapType map_;

208  struct PointerHasher {
209  size_t operator() (const void *arg) const {
210  // the last few bits tend to be very predictable, for alignment reasons (CUDA
211  // allocation may align on 256 byte or 512 byte boundaries or something similar).
212  size_t temp = reinterpret_cast<size_t>(arg);
213  return (temp >> 4) + (temp >> 9);
214  }
215  };
