62 template<typename Real>
63 void CuTpMatrix<Real>::Invert() {
64 #if HAVE_CUDA==1
65  if (CuDevice::Instantiate().Enabled()) {
66  if (this->num_rows_ == 0) return;
67  Timer tim;
68  int dimBlock(CU2DBLOCK);
69  int dimGrid(n_blocks(this->NumRows(), CU2DBLOCK));
70  CuMatrix<Real> tmp(this->NumRows(), this->NumRows());
71  int dim = this->NumRows();
72  Real alpha = 1.0;
73  cuda_set_diag(dimGrid, dimBlock, tmp.Data(), alpha, tmp.Dim());
74  CU_SAFE_CALL(cudaGetLastError());
75  CuMatrix<Real> tmp2(dim, dim);
76  tmp2.CopyFromTp(*this);
77  CU_SAFE_CALL(cublas_trsm(GetCublasHandle(), dim, dim, alpha, tmp2.Data(), tmp2.Dim().stride,
78  tmp.Data(), tmp.Dim().stride));
79  this->CopyFromMat(tmp, kNoTrans);
80  } else
81 #endif
82  {
83  Mat().Invert();
84  }

87 template<typename Real>
88 void CuTpMatrix<Real>::CopyFromMat(const CuMatrixBase<Real> &M,
89  MatrixTransposeType Trans) {
90 #if HAVE_CUDA==1
91  if (CuDevice::Instantiate().Enabled()) {
92  MatrixIndexT num_rows = this->num_rows_;
93  KALDI_ASSERT(num_rows == M.NumRows() && this->num_rows_ == M.NumCols());
94  if (num_rows == 0) return;
95  Timer tim;
96  dim3 dimBlock(CU2DBLOCK, CU2DBLOCK);
97  dim3 dimGrid(n_blocks(num_rows, CU2DBLOCK), n_blocks(num_rows, CU2DBLOCK));
98  if (Trans == kNoTrans) {
99  cuda_take_lower(dimGrid, dimBlock, M.Data(), this->data_, M.Dim());
100  } else {
101  cuda_take_upper(dimGrid, dimBlock, M.Data(), this->data_, M.Dim());
102  }
103  CU_SAFE_CALL(cudaGetLastError());
104  } else
105 #endif
106  {
107  Mat().CopyFromMat(M.Mat(), Trans);
108  }
109 }
