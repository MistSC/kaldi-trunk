40 void FeatureTransformEstimate::EstimateInternal(
41  const FeatureTransformEstimateOptions &opts,
42  const SpMatrix<double> &total_covar,
43  const SpMatrix<double> &between_covar,
44  const Vector<double> &total_mean,
45  Matrix<BaseFloat> *M,
46  TpMatrix<BaseFloat> *C) {
47 
48  int32 target_dim = opts.dim, dim = total_covar.NumRows();
49  // Interpret zero or negative target_dim as the full dim
50  if (target_dim <= 0)
51  target_dim = dim;
52  // between-class covar is of most rank C-1
53  KALDI_ASSERT(target_dim <= dim);
54 
55  // within-class covariance
56  SpMatrix<double> wc_covar(total_covar);
57  wc_covar.AddSp(-1.0, between_covar);
58  TpMatrix<double> wc_covar_sqrt(dim);
59  try {
60  wc_covar_sqrt.Cholesky(wc_covar);
61  if (C != NULL) {
62  C->Resize(dim);
63  C->CopyFromTp(wc_covar_sqrt);
64  }
65  } catch (...) {
66  BaseFloat smooth = 1.0e-03 * wc_covar.Trace() / wc_covar.NumRows();
67  KALDI_LOG << "Cholesky failed (possibly not +ve definite), so adding " << smooth
68  << " to diagonal and trying again.\n";
69  for (int32 i = 0; i < wc_covar.NumRows(); i++)
70  wc_covar(i, i) += smooth;
71  wc_covar_sqrt.Cholesky(wc_covar);
72  }
73  Matrix<double> wc_covar_sqrt_mat(wc_covar_sqrt);
74  wc_covar_sqrt_mat.Invert();
75 
76  SpMatrix<double> tmp_sp(dim);
77  tmp_sp.AddMat2Sp(1.0, wc_covar_sqrt_mat, kNoTrans, between_covar, 0.0);
78  Matrix<double> tmp_mat(tmp_sp);
79  Matrix<double> svd_u(dim, dim), svd_vt(dim, dim);
80  Vector<double> svd_d(dim);
81  tmp_mat.Svd(&svd_d, &svd_u, &svd_vt);
82  SortSvd(&svd_d, &svd_u);
83 
84  KALDI_LOG << "LDA singular values are " << svd_d;
85 
86  KALDI_LOG << "Sum of all singular values is " << svd_d.Sum();
87  KALDI_LOG << "Sum of selected singular values is " <<
88  SubVector<double>(svd_d, 0, target_dim).Sum();
89 
90  Matrix<double> lda_mat(dim, dim);
91  lda_mat.AddMatMat(1.0, svd_u, kTrans, wc_covar_sqrt_mat, kNoTrans, 0.0);
92 
93  // finally, copy first target_dim rows to m
94  M->Resize(target_dim, dim);
95  M->CopyFromMat(lda_mat.Range(0, target_dim, 0, dim));
96 
97  if (opts.within_class_factor != 1.0) {
98  for (int32 i = 0; i < svd_d.Dim(); i++) {
99  BaseFloat old_var = 1.0 + svd_d(i), // the total variance of that dim..
100  new_var = opts.within_class_factor + svd_d(i), // the variance we want..
101  scale = sqrt(new_var / old_var);
102  if (i < M->NumRows())
103  M->Row(i).Scale(scale);
104  }
105  }
106 
107  if (opts.max_singular_value > 0.0) {
108  int32 rows = M->NumRows(), cols = M->NumCols(),
109  min_dim = std::min(rows, cols);
110  Matrix<BaseFloat> U(rows, min_dim), Vt(min_dim, cols);
111  Vector<BaseFloat> s(min_dim);
112  M->Svd(&s, &U, &Vt); // decompose m = U diag(s) Vt.
113  BaseFloat max_s = s.Max();
114  int32 n = s.ApplyCeiling(opts.max_singular_value);
115  if (n > 0) {
116  KALDI_LOG << "Applied ceiling to " << n << " out of " << s.Dim()
117  << " singular values of transform using ceiling "
118  << opts.max_singular_value << ", max is " << max_s;
119  Vt.MulRowsVec(s);
120  // reconstruct m with the modified singular values:
121  M->AddMatMat(1.0, U, kNoTrans, Vt, kNoTrans, 0.0);
122  }
123  }
124 
125  if (opts.remove_offset)
126  AddMeanOffset(total_mean, M);
127 }

129 void FeatureTransformEstimateMulti::EstimateTransformPart(
130  const FeatureTransformEstimateOptions &opts,
131  const std::vector<int32> &indexes,
132  const SpMatrix<double> &total_covar,
133  const SpMatrix<double> &between_covar,
134  const Vector<double> &mean,
135  Matrix<BaseFloat> *M) const {
136 
137  int32 full_dim = Dim(), proj_dim = indexes.size();
138  Matrix<double> transform(proj_dim, full_dim); // projects from full to projected dim.
139  for (int32 i = 0; i < proj_dim; i++)
140  transform(i, indexes[i]) = 1.0;
141 
142  SpMatrix<double> total_covar_proj(proj_dim), between_covar_proj(proj_dim);
143  Vector<double> mean_proj(proj_dim);
144  total_covar_proj.AddMat2Sp(1.0, transform, kNoTrans, total_covar, 0.0);
145  between_covar_proj.AddMat2Sp(1.0, transform, kNoTrans, between_covar, 0.0);
146  mean_proj.AddMatVec(1.0, transform, kNoTrans, mean, 0.0);
147 
148  Matrix<BaseFloat> M_proj;
149  FeatureTransformEstimateOptions opts_tmp(opts);
150  opts_tmp.dim = proj_dim;
151  EstimateInternal(opts_tmp, total_covar_proj, between_covar_proj, mean_proj,
152  &M_proj, NULL);
153  if (M_proj.NumCols() == proj_dim + 1) { // Extend transform to add the extra "1" that we
154  // use to handle mean shifts..
155  transform.Resize(proj_dim + 1, full_dim + 1, kCopyData);
156  transform(proj_dim, full_dim) = 1.0;
157  }
158  M->Resize(proj_dim, transform.NumCols());
159  // Produce output..
160  M->AddMatMat(1.0, M_proj, kNoTrans, Matrix<BaseFloat>(transform),
161  kNoTrans, 0.0);
162 }

164 void FeatureTransformEstimateMulti::Estimate(
165  const FeatureTransformEstimateOptions &opts,
166  const std::vector<std::vector<int32> > &indexes,
167  Matrix<BaseFloat> *M) const {
168 
169  int32 input_dim = Dim(), output_dim = 0, num_transforms = indexes.size();
170  for (int32 i = 0; i < num_transforms; i++) { // some input-checking.
171  KALDI_ASSERT(indexes[i].size() > 0);
172  std::vector<int32> this_indexes(indexes[i]);
173  std::sort(this_indexes.begin(), this_indexes.end());
174  KALDI_ASSERT(IsSortedAndUniq(this_indexes)); // check for duplicates.
175  KALDI_ASSERT(this_indexes.front() >= 0);
176  KALDI_ASSERT(this_indexes.back() < input_dim);
177  output_dim += this_indexes.size();
178  }
179 
180  int32 input_dim_ext = (opts.remove_offset ? input_dim + 1 : input_dim);
181  M->Resize(output_dim, input_dim_ext);
182 
183  double count;
184  Vector<double> total_mean;
185  SpMatrix<double> total_covar, between_covar;
186  GetStats(&total_covar, &between_covar, &total_mean, &count);
187 
188  int32 cur_output_index = 0;
189  for (int32 i = 0; i < num_transforms; i++) {
190  Matrix<BaseFloat> M_tmp;
191  EstimateTransformPart(opts, indexes[i], total_covar, between_covar,
192  total_mean, &M_tmp);
193  int32 this_output_dim = indexes[i].size();
194  M->Range(cur_output_index, this_output_dim, 0, M->NumCols()).
195  CopyFromMat(M_tmp);
196  cur_output_index += this_output_dim;
197  }
198 
199 }
