39  "Markov Chain Monte-Carlo.\n"

113  const BaseFloat& momentum_max = trn_opts.momentum_max;

155  // skip too long segments (avoid runinning out of memory)

178  }
179 
180  // randomize
181  feature_randomizer.Randomize(randomizer_mask.Generate(feature_randomizer.NumFrames()));
182 
183  // train with data from randomizer (using mini-batches)
184  for( ; !feature_randomizer.Done(); feature_randomizer.Next()) {
185  // get block of feature/target pairs
186  const CuMatrixBase<BaseFloat>& pos_vis = feature_randomizer.Value();
187  // get the dims
188  int32 num_frames = pos_vis.NumRows(),
189  dim_hid = rbm.OutputDim();
190  // Create dummy frame-weights for Mse::Eval,
191  Vector<BaseFloat> dummy_weights(num_frames);
192  dummy_weights.Set(1.0);
193 
194  // TRAIN with CD1
195  // forward pass
196  rbm.Propagate(pos_vis, &pos_hid);
197 
198  // alter the hidden values, so we can generate negative example
199  if (rbm.HidType() == Rbm::Bernoulli) {
200  pos_hid_aux.Resize(num_frames, dim_hid);
201  cu_rand.BinarizeProbs(pos_hid, &pos_hid_aux);
202  } else {
203  // assume HidType Rbm::GAUSSIAN
204  pos_hid_aux.Resize(num_frames, dim_hid);
205  pos_hid_aux.CopyFromMat(pos_hid);
206  cu_rand.AddGaussNoise(&pos_hid_aux);
207  }
208 
209  // reconstruct pass
210  rbm.Reconstruct(pos_hid_aux, &neg_vis);
211  // propagate negative examples
212  rbm.Propagate(neg_vis, &neg_hid);
213  // update step
214  rbm.RbmUpdate(pos_vis, pos_hid, neg_vis, neg_hid);
215  // evaluate mean square error
216  mse.Eval(dummy_weights, neg_vis, pos_vis, &dummy_mse_mat);
217 
218  total_frames += num_frames;
219 
220  // change the momentum progressively per 0.5million samples of the data
221  {
222  static int32 n_prev = -1;
223  BaseFloat step = (momentum_max - momentum) / momentum_steps;
224  int32 n = total_frames / momentum_step_period; //change every momentum_step_period data
225  BaseFloat momentum_actual;
226  if(n > momentum_steps) {
227  momentum_actual = momentum_max;
228  } else {
229  momentum_actual = momentum + n*step;
230  }
231  if(n - n_prev > 0) {
232  n_prev = n;
233  BaseFloat learning_rate_actual = learn_rate*(1-momentum_actual);
234  KALDI_VLOG(1) << "Setting momentum " << (with_bug ? momentum_max : momentum_actual)
235  << " and learning rate " << learning_rate_actual
236  << " after processing "
237  << static_cast<double>(total_frames)/360000 << "h";
238  // pass values to rbm
239  trn_opts_rbm.momentum = (with_bug ? momentum_max : momentum_actual);
240  trn_opts_rbm.learn_rate = learning_rate_actual;
241  rbm.SetRbmTrainOptions(trn_opts_rbm);
242  }
243  }
244  }
245 
246  // reopen the feature stream if we will run another iteration
247  if (feature_reader.Done() && (iter < num_iters)) {
248  iter++;
249  KALDI_LOG << "Iteration " << iter << "/" << num_iters;
250  feature_reader.Close();
251  feature_reader.Open(feature_rspecifier);
252  }
253  }
254 
255  nnet.Write(target_model_filename, binary);
256 
257  KALDI_LOG << "Done " << iter << " iterations, " << num_done << " files, "
258  << "skipped " << num_other_error << " files. "
259  << "[" << time.Elapsed()/60 << " min, fps" << total_frames/time.Elapsed()
260  << "]";
261 
262  KALDI_LOG << mse.Report();
263 
264 #if HAVE_CUDA==1
265  CuDevice::Instantiate().PrintProfile();
266 #endif
267  return 0;
268  } catch(const std::exception &e) {
269  std::cerr << e.what();
270  return -1;
271  }
