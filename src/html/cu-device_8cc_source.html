60 static bool GetCudaContext(int32 num_gpus, std::string *debug_str) {
61  std::ostringstream debug_stream;
62  debug_stream << "num-gpus=" << num_gpus << ". ";
63  for (int32 device = 0; device < num_gpus; device++) {
64  cudaSetDevice(device);
65  cudaError_t e = cudaDeviceSynchronize(); // << CUDA context gets created here.
66  if (e == cudaSuccess) {
67  *debug_str = debug_stream.str();
68  return true;
69  }
70  debug_stream << "Device " << device << ": " << cudaGetErrorString(e) << ". ";
71  cudaGetLastError(); // Make sure the error state doesn't get returned in
72  // the next cudaGetLastError().
73  }
74  *debug_str = debug_stream.str();
75  return false;
76 }

97 void CuDevice::SelectGpuId(std::string use_gpu) {
98  // Possible modes
99  if (use_gpu != "yes" && use_gpu != "no" && use_gpu != "optional" && use_gpu != "wait") {
100  KALDI_ERR << "Please choose : --use-gpu=yes|no|optional|wait, passed '" << use_gpu << "'";
101  }
102 
103  // Make sure this function is not called twice!
104  if (Enabled()) {
105  KALDI_ERR << "There is already an active GPU " << active_gpu_id_
106  << ", cannot change it on the fly!";
107  }
108  // Allow the GPU to stay disabled
109  if (!Enabled() && use_gpu == "no") {
110  KALDI_LOG << "Manually selected to compute on CPU.";
111  return;
112  }
113 
114  // Check that we have a gpu available
115  int32 num_gpus = 0;
116 
117  cudaError_t e = cudaGetDeviceCount(&num_gpus);
118 
119  if (num_gpus == 0) {
120  if (use_gpu == "yes" || use_gpu == "wait") {
121  KALDI_CUDA_ERR(e, "No CUDA GPU detected!");
122  }
123  if (use_gpu == "optional") {
124  KALDI_WARN << "Running on CPU!!! No CUDA GPU detected...";
125  return;
126  }
127  }
128 
129  // Create a CUDA context.
130  std::string debug_str;
131  bool got_context = GetCudaContext(num_gpus, &debug_str);
132 
133  if (use_gpu != "wait") {
134  if (!got_context) {
135  // So far no we don't have context, sleep a bit and retry.
136  int32 sec_sleep = (use_gpu == "yes" ? 20 : 2);
137  KALDI_WARN << "Will try again to get a GPU after " << sec_sleep
138  << " seconds.";
139  Sleep(sec_sleep);
140  if (!GetCudaContext(num_gpus, &debug_str)) {
141  if (use_gpu == "yes") {
142  {
143  Input input;
144  input.Open("nvidia-smi 1>&2 |");
145  }
146  KALDI_LOG << debug_str;
147  KALDI_ERR << "Failed to create CUDA context, no more unused GPUs? ";
148  }
149  if (use_gpu == "optional") {
150  KALDI_WARN << "Running on CPU!!! No more unused CUDA GPUs?";
151  return;
152  }
153  }
154  }
155  } else {
156  int32 num_times = 0;
157  BaseFloat wait_time = 0.0;
158  while (! got_context) {
159  int32 sec_sleep = 5;
160  if (num_times == 0)
161  KALDI_WARN << "Will try again indefinitely every " << sec_sleep
162  << " seconds to get a GPU.";
163  num_times++;
164  wait_time += sec_sleep;
165  Sleep(sec_sleep);
166  got_context = GetCudaContext(num_gpus, &debug_str);
167  }
168 
169  KALDI_WARN << "Waited " << wait_time
170  << " seconds before creating CUDA context";
171  }
172 
173  // Re-assure we have the context
174  KALDI_ASSERT(cudaSuccess == cudaThreadSynchronize());
175 
176  // Check if the machine use compute exclusive mode
177  if (IsComputeExclusive()) {
178  FinalizeActiveGpu();
179  return;
180  } else {
181  // Or suggest to use compute exclusive mode
182  if (num_gpus > 1) {
183  KALDI_WARN << "Suggestion: use 'nvidia-smi -c 1' to set compute exclusive mode";
184  }
185  // And select the GPU according to proportion of free memory
186  if (SelectGpuIdAuto()) {
187  FinalizeActiveGpu();
188  return;
189  } else {
190  // Could not get GPU, after prevously having the CUDA context?
191  // Strange but not impossible...
192  if (use_gpu == "yes") {
193  KALDI_ERR << "Error acquiring GPU.";
194  }
195  if (use_gpu == "optional") {
196  KALDI_WARN << "Running on CPU!!! Error acquiring GPU.";
197  return;
198  }
199  }
200  }
201 }

204 void CuDevice::FinalizeActiveGpu() {
205  // The device at this point should have active GPU, so we can query its name
206  // and memory stats and notify user which GPU is finally used.
207 
208  // Get the device-id of active device:
209  {
210  int32 act_gpu_id;
211  cudaError_t e = cudaGetDevice(&act_gpu_id);
212  if (e != cudaSuccess) {
213  KALDI_CUDA_ERR(e, "Failed to get device-id of active device.");
214  }
215  // Remember the id of active GPU
216  active_gpu_id_ = act_gpu_id; // CuDevice::Enabled() is true from now on
217  // Initialize the CUBLAS
218  CU_SAFE_CALL(cublasCreate(&handle_));
219 
220  // Notify user which GPU is finally used
221  char name[128];
222  DeviceGetName(name,128,act_gpu_id);
223 
224  CU_SAFE_CALL(cudaGetDeviceProperties(&properties_, act_gpu_id));
225 
226  KALDI_LOG << "The active GPU is [" << act_gpu_id << "]: " << name << "\t"
227  << GetFreeMemory(&free_memory_at_startup_, NULL) << " version "
228  << properties_.major << "." << properties_.minor;
229  }
230  return;
231 }

241 bool CuDevice::IsComputeExclusive() {
242  // assume we already have an CUDA context created
243  KALDI_ASSERT(cudaSuccess == cudaThreadSynchronize());
244 
245  // get the device-id and its device-properties
246  int32 gpu_id = -1;
247  cudaError_t e = cudaGetDevice(&gpu_id);
248  if (e != cudaSuccess) {
249  KALDI_CUDA_ERR(e, "Failed to get current device");
250  }
251  struct cudaDeviceProp gpu_prop;
252  e = cudaGetDeviceProperties(&gpu_prop, gpu_id);
253  if (e != cudaSuccess) {
254  KALDI_CUDA_ERR(e, "Failed to get device properties");
255  }
256  // find out whether compute exclusive mode is used
257  switch (gpu_prop.computeMode) {
258  case cudaComputeModeExclusive :
259  KALDI_LOG << "CUDA setup operating under Compute Exclusive Mode.";
260  return true;
261  break;
262 #if (CUDA_VERSION >= 4000)
263  case cudaComputeModeExclusiveProcess :
264  KALDI_LOG << "CUDA setup operating under Compute Exclusive Process Mode.";
265  return true;
266  break;
267 #endif
268  default :
269  // The computation mode is not compute-exclusive,
270  // in this case we release the GPU context...
271  e = cudaThreadExit(); // deprecated, but for legacy reason not cudaDeviceReset
272  if (e != cudaSuccess) {
273  KALDI_CUDA_ERR(e, "Failed to release CUDA context on a GPU");
274  }
275  return false;
276  }

284 bool CuDevice::SelectGpuIdAuto() {
285  // Check that we have at least one gpu
286  int32 num_gpus = 0;
287  cudaError_t e = cudaGetDeviceCount(&num_gpus);
288  if (num_gpus == 0) {
289  KALDI_WARN << "No CUDA devices found";
290  if (e != cudaSuccess) {
291  KALDI_WARN << "cudaGetDeviceCount() returned " << e
292  <<", meaning: \"" << cudaGetErrorString(e) << "\"";
293  }
294  return false;
295  }
296 
297  // The GPU is selected according to maximal free memory ratio
298  std::vector< std::pair<int, float> > free_mem_ratio(num_gpus);
299 
300  // Get ratios of memory use, if possible
301  KALDI_LOG << "Selecting from " << num_gpus << " GPUs";
302  for(int32 n = 0; n < num_gpus; n++) {
303  int32 ret = cudaSetDevice(n);
304  switch(ret) {
305  case cudaSuccess : {
306  // create the CUDA context for the thread
307  cudaThreadSynchronize(); // deprecated, but for legacy not cudaDeviceSynchronize
308  // get GPU name
309  char name[128];
310  DeviceGetName(name,128,n);
311  // get GPU memory stats
312  int64 free, total;
313  std::string mem_stats;
314  mem_stats = GetFreeMemory(&free, &total);
315  // log
316  KALDI_LOG << "cudaSetDevice(" << n << "): "
317  << name << "\t" << mem_stats;
318 
319  // We have seen that in some cases GetFreeMemory returns zero
320  // That will produce nan after division, which might confuse
321  // the sorting routine. Or maybe not, but let's keep it clean
322  if (total <= 0) {
323  KALDI_LOG << "Total memory reported for device " << n << " is zero (or less).";
324  }
325  float mem_ratio = total > 0 ? free/(float)total : 0;
326  free_mem_ratio[n] = std::make_pair(n, mem_ratio);
327 
328  // destroy the CUDA context for the thread
329  cudaThreadExit(); // deprecated, but for legacy reason not cudaDeviceReset
330  } break;
331 
332 #if (CUDA_VERSION > 3020)
333  case cudaErrorDeviceAlreadyInUse :
334  KALDI_LOG << "cudaSetDevice(" << n << "): "
335  << "Device cannot be accessed, used EXCLUSIVE-THREAD mode...";
336  break;
337 #endif
338  case cudaErrorInvalidDevice :
339  KALDI_LOG << "cudaSetDevice(" << n << "): "
340  << "Device cannot be accessed, not a VALID CUDA device!";
341  break;
342  default :
343  KALDI_LOG << "cudaSetDevice(" << n << "): "
344  << "returned " << ret << ", "
345  << cudaGetErrorString((cudaError_t)ret);
346  }
347  }
348  // find GPU with max free memory
349  int32 max_id=0;
350  std::sort(free_mem_ratio.begin(), free_mem_ratio.end(),
351  greater_pair<int, float>);
352  // the free_mem_ratio should be bigger than zero
353  KALDI_ASSERT(free_mem_ratio[max_id].second > 0.0);
354 
355  float dev_id;
356  float mem_ratio;
357  do {
358  // try to select the GPU in the best to worst order
359  // Note we have to check the return codes manually, as the CU_SAFE_CALL
360  // contains call to KALDI_ERR (which will cause the program to abort)
361 
362  dev_id = free_mem_ratio[max_id].first;
363  mem_ratio = free_mem_ratio[max_id].second;
364 
365  KALDI_LOG << "Trying to select device: " << dev_id << " (automatically), mem_ratio: " << mem_ratio;
366  e = cudaSetDevice(dev_id);
367  if (e != cudaSuccess) {
368  KALDI_WARN << "Cannot select this device: return code " << e
369  << ", Error message: \"" << cudaGetErrorString(e) << "\"";
370  } else {
371  e = cudaThreadSynchronize(); // deprecated, but for legacy not cudaDeviceSynchronize
372  if (e != cudaSuccess) {
373  KALDI_WARN << "Cannot select this device: return code " << e
374  << ", Error message: \"" << cudaGetErrorString(e) << "\"";
375  }
376  }
377  max_id++;
378  } while ((e != cudaSuccess) && (max_id < free_mem_ratio.size()));
379 
380  if (e != cudaSuccess) {
381  KALDI_WARN << "Failed to (automatically) select any device";
382  return false;
383  }
384  KALDI_LOG << "Success selecting device " << dev_id << " free mem ratio: " << mem_ratio;
385  return true;
386 }

406 void CuDevice::PrintProfile() {
407  if (verbose_ && Enabled()) {
408  std::ostringstream os;
409  os << "-----\n[cudevice profile]\n";
410  unordered_map<std::string, double, StringHasher>::iterator it;
411  std::vector<std::pair<double, std::string> > pairs;
412  double total_time = 0.0;
413  for(it = profile_map_.begin(); it != profile_map_.end(); ++it) {
414  std::string function_name = it->first;
415  double elapsed_time = it->second;
416  total_time += elapsed_time;
417  pairs.push_back(std::make_pair(elapsed_time, function_name));
418  }
419  // display from shortest to longest time, so tail will show the longest
420  // times at the end.
421  std::sort(pairs.begin(), pairs.end());
422  size_t max_print = 15, start_pos = (pairs.size() <= max_print ?
423  0 : pairs.size() - max_print);
424  for (size_t i = start_pos; i < pairs.size(); i++)
425  os << pairs[i].second << "\t" << pairs[i].first << "s\n";
426  os << "Total GPU time:\t" << total_time << "s (may involve some double-counting)\n";
427  os << "-----";
428  KALDI_LOG << os.str();
429  PrintMemoryUsage();
430  }
431 }

434 std::string CuDevice::GetFreeMemory(int64* free, int64* total) const {
435  // WARNING! the CUDA API is inconsistent accross versions!
436 #ifdef _MSC_VER
437  size_t mem_free, mem_total;
438  cuMemGetInfo_v2(handle_, &mem_free, &mem_total);
439 #else
440 #if (CUDA_VERSION >= 3020)
441  // define the function signature type
442  size_t mem_free, mem_total;
443 #else
444  unsigned int mem_free, mem_total;
445 #endif
446  {
447  // we will load cuMemGetInfo_v2 dynamically from libcuda.so
448  // pre-fill ``safe'' values that will not cause problems
449  mem_free = 1; mem_total = 1;
450 #ifdef _MSC_VER
451  cuMemGetInfo_v2(handle_, &mem_free, &mem_total);
452 #else
453  // open libcuda.so
454  void* libcuda = dlopen("libcuda.so",RTLD_LAZY);
455  if (NULL == libcuda) {
456  KALDI_WARN << "cannot open libcuda.so";
457  } else {
458  // define the function signature type
459  // and get the symbol
460 #if (CUDA_VERSION >= 3020)
461  typedef CUresult (*cu_fun_ptr)(size_t*, size_t*);
462  cu_fun_ptr dl_cuMemGetInfo = (cu_fun_ptr)dlsym(libcuda,"cuMemGetInfo_v2");
463 #else
464  typedef CUresult (*cu_fun_ptr)(int*, int*);
465  cu_fun_ptr dl_cuMemGetInfo = (cu_fun_ptr)dlsym(libcuda,"cuMemGetInfo");
466 #endif
467  if (NULL == dl_cuMemGetInfo) {
468  KALDI_WARN << "cannot load cuMemGetInfo from libcuda.so";
469  } else {
470  // call the function
471  dl_cuMemGetInfo(&mem_free, &mem_total);
472  }
473  // close the library
474  dlclose(libcuda);
475  }
476 #endif
477  }
478 #endif
479  // copy the output values outside
480  if (NULL != free) *free = mem_free;
481  if (NULL != total) *total = mem_total;
482  // prepare the text output
483  std::ostringstream os;
484  os << "free:" << mem_free/(1024*1024) << "M, "
485  << "used:" << (mem_total-mem_free)/(1024*1024) << "M, "
486  << "total:" << mem_total/(1024*1024) << "M, "
487  << "free/total:" << mem_free/(float)mem_total;
488  return os.str();
489 }
