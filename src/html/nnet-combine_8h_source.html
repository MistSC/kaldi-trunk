36 struct NnetCombineConfig {
37  int32 num_iters; // The dimension of the space we are optimizing in is fairly
38  // small (equal to the number of components times the number
39  // of neural nets we were given), so we optimize with BFGS
40  // (internally the code uses L-BFGS, but we set the the
41  // number of vectors to be the same as the dimension of the
42  // space, so it actually is regular BFGS. The num-iters
43  // corresponds to the number of function evaluations.
44 
45 
46  BaseFloat initial_impr;
47  int32 max_effective_inputs;
48  bool test_gradient;
49  bool enforce_positive_weights;
50  bool enforce_sum_to_one;
51  bool separate_weights_per_component;
52  NnetCombineConfig(): num_iters(60),
53  initial_impr(0.01),
54  max_effective_inputs(15),
55  test_gradient(false),
56  enforce_positive_weights(false),
57  enforce_sum_to_one(false),
58  separate_weights_per_component(true) { }
59 
60  void Register(OptionsItf *po) {
61  po->Register("num-iters", &num_iters, "Maximum number of function "
62  "evaluations for BFGS to use when optimizing combination weights");
63  po->Register("max-effective-inputs", &max_effective_inputs, "Limits the number of "
64  "parameters that have to be learn to be equivalent to the number of "
65  "parameters we'd have to learn if the number of inputs nnets equalled "
66  "this number. Does this by using averages of nnets at close points "
67  "in the sequence of inputs, as the actual inputs to the computation.");
68  po->Register("initial-impr", &initial_impr, "Amount of objective-function change "
69  "we aim for on the first iteration (controls the initial step size).");
70  po->Register("test-gradient", &test_gradient, "If true, activate code that "
71  "tests the gradient is accurate.");
72  po->Register("enforce-positive-weights", &enforce_positive_weights,
73  "If true, enforce that all weights are positive.");
74  po->Register("enforce-sum-to-one", &enforce_sum_to_one, "If true, enforce that "
75  "the model weights for each component should sum to one.");
76  po->Register("separate-weights-per-component", &separate_weights_per_component,
77  "If true, have a separate weight for each updatable component in "
78  "the nnet.");
79  }
80 };

83 /*
84  You should use this class as follows:
85  - Call the constructor, giving it the egs and the first nnet.
86  - Call AcceptNnet to provide all the other nnets. (the nnets will
87  be stored in a matrix in CPU memory, to avoid filing up GPU memory).
88  - Call Combine()
89  - Get the resultant nnet with GetNnet().
90  */
