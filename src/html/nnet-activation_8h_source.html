73  void InitData(std::istream &is) {
74  // parse config
75  std::string token,
76  dims_str;
77  while (!is.eof()) {
78  ReadToken(is, false, &token);
79  if (token == "<BlockDims>") is >> dims_str;
80  else KALDI_ERR << "Unknown token " << token << ", a typo in config?"
81  << " (BlockDims)";
82  is >> std::ws; // eat-up whitespace
83  }
84  // parse dims,
85  if (!kaldi::SplitStringToIntegers(dims_str, ",:", false, &block_dims))
86  KALDI_ERR << "Invalid block-dims " << dims_str;
87  // sanity check
88  int32 sum = 0;
89  for (int32 i=0; i<block_dims.size(); i++) {
90  sum += block_dims[i];
91  }
92  KALDI_ASSERT(sum == OutputDim());
93  }

95  void ReadData(std::istream &is, bool binary) {
96  ReadIntegerVector(is, binary, &block_dims);
97  block_offset.resize(block_dims.size()+1, 0);
98  for (int32 i = 0; i < block_dims.size(); i++) {
99  block_offset[i+1] = block_offset[i] + block_dims[i];
100  }
101  // check
102  KALDI_ASSERT(OutputDim() == block_offset[block_offset.size()-1]);
103  }

109  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
110  // perform softmax per block:
111  for (int32 bl = 0; bl < block_dims.size(); bl++) {
112  CuSubMatrix<BaseFloat> in_bl = in.ColRange(block_offset[bl], block_dims[bl]);
113  CuSubMatrix<BaseFloat> out_bl = out->ColRange(block_offset[bl], block_dims[bl]);
114  // y = e^x_j/sum_j(e^x_j)
115  out_bl.ApplySoftMaxPerRow(in_bl);
116  }
117  }

119  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
120  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
121  // copy the error derivative:
122  // (assuming we already got softmax-cross-entropy derivative in out_diff)
123  in_diff->CopyFromMat(out_diff);
124 
125  // zero-out line-in-block, where sum different from zero,
126  // process per block:
127  for (int32 bl = 0; bl < block_dims.size(); bl++) {
128  CuSubMatrix<BaseFloat> diff_bl = in_diff->ColRange(block_offset[bl], block_dims[bl]);
129  CuVector<BaseFloat> row_sum(diff_bl.NumRows());
130  row_sum.AddColSumMat(1.0, diff_bl, 0.0); // 0:keep, 1:zero-out
131  // we'll scale rows by 0/1 masks
132  CuVector<BaseFloat> row_diff_mask(row_sum);
133  row_diff_mask.Scale(-1.0); // 0:keep, -1:zero-out
134  row_diff_mask.Add(1.0); // 1:keep, 0:zero-out
135  // here we should have only 0 and 1
136  diff_bl.MulRowsVec(row_diff_mask);
137  }
138  }

142  }

238  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
239  out->CopyFromMat(in);
240  // switch off 50% of the inputs...
241  dropout_mask_.Resize(out->NumRows(),out->NumCols());
242  dropout_mask_.Set(dropout_retention_);
243  rand_.BinarizeProbs(dropout_mask_,&dropout_mask_);
244  out->MulElements(dropout_mask_);
245  // rescale to keep same dynamic range as w/o dropout
246  out->Scale(1.0/dropout_retention_);
247  }

249  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
250  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
251  in_diff->CopyFromMat(out_diff);
252  // use same mask on the error derivatives...
253  in_diff->MulElements(dropout_mask_);
254  // enlarge output to fit dynamic range w/o dropout
255  in_diff->Scale(1.0/dropout_retention_);
256  }
