123 // This class allows you to compute the matrix exponential function
124 // B = I + A + 1/2! A^2 + 1/3! A^3 + ...
125 // This method is most accurate where the result is of the same order of
126 // magnitude as the unit matrix (it will typically not work well when
127 // the answer has almost-zero eigenvalues or is close to zero).
128 // It also provides a function that allows you do back-propagate the
129 // derivative of a scalar function through this calculation.
130 // The
131 template<typename Real>
132 class MatrixExponential {
133  public:
134  MatrixExponential() { }
135 
136  void Compute(const MatrixBase<Real> &M, MatrixBase<Real> *X); // does *X = exp(M)
137 
138  // Version for symmetric matrices (it just copies to full matrix).
139  void Compute(const SpMatrix<Real> &M, SpMatrix<Real> *X); // does *X = exp(M)
140 
141  void Backprop(const MatrixBase<Real> &hX, MatrixBase<Real> *hM) const; // Propagates
142  // the gradient of a scalar function f backwards through this operation, i.e.:
143  // if the parameter dX represents df/dX (with no transpose, so element i, j of dX
144  // is the derivative of f w.r.t. E(i, j)), it sets dM to df/dM, again with no
145  // transpose (of course, only the part thereof that comes through the effect of
146  // A on B). This applies to the values of A and E that were called most recently
147  // with Compute().
148 
149  // Version for symmetric matrices (it just copies to full matrix).
150  void Backprop(const SpMatrix<Real> &hX, SpMatrix<Real> *hM) const;
151 
152  private:
153  void Clear();

155  static MatrixIndexT ComputeN(const MatrixBase<Real> &M);
156 
157  // This is intended for matrices P with small norms: compute B_0 = exp(P) - I.
158  // Keeps adding terms in the Taylor series till there is no further
159  // change in the result. Stores some of the powers of A in powers_,
160  // and the number of terms K as K_.
161  void ComputeTaylor(const MatrixBase<Real> &P, MatrixBase<Real> *B0);
162 
163  // Backprop through the Taylor-series computation above.
164  // note: hX is \hat{X} in the math; hM is \hat{M} in the math.
165  void BackpropTaylor(const MatrixBase<Real> &hX,
166  MatrixBase<Real> *hM) const;
167 
168  Matrix<Real> P_; // Equals M * 2^(-N_)
169  std::vector<Matrix<Real> > B_; // B_[0] = exp(P_) - I,
170  // B_[k] = 2 B_[k-1] + B_[k-1]^2 [k > 0],
171  // ( = exp(P_)^k - I )
172  // goes from 0..N_ [size N_+1].
173 
174  std::vector<Matrix<Real> > powers_; // powers (>1) of P_ stored here,
175  // up to all but the last one used in the Taylor expansion (this is the
176  // last one we need in the backprop). The index is the power minus 2.
177 
178  MatrixIndexT N_; // Power N_ >=0 such that P_ = A * 2^(-N_),
179  // we choose it so that P_ has a sufficiently small norm
180  // that the Taylor series will converge fast.
181 };
