46 void LatticeAcousticRescore(const Matrix<BaseFloat> &log_like,
47  const TransitionModel &trans_model,
48  const std::vector<int32> &state_times,
49  Lattice *lat) {
50  kaldi::uint64 props = lat->Properties(fst::kFstProperties, false);
51  if (!(props & fst::kTopSorted))
52  KALDI_ERR << "Input lattice must be topologically sorted.";
53 
54  KALDI_ASSERT(!state_times.empty());
55  std::vector<std::vector<int32> > time_to_state(log_like.NumRows());
56  for (size_t i = 0; i < state_times.size(); i++) {
57  KALDI_ASSERT(state_times[i] >= 0);
58  if (state_times[i] < log_like.NumRows()) // end state may be past this..
59  time_to_state[state_times[i]].push_back(i);
60  else
61  KALDI_ASSERT(state_times[i] == log_like.NumRows()
62  && "There appears to be lattice/feature mismatch.");
63  }
64 
65  for (int32 t = 0; t < log_like.NumRows(); t++) {
66  for (size_t i = 0; i < time_to_state[t].size(); i++) {
67  int32 state = time_to_state[t][i];
68  for (fst::MutableArcIterator<Lattice> aiter(lat, state); !aiter.Done();
69  aiter.Next()) {
70  LatticeArc arc = aiter.Value();
71  int32 trans_id = arc.ilabel;
72  if (trans_id != 0) { // Non-epsilon input label on arc
73  int32 pdf_id = trans_model.TransitionIdToPdf(trans_id);
74  arc.weight.SetValue2(-log_like(t, pdf_id) + arc.weight.Value2());
75  aiter.SetValue(arc);
76  }
77  }
78  }
79  }
80 }

94  "The network weights are updated on each utterance.\n"
95  "Usage: nnet-train-mmi-sequential [options] <model-in> <transition-model-in> "

136  po.Read(argc, argv);

144  transition_model_filename = po.GetArg(2),

146  den_lat_rspecifier = po.GetArg(4),

165  }
166 
167  Nnet nnet;
168  nnet.Read(model_filename);
169  // using activations directly: remove softmax, if present
170  if (nnet.GetComponent(nnet.NumComponents()-1).GetType() ==
171  kaldi::nnet1::Component::kSoftmax) {
172  KALDI_LOG << "Removing softmax from the nnet " << model_filename;
173  nnet.RemoveComponent(nnet.NumComponents()-1);
174  } else {
175  KALDI_LOG << "The nnet was without softmax " << model_filename;
176  }
177  nnet.SetTrainOptions(trn_opts);
178 
179  // Read the class-frame-counts, compute priors
180  PdfPrior log_prior(prior_opts);
181 
182  // Read transition model
183  TransitionModel trans_model;
184  ReadKaldiObject(transition_model_filename, &trans_model);
185 
186  SequentialBaseFloatMatrixReader feature_reader(feature_rspecifier);
187  RandomAccessLatticeReader den_lat_reader(den_lat_rspecifier);
188  RandomAccessInt32VectorReader num_ali_reader(num_ali_rspecifier);
189 
190  CuMatrix<BaseFloat> feats, feats_transf, nnet_out, nnet_diff;
191  Matrix<BaseFloat> nnet_out_h, nnet_diff_h;
192 
193  if (drop_frames) {
194  KALDI_LOG << "--drop-frames=true :"
195  " we will zero gradient for frames with total den/num mismatch."
196  " The mismatch is likely to be caused by missing correct path "
197  " from den-lattice due wrong annotation or search error."
198  " Leaving such frames out stabilizes the training.";
199  }
200 
201  Timer time;
202  double time_now = 0;
203  KALDI_LOG << "TRAINING STARTED";
204 
205  int32 num_done = 0, num_no_num_ali = 0, num_no_den_lat = 0,
206  num_other_error = 0, num_frm_drop = 0;
207 
208  kaldi::int64 total_frames = 0;
209  double lat_like; // total likelihood of the lattice
210  double lat_ac_like; // acoustic likelihood weighted by posterior.
211  double total_mmi_obj = 0.0, mmi_obj = 0.0;
212  double total_post_on_ali = 0.0, post_on_ali = 0.0;
213 
214  // do per-utterance processing
215  for( ; !feature_reader.Done(); feature_reader.Next()) {
216  std::string utt = feature_reader.Key();
217  if (!den_lat_reader.HasKey(utt)) {
218  KALDI_WARN << "Utterance " << utt << ": found no lattice.";
219  num_no_den_lat++;
220  continue;
221  }
222  if (!num_ali_reader.HasKey(utt)) {
223  KALDI_WARN << "Utterance " << utt << ": found no reference alignment.";
224  num_no_num_ali++;
225  continue;
226  }
227 
228  // 1) get the features, numerator alignment
229  const Matrix<BaseFloat> &mat = feature_reader.Value();
230  const std::vector<int32> &num_ali = num_ali_reader.Value(utt);
231  // check for temporal length of numerator alignments
232  if ((int32)num_ali.size() != mat.NumRows()) {
233  KALDI_WARN << "Numerator alignment has wrong length "
234  << num_ali.size() << " vs. "<< mat.NumRows();
235  num_other_error++;
236  continue;
237  }
238  if (mat.NumRows() > max_frames) {
239  KALDI_WARN << "Utterance " << utt << ": Skipped because it has " << mat.NumRows() <<
240  " frames, which is more than " << max_frames << ".";
241  num_other_error++;
242  continue;
243  }
244 
245  // 2) get the denominator lattice, preprocess
246  Lattice den_lat = den_lat_reader.Value(utt);
247  if (den_lat.Start() == -1) {
248  KALDI_WARN << "Empty lattice for utt " << utt;
249  num_other_error++;
250  continue;
251  }
252  if (old_acoustic_scale != 1.0) {
253  fst::ScaleLattice(fst::AcousticLatticeScale(old_acoustic_scale), &den_lat);
254  }
255  // optional sort it topologically
256  kaldi::uint64 props = den_lat.Properties(fst::kFstProperties, false);
257  if (!(props & fst::kTopSorted)) {
258  if (fst::TopSort(&den_lat) == false)
259  KALDI_ERR << "Cycles detected in lattice.";
260  }
261  // get the lattice length and times of states
262  vector<int32> state_times;
263  int32 max_time = kaldi::LatticeStateTimes(den_lat, &state_times);
264  // check for temporal length of denominator lattices
265  if (max_time != mat.NumRows()) {
266  KALDI_WARN << "Denominator lattice has wrong length "
267  << max_time << " vs. " << mat.NumRows();
268  num_other_error++;
269  continue;
270  }
271 
272  // get actual dims for this utt and nnet
273  int32 num_frames = mat.NumRows(),
274  num_fea = mat.NumCols(),
275  num_pdfs = nnet.OutputDim();
276 
277  // 3) propagate the feature to get the log-posteriors (nnet w/o sofrmax)
278  // push features to GPU
279  feats.Resize(num_frames, num_fea, kUndefined);
280  feats.CopyFromMat(mat);
281  // possibly apply transform
282  nnet_transf.Feedforward(feats, &feats_transf);
283  // propagate through the nnet (assuming w/o softmax)
284  nnet.Propagate(feats_transf, &nnet_out);
285  // subtract the log_prior
286  if(prior_opts.class_frame_counts != "") {
287  log_prior.SubtractOnLogpost(&nnet_out);
288  }
289  // transfer it back to the host
290  nnet_out_h.Resize(num_frames,num_pdfs, kUndefined);
291  nnet_out.CopyToMat(&nnet_out_h);
292  // release the buffers we don't need anymore
293  feats.Resize(0,0);
294  feats_transf.Resize(0,0);
295  nnet_out.Resize(0,0);
296 
297  // 4) rescore the latice
298  LatticeAcousticRescore(nnet_out_h, trans_model, state_times, &den_lat);
299  if (acoustic_scale != 1.0 || lm_scale != 1.0)
300  fst::ScaleLattice(fst::LatticeScale(lm_scale, acoustic_scale), &den_lat);
301 
302  // 5) get the posteriors
303  kaldi::Posterior post;
304  lat_like = kaldi::LatticeForwardBackward(den_lat, &post, &lat_ac_like);
305 
306  // 6) convert the Posterior to a matrix
307  nnet_diff_h.Resize(num_frames, num_pdfs, kSetZero);
308  for (int32 t = 0; t < post.size(); t++) {
309  for (int32 arc = 0; arc < post[t].size(); arc++) {
310  int32 pdf = trans_model.TransitionIdToPdf(post[t][arc].first);
311  nnet_diff_h(t, pdf) += post[t][arc].second;
312  }
313  }
314 
315  // 7) Calculate the MMI-objective function
316  // Calculate the likelihood of correct path from acoustic score,
317  // the denominator likelihood is the total likelihood of the lattice.
318  double path_ac_like = 0.0;
319  for(int32 t=0; t<num_frames; t++) {
320  int32 pdf = trans_model.TransitionIdToPdf(num_ali[t]);
321  path_ac_like += nnet_out_h(t,pdf);
322  }
323  path_ac_like *= acoustic_scale;
324  mmi_obj = path_ac_like - lat_like;
325  //
326  // Note: numerator likelihood does not include graph score,
327  // while denominator likelihood contains graph scores.
328  // The result is offset at the MMI-objective.
329  // However the offset is constant for given alignment,
330  // so it is not harmful.
331 
332  // Sum the den-posteriors under the correct path:
333  post_on_ali = 0.0;
334  for(int32 t=0; t<num_frames; t++) {
335  int32 pdf = trans_model.TransitionIdToPdf(num_ali[t]);
336  double posterior = nnet_diff_h(t, pdf);
337  post_on_ali += posterior;
338  }
339 
340  // Report
341  KALDI_VLOG(1) << "Lattice #" << num_done + 1 << " processed"
342  << " (" << utt << "): found " << den_lat.NumStates()
343  << " states and " << fst::NumArcs(den_lat) << " arcs.";
344 
345  KALDI_VLOG(1) << "Utterance " << utt << ": Average MMI obj. value = "
346  << (mmi_obj/num_frames) << " over " << num_frames
347  << " frames."
348  << " (Avg. den-posterior on ali " << post_on_ali/num_frames << ")";
349 
350 
351  // 7a) Search for the frames with num/den mismatch
352  int32 frm_drop = 0;
353  std::vector<int32> frm_drop_vec;
354  for(int32 t=0; t<num_frames; t++) {
355  int32 pdf = trans_model.TransitionIdToPdf(num_ali[t]);
356  double posterior = nnet_diff_h(t, pdf);
357  if(posterior < 1e-20) {
358  frm_drop++;
359  frm_drop_vec.push_back(t);
360  }
361  }
362 
363  // 8) subtract the pdf-Viterbi-path
364  for(int32 t=0; t<nnet_diff_h.NumRows(); t++) {
365  int32 pdf = trans_model.TransitionIdToPdf(num_ali[t]);
366  nnet_diff_h(t, pdf) -= 1.0;
367  }
368 
369  // 9) Drop mismatched frames from the training by zeroing the derivative
370  if(drop_frames) {
371  for(int32 i=0; i<frm_drop_vec.size(); i++) {
372  nnet_diff_h.Row(frm_drop_vec[i]).Set(0.0);
373  }
374  num_frm_drop += frm_drop;
375  }
376  // Report the frame dropping
377  if (frm_drop > 0) {
378  std::stringstream ss;
379  ss << (drop_frames?"Dropped":"[dropping disabled] Would drop")
380  << " frames in " << utt << " " << frm_drop << "/" << num_frames << ",";
381  //get frame intervals from vec frm_drop_vec
382  ss << " intervals :";
383  //search for streaks of consecutive numbers:
384  int32 beg_streak=frm_drop_vec[0];
385  int32 len_streak=0;
386  int32 i;
387  for(i=0; i<frm_drop_vec.size(); i++,len_streak++) {
388  if(beg_streak + len_streak != frm_drop_vec[i]) {
389  ss << " " << beg_streak << ".." << frm_drop_vec[i-1] << "frm";
390  beg_streak = frm_drop_vec[i];
391  len_streak = 0;
392  }
393  }
394  ss << " " << beg_streak << ".." << frm_drop_vec[i-1] << "frm";
395  //print
396  KALDI_WARN << ss.str();
397  }
398 
399  // 10) backpropagate through the nnet
400  nnet_diff.Resize(num_frames, num_pdfs, kUndefined);
401  nnet_diff.CopyFromMat(nnet_diff_h);
402  nnet.Backpropagate(nnet_diff, NULL);
403  // relase the buffer, we don't need anymore
404  nnet_diff.Resize(0,0);
405 
406  // increase time counter
407  total_mmi_obj += mmi_obj;
408  total_post_on_ali += post_on_ali;
409  total_frames += num_frames;
410  num_done++;
411 
412  if (num_done % 100 == 0) {
413  time_now = time.Elapsed();
414  KALDI_VLOG(1) << "After " << num_done << " utterances: time elapsed = "
415  << time_now/60 << " min; processed " << total_frames/time_now
416  << " frames per second.";
417 #if HAVE_CUDA==1
418  // check the GPU is not overheated
419  CuDevice::Instantiate().CheckGpuHealth();
420 #endif
421  }
422 
423  // Gradient logging,
424  if (kaldi::g_kaldi_verbose_level >= 1 &&
425  (num_done == 1 || num_done % 1000 == 0)) {
426  // first utterance, or every 1000th utterance,
427  KALDI_VLOG(1) << nnet.InfoPropagate();
428  KALDI_VLOG(1) << nnet.InfoBackPropagate();
429  KALDI_VLOG(1) << nnet.InfoGradient();
430  } else if (kaldi::g_kaldi_verbose_level >= 2) {
431  // vlog 2 => every utterance,
432  KALDI_VLOG(2) << nnet.InfoPropagate();
433  KALDI_VLOG(2) << nnet.InfoBackPropagate();
434  KALDI_VLOG(2) << nnet.InfoGradient();
435  }
436  }
437 
438  // after last utterance,
439  KALDI_VLOG(1) << nnet.InfoPropagate();
440  KALDI_VLOG(1) << nnet.InfoBackPropagate();
441  KALDI_VLOG(1) << nnet.InfoGradient();
442 
443  // add the softmax layer back before writing,
444  KALDI_LOG << "Appending the softmax " << target_model_filename;
445  nnet.AppendComponent(new Softmax(nnet.OutputDim(),nnet.OutputDim()));
446  // store the nnet,
447  nnet.Write(target_model_filename, binary);
448 
449  time_now = time.Elapsed();
450  KALDI_LOG << "TRAINING FINISHED; "
451  << "Time taken = " << time_now/60 << " min; processed "
452  << (total_frames/time_now) << " frames per second.";
453 
454  KALDI_LOG << "Done " << num_done << " files, "
455  << num_no_num_ali << " with no numerator alignments, "
456  << num_no_den_lat << " with no denominator lattices, "
457  << num_other_error << " with other errors.";
458 
459  KALDI_LOG << "Overall MMI-objective/frame is "
460  << std::setprecision(8) << (total_mmi_obj/total_frames)
461  << " over " << total_frames << " frames."
462  << " (average den-posterior on ali " << (total_post_on_ali/total_frames) << ","
463  << " dropped " << num_frm_drop << " frames with num/den mismatch)";
464 
465 #if HAVE_CUDA == 1
466  CuDevice::Instantiate().PrintProfile();
467 #endif
468 
469  return 0;
470  } catch(const std::exception &e) {
471  std::cerr << e.what();
472  return -1;
473  }
