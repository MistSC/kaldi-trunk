30 // returns an integer randomly drawn with expected value "expected_count"
31 // (will be either floor(expected_count) or ceil(expected_count)).
32 int32 GetCount(double expected_count) {
33  KALDI_ASSERT(expected_count >= 0.0);
34  int32 ans = floor(expected_count);
35  expected_count -= ans;
36  if (WithProb(expected_count))
37  ans++;
38  return ans;
39 }

48 bool ContainsSingleExample(const NnetExample &eg,
49  int32 *min_input_t,
50  int32 *max_input_t,
51  int32 *min_output_t,
52  int32 *max_output_t) {
53  bool done_input = false, done_output = false;
54  int32 num_indexes = eg.io.size();
55  for (int32 i = 0; i < num_indexes; i++) {
56  const NnetIo &io = eg.io[i];
57  std::vector<Index>::const_iterator iter = io.indexes.begin(),
58  end = io.indexes.end();
59  // Should not have an empty input/output type.
60  KALDI_ASSERT(!io.indexes.empty());
61  if (io.name == "input" || io.name == "output") {
62  int32 min_t = iter->t, max_t = iter->t;
63  for (; iter != end; ++iter) {
64  int32 this_t = iter->t;
65  min_t = std::min(min_t, this_t);
66  max_t = std::max(max_t, this_t);
67  if (iter->n != 0) {
68  KALDI_WARN << "Example does not contain just a single example; "
69  << "too late to do frame selection or reduce context.";
70  return false;
71  }
72  }
73  if (io.name == "input") {
74  done_input = true;
75  *min_input_t = min_t;
76  *max_input_t = max_t;
77  } else {
78  KALDI_ASSERT(io.name == "output");
79  done_output = true;
80  *min_output_t = min_t;
81  *max_output_t = max_t;
82  }
83  } else {
84  for (; iter != end; ++iter) {
85  if (iter->n != 0) {
86  KALDI_WARN << "Example does not contain just a single example; "
87  << "too late to do frame selection or reduce context.";
88  return false;
89  }
90  }
91  }
92  }
93  if (!done_input) {
94  KALDI_WARN << "Example does not have any input named 'input'";
95  return false;
96  }
97  if (!done_output) {
98  KALDI_WARN << "Example does not have any output named 'output'";
99  return false;
100  }
101  return true;
102 }

111 void FilterExample(const NnetExample &eg,
112  int32 min_input_t,
113  int32 max_input_t,
114  int32 min_output_t,
115  int32 max_output_t,
116  NnetExample *eg_out) {
117  eg_out->io.clear();
118  eg_out->io.resize(eg.io.size());
119  for (size_t i = 0; i < eg.io.size(); i++) {
120  bool is_input_or_output;
121  int32 min_t, max_t;
122  const NnetIo &io_in = eg.io[i];
123  NnetIo &io_out = eg_out->io[i];
124  const std::string &name = io_in.name;
125  io_out.name = name;
126  if (name == "input") {
127  min_t = min_input_t;
128  max_t = max_input_t;
129  is_input_or_output = true;
130  } else if (name == "output") {
131  min_t = min_output_t;
132  max_t = max_output_t;
133  is_input_or_output = true;
134  } else {
135  is_input_or_output = false;
136  }
137  if (!is_input_or_output) { // Just copy everything.
138  io_out.indexes = io_in.indexes;
139  io_out.features = io_in.features;
140  } else {
141  const std::vector<Index> &indexes_in = io_in.indexes;
142  std::vector<Index> &indexes_out = io_out.indexes;
143  indexes_out.reserve(indexes_in.size());
144  int32 num_indexes = indexes_in.size(), num_kept = 0;
145  KALDI_ASSERT(io_in.features.NumRows() == num_indexes);
146  std::vector<bool> keep(num_indexes, false);
147  std::vector<Index>::const_iterator iter_in = indexes_in.begin(),
148  end_in = indexes_in.end();
149  std::vector<bool>::iterator iter_out = keep.begin();
150  for (; iter_in != end_in; ++iter_in,++iter_out) {
151  int32 t = iter_in->t;
152  bool is_within_range = (t >= min_t && t <= max_t);
153  *iter_out = is_within_range;
154  if (is_within_range) {
155  indexes_out.push_back(*iter_in);
156  num_kept++;
157  }
158  }
159  KALDI_ASSERT(iter_out == keep.end());
160  if (num_kept == 0)
161  KALDI_ERR << "FilterExample removed all indexes for '" << name << "'";
162 
163  FilterGeneralMatrixRows(io_in.features, keep,
164  &io_out.features);
165  KALDI_ASSERT(io_out.features.NumRows() == num_kept &&
166  indexes_out.size() == static_cast<size_t>(num_kept));
167  }
168  }
169 }

188 bool SelectFromExample(const NnetExample &eg,
189  std::string frame_str,
190  int32 left_context,
191  int32 right_context,
192  int32 frame_shift,
193  NnetExample *eg_out) {
194  int32 min_input_t, max_input_t,
195  min_output_t, max_output_t;
196  if (!ContainsSingleExample(eg, &min_input_t, &max_input_t,
197  &min_output_t, &max_output_t))
198  KALDI_ERR << "Too late to perform frame selection/context reduction on "
199  << "these examples (already merged?)";
200  if (frame_str != "") {
201  // select one frame.
202  if (frame_str == "random") {
203  min_output_t = max_output_t = RandInt(min_output_t,
204  max_output_t);
205  } else {
206  int32 frame;
207  if (!ConvertStringToInteger(frame_str, &frame))
208  KALDI_ERR << "Invalid option --frame='" << frame_str << "'";
209  if (frame < min_output_t || frame > max_output_t) {
210  // Frame is out of range. Should happen only rarely. Calling code
211  // makes sure of this.
212  return false;
213  }
214  min_output_t = max_output_t = frame;
215  }
216  }
217  // There may come a time when we want to remove or make it possible to disable
218  // the error messages below. The std::max and std::min expressions may seem
219  // unnecessary but are intended to make life easier if and when we do that.
220  if (left_context != -1) {
221  if (min_input_t > min_output_t - left_context)
222  KALDI_ERR << "You requested --left-context=" << left_context
223  << ", but example only has left-context of "
224  << (min_output_t - min_input_t);
225  min_input_t = std::max(min_input_t, min_output_t - left_context);
226  }
227  if (right_context != -1) {
228  if (max_input_t < max_output_t + right_context)
229  KALDI_ERR << "You requested --right-context=" << right_context
230  << ", but example only has right-context of "
231  << (max_input_t - max_output_t);
232  max_input_t = std::min(max_input_t, max_output_t + right_context);
233  }
234  FilterExample(eg,
235  min_input_t, max_input_t,
236  min_output_t, max_output_t,
237  eg_out);
238  if (frame_shift != 0) {
239  std::vector<std::string> exclude_names; // we can later make this
240  exclude_names.push_back(std::string("ivector")); // configurable.
241  ShiftExampleTimes(frame_shift, exclude_names, eg_out);
242  }
243  return true;
244 }

250 int main(int argc, char *argv[]) {
251  try {
252  using namespace kaldi;
253  using namespace kaldi::nnet3;
254  typedef kaldi::int32 int32;
255  typedef kaldi::int64 int64;

257  const char *usage =
258  "Copy examples (single frames or fixed-size groups of frames) for neural\n"
259  "network training, possibly changing the binary mode. Supports multiple wspecifiers, in\n"
260  "which case it will write the examples round-robin to the outputs.\n"
261  "\n"
262  "Usage: nnet3-copy-egs [options] <egs-rspecifier> <egs-wspecifier1> [<egs-wspecifier2> ...]\n"
263  "\n"
264  "e.g.\n"
265  "nnet3-copy-egs ark:train.egs ark,t:text.egs\n"
266  "or:\n"
267  "nnet3-copy-egs ark:train.egs ark:1.egs ark:2.egs\n";
268 
269  bool random = false;
270  int32 srand_seed = 0;
271  int32 frame_shift = 0;
272  BaseFloat keep_proportion = 1.0;
273 
274  // The following config variables, if set, can be used to extract a single
275  // frame of labels from a multi-frame example, and/or to reduce the amount
276  // of context.
277  int32 left_context = -1, right_context = -1;
278 
279  // you can set frame to a number to select a single frame with a particular
280  // offset, or to 'random' to select a random single frame.
281  std::string frame_str;
282 
283  ParseOptions po(usage);
284  po.Register("random", &random, "If true, will write frames to output "
285  "archives randomly, not round-robin.");
286  po.Register("frame-shift", &frame_shift, "Allows you to shift time values "
287  "in the supervision data (excluding iVector data). Only really "
288  "useful in clockwork topologies (i.e. any topology for which "
289  "modulus != 1). Shifting is done after any frame selection.");
290  po.Register("keep-proportion", &keep_proportion, "If <1.0, this program will "
291  "randomly keep this proportion of the input samples. If >1.0, it will "
292  "in expectation copy a sample this many times. It will copy it a number "
293  "of times equal to floor(keep-proportion) or ceil(keep-proportion).");
294  po.Register("srand", &srand_seed, "Seed for random number generator "
295  "(only relevant if --random=true or --keep-proportion != 1.0)");
296  po.Register("frame", &frame_str, "This option can be used to select a single "
297  "frame from each multi-frame example. Set to a number 0, 1, etc. "
298  "to select a frame with a given index, or 'random' to select a "
299  "random frame.");
300  po.Register("left-context", &left_context, "Can be used to truncate the "
301  "feature left-context that we output.");
302  po.Register("right-context", &right_context, "Can be used to truncate the "
303  "feature right-context that we output.");
304 
305 
306  po.Read(argc, argv);
307 
308  srand(srand_seed);
309 
310  if (po.NumArgs() < 2) {
311  po.PrintUsage();
312  exit(1);
313  }
314 
315  std::string examples_rspecifier = po.GetArg(1);
316 
317  SequentialNnetExampleReader example_reader(examples_rspecifier);
318 
319  int32 num_outputs = po.NumArgs() - 1;
320  std::vector<NnetExampleWriter*> example_writers(num_outputs);
321  for (int32 i = 0; i < num_outputs; i++)
322  example_writers[i] = new NnetExampleWriter(po.GetArg(i+2));
323 
324 
325  int64 num_read = 0, num_written = 0;
326  for (; !example_reader.Done(); example_reader.Next(), num_read++) {
327  // count is normally 1; could be 0, or possibly >1.
328  int32 count = GetCount(keep_proportion);
329  std::string key = example_reader.Key();
330  const NnetExample &eg = example_reader.Value();
331  for (int32 c = 0; c < count; c++) {
332  int32 index = (random ? Rand() : num_written) % num_outputs;
333  if (frame_str == "" && left_context == -1 && right_context == -1 &&
334  frame_shift == 0) {
335  example_writers[index]->Write(key, eg);
336  num_written++;
337  } else { // the --frame option or context options were set.
338  NnetExample eg_modified;
339  if (SelectFromExample(eg, frame_str, left_context, right_context,
340  frame_shift, &eg_modified)) {
341  // this branch of the if statement will almost always be taken (should only
342  // not be taken for shorter-than-normal egs from the end of a file.
343  example_writers[index]->Write(key, eg_modified);
344  num_written++;
345  }
346  }
347  }
348  }
349 
350  for (int32 i = 0; i < num_outputs; i++)
351  delete example_writers[i];
352  KALDI_LOG << "Read " << num_read << " neural-network training examples, wrote "
353  << num_written;
354  return (num_written == 0 ? 1 : 0);
355  } catch(const std::exception &e) {
356  std::cerr << e.what() << '\n';
357  return -1;
358  }
359 }
