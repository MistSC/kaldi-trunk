100 template<typename Real>
101 static void UnitTestCuMathSplice() {
102  int32 M = 100 + Rand() % 200, N = 100 + Rand() % 200;
103  CuMatrix<Real> src(M, N);
104  CuArray<int32> frame_offsets;
105 
106  src.SetRandn();
107  int32 n_rows = src.NumRows();
108  int32 n_columns = src.NumCols();
109  std::vector<int32> frame_offsets_vec;
110 
111  // The number of columns of tgt is rows(src)
112  // times n_frame_offsets, so we keep n_frame_offsets
113  // reasonably small (2 <= n <= 6).
114  int32 n_frame_offsets = Rand() % 7 + 2;
115  for (int32 i = 0; i < n_frame_offsets; i++) {
116  frame_offsets_vec.push_back(Rand() % 2 * n_columns - n_columns);
117  }
118 
119  CuMatrix<Real> tgt(M, N * n_frame_offsets);
120  frame_offsets.CopyFromVec(frame_offsets_vec);
121  cu::Splice(src, frame_offsets, &tgt);
122 
123  Matrix<Real> src_copy(src), tgt_copy(tgt);
124  for (int32 i = 0; i < n_rows; i++) {
125  for (int32 k = 0; k < n_frame_offsets; k++) {
126  for (int32 j = 0; j < n_columns; j++) {
127  Real src_val;
128  if (i + frame_offsets_vec.at(k) >= n_rows) {
129  src_val = src_copy(n_rows-1, j);
130  } else if (i + frame_offsets_vec.at(k) <= 0) {
131  src_val = src_copy(0, j);
132  } else {
133  src_val = src_copy(i + frame_offsets_vec.at(k), j);
134  }
135  Real tgt_val = tgt_copy(i, k * n_columns + j);
136  AssertEqual(src_val, tgt_val);
137  }
138  }
139  }
