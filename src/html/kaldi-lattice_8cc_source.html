62 bool WriteCompactLattice(std::ostream &os, bool binary,
63  const CompactLattice &t) {
64  if (binary) {
65  fst::FstWriteOptions opts;
66  // Leave all the options default. Normally these lattices wouldn't have any
67  // osymbols/isymbols so no point directing it not to write them (who knows what
68  // we'd want to if we had them).
69  return t.Write(os, opts);
70  } else {
71  // Text-mode output. Note: we expect that t.InputSymbols() and
72  // t.OutputSymbols() would always return NULL. The corresponding input
73  // routine would not work if the FST actually had symbols attached.
74  // Write a newline after the key, so the first line of the FST appears
75  // on its own line.
76  os << '\n';
77  bool acceptor = true, write_one = false;
78 #ifdef HAVE_OPENFST_GE_10400
79  fst::FstPrinter<CompactLatticeArc> printer(t, t.InputSymbols(),
80  t.OutputSymbols(),
81  NULL, acceptor, write_one, "\t");
82 #else
83  fst::FstPrinter<CompactLatticeArc> printer(t, t.InputSymbols(),
84  t.OutputSymbols(),
85  NULL, acceptor, write_one);
86 #endif
87  printer.Print(&os, "<unknown>");
88  if (os.fail())
89  KALDI_WARN << "Stream failure detected.";
90  // Write another newline as a terminating character. The read routine will
91  // detect this [this is a Kaldi mechanism, not somethig in the original
92  // OpenFst code].
93  os << '\n';
94  return os.good();
95  }

114  static std::pair<Lattice*, CompactLattice*> ReadText(
115  std::istream &is) {
116  typedef std::pair<Lattice*, CompactLattice*> PairT;
117  using std::string;
118  using std::vector;
119  Lattice *fst = new Lattice();
120  CompactLattice *cfst = new CompactLattice();
121  string line;
122  size_t nline = 0;
123  string separator = FLAGS_fst_field_separator + "\r\n";
124  while (std::getline(is, line)) {
125  nline++;
126  vector<string> col;
127  // on Windows we'll write in text and read in binary mode.
128  SplitStringToVector(line, separator.c_str(), true, &col);
129  if (col.size() == 0) break; // Empty line is a signal to stop, in our
130  // archive format.
131  if (col.size() > 5) {
132  KALDI_WARN << "Reading lattice: bad line in FST: " << line;
133  delete fst;
134  delete cfst;
135  return PairT(static_cast<Lattice*>(NULL),
136  static_cast<CompactLattice*>(NULL));
137  }
138  StateId s;
139  if (!ConvertStringToInteger(col[0], &s)) {
140  KALDI_WARN << "FstCompiler: bad line in FST: " << line;
141  delete fst;
142  delete cfst;
143  return PairT(static_cast<Lattice*>(NULL),
144  static_cast<CompactLattice*>(NULL));
145  }
146  if (fst)
147  while (s >= fst->NumStates())
148  fst->AddState();
149  if (cfst)
150  while (s >= cfst->NumStates())
151  cfst->AddState();
152  if (nline == 1) {
153  if (fst) fst->SetStart(s);
154  if (cfst) cfst->SetStart(s);
155  }
156 
157  if (fst) { // we still have fst; try to read that arc.
158  bool ok = true;
159  Arc arc;
160  Weight w;
161  StateId d = s;
162  switch (col.size()) {
163  case 1 :
164  fst->SetFinal(s, Weight::One());
165  break;
166  case 2:
167  if (!StrToWeight(col[1], true, &w)) ok = false;
168  else fst->SetFinal(s, w);
169  break;
170  case 3: // 3 columns not ok for Lattice format; it's not an acceptor.
171  ok = false;
172  break;
173  case 4:
174  ok = ConvertStringToInteger(col[1], &arc.nextstate) &&
175  ConvertStringToInteger(col[2], &arc.ilabel) &&
176  ConvertStringToInteger(col[3], &arc.olabel);
177  if (ok) {
178  d = arc.nextstate;
179  arc.weight = Weight::One();
180  fst->AddArc(s, arc);
181  }
182  break;
183  case 5:
184  ok = ConvertStringToInteger(col[1], &arc.nextstate) &&
185  ConvertStringToInteger(col[2], &arc.ilabel) &&
186  ConvertStringToInteger(col[3], &arc.olabel) &&
187  StrToWeight(col[4], false, &arc.weight);
188  if (ok) {
189  d = arc.nextstate;
190  fst->AddArc(s, arc);
191  }
192  break;
193  default:
194  ok = false;
195  }
196  while (d >= fst->NumStates())
197  fst->AddState();
198  if (!ok) {
199  delete fst;
200  fst = NULL;
201  }
202  }
203  if (cfst) {
204  bool ok = true;
205  CArc arc;
206  CWeight w;
207  StateId d = s;
208  switch (col.size()) {
209  case 1 :
210  cfst->SetFinal(s, CWeight::One());
211  break;
212  case 2:
213  if (!StrToCWeight(col[1], true, &w)) ok = false;
214  else cfst->SetFinal(s, w);
215  break;
216  case 3: // compact-lattice is acceptor format: state, next-state, label.
217  ok = ConvertStringToInteger(col[1], &arc.nextstate) &&
218  ConvertStringToInteger(col[2], &arc.ilabel);
219  if (ok) {
220  d = arc.nextstate;
221  arc.olabel = arc.ilabel;
222  arc.weight = CWeight::One();
223  cfst->AddArc(s, arc);
224  }
225  break;
226  case 4:
227  ok = ConvertStringToInteger(col[1], &arc.nextstate) &&
228  ConvertStringToInteger(col[2], &arc.ilabel) &&
229  StrToCWeight(col[3], false, &arc.weight);
230  if (ok) {
231  d = arc.nextstate;
232  arc.olabel = arc.ilabel;
233  cfst->AddArc(s, arc);
234  }
235  break;
236  case 5: default:
237  ok = false;
238  }
239  while (d >= cfst->NumStates())
240  cfst->AddState();
241  if (!ok) {
242  delete cfst;
243  cfst = NULL;
244  }
245  }
246  if (!fst && !cfst) {
247  KALDI_WARN << "Bad line in lattice text format: " << line;
248  // read until we get an empty line, so at least we
249  // have a chance to read the next one (although this might
250  // be a bit futile since the calling code will get unhappy
251  // about failing to read this one.
252  while (std::getline(is, line)) {
253  SplitStringToVector(line, separator.c_str(), true, &col);
254  if (col.empty()) break;
255  }
256  return PairT(static_cast<Lattice*>(NULL),
257  static_cast<CompactLattice*>(NULL));
258  }
259  }
260  return PairT(fst, cfst);
261  }

310 bool ReadCompactLattice(std::istream &is, bool binary,
311  CompactLattice **clat) {
312  KALDI_ASSERT(*clat == NULL);
313  if (binary) {
314  fst::FstHeader hdr;
315  if (!hdr.Read(is, "<unknown>")) {
316  KALDI_WARN << "Reading compact lattice: error reading FST header.";
317  return false;
318  }
319  if (hdr.FstType() != "vector") {
320  KALDI_WARN << "Reading compact lattice: unsupported FST type: "
321  << hdr.FstType();
322  return false;
323  }
324  fst::FstReadOptions ropts("<unspecified>",
325  &hdr);
326 
327  typedef fst::CompactLatticeWeightTpl<fst::LatticeWeightTpl<float>, int32> T1;
328  typedef fst::CompactLatticeWeightTpl<fst::LatticeWeightTpl<double>, int32> T2;
329  typedef fst::LatticeWeightTpl<float> T3;
330  typedef fst::LatticeWeightTpl<double> T4;
331  typedef fst::VectorFst<fst::ArcTpl<T1> > F1;
332  typedef fst::VectorFst<fst::ArcTpl<T2> > F2;
333  typedef fst::VectorFst<fst::ArcTpl<T3> > F3;
334  typedef fst::VectorFst<fst::ArcTpl<T4> > F4;
335 
336  CompactLattice *ans = NULL;
337  if (hdr.ArcType() == T1::Type()) {
338  ans = ConvertToCompactLattice(F1::Read(is, ropts));
339  } else if (hdr.ArcType() == T2::Type()) {
340  ans = ConvertToCompactLattice(F2::Read(is, ropts));
341  } else if (hdr.ArcType() == T3::Type()) {
342  ans = ConvertToCompactLattice(F3::Read(is, ropts));
343  } else if (hdr.ArcType() == T4::Type()) {
344  ans = ConvertToCompactLattice(F4::Read(is, ropts));
345  } else {
346  KALDI_WARN << "FST with arc type " << hdr.ArcType()
347  << " cannot be converted to CompactLattice.\n";
348  return false;
349  }
350  if (ans == NULL) {
351  KALDI_WARN << "Error reading compact lattice (after reading header).";
352  return false;
353  }
354  *clat = ans;
355  return true;
356  } else {
357  // The next line would normally consume the \r on Windows, plus any
358  // extra spaces that might have got in there somehow.
359  while (std::isspace(is.peek()) && is.peek() != '\n') is.get();
360  if (is.peek() == '\n') is.get(); // consume the newline.
361  else { // saw spaces but no newline.. this is not expected.
362  KALDI_WARN << "Reading compact lattice: unexpected sequence of spaces "
363  << " at file position " << is.tellg();
364  return false;
365  }
366  *clat = ReadCompactLatticeText(is); // that routine will warn on error.
367  return (*clat != NULL);
368  }

372 bool CompactLatticeHolder::Read(std::istream &is) {
373  Clear(); // in case anything currently stored.
374  int c = is.peek();
375  if (c == -1) {
376  KALDI_WARN << "End of stream detected reading CompactLattice.";
377  return false;
378  } else if (isspace(c)) { // The text form of the lattice begins
379  // with space (normally, '\n'), so this means it's text (the binary form
380  // cannot begin with space because it starts with the FST Type() which is not
381  // space).
382  return ReadCompactLattice(is, false, &t_);
383  } else if (c != 214) { // 214 is first char of FST magic number,
384  // on little-endian machines which is all we support (\326 octal)
385  KALDI_WARN << "Reading compact lattice: does not appear to be an FST "
386  << " [non-space but no magic number detected], file pos is "
387  << is.tellg();
388  return false;
389  } else {
390  return ReadCompactLattice(is, true, &t_);
391  }
392 }
