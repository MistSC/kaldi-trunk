38 static void InitRand(MatrixBase<Real> *M) {
39  do {
40  for (MatrixIndexT i = 0;i < M->NumRows();i++)
41  for (MatrixIndexT j = 0;j < M->NumCols();j++)
42  (*M)(i, j) = RandGauss();
43  } while (M->NumRows() != 0 && M->Cond() > 100);

78 void UnitTestMatrixRandomizer() {
79  Matrix<BaseFloat> m(1111,10);
80  InitRand(&m);
81  CuMatrix<BaseFloat> m2(m);
82  // config
83  NnetDataRandomizerOptions c;
84  c.randomizer_size = 1000;
85  c.minibatch_size = 100;
86  // randomizer
87  MatrixRandomizer r;
88  r.Init(c);
89  r.AddData(m2);
90  KALDI_ASSERT(r.IsFull());
91  // create vector with consecutive indices
92  std::vector<int32> mask(1111);
93  for(int32 i=0; i<1111; i++) { mask[i]=i; }
94  r.Randomize(mask); // no shuffling
95  // make sure we get same data we put to randomizer
96  int32 i=0;
97  for( ; !r.Done(); r.Next(), i++) {
98  KALDI_LOG << i;
99  const CuMatrixBase<BaseFloat> &m3 = r.Value();
100  Matrix<BaseFloat> m4(m3.NumRows(),m3.NumCols()); m3.CopyToMat(&m4);
101  AssertEqual(m4,m.RowRange(i*c.minibatch_size,c.minibatch_size));
102  }
103  KALDI_ASSERT(i == 11); // 11 minibatches
104 
105  KALDI_LOG << "Filling for 2nd time";
106  // try to fill buffer one more time, and empty it
107  KALDI_ASSERT(!r.IsFull());
108  r.AddData(m2);
109  KALDI_ASSERT(r.IsFull());
110  KALDI_ASSERT(r.NumFrames() == 11 + 1111);
111  { // check last 11 rows were copied to the front in the buffer
112  const CuMatrixBase<BaseFloat> &m3 = r.Value();
113  Matrix<BaseFloat> m4(m3.NumRows(),m3.NumCols()); m3.CopyToMat(&m4);
114  AssertEqual(m4.RowRange(0,11),m.RowRange(1100,11));
115  }
116  KALDI_ASSERT(!r.Done());
117  for( ; !r.Done(); r.Next(), i++) {
118  KALDI_LOG << i;
119  const CuMatrixBase<BaseFloat>& m3 = r.Value();
120  static_cast<const void>(m3); // variable no longer unused,
121  }
122  KALDI_ASSERT(i == 22); // 22 minibatches
123 }
