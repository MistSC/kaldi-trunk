33 // test that determinization proceeds correctly on general
34 // FSTs (not guaranteed determinzable, but we use the
35 // max-states option to stop it getting out of control).
36 template<class Arc> void TestDeterminizeLatticePruned() {
37  typedef kaldi::int32 Int;
38  typedef typename Arc::Weight Weight;
39  typedef ArcTpl<CompactLatticeWeightTpl<Weight, Int> > CompactArc;
40 
41  for(int i = 0; i < 100; i++) {
42  RandFstOptions opts;
43  opts.n_states = 4;
44  opts.n_arcs = 10;
45  opts.n_final = 2;
46  opts.allow_empty = false;
47  opts.weight_multiplier = 0.5; // impt for the randomly generated weights
48  opts.acyclic = true;
49  // to be exactly representable in float,
50  // or this test fails because numerical differences can cause symmetry in
51  // weights to be broken, which causes the wrong path to be chosen as far
52  // as the string part is concerned.
53 
54  VectorFst<Arc> *fst = RandPairFst<Arc>(opts);
55 
56  bool sorted = TopSort(fst);
57  KALDI_ASSERT(sorted);
58 
59  ILabelCompare<Arc> ilabel_comp;
60  if (kaldi::Rand() % 2 == 0)
61  ArcSort(fst, ilabel_comp);
62 
63  std::cout << "FST before lattice-determinizing is:\n";
64  {
65 #ifdef HAVE_OPENFST_GE_10400
66  FstPrinter<Arc> fstprinter(*fst, NULL, NULL, NULL, false, true, "\t");
67 #else
68  FstPrinter<Arc> fstprinter(*fst, NULL, NULL, NULL, false, true);
69 #endif
70  fstprinter.Print(&std::cout, "standard output");
71  }
72  VectorFst<Arc> det_fst;
73  try {
74  DeterminizeLatticePrunedOptions lat_opts;
75  lat_opts.max_mem = ((kaldi::Rand() % 2 == 0) ? 100 : 1000);
76  lat_opts.max_states = ((kaldi::Rand() % 2 == 0) ? -1 : 20);
77  lat_opts.max_arcs = ((kaldi::Rand() % 2 == 0) ? -1 : 30);
78  bool ans = DeterminizeLatticePruned<Weight>(*fst, 10.0, &det_fst, lat_opts);
79 
80  std::cout << "FST after lattice-determinizing is:\n";
81  {
82 #ifdef HAVE_OPENFST_GE_10400
83  FstPrinter<Arc> fstprinter(det_fst, NULL, NULL, NULL, false, true, "\t");
84 #else
85  FstPrinter<Arc> fstprinter(det_fst, NULL, NULL, NULL, false, true);
86 #endif
87  fstprinter.Print(&std::cout, "standard output");
88  }
89  KALDI_ASSERT(det_fst.Properties(kIDeterministic, true) & kIDeterministic);
90  // OK, now determinize it a different way and check equivalence.
91  // [note: it's not normal determinization, it's taking the best path
92  // for any input-symbol sequence....
93 
94 
95  VectorFst<Arc> pruned_fst(*fst);
96  if (pruned_fst.NumStates() != 0)
97  kaldi::PruneLattice(10.0, &pruned_fst);
98 
99  VectorFst<CompactArc> compact_pruned_fst, compact_pruned_det_fst;
100  ConvertLattice<Weight, Int>(pruned_fst, &compact_pruned_fst, false);
101  std::cout << "Compact pruned FST is:\n";
102  {
103 #ifdef HAVE_OPENFST_GE_10400
104  FstPrinter<CompactArc> fstprinter(compact_pruned_fst, NULL, NULL, NULL, false, true, "\t");
105 #else
106  FstPrinter<CompactArc> fstprinter(compact_pruned_fst, NULL, NULL, NULL, false, true);
107 #endif
108  fstprinter.Print(&std::cout, "standard output");
109  }
110  ConvertLattice<Weight, Int>(det_fst, &compact_pruned_det_fst, false);
111 
112  std::cout << "Compact version of determinized FST is:\n";
113  {
114 #ifdef HAVE_OPENFST_GE_10400
115  FstPrinter<CompactArc> fstprinter(compact_pruned_det_fst, NULL, NULL, NULL, false, true, "\t");
116 #else
117  FstPrinter<CompactArc> fstprinter(compact_pruned_det_fst, NULL, NULL, NULL, false, true);
118 #endif
119  fstprinter.Print(&std::cout, "standard output");
120  }
121 
122  if (ans)
123  KALDI_ASSERT(RandEquivalent(compact_pruned_det_fst, compact_pruned_fst, 5/*paths*/, 0.01/*delta*/, kaldi::Rand()/*seed*/, 100/*path length, max*/));
124  } catch (...) {
125  std::cout << "Failed to lattice-determinize this FST (probably not determinizable)\n";
126  }
127  delete fst;
128  }
129 }
