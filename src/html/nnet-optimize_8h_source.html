154  // The access queue for keeping track of the freshness of computation.
155  // Most-recently-accessed computation is at the end, and
156  // least-recently-accessed computaiton is at the beginning.
157  // Together with computation_cache_, this forms a most-recently-used (MRU)
158  // cache for Computations, indexed by ComputationRequest. Pointers
159  // are owned in computation_cache_.
160  typedef std::list<const ComputationRequest*> AqType;
161  AqType access_queue_;
162 
163  // Map from computation-request to pair of (computation, and position in
164  // access_queue_). Used for fast lookup of previously compiled computations.
165  // All pointers are owned here.
166  typedef unordered_map<const ComputationRequest*, std::pair<NnetComputation*,
167  AqType::iterator>, ComputationRequestHasher,
168  ComputationRequestPtrEqual> CacheType;
169  CacheType computation_cache_;
170 
171  // This function updates the computation cache. It is called within
172  // Compile(). It insert the request to the end of the queue, and purge
173  // the least-recently-accessed request from the queue and the cache
174  // if the capacity is reached.
175  void UpdateCache(const ComputationRequest *request,
176  NnetComputation *computation);
177  // This function updates the recently accessed queue.
178  void UpdateAccessQueue(CacheType::iterator &cit);
179  // This configuration value determines how many unique Computations
180  // to cache in our most-recently-used cache.
181  int32 cache_capacity_;
182 };
