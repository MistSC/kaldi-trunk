167 void UnitTestReadConfig() {
168  std::string str = " alpha=aa beta=\"b b\"# String test\n"
169  "beta2='b c' beta3=bd # \n"
170  "gamma=1:2:3:4 # Int Vector test\n"
171  " de1ta=f # Bool + Integer in key Comment test delta=t \n"
172  "_epsilon=-1 # Int Vector test _epsilon=1 \n"
173  " zet-_a=0.15 theta=1.1# Float, -, _ test\n"
174  "quoted='a b c' # quoted string\n"
175  "quoted2=\"d e 'a b=c' f\" # string quoted with double quotes";
176 
177  std::istringstream is(str);
178  std::vector<std::string> lines;
179  ReadConfigFile(is, &lines);
180  KALDI_ASSERT(lines.size() == 8);
181 
182  ConfigLine cfl;
183  for (size_t i = 0; i < lines.size(); i++) {
184  KALDI_ASSERT(cfl.ParseLine(lines[i]));
185  }
186  KALDI_ASSERT(cfl.GetValue("beta2", &str) && str == "b c");
187  KALDI_ASSERT(cfl.GetValue("_epsilon", &str) && str == "-1");
188  KALDI_ASSERT(cfl.GetValue("quoted", &str) && str == "a b c");
189  KALDI_ASSERT(cfl.GetValue("quoted2", &str) && str == "d e 'a b=c' f");
190 
191  BaseFloat float_val = 0;
192  KALDI_ASSERT(cfl.GetValue("zet-_a", &float_val) && ApproxEqual(float_val, 0.15));
193 }

195 void UnitTestDescriptorTokenize() {
196  std::vector<std::string> lines;
197 
198  std::string str = "(,test )";
199  KALDI_ASSERT(DescriptorTokenize(str, &lines));
200  KALDI_ASSERT(lines[0] == "(" && lines[1] == "," && lines[2] == "test" && lines[3] == ")");
201 
202  str = "(,1test )";
203  KALDI_ASSERT(!DescriptorTokenize(str, &lines));
204 
205  str = "t (,-1 )";
206  KALDI_ASSERT(DescriptorTokenize(str, &lines));
207  KALDI_ASSERT(lines.size() == 5 && lines[0] == "t" && lines[3] == "-1");
208 
209  str = " sd , -112 )";
210  KALDI_ASSERT(DescriptorTokenize(str, &lines));
211  KALDI_ASSERT(lines.size() == 4 && lines[0] == "sd" && lines[2] == "-112");
212 
213  str = " sd , +112 )";
214  KALDI_ASSERT(DescriptorTokenize(str, &lines));
215  KALDI_ASSERT(lines.size() == 4 && lines[0] == "sd" && lines[2] == "+112");
216 
217  str = "foo";
218  KALDI_ASSERT(DescriptorTokenize(str, &lines));
219  KALDI_ASSERT(lines.size() == 1 && lines[0] == "foo");
220 
221 }
