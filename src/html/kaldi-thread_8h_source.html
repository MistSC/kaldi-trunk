44 // Description of MultiThreadPool and its usage:
45 //
46 // Usage of the RunMultiThreadedPersistent is the same as the usage of
47 // RunMultiThreaded, except that the object provided ust inherit MultiThreadable
48 // and it's run method isn't called, but operator() is called directly instead.
49 // Member variables num_threads_ and thread_id_ must NOT be redefined in the
50 // classes used, as they are called when using MultiThreadable*
51 //
52 // MultiThreadPool is a singleton class, it's instance is obtained using
53 // MultiThreadPool::Instantiate(). First instantiation initializes the thread
54 // pool using g_num_threads threads, each of those threads runs infinite loop in
55 // ThreadWorker::run(). When RunMultiThreadedPersistent(c) is called, each
56 // ThreadWorker is given a pointer to a copy of c and calls c() in it's thread.
57 // After doing this, it Waits on barrier to sync with all the threads and the
58 // main one, then Waits again until it receives another job.

124 template<class C>
125 class MultiThreader {
126  public:
127  MultiThreader(int32 num_threads,
128  const C &c_in):
129  threads_(new pthread_t[std::max<int32>(1, num_threads)]),
130  cvec_(std::max<int32>(1, num_threads), c_in) {
131  if (num_threads == 0) {
132  // This is a special case with num_threads == 0, which behaves like with
133  // num_threads == 1 but without creating extra threads. This can be
134  // useful in GPU computations where threads cannot be used.
135  KALDI_PTHREAD_PTR(threads_[0]) = 0;
136  cvec_[0].thread_id_ = 0;
137  cvec_[0].num_threads_ = 1;
138  (cvec_[0])();
139  } else {
140  pthread_attr_t pthread_attr;
141  pthread_attr_init(&pthread_attr);
142  for (int32 thread = 0; thread < num_threads; thread++) {
143  cvec_[thread].thread_id_ = thread;
144  cvec_[thread].num_threads_ = num_threads;
145  int32 ret;
146  if ((ret=pthread_create(&(threads_[thread]),
147  &pthread_attr, C::run, &(cvec_[thread])))) {
148  const char *c = strerror(ret);
149  if (c == NULL) { c = "[NULL]"; }
150  KALDI_ERR << "Error creating thread, errno was: " << c;
151  }
152  }
153  }
154  }
155  ~MultiThreader() {
156  for (size_t thread = 0; thread < cvec_.size(); thread++)
157  if (KALDI_PTHREAD_PTR(threads_[thread]) != 0)
158  if (pthread_join(threads_[thread], NULL))
159  KALDI_ERR << "Error rejoining thread.";
160  delete [] threads_;
161  }
162  private:
163  pthread_t *threads_;
164  std::vector<C> cvec_;
165 };
