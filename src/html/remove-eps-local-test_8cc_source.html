121 static void TestRemoveEpsLocalSpecial() {
122  // test that RemoveEpsLocalSpecial preserves equivalence in tropical while
123  // maintaining stochasticity in log.
124  typedef VectorFst<LogArc> Fst;
125  typedef LogArc::Weight Weight;
126  typedef LogArc::StateId StateId;
127  typedef LogArc Arc;
128  VectorFst<LogArc> *logfst = RandFst<LogArc>();
129 
130  { // Make the FST stochastic.
131  for (StateId s = 0; s < logfst->NumStates(); s++) {
132  Weight w = logfst->Final(s);
133  for (ArcIterator<Fst> aiter(*logfst, s); !aiter.Done(); aiter.Next()) {
134  w = Plus(w, aiter.Value().weight);
135  }
136  if (w != Weight::Zero()) {
137  logfst->SetFinal(s, Divide(logfst->Final(s), w, DIVIDE_ANY));
138  for (MutableArcIterator<Fst> aiter(logfst, s); !aiter.Done(); aiter.Next()) {
139  Arc a = aiter.Value();
140  a.weight = Divide(a.weight, w, DIVIDE_ANY);
141  aiter.SetValue(a);
142  }
143  }
144  }
145  }
146 #ifndef _MSC_VER
147  assert(IsStochasticFst(*logfst, kDelta*10));
148 #endif
149  {
150  std::cout << "logfst = \n";
151 #ifdef HAVE_OPENFST_GE_10400
152  FstPrinter<LogArc> fstprinter(*logfst, NULL, NULL, NULL, false, true, "\t");
153 #else
154  FstPrinter<LogArc> fstprinter(*logfst, NULL, NULL, NULL, false, true);
155 #endif
156  fstprinter.Print(&std::cout, "standard output");
157  }
158 
159  VectorFst<StdArc> fst;
160  Cast(*logfst, &fst);
161  VectorFst<StdArc> fst_copy(fst);
162  RemoveEpsLocalSpecial(&fst); // removes eps in std-arc but keep stochastic in log-arc
163  // make sure equivalent.
164  assert(RandEquivalent(fst, fst_copy, 5/*paths*/, 0.01/*delta*/, kaldi::Rand()/*seed*/, 100/*path length-- max?*/));
165  VectorFst<LogArc> logfst2;

171  FstPrinter<LogArc> fstprinter(logfst2, NULL, NULL, NULL, false, true, "\t");
172 #else
173  FstPrinter<LogArc> fstprinter(logfst2, NULL, NULL, NULL, false, true);
174 #endif
175  fstprinter.Print(&std::cout, "standard output");
176  }
177  if (ApproxEqual(ShortestDistance(*logfst), ShortestDistance(logfst2))) {
178  // make sure we preserved stochasticity in cases where doing so was
179  // possible... if the log-semiring total weight changed, then it is
180  // not possible so don't assert this.
181  assert(IsStochasticFst(logfst2, kDelta*10));
182  }
183  delete logfst;
184 }
