279  void ResetLstmStreams(const std::vector<int32> &stream_reset_flag) {
280  // allocate prev_nnet_state_ if not done yet,
281  if (nstream_ == 0) {
282  // Karel: we just got number of streams! (before the 1st batch comes)
283  nstream_ = stream_reset_flag.size();
284  prev_nnet_state_.Resize(nstream_, 7*ncell_ + 1*nrecur_, kSetZero);
285  KALDI_LOG << "Running training with " << nstream_ << " streams.";
286  }

296  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
297  int DEBUG = 0;
298 
299  static bool do_stream_reset = false;
300  if (nstream_ == 0) {
301  do_stream_reset = true;
302  nstream_ = 1; // Karel: we are in nnet-forward, so 1 stream,
303  prev_nnet_state_.Resize(nstream_, 7*ncell_ + 1*nrecur_, kSetZero);
304  KALDI_LOG << "Running nnet-forward with per-utterance LSTM-state reset";
305  }
306  if (do_stream_reset) prev_nnet_state_.SetZero();
307  KALDI_ASSERT(nstream_ > 0);
308 
309  KALDI_ASSERT(in.NumRows() % nstream_ == 0);
310  int32 T = in.NumRows() / nstream_;
311  int32 S = nstream_;
312 
313  // 0:forward pass history, [1, T]:current sequence, T+1:dummy
314  propagate_buf_.Resize((T+2)*S, 7 * ncell_ + nrecur_, kSetZero);
315  propagate_buf_.RowRange(0*S,S).CopyFromMat(prev_nnet_state_);
316 
317  // disassemble entire neuron activation buffer into different neurons
318  CuSubMatrix<BaseFloat> YG(propagate_buf_.ColRange(0*ncell_, ncell_));
319  CuSubMatrix<BaseFloat> YI(propagate_buf_.ColRange(1*ncell_, ncell_));
320  CuSubMatrix<BaseFloat> YF(propagate_buf_.ColRange(2*ncell_, ncell_));
321  CuSubMatrix<BaseFloat> YO(propagate_buf_.ColRange(3*ncell_, ncell_));
322  CuSubMatrix<BaseFloat> YC(propagate_buf_.ColRange(4*ncell_, ncell_));
323  CuSubMatrix<BaseFloat> YH(propagate_buf_.ColRange(5*ncell_, ncell_));
324  CuSubMatrix<BaseFloat> YM(propagate_buf_.ColRange(6*ncell_, ncell_));
325  CuSubMatrix<BaseFloat> YR(propagate_buf_.ColRange(7*ncell_, nrecur_));
326 
327  CuSubMatrix<BaseFloat> YGIFO(propagate_buf_.ColRange(0, 4*ncell_));
328 
329  // x -> g, i, f, o, not recurrent, do it all in once
330  YGIFO.RowRange(1*S,T*S).AddMatMat(1.0, in, kNoTrans, w_gifo_x_, kTrans, 0.0);
334  //if (dropout_rate_ != 0.0) {
335  // dropout_mask_.Resize(in.NumRows(), 4*ncell_, kUndefined);
336  // dropout_mask_.SetRandUniform(); // [0,1]
337  // dropout_mask_.Add(-dropout_rate_); // [-dropout_rate, 1-dropout_rate_],
338  // dropout_mask_.ApplyHeaviside(); // -tive -> 0.0, +tive -> 1.0
339  // YGIFO.RowRange(1*S,T*S).MulElements(dropout_mask_);
340  //}
341 
342  // bias -> g, i, f, o
343  YGIFO.RowRange(1*S,T*S).AddVecToRows(1.0, bias_);
344 
345  for (int t = 1; t <= T; t++) {
346  // multistream buffers for current time-step
347  CuSubMatrix<BaseFloat> y_g(YG.RowRange(t*S,S));
348  CuSubMatrix<BaseFloat> y_i(YI.RowRange(t*S,S));
349  CuSubMatrix<BaseFloat> y_f(YF.RowRange(t*S,S));
350  CuSubMatrix<BaseFloat> y_o(YO.RowRange(t*S,S));
351  CuSubMatrix<BaseFloat> y_c(YC.RowRange(t*S,S));
352  CuSubMatrix<BaseFloat> y_h(YH.RowRange(t*S,S));
353  CuSubMatrix<BaseFloat> y_m(YM.RowRange(t*S,S));
354  CuSubMatrix<BaseFloat> y_r(YR.RowRange(t*S,S));
355 
356  CuSubMatrix<BaseFloat> y_gifo(YGIFO.RowRange(t*S,S));
357 
358  // r(t-1) -> g, i, f, o
359  y_gifo.AddMatMat(1.0, YR.RowRange((t-1)*S,S), kNoTrans, w_gifo_r_, kTrans, 1.0);
360 
361  // c(t-1) -> i(t) via peephole
362  y_i.AddMatDiagVec(1.0, YC.RowRange((t-1)*S,S), kNoTrans, peephole_i_c_, 1.0);
363 
364  // c(t-1) -> f(t) via peephole
365  y_f.AddMatDiagVec(1.0, YC.RowRange((t-1)*S,S), kNoTrans, peephole_f_c_, 1.0);
366 
367  // i, f sigmoid squashing
368  y_i.Sigmoid(y_i);
369  y_f.Sigmoid(y_f);
370 
371  // g tanh squashing
372  y_g.Tanh(y_g);
373 
374  // g -> c
375  y_c.AddMatMatElements(1.0, y_g, y_i, 0.0);
376 
377  // c(t-1) -> c(t) via forget-gate
378  y_c.AddMatMatElements(1.0, YC.RowRange((t-1)*S,S), y_f, 1.0);
379 
380  y_c.ApplyFloor(-50); // optional clipping of cell activation
381  y_c.ApplyCeiling(50); // google paper Interspeech2014: LSTM for LVCSR
382 
383  // h tanh squashing
384  y_h.Tanh(y_c);
385 
386  // c(t) -> o(t) via peephole (non-recurrent) & o squashing
387  y_o.AddMatDiagVec(1.0, y_c, kNoTrans, peephole_o_c_, 1.0);
388 
389  // o sigmoid squashing
390  y_o.Sigmoid(y_o);
391 
392  // h -> m via output gate
393  y_m.AddMatMatElements(1.0, y_h, y_o, 0.0);
394 
395  // m -> r
396  y_r.AddMatMat(1.0, y_m, kNoTrans, w_r_m_, kTrans, 0.0);
397 
398  if (DEBUG) {
399  std::cerr << "forward-pass frame " << t << "\n";
400  std::cerr << "activation of g: " << y_g;
401  std::cerr << "activation of i: " << y_i;
402  std::cerr << "activation of f: " << y_f;
403  std::cerr << "activation of o: " << y_o;
404  std::cerr << "activation of c: " << y_c;
405  std::cerr << "activation of h: " << y_h;
406  std::cerr << "activation of m: " << y_m;
407  std::cerr << "activation of r: " << y_r;
408  }
409  }
410 
411  // recurrent projection layer is also feed-forward as LSTM output
412  out->CopyFromMat(YR.RowRange(1*S,T*S));
413 
414  // now the last frame state becomes previous network state for next batch
415  prev_nnet_state_.CopyFromMat(propagate_buf_.RowRange(T*S,S));
416  }

418  void BackpropagateFnc(const CuMatrixBase<BaseFloat> &in, const CuMatrixBase<BaseFloat> &out,
419  const CuMatrixBase<BaseFloat> &out_diff, CuMatrixBase<BaseFloat> *in_diff) {
420 
421  int DEBUG = 0;
422 
423  int32 T = in.NumRows() / nstream_;
424  int32 S = nstream_;
425 
426  // disassemble propagated buffer into neurons
427  CuSubMatrix<BaseFloat> YG(propagate_buf_.ColRange(0*ncell_, ncell_));
428  CuSubMatrix<BaseFloat> YI(propagate_buf_.ColRange(1*ncell_, ncell_));
429  CuSubMatrix<BaseFloat> YF(propagate_buf_.ColRange(2*ncell_, ncell_));
430  CuSubMatrix<BaseFloat> YO(propagate_buf_.ColRange(3*ncell_, ncell_));
431  CuSubMatrix<BaseFloat> YC(propagate_buf_.ColRange(4*ncell_, ncell_));
432  CuSubMatrix<BaseFloat> YH(propagate_buf_.ColRange(5*ncell_, ncell_));
433  CuSubMatrix<BaseFloat> YM(propagate_buf_.ColRange(6*ncell_, ncell_));
434  CuSubMatrix<BaseFloat> YR(propagate_buf_.ColRange(7*ncell_, nrecur_));
435 
436  // 0:dummy, [1,T] frames, T+1 backward pass history
437  backpropagate_buf_.Resize((T+2)*S, 7 * ncell_ + nrecur_, kSetZero);
438 
439  // disassemble backpropagate buffer into neurons
440  CuSubMatrix<BaseFloat> DG(backpropagate_buf_.ColRange(0*ncell_, ncell_));
441  CuSubMatrix<BaseFloat> DI(backpropagate_buf_.ColRange(1*ncell_, ncell_));
442  CuSubMatrix<BaseFloat> DF(backpropagate_buf_.ColRange(2*ncell_, ncell_));
443  CuSubMatrix<BaseFloat> DO(backpropagate_buf_.ColRange(3*ncell_, ncell_));
444  CuSubMatrix<BaseFloat> DC(backpropagate_buf_.ColRange(4*ncell_, ncell_));
445  CuSubMatrix<BaseFloat> DH(backpropagate_buf_.ColRange(5*ncell_, ncell_));
446  CuSubMatrix<BaseFloat> DM(backpropagate_buf_.ColRange(6*ncell_, ncell_));
447  CuSubMatrix<BaseFloat> DR(backpropagate_buf_.ColRange(7*ncell_, nrecur_));
448 
449  CuSubMatrix<BaseFloat> DGIFO(backpropagate_buf_.ColRange(0, 4*ncell_));
450 
451  // projection layer to LSTM output is not recurrent, so backprop it all in once
452  DR.RowRange(1*S,T*S).CopyFromMat(out_diff);
453 
454  for (int t = T; t >= 1; t--) {
455  CuSubMatrix<BaseFloat> y_g(YG.RowRange(t*S,S));
456  CuSubMatrix<BaseFloat> y_i(YI.RowRange(t*S,S));
457  CuSubMatrix<BaseFloat> y_f(YF.RowRange(t*S,S));
458  CuSubMatrix<BaseFloat> y_o(YO.RowRange(t*S,S));
459  CuSubMatrix<BaseFloat> y_c(YC.RowRange(t*S,S));
460  CuSubMatrix<BaseFloat> y_h(YH.RowRange(t*S,S));
461  CuSubMatrix<BaseFloat> y_m(YM.RowRange(t*S,S));
462  CuSubMatrix<BaseFloat> y_r(YR.RowRange(t*S,S));
463 
464  CuSubMatrix<BaseFloat> d_g(DG.RowRange(t*S,S));
465  CuSubMatrix<BaseFloat> d_i(DI.RowRange(t*S,S));
466  CuSubMatrix<BaseFloat> d_f(DF.RowRange(t*S,S));
467  CuSubMatrix<BaseFloat> d_o(DO.RowRange(t*S,S));
468  CuSubMatrix<BaseFloat> d_c(DC.RowRange(t*S,S));
469  CuSubMatrix<BaseFloat> d_h(DH.RowRange(t*S,S));
470  CuSubMatrix<BaseFloat> d_m(DM.RowRange(t*S,S));
471  CuSubMatrix<BaseFloat> d_r(DR.RowRange(t*S,S));
472 
473  // r
474  // Version 1 (precise gradients):
475  // backprop error from g(t+1), i(t+1), f(t+1), o(t+1) to r(t)
476  d_r.AddMatMat(1.0, DGIFO.RowRange((t+1)*S,S), kNoTrans, w_gifo_r_, kNoTrans, 1.0);
477 
478  /*
479  // Version 2 (Alex Graves' PhD dissertation):
480  // only backprop g(t+1) to r(t)
481  CuSubMatrix<BaseFloat> w_g_r_(w_gifo_r_.RowRange(0, ncell_));
482  d_r.AddMatMat(1.0, DG.RowRange((t+1)*S,S), kNoTrans, w_g_r_, kNoTrans, 1.0);
483  */
484 
485  /*
486  // Version 3 (Felix Gers' PhD dissertation):
487  // truncate gradients of g(t+1), i(t+1), f(t+1), o(t+1) once they leak out memory block
488  // CEC(with forget connection) is the only "error-bridge" through time
489  */
490 
491  // r -> m
492  d_m.AddMatMat(1.0, d_r, kNoTrans, w_r_m_, kNoTrans, 0.0);
493 
494  // m -> h via output gate
495  d_h.AddMatMatElements(1.0, d_m, y_o, 0.0);
496  d_h.DiffTanh(y_h, d_h);
497 
498  // o
499  d_o.AddMatMatElements(1.0, d_m, y_h, 0.0);
500  d_o.DiffSigmoid(y_o, d_o);
501 
502  // c
503  // 1. diff from h(t)
504  // 2. diff from c(t+1) (via forget-gate between CEC)
505  // 3. diff from i(t+1) (via peephole)
506  // 4. diff from f(t+1) (via peephole)
507  // 5. diff from o(t) (via peephole, not recurrent)
508  d_c.AddMat(1.0, d_h);
509  d_c.AddMatMatElements(1.0, DC.RowRange((t+1)*S,S), YF.RowRange((t+1)*S,S), 1.0);
510  d_c.AddMatDiagVec(1.0, DI.RowRange((t+1)*S,S), kNoTrans, peephole_i_c_, 1.0);
511  d_c.AddMatDiagVec(1.0, DF.RowRange((t+1)*S,S), kNoTrans, peephole_f_c_, 1.0);
512  d_c.AddMatDiagVec(1.0, d_o , kNoTrans, peephole_o_c_, 1.0);
513 
514  // f
515  d_f.AddMatMatElements(1.0, d_c, YC.RowRange((t-1)*S,S), 0.0);
516  d_f.DiffSigmoid(y_f, d_f);
517 
518  // i
519  d_i.AddMatMatElements(1.0, d_c, y_g, 0.0);
520  d_i.DiffSigmoid(y_i, d_i);
521 
522  // c -> g via input gate
523  d_g.AddMatMatElements(1.0, d_c, y_i, 0.0);
524  d_g.DiffTanh(y_g, d_g);
525 
526  // debug info
527  if (DEBUG) {
528  std::cerr << "backward-pass frame " << t << "\n";
529  std::cerr << "derivative wrt input r " << d_r;
530  std::cerr << "derivative wrt input m " << d_m;
531  std::cerr << "derivative wrt input h " << d_h;
532  std::cerr << "derivative wrt input o " << d_o;
533  std::cerr << "derivative wrt input c " << d_c;
534  std::cerr << "derivative wrt input f " << d_f;
535  std::cerr << "derivative wrt input i " << d_i;
536  std::cerr << "derivative wrt input g " << d_g;
537  }
538  }
539 
540  // g,i,f,o -> x, do it all in once
541  in_diff->AddMatMat(1.0, DGIFO.RowRange(1*S,T*S), kNoTrans, w_gifo_x_, kNoTrans, 0.0);
542 
544  //if (dropout_rate_ != 0.0) {
545  // in_diff->MulElements(dropout_mask_);
546  //}
547 
548  // calculate delta
549  const BaseFloat mmt = opts_.momentum;
550 
551  // weight x -> g, i, f, o
552  w_gifo_x_corr_.AddMatMat(1.0, DGIFO.RowRange(1*S,T*S), kTrans,
553  in , kNoTrans, mmt);
554  // recurrent weight r -> g, i, f, o
555  w_gifo_r_corr_.AddMatMat(1.0, DGIFO.RowRange(1*S,T*S), kTrans,
556  YR.RowRange(0*S,T*S) , kNoTrans, mmt);
557  // bias of g, i, f, o
558  bias_corr_.AddRowSumMat(1.0, DGIFO.RowRange(1*S,T*S), mmt);
559 
560  // recurrent peephole c -> i
561  peephole_i_c_corr_.AddDiagMatMat(1.0, DI.RowRange(1*S,T*S), kTrans,
562  YC.RowRange(0*S,T*S), kNoTrans, mmt);
563  // recurrent peephole c -> f
564  peephole_f_c_corr_.AddDiagMatMat(1.0, DF.RowRange(1*S,T*S), kTrans,
565  YC.RowRange(0*S,T*S), kNoTrans, mmt);
566  // peephole c -> o
567  peephole_o_c_corr_.AddDiagMatMat(1.0, DO.RowRange(1*S,T*S), kTrans,
568  YC.RowRange(1*S,T*S), kNoTrans, mmt);
569 
570  w_r_m_corr_.AddMatMat(1.0, DR.RowRange(1*S,T*S), kTrans,
571  YM.RowRange(1*S,T*S), kNoTrans, mmt);
572 
573  if (clip_gradient_ > 0.0) {
574  w_gifo_x_corr_.ApplyFloor(-clip_gradient_);
575  w_gifo_x_corr_.ApplyCeiling(clip_gradient_);
576  w_gifo_r_corr_.ApplyFloor(-clip_gradient_);
577  w_gifo_r_corr_.ApplyCeiling(clip_gradient_);
578  bias_corr_.ApplyFloor(-clip_gradient_);
579  bias_corr_.ApplyCeiling(clip_gradient_);
580  w_r_m_corr_.ApplyFloor(-clip_gradient_);
581  w_r_m_corr_.ApplyCeiling(clip_gradient_);
582  peephole_i_c_corr_.ApplyFloor(-clip_gradient_);
583  peephole_i_c_corr_.ApplyCeiling(clip_gradient_);
584  peephole_f_c_corr_.ApplyFloor(-clip_gradient_);
585  peephole_f_c_corr_.ApplyCeiling(clip_gradient_);
586  peephole_o_c_corr_.ApplyFloor(-clip_gradient_);
587  peephole_o_c_corr_.ApplyCeiling(clip_gradient_);
588  }
589 
590  if (DEBUG) {
591  std::cerr << "gradients(with optional momentum): \n";
592  std::cerr << "w_gifo_x_corr_ " << w_gifo_x_corr_;
593  std::cerr << "w_gifo_r_corr_ " << w_gifo_r_corr_;
594  std::cerr << "bias_corr_ " << bias_corr_;
595  std::cerr << "w_r_m_corr_ " << w_r_m_corr_;
596  std::cerr << "peephole_i_c_corr_ " << peephole_i_c_corr_;
597  std::cerr << "peephole_f_c_corr_ " << peephole_f_c_corr_;
598  std::cerr << "peephole_o_c_corr_ " << peephole_o_c_corr_;
599  }

603  const BaseFloat lr = opts_.learn_rate;
604 
605  w_gifo_x_.AddMat(-lr, w_gifo_x_corr_);

615 // /*
616 // Here we deal with the famous "vanishing & exploding difficulties" in RNN learning.
617 //
618 // *For gradients vanishing*
619 // LSTM architecture introduces linear CEC as the "error bridge" across long time distance
620 // solving vanishing problem.
621 //
622 // *For gradients exploding*
623 // LSTM is still vulnerable to gradients explosing in BPTT(with large weight & deep time expension).
624 // To prevent this, we tried L2 regularization, which didn't work well
625 //
626 // Our approach is a *modified* version of Max Norm Regularization:
627 // For each nonlinear neuron,
628 // 1. fan-in weights & bias model a seperation hyper-plane: W x + b = 0
629 // 2. squashing function models a differentiable nonlinear slope around this hyper-plane.
630 //
631 // Conventional max norm regularization scale W to keep its L2 norm bounded,
632 // As a modification, we scale down large (W & b) *simultaneously*, this:
633 // 1. keeps all fan-in weights small, prevents gradients from exploding during backward-pass.
634 // 2. keeps the location of the hyper-plane unchanged, so we don't wipe out already learned knowledge.
635 // 3. shrinks the "normal" of the hyper-plane, smooths the nonlinear slope, improves generalization.
636 // 4. makes the network *well-conditioned* (weights are constrained in a reasonible range).
637 //
638 // We've observed faster convergence and performance gain by doing this.
639 // */
