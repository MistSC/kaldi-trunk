130 void AddToClustersOptimized(const std::vector<Clusterable*> &stats,
131  const std::vector<int32> &assignments,
132  const Clusterable &total,
133  std::vector<Clusterable*> *clusters) {

138  tot_norm = total.Normalizer();

140  }

144  int32 size = stats.size();
145  if (size == 0) return; // Nothing to do.
146  KALDI_ASSERT(clusters != NULL);
147  int32 num_clust = 1 + *std::max_element(assignments.begin(),
148  assignments.end());
149  if (static_cast<int32> (clusters->size()) < num_clust)
150  clusters->resize(num_clust, NULL); // extend with NULLs.
151  std::vector<int32> num_stats_for_cluster(num_clust, 0);
152  int32 num_total_stats = 0;
153  for (int32 i = 0; i < size; i++) {
154  if (stats[i] != NULL) {
155  num_total_stats++;
156  num_stats_for_cluster[assignments[i]]++;
157  }
158  }
159  if (num_total_stats == 0) return; // Nothing to do.
160 
161  // it will only ever be efficient to subtract for at most one index.
162  int32 subtract_index = -1;
163  for (int32 c = 0; c < num_clust; c++) {
164  if (num_stats_for_cluster[c] > num_total_stats - num_stats_for_cluster[c]) {
165  subtract_index = c;
166  if ((*clusters)[c] == NULL)
167  (*clusters)[c] = total.Copy();
168  else
169  (*clusters)[c]->Add(total);
170  break;
171  }
172  }
173 
174  for (int32 i = 0; i < size; i++) {
175  if (stats[i] != NULL) {
176  int32 assignment = assignments[i];
177  if (assignment != (int32) subtract_index) {
178  if ((*clusters)[assignment] == NULL)
179  (*clusters)[assignment] = stats[i]->Copy();
180  else
181  (*clusters)[assignment]->Add(*(stats[i]));
182  }
183  if (subtract_index != -1 && assignment != subtract_index)
184  (*clusters)[subtract_index]->Sub(*(stats[i]));
185  }
186  }

345 void BottomUpClusterer::MergeClusters(int32 i, int32 j) {
346  KALDI_ASSERT(i != j && i < npoints_ && j < npoints_);
347  (*clusters_)[i]->Add(*((*clusters_)[j]));
348  delete (*clusters_)[j];
349  (*clusters_)[j] = NULL;
350  // note that we may have to follow the chain within "assignment_" to get
351  // final assignments.
352  (*assignments_)[j] = i;
353  // subtract negated objective function change, i.e. add objective function
354  // change.
355  ans_ -= dist_vec_[(i * (i - 1)) / 2 + j];
356  nclusters_--;
357  // Now update "distances".
358  for (int32 k = 0; k < npoints_; k++) {
359  if (k != i && (*clusters_)[k] != NULL) {
360  if (k < i)
361  SetDistance(i, k); // SetDistance requires k < i.
362  else
363  SetDistance(k, i);
364  }
365  }

516 void CompartmentalizedBottomUpClusterer::Renumber(int32 comp) {
517  // first free up memory by getting rid of queue. this is a special trick
518  // to force STL to free memory.
519  {
520  QueueType tmp;
521  std::swap(tmp, queue_);
522  }
523 
524  // First find the number of surviving clusters in the compartment.
525  int32 clusts_in_compartment = 0;
526  for (int32 i = 0; i < npoints_[comp]; i++) {
527  if (clusters_[comp][i] != NULL)
528  clusts_in_compartment++;
529  }
530  KALDI_ASSERT(clusts_in_compartment <= nclusters_);
531 
532  // mapping from intermediate to final clusters.
533  vector<uint_smaller> mapping(npoints_[comp], static_cast<uint_smaller> (-1));
534  vector<Clusterable*> new_clusters(clusts_in_compartment);
535 
536  // Now copy the surviving clusters in a fresh array.
537  clusts_in_compartment = 0;
538  for (int32 i = 0; i < npoints_[comp]; i++) {
539  if (clusters_[comp][i] != NULL) {
540  new_clusters[clusts_in_compartment] = clusters_[comp][i];
541  mapping[i] = clusts_in_compartment;
542  clusts_in_compartment++;
543  }
544  }
545 
546  // Next, process the assignments.
547  std::vector<int32> new_assignments(npoints_[comp]);
548  for (int32 i = 0; i < npoints_[comp]; i++) {
549  int32 ii = i;
550  while (assignments_[comp][ii] != ii)
551  ii = assignments_[comp][ii]; // follow the chain.
552  // cannot assign to nonexistent cluster.
553  KALDI_ASSERT(clusters_[comp][ii] != NULL);
554  KALDI_ASSERT(mapping[ii] != static_cast<uint_smaller>(-1));
555  new_assignments[i] = mapping[ii];
556  }
557  clusters_[comp].swap(new_clusters);
558  assignments_[comp].swap(new_assignments);
559 }

598  clusters_[comp][j] = NULL;
599  // note that we may have to follow the chain within "assignment_" to get
600  // final assignments.
601  assignments_[comp][j] = i;
602  // objective function change.
603  BaseFloat ans = -dist_vec_[comp][(i * (i - 1)) / 2 + j];
604  nclusters_--;
605  // Now update "distances".
606  for (int32 k = 0; k < npoints_[comp]; k++) {
607  if (k != i && clusters_[comp][k] != NULL) {
608  if (k < i)
609  SetDistance(comp, i, k); // SetDistance requires k < i.
610  else
611  SetDistance(comp, k, i);
612  }
613  }
614  // Control memory use by getting rid of orphaned queue entries every time
615  // it's at least twice the maximum possible size.
616  if (queue_.size() >= static_cast<size_t> (nclusters_ * nclusters_)) {
617  ReconstructQueue();
618  }
619  return ans;
620 }

629  for (int32 i = 0; i < npoints_[comp]; i++) {

631  for (int32 j = 0; j < i; j++) {
632  if (clusters_[comp][j] == NULL) continue;
633  SetDistance(comp, i, j);

685 class RefineClusterer {

687  public:
688  // size used in point_info structure (we store a lot of these so don't want
689  // to just make it int32). Also used as a time-id (cannot have more moves of
690  // points, than can fit in this time). Must be big enough to store num-clust.

692  typedef uint_smaller ClustIndexInt;
693 
694  RefineClusterer(const std::vector<Clusterable*> &points,
695  std::vector<Clusterable*> *clusters,
696  std::vector<int32> *assignments,
697  RefineClustersOptions cfg)
698  : points_(points), clusters_(clusters), assignments_(assignments),
699  cfg_(cfg) {
700  KALDI_ASSERT(cfg_.top_n >= 2);
701  num_points_ = points_.size();
702  num_clust_ = static_cast<int32> (clusters->size());
703 
704  // so can fit clust-id in LocalInt
705  if (cfg_.top_n > (int32) num_clust_) cfg_.top_n
706  = static_cast<int32> (num_clust_);
707  KALDI_ASSERT(cfg_.top_n == static_cast<int32>(static_cast<ClustIndexInt>(cfg_.top_n)));
708  t_ = 0;
709  my_clust_index_.resize(num_points_);
710  // will set all PointInfo's to 0 too (they will be up-to-date).
711  clust_time_.resize(num_clust_, 0);
712  clust_objf_.resize(num_clust_);
713  for (int32 i = 0; i < num_clust_; i++)
714  clust_objf_[i] = (*clusters_)[i]->Objf();
715  info_.resize(num_points_ * cfg_.top_n);
716  ans_ = 0;
717  InitPoints();
718  }
719 
720  BaseFloat Refine() {
721  if (cfg_.top_n <= 1) return 0.0; // nothing to do.
722  Iterate();
723  return ans_;
724  }
725  // at some point check cfg_.top_n > 1 after maxing to num_clust_.
726  private:
727  void InitPoint(int32 point) {
728  // Find closest clusters to this point.
729  // distances are really negated objf changes, ignoring terms that don't vary with the "other" cluster.
730 
731  std::vector<std::pair<BaseFloat, LocalInt> > distances;
732  distances.reserve(num_clust_-1);
733  int32 my_clust = (*assignments_)[point];
734  Clusterable *point_cl = points_[point];
735 
736  for (int32 clust = 0;clust < num_clust_;clust++) {
737  if (clust != my_clust) {
738  Clusterable *tmp = (*clusters_)[clust]->Copy();
739  tmp->Add(*point_cl);
740  BaseFloat other_clust_objf = clust_objf_[clust];
741  BaseFloat other_clust_plus_me_objf = (*clusters_)[clust]->ObjfPlus(* (points_[point]));
742 
743  BaseFloat distance = other_clust_objf-other_clust_plus_me_objf; // negated delta-objf, with only "varying" terms.
744  distances.push_back(std::make_pair(distance, (LocalInt)clust));
745  delete tmp;
746  }
747  }
748  if ((cfg_.top_n-1-1) >= 0) {
749  std::nth_element(distances.begin(), distances.begin()+(cfg_.top_n-1-1), distances.end());
750  }
751  // top_n-1 is the # of elements we want to retain. -1 because we need the iterator
752  // that points to the end of that range (i.e. not potentially off the end of the array).
753 
754  for (int32 index = 0;index < cfg_.top_n-1;index++) {
755  point_info &info = GetInfo(point, index);
756  int32 clust = distances[index].second;
757  info.clust = clust;
758  BaseFloat distance = distances[index].first;
759  BaseFloat other_clust_objf = clust_objf_[clust];
760  BaseFloat other_clust_plus_me_objf = -(distance - other_clust_objf);
761  info.objf = other_clust_plus_me_objf;
762  info.time = 0;
763  }
764  // now put the last element in, which is my current cluster.
765  point_info &info = GetInfo(point, cfg_.top_n-1);
766  info.clust = my_clust;
767  info.time = 0;
768  info.objf = (*clusters_)[my_clust]->ObjfMinus(*(points_[point]));
769  my_clust_index_[point] = cfg_.top_n-1;
770  }
771  void InitPoints() {
772  // finds, for each point, the closest cfg_.top_n clusters (including its own cluster).
773  // this may be the most time-consuming step of the algorithm.
774  for (int32 p = 0;p < num_points_;p++) InitPoint(p);
775  }
776  void Iterate() {
777  int32 iter, num_iters = cfg_.num_iters;
778  for (iter = 0;iter < num_iters;iter++) {
779  int32 cur_t = t_;
780  for (int32 point = 0;point < num_points_;point++) {
781  if (t_+1 == 0) {
782  KALDI_WARN << "Stopping iterating at int32 moves";
783  return; // once we use up all time points, must return-- this
784  // should rarely happen as int32 is large.
785  }
786  ProcessPoint(point);
787  }
788  if (t_ == cur_t) break; // nothing changed so we converged.
789  }
790  }
791  void MovePoint(int32 point, int32 new_index) {
792  // move point to a different cluster.
793  t_++;
794  int32 old_index = my_clust_index_[point]; // index into info
795  // array corresponding to current cluster.
796  KALDI_ASSERT(new_index < cfg_.top_n && new_index != old_index);
797  point_info &old_info = GetInfo(point, old_index),
798  &new_info = GetInfo(point, new_index);
799  my_clust_index_[point] = new_index; // update to new index.
800 
801  int32 old_clust = old_info.clust, new_clust = new_info.clust;
802  KALDI_ASSERT( (*assignments_)[point] == old_clust);
803  (*assignments_)[point] = new_clust;
804  (*clusters_)[old_clust]->Sub( *(points_[point]) );
805  (*clusters_)[new_clust]->Add( *(points_[point]) );
806  UpdateClust(old_clust);
807  UpdateClust(new_clust);
808  }
809  void UpdateClust(int32 clust) {
810  KALDI_ASSERT(clust < num_clust_);
811  clust_objf_[clust] = (*clusters_)[clust]->Objf();
812  clust_time_[clust] = t_;
813  }
814  void ProcessPoint(int32 point) {
815  // note: calling code uses the fact
816  // that it only ever increases t_ by one.
817  KALDI_ASSERT(point < num_points_);
818  // (1) Make sure own-cluster like is updated.
819  int32 self_index = my_clust_index_[point]; // index <cfg_.top_n of own cluster.
820  point_info &self_info = GetInfo(point, self_index);
821  int32 self_clust = self_info.clust; // cluster index of own cluster.
822  KALDI_ASSERT(self_index < cfg_.top_n);
823  UpdateInfo(point, self_index);
824 
825  float own_clust_objf = clust_objf_[self_clust];
826  float own_clust_minus_me_objf = self_info.objf; // objf of own cluster minus self.
827  // Now check the other "close" clusters and see if we want to move there.
828  for (int32 index = 0;index < cfg_.top_n;index++) {
829  if (index != self_index) {
830  UpdateInfo(point, index);
831  point_info &other_info = GetInfo(point, index);
832  BaseFloat other_clust_objf = clust_objf_[other_info.clust];
833  BaseFloat other_clust_plus_me_objf = other_info.objf;
834  BaseFloat impr = other_clust_plus_me_objf + own_clust_minus_me_objf
835  - other_clust_objf - own_clust_objf;
836  if (impr > 0) { // better to switch...
837  ans_ += impr;
838  MovePoint(point, index);
839  return; // the stuff we precomputed at the top is invalidated now, and it's
840  // easiest just to wait till next time we visit this point.
841  }
842  }
843  }
844  }

851  tmp_cl->Sub( *(points_[point]) );
852  } else{
853  tmp_cl->Add( *(points_[point]) );
854  }
855  pinfo.time = t_;
856  pinfo.objf = tmp_cl->Objf();
857  delete tmp_cl;
858  }

917 BaseFloat ClusterKMeansOnce(const std::vector<Clusterable*> &points,
918  int32 num_clust,
919  std::vector<Clusterable*> *clusters_out,
920  std::vector<int32> *assignments_out,
921  ClusterKMeansOptions &cfg) {
922  std::vector<int32> my_assignments;
923  int32 num_points = points.size();
924  KALDI_ASSERT(clusters_out != NULL);
925  KALDI_ASSERT(num_points != 0);
926  KALDI_ASSERT(num_clust <= num_points);
927 
928  KALDI_ASSERT(clusters_out->empty()); // or we wouldn't know what to do with pointers in there.
929  clusters_out->resize(num_clust, (Clusterable*)NULL);
930  assignments_out->resize(num_points);
931 
932  { // This block assigns points to clusters.
933  // This is done pseudo-randomly using Rand() so that
934  // if we call ClusterKMeans multiple times we get different answers (so we can choose
935  // the best if we want).
936  int32 skip; // randomly choose a "skip" that's coprime to num_points.
937  if (num_points == 1) {
938  skip = 1;
939  } else {
940  skip = 1 + (Rand() % (num_points-1)); // a number between 1 and num_points-1.
941  while (Gcd(skip, num_points) != 1) { // while skip is not coprime to num_points...
942  if (skip == num_points-1) skip = 0;
943  skip++; // skip is now still betweeen 1 and num_points-1. will cycle through
944  // all of 1...num_points-1.
945  }
946  }
947  int32 i, j, count = 0;
948  for (i = 0, j = 0; count != num_points;i = (i+skip)%num_points, j = (j+1)%num_clust, count++) {
949  // i cycles pseudo-randomly through all points; j skips ahead by 1 each time
950  // modulo num_points.
951  // assign point i to cluster j.
952  if ((*clusters_out)[j] == NULL) (*clusters_out)[j] = points[i]->Copy();
953  else (*clusters_out)[j]->Add(*(points[i]));
954  (*assignments_out)[i] = j;
955  }
956  }
957 
958 
959  BaseFloat normalizer = SumClusterableNormalizer(*clusters_out);
960  BaseFloat ans;
961  { // work out initial value of "ans" (objective function improvement).
962  Clusterable *all_stats = SumClusterable(*clusters_out);
963  ans = SumClusterableObjf(*clusters_out) - all_stats->Objf(); // improvement just from the random
964  // initialization.
965  if (ans < -0.01 && ans < -0.01 * fabs(all_stats->Objf())) { // something bad happend.
966  KALDI_WARN << "ClusterKMeans: objective function after random assignment to clusters is worse than in single cluster: "<< (all_stats->Objf()) << " changed by " << ans << ". Perhaps your stats class has the wrong properties?";
967  }
968  delete all_stats;
969  }
970  for (int32 iter = 0;iter < cfg.num_iters;iter++) {
971  // Keep refining clusters by reassigning points.
972  BaseFloat objf_before;
973  if (cfg.verbose) objf_before =SumClusterableObjf(*clusters_out);
974  BaseFloat impr = RefineClusters(points, clusters_out, assignments_out, cfg.refine_cfg);
975  BaseFloat objf_after;
976  if (cfg.verbose) objf_after = SumClusterableObjf(*clusters_out);
977  ans += impr;
978  if (cfg.verbose)
979  KALDI_LOG << "ClusterKMeans: on iteration "<<(iter)<<", objf before = "<<(objf_before)<<", impr = "<<(impr)<<", objf after = "<<(objf_after)<<", normalized by "<<(normalizer)<<" = "<<(objf_after/normalizer);
980  if (impr == 0) break;
981  }
982  return ans;
983 }

985 BaseFloat ClusterKMeans(const std::vector<Clusterable*> &points,
986  int32 num_clust,
987  std::vector<Clusterable*> *clusters_out,
988  std::vector<int32> *assignments_out,
989  ClusterKMeansOptions cfg) {
990  if (points.size() == 0) {
991  if (clusters_out) KALDI_ASSERT(clusters_out->empty()); // or we wouldn't know whether to free the pointers.
992  if (assignments_out) assignments_out->clear();
993  return 0.0;
994  }
995  KALDI_ASSERT(cfg.num_tries>=1 && cfg.num_iters>=1);
996  if (clusters_out) KALDI_ASSERT(clusters_out->empty()); // or we wouldn't know whether to deallocate.
997  if (cfg.num_tries == 1) {
998  std::vector<int32> assignments;
999  return ClusterKMeansOnce(points, num_clust, clusters_out, (assignments_out != NULL?assignments_out:&assignments), cfg);
1000  } else { // multiple tries.
1001  if (clusters_out) {
1002  KALDI_ASSERT(clusters_out->empty()); // we don't know the ownership of any pointers in there, otherwise.
1003  }
1004  BaseFloat best_ans = 0.0;
1005  for (int32 i = 0;i < cfg.num_tries;i++) {
1006  std::vector<Clusterable*> clusters_tmp;
1007  std::vector<int32> assignments_tmp;
1008  BaseFloat ans = ClusterKMeansOnce(points, num_clust, &clusters_tmp, &assignments_tmp, cfg);
1009  KALDI_ASSERT(!ContainsNullPointers(clusters_tmp));
1010  if (i == 0 || ans > best_ans) {
1011  best_ans = ans;
1012  if (clusters_out) {
1013  if (clusters_out->size()) DeletePointers(clusters_out);
1014  *clusters_out = clusters_tmp;
1015  clusters_tmp.clear(); // suppress deletion of pointers.
1016  }
1017  if (assignments_out) *assignments_out = assignments_tmp;
1018  }
1019  // delete anything remaining in clusters_tmp (we cleared it if we used
1020  // the pointers.
1021  DeletePointers(&clusters_tmp);
1022  }
1023  return best_ans;
1024  }
