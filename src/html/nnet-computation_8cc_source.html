29 bool ComputationRequest::NeedDerivatives() const {
30  bool ans = false;
31  if (need_model_derivative)
32  ans = true;
33  for (size_t i = 0; i < inputs.size(); i++) {
34  if (inputs[i].has_deriv) { // derivative requested for this input
35  ans = true;
36  break;
37  }
38  }
39  if (ans) {
40  // check that the output actually provides a derivative, else the
41  // request could not be meaningfully satisfied.
42  size_t i;
43  for (i = 0; i < outputs.size(); i++)
44  if (outputs[i].has_deriv)
45  break;
46  if (i == outputs.size()) {
47  KALDI_ERR << "You requested model derivatives or input derivatives, but "
48  << "provide no derivatives at the output.";
49  }
50  }
51  return ans;
52 }

101 int32 NnetComputation::NewSubMatrix(int32 base_submatrix,
102  int32 row_offset, int32 num_rows,
103  int32 col_offset, int32 num_cols) {
104  KALDI_ASSERT(base_submatrix > 0 &&
105  static_cast<size_t>(base_submatrix) < submatrices.size());
106  const SubMatrixInfo &base_info = submatrices[base_submatrix];
107  int32 base_matrix = base_info.matrix_index;
108  KALDI_ASSERT(base_matrix > 0 &&
109  static_cast<size_t>(base_matrix) < matrices.size());
110  if (num_rows == -1) // we interpret this to mean 'as many as possible'.
111  num_rows = base_info.num_rows - row_offset;
112  if (num_cols == -1) // we interpret this to mean 'as many as possible'.
113  num_cols = base_info.num_cols - col_offset;
114  KALDI_ASSERT(row_offset + num_rows <= base_info.num_rows &&
115  col_offset + num_cols <= base_info.num_cols &&
116  row_offset >= 0 && col_offset >= 0 &&
117  num_rows > 0 && num_cols > 0);
118  int32 matrix_row_offset = base_info.row_offset + row_offset,
119  matrix_col_offset = base_info.col_offset + col_offset;
120  int32 ans = submatrices.size();
121  submatrices.push_back(
122  NnetComputation::SubMatrixInfo(base_matrix, matrix_row_offset, num_rows,
123  matrix_col_offset, num_cols));
124  return ans;
125 }

165  << submat.col_offset << ':' << (submat.col_offset + submat.num_cols - 1)

169  }

213  col_end = col_start + submat.num_cols;
214  if (!(row_index < submat.num_rows &&

219  << ':' << (submat.row_offset + submat.num_rows - 1)
220  << ',' << submat.col_offset << ':'
221  << (submat.col_offset + submat.num_cols - 1) << ") has "
222  << submat.num_rows << " rows, but you access row "
223  << row_index;
224  }
225  if (col_start == 0 && col_end == mat.num_cols)
226  os << 'm' << submat.matrix_index << '(' << row << ",:)";
227  else
228  os << 'm' << submat.matrix_index << '(' << row << ',' << col_start
229  << ':' << (col_end - 1) << ')';

231  if (j + 1 < size) os << ",";
232  }
233  os << "]";
234  (*indexes_multi_strings)[i] = os.str();

239 // writes to "os" the statement for this command.
240 static void PrintCommand(std::ostream &os,
241  const Nnet &nnet,
242  const NnetComputation &computation,
243  int32 command_index,
244  const std::vector<std::string> &submatrix_strings,
245  const std::vector<std::string> &indexes_strings,
246  const std::vector<std::string> &indexes_multi_strings) {
247  KALDI_ASSERT(command_index < computation.commands.size());
248  os << "c" << command_index << ": ";
249  const NnetComputation::Command &c = computation.commands[command_index];
250  switch (c.command_type) {
251  case kAllocMatrixZeroed:
252  os << "m" << c.arg1 << " = zeros("
253  << computation.matrices[c.arg1].num_rows
254  << ',' << computation.matrices[c.arg1].num_cols << ")\n";
255  break;
256  case kAllocMatrixUndefined:
257  os << "m" << c.arg1 << " = undefined("
258  << computation.matrices[c.arg1].num_rows
259  << ',' << computation.matrices[c.arg1].num_cols << ")\n";
260  break;
261  case kDeallocMatrix:
262  os << "m" << c.arg1 << " = []\n";
263  break;
264  case kAllocMatrixFromOther:
265  os << "m" << c.arg1 << ".swap(m" << c.arg2 << ") [dim = "
266  << computation.matrices[c.arg1].num_rows << " x "
267  << computation.matrices[c.arg1].num_cols << "]\n";
268  break;
269  case kAllocMatrixFromOtherZeroed:
270  os << "m" << c.arg1 << ".swap(m" << c.arg2 << ") [dim = "
271  << computation.matrices[c.arg1].num_rows << " x "
272  << computation.matrices[c.arg1].num_cols << "]; m"
273  << c.arg1 << ".zero();\n";
274  break;
275  case kPropagate:
276  os << nnet.GetComponentName(c.arg1) << ".Propagate(";
277  if (c.arg2 == 0) os << "NULL, ";
278  else os << "precomputed_indexes[" << c.arg2 << "], ";
279  os << submatrix_strings[c.arg3] << ", &" << submatrix_strings[c.arg4]
280  << ")\n";
281  break;
282  case kStoreStats:
283  os << nnet.GetComponentName(c.arg1) << ".StoreStats("
284  << submatrix_strings[c.arg2] << ")\n";
285  break;
286  case kBackprop:
287  case kBackpropNoModelUpdate: {
288  int32 component_index = c.arg1;
289  os << nnet.GetComponentName(component_index) << ".Backprop(";
290  if (c.arg2 == 0) os << "NULL, ";
291  else os << "precomputed_indexes[" << c.arg2 << "], ";
292  os << submatrix_strings[c.arg3] << ", "
293  << submatrix_strings[c.arg4] << ", "
294  << submatrix_strings[c.arg5] << ", "
295  << (computation.need_model_derivative &&
296  c.command_type == kBackprop ?
297  "[component-pointer], " : "NULL, ")
298  << (c.arg6 == 0 ? std::string("NULL") :
299  std::string("&") + submatrix_strings[c.arg6]) << ")\n";
300  break;
301  }
302  case kMatrixCopy:
303  os << submatrix_strings[c.arg1] << " = "
304  << submatrix_strings[c.arg2] << "\n";
305  break;
306  case kMatrixAdd:
307  os << submatrix_strings[c.arg1] << " += "
308  << submatrix_strings[c.arg2] << "\n";
309  break;
310  case kAddRows:
311  case kCopyRows:
312  os << submatrix_strings[c.arg1] << "."
313  << (c.command_type == kAddRows ? "AddRows" :
314  "CopyRows") << "(" << submatrix_strings[c.arg2]
315  << indexes_strings[c.arg3] << ")\n";
316  break;
317  case kAddRowsMulti:
318  case kAddToRowsMulti:
319  case kCopyRowsMulti:
320  case kCopyToRowsMulti: {
321  CommandType ct = c.command_type;
322  os << submatrix_strings[c.arg1] << "."
323  << (ct == kAddRowsMulti ? "AddRowsMulti" :
324  (ct == kAddToRowsMulti? "AddToRowsMulti" :
325  (ct == kCopyRowsMulti ? "CopyRowsMulti" :
326  "CopyToRowsMulti"))) << "("
327  << indexes_multi_strings[c.arg2] << ")\n";
328  break;
329  }
330  case kAddRowRanges: {
331  os << submatrix_strings[c.arg1] << ".AddRowRanges("
332  << submatrix_strings[c.arg2] << ", [";
333  const std::vector<std::pair<int32, int32> > &pairs =
334  computation.indexes_ranges[c.arg3];
335  for (size_t i = 0; i < pairs.size(); i++) {
336  os << pairs[i].first << ":" << (pairs[i].second - 1);
337  if (i + 1 < pairs.size()) os << ",";
338  }
339  os << "])\n";
340  break;
341  }
342  case kNoOperation:
343  os << "[no-op]\n";
344  break;
345  case kNoOperationMarker:
346  os << "# begin backward commands\n";
347  break;
348  default:
349  KALDI_ERR << "Un-handled command type.";
350  }
351 }

354 static void PrintComputationPreamble(
355  std::ostream &os,
356  const NnetComputation &c,
357  const Nnet &nnet,
358  const std::vector<std::string> &submatrix_strings,
359  const std::vector<std::string> &indexes_strings,
360  const std::vector<std::string> &indexes_multi_strings) {
361 
362  // First print info about the matrices.
363  os << "matrix ";
364  for (int32 i = 1; i < c.matrices.size(); i++) {
365  os << "m" << i << "(" << c.matrices[i].num_rows
366  << ", " << c.matrices[i].num_cols << ")";
367  if (i + 1 < c.matrices.size())
368  os << ", ";
369  }
370  os << "\n";
371  // show which matrices the inputs and outputs map to.
372  for (unordered_map<int32, std::pair<int32, int32> >::const_iterator iter =
373  c.input_output_info.begin(); iter != c.input_output_info.end();
374  ++iter) {
375  int32 node_index = iter->first,
376  value_matrix_index = iter->second.first,
377  deriv_matrix_index = iter->second.second;
378  os << nnet.GetNodeName(node_index) << ".value -> m"
379  << value_matrix_index << "\n";
380  if (deriv_matrix_index != 0) {
381  os << nnet.GetNodeName(node_index) << ".deriv -> m"
382  << deriv_matrix_index << "\n";
383  }
384  }
385  if (!c.matrix_debug_info.empty()) {
386  os << "# The following show how matrices correspond to network-nodes and\n"
387  << "# cindex-ids. Format is: matrix = <node-id>.[value|deriv][ <list-of-cindex-ids> ]\n"
388  << "# where a cindex-id is written as (n,t[,x]) but ranges of t values are compressed\n"
389  << "# so we write (n, tfirst:tlast).\n";
390  KALDI_ASSERT(c.matrix_debug_info.size() == c.matrices.size());
391  for (int32 i = 1; i < c.matrices.size(); i++) {
392  const NnetComputation::MatrixDebugInfo &debug_info =
393  c.matrix_debug_info[i];
394  os << "m" << i << " == " << (debug_info.is_deriv ? "deriv: " : "value: ");
395  PrintCindexes(os, debug_info.cindexes, nnet.GetNodeNames());
396  os << "\n";
397  }
398  }

401 void NnetComputation::Print(std::ostream &os, const Nnet &nnet) const {
402  std::vector<std::string> submatrix_strings, indexes_strings,
403  indexes_multi_strings;
404  this->GetSubmatrixStrings(nnet, &submatrix_strings);
