48  void PropagateFnc(const CuMatrixBase<BaseFloat> &in, CuMatrixBase<BaseFloat> *out) {
49  if (kl_inv_q_.NumRows() == 0) {
50  // Copy the CudaMatrix to a Matrix
51  Matrix<BaseFloat> in_tmp(in.NumRows(), in.NumCols());
52  in.CopyToMat(&in_tmp);
53  // Just check if there are posteriors in the Matrix (just check the first row)
54  BaseFloat post_sum=in_tmp.Row(0).Sum();
55  KALDI_ASSERT(ApproxEqual(post_sum, 1.0));
56  // Get a tmp Matrix of the stats
57  Matrix<BaseFloat> kl_stats_tmp(kl_stats_);
58  // Init a vector to get the sum of the rows (for normalization)
59  Vector<BaseFloat> row_sum(kl_stats_.NumRows(), kSetZero);
60  // Get the sum of the posteriors for normalization
61  row_sum.AddColSumMat(1, kl_stats_tmp);
62  // Apply floor to make sure there is no zero
63  row_sum.ApplyFloor(1e-20);
64  // Invert the sum (to normalize)
65  row_sum.InvertElements();
66  // Normalizing the statistics vector
67  kl_stats_tmp.MulRowsVec(row_sum);
68  //Apply floor before inversion and logarithm
69  kl_stats_tmp.ApplyFloor(1e-20);
70  //Apply invesion
71  kl_stats_tmp.InvertElements();
72  //Apply logarithm
73  kl_stats_tmp.ApplyLog();
74  //Inverted and logged values
75  kl_inv_q_.Resize(kl_stats_.NumRows(),kl_stats_.NumCols());
76  //Holds now log (1/Q)
77  kl_inv_q_.CopyFromMat(kl_stats_tmp);
78  }
79  // Get the logarithm of the features for the Entropy calculation
80  // Copy the CudaMatrix to a Matrix
81  Matrix<BaseFloat> in_log_tmp(in.NumRows(), in.NumCols());
82  in.CopyToMat(&in_log_tmp);
83  // Flooring and log
84  in_log_tmp.ApplyFloor(1e-20);
85  in_log_tmp.ApplyLog();
86  CuMatrix<BaseFloat> log_in(in.NumRows(), in.NumCols());
87  log_in.CopyFromMat(in_log_tmp);
88  // P*logP
89  CuMatrix<BaseFloat> tmp_entropy(in);
90  tmp_entropy.MulElements(log_in);
91  // Getting the entropy (sum P*logP)
92  CuVector<BaseFloat> in_entropy(in.NumRows(), kSetZero);
93  in_entropy.AddColSumMat(1,tmp_entropy);
94  // sum P*log (1/Q)
95  out->AddMatMat(1, in, kNoTrans, kl_inv_q_, kTrans, 0);
96  // (sum P*logP) + (sum P*log(1/Q)
97  out->AddVecToCols(1, in_entropy);
98  // return the negative KL-divergence
99  out->Scale(-1);
100  }

128  void Accumulate (const Matrix<BaseFloat> &posteriors, const std::vector<int32> &alignment) {
129  KALDI_ASSERT(posteriors.NumRows() == alignment.size());
130  KALDI_ASSERT(posteriors.NumCols() == kl_stats_.NumCols());
131  int32 num_frames = alignment.size();
132  for(int32 i = 0; i < num_frames; i++) {
133  //Convertin the float posterior into a double (to have higher precision during collection)
134  Vector<double> temp(posteriors.Row(i));
135  //Sum all the postiors associated with a particular state
136  kl_stats_.Row(alignment[i]).AddVec(1,temp);
137  }
138  }
