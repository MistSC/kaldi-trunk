271 void ConstArpaLmBuilder::ConsumeNGram(const NGram& ngram) {
272  int32 cur_order = ngram.words.size();
273  // If <ngram_order_> is larger than 1, then we do not create LmState for
274  // the final order entry. We only keep the log probability for it.
275  LmState *lm_state = NULL;
276  if (cur_order != ngram_order_ || ngram_order_ == 1) {
277  lm_state = new LmState(cur_order == 1,
278  cur_order == ngram_order_ - 1,
279  ngram.logprob, ngram.backoff);
280 
281  KALDI_ASSERT(seq_to_state_.find(ngram.words) == seq_to_state_.end());
282  seq_to_state_[ngram.words] = lm_state;
283  }
284 
285  // If n-gram order is larger than 1, we have to add possible child to
286  // existing LmStates. We have the following two assumptions:
287  // 1. N-grams are processed from small order to larger ones, i.e., from
288  // 1, 2, ... to the highest order.
289  // 2. If a n-gram exists in the Arpa format language model, then the
290  // "history" n-gram also exists. For example, if "A B C" is a valid
291  // n-gram, then "A B" is also a valid n-gram.
292  int32 last_word = ngram.words[cur_order - 1];
293  if (cur_order > 1) {
294  std::vector<int32> hist(ngram.words.begin(), ngram.words.end() - 1);
295  unordered_map<std::vector<int32>,
296  LmState*, VectorHasher<int32> >::iterator hist_iter;
297  hist_iter = seq_to_state_.find(hist);
298  if (hist_iter == seq_to_state_.end()) {
299  std::ostringstream ss;
300  for (int i = 0; i < cur_order; ++i)
301  ss << (i == 0 ? '[' : ' ') << ngram.words[i];
302  KALDI_ERR << "In line " << LineNumber() << ": "
303  << cur_order << "-gram " << ss.str() << "] does not have "
304  << "a parent model " << cur_order << "-gram.";
305  }
306  if (cur_order != ngram_order_ || ngram_order_ == 1) {
307  KALDI_ASSERT(lm_state != NULL);
308  KALDI_ASSERT(!hist_iter->second->IsChildFinalOrder());
309  hist_iter->second->AddChild(last_word, lm_state);
310  } else {
311  KALDI_ASSERT(lm_state == NULL);
312  KALDI_ASSERT(hist_iter->second->IsChildFinalOrder());
313  hist_iter->second->AddChild(last_word, ngram.logprob);
314  }
315  } else {
316  // Figures out <max_word_id>.
317  num_words_ = std::max(num_words_, last_word + 1);
318  }
319 }

321 // ConstArpaLm can be built in the following steps, assuming we have already
322 // created LmStates <seq_to_state_>:
323 // 1. Sort LmStates lexicographically.
324 // This enables us to compute relative address. When we say lexicographic, we
325 // treat the word-ids as letters. After sorting, the LmStates are in the
326 // following order:
327 // ...
328 // A B
329 // A B A
330 // A B B
331 // A B C
332 // ...
333 // where each line represents a LmState.
334 // 2. Update <my_address> in LmState, which is relative to the first element in
335 // <sorted_vec>.
336 // 3. Put the following structure into the memory block
337 // struct LmState {
338 // float logprob;
339 // float backoff_logprob;
340 // int32 num_children;
341 // std::pair<int32, int32> [] children;
342 // }
343 //
344 // At the same time, we will also create two special buffers:
345 // <unigram_states_>
346 // <overflow_buffer_>
347 void ConstArpaLmBuilder::ReadComplete() {
348  // STEP 1: sorting LmStates lexicographically.
349  // Vector for holding the sorted LmStates.
350  std::vector<std::pair<std::vector<int32>*, LmState*> > sorted_vec;
351  unordered_map<std::vector<int32>,
352  LmState*, VectorHasher<int32> >::iterator iter;
353  for (iter = seq_to_state_.begin(); iter != seq_to_state_.end(); ++iter) {
354  if (iter->second->MemSize() > 0) {
355  sorted_vec.push_back(
356  std::make_pair(const_cast<std::vector<int32>*>(&(iter->first)),
357  iter->second));
358  }
359  }
360 
361  std::sort(sorted_vec.begin(), sorted_vec.end(),
362  WordsAndLmStatePairLessThan());
363 
364  // STEP 2: updating <my_address> in LmState.
365  for (int32 i = 0; i < sorted_vec.size(); ++i) {
366  lm_states_size_ += sorted_vec[i].second->MemSize();
367  if (i == 0) {
368  sorted_vec[i].second->SetMyAddress(0);
369  } else {
370  sorted_vec[i].second->SetMyAddress(sorted_vec[i - 1].second->MyAddress()
371  + sorted_vec[i - 1].second->MemSize());
372  }
373  }
374 
375  // STEP 3: creating memory block to store LmStates.
376  // Reserves a memory block for LmStates.
377  int64 lm_states_index = 0;
378  try {
379  lm_states_ = new int32[lm_states_size_];
380  } catch(const std::exception &e) {
381  KALDI_ERR << e.what();
382  }
383 
384  // Puts data into memory block.
385  unigram_states_ = new int32*[num_words_];
386  std::vector<int32*> overflow_buffer_vec;
387  for (int32 i = 0; i < num_words_; ++i) {
388  unigram_states_[i] = NULL;
389  }
390  for (int32 i = 0; i < sorted_vec.size(); ++i) {
391  // Current address.
392  int32* parent_address = lm_states_ + lm_states_index;
393 
394  // Adds logprob.
395  Int32AndFloat logprob_f(sorted_vec[i].second->Logprob());
396  lm_states_[lm_states_index++] = logprob_f.i;
397 
398  // Adds backoff_logprob.
399  Int32AndFloat backoff_logprob_f(sorted_vec[i].second->BackoffLogprob());
400  lm_states_[lm_states_index++] = backoff_logprob_f.i;
401 
402  // Adds num_children.
403  lm_states_[lm_states_index++] = sorted_vec[i].second->NumChildren();
404 
405  // Adds children, there are 3 cases:
406  // 1. Child is a leaf and not unigram
407  // 2. Child is not a leaf or is unigram
408  // 2.1 Relative address can be represented by 30 bits
409  // 2.2 Relative address cannot be represented by 30 bits
410  sorted_vec[i].second->SortChildren();
411  for (int32 j = 0; j < sorted_vec[i].second->NumChildren(); ++j) {
412  int32 child_info;
413  if (sorted_vec[i].second->IsChildFinalOrder() ||
414  sorted_vec[i].second->GetChild(j).second.state->MemSize() == 0) {
415  // Child is a leaf and not unigram. In this case we will not create an
416  // entry in <lm_states_>; instead, we put the logprob in the place where
417  // we normally store the poitner.
418  Int32AndFloat child_logprob_f;
419  if (sorted_vec[i].second->IsChildFinalOrder()) {
420  child_logprob_f.f = sorted_vec[i].second->GetChild(j).second.prob;
421  } else {
422  child_logprob_f.f =
423  sorted_vec[i].second->GetChild(j).second.state->Logprob();
424  }
425  child_info = child_logprob_f.i;
426  child_info &= ~1; // Sets the last bit to 0 so <child_info> is even.
427  } else {
428  // Child is not a leaf or is unigram.
429  int64 offset =
430  sorted_vec[i].second->GetChild(j).second.state->MyAddress()
431  - sorted_vec[i].second->MyAddress();
432  KALDI_ASSERT(offset > 0);
433  if (offset <= max_address_offset_) {
434  // Relative address can be represented by 30 bits.
435  child_info = offset * 2;
436  child_info |= 1;
437  } else {
438  // Relative address cannot be represented by 30 bits, we have to put
439  // the child address into <overflow_buffer_>.
440  int32* abs_address = parent_address + offset;
441  overflow_buffer_vec.push_back(abs_address);
442  int32 overflow_buffer_index = overflow_buffer_vec.size() - 1;
443  child_info = overflow_buffer_index * 2;
444  child_info |= 1;
445  child_info *= -1;
446  }
447  }
448  // Child word.
449  lm_states_[lm_states_index++] = sorted_vec[i].second->GetChild(j).first;
450  // Child info.
451  lm_states_[lm_states_index++] = child_info;
452  }
453 
454  // If the current state corresponds to an unigram, then create a separate
455  // loop up table to improve efficiency, since those will be looked up pretty
456  // frequently.
457  if (sorted_vec[i].second->IsUnigram()) {
458  KALDI_ASSERT(sorted_vec[i].first->size() == 1);
459  unigram_states_[(*sorted_vec[i].first)[0]] = parent_address;
460  }
461  }
462  KALDI_ASSERT(lm_states_size_ == lm_states_index);
463 
464  // Move <overflow_buffer_> from vector holder to array.
465  overflow_buffer_size_ = overflow_buffer_vec.size();
466  overflow_buffer_ = new int32*[overflow_buffer_size_];
467  for (int32 i = 0; i < overflow_buffer_size_; ++i) {
468  overflow_buffer_[i] = overflow_buffer_vec[i];
469  }
470 
471  is_built_ = true;
472 }

488 void ConstArpaLm::Write(std::ostream &os, bool binary) const {
489  KALDI_ASSERT(initialized_);
490  if (!binary) {
491  KALDI_ERR << "text-mode writing is not implemented for ConstArpaLm.";
492  }
493 
494  WriteToken(os, binary, "<ConstArpaLm>");
495 
496  // Misc info.
497  WriteToken(os, binary, "<LmInfo>");
498  WriteBasicType(os, binary, bos_symbol_);
499  WriteBasicType(os, binary, eos_symbol_);
500  WriteBasicType(os, binary, unk_symbol_);
501  WriteBasicType(os, binary, ngram_order_);
502  WriteToken(os, binary, "</LmInfo>");
503 
504  // LmStates section.
505  WriteToken(os, binary, "<LmStates>");
506  WriteBasicType(os, binary, lm_states_size_);
507  os.write(reinterpret_cast<char *>(lm_states_),
508  sizeof(int32) * lm_states_size_);
509  if (!os.good()) {
510  KALDI_ERR << "ConstArpaLm <LmStates> section writing failed.";
511  }
512  WriteToken(os, binary, "</LmStates>");
513 
514  // Unigram section. We write memory offset to disk instead of the absolute
515  // pointers.
516  WriteToken(os, binary, "<LmUnigram>");
517  WriteBasicType(os, binary, num_words_);
518  int64* tmp_unigram_address = new int64[num_words_];
519  for (int32 i = 0; i < num_words_; ++i) {
520  // The relative address here is a little bit tricky:
521  // 1. If the original address is NULL, then we set the relative address to
522  // zero.
523  // 2. If the original address is not NULL, we set it to the following:
524  // unigram_states_[i] - lm_states_ + 1
525  // we plus 1 to ensure that the above value is positive.
526  tmp_unigram_address[i] = (unigram_states_[i] == NULL) ? 0 :
527  unigram_states_[i] - lm_states_ + 1;
528  }
529  os.write(reinterpret_cast<char *>(tmp_unigram_address),
530  sizeof(int64) * num_words_);
531  if (!os.good()) {
532  KALDI_ERR << "ConstArpaLm <LmUnigram> section writing failed.";
533  }
534  delete[] tmp_unigram_address; // Releases the memory.
535  tmp_unigram_address = NULL;
536  WriteToken(os, binary, "</LmUnigram>");
537 
538  // Overflow section. We write memory offset to disk instead of the absolute
539  // pointers.
540  WriteToken(os, binary, "<LmOverflow>");
541  WriteBasicType(os, binary, overflow_buffer_size_);
542  int64* tmp_overflow_address = new int64[overflow_buffer_size_];
543  for (int32 i = 0; i < overflow_buffer_size_; ++i) {
544  // The relative address here is a little bit tricky:
545  // 1. If the original address is NULL, then we set the relative address to
546  // zero.
547  // 2. If the original address is not NULL, we set it to the following:
548  // overflow_buffer_[i] - lm_states_ + 1
549  // we plus 1 to ensure that the above value is positive.
550  tmp_overflow_address[i] = (overflow_buffer_[i] == NULL) ? 0 :
551  overflow_buffer_[i] - lm_states_ + 1;
552  }
553  os.write(reinterpret_cast<char *>(tmp_overflow_address),
554  sizeof(int64) * overflow_buffer_size_);
555  if (!os.good()) {
556  KALDI_ERR << "ConstArpaLm <LmOverflow> section writing failed.";
557  }
558  delete[] tmp_overflow_address;
559  tmp_overflow_address = NULL;
560  WriteToken(os, binary, "</LmOverflow>");
561  WriteToken(os, binary, "</ConstArpaLm>");
562 }

739 float ConstArpaLm::GetNgramLogprob(const int32 word,
740  const std::vector<int32>& hist) const {
741  KALDI_ASSERT(initialized_);
742 
743  // If the history size plus one is larger than <ngram_order_>, remove the old
744  // words.
745  std::vector<int32> mapped_hist(hist);
746  while (mapped_hist.size() >= ngram_order_) {
747  mapped_hist.erase(mapped_hist.begin(), mapped_hist.begin() + 1);
748  }
749  KALDI_ASSERT(mapped_hist.size() + 1 <= ngram_order_);
750 
751  // TODO(guoguo): check with Dan if this is reasonable.
752  // Maps possible out-of-vocabulary words to <unk>. If a word does not have a
753  // corresponding LmState, we treat it as <unk>. We map it to <unk> if <unk> is
754  // specified.
755  int32 mapped_word = word;
756  if (unk_symbol_ != -1) {
757  KALDI_ASSERT(mapped_word >= 0);
758  if (mapped_word >= num_words_ || unigram_states_[mapped_word] == NULL) {
759  mapped_word = unk_symbol_;
760  }
761  for (int32 i = 0; i < mapped_hist.size(); ++i) {
762  KALDI_ASSERT(mapped_hist[i] >= 0);
763  if (mapped_hist[i] >= num_words_ ||
764  unigram_states_[mapped_hist[i]] == NULL) {
765  mapped_hist[i] = unk_symbol_;
766  }
767  }
768  }
769 
770  // Loops up n-gram probability.
771  return GetNgramLogprobRecurse(mapped_word, mapped_hist);
772 }

774 float ConstArpaLm::GetNgramLogprobRecurse(
775  const int32 word, const std::vector<int32>& hist) const {
776  KALDI_ASSERT(initialized_);
777  KALDI_ASSERT(hist.size() + 1 <= ngram_order_);
778 
779  // Unigram case.
780  if (hist.size() == 0) {
781  if (word >= num_words_ || unigram_states_[word] == NULL) {
782  // If <unk> is defined, then the word sequence should have already been
783  // mapped to <unk> is necessary; this is for the case where <unk> is not
784  // defined.
785  return std::numeric_limits<float>::min();
786  } else {
787  Int32AndFloat logprob_i(*unigram_states_[word]);
788  return logprob_i.f;
789  }
790  }
791 
792  // High n-gram orders.
793  float logprob = 0.0;
794  float backoff_logprob = 0.0;
795  int32* state;
796  if ((state = GetLmState(hist)) != NULL) {
797  int32 child_info;
798  int32* child_lm_state = NULL;
799  if (GetChildInfo(word, state, &child_info)) {
800  DecodeChildInfo(child_info, state, &child_lm_state, &logprob);
801  return logprob;
802  } else {
803  Int32AndFloat backoff_logprob_i(*(state + 1));
804  backoff_logprob = backoff_logprob_i.f;
805  }
806  }
807  std::vector<int32> new_hist(hist);
808  new_hist.erase(new_hist.begin(), new_hist.begin() + 1);
809  return backoff_logprob + GetNgramLogprobRecurse(word, new_hist);
810 }

812 int32* ConstArpaLm::GetLmState(const std::vector<int32>& seq) const {
813  KALDI_ASSERT(initialized_);
814 
815  // No LmState exists for empty word sequence.
816  if (seq.size() == 0) return NULL;
817 
818  // If <unk> is defined, then the word sequence should have already been mapped
819  // to <unk> is necessary; this is for the case where <unk> is not defined.
820  if (seq[0] >= num_words_ || unigram_states_[seq[0]] == NULL) return NULL;
821  int32* parent = unigram_states_[seq[0]];
822 
823  int32 child_info;
824  int32* child_lm_state = NULL;
825  float logprob;
826  for (int32 i = 1; i < seq.size(); ++i) {
827  if (!GetChildInfo(seq[i], parent, &child_info)) {
828  return NULL;
829  }
830  DecodeChildInfo(child_info, parent, &child_lm_state, &logprob);
831  if (child_lm_state == NULL) {
832  return NULL;
833  } else {
834  parent = child_lm_state;
835  }
836  }
837  return parent;
838 }

840 bool ConstArpaLm::GetChildInfo(const int32 word,
841  int32* parent, int32* child_info) const {
842  KALDI_ASSERT(initialized_);
843 
844  KALDI_ASSERT(parent != NULL);
845  KALDI_ASSERT(parent >= lm_states_);
846  KALDI_ASSERT(child_info != NULL);
847 
848  KALDI_ASSERT(parent + 2 <= lm_states_end_);
849  int32 num_children = *(parent + 2);
850  KALDI_ASSERT(parent + 2 + 2 * num_children <= lm_states_end_);
851 
852  if (num_children == 0) return false;
853 
854  // A binary search into the children memory block.
855  int32 start_index = 1;
856  int32 end_index = num_children;
857  while (start_index <= end_index) {
858  int32 mid_index = round((start_index + end_index) / 2);
859  int32 mid_word = *(parent + 1 + 2 * mid_index);
860  if (mid_word == word) {
861  *child_info = *(parent + 2 + 2 * mid_index);
862  return true;
863  } else if (mid_word < word) {
864  start_index = mid_index + 1;
865  } else {
866  end_index = mid_index - 1;
867  }
868  }
869 
870  return false;
871 }

873 void ConstArpaLm::DecodeChildInfo(const int32 child_info,
874  int32* parent,
875  int32** child_lm_state,
876  float* logprob) const {
877  KALDI_ASSERT(initialized_);
878 
879  KALDI_ASSERT(logprob != NULL);
880  if (child_info % 2 == 0) {
881  // Child is a leaf, only returns the log probability.
882  *child_lm_state = NULL;
883  Int32AndFloat logprob_i(child_info);
884  *logprob = logprob_i.f;
885  } else {
886  int32 child_offset = child_info / 2;
887  if (child_offset > 0) {
888  *child_lm_state = parent + child_offset;
889  Int32AndFloat logprob_i(**child_lm_state);
890  *logprob = logprob_i.f;
891  } else {
892  KALDI_ASSERT(-child_offset < overflow_buffer_size_);
893  *child_lm_state = overflow_buffer_[-child_offset];
894  Int32AndFloat logprob_i(**child_lm_state);
895  *logprob = logprob_i.f;
896  }
897  KALDI_ASSERT(*child_lm_state >= lm_states_);
898  KALDI_ASSERT(*child_lm_state <= lm_states_end_);

902 void ConstArpaLm::WriteArpaRecurse(int32* lm_state,
903  const std::vector<int32>& seq,
904  std::vector<ArpaLine> *output) const {
905  if (lm_state == NULL) return;
906 
907  KALDI_ASSERT(lm_state >= lm_states_);
908  KALDI_ASSERT(lm_state + 2 <= lm_states_end_);
909 
910  // Inserts the current LmState to <output>.
911  ArpaLine arpa_line;
912  arpa_line.words = seq;
913  Int32AndFloat logprob_i(*lm_state);
914  arpa_line.logprob = logprob_i.f;
915  Int32AndFloat backoff_logprob_i(*(lm_state + 1));
916  arpa_line.backoff_logprob = backoff_logprob_i.f;
917  output->push_back(arpa_line);
918 
919  // Scans for possible children, and recursively adds child to <output>.
920  int32 num_children = *(lm_state + 2);
921  KALDI_ASSERT(lm_state + 2 + 2 * num_children <= lm_states_end_);
922  for (int32 i = 0; i < num_children; ++i) {
923  std::vector<int32> new_seq(seq);
924  new_seq.push_back(*(lm_state + 3 + 2 * i));
925  int32 child_info = *(lm_state + 4 + 2 * i);
926  float logprob;
927  int32* child_lm_state = NULL;
928  DecodeChildInfo(child_info, lm_state, &child_lm_state, &logprob);
929 
930  if (child_lm_state == NULL) {
931  // Leaf case.
932  ArpaLine child_arpa_line;
933  child_arpa_line.words = new_seq;
934  child_arpa_line.logprob = logprob;
935  child_arpa_line.backoff_logprob = 0.0;
936  output->push_back(child_arpa_line);
937  } else {
938  WriteArpaRecurse(child_lm_state, new_seq, output);
939  }
940  }
